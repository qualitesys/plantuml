19   package jcckit.util;
20   
21   import java.applet.Applet;
22   
23   /**
24    *  Implementation of {@link FlatConfigData} based on
25    *  <tt>java.applet.Applet</tt>.
26    *
27    *  @author Franz-Josef Elmer
28    */
29   public class AppletBasedConfigData extends FlatConfigData {
30     private final Applet _applet;
31   
32     /**
33      * Creates an instance based on the specified applet.
34      * The path is undefined.
35      */
36     public AppletBasedConfigData(Applet applet) {
37       this(applet, null);
38     }
39   
40     /** Creates an instance based on the specified properties and path. */
41     private AppletBasedConfigData(Applet applet, String path) {
42       super(path);
43       _applet = applet;
44     }
45   
46     protected String getValue(String fullKey) {
47       return _applet.getParameter(fullKey);
48     }
49   
50     protected ConfigData createConfigData(String path) {
51       return new AppletBasedConfigData(_applet, path);
52     }
53   }
