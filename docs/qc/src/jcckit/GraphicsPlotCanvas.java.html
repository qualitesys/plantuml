36   /**
37    * Class which handles plotting into a <tt>Graphics</tt> context based on the
38    * {link jcckit.renderer.GraphicsRenderer}. This class is not a subclass of
39    * <tt>java.awt.Component</tt>. The actual AWT component presenting the plot
40    * is an innerclass. Its instance wrapped by <tt>GraphicsPlotCanvas</tt> can
41    * be obtained with {link #getGraphicsCanvas}.
42    * <p>
43    * The plot is painted by using double-buffering and pre-rendered view of the
44    * coordinate system. That is, the coordinate system is drawn into an off-screen
45    * image. It will be redrawn only if the size of the embedding AWT component is
46    * changed.
47    *
48    * @author Franz-Josef Elmer
49    */
50   public class GraphicsPlotCanvas extends PlotCanvas {
51       /** Key of a configuration parameter. */
52       public static final String BACKGROUND_KEY = "background";
53       public static final String FOREGROUND_KEY = "foreground";
54       public static final String DOUBLE_BUFFERING_KEY = "doubleBuffering";
55   
56       /**
57        * Class which does the actual painting. Needs the <tt>Component</tt> into
58        * which the plot is painted for some resources like size, background color,
59        * etc.
60        *
61        * @author Franz-Josef Elmer
62        */
63       private final BufferedImage img3;
64

66

68       private String _renderer = "jcckit.renderer.GraphicsRenderer";

72       /**
73        * Creates an instance from the specified configuration parameters. <table
74        * border=1 cellpadding=5>
75        * <tr>
76        * <th>Key &amp; Default Value</th>
77        * <th>Type</th>
78        * <th>Mandatory</th>
79        * <th>Description</th>
80        * </tr>
81        * <tr>
82        * <td><tt>background = </tt><i>default background color of the wrapped
83        * AWT component</i></td>
84        * <td><tt>Color</tt></td>
85        * <td>no</td>
86        * <td>Background color of the wrapped AWT component.</td>
87        * </tr>
88        * <tr>
89        * <td><tt>foreground = </tt><i>default foreground color of the wrapped
90        * AWT component</i></td>
91        * <td><tt>Color</tt></td>
92        * <td>no</td>
93        * <td>Foreground color of the wrapped AWT component.</td>
94        * </tr>
95        * <tr>
96        * <td><tt>doubleBuffering = true</tt></td>
97        *     <td><tt>boolean</tt></td><td>no</td>
98        *     <td>If <tt>true</tt> the plot will be painted by using
99        *         double-buffering and pre-rendered view of the coordinate system.
100        *     </td></tr>
101        * </table>
102        * In addition the configuration parameters of the
103        * <a href="plot/PlotCanvas.html#PlotCanvas(jcckit.util.ConfigParameters)">
104        * constructor</a> of the superclass {@link jcckit.plot.PlotCanvas} apply.
105        */
106       public GraphicsPlotCanvas(ConfigParameters config, BufferedImage img3) {
107           super(config);
108           this.img3 = img3;
109           setRenderer("jcckit.renderer.Graphics2DRenderer");
110   
111           g3 = img3.createGraphics();
112           g3.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
113   
114           // _doubleBuffering = config.getBoolean(DOUBLE_BUFFERING_KEY, true);
115           background = config.getColor(BACKGROUND_KEY, Color.WHITE);
116           foreground = config.getColor(FOREGROUND_KEY, Color.BLACK);
117       }
118   
119       private final Color background;
120       private final Color foreground;
121   
122       /**
123        * Paints the plot. If {link GraphicsPlotCanvas#_doubleBuffering} is set
124        * double-buffering and pre-rendered view of the coordinate system is used.
125        */
126       public void paint() {
127           Dimension size = new Dimension(img3.getWidth(), img3.getHeight());
128           g3.setColor(background);
129           g3.fillRect(0, 0, size.width + 1, size.height + 1);
130   
131           init(size);
132   
133           _transformation.apply(g3);
134           Plot plot = getPlot();
135           drawCoordinateSystem(size, plot);
136           drawPlot(plot);
137           if (_marker != null) {
138               _marker.renderWith(createRenderer());
139           }
140       }
141   
142       private void drawPlot(Plot plot) {
143           prepare();
144           Renderer renderer = createRenderer();
145           GraphicalElement[] curves = plot.getCurves();
146           for (int i = 0; i < curves.length; i++) {
147               curves[i].renderWith(renderer);
148           }
149           GraphicalElement annotation = plot.getAnnotation();
150           if (annotation != null) {
151               annotation.renderWith(renderer);
152           }
153           if (plot.isLegendVisible()) {
154               plot.getLegend().renderWith(renderer);
155           }
156       }
157   
158       private void init(Dimension size) {
159           calculateTransformation(size);
160       }
161   
162       private void drawCoordinateSystem(Dimension size, Plot plot) {
163           g3.setColor(foreground);
164           plot.getCoordinateSystem().renderWith(createRenderer());
165       }

174       /**
175        * Calculate the transformation form device-independent coordinates into
176        * device-dependent coordinates according to the specified canvas size.
177        */
178       protected void calculateTransformation(Dimension size) {
179           _transformation = new Transformation(size.width, size.height, getPaper(), getHorizontalAnchor(),
180                   getVerticalAnchor());
181       }

191       }
