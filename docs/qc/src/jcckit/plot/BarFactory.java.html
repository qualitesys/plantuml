19   package jcckit.plot;
20   
21   import jcckit.graphic.ClippingShape;
22   import jcckit.graphic.GraphPoint;
23   import jcckit.graphic.GraphicAttributes;
24   import jcckit.graphic.GraphicalElement;
25   import jcckit.graphic.Rectangle;
26   import jcckit.util.ConfigParameters;
27   
28   /**
29    * A factory of bars. The bars are {@link Rectangle Rectangles}. 
30    * Depending on the configuration parameters the bars can be 
31    * horizontal or vertical. Bars of several curves can be side by side or
32    * stacked. The bar length is determined by the x or y value of the 
33    * curve point in device-independent coordinates. If the value is negative 
34    * the bar goes into the negative direction. For stacked bars the values
35    * should always be positive.
36    * <p>
37    * When used inside a {@link SimpleCurve} soft clipping should always be
38    * switched off (see 
39    * {@link SimpleCurve#SimpleCurve(ConfigParameters, int, int, ClippingShape, Legend)}). 
40    *
41    * @author Franz-Josef Elmer
42    */
43   public class BarFactory extends AbstractSymbolFactory {
44     /** Configuration parameter key. */
45     public static final String STACKED_KEY = "stacked",
46                                HORIZONTAL_BARS_KEY = "horizontalBars";
47   
48     private final boolean _stacked;
49     private final boolean _horizontalBars;
50   
51     /**
52      * Creates an instance from the specified configuration parameters.
53      * <table border=1 cellpadding=5>
54      * <tr><th>Key &amp; Default Value</th><th>Type</th><th>Mandatory</th>
55      *     <th>Description</th></tr>
56      * <tr><td><tt>horizontalBars = false</tt></td>
57      *     <td><tt>boolean</tt></td><td>no</td>
58      *     <td>If <tt>true</tt> horizontal bars will be drawn. Otherwise 
59      *         vertical bars are drawn.</td></tr>
60      * <tr><td><tt>stacked = false</tt></td>
61      *     <td><tt>boolean</tt></td><td>no</td>
62      *     <td>If <tt>true</tt> the bars of several curves will be 
63      *         stacked.</td></tr>
64      * </table>
65      * In addition the configuration parameters of the
66      * <a href="AbstractSymbolFactory.html#AbstractSymbolFactory(jcckit.util.ConfigParameters)">
67      * constructor</a> of the superclass {@link AbstractSymbolFactory} apply.
68      */
69     public BarFactory(ConfigParameters config) {
70       super(config);
71       _horizontalBars = config.getBoolean(HORIZONTAL_BARS_KEY, false);
72       _stacked = config.getBoolean(STACKED_KEY, false);
73     }
74   
75     /**
76      * Creates a bar at the specified point.
77      * If <tt>hintFromPreviousCurve</tt>
78      * is not an instance of {@link PositionHint} the values of 
79      * origin and position will be (0,0).
80      * @param hintFromPreviousCurve Hint from previous curve. Will be used
81      *        to calculate symbol shape and hint for the next curve.
82      */
83     protected Symbol createSymbol(GraphPoint point, GraphicAttributes attributes,
84                                Hint hintForNextPoint,
85                                Hint hintFromPreviousCurve) {
86       GraphPoint origin = new GraphPoint(null);
87       GraphPoint position = origin;
88       if (hintFromPreviousCurve instanceof PositionHint) {
89         origin = ((PositionHint) hintFromPreviousCurve).getOrigin();
90         position = ((PositionHint) hintFromPreviousCurve).getPosition();
91       }
92       double px = position.getX();
93       double py = position.getY();
94       double x = point.getX() - origin.getX();
95       double y = point.getY() - origin.getY();
96       if (_horizontalBars) {
97         y = _size;
98         position = new GraphPoint(px + 0.5 * x, point.getY() + py);
99         px += _stacked ? x : 0;
100         py += _stacked ? 0 : _size;
101       } else {
102         x = _size;
103         position = new GraphPoint(point.getX() + px, py + 0.5 * y);
104         px += _stacked ? 0 : _size;
105         py += _stacked ? y : 0;
106       }
107       Hint hintForNextCurve = new PositionHint(origin, new GraphPoint(px, py));
108       return new Symbol(new Rectangle(position, Math.abs(x), Math.abs(y), 
109                                       attributes),
110                         hintForNextPoint, hintForNextCurve);
111     }
112   
113     /**
114      *  Creates a symbol for the legend at the specified position.
115      *  @param centerPosition Center position of the symbol.
116      *  @param size The size of the symbol.
117      */
118     public GraphicalElement createLegendSymbol(GraphPoint centerPosition,
119                                                double size) {
120       return new Rectangle(centerPosition, size, size, _attributes);
121     }
122     
123     /** 
124      * Returns <tt>null</tt> because this method isn't needed but has to be
125      * implemented.
126      */
127     protected GraphicalElement createPlainSymbol(
128         GraphPoint centerPosition, double size, GraphicAttributes attributes) {
129       return null;
130     }
131   }
