19   package jcckit.plot;
20   
21   import jcckit.graphic.GraphicalElement;
22   
23   /**
24    * Immutable class holding the graphical represention of the symbol and
25    * two {@link Hint Hints}.
26    * 
27    * @author Franz-Josef Elmer
28    */
29   public class Symbol {
30     private final GraphicalElement _symbol;
31     private final Hint _hintForNextPoint;
32     private final Hint _hintForNextCurve;
33   
34     /** Creates an instance for the specified symbol and hints. */
35     public Symbol(GraphicalElement symbol, Hint hintForNextPoint,
36                   Hint hintForNextCurve) {
37       _symbol = symbol;
38       _hintForNextPoint = hintForNextPoint;
39       _hintForNextCurve = hintForNextCurve;
40     }
41   
42     /** Returns the graphical symbol. */
43     public GraphicalElement getSymbol() {
44       return _symbol;
45     }
46   
47     /** Returns the hint for the next point. */
48     public Hint getHintForNextPoint() {
49       return _hintForNextPoint;
50     }
51   
52     /** Returns the hint for the next curve. */
53     public Hint getHintForNextCurve() {
54       return _hintForNextCurve;
55     }
56   }
