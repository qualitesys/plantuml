19   package jcckit.plot;
20   
21   import jcckit.data.DataPlot;
22   import jcckit.graphic.Anchor;
23   import jcckit.graphic.ClippingRectangle;
24   import jcckit.util.ConfigParameters;
25   
26   /**
27    * An abstract canvas containg a single {@link Plot}. The canvas is specified
28    * by a {@link ClippingRectangle}, called <em>paper</em>. A horizontal and
29    * vertical {@link Anchor} determine the position of the paper on the actual
30    * device.
31    * 
32    * @author Franz-Josef Elmer
33    */
34   public class PlotCanvas implements PlotListener {
35       /** Configuration parameter key. */
36       public static final String PAPER_KEY = "paper", HORIZONTAL_ANCHOR_KEY = "horizontalAnchor",
37               VERTICAL_ANCHOR_KEY = "verticalAnchor", PLOT_KEY = "plot";
38       private final ClippingRectangle _paper;
39       private final Anchor _horizontalAnchor;
40       private final Anchor _verticalAnchor;
41       private final Plot _plot;
42   
43       /**
44        * Creates an instance from the specified configuration parameters. <table
45        * border=1 cellpadding=5>
46        * <tr>
47        * <th>Key &amp; Default Value</th>
48        * <th>Type</th>
49        * <th>Mandatory</th>
50        * <th>Description</th>
51        * </tr>
52        * <tr>
53        * <td><tt>horizontalAnchor = center</tt></td>
54        * <td><tt>String</tt></td>
55        * <td>no</td>
56        * <td>Horizontal position of the paper relative to the device border.
57        * Possible values are <tt>left</tt>, <tt>center</tt>, and
58        * <tt>right</tt>.</td>
59        * </tr>
60        * <tr>
61        * <td><tt>paper = 0,&nbsp;0,&nbsp;1,&nbsp;0.6</tt></td>
62        * <td><tt>double[]</tt></td>
63        * <td>no</td>
64        * <td>Rectangle defining the paper. The first two values determine the x-
65        * and y- coordinates (in device-independent units) of the lower-left
66        * corner. The last two values determine the upper-right corner.</td>
67        * </tr>
68        * <tr>
69        * <td><tt>plot = </tt>default values of {@link Plot}</td>
70        * <td><tt>ConfigParameters</tt></td>
71        * <td>no</td>
72        * <td>Definition of the {@link Plot}.</td>
73        * </tr>
74        * <tr>
75        * <td><tt>verticalAnchor = center</tt></td>
76        * <td><tt>String</tt></td>
77        * <td>no</td>
78        * <td>Vertical position of the paper relative to the device border.
79        * Possible values are <tt>top</tt>, <tt>center</tt>, and
80        * <tt>bottom</tt>.</td>
81        * </tr>
82        * </table>
83        * <p>
84        * Note, that this instance registers itself at the wrapped {@link Plot}
85        * instance.
86        */
87       public PlotCanvas(ConfigParameters config) {
88           double[] paper = config.getDoubleArray(PAPER_KEY, new double[] { 0, 0, 1, 0.6 });
89           _paper = new ClippingRectangle(paper[0], paper[1], paper[2], paper[3]);
90           _horizontalAnchor = Anchor.getHorizontalAnchor(config, HORIZONTAL_ANCHOR_KEY, Anchor.CENTER);
91           _verticalAnchor = Anchor.getVerticalAnchor(config, VERTICAL_ANCHOR_KEY, Anchor.CENTER);
92           _plot = new Plot(config.getNode(PLOT_KEY));
93           _plot.addPlotListener(this);
94       }
95   
96       /** Returns the paper definition. */
97       public ClippingRectangle getPaper() {
98           return _paper;
99       }
100   
101       /** Returns the horizontal anchor. */
102       public Anchor getHorizontalAnchor() {
103           return _horizontalAnchor;
104       }
105   
106       /** Returns the vertical anchor. */
107       public Anchor getVerticalAnchor() {
108           return _verticalAnchor;
109       }
110   
111       /** Returns the plot. */
112       public Plot getPlot() {
113           return _plot;
114       }
115   
116       /**
117        * Connects the wrapped {@link Plot} instance with the specified
118        * {@link DataPlot}.
119        * 
120        * @param dataPlot
121        *            Data to be connected with this plot canvas. Can be
122        *            <tt>null</tt> in order to disconnect this instance from a
123        *            <tt>DataPlot</tt>.
124        */
125       public void connect(DataPlot dataPlot) {
126           _plot.connect(dataPlot);
127       }
128   
129       /**
130        * Handles the spcified event. Here nothing is done. But subclass may
131        * override this method.
132        */
133       public void plotChanged(PlotEvent event) {
134       }
135   }
