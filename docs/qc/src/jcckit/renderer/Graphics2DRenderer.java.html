52   /**
53    * Renderer who draws the {@link jcckit.graphic.GraphicalElement
54    * GraphicalElements} into a <tt>java.awt.Graphics2D</tt> context.
55    * <p>
56    * The default color for lines and texts is determined by the current color of
57    * the <tt>Graphics2D</tt> context when a new instance of
58    * <tt>Graphics2DRenderer</tt> is created.
59    * <p>
60    * The default font is <tt>SansSerif-12</tt>.
61    * 
62    * @author Franz-Josef Elmer
63    */
64   public class Graphics2DRenderer implements GraphicalCompositeRenderer, PolygonRenderer, OvalRenderer, TextRenderer,
65           RectangleRenderer {
66       private static final int FS = 1;
67       private static final String DEFAULT_FONT_NAME = "SansSerif";
68       private static final FontStyle DEFAULT_FONT_STYLE = FontStyle.NORMAL;
69       private static final int DEFAULT_FONT_SIZE = 12;
70   
71       private Color _defaultColor;
72       private Graphics2D _graphics;
73   
74       /**
75        * Initializes this instance. During renderering the current transformation
76        * will be leaved unchanged. But the current Clip may be cleared.
77        * 
78        * @param graphics
79        *            Graphics2D context into which the
80        *            {@link BasicGraphicalElement BaiscGraphicalElements} are
81        *            painted.
82        * @return this instance.
83        */
84       public Graphics2DRenderer init(Graphics2D graphics) {
85           _graphics = graphics;
86           _defaultColor = graphics.getColor(); // the foreground color
87           return this;
88       }

90       /**
91        * Starts rendering of the specified composite. Does nothing except if
92        * <tt>composite</tt> has a {@link ClippingShape}. In this case the Clip
93        * of the <tt>Graphics2D</tt> context becomes the clipping rectangle
94        * determined by the bounding box of the <tt>ClippingShape</tt>.
95        */
96       public void startRendering(GraphicalComposite composite) {
97           ClippingShape shape = composite.getClippingShape();
98           if (shape != null) {
99               ClippingRectangle rect = shape.getBoundingBox();
100               _graphics.clip(new Rectangle2D.Double(rect.getMinX(), rect.getMinY(), rect.getMaxX() - rect.getMinX(), rect
101                       .getMaxY()
102                       - rect.getMinY()));
103           }
104       }
105   
106       /**
107        * Finishes rendering of the specified composite. Does nothing except if
108        * <tt>composite</tt> has a {@link ClippingShape}. In this case the Clip
109        * of the <tt>Graphics2D</tt> context will be cleared.
110        */
111       public void finishRendering(GraphicalComposite composite) {
112           _graphics.setClip(null);
113       }

115       /** Paints the specified polygon into the <tt>Graphics2D</tt> context. */
116       public void render(Polygon polygon) {
117           int numberOfPoints = polygon.getNumberOfPoints();
118           if (numberOfPoints > 0) {
119               Color currentColor = _graphics.getColor();
120               GeneralPath p = new GeneralPath(GeneralPath.WIND_EVEN_ODD, numberOfPoints);
121               p.moveTo((float) polygon.getPoint(0).getX(), (float) polygon.getPoint(0).getY());
122               for (int i = 1; i < numberOfPoints; i++) {
123                   p.lineTo((float) polygon.getPoint(i).getX(), (float) polygon.getPoint(i).getY());
124               }
125               if (polygon.isClosed()) {
126                   p.closePath();
127               }
128               drawShape(p, polygon, currentColor);
129           }
130       }

132       /**
133        * Paints the specified rectangle into the current <tt>Graphics</tt>
134        * context.
135        */
136       public void render(Rectangle rectangle) {
137           Color currentColor = _graphics.getColor();
138           GraphPoint center = rectangle.getCenter();
139           double width = rectangle.getWidth();
140           double height = rectangle.getHeight();
141           Rectangle2D rect = new Rectangle2D.Double(center.getX() - 0.5 * width, center.getY() - 0.5 * height, width,
142                   height);
143           drawShape(rect, rectangle, currentColor);
144       }

146       /**
147        * Paints the specified oval into the current <tt>Graphics</tt> context.
148        */
149       public void render(Oval oval) {
150           Color currentColor = _graphics.getColor();
151           GraphPoint center = oval.getCenter();
152           double width = oval.getWidth();
153           double height = oval.getHeight();
154           Ellipse2D ellipse = new Ellipse2D.Double(center.getX() - 0.5 * width, center.getY() - 0.5 * height, width,
155                   height);
156           drawShape(ellipse, oval, currentColor);
157       }

159       private void drawShape(Shape shape, BasicGraphicalElement element, Color backupColor) {
160           GraphicAttributes attributes = element.getGraphicAttributes();
161           Color fillColor = null;
162           if (element.isClosed() && attributes instanceof FillAttributes) {
163               fillColor = ((FillAttributes) attributes).getFillColor();
164           }
165           if (fillColor != null) {
166               _graphics.setColor(fillColor);
167               _graphics.fill(shape);
168           }
169           Color lineColor = _defaultColor;
170           if (attributes instanceof LineAttributes) {
171               LineAttributes la = (LineAttributes) attributes;
172               BasicStroke stroke = new BasicStroke((float) la.getLineThickness());
173               double[] linePattern = la.getLinePattern();
174               if (linePattern != null) {
175                   float[] dash = new float[linePattern.length];
176                   for (int i = 0; i < dash.length; i++) {
177                       dash[i] = (float) la.getLinePattern()[i];
178                   }
179                   stroke = new BasicStroke(stroke.getLineWidth(), BasicStroke.CAP_BUTT, BasicStroke.JOIN_MITER, 10f,
180                           dash, 0f);
181               }
182               _graphics.setStroke(stroke);
183               if (la.getLineColor() != null || fillColor != null) {
184                   lineColor = la.getLineColor();
185               }
186           }
187           if (lineColor != null) {
188               _graphics.setColor(lineColor);
189               _graphics.draw(shape);
190           }
191           _graphics.setColor(backupColor);
192       }

194       /**
195        * Paints the specified text into the current <tt>Graphics</tt> context.
196        * <p>
197        * If the font size is zero the default font size will be used.
198        * <p>
199        * If the orientation angle is unequal zero the text will first be painted
200        * into an off-screen image and rotated. Finally, it will be drawn into the
201        * current <tt>Graphics</tt> context. Note, that only integer multiples of
202        * 90 degree rotation are performed. Other orientation angles will be
203        * adjusted to the nearest integer multiple of 90 degree.
204        */
205       public void render(Text text) {
206           final GraphicAttributes ga = text.getGraphicAttributes();
207           if (ga instanceof TextAttributes) {
208               final TextAttributes ta = (TextAttributes) ga;
209               final Color currentColor = _graphics.getColor();
210               Color fontColor = ta.getTextColor();
211               if (fontColor == null) {
212                   fontColor = _defaultColor;
213               }
214               _graphics.setColor(fontColor);
215   
216               final double scale = _graphics.getTransform().getScaleX();
217               final String str = text.getText();
218   
219               AffineTransform before = _graphics.getTransform();
220               _graphics.setTransform(new AffineTransform());
221   
222               double fs = ta.getFontSize();
223               fs = fs == 0 ? 1 : fs * scale / DEFAULT_FONT_SIZE;
224   
225               Font font = createFont(ta, 0);
226   
227               AffineTransform fontTransform = new AffineTransform();
228               fontTransform.scale(fs, fs);
229               fontTransform.rotate(-ta.getOrientationAngle() * Math.PI / 180);
230               font = font.deriveFont(fontTransform);
231               _graphics.setFont(font);
232               Rectangle2D bounds = _graphics.getFontMetrics().getStringBounds(str, _graphics);
233   
234               fontTransform.rotate(-ta.getOrientationAngle() * Math.PI / 180);
235   
236               final double yy = bounds.getHeight() + bounds.getY();
237   
238               Point2D.Double pos = new Point2D.Double(text.getPosition().getX(), text.getPosition().getY());
239               before.transform(pos, pos);
240   
241               double x = 0;
242               double y = 0;
243               if (ta.getOrientationAngle() == 0) {
244                   x = -0.5 * ta.getHorizontalAnchor().getFactor() * bounds.getWidth();
245                   y = 0.5 * ta.getVerticalAnchor().getFactor() * bounds.getHeight() - yy;
246                   x = pos.x + x;
247                   y = pos.y + y;
248               } else {
249                   x = 0.5 * ta.getVerticalAnchor().getFactor() * bounds.getHeight();
250                   y = 0.5 * ta.getHorizontalAnchor().getFactor() * bounds.getWidth();
251                   // System.err.println("yy="+y+" dx="+x+" dy="+y);
252                   // x = 0;
253                   // y = 0;
254                   x = pos.x + x;
255                   y = pos.y + y;
256               }
257   
258   //            if (ta.getOrientationAngle() == 0) {
259   ////                System.err.println("x0=" + x);
260   ////                System.err.println("y0=" + y);
261   //            } else {
262   //                System.err.println("bounds=" + bounds + " y=" + bounds.getY() + " h=" + bounds.getHeight() + " vert="
263   //                        + ta.getVerticalAnchor().getFactor()+" horz="+ta.getHorizontalAnchor().getFactor());
264   //                System.err.println("x1=" + x);
265   //                System.err.println("y1=" + y);
266   //            }
267   
268               
269               _graphics.drawString(str, (float) x, (float) y);
270               // _graphics.fillRect((int)x, (int)y, 5, 5);
271               _graphics.setTransform(before);
272               _graphics.setColor(currentColor);
273           }
274       }

276       /**
277        * Creates a font instance based on the specified text attributes and font
278        * size.
279        * 
280        * @param attributes
281        *            Text attributes (font name and style).
282        * @param size
283        *            Font size in pixel. If 0 {@link #DEFAULT_FONT_SIZE} will be
284        *            used.
285        * @return new font instance.
286        */
287       static Font createFont(TextAttributes attributes, int size) {
288           String fontName = attributes.getFontName();
289           if (fontName == null) {
290               fontName = DEFAULT_FONT_NAME;
291           }
292   
293           FontStyle fontStyle = attributes.getFontStyle();
294           if (fontStyle == null) {
295               fontStyle = DEFAULT_FONT_STYLE;
296           }
297           int style = Font.PLAIN;
298           if (fontStyle == FontStyle.BOLD) {
299               style = Font.BOLD;
300           } else if (fontStyle == FontStyle.ITALIC) {
301               style = Font.ITALIC;
302           } else if (fontStyle == FontStyle.BOLD_ITALIC) {
303               style = Font.BOLD + Font.ITALIC;
304           }
305   
306           if (size == 0) {
307               size = DEFAULT_FONT_SIZE;
308           }
309   
310           return new Font(fontName, style, size);
311       }
