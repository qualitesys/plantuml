19   package jcckit.renderer;
20   
21   import java.awt.Graphics2D;
22   
23   import jcckit.graphic.Anchor;
24   import jcckit.graphic.ClippingRectangle;
25   import jcckit.graphic.GraphPoint;
26   
27   /**
28    *  Transformation between device-independent coordinates
29    *  and standard Java coordinates. The aspect-ratio will
30    *  be the same. The position in the canvas is determined by a
31    *  {@link jcckit.graphic.Rectangle Rectangle} defining a (virtual)
32    *  paper which is placed in the canvas according to an anchor point.
33    *  Depending on the aspect ratio  of the canvas the paper width or
34    *  height occupies the canvas width or height.
35    *
36    *  @author Franz-Josef Elmer
37    */
38   public class Transformation {
39     private final double _scale, _x0, _y0;
40     
41     public String toString() {
42           return "_scale=" + _scale + " _x0=" + _x0 + " _y0=" + _y0;
43       }
44   
45     /**
46      *  Creates an instance for the specified canvas size, paper size,
47      *  and anchor points of the paper.
48      *  @param width Width of the canvas.
49      *  @param height Height of the canvas.
50      *  @param paper Rectangle defining the paper in device-independent
51      *         coordinates.
52      *  @param horizontalAnchor Horizontal anchor of the paper in the canvas.
53      *  @param verticalAnchor Vertical anchor of the paper in the canvas.
54      */
55     public Transformation(int width, int height, ClippingRectangle paper,
56                           Anchor horizontalAnchor, Anchor verticalAnchor) {
57       double pWidth = paper.getMaxX() - paper.getMinX();
58       double pHeight = paper.getMaxY() - paper.getMinY();
59       _scale = Math.min(width / pWidth, height / pHeight);
60       _x0 = 0.5 * horizontalAnchor.getFactor() * (width - _scale * pWidth)
61             - _scale * paper.getMinX();
62       _y0 = 0.5 * verticalAnchor.getFactor() * (_scale * pHeight - height)
63             + height + _scale * + paper.getMinY();
64     }
65   
66     /** Transforms the device-independent x coordinate into Java coordinates. */
67     public int transformX(double x) {
68       return trim(_scale * x + _x0);
69     }
70   
71     /** Transforms the device-independent y coordinate into Java coordinates. */
72     public int transformY(double y) {
73       return trim(_y0 - _scale * y);
74     }
75   
76     /** Transforms the device-independent width into Java width. */
77     public int transformWidth(double width) {
78       return trim(_scale * width + 0.5);
79     }
80   
81     /** Transforms the device-independent height into Java height. */
82     public int transformHeight(double height) {
83       return trim(_scale * height + 0.5);
84     }
85     
86     private static int trim(double number)
87     {
88       return number > Short.MAX_VALUE 
89                 ? Short.MAX_VALUE 
90                 : (number < Short.MIN_VALUE ? Short.MIN_VALUE : (int) number); 
91     }
92   
93     /**
94      *  Transforms a point in Java coordinates back into device-independent
95      *  coordinates.
96      */
97     public GraphPoint transformBack(int x, int y) {
98       return new GraphPoint((x - _x0) / _scale, (_y0 - y) / _scale);
99     }
100   
101   public void apply(Graphics2D g) {
102       g.translate(_x0, _y0);
103       g.scale(_scale, -_scale);
104       
105   }
106   }
