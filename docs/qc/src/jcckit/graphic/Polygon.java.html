19   package jcckit.graphic;
20   
21   import java.util.Vector;
22   
23   /**
24    *  A polygon or polyline.
25    *
26    *  @author Franz-Josef Elmer
27    */
28   public class Polygon extends BasicGraphicalElement {
29     private final Vector _points = new Vector();
30     private final boolean _closed;
31   
32     /**
33      * Creates an instance of the specified graphic attributes.
34      * @param attributes attributes of the instance to be created
35      * @param closed <tt>true</tt> if this polygon is closed.
36      */
37     public Polygon(GraphicAttributes attributes, boolean closed) {
38       super(attributes);
39       _closed = closed;
40     }
41   
42     /** Returns <tt>true</tt> if this polygon is closed. */
43     public boolean isClosed() {
44       return _closed;
45     }
46   
47     /** Returns the number points. */
48     public int getNumberOfPoints() {
49       return _points.size();
50     }
51   
52     /** Returns the point for the specified index. */
53     public GraphPoint getPoint(int index) {
54       return (GraphPoint) _points.elementAt(index);
55     }
56   
57     /** Adds a new point to the end of the list of points. */
58     public void addPoint(GraphPoint point) {
59       _points.addElement(point);
60     }
61   
62     /** Removes all points. */
63     public void removeAllPoints() {
64       _points.removeAllElements();
65     }
66   
67     /** Replaces the point at the specified index by a new one. */
68     public void replacePointAt(int index, GraphPoint point) {
69       _points.setElementAt(point, index);
70     }
71   
72     /**
73      *  Renders this line with the specified {@link Renderer}.
74      *  @param renderer An instance of {@link PolygonRenderer}.
75      *  @throws IllegalArgumentException if <tt>renderer</tt> is not
76      *          an instance of <tt>PolygonRenderer</tt>.
77      */
78     public void renderWith(Renderer renderer) {
79       if (renderer instanceof PolygonRenderer) {
80         ((PolygonRenderer) renderer).render(this);
81       } else {
82         throw new IllegalArgumentException(renderer
83                               + " does not implements PolygonRenderer.");
84       }
85     }
86   }
