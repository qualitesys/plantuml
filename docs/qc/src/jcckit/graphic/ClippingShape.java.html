19   package jcckit.graphic;
20   
21   /**
22    * Defining a clipping shape applied to all {@link GraphicalElement
23    * GraphicalElements} of a {@link GraphicalComposite}.
24    *
25    * @author Franz-Josef Elmer
26    */
27   public interface ClippingShape {
28     /**
29      * Returns <tt>true</tt> if the specified point is inside this
30      * clipping shape.
31      */
32     public boolean isInside(GraphPoint point);
33   
34     /**
35      * Returns the bounding box of this clipping shape.
36      * This method will be used by renderers who supports only
37      * rectangular clipping shapes.
38      */
39     public ClippingRectangle getBoundingBox();
40     
41     /**
42      * Returns a basic graphical element (such as {@link Rectangle}
43      * or {@link Polygon}) which may be used by renderers to
44      * define the clipping shape for the output device.
45      */
46     public BasicGraphicalElement getGraphicalElement();
47   }
