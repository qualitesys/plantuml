52       /**
53        *
54        * <p>An outline of the inner workings of this very important (and monstrous)
55        * constructor is presented here. Boundary processing is the first step
56        * of the process:</p>
57        *
58        * <ol>
59        *   <li>Copy the grid into a work grid and remove all type-on-line
60        *       and point markers from the work grid</li>
61        *   <li>Split grid into distinct shapes by plotting the grid
62        *          onto an AbstractionGrid and its getDistinctShapes() method.</li>
63        *   <li>Find all the possible boundary sets of each of the
64        *       distinct shapes. This can produce duplicate shapes (if the boundaries
65        *       are the same when filling from the inside and the outside).</li>
66        *   <li>Remove duplicate boundaries.</li>
67        *   <li>Remove obsolete boundaries. Obsolete boundaries are the ones that are
68        *       the sum of their parts when plotted as filled shapes. (see method
69        *       removeObsoleteShapes())</li>
70        *   <li>Seperate the found boundary sets to open, closed or mixed
71        *       (See CellSet class on how its done).</li>
72        *   <li>Are there any closed boundaries?
73        *        <ul>
74        *           <li>YES. Subtract all the closed boundaries from each of the
75        *           open ones. That should convert the mixed shapes into open.</li>
76        *           <li>NO. In this (harder) case, we use the method
77        *           breakTrulyMixedBoundaries() of CellSet to break boundaries
78        *           into open and closed shapes (would work in any case, but it's
79        *           probably slower than the other method). This method is based
80        *           on tracing from the lines' ends and splitting when we get to
81        *           an intersection.</li>
82        *        </ul>
83        *   </li>
84        *   <li>If we had to eliminate any mixed shapes, we seperate the found
85        *   boundary sets again to open, closed or mixed.</li>
86        * </ol>
87        *
88        * <p>At this stage, the boundary processing is all complete and we
89        * proceed with using those boundaries to create the shapes:</p>
90        *
91        * <ol>
92        *   <li>Create closed shapes.</li>
93        *   <li>Create open shapes. That's when the line end corrections are
94        *   also applied, concerning the positioning of the ends of lines
95        *   see methods connectEndsToAnchors() and moveEndsToCellEdges() of
96        *   DiagramShape.</li>
97        *   <li>Assign color codes to closed shapes.</li>
98        *   <li>Assing extended markup tags to closed shapes.</p>
99        *   <li>Create arrowheads.</p>
100        *   <li>Create point markers.</p>
101        * </ol>
102        *
103        * <p>Finally, the text processing occurs: [pending]</p>
104        *
105        * @param grid
106        * @param options
107        * @param processingOptions
108        */
109       public Diagram(TextGrid grid, ConversionOptions options, ProcessingOptions processingOptions) {
110   
111           this.cellWidth = options.renderingOptions.getCellWidth();
112           this.cellHeight = options.renderingOptions.getCellHeight();
113   
114           FontMeasurer fontMeasurer = new FontMeasurer(
115               options.renderingOptions.getFont(), options.renderingOptions.getForceFontSize());
116   
117           width = grid.getWidth() * cellWidth;
118           height = grid.getHeight() * cellHeight;
119   
120           TextGrid workGrid = new TextGrid(grid);
121           workGrid.replaceTypeOnLine();
122           workGrid.replacePointMarkersOnLine();
123           if(DEBUG) workGrid.printDebug();
124   
125           int width = grid.getWidth();
126           int height = grid.getHeight();
127   
128   
129           //split distinct shapes using AbstractionGrid
130           AbstractionGrid temp = new AbstractionGrid(workGrid, workGrid.getAllBoundaries());
131           ArrayList boundarySetsStep1 = temp.getDistinctShapes();
132   
133           if(DEBUG){
134               System.out.println("******* Distinct shapes found using AbstractionGrid *******");
135               Iterator dit = boundarySetsStep1.iterator();
136               while (dit.hasNext()) {
137                   CellSet set = (CellSet) dit.next();
138                   set.printAsGrid();
139               }
140               System.out.println("******* Same set of shapes after processing them by filling *******");
141           }
142   
143   
144           //Find all the boundaries by using the special version of the filling method
145           //(fills in a different buffer than the buffer it reads from)
146           ArrayList boundarySetsStep2 = new ArrayList();
147           Iterator boundarySetIt = boundarySetsStep1.iterator();
148           while (boundarySetIt.hasNext()) {
149               CellSet set = (CellSet) boundarySetIt.next();
150   
151               //the fill buffer keeps track of which cells have been
152               //filled already
153               TextGrid fillBuffer = new TextGrid(width * 3, height * 3);
154   
155               for(int yi = 0; yi < height * 3; yi++){
156                   for(int xi = 0; xi < width * 3; xi++){
157                       if(fillBuffer.isBlank(xi, yi)){
158   
159                           TextGrid copyGrid = new AbstractionGrid(workGrid, set).getCopyOfInternalBuffer();
160   
161                           CellSet boundaries =
162                               copyGrid
163                               .findBoundariesExpandingFrom(copyGrid.new Cell(xi, yi));
164                           if(boundaries.size() == 0) continue; //i'm not sure why these occur
165                           boundarySetsStep2.add(boundaries.makeScaledOneThirdEquivalent());
166   
167                           copyGrid = new AbstractionGrid(workGrid, set).getCopyOfInternalBuffer();
168                           CellSet filled =
169                               copyGrid
170                               .fillContinuousArea(copyGrid.new Cell(xi, yi), '*');
171                           fillBuffer.fillCellsWith(filled, '*');
172                           fillBuffer.fillCellsWith(boundaries, '-');
173   
174                           if(DEBUG){
175                               //System.out.println("Fill buffer:");
176                               //fillBuffer.printDebug();
177                               boundaries.makeScaledOneThirdEquivalent().printAsGrid();
178                               System.out.println("-----------------------------------");
179                           }
180   
181                       }
182                   }
183               }
184           }
185   
186           if (DEBUG)
187               System.out.println("******* Removed duplicates *******");
188   
189           boundarySetsStep2 = CellSet.removeDuplicateSets(boundarySetsStep2);
190   
191           if(DEBUG){
192               Iterator dit = boundarySetsStep2.iterator();
193               while (dit.hasNext()) {
194                   CellSet set = (CellSet) dit.next();
195                   set.printAsGrid();
196               }
197           }
198   
199           int originalSize = boundarySetsStep2.size();
200           boundarySetsStep2 = CellSet.removeDuplicateSets(boundarySetsStep2);
201           if(DEBUG) {
202               System.out.println(
203                   "******* Removed duplicates: there were "
204                   +originalSize
205                   +" shapes and now there are "
206                   +boundarySetsStep2.size());
207           }
208   
209   
210           //split boundaries to open, closed and mixed
211   
212           if (DEBUG)
213               System.out.println("******* First evaluation of openess *******");
214   
215           ArrayList open = new ArrayList();
216           ArrayList closed = new ArrayList();
217           ArrayList mixed = new ArrayList();
218   
219           Iterator sets = boundarySetsStep2.iterator();
220           while(sets.hasNext()){
221               CellSet set = (CellSet) sets.next();
222               int type = set.getType(workGrid);
223               if(type == CellSet.TYPE_CLOSED) closed.add(set);
224               else if(type == CellSet.TYPE_OPEN) open.add(set);
225               else if(type == CellSet.TYPE_MIXED) mixed.add(set);
226               if(DEBUG){
227                   if(type == CellSet.TYPE_CLOSED) System.out.println("Closed boundaries:");
228                   else if(type == CellSet.TYPE_OPEN) System.out.println("Open boundaries:");
229                   else if(type == CellSet.TYPE_MIXED) System.out.println("Mixed boundaries:");
230                   set.printAsGrid();
231               }
232           }
233   
234           boolean hadToEliminateMixed = false;
235   
236           if(mixed.size() > 0 && closed.size() > 0) {
237                               // mixed shapes can be eliminated by
238                               // subtracting all the closed shapes from them
239               if (DEBUG)
240                   System.out.println("******* Eliminating mixed shapes (basic algorithm) *******");
241   
242               hadToEliminateMixed = true;
243   
244               //subtract from each of the mixed sets all the closed sets
245               sets = mixed.iterator();
246               while(sets.hasNext()){
247                   CellSet set = (CellSet) sets.next();
248                   Iterator closedSets = closed.iterator();
249                   while(closedSets.hasNext()){
250                       CellSet closedSet = (CellSet) closedSets.next();
251                       set.subtractSet(closedSet);
252                   }
253                   // this is necessary because some mixed sets produce
254                   // several distinct open sets after you subtract the
255                   // closed sets from them
256                   if(set.getType(workGrid) == CellSet.TYPE_OPEN) {
257                       boundarySetsStep2.remove(set);
258                       boundarySetsStep2.addAll(set.breakIntoDistinctBoundaries(workGrid));
259                   }
260               }
261   
262           } else if(mixed.size() > 0 && closed.size() == 0) {
263                               // no closed shape exists, will have to
264                               // handle mixed shape on its own
265               // an example of this case is the following:
266               // +-----+
267               // |  A  |C                 B
268               // +  ---+-------------------
269               // |     |
270               // +-----+
271   
272               hadToEliminateMixed = true;
273   
274               if (DEBUG)
275                   System.out.println("******* Eliminating mixed shapes (advanced algorithm for truly mixed shapes) *******");
276   
277               sets = mixed.iterator();
278               while(sets.hasNext()){
279                   CellSet set = (CellSet) sets.next();
280                   boundarySetsStep2.remove(set);
281                   boundarySetsStep2.addAll(set.breakTrulyMixedBoundaries(workGrid));
282               }
283   
284           } else {
285               if (DEBUG)
286                   System.out.println("No mixed shapes found. Skipped mixed shape elimination step");
287           }
288   
289   
290           if(hadToEliminateMixed){
291               if (DEBUG)
292                   System.out.println("******* Second evaluation of openess *******");
293   
294               //split boundaries again to open, closed and mixed
295               open = new ArrayList();
296               closed = new ArrayList();
297               mixed = new ArrayList();
298   
299               sets = boundarySetsStep2.iterator();
300               while(sets.hasNext()){
301                   CellSet set = (CellSet) sets.next();
302                   int type = set.getType(workGrid);
303                   if(type == CellSet.TYPE_CLOSED) closed.add(set);
304                   else if(type == CellSet.TYPE_OPEN) open.add(set);
305                   else if(type == CellSet.TYPE_MIXED) mixed.add(set);
306                   if(DEBUG){
307                       if(type == CellSet.TYPE_CLOSED) System.out.println("Closed boundaries:");
308                       else if(type == CellSet.TYPE_OPEN) System.out.println("Open boundaries:");
309                       else if(type == CellSet.TYPE_MIXED) System.out.println("Mixed boundaries:");
310                       set.printAsGrid();
311                   }
312               }
313           }
314   
315           boolean removedAnyObsolete = removeObsoleteShapes(workGrid, closed);
316   
317           boolean allCornersRound = false;
318           if(processingOptions.areAllCornersRound()) allCornersRound = true;
319   
320           //make shapes from the boundary sets
321           //make closed shapes
322           ArrayList closedShapes = new ArrayList();
323           sets = closed.iterator();
324           while(sets.hasNext()){
325               CellSet set = (CellSet) sets.next();
326               DiagramComponent shape = DiagramComponent.createClosedFromBoundaryCells(workGrid, set, cellWidth, cellHeight, allCornersRound);
327               if(shape != null){
328                   if(shape instanceof DiagramShape){
329                       addToShapes((DiagramShape) shape);
330                       closedShapes.add(shape);
331                   } else if(shape instanceof CompositeDiagramShape)
332                       addToCompositeShapes((CompositeDiagramShape) shape);
333               }
334           }
335   
336           if(processingOptions.performSeparationOfCommonEdges())
337               separateCommonEdges(closedShapes);
338   
339           //make open shapes
340           sets = open.iterator();
341           while(sets.hasNext()){
342               CellSet set = (CellSet) sets.next();
343               if(set.size() == 1){ //single cell "shape"
344                   TextGrid.Cell cell = (TextGrid.Cell) set.getFirst();
345                   if(!grid.cellContainsDashedLineChar(cell)) {
346                       DiagramShape shape = DiagramShape.createSmallLine(workGrid, cell, cellWidth, cellHeight);
347                       if(shape != null) {
348                           addToShapes(shape);
349                           shape.connectEndsToAnchors(workGrid, this);
350                       }
351                   }
352               } else { //normal shape
353                   DiagramComponent shape =
354                       CompositeDiagramShape
355                           .createOpenFromBoundaryCells(
356                                   workGrid, set, cellWidth, cellHeight, allCornersRound);
357   
358                   if(shape != null){
359                       if(shape instanceof CompositeDiagramShape){
360                           addToCompositeShapes((CompositeDiagramShape) shape);
361                           ((CompositeDiagramShape) shape).connectEndsToAnchors(workGrid, this);
362                       } else if(shape instanceof DiagramShape) {
363                           addToShapes((DiagramShape) shape);
364                           ((DiagramShape) shape).connectEndsToAnchors(workGrid, this);
365                           ((DiagramShape) shape).moveEndsToCellEdges(grid, this);
366                       }
367                   }
368   
369               }
370           }
371   
372           //assign color codes to shapes
373           //TODO: text on line should not change its color
374           //TODO: each color tag should be assigned to the smallest containing shape (like shape tags)

386               }
387           }
388

390           Iterator cellTagPairs = grid.findMarkupTags().iterator();
391           while(cellTagPairs.hasNext()){
392               TextGrid.CellTagPair pair =
393                   (TextGrid.CellTagPair) cellTagPairs.next();
394               ShapePoint point =
395                   new ShapePoint(getCellMidX(pair.cell), getCellMidY(pair.cell));
396   
397               //find the smallest shape that contains the tag
398               DiagramShape containingShape = null;
399               Iterator shapes = getShapes().iterator();
400               while(shapes.hasNext()){
401                   DiagramShape shape = (DiagramShape) shapes.next();
402                   if(shape.contains(point)){
403                       if(containingShape == null){
404                           containingShape = shape;
405                       } else {
406                           if(shape.isSmallerThan(containingShape)){
407                               containingShape = shape;
408                           }
409                       }
410                   }
411               }
412   
413               //this tag is not within a shape, skip
414               if(containingShape == null) continue;
415   
416               //TODO: the code below could be a lot more concise
417               if(pair.tag.equals("d")){
418                   CustomShapeDefinition def =
419                       processingOptions.getFromCustomShapes("d");
420                   if(def == null)
421                       containingShape.setType(DiagramShape.TYPE_DOCUMENT);
422                   else {
423                       containingShape.setType(DiagramShape.TYPE_CUSTOM);
424                       containingShape.setDefinition(def);
425                   }
426               } else if(pair.tag.equals("s")){
427                   CustomShapeDefinition def =
428                       processingOptions.getFromCustomShapes("s");
429                   if(def == null)
430                       containingShape.setType(DiagramShape.TYPE_STORAGE);
431                   else {
432                       containingShape.setType(DiagramShape.TYPE_CUSTOM);
433                       containingShape.setDefinition(def);
434                   }
435               } else if(pair.tag.equals("io")){
436                   CustomShapeDefinition def =
437                       processingOptions.getFromCustomShapes("io");
438                   if(def == null)
439                       containingShape.setType(DiagramShape.TYPE_IO);
440                   else {
441                       containingShape.setType(DiagramShape.TYPE_CUSTOM);
442                       containingShape.setDefinition(def);
443                   }
444               } else if(pair.tag.equals("c")){
445                   CustomShapeDefinition def =
446                       processingOptions.getFromCustomShapes("c");
447                   if(def == null)
448                       containingShape.setType(DiagramShape.TYPE_DECISION);
449                   else {
450                       containingShape.setType(DiagramShape.TYPE_CUSTOM);
451                       containingShape.setDefinition(def);
452                   }
453               } else if(pair.tag.equals("mo")){
454                   CustomShapeDefinition def =
455                       processingOptions.getFromCustomShapes("mo");
456                   if(def == null)
457                       containingShape.setType(DiagramShape.TYPE_MANUAL_OPERATION);
458                   else {
459                       containingShape.setType(DiagramShape.TYPE_CUSTOM);
460                       containingShape.setDefinition(def);
461                   }
462               } else if(pair.tag.equals("tr")){
463                   CustomShapeDefinition def =
464                       processingOptions.getFromCustomShapes("tr");
465                   if(def == null)
466                       containingShape.setType(DiagramShape.TYPE_TRAPEZOID);

785       private void separateCommonEdges(ArrayList shapes){
786   
787           float offset = getMinimumOfCellDimension() / 5;
788   
789           ArrayList<ShapeEdge> edges = new ArrayList<ShapeEdge>();
790   
791           //get all adges
792           Iterator it = shapes.iterator();
793           while (it.hasNext()) {
794               DiagramShape shape = (DiagramShape) it.next();
795               edges.addAll(shape.getEdges());
796           }
797   
798           //group edges into pairs of touching edges
799           ArrayList<Pair<ShapeEdge, ShapeEdge>> listOfPairs = new ArrayList<Pair<ShapeEdge, ShapeEdge>>();
800           it = edges.iterator();
801   
802           //all-against-all touching test for the edges
803           int startIndex = 1; //skip some to avoid duplicate comparisons and self-to-self comparisons
804   
805           while(it.hasNext()){
806               ShapeEdge edge1 = (ShapeEdge) it.next();
807   
808               for(int k = startIndex; k < edges.size(); k++) {
809                   ShapeEdge edge2 =  edges.get(k);
810   
811                   if(edge1.touchesWith(edge2)) {
812                       listOfPairs.add(new Pair<ShapeEdge, ShapeEdge>(edge1, edge2));
813                   }
814               }
815               startIndex++;
816           }
817   
818           ArrayList<ShapeEdge> movedEdges = new ArrayList<ShapeEdge>();
819   
820           //move equivalent edges inwards
821           it = listOfPairs.iterator();
822           while(it.hasNext()){
823               Pair<ShapeEdge, ShapeEdge> pair = (Pair<ShapeEdge, ShapeEdge>) it.next();
824               if(!movedEdges.contains(pair.first)) {
825                   pair.first.moveInwardsBy(offset);
826                   movedEdges.add(pair.first);
827               }
828               if(!movedEdges.contains(pair.second)) {
829                   pair.second.moveInwardsBy(offset);
830                   movedEdges.add(pair.second);
831               }
832           }
