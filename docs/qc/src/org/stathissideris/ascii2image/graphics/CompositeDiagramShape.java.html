57       public static DiagramComponent createOpenFromBoundaryCells(
58               final TextGrid grid,
59               final CellSet boundaryCells,
60               final int cellWidth,
61               final int cellHeight,
62               boolean allRound) {
63           
64           if(boundaryCells.getType(grid) != CellSet.TYPE_OPEN) throw new IllegalArgumentException("This shape is closed and cannot be handled by this method");
65           if(boundaryCells.size() == 0) return null;
66   
67           
68           CompositeDiagramShape compositeShape = new CompositeDiagramShape();
69           TextGrid workGrid = new TextGrid(grid.getWidth(), grid.getHeight());
70           grid.copyCellsTo(boundaryCells, workGrid);
71   
72           if(DEBUG) {
73               System.out.println("Making composite shape from grid:");
74               workGrid.printDebug();
75           }
76           
77           
78           CellSet visitedCells = new CellSet();
79           
80           List<DiagramShape> shapes = new ArrayList<DiagramShape>(100);
81           
82           for(TextGrid.Cell cell : boundaryCells) {
83               if(workGrid.isLinesEnd(cell)) {
84                   CellSet nextCells = workGrid.followCell(cell);
85                   shapes.addAll(growEdgesFromCell(workGrid, cellWidth, cellHeight, allRound, nextCells.getFirst(), cell, visitedCells));
86                   break;
87               }
88           }
89           
90           //dashed shapes should "infect" the rest of the shapes
91           boolean dashedShapeExists = false;
92           for(DiagramShape shape : shapes)
93               if(shape.isStrokeDashed())
94                   dashedShapeExists = true;
95           
96           for(DiagramShape shape : shapes) {
97               if(dashedShapeExists) shape.setStrokeDashed(true);
98               compositeShape.addToShapes(shape);
99           }
100           
101           return compositeShape;
102       }

105       private static List<DiagramShape> growEdgesFromCell(
106               TextGrid workGrid,
107               final int cellWidth,
108               final int cellHeight,
109               boolean allRound,
110               TextGrid.Cell cell, 
111               TextGrid.Cell previousCell, 
112               CellSet visitedCells) {
113           
114           List<DiagramShape> result = new ArrayList<DiagramShape>(50); 
115           
116           visitedCells.add(previousCell);
117           
118           DiagramShape shape = new DiagramShape();
119           
120           shape.addToPoints(makePointForCell(previousCell, workGrid, cellWidth, cellHeight, allRound));
121           if(DEBUG) System.out.println("point at "+previousCell+" (call from line: "+DebugUtils.getLineNumber()+")");
122           if(workGrid.cellContainsDashedLineChar(previousCell)) shape.setStrokeDashed(true);
123   
124           boolean finished = false;
125           int nb = 0;
126           while(!finished) {
127               // https://github.com/plantuml/plantuml/issues/1036
128               nb++;
129               if (nb > 1000)
130                   return result;
131               visitedCells.add(cell);
132               if(workGrid.isPointCell(cell)) {
133                   if(DEBUG) System.out.println("point at "+cell+" (call from line: "+DebugUtils.getLineNumber()+")");
134                   shape.addToPoints(makePointForCell(cell, workGrid, cellWidth, cellHeight, allRound));
135               }
136               
137               if(workGrid.cellContainsDashedLineChar(cell)) shape.setStrokeDashed(true);
138   
139               if(workGrid.isLinesEnd(cell)){
140                   finished = true;
141                   if(DEBUG) System.out.println("finished shape");
142               }
143               
144               CellSet nextCells = workGrid.followCell(cell, previousCell);
145               if(nextCells.size() == 1) {
146                   previousCell = cell;
147                   cell = (TextGrid.Cell) nextCells.getFirst();
148               } else if(nextCells.size() > 1) {//3- or 4- way intersection
149                   finished = true;
150                   if(DEBUG) System.out.println("finished shape");
151                   for(TextGrid.Cell nextCell : nextCells)
152                       result.addAll(growEdgesFromCell(workGrid, cellWidth, cellHeight, allRound, nextCell, cell, visitedCells));
153               }
154           }
155           
156           result.add(shape);
157           return result;
158       }

167       public DiagramComponent connectLines(){
168           CompositeDiagramShape result = new CompositeDiagramShape();
169   
170           //find all lines
171           ArrayList<DiagramShape> lines = new ArrayList<DiagramShape>();
172           Iterator it = shapes.iterator();
173           while(it.hasNext()){
174               DiagramShape shape = (DiagramShape) it.next();
175               if(shape.getPoints().size() == 2){
176                   lines.add(shape);
177               }
178           }
179           
180           it = lines.iterator();
181           while(it.hasNext()){
182               DiagramShape line1 = (DiagramShape) it.next();
183               Iterator it2 = lines.iterator();
184               while(it2.hasNext()){
185                   DiagramShape line2 = (DiagramShape) it.next();
186                   ShapePoint commonPoint = null;
187                   ShapePoint line1UncommonPoint = null;
188                   ShapePoint line2UncommonPoint = null;
189                   if(line1.getPoint(0).equals(line2.getPoint(0))){
190                       commonPoint = line1.getPoint(0);
191                       line1UncommonPoint = line1.getPoint(1);
192                       line2UncommonPoint = line2.getPoint(1);
193                   }
194                   if(line1.getPoint(0).equals(line2.getPoint(1))){
195                       commonPoint = line1.getPoint(0);
196                       line1UncommonPoint = line1.getPoint(1);
197                       line2UncommonPoint = line2.getPoint(0);
198                   }
199                   if(line1.getPoint(1).equals(line2.getPoint(0))){
200                       commonPoint = line1.getPoint(1);
201                       line1UncommonPoint = line1.getPoint(0);
202                       line2UncommonPoint = line2.getPoint(1);
203                   }
204                   if(line1.getPoint(1).equals(line2.getPoint(1))){
205                       commonPoint = line1.getPoint(1);
206                       line1UncommonPoint = line1.getPoint(0);
207                       line2UncommonPoint = line2.getPoint(0);
208                   }
209                   if(commonPoint != null){
210                       
211                   }
212               }
213           }
214           
215           return result;

228       private static DiagramShape makeLine(TextGrid grid, TextGrid.Cell start, TextGrid.Cell end, int cellWidth, int cellHeight){
229           DiagramShape line = new DiagramShape();
230           
231           if(grid.isHorizontalLine(start)){
232               if(start.isWestOf(end)){
233                   line.addToPoints(new ShapePoint(
234                               Diagram.getCellMinX(start, cellWidth),
235                               Diagram.getCellMidY(start, cellHeight)));
236               } else {
237                   line.addToPoints(new ShapePoint(
238                               Diagram.getCellMaxX(start, cellWidth),
239                               Diagram.getCellMidY(start, cellHeight)));
240               }
241           } else if(grid.isVerticalLine(start)){
242               if(start.isNorthOf(end)){
243                   line.addToPoints(new ShapePoint(
244                               Diagram.getCellMidX(start, cellWidth),
245                               Diagram.getCellMinY(start, cellHeight)));
246               } else {
247                   line.addToPoints(new ShapePoint(
248                               Diagram.getCellMidX(start, cellWidth),
249                               Diagram.getCellMaxY(start, cellHeight)));
250               }            
251           } else { //corner
252               if(DEBUG) System.out.println("Corner");
253               int type = (grid.isRoundCorner(start))?ShapePoint.TYPE_ROUND:ShapePoint.TYPE_NORMAL;
254               line.addToPoints(new ShapePoint(
255                           Diagram.getCellMidX(start, cellWidth),
256                           Diagram.getCellMidY(start, cellHeight),
257                           type));
258               
259           }
260   
261           if(grid.isHorizontalLine(end)){
262               if(start.isWestOf(start)){
263                   line.addToPoints(new ShapePoint(
264                               Diagram.getCellMinX(end, cellWidth),
265                               Diagram.getCellMidY(end, cellHeight)));
266               } else {
267                   line.addToPoints(new ShapePoint(
268                               Diagram.getCellMaxX(end, cellWidth),
269                               Diagram.getCellMidY(end, cellHeight)));
270               }
271           } else if(grid.isVerticalLine(end)){
272               if(start.isNorthOf(start)){
273                   line.addToPoints(new ShapePoint(
274                               Diagram.getCellMidX(end, cellWidth),
275                               Diagram.getCellMinY(end, cellHeight)));
276               } else {
277                   line.addToPoints(new ShapePoint(
278                               Diagram.getCellMidX(end, cellWidth),
279                               Diagram.getCellMaxY(end, cellHeight)));
280               }            
281           } else { //corner
282               int type = (grid.isRoundCorner(end))?ShapePoint.TYPE_ROUND:ShapePoint.TYPE_NORMAL;
283               if(DEBUG) System.out.println("Corner");
284               line.addToPoints(new ShapePoint(
285                           Diagram.getCellMidX(end, cellWidth),
286                           Diagram.getCellMidY(end, cellHeight),
287                           type));
288               
289           }
290   
291           
292           return line;
293       }
