56       private float getDistanceFromOrigin() {
57           int type = this.getType();
58           if(type == TYPE_SLOPED)
59               throw new RuntimeException("Cannot calculate distance of sloped edge from origin");
60           if(type == TYPE_HORIZONTAL)
61               return startPoint.y;
62           return startPoint.x; //vertical
63       }

65       //TODO: moveInwardsBy() not implemented
66       public void moveInwardsBy(float offset){
67           int type = this.getType();
68           if(type == TYPE_SLOPED)
69               throw new RuntimeException("Cannot move a sloped egde inwards: "+this);
70           
71           float xOffset = 0;
72           float yOffset = 0;
73           
74           ShapePoint middle = getMiddle();
75           GeneralPath path = owner.makeIntoPath();
76           if(type == TYPE_HORIZONTAL){
77               xOffset = 0;
78               ShapePoint up = new ShapePoint(middle.x, middle.y - 0.05f);
79               ShapePoint down = new ShapePoint(middle.x, middle.y + 0.05f);
80               if(path.contains(up)) yOffset = -offset;
81               else if(path.contains(down)) yOffset = offset;
82           } else if(type == TYPE_VERTICAL){
83               yOffset = 0;
84               ShapePoint left = new ShapePoint(middle.x - 0.05f, middle.y);
85               ShapePoint right = new ShapePoint(middle.x + 0.05f, middle.y);
86               if(path.contains(left)) xOffset = -offset;
87               else if(path.contains(right)) xOffset = offset;
88           }
89           if(DEBUG) System.out.println("Moved edge "+this+" by "+xOffset+", "+yOffset);
90           translate(xOffset, yOffset);
91       }

93       public void translate(float dx, float dy){
94           startPoint.x += dx;
95           startPoint.y += dy;
96           endPoint.x += dx;
97           endPoint.y += dy;
98       }

104           );
105       }

161       public boolean equals(Object object){
162           if(!(object instanceof ShapeEdge)) return false;
163           ShapeEdge edge = (ShapeEdge) object;
164           if(startPoint.equals(edge.getStartPoint())
165               && endPoint.equals(edge.getEndPoint())) return true;
166           if(startPoint.equals(edge.getEndPoint())
167               && endPoint.equals(edge.getStartPoint())) return true;
168           return false;
169       }

171       public boolean touchesWith(ShapeEdge other){
172           if(this.equals(other)) return true;
173           
174           if(this.isHorizontal() && other.isVertical()) return false;
175           if(other.isHorizontal() && this.isVertical()) return false;
176           
177           if(this.getDistanceFromOrigin() != other.getDistanceFromOrigin()) return false;
178   
179           //covering this corner case (should produce false):
180           //      ---------
181           //              ---------
182           
183           ShapeEdge first = new ShapeEdge(this);
184           ShapeEdge second = new ShapeEdge(other);
185           
186           if(first.isVertical()) {
187               first.changeAxis();
188               second.changeAxis();
189           }
190           
191           first.fixDirection();
192           second.fixDirection();
193           
194           if(first.startPoint.x > second.startPoint.x) {
195               ShapeEdge temp = first;
196               first = second;
197               second = temp;
198           }
199           
200           if(first.endPoint.equals(second.startPoint)) return false;
201           
202           // case 1:
203           // ----------
204           //      -----------
205           
206           // case 2:
207           //         ------
208           // -----------------
209           
210           if(this.startPoint.isWithinEdge(other) || this.endPoint.isWithinEdge(other)) return true;
211           if(other.startPoint.isWithinEdge(this) || other.endPoint.isWithinEdge(this)) return true;    
212           
213           
214           return false;
215       }

217       private void changeAxis(){
218           ShapePoint temp = new ShapePoint(startPoint);
219           startPoint = new ShapePoint(endPoint.y, endPoint.x);
220           endPoint = new ShapePoint(temp.y, temp.x);
221       }
222       
223       /**
224        * if horizontal flips start and end points so that start is left of end
225        * if verical flips start and end points so that start is over of end
226        *
227        */
228       private void fixDirection(){
229           if(isHorizontal()) {
230               if(startPoint.x > endPoint.x) flipDirection();
231           } else if(isVertical()) {
232               if(startPoint.y > endPoint.y) flipDirection();
233           } else {
234               throw new RuntimeException("Cannot fix direction of sloped egde");
235           }
236       }
237       
238       private void flipDirection(){
239           ShapePoint temp = startPoint;
240           startPoint = endPoint;
241           endPoint = temp;
242       }

245           if(startPoint.y == endPoint.y) return true;
246           return false;
247       }

249       public boolean isVertical(){
250           if(startPoint.x == endPoint.x) return true;
251           return false;
252       }
