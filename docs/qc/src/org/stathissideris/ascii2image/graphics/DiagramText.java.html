21   package org.stathissideris.ascii2image.graphics;
22   
23   import java.awt.Color;
24   import java.awt.Font;
25   import java.awt.geom.Rectangle2D;

27   /**
28    * 
29    * @author Efstathios Sideris
30    */
31   public class DiagramText extends DiagramComponent {
32   
33       public static final Color DEFAULT_COLOR = Color.black;
34       
35       private String text;
36       private Font font;
37       private int xPos, yPos;
38       private Color color = Color.black;
39       private boolean isTextOnLine = false;
40       private boolean hasOutline = false;
41       private Color outlineColor = Color.white;
42       private final FontMeasurer fontMeasurer;
43   
44       public DiagramText(int x, int y, String text, Font font, FontMeasurer fontMeasurer){
45           if(text == null) throw new IllegalArgumentException("DiagramText cannot be initialised with a null string");
46           if(font == null) throw new IllegalArgumentException("DiagramText cannot be initialised with a null font");
47   
48           this.xPos = x;
49           this.yPos = y;
50           this.text = text;
51           this.font = font;
52           this.fontMeasurer = fontMeasurer;
53       }
54   
55       public void centerInBounds(Rectangle2D bounds){
56           centerHorizontallyBetween((int) bounds.getMinX(), (int) bounds.getMaxX());
57           centerVerticallyBetween((int) bounds.getMinY(), (int) bounds.getMaxY());
58       }
59   
60       public void centerHorizontallyBetween(int minX, int maxX){
61           int width = fontMeasurer.getWidthFor(text, font);
62           int center = Math.abs(maxX - minX) / 2;
63           xPos += Math.abs(center - width / 2);
64           
65       }
66   
67       public void centerVerticallyBetween(int minY, int maxY){
68           int zHeight = fontMeasurer.getZHeight(font);
69           int center = Math.abs(maxY - minY) / 2;
70           yPos -= Math.abs(center - zHeight / 2);
71       }
72   
73       public void alignRightEdgeTo(int x){
74           int width = fontMeasurer.getWidthFor(text, font);
75           xPos = x - width;
76       }
77   
78   
79       /**
80        * @return
81        */
82       public Color getColor() {
83           return color;
84       }
85   
86       /**
87        * @return
88        */
89       public Font getFont() {
90           return font;
91       }
92   
93       /**
94        * @return
95        */
96       public String getText() {
97           return text;
98       }
99   
100       /**
101        * @return
102        */
103       public int getXPos() {
104           return xPos;
105       }
106   
107       /**
108        * @return
109        */
110       public int getYPos() {
111           return yPos;
112       }
113   
114       /**
115        * @param color
116        */
117       public void setColor(Color color) {
118           this.color = color;
119       }
120   
121       /**
122        * @param font
123        */
124       public void setFont(Font font) {
125           this.font = font;
126       }
127   
128       /**
129        * @param string
130        */
131       public void setText(String string) {
132           text = string;
133       }
134   
135       /**
136        * @param i
137        */
138       public void setXPos(int i) {
139           xPos = i;
140       }
141   
142       /**
143        * @param i
144        */
145       public void setYPos(int i) {
146           yPos = i;
147       }
148   
149       public Rectangle2D getBounds(){
150           Rectangle2D bounds = fontMeasurer.getBoundsFor(text, font);
151           bounds.setRect(
152               bounds.getMinX() + xPos,
153               bounds.getMinY() + yPos,
154               bounds.getWidth(),
155               bounds.getHeight());
156           return bounds;
157       }
158       
159       public String toString(){
160           return "DiagramText, at ("+xPos+", "+yPos+"), within "+getBounds()+" '"+text+"', "+color+" "+font;
161       }
162   
163       /**
164        * @return
165        */
166       public boolean isTextOnLine() {
167           return isTextOnLine;
168       }
169   
170       /**
171        * @param b
172        */
173       public void setTextOnLine(boolean b) {
174           isTextOnLine = b;
175       }
176   
177       public boolean hasOutline() {
178           return hasOutline;
179       }
180   
181       public void setHasOutline(boolean hasOutline) {
182           this.hasOutline = hasOutline;
183       }
184   
185       public Color getOutlineColor() {
186           return outlineColor;
187       }
188

190           this.outlineColor = outlineColor;
191       }
