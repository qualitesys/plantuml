154       public static DiagramShape createSmallLine(TextGrid grid, TextGrid.Cell cell, int cellXSize, int cellYSize) {
155           if (grid.isLine(cell)) {
156               DiagramShape shape = new DiagramShape();
157               if (grid.isHorizontalLine(cell)) {
158                   shape.addToPoints(
159                       new ShapePoint(
160                           cell.x * cellXSize,
161                           cell.y * cellYSize + cellYSize / 2));
162                   shape.addToPoints(
163                       new ShapePoint(
164                           cell.x * cellXSize + cellXSize - 1,
165                           cell.y * cellYSize + cellYSize / 2));
166               } else if (grid.isVerticalLine(cell)) {
167                   shape.addToPoints(
168                       new ShapePoint(
169                           cell.x * cellXSize + cellXSize / 2,
170                           cell.y * cellYSize));
171                   shape.addToPoints(
172                       new ShapePoint(
173                           cell.x * cellXSize + cellXSize / 2,
174                           cell.y * cellYSize + cellYSize - 1));
175               }
176               
177               //the -1 above, make a difference: the second point
178               //should not fall into the next cell, because this
179               //results in a failure of a proper end-of-line
180               //plotting correction
181               return shape;
182           }
183           return null;
184       }

875           path.curveTo(bottom.x - xOff, bottom.y, left.x, left.y + yOff, left.x, left.y);
876           path.curveTo(left.x, left.y - yOff, top.x - xOff, top.y, top.x, top.y);
