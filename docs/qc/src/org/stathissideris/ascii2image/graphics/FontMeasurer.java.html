31    * 
32    * @author Efstathios Sideris
33    */
34   public class FontMeasurer {
35   
36       private final Font baseFont;
37       private boolean forceFontSize;
38       private FontRenderContext fakeRenderContext;
39       private Graphics2D fakeGraphics;
40   
41       public FontMeasurer(Font font, boolean forceFontSize){
42           this.baseFont = font;
43           this.forceFontSize = forceFontSize;
44   
45           BufferedImage image = new BufferedImage(1,1, BufferedImage.TYPE_INT_RGB);
46           fakeGraphics = image.createGraphics();
47           
48           fakeRenderContext = fakeGraphics.getFontRenderContext();
49       }
50   
51       public int getWidthFor(String str, int pixelHeight){
52           Font font = getFontFor(pixelHeight);
53           Rectangle2D rectangle = font.getStringBounds(str, fakeRenderContext);
54           return (int) rectangle.getWidth();
55       }
56   
57       public int getHeightFor(String str, int pixelHeight){
58           Font font = getFontFor(pixelHeight);
59           Rectangle2D rectangle = font.getStringBounds(str, fakeRenderContext);
60           return (int) rectangle.getHeight();
61       }
62   
63       public int getWidthFor(String str, Font font){
64           Rectangle2D rectangle = font.getStringBounds(str, fakeRenderContext);
65           return (int) rectangle.getWidth();
66       }
67   
68       public int getHeightFor(String str, Font font){
69           Rectangle2D rectangle = font.getStringBounds(str, fakeRenderContext);
70           return (int) rectangle.getHeight();
71       }
72   
73       public Rectangle2D getBoundsFor(String str, Font font){
74           return font.getStringBounds(str, fakeRenderContext);
75       }
76   
77       public int getAscent(Font font){
78           fakeGraphics.setFont(font);
79           FontMetrics metrics = fakeGraphics.getFontMetrics();
80           return metrics.getAscent();
81       }
82   
83       public int getZHeight(Font font){
84           return (int) font.createGlyphVector(fakeRenderContext, "Z").getOutline().getBounds().getHeight();
85       }
86   
87       public Font getFontFor(final int maxWidth, final String string){
88           if (forceFontSize)
89           {
90               return baseFont;
91           }
92           else
93           {
94               FontPredicate predicate = new FontPredicate() {
95                   @Override
96                   public boolean test(Font font)
97                   {
98                       int width = getWidthFor(string, font);
99                       return width > maxWidth;
100                   }
101               };
102               return deriveFont(predicate, 1.0f);
103           }
104       }
105   
106       public Font getFontFor(final int pixelHeight){
107           if (forceFontSize)
108           {
109               return baseFont;
110           }
111           else
112           {
113               FontPredicate predicate = new FontPredicate() {
114                   @Override
115                   public boolean test(Font font)
116                   {
117                       //ascent is the distance between the baseline and the tallest character
118                       int ascent = getAscent(font);
119                       return ascent > pixelHeight;
120                   }
121               };
122               return deriveFont(predicate, 0.5f);
123           }
124       }
125   
126       private Font deriveFont(FontPredicate predicate, float sizeDelta)
127       {
128           Font currentFont = baseFont;
129           float size = baseFont.getSize2D();
130   
131           int direction; //direction of size change (towards smaller or bigger)
132           if(predicate.test(currentFont)){
133               currentFont = currentFont.deriveFont(size - 1f);
134               size--;
135               direction = -1;
136           } else {
137               currentFont = currentFont.deriveFont(size + 1f);
138               size++;
139               direction = 1;
140           }
141   
142           while(size > 0){
143               currentFont = currentFont.deriveFont(size);
144               //rectangle = currentFont.getStringBounds(testString, frc);
145               if (direction == 1) {
146                   if (predicate.test(currentFont)) {
147                       size = size - sizeDelta;
148                       return currentFont.deriveFont(size);
149                   } else {
150                       size = size + sizeDelta;
151                   }
152               } else {
153                   if (!predicate.test(currentFont)) {
154                       return currentFont;
155                   } else {
156                       size = size - sizeDelta;
157                   }
158               }
159           }
160           return null;
161       }
162   
163       private interface FontPredicate {
164           boolean test(Font font);
165       }
166   }
