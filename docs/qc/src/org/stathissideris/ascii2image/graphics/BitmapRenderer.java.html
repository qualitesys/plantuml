74       public RenderedImage render(Diagram diagram, BufferedImage image,  RenderingOptions options){
75           RenderedImage renderedImage = image;
76           Graphics2D g2 = image.createGraphics();
77   
78           Object antialiasSetting = RenderingHints.VALUE_ANTIALIAS_OFF;
79           if(options.performAntialias())
80               antialiasSetting = RenderingHints.VALUE_ANTIALIAS_ON;
81           
82           g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, antialiasSetting);
83   
84           g2.setColor(options.getBackgroundColor());
85           //TODO: find out why the next line does not work
86           g2.fillRect(0, 0, image.getWidth()+10, image.getHeight()+10);
87           /*for(int y = 0; y < diagram.getHeight(); y ++)
88               g2.drawLine(0, y, diagram.getWidth(), y);*/
89           
90           g2.setStroke(new BasicStroke(1, BasicStroke.CAP_SQUARE, BasicStroke.JOIN_ROUND));
91   
92           ArrayList shapes = diagram.getAllDiagramShapes();
93   
94           if(DEBUG) System.out.println("Rendering "+shapes.size()+" shapes (groups flattened)");
95   
96           Iterator shapesIt;
97           if(options.dropShadows()){
98               //render shadows
99               shapesIt = shapes.iterator();
100               while(shapesIt.hasNext()){
101                   DiagramShape shape = (DiagramShape) shapesIt.next();
102   
103                   if(shape.getPoints().isEmpty()) continue;
104   
105                   //GeneralPath path = shape.makeIntoPath();
106                   GeneralPath path;
107                   path = shape.makeIntoRenderPath(diagram);            
108                               
109                   float offset = diagram.getMinimumOfCellDimension() / 3.333f;
110               
111                   if(path != null
112                           && shape.dropsShadow()
113                           && shape.getType() != DiagramShape.TYPE_CUSTOM){
114                       GeneralPath shadow = new GeneralPath(path);
115                       AffineTransform translate = new AffineTransform();
116                       translate.setToTranslation(offset, offset);
117                       shadow.transform(translate);
118                       g2.setColor(new Color(150,150,150));
119                       g2.fill(shadow);
120                   
121                   }
122               }
123   
124           
125               //blur shadows
126           
127               if(true) {
128                   int blurRadius = 6;
129                   int blurRadius2 = blurRadius * blurRadius;
130                   float blurRadius2F = blurRadius2;
131                   float weight = 1.0f / blurRadius2F;
132                   float[] elements = new float[blurRadius2];
133                   for (int k = 0; k < blurRadius2; k++)
134                       elements[k] = weight;
135                   Kernel myKernel = new Kernel(blurRadius, blurRadius, elements);
136   
137                   //if EDGE_NO_OP is not selected, EDGE_ZERO_FILL is the default which creates a black border 
138                   ConvolveOp simpleBlur =
139                       new ConvolveOp(myKernel, ConvolveOp.EDGE_NO_OP, null);
140                                   
141                   BufferedImage destination =
142                       new BufferedImage(
143                           image.getWidth(),
144                           image.getHeight(),
145                           image.getType());
146   
147                   simpleBlur.filter(image, (BufferedImage) destination);
148   
149                   //destination = destination.getSubimage(blurRadius/2, blurRadius/2, image.getWidth(), image.getHeight()); 
150                   g2 = (Graphics2D) destination.getGraphics();
151                   g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, antialiasSetting);
152                   renderedImage = (RenderedImage) destination;
153               }
154           }
155   
156           
157           //fill and stroke
158           
159           float dashInterval = Math.min(diagram.getCellWidth(), diagram.getCellHeight()) / 2;
160           //Stroke normalStroke = g2.getStroke();
161           
162           float strokeWeight = diagram.getMinimumOfCellDimension() / 10;
163           
164           normalStroke =
165             new BasicStroke(
166               strokeWeight,
167               //10,
168               BasicStroke.CAP_ROUND,
169               BasicStroke.JOIN_ROUND
170             );
171   
172           dashStroke = 
173             new BasicStroke(
174               strokeWeight,
175               BasicStroke.CAP_BUTT,
176               BasicStroke.JOIN_ROUND,
177               0,
178               new float[] {dashInterval}, 
179               0
180             );
181           
182           //TODO: at this stage we should draw the open shapes first in order to make sure they are at the bottom (this is useful for the {mo} shape) 
183           
184           
185           //find storage shapes
186           ArrayList storageShapes = new ArrayList();
187           shapesIt = shapes.iterator();
188           while(shapesIt.hasNext()){
189               DiagramShape shape = (DiagramShape) shapesIt.next();
190               if(shape.getType() == DiagramShape.TYPE_STORAGE) {
191                   storageShapes.add(shape);
192                   continue;
193               } 
194           }
195   
196           //render storage shapes
197           //special case since they are '3d' and should be
198           //rendered bottom to top
199           //TODO: known bug: if a storage object is within a bigger normal box, it will be overwritten in the main drawing loop
200           //(BUT this is not possible since tags are applied to all shapes overlaping shapes)
201   
202           
203           Collections.sort(storageShapes, new Shape3DOrderingComparator());
204           
205           g2.setStroke(normalStroke);
206           shapesIt = storageShapes.iterator();
207           while(shapesIt.hasNext()){
208               DiagramShape shape = (DiagramShape) shapesIt.next();
209   
210               GeneralPath path;
211               path = shape.makeIntoRenderPath(diagram);
212               
213               if(!shape.isStrokeDashed()) {
214                   if(shape.getFillColor() != null)
215                       g2.setColor(shape.getFillColor());
216                   else
217                       g2.setColor(Color.white);
218                   g2.fill(path);
219               }
220   
221               if(shape.isStrokeDashed())
222                   g2.setStroke(dashStroke);
223               else
224                   g2.setStroke(normalStroke);
225               g2.setColor(shape.getStrokeColor());
226               g2.draw(path);
227           }
228   
229   
230           //render the rest of the shapes
231           ArrayList pointMarkers = new ArrayList();
232           shapesIt = shapes.iterator();
233           while(shapesIt.hasNext()){
234               DiagramShape shape = (DiagramShape) shapesIt.next();
235               if(shape.getType() == DiagramShape.TYPE_POINT_MARKER) {
236                   pointMarkers.add(shape);
237                   continue;
238               } 
239               if(shape.getType() == DiagramShape.TYPE_STORAGE) {
240                   continue;
241               } 
242               if(shape.getType() == DiagramShape.TYPE_CUSTOM){
243                   renderCustomShape(shape, g2);
244                   continue;
245               }
246   
247               if(shape.getPoints().isEmpty()) continue;
248   
249               int size = shape.getPoints().size();
250               
251               GeneralPath path;
252               path = shape.makeIntoRenderPath(diagram);
253               
254               //fill
255               if(path != null && shape.isClosed() && !shape.isStrokeDashed()){
256                   if(shape.getFillColor() != null)
257                       g2.setColor(shape.getFillColor());
258                   else
259                       g2.setColor(Color.white);
260                   g2.fill(path);
261               }
262               
263               //draw
264               if(shape.getType() != DiagramShape.TYPE_ARROWHEAD){
265                   g2.setColor(shape.getStrokeColor());
266                   if(shape.isStrokeDashed())
267                       g2.setStroke(dashStroke);
268                   else
269                       g2.setStroke(normalStroke);
270                   g2.draw(path);
271               }
272           }
273           
274           //render point markers
275           
276           g2.setStroke(normalStroke);
277           shapesIt = pointMarkers.iterator();
278           while(shapesIt.hasNext()){
279               DiagramShape shape = (DiagramShape) shapesIt.next();
280               //if(shape.getType() != DiagramShape.TYPE_POINT_MARKER) continue;
281   
282               GeneralPath path;
283               path = shape.makeIntoRenderPath(diagram);
284               
285               g2.setColor(Color.white);
286               g2.fill(path);
287               g2.setColor(shape.getStrokeColor());
288               g2.draw(path);
289           }        
290           
291           //handle text
292           //g2.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
293           //renderTextLayer(diagram.getTextObjects().iterator());
294           
295           Iterator textIt = diagram.getTextObjects().iterator();
296           while(textIt.hasNext()){
297               DiagramText text = (DiagramText) textIt.next();
298               g2.setFont(text.getFont());
299               if(text.hasOutline()){
300                   g2.setColor(text.getOutlineColor());
301                   g2.drawString(text.getText(), text.getXPos() + 1, text.getYPos());
302                   g2.drawString(text.getText(), text.getXPos() - 1, text.getYPos());
303                   g2.drawString(text.getText(), text.getXPos(), text.getYPos() + 1);
304                   g2.drawString(text.getText(), text.getXPos(), text.getYPos() - 1);
305               }
306               g2.setColor(text.getColor());
307               g2.drawString(text.getText(), text.getXPos(), text.getYPos());
308           }
309           
310           if(options.renderDebugLines() || DEBUG){
311               Stroke debugStroke =
312                 new BasicStroke(
313                   1,
314                   BasicStroke.CAP_ROUND,
315                   BasicStroke.JOIN_ROUND
316                 );
317               g2.setStroke(debugStroke);
318               g2.setColor(new Color(170, 170, 170));
319               g2.setXORMode(Color.white);
320               for(int x = 0; x < diagram.getWidth(); x += diagram.getCellWidth())
321                   g2.drawLine(x, 0, x, diagram.getHeight());
322               for(int y = 0; y < diagram.getHeight(); y += diagram.getCellHeight())
323                   g2.drawLine(0, y, diagram.getWidth(), y);
324           }
325           
326   
327           g2.dispose();
328           
329           return renderedImage;
330       }

366       private void renderCustomShape(DiagramShape shape, Graphics2D g2){
367           CustomShapeDefinition definition = shape.getDefinition();
368           
369           Rectangle bounds = shape.getBounds();
370           
371           if(definition.hasBorder()){
372               g2.setColor(shape.getStrokeColor());
373               if(shape.isStrokeDashed())
374                   g2.setStroke(dashStroke);
375               else
376                   g2.setStroke(normalStroke);
377               g2.drawLine(bounds.x, bounds.y, bounds.x + bounds.width, bounds.y);
378               g2.drawLine(bounds.x + bounds.width, bounds.y, bounds.x + bounds.width, bounds.y + bounds.height);
379               g2.drawLine(bounds.x, bounds.y + bounds.height, bounds.x + bounds.width, bounds.y + bounds.height);
380               g2.drawLine(bounds.x, bounds.y, bounds.x, bounds.y + bounds.height);
381               
382   //            g2.drawRect(bounds.x, bounds.y, bounds.width, bounds.height); //looks different!            
383           }
384           
385           //TODO: custom shape distintion relies on filename extension. Make this more intelligent
386           if(definition.getFilename().endsWith(".png")){
387               renderCustomPNGShape(shape, g2);
388           } else if(definition.getFilename().endsWith(".svg")){
389               // renderCustomSVGShape(shape, g2);
390               throw new UnsupportedOperationException();
391           }
392       }

412       private void renderCustomPNGShape(DiagramShape shape, Graphics2D g2){
413           CustomShapeDefinition definition = shape.getDefinition();
414           Rectangle bounds = shape.getBounds();
415           Image graphic = ImageHandler.instance().loadImage(definition.getFilename());
416           
417           int xPos, yPos, width, height;
418           
419           if(definition.stretches()){ //occupy all available space
420               xPos = bounds.x; yPos = bounds.y;
421               width = bounds.width; height = bounds.height;
422           } else { //decide how to fit
423               int newHeight = bounds.width * graphic.getHeight(null) / graphic.getWidth(null);
424               if(newHeight < bounds.height){ //expand to fit width
425                   height = newHeight;
426                   width = bounds.width;
427                   xPos = bounds.x;
428                   yPos = bounds.y + bounds.height / 2 - graphic.getHeight(null) / 2;
429               } else { //expand to fit height
430                   width = graphic.getWidth(null) * bounds.height / graphic.getHeight(null);
431                   height = bounds.height;
432                   xPos = bounds.x + bounds.width / 2 - graphic.getWidth(null) / 2;
433                   yPos = bounds.y;
434               }
435           }
436           
437           g2.drawImage(graphic, xPos, yPos, width, height, null);        
438       }

440       public static boolean isColorDark(Color color){
441           int brightness = Math.max(color.getRed(), color.getGreen());
442           brightness = Math.max(color.getBlue(), brightness);
443           if(brightness < 200) {
444               if(DEBUG) System.out.println("Color "+color+" is dark");
445               return true;
446           }
447           if(DEBUG) System.out.println("Color "+color+" is not dark");
448           return false;
449       }
