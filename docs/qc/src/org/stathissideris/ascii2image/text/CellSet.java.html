155               type = TYPE_OPEN;
156               return TYPE_OPEN;
157           }

159               type = TYPE_CLOSED;

175           type = TYPE_UNDETERMINED;
176           return TYPE_UNDETERMINED;
177       }

253       private int getTypeAccordingToFillMethod(TextGrid grid){
254           if(size() == 0) return TYPE_OPEN;
255           
256           CellSet tempSet = copyCellSet(this);
257           tempSet.translate( -this.getMinX() + 1, -this.getMinY() + 1);
258           TextGrid subGrid = grid.getSubGrid(getMinX() - 1, getMinY() - 1, getWidth() + 3, getHeight() + 3);
259           AbstractionGrid abstraction = new AbstractionGrid(subGrid, tempSet);
260           TextGrid temp = abstraction.getCopyOfInternalBuffer();
261   
262           TextGrid.Cell cell1 = null;
263           TextGrid.Cell cell2 = null;
264   
265           int width = temp.getWidth();
266           int height = temp.getHeight();
267           
268           TextGrid.Cell fillCell = null;
269           for(int y = 0; y < height; y++){
270               for(int x = 0; x < width; x++){
271                   TextGrid.Cell cCell = temp.new Cell(x, y);
272                   if(temp.isBlank(cCell)){
273                       fillCell = cCell;
274                       break;
275                   }
276               }
277           }
278           
279           if(fillCell == null){
280               System.err.println("Unexpected error: fill method cannot fill anywhere");
281               return TYPE_UNDETERMINED;
282           }
283           
284           temp.fillContinuousArea(fillCell, '*');
285           if(VERBOSE_DEBUG) {System.out.println("Buffer after filling:"); temp.printDebug();}
286           
287           if(temp.hasBlankCells()) return TYPE_HAS_CLOSED_AREA;
288           else return TYPE_OPEN;
289       }

579        * @return a list of boundaries that are either open or closed but not mixed
580        * and they are equivalent to the <code>this</code>
581        */
582       public ArrayList breakTrulyMixedBoundaries(TextGrid grid){
583           ArrayList result = new ArrayList();
584           CellSet visitedEnds = new CellSet();
585           
586           TextGrid workGrid = TextGrid.makeSameSizeAs(grid);
587           grid.copyCellsTo(this, workGrid);
588   
589           if (DEBUG){
590               System.out.println("Breaking truly mixed boundaries below:");
591               workGrid.printDebug();
592           }
593   
594           Iterator it = iterator();
595           while(it.hasNext()){
596               TextGrid.Cell start = (TextGrid.Cell) it.next();
597               if(workGrid.isLinesEnd(start) && !visitedEnds.contains(start)){
598                   
599                   if (DEBUG)
600                       System.out.println("Starting new subshape:");
601                   
602                   CellSet set = new CellSet();
603                   set.add(start);
604                   if(DEBUG) System.out.println("Added boundary "+start);
605                   
606                   TextGrid.Cell previous = start;
607                   TextGrid.Cell cell = null;
608                   CellSet nextCells = workGrid.followCell(previous);
609                   if(nextCells.size() == 0)
610                       throw new IllegalArgumentException("This shape is either open but multipart or has only one cell, and cannot be processed by this method");
611                   cell = (TextGrid.Cell) nextCells.getFirst();
612                   set.add(cell);
613                   if(DEBUG) System.out.println("Added boundary "+cell);
614                   
615                   boolean finished = false;
616                   if(workGrid.isLinesEnd(cell)){
617                       visitedEnds.add(cell);
618                       finished = true;                    
619                   }
620                   
621                   while(!finished){
622                       nextCells = workGrid.followCell(cell, previous);
623                       if(nextCells.size() == 1) {
624                           set.add(cell);
625                           if(DEBUG) System.out.println("Added boundary " + cell);
626                           previous = cell;
627                           cell = (TextGrid.Cell) nextCells.getFirst();
628                           //if(!cell.equals(start) && grid.isPointCell(cell))
629                           //    s.addToPoints(makePointForCell(cell, workGrid, cellWidth, cellHeight, allRound));
630                           if(workGrid.isLinesEnd(cell)){
631                               visitedEnds.add(cell);
632                               finished = true;
633                           }
634                       } else if(nextCells.size() > 1) {
635                           finished = true;
636                       }
637                   }
638                   result.add(set);
639               }
640           }
641   
642           //substract all boundary sets from this CellSet
643           CellSet whatsLeft = new CellSet(this);
644           it = result.iterator();
645           while (it.hasNext()) {
646               CellSet set = (CellSet) it.next();
647               whatsLeft.subtractSet(set);
648               if(DEBUG) set.printAsGrid();
649           }
650           result.add(whatsLeft);
651           if(DEBUG) whatsLeft.printAsGrid();
652           
653           return result;
654       }
