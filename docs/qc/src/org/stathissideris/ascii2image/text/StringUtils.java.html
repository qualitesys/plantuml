75       public static String insertSpaceAtCaps(String string){
76   
77           int uppers = 0;
78   
79           //first we count
80           for(int i=0; i < string.length(); i++){
81               if(Character.isUpperCase(string.charAt(i))) uppers++;
82           }
83   
84           int[] indexes = null;
85           
86           if(Character.isUpperCase(string.charAt(0))){        
87               indexes = new int[uppers];
88           } else {
89               indexes = new int[++uppers];
90           }
91           indexes[0] = 0;        
92           int k = 1;
93   
94           //then we find the indexes (we have ckecked the first char already)
95           for(int j =1; j < string.length(); j++){
96               if(Character.isUpperCase(string.charAt(j))) indexes[k++] = j;
97           }
98           
99           StringBuffer buffer = new StringBuffer("");
100           //and finally we breakup the String
101           for(int i =0; i < indexes.length; i++){
102               if(i+1 < indexes.length){
103                   buffer.append(string.substring(indexes[i], indexes[i+1]));
104                   buffer.append(" ");
105               } else {
106                   buffer.append(string.substring(indexes[i]));
107               }            
108           }
109           return buffer.toString();
110       }
