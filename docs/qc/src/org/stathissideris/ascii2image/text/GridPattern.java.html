27   /**
28    * This is a TextGrid (usually 3x3) that contains the equivalent of a
29    * 2D reqular expression (which uses custom syntax to make things more
30    * visual, but standard syntax is also possible).
31    * 
32    * The custom syntax is:
33    * . means anything
34    * b means any boundary (any of  - = / \ + | :)
35    * ! means not boundary (none of  - = / \ + | :)
36    * - means - or =
37    * | means | or :
38    * [ means not | nor :
39    * ~ means not - nor =
40    * ^ means a boundary but not - nor =
41    * ( means a boundary but not | nor :
42    * s means a straight boundary (one of - = + | :)
43    * S means not a straight boundary (none of - = + | :)
44    * 
45    * 1 means a cell that has entry point 1
46    * 2 means a cell that has entry point 2
47    * 3 means a cell that has entry point 3 etc. up to number 8
48    * 
49    * %1 means a cell that does not have entry point 1 etc.
50    * 
51    * See below for an explanation of entry points
52    * 
53    * +, \, / and the space are literal (as is any other character)
54    * 
55    * 
56    * Entry points
57    * 
58    * <pre>
59    * 1   2   3
60    *  *--*--*
61    *  |     |
62    * 8*     *4
63    *  |     |
64    *  *--*--*
65    * 7   6   5
66    * </pre>
67    * 
68    * We number the entry points for each cell as in the diagram
69    * above. If a cell is occupied by a character, we define as
70    * entry points the points of the above diagram that the character
71    * can touch with the end of its lines. For example - has
72    * entry points 8 and 4, | and : have entry points 2 and 6,
73    * / has 3 and 7, \ has 1 and 5, + has 2, 6, 8 and 4 etc.   
74    * 
75    * 
76    * @author Efstathios Sideris
77    */

106       public boolean isMatchedBy(TextGrid grid){
107           /*if(grid.getHeight() != this.getHeight()
108               || grid.getWidth() != this.getWidth()) return false;*/
109           if(!regExpsAreValid) prepareRegExps(); 
110   
111           for(int i = 0; i < grid.getHeight(); i++) {
112               String row = grid.getRow(i).toString();
113               Pattern regexp = regExps.get(i);
114               if(!regexp.matcher(row).matches()) {
115                   if(DEBUG)
116                       System.out.println(row+" does not match "+regexp);
117                   return false;
118               } 
119           }
120           return true;
121       }

123       private void prepareRegExps(){
124           regExpsAreValid = true;
125           regExps.clear();
126           if (DEBUG)
127               System.out.println("Trying to match:");
128           if(!usesStandardSyntax){
129               Iterator<StringBuffer> it = getRows().iterator();
130               while (it.hasNext()) {
131                   String row = it.next().toString();
132                   regExps.add(Pattern.compile(makeRegExp(row)));
133                   if(DEBUG)
134                       System.out.println(row+" becomes "+makeRegExp(row));
135               }            
136           } else {
137               Iterator<StringBuffer> it = getRows().iterator();
138               while (it.hasNext()) {
139                   String row = it.next().toString();
140                   regExps.add(Pattern.compile(row));
141               }
142           }
143       }

145       private String makeRegExp(String pattern){
146           StringBuilder result = new StringBuilder();
147           int tokensHandled = 0;
148           for(int i = 0; i < pattern.length() && tokensHandled < 3; i++){
149               char c = pattern.charAt(i);
150               if(c == '[') {
151                   result.append("[^|:]");
152               } else if(c == '|') {
153                   result.append("[|:]");
154               } else if(c == '-') {
155                   result.append("-");
156               } else if(c == '!') {
157                   result.append("[^-=\\/\\\\+|:]");
158               } else if(c == 'b') {
159                   result.append("[-=\\/\\\\+|:]");
160               } else if(c == '^') {
161                   result.append("[\\/\\\\+|:]");
162               } else if(c == '(') {
163                   result.append("[-=\\/\\\\+]");
164               } else if(c == '~') {
165                   result.append(".");
166               } else if(c == '+') {
167                   result.append("\\+");
168               } else if(c == '\\') {
169                   result.append("\\\\");
170               } else if(c == 's') {
171                   result.append("[-=+|:]");
172               } else if(c == 'S') {
173                   result.append("[\\/\\\\]");
174               } else if(c == '*') {
175                   result.append("\\*");
176               
177               //entry points
178               } else if(c == '1') {
179                   result.append("[\\\\]");
180   
181               } else if(c == '2') {
182                   result.append("[|:+\\/\\\\]");
183   
184               } else if(c == '3') {
185                   result.append("[\\/]");
186   
187               } else if(c == '4') {
188                   result.append("[-=+\\/\\\\]");
189   
190               } else if(c == '5') {
191                   result.append("[\\\\]");
192   
193               } else if(c == '6') {
194                   result.append("[|:+\\/\\\\]");
195   
196               } else if(c == '7') {
197                   result.append("[\\/]");
198   
199               } else if(c == '8') {
200                   result.append("[-=+\\/\\\\]");
201                   
202               //entry point negations
203               } else if(c == '%') {
204                   if(i+1 > pattern.length()){
205                       throw new RuntimeException("Invalid pattern, found % at the end");
206                   }
207                   c = pattern.charAt(++i);
208                   
209                   if(c == '1') {
210                       result.append("[^\\\\]");
211   
212                   } else if(c == '2') {
213                       result.append("[^|:+\\/\\\\]");
214   
215                   } else if(c == '3') {
216                       result.append("[^\\/]");
217   
218                   } else if(c == '4') {
219                       result.append("[^-=+\\/\\\\]");
220   
221                   } else if(c == '5') {
222                       result.append("[^\\\\]");
223   
224                   } else if(c == '6') {
225                       result.append("[^|:+\\/\\\\]");
226   
227                   } else if(c == '7') {
228                       result.append("[^\\/]");
229   
230                   } else if(c == '8') {
231                       result.append("[^-=+\\/\\\\]");
232                   }
233               } else result.append(String.valueOf(c));
234               tokensHandled++;
235           }
236           return result.toString();
237       }
