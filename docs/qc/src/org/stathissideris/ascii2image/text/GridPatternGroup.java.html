31       public boolean areAllMatchedBy(TextGrid grid){
32           Iterator<GridPattern> it = iterator();
33           while (it.hasNext()) {
34               GridPattern pattern = it.next();
35               if(!pattern.isMatchedBy(grid)) return false;
36           }
37           return true;
38       }
39   
40       public boolean isAnyMatchedBy(TextGrid grid){
41           Iterator<GridPattern> it = iterator();
42           while (it.hasNext()) {
43               GridPattern pattern = it.next();
44               if(pattern.isMatchedBy(grid)) return true;
45           }
46           return false;
47       }
