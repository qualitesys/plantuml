21   /**
22    * <p>Encapsulates the result of detecting a barcode in an image. This includes the raw
23    * matrix of black/white pixels corresponding to the barcode, and possibly points of interest
24    * in the image, like the location of finder patterns or corners of the barcode in the image.</p>
25    *
26    * @author Sean Owen
27    */
28   public final class DetectorResult {
29   
30     private final BitMatrix bits;
31     private final ResultPoint[] points;
32   
33     public DetectorResult(BitMatrix bits, ResultPoint[] points) {
34       this.bits = bits;
35       this.points = points;
36     }
37   
38     public BitMatrix getBits() {
39       return bits;
40     }
41   
42     public ResultPoint[] getPoints() {
43       return points;
44     }
45   
46   }
