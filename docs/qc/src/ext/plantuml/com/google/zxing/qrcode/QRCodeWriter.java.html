31   /**
32    * This object renders a QR Code as a BitMatrix 2D array of greyscale values.
33    * 
34    * @author dswitkin@google.com (Daniel Switkin)
35    */
36   public final class QRCodeWriter implements Writer {
37   
38       private static final int QUIET_ZONE_SIZE = 4;
39   
40       public BitMatrix encode(String contents, BarcodeFormat format, int multiple) throws WriterException {
41   
42           return encode(contents, format, multiple, null);
43       }
44   
45       public BitMatrix encode(String contents, BarcodeFormat format, int multiple, Hashtable hints) throws WriterException {
46   
47           if (contents == null || contents.length() == 0) {
48               throw new IllegalArgumentException("Found empty contents");
49           }
50   
51           if (format != BarcodeFormat.QR_CODE) {
52               throw new IllegalArgumentException("Can only encode QR_CODE, but got " + format);
53           }
54   
55           ErrorCorrectionLevel errorCorrectionLevel = ErrorCorrectionLevel.L;
56           if (hints != null) {
57               ErrorCorrectionLevel requestedECLevel = (ErrorCorrectionLevel) hints.get(EncodeHintType.ERROR_CORRECTION);
58               if (requestedECLevel != null) {
59                   errorCorrectionLevel = requestedECLevel;
60               }
61           }
62   
63           QRCode code = new QRCode();
64           Encoder.encode(contents, errorCorrectionLevel, hints, code);
65           return renderResult(code, multiple);
66       }
67   
68       // Note that the input matrix uses 0 == white, 1 == black, while the output
69       // matrix uses
70       // 0 == black, 255 == white (i.e. an 8 bit greyscale bitmap).
71       private static BitMatrix renderResult(QRCode code, int multiple) {
72           ByteMatrix input = code.getMatrix();
73           int inputWidth = input.getWidth();
74           int inputHeight = input.getHeight();
75           int qrWidth = inputWidth + (QUIET_ZONE_SIZE << 1);
76           int qrHeight = inputHeight + (QUIET_ZONE_SIZE << 1);
77           // int outputWidth = Math.max(width, qrWidth);
78           // int outputHeight = Math.max(height, qrHeight);
79   
80           // int multiple = Math.min(outputWidth / qrWidth, outputHeight /
81           // qrHeight);
82           // Padding includes both the quiet zone and the extra white pixels to
83           // accommodate the requested
84           // dimensions. For example, if input is 25x25 the QR will be 33x33
85           // including the quiet zone.
86           // If the requested size is 200x160, the multiple will be 4, for a QR of
87           // 132x132. These will
88           // handle all the padding from 100x100 (the actual QR) up to 200x160.
89           int leftPadding = QUIET_ZONE_SIZE * multiple;
90           int topPadding = QUIET_ZONE_SIZE * multiple;
91   
92           BitMatrix output = new BitMatrix(qrWidth * multiple, qrHeight * multiple);
93   
94           for (int inputY = 0, outputY = topPadding; inputY < inputHeight; inputY++, outputY += multiple) {
95               // Write the contents of this row of the barcode
96               for (int inputX = 0, outputX = leftPadding; inputX < inputWidth; inputX++, outputX += multiple) {
97                   if (input.get(inputX, inputY) == 1) {
98                       output.setRegion(outputX, outputY, multiple, multiple);
99                   }
100               }
101           }
102   
103           return output;
104       }
105   
106   }
