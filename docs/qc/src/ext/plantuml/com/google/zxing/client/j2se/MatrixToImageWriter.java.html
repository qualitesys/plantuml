34

37     /**
38      * Renders a {@link BitMatrix} as an image, where "false" bits are rendered
39      * as white, and "true" bits are rendered as black.
40      */
41     public static BufferedImage toBufferedImage(BitMatrix matrix, int fore, int back) {
42       int width = matrix.getWidth();
43       int height = matrix.getHeight();
44       BufferedImage image = new BufferedImage(width, height, BufferedImage.TYPE_INT_ARGB);
45       for (int x = 0; x < width; x++) {
46         for (int y = 0; y < height; y++) {
47           image.setRGB(x, y, matrix.get(x, y) ? fore : back);
48         }
49       }
50       return image;
51     }
