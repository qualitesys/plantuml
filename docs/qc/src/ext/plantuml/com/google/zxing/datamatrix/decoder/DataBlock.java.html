17   package ext.plantuml.com.google.zxing.datamatrix.decoder;
18   
19   /**
20    * <p>Encapsulates a block of data within a Data Matrix Code. Data Matrix Codes may split their data into
21    * multiple blocks, each of which is a unit of data and error-correction codewords. Each
22    * is represented by an instance of this class.</p>
23    *
24    * @author bbrown@google.com (Brian Brown)
25    */
26   final class DataBlock {
27   
28     private final int numDataCodewords;
29     private final byte[] codewords;
30   
31     private DataBlock(int numDataCodewords, byte[] codewords) {
32       this.numDataCodewords = numDataCodewords;
33       this.codewords = codewords;
34     }
35   
36     /**
37      * <p>When Data Matrix Codes use multiple data blocks, they actually interleave the bytes of each of them.
38      * That is, the first byte of data block 1 to n is written, then the second bytes, and so on. This
39      * method will separate the data into original blocks.</p>
40      *
41      * @param rawCodewords bytes as read directly from the Data Matrix Code
42      * @param version version of the Data Matrix Code
43      * @return {@link DataBlock}s containing original bytes, "de-interleaved" from representation in the
44      *         Data Matrix Code
45      */
46     static DataBlock[] getDataBlocks(byte[] rawCodewords,
47                                      Version version) {
48       // Figure out the number and size of data blocks used by this version
49       Version.ECBlocks ecBlocks = version.getECBlocks();
50   
51       // First count the total number of data blocks
52       int totalBlocks = 0;
53       Version.ECB[] ecBlockArray = ecBlocks.getECBlocks();
54       for (int i = 0; i < ecBlockArray.length; i++) {
55         totalBlocks += ecBlockArray[i].getCount();
56       }
57   
58       // Now establish DataBlocks of the appropriate size and number of data codewords
59       DataBlock[] result = new DataBlock[totalBlocks];
60       int numResultBlocks = 0;
61       for (int j = 0; j < ecBlockArray.length; j++) {
62         Version.ECB ecBlock = ecBlockArray[j];
63         for (int i = 0; i < ecBlock.getCount(); i++) {
64           int numDataCodewords = ecBlock.getDataCodewords();
65           int numBlockCodewords = ecBlocks.getECCodewords() + numDataCodewords;
66           result[numResultBlocks++] = new DataBlock(numDataCodewords, new byte[numBlockCodewords]);
67         }
68       }
69   
70       // All blocks have the same amount of data, except that the last n
71       // (where n may be 0) have 1 less byte. Figure out where these start.
72       // TODO(bbrown): There is only one case where there is a difference for Data Matrix for size 144
73       int longerBlocksTotalCodewords = result[0].codewords.length;
74       //int shorterBlocksTotalCodewords = longerBlocksTotalCodewords - 1;
75   
76       int longerBlocksNumDataCodewords = longerBlocksTotalCodewords - ecBlocks.getECCodewords();
77       int shorterBlocksNumDataCodewords = longerBlocksNumDataCodewords - 1;
78       // The last elements of result may be 1 element shorter for 144 matrix
79       // first fill out as many elements as all of them have minus 1
80       int rawCodewordsOffset = 0;
81       for (int i = 0; i < shorterBlocksNumDataCodewords; i++) {
82         for (int j = 0; j < numResultBlocks; j++) {
83           result[j].codewords[i] = rawCodewords[rawCodewordsOffset++];
84         }
85       }
86       
87       // Fill out the last data block in the longer ones
88       boolean specialVersion = version.getVersionNumber() == 24;
89       int numLongerBlocks = specialVersion ? 8 : numResultBlocks;
90       for (int j = 0; j < numLongerBlocks; j++) {
91         result[j].codewords[longerBlocksNumDataCodewords - 1] = rawCodewords[rawCodewordsOffset++];
92       }
93       
94       // Now add in error correction blocks
95       int max = result[0].codewords.length;
96       for (int i = longerBlocksNumDataCodewords; i < max; i++) {
97         for (int j = 0; j < numResultBlocks; j++) {
98           int iOffset = (specialVersion && j > 7) ? i - 1 : i;
99           result[j].codewords[iOffset] = rawCodewords[rawCodewordsOffset++];
100         }
101       }
102   
103       if (rawCodewordsOffset != rawCodewords.length) {
104         throw new IllegalArgumentException();
105       }
106   
107       return result;
108     }
109   
110     int getNumDataCodewords() {
111       return numDataCodewords;
112     }
113   
114     byte[] getCodewords() {
115       return codewords;
116     }
117   
118   }
