23   /**
24    * The base class for all objects which encode/generate a barcode image.
25    *
26    * @author dswitkin@google.com (Daniel Switkin)
27    */
28   public interface Writer {
29   
30     /**
31      * Encode a barcode using the default settings.
32      *
33      * @param contents The contents to encode in the barcode
34      * @param format The barcode format to generate
35      * @return The generated barcode as a Matrix of unsigned bytes (0 == black, 255 == white)
36      */
37     BitMatrix encode(String contents, BarcodeFormat format, int multiple)
38         throws WriterException;
39   
40     /**
41      *
42      * @param contents The contents to encode in the barcode
43      * @param format The barcode format to generate
44      * @param hints Additional parameters to supply to the encoder
45      * @return The generated barcode as a Matrix of unsigned bytes (0 == black, 255 == white)
46      */
47     BitMatrix encode(String contents, BarcodeFormat format, int multiple, Hashtable hints)
48         throws WriterException;
49   
50   }
