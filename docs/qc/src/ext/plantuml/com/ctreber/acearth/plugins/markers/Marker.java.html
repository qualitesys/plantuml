82       private static Marker createFromLine(String pLine) {
83           final List lWords = StringParser.parse(pLine);
84   
85           final double lLat = Double.parseDouble((String) lWords.get(0));
86           final double lLong = Double.parseDouble((String) lWords.get(1));
87           final String lLabel = (String) lWords.get(2);
88   
89           int lAlign = MarkerAlignDefault;
90           if (lWords.size() >= 4) {
91               String lAlignString = (String) lWords.get(3);
92               if (lAlignString.equalsIgnoreCase("left")) {
93                   lAlign = MarkerAlignLeft;
94               }
95               if (lAlignString.equalsIgnoreCase("right")) {
96                   lAlign = MarkerAlignRight;
97               }
98               if (lAlignString.equalsIgnoreCase("above")) {
99                   lAlign = MarkerAlignAbove;
100               }
101               if (lAlignString.equalsIgnoreCase("below")) {
102                   lAlign = MarkerAlignBelow;
103               }
104           }
105   
106           final Coordinate lPos = new Coordinate(lLat, lLong);
107           if (!lPos.check()) {
108               // ACearth.logError("latitude must be between -90 and 90, and
109               // longitude must be between -180 and 180");
110               return null;
111           }
112   
113           return new Marker(lPos, lLabel, lAlign);
114       }

141       public void render(PixelCanvas pCanvas, Projection pProjection) {
142           final Point3D lPos = pProjection.rotate(fCoordinate.getPoint3D());
143   
144           if ((pProjection instanceof ProjectionOrtho) && (lPos.getZ() <= 0)) {
145               // Back side of the Earth.
146               // Insight: We don't need to check if the marker is visible in other
147               // projections because they always show the whole earth - and all
148               // markers!
149               return;
150           }
151   
152           Point2D lPoint = pProjection.finalize(pProjection.project2D(lPos));
153           final int lX = (int) lPoint.getX();
154           final int lY = (int) lPoint.getY();
155   
156           // Draw a circle
157           Graphics2D g2d = pCanvas.getGraphics2D();
158           g2d.setColor(Color.red);
159           // pCanvas.drawCircle(lX, lY, MARKER_SIZE);
160           g2d.drawOval(lX, lY, MARKER_SIZE, MARKER_SIZE);
161   
162           if (fLabel != null) {
163               switch (fAlign) {
164               case Marker.MarkerAlignLeft:
165                   break;
166   
167               case Marker.MarkerAlignRight:
168               case Marker.MarkerAlignDefault:
169                   // pCanvas.drawText(lX + MARKER_SIZE, lY + 4, fLabel);
170                   // fRenderTarget.setTextFont(fRenderTarget.getTextFont().deriveFont(9.0f));
171                   g2d.setFont(new Font("", Font.PLAIN, 9));
172                   g2d.drawString(fLabel, lX + MARKER_SIZE + 1, lY + 2);
173                   break;
174   
175               case Marker.MarkerAlignAbove:
176                   break;
177   
178               case Marker.MarkerAlignBelow:
179                   break;
180               }
181           }
182       }
