14    */
15   abstract public class Projection
16   {
17     // Target information
18     int fImageHeight;
19     int fImageWidth;
20     private double fXOffset;
21     private double fYOffset;
22   
23     //Viewing information
24     private int fShiftX;
25     private int fShiftY;
26     double fScale;
27   
28     private Coordinate fViewPos;
29     /** <p>In rads */
30     private double fViewRotation;
31     double fViewMagnification;
32   
33   
34     //Transformation matrix parameters */
35     private double fCosLat;
36     private double fSinLat;
37     private double fCosLon;
38     private double fSinLon;
39     private double fCosRot;
40     private double fSinRot;
41   
42     /**
43      * <p>Initialize transform parameters, set offset to center of image
44      * (plus shifts), set scale
45      */
46     public void initTransformTable()
47     {
48       // Set transformation parameters
49       fCosLat = Math.cos(Toolkit.degsToRads(fViewPos.getLat()));
50       fSinLat = Math.sin(Toolkit.degsToRads(fViewPos.getLat()));
51       fCosLon = Math.cos(Toolkit.degsToRads(fViewPos.getLong()));
52       fSinLon = Math.sin(Toolkit.degsToRads(fViewPos.getLong()));
53       fCosRot = Math.cos(Toolkit.degsToRads(fViewRotation));
54       fSinRot = Math.sin(Toolkit.degsToRads(fViewRotation));
55   
56       fXOffset = (double)fImageWidth / 2 + fShiftX;
57       fYOffset = (double)fImageHeight / 2 + fShiftY;
58   
59       setScale();
60     }

88      *

95     /**
96      * <p>Imagine view the globe, N is up, S is down, and N0E0 is in the center.
97      * x is right/left, y is up/down, and z is front/rear.
98      *
99      * <p>Map points are located on the surface of a unit sphere (diameter = 1).
100      * Latitude is the angle between x and y or z and y. Longitude is the angle
101      * between x and z.
102      *
103      * <p>Why? The way we choose our global coordinate system, longitude circles
104      * (latidude variable) always have the same size while the size of
105      * latidude circles (longitude variable) depends on the latitude.
106      *
107      * @param pPoint
108      * @return
109      */
110     public Point2D project2D(Point3D pPoint)
111     {
112       return new Point2D(projectX(pPoint.getX(), pPoint.getZ()),
113         projectY(pPoint.getY()));
114     }

116     public Point2D finalize(Point2D pPoint)
117     {
118       return new Point2D(finalizeX(pPoint.getX()), finalizeY(pPoint.getY()));
119     }
120   
121     /**
122      * <p>Since the final mapping is relative to the center of the image
123      * -PI and PI get mapped to the left and right border respectively.
124      * But see ProjectionOrtho.setScale().
125      */
126     public double finalizeX(double pX)
127     {
128       return fXOffset + fScale * pX;
129     }
130   
131     /**
132      * <p>Since the final mapping is relative to the center of the image
133      * -PI and PI get mapped to the bottom and top border respectively.
134      * But see ProjectionOrtho.setScale().
135      */
136     public double finalizeY(double pY)
137     {
138       return fYOffset - fScale * pY;
139     }
140   
141     /**
142      * <p>Transform screen to image coordinates.
143      */
144     public double inverseFinalizeX(double x)
145     {
146       return (x - fXOffset) / fScale;
147     }
148   
149     /**
150      * <p>Transform screen to image coordinates.
151      */
152     public double inverseFinalizeY(double y)
153     {
154       return (fYOffset - y) / fScale;
155     }

157     /**
158      * <p>Rotate the point according to the current rotation of Earth.
159      */
160     public Point3D rotate(Point3D pPoint)
161     {
162       double lX = pPoint.getX();
163       double lY = pPoint.getY();
164       double lZ = pPoint.getZ();
165   
166       // Do NOT inline vars - it does not work (just inline _t_ for a try).
167       double _c_ = fCosLon;
168       double _s_ = fSinLon;
169       double _t_ = _c_ * lX - _s_ * lZ;
170       lZ = _s_ * lX + _c_ * lZ;
171       lX = _t_;
172   
173       _c_ = fCosLat;
174       _s_ = fSinLat;
175       _t_ = (_c_ * lY) - (_s_ * lZ);
176       lZ = (_s_ * lY) + (_c_ * lZ);
177       lY = _t_;
178   
179       _c_ = fCosRot;
180       _s_ = fSinRot;
181       _t_ = (_c_ * lX) - (_s_ * lY);
182       lY = (_s_ * lX) + (_c_ * lY);
183       lX = _t_;
184   
185       return new Point3D(lX, lY, lZ);
186     }

188     public Point3D rotateReverse(Point3D pPoint)
189     {
190       // Set transformation parameters
191       final double fCosLat = Math.cos(Toolkit.degsToRads(-fViewPos.getLat()));
192       final double fSinLat = Math.sin(Toolkit.degsToRads(-fViewPos.getLat()));
193       final double fCosLon = Math.cos(Toolkit.degsToRads(-fViewPos.getLong()));
194       final double fSinLon = Math.sin(Toolkit.degsToRads(-fViewPos.getLong()));
195       final double fCosRot = Math.cos(Toolkit.degsToRads(-fViewRotation));
196       final double fSinRot = Math.sin(Toolkit.degsToRads(-fViewRotation));
197   
198       double lX = pPoint.getX();
199       double lY = pPoint.getY();
200       double lZ = pPoint.getZ();
201   
202       // Do NOT inline vars - it does not work (just inline lTmp for a try).
203       double lCosFac;
204       double lSinFac;
205       double lTmp;
206   
207       // Note that the order of the three rotation had to be reversed as well.
208       lCosFac = fCosRot;
209       lSinFac = fSinRot;
210       lTmp = (lCosFac * lX) - (lSinFac * lY);
211       lY = (lSinFac * lX) + (lCosFac * lY);
212       lX = lTmp;
213   
214       lCosFac = fCosLat;
215       lSinFac = fSinLat;
216       lTmp = (lCosFac * lY) - (lSinFac * lZ);
217       lZ = (lSinFac * lY) + (lCosFac * lZ);
218       lY = lTmp;
219   
220       lCosFac = fCosLon;
221       lSinFac = fSinLon;
222       lTmp = (lCosFac * lX) - (lSinFac * lZ);
223       lZ = (lSinFac * lX) + (lCosFac * lZ);
224       lX = lTmp;
225   
226       return new Point3D(lX, lY, lZ);
227     }
