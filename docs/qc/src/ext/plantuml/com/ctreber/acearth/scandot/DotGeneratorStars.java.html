2   
3   import java.util.Random;
4   
5   /**
6    * <p>Generate random stars as dots.
7    *
8    * <p>&copy; 2002 Christian Treber, ct@ctreber.com
9    * @author Christian Treber, ct@ctreber.com
10    *
11    */
12   public class DotGeneratorStars extends ScanDotGenerator
13   {
14     private final int fImageWidth;
15     private final int fImageHeight;
16     private int fBigStars;
17     private double fStarFrequency;
18     private final Random lRandom;
19   
20     public DotGeneratorStars(int pWidth, int pHeight,
21       double pStarFrequency, int pBigStars, Random rnd)
22     {
23       lRandom = rnd;
24       fImageWidth = pWidth;
25       fImageHeight = pHeight;
26       fStarFrequency = pStarFrequency;
27       fBigStars = pBigStars;
28     }
29   
30     public void generateScanDots()
31     {
32       // Make sure stars don't jump around between updates.
33           // final Random lRandom = new Random(ACearth.getStartTime());
34   
35       final int lStarsMax = (int)(fImageWidth * fImageHeight * fStarFrequency);
36       for(int i = 0; i < lStarsMax; i++)
37       {
38         // "-1" to leave space for big stars.
39         int x = (int)(lRandom.nextDouble() * (fImageWidth - 1));
40         int y = (int)(lRandom.nextDouble() * fImageHeight);
41   
42         fDots.add(new ScanDot(ScanDot.DotTypeStar, x, y));
43   
44         // A big star is just two pixels wide.
45         if((fBigStars != 0) && (Math.random() * 100 < fBigStars))
46         {
47           fDots.add(new ScanDot(ScanDot.DotTypeStar, x + 1, y));
48         }
49       }
50     }
51   }
