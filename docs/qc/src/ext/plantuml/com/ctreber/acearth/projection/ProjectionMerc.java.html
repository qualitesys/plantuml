22     public boolean isVisible(Point3D pPoint)

27     public Coordinate getLocation(int pX, int pY)
28     {
29       final Coordinate lRaw = new Coordinate(
30         2 * (Math.atan(Math.exp(inverseFinalizeY(pY))) - Math.PI / 4),
31         inverseFinalizeX(pX));
32   
33       return rotateReverse(lRaw.getPoint3DRads()).getCoordinate();
34     }

36     /**
37      * <p>The scale is set so that a value of

39      * But see ProjectionOrtho.setScale().
40      */
41     protected void setScale()
42     {
43       // Makes 2PI come out as full image width

45     }

47     /**

49      */
50     public double projectX(double pX, double pZ)
51     {
52       return Math.atan2(pX, pZ);
53     }

55     public double inverseProjectX(double pX)
56     {
57       return Math.sin(pX);
58     }
59   
60     /**
61      * @return Latitude (-PI/2 to PI/2), projected from center of Earth on
62      * y axis with a twist and a log scale.
63      */
64     public double projectY(double pY)
65     {
66       return (pY >= 0.9999999999) ? 1e6
67         : (pY <= -0.9999999999) ? -1e6
68         : Math.log(Math.tan(Math.asin(pY) / 2 + Math.PI / 4));
69     }
70   
71     public double inverseProjectY(double y)
72     {
73       return Math.sin(2 * (Math.atan(Math.exp(y)) - Math.PI / 4));
74     }
