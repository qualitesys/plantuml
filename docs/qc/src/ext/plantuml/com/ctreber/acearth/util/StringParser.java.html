1   package ext.plantuml.com.ctreber.acearth.util;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * <p>Cuts a string in words separated by white space. Quotes and square
8    * brackets are recognized (that is, white space within is ignored).
9    *
10    * <p>&copy; 2002 Christian Treber, ct@ctreber.com
11    * @author Christian Treber, ct@ctreber.com
12    *
13    */
14   public class StringParser
15   {
16     public static List parse(String pLine)
17     {
18       final List lSections = new ArrayList();
19   
20       // True if within word.
21       boolean lInSectionP = false;
22       // Current char
23       char lChar;
24       // Wait for this character before switching back to normal parsing.
25       char lSeparator = ' ';
26       // Part count.
27       int lSectionNo = 0;
28       // Part start position
29       int lSectionStart = 0;
30       // Part end position
31       int lSectionEnd = 0;
32   
33       final int lLen = pLine.length();
34       for(int lCharNo = 0; lCharNo <= lLen; lCharNo++)
35       {
36         if(lCharNo < lLen)
37         {
38           lChar = pLine.charAt(lCharNo);
39         } else
40         {
41           // This is a fictional last character.
42           lChar = ' ';
43         }
44   
45         if(lInSectionP)
46         {
47           // In section. Termination is by space or specific separator.
48           if((lChar != ' ') || (lSeparator != ' '))
49           {
50             // It's not a space, or it is a space, but we wait for a special separator.
51             if(lChar == lSeparator)
52             {
53               // We waited for this separator. Switch back to normal parsing.
54               lSeparator = ' ';
55               lSectionEnd = lCharNo - 1;
56             } else
57             {
58               lSectionEnd = lCharNo;
59             }
60           } else
61           {
62             // Section has ended (with a space).
63             lSections.add(pLine.substring(lSectionStart, lSectionEnd + 1));
64             lSectionNo++;
65             lInSectionP = false;
66           }
67         } else
68         {
69           // Not in a section, skipping white space.
70           if(lChar != ' ')
71           {
72             // No white space: a section has started.
73             if(lChar == '"')
74             {
75               // Special parsing "string"
76               lSeparator = '"';
77               lSectionStart = lCharNo + 1;
78             } else if(lChar == '[')
79             {
80               // Special parsing "square brackets"
81               lSeparator = ']';
82               lSectionStart = lCharNo + 1;
83             } else
84             {
85               // Use normal parsing.
86               lSeparator = ' ';
87               lSectionEnd = lSectionStart = lCharNo;
88             }
89             lInSectionP = true;
90           } else
91           {
92             // More void...
93           }
94         }
95       }
96   
97       return lSections;
98     }
99   }
