6   /**
7    * <p>Renders a row of ScanDots to pixel types.</p>
8    *
9    * <p>&copy; 2002 Christian Treber, ct@ctreber.com (Nov 11, 2002)</p>
10    * @author Christian Treber, ct@ctreber.com
11    *
12    */
13   public class RowTypeRendererScanDot implements RowTypeRenderer
14   {
15     private int fScanDotIndex;
16     private ScanDot[] fScanDots;
17   
18     public void startNewRun()
19     {
20       fScanDotIndex = 0;
21     }
22   
23     public void getPixelTypes(int pRowNo, int[] pPixelTypes)
24     {
25       // For all ScanDots in specified row...
26       while((fScanDotIndex < fScanDots.length) &&
27         (fScanDots[fScanDotIndex].getY() == pRowNo))
28       {
29         ScanDot lDot = fScanDots[fScanDotIndex];
30   
31         if(lDot.getType() == ScanDot.DotTypeStar)
32         {
33           if(pPixelTypes[lDot.getX()] == BitGeneratorMap.PixTypeSpace)
34           {
35             // Stars get only painted on Space.
36             pPixelTypes[lDot.getX()] = BitGeneratorMap.PixTypeStar;
37           }
38         } else
39         {
40           // The only other type for a dot (so far) is "grid".
41           switch(pPixelTypes[lDot.getX()])
42           {
43             case BitGeneratorMap.PixTypeLand:
44               pPixelTypes[lDot.getX()] = BitGeneratorMap.PixTypeGridLand;
45               break;
46   
47             case BitGeneratorMap.PixTypeWater:
48               pPixelTypes[lDot.getX()] = BitGeneratorMap.PixTypeGridWater;
49               break;
50           }
51         }
52         fScanDotIndex++;
53       }
54     }
55   
56     public void setScanDots(ScanDot[] pScanDots)
57     {
58       fScanDots = pScanDots;
59     }
60   }
