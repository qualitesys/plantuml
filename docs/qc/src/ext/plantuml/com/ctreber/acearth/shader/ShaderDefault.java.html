1   package ext.plantuml.com.ctreber.acearth.shader;
2   
3   import java.awt.Color;
4   
5   /**
6    * <p>Shader for projections which display the whole surface.</p>
7    *
8    * <p>&copy; 2002 Christian Treber, ct@ctreber.com (Nov 11, 2002)</p>
9    * @author Christian Treber, ct@ctreber.com
10    *
11    */
12   public class ShaderDefault extends Shader
13   {
14     public Color[] getShadedColors(int pRowNo, int[] pRowTypes)
15     {
16       final double y = fProjection.inverseProjectY(fProjection.inverseFinalizeY(pRowNo));
17   
18       // conceptually, on each iteration of the i loop, we want:
19       //
20       //   x = Math.sin(INV_XPROJECT(i)) * sqrt(1 - (y*y));
21       //   z = cos(INV_XPROJECT(i)) * sqrt(1 - (y*y));
22       //
23       // computing this directly is rather expensive, however, so we only
24       // compute the first (i=0) pair of values directly; all other pairs
25       // (i>0) are obtained through successive rotations of the original
26       // pair (by inv_proj_scale radians).
27       //
28   
29       // compute initial (x, z) values
30       double tmp = Math.sqrt(1 - (y * y));
31       double x = Math.sin(fProjection.inverseFinalizeX(0)) * tmp;
32       double z = Math.cos(fProjection.inverseFinalizeX(0)) * tmp;
33   
34       // compute rotation coefficients used
35       // to find subsequent (x, z) values
36       tmp = 1 / fProjection.getScale();
37       final double sin_theta = Math.sin(tmp);
38       final double cos_theta = Math.cos(tmp);
39   
40       // save a little computation in the inner loop
41       final double lYBySunVectorY = y * fLightVector.getY();
42   
43       // use i_lim to encourage compilers to register loop limit
44       final Color[] lColors = new Color[pRowTypes.length];
45       for(int lColNo = 0; lColNo < pRowTypes.length; lColNo++)
46       {
47         double lSunValue = (x * fLightVector.getX()) + lYBySunVectorY +
48           (z * fLightVector.getZ());
49         lColors[lColNo] = getShadedColorForType(pRowTypes[lColNo], lSunValue);
50   
51         // compute next (x, z) values via 2-d rotation
52         tmp = (cos_theta * z) - (sin_theta * x);
53         x = (sin_theta * z) + (cos_theta * x);
54         z = tmp;
55       }
56   
57       return lColors;
58     }
59   }
