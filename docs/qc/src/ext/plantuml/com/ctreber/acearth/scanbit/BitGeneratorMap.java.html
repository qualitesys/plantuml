25   public abstract class BitGeneratorMap extends ScanBitGenerator {
26       // Types of pixels.
27       public static final int PixTypeSpace = 0;
28       public static final int PixTypeLand = 1;
29       public static final int PixTypeWater = 2;
30       public static final int PixTypeStar = 3;
31       public static final int PixTypeGridLand = 4;
32       public static final int PixTypeGridWater = 5;
33   
34       // Parameters influencing generateScanBits buffer genertion.
35       private Polygon[] fMapData;
36       Projection fProjection;
37       private List fScanbitsVector = new ArrayList();
38       // Created by scanPolygon
39       List fEdgeCrossings;
40   
41       abstract protected ScanBuf scanOutline();
42   
43       abstract protected void handleCrossings(ScanBuf pScanBuf, EdgeCrossing[] pEdgeCrossings);
44   
45       abstract protected Comparator getEdgeXingComparator();
46   
47       abstract protected void scanPolygon(ScanBuf pScanBuf, Point3D[] pPoints3D, Point2D[] pPoints2D, int pIndex);
48   
49       public BitGeneratorMap(Projection pProjection) {
50           fProjection = pProjection;
51       }

53       /**
54        * <p>
55        * Create outline for the map, scan all polygons.
56        */
57       public void generateScanBits() {
58           // Prepare data.
59           fScanbitsVector = new ArrayList();
60           fProjection.setImageWidth(fImageWidth);
61           fProjection.setImageHeight(fImageHeight);
62           fProjection.initTransformTable();
63   
64           // Trace outling and polygons.
65           final ScanBuf lScanBuf = scanOutline();
66           fScanbitsVector.addAll(lScanBuf.getScanbits(64));
67           scanPolygons();
68   
69           // Dress results.
70           final CTSort lSort = new QuickSort();
71           fScanBitsArray = (ScanBit[]) fScanbitsVector.toArray(new ScanBit[0]);
72           lSort.sort(fScanBitsArray);
73       }
74   
75       private void scanPolygons() {
76           for (int lPolyNo = 0; lPolyNo < fMapData.length; lPolyNo++) {
77               Polygon lPolygon = fMapData[lPolyNo];
78   
79               Point3D[] lPoints3D = new Point3D[lPolygon.getSize()];
80               Point2D[] lPoints2D = new Point2D[lPolygon.getSize()];
81               transformPolygonPoints(lPolygon, lPoints3D, lPoints2D);
82   
83               // For all points in polygon...
84               fEdgeCrossings = new ArrayList();
85               ScanBuf lScanBuf = new ScanBuf(fImageHeight, fImageWidth);
86               for (int i = 0; i < lPoints2D.length; i++) {
87                   scanPolygon(lScanBuf, lPoints3D, lPoints2D, i);
88               }
89   
90               if (fEdgeCrossings.size() > 0) {
91                   // Edge crossings have been generated, deal with them.
92                   final EdgeCrossing[] xings = (EdgeCrossing[]) fEdgeCrossings.toArray(new EdgeCrossing[0]);
93                   final CTSort lSort = new QuickSort();
94                   lSort.sort(xings, getEdgeXingComparator());
95                   handleCrossings(lScanBuf, xings);
96               }
97   
98               if (lScanBuf.containsPoints()) {
99                   // Scan lines have been generated, transform them into ScanBit.
100                   fScanbitsVector.addAll(lScanBuf.getScanbits(lPolygon.getType()));
101               }
102           }
103       }
104   
105       /**
106        * The transformation rotates 3D and projects 2D points from it
107        */
108       private void transformPolygonPoints(Polygon pPolygon, Point3D[] pPoints3D, Point2D[] pPoints2D) {
109           for (int i = 0; i < pPolygon.getPoints().length; i++) {
110               Point3D lPoint = pPolygon.getPoints()[i];
111   
112               Point3D lPointRotated = fProjection.rotate(lPoint);
113               pPoints3D[i] = lPointRotated;
114               pPoints2D[i] = fProjection.project2D(lPointRotated);
115           }
116       }
117

119           fMapData = pMapData;
120       }
121   
122       protected void addEdgeXing(EdgeCrossing pEdgeXing) {
123           fEdgeCrossings.add(pEdgeXing);
124       }
