1   package ext.plantuml.com.ctreber.acearth.util;
2   
3   import java.util.Calendar;
4   import java.util.Date;
5   import java.util.TimeZone;
6   
7   /**
8    * <p>Calculates the position of the point on Earth which is directly
9    * below the sun or the moon.
10    *
11    * <p>&copy; 2002 Christian Treber, ct@ctreber.com
12    * @author Christian Treber, ct@ctreber.com
13    *
14    */
15   public class SunPositionCalculator
16   {
17     /*
18       * the epoch upon which these astronomical calculations are based is
19       * 1990 january 0.0, 631065600 seconds since the beginning of the
20       * "unix epoch" (00:00:00 GMT, Jan. 1, 1970)
21       *
22       * given a number of seconds since the start of the unix epoch,
23       * daysSinceEpoch() computes the number of days since the start of the
24       * astronomical epoch (1990 january 0.0)
25       */
26   
27     private static final long EPOCH_START = 631065600000l;
28   
29     /*
30     * assuming the apparent orbit of the sun about the earth is circular,
31     * the rate at which the orbit progresses is given by RadsPerDay --
32     * TWOPI radians per orbit divided by 365.242191 days per year:
33     */
34   
35     private static final double RADS_PER_DAY = Toolkit.TWOPI / 365.242191;
36   
37     /*
38     * details of sun's apparent orbit at epoch 1990.0 (after
39     * duffett-smith, table 6, section 46)
40     *
41     * Epsilon_g    (ecliptic longitude at epoch 1990.0) 279.403303 degrees
42     * OmegaBar_g   (ecliptic longitude of perigee)      282.768422 degrees
43     * Eccentricity (eccentricity of orbit)                0.016713
44     */
45   
46     private static final double EPSILON_G = Toolkit.degsToRads(279.403303);
47     private static final double OMEGA_BAR_G = Toolkit.degsToRads(282.768422);
48     private static final double ECCENTRICITY = 0.016713;
49   
50     /*
51     * Lunar parameters, epoch January 0, 1990.0
52     */
53     private static final double MOON_MEAN_LONGITUDE = Toolkit.degsToRads(318.351648);
54     private static final double MOON_MEAN_LONGITUDE_PERIGEE = Toolkit.degsToRads(36.340410);
55     private static final double MOON_MEAN_LONGITUDE_NODE = Toolkit.degsToRads(318.510107);
56     private static final double MOON_INCLINATION = Toolkit.degsToRads(5.145396);
57   
58     private static final double SIDERAL_MONTH = 27.3217;
59   
60     /**
61      * <p>Calculate the position of the mean sun: where the sun would
62      * be if the earth's orbit were circular instead of ellipictal.
63      *
64      * <p>Verified.
65      *
66      * @param pDays days since ephemeris epoch
67      */
68     private static double getMeanSunLongitude(double pDays)
69     {
70       double N, M;
71   
72       N = RADS_PER_DAY * pDays;
73       N = Toolkit.fmod(N, 0, Toolkit.TWOPI);
74       if(N < 0)
75       {
76         N += Toolkit.TWOPI;
77       }
78   
79       M = N + EPSILON_G - OMEGA_BAR_G;
80       if(M < 0)
81       {
82         M += Toolkit.TWOPI;
83       }
84   
85       return M;
86     }
87   
88     /**
89      * <p>Compute ecliptic longitude of sun (in radians)
90      * (after duffett-smith, section 47)
91      *
92      * <p>Verified.
93      *
94      * @param pMillis Milliseconds since unix epoch
95      */
96     private static double getSunEclipticLongitude(long pMillis)
97     {
98       final double lDays = daysSinceEpoch(pMillis);
99       final double M_sun = getMeanSunLongitude(lDays);
100   
101       final double E = doKepler(M_sun);
102       final double v = 2 * Math.atan(Math.sqrt((1 + ECCENTRICITY) / (1 - ECCENTRICITY)) * Math.tan(E / 2));
103   
104       return (v + OMEGA_BAR_G);
105     }
106   
107     static double daysSinceEpoch(long pMillis)
108     {
109       return (double)(pMillis - EPOCH_START) / 24 / 3600 / 1000;
110     }
111   
112     /**
113      * solve Kepler's equation via Newton's method
114      * (after duffett-smith, section 47)
115      *
116      * <p>Verified.
117      */
118     private static double doKepler(double M)
119     {
120       double E;
121       double lDelta;
122   
123       E = M;
124       while(true)
125       {
126         lDelta = E - ECCENTRICITY * Math.sin(E) - M;
127         if(Math.abs(lDelta) <= 1e-10)
128         {
129           break;
130         }
131         E -= lDelta / (1 - ECCENTRICITY * Math.cos(E));
132       }
133   
134       return E;
135     }

138     /**
139      * <p>computing julian dates (assuming gregorian calendar, thus this is
140      * only valid for dates of 1582 oct 15 or later)
141      * (after duffett-smith, section 4)
142      *
143      * <p>Verified.
144      *
145      * @param pYear year (e.g. 19xx)
146      * @param pMonth month (jan=1, feb=2, ...)
147      * @param pDay day of month
148      */
149     private static double getJulianDate(int pYear, int pMonth, int pDay)
150     {
151       if((pMonth == 1) || (pMonth == 2))
152       {
153         pYear -= 1;
154         pMonth += 12;
155       }
156   
157       final int A = pYear / 100;
158       final int B = 2 - A + (A / 4);
159       final int C = (int)(365.25 * pYear);
160       final int D = (int)(30.6001 * (pMonth + 1));
161   
162       return B + C + D + pDay + 1720994.5;
163     }

166     /**
167      * <p>compute greenwich mean sidereal time (getGST) corresponding to a given
168      * number of milliseconds since the unix epoch
169      * (after duffett-smith, section 12)
170      *
171      * <p>Verified.
172      */
173     private static double getGST(long pMillis)
174     {
175       final Calendar lCal = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
176       lCal.setTime(new Date(pMillis));
177   
178       final double lJulianDate = getJulianDate(lCal.get(Calendar.YEAR), lCal.get(Calendar.MONTH) + 1,
179         lCal.get(Calendar.DAY_OF_MONTH));
180       final double T = (lJulianDate - 2451545) / 36525;
181       double T0 = ((T + 2.5862e-5) * T + 2400.051336) * T + 6.697374558;
182   
183       T0 = Toolkit.fmod(T0, 0, 24.0);
184       if(T0 < 0)
185       {
186         T0 += 24;
187       }
188   
189       final double UT = lCal.get(Calendar.HOUR_OF_DAY) +
190         (lCal.get(Calendar.MINUTE) + lCal.get(Calendar.SECOND) / 60.0) / 60.0;
191   
192       T0 += UT * 1.002737909;
193       T0 = Toolkit.fmod(T0, 0, 24.0);
194       if(T0 < 0)
195       {
196         T0 += 24;
197       }
198   
199       return T0;
200     }

202     /**
203      * <p>Given a particular time (expressed in milliseconds since the unix
204      * epoch), compute position on the earth (lat, lon) such that sun is
205      * directly overhead.
206      *
207      * <p>Verified.
208      *
209      * @param pMillis seconds since unix epoch
210      *
211      */
212     public static Coordinate getSunPositionOnEarth(long pMillis)
213     {
214       final Coordinate lSunPosEc = new Coordinate(0.0, getSunEclipticLongitude(pMillis));
215       final Coordinate lSunPosEq = lSunPosEc.eclipticToEquatorial();
216   
217       final double lRA = Toolkit.limitRads(lSunPosEq.getRA() - (Toolkit.TWOPI / 24) * getGST(pMillis));
218   
219       return new Coordinate(Toolkit.radsToDegs(lSunPosEq.getDE()), Toolkit.radsToDegs(lRA));
220     }
221   
222     /**
223      * <p>Given a particular time (expressed in milliseconds since the unix
224      * epoch), compute position on the earth (lat, lon) such that the
225      * moon is directly overhead.
226      *
227      * Based on duffett-smith **2nd ed** section 61; combines some steps
228      * into single expressions to reduce the number of extra variables.
229      *
230      * <p>Verified.
231      */
232     public static Coordinate getMoonPositionOnEarth(long pMillis)
233     {
234       final double lDays = daysSinceEpoch(pMillis);
235       double lSunLongEc = getSunEclipticLongitude(pMillis);
236       final double Ms = getMeanSunLongitude(lDays);
237   
238       double L = Toolkit.limitRads(Toolkit.fmod(lDays / SIDERAL_MONTH, 0, 1.0) * Toolkit.TWOPI + MOON_MEAN_LONGITUDE);
239       double Mm = Toolkit.limitRads(L - Toolkit.degsToRads(0.1114041 * lDays) - MOON_MEAN_LONGITUDE_PERIGEE);
240       double N = Toolkit.limitRads(MOON_MEAN_LONGITUDE_NODE - Toolkit.degsToRads(0.0529539 * lDays));
241       final double Ev = Toolkit.degsToRads(1.2739) * Math.sin(2.0 * (L - lSunLongEc) - Mm);
242       final double Ae = Toolkit.degsToRads(0.1858) * Math.sin(Ms);
243       Mm += Ev - Ae - Toolkit.degsToRads(0.37) * Math.sin(Ms);
244       final double Ec = Toolkit.degsToRads(6.2886) * Math.sin(Mm);
245       L += Ev + Ec - Ae + Toolkit.degsToRads(0.214) * Math.sin(2.0 * Mm);
246       L += Toolkit.degsToRads(0.6583) * Math.sin(2.0 * (L - lSunLongEc));
247       N -= Toolkit.degsToRads(0.16) * Math.sin(Ms);
248   
249       L -= N;
250       lSunLongEc = Toolkit.limitRads((Math.abs(Math.cos(L)) < 1e-12) ?
251         (N + Math.sin(L) * Math.cos(MOON_INCLINATION) * Math.PI / 2) :
252         (N + Math.atan2(Math.sin(L) * Math.cos(MOON_INCLINATION), Math.cos(L))));
253       final double lSunLatEc = Math.asin(Math.sin(L) * Math.sin(MOON_INCLINATION));
254   
255       final Coordinate lSunPosEq = new Coordinate(lSunLatEc, lSunLongEc).eclipticToEquatorial();
256       final double lRA = Toolkit.limitRads(lSunPosEq.getRA() - (Toolkit.TWOPI / 24) * getGST(pMillis));
257   
258       return new Coordinate(Toolkit.radsToDegs(lSunPosEq.getDE()), Toolkit.radsToDegs(lRA));
259     }
