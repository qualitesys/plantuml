8   /**
9    * <p>Generate latitude and longitude grid as dots.
10    *
11    * <p>Refactored 08.11.2002
12    *
13    * <p>&copy; 2002 Christian Treber, ct@ctreber.com
14    * @author Christian Treber, ct@ctreber.com
15    *
16    */
17   public class DotGeneratorLines extends ScanDotGenerator
18   {
19     private Projection fProjection;
20     private int fLineDivider;
21     private int fPointDivider;
22     private static final double PI = Math.PI;
23     private static final double TWOPI = 2 * PI;
24     private static final double HALFPI = PI / 2;
25   
26     public DotGeneratorLines(Projection pProjection,
27       int pLineDevider, int pPointDivider)
28     {
29       fProjection = pProjection;
30       fLineDivider = pLineDevider;
31       fPointDivider = pPointDivider;
32     }
33   
34     /**
35      * <p>Paint grid.
36      */
37     public void generateScanDots()
38     {
39       double lLonStep = TWOPI / (fLineDivider * 4);
40       double lLatStep = PI / (fLineDivider * 2 * fPointDivider);
41       for(double lLon = -PI; lLon <= PI; lLon += lLonStep)
42       {
43         for(double lLat = -HALFPI; lLat <= HALFPI; lLat += lLatStep)
44         {
45           transformAndAddDot(new Coordinate(lLat, lLon));
46         }
47       }
48   
49       lLatStep = TWOPI / (fLineDivider * 4);
50       lLonStep = PI / (fLineDivider * 2 * fPointDivider);
51       for(double lLat = -HALFPI; lLat <= HALFPI; lLat += lLatStep)
52       {
53         for(double lLon = -PI; lLon <= PI; lLon += lLonStep)
54         {
55           transformAndAddDot(new Coordinate(lLat, lLon));
56         }
57       }
58     }
59   
60     private void transformAndAddDot(Coordinate pPos)
61     {
62       final Point3D lPointRotated = fProjection.rotate(pPos.getPoint3DRads());
63       if(fProjection.isVisible(lPointRotated))
64       {
65         Point2D lPoint = fProjection.finalize(fProjection.project2D(lPointRotated));
66         if(fProjection.isWithinImage(lPoint))
67         {
68           fDots.add(new ScanDot(ScanDot.DotTypeGrid, lPoint));
69         }
70       }
71     }
72   }
