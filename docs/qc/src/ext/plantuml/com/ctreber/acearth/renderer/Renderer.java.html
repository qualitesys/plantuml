10   /**
11    * <p>
12    * Uses defined RowTypeRenderers and Shader to render to render target.
13    * 
14    * <p>
15    * &copy; 2002 Christian Treber, ct@ctreber.com
16    * 
17    * @author Christian Treber, ct@ctreber.com
18    * 
19    */
20   public class Renderer {
21       private Shader fShader;
22       private RenderTarget fRenderTarget;
23       private List fRowTypeRenderers = new ArrayList();
24   
25       public Renderer(RenderTarget pRenderTarget) {
26           fRenderTarget = pRenderTarget;
27       }
28   
29       public void render() {
30           final Iterator lIt = fRowTypeRenderers.iterator();
31           while (lIt.hasNext()) {
32               RowTypeRenderer lRowRenderer = (RowTypeRenderer) lIt.next();
33               lRowRenderer.startNewRun();
34           }
35           renderRows();
36       }
37   
38       private void renderRows() {
39           for (int lRowNo = 0; lRowNo < fRenderTarget.getImageHeight(); lRowNo++) {
40               int[] lPixelTypes = getPixelTypes(lRowNo);
41               renderRow(lRowNo, lPixelTypes);
42           }
43       }
44   
45       /**
46        * <p>
47        * Get pixel types for whole row from all registered RowRenderers.
48        * 
49        * @param pRowNo
50        *            Row number.
51        * @return Pixel types for row.
52        */
53       private int[] getPixelTypes(int pRowNo) {
54           // Create the types array
55           final int[] lPixelTypes = new int[fRenderTarget.getImageWidth()];
56   
57           final Iterator lIt = fRowTypeRenderers.iterator();
58           while (lIt.hasNext()) {
59               RowTypeRenderer lRowRenderer = (RowTypeRenderer) lIt.next();
60               lRowRenderer.getPixelTypes(pRowNo, lPixelTypes);
61           }
62   
63           return lPixelTypes;
64       }
65   
66       /**
67        * <p>
68        * With help of Shader, render pixel types to actual colored pixels.
69        * 
70        * @param pRowNo
71        * @param pPixelTypes
72        */
73       private void renderRow(int pRowNo, int[] pPixelTypes) {
74           // For each pixel in row, render it.
75           final Color[] lPixelColors = fShader.getShadedColors(pRowNo, pPixelTypes);
76           for (int lColNo = 0; lColNo < fRenderTarget.getImageWidth(); lColNo++) {
77               fRenderTarget.setPixel(lColNo, pRowNo, lPixelColors[lColNo]);
78           }
79       }
80   
81       public void setShader(Shader pShader) {
82           fShader = pShader;
83       }
84   
85       public void setRenderTarget(RenderTarget pRenderTarget) {
86           fRenderTarget = pRenderTarget;
87       }
88   
89       public void addRowTypeRenderer(RowTypeRenderer pRowRenderer) {
90           fRowTypeRenderers.add(pRowRenderer);
91       }
92   }
