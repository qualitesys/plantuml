55     {
56       while(pValue < 0)

64

66     }

69      * <p>Examples: min -2, max 2: range 4
70      *
71      * <ul>
72      * <li> value 1: lFact = 0
73      * <li> value 3: lFact = 1, value -1
74      * <li> value 9: lFact = 2, value 1
75      * <li> value -3: lFact = -1, value 1
76      * </ul>
77      */
78     public static double fmod(double pValue, double pMinValue, double pMaxValue)
79     {
80       final double lRange = pMaxValue - pMinValue;
81       int lFact = (int)((pValue - pMinValue) / lRange);
82       if(pValue < pMinValue)
83       {
84         lFact--;
85       }
86   
87       return pValue - lFact * lRange;
88     }

90     /**
91      * <p>Capitalize String. Uppercase words smaller/equal than 3 chars,
92      * lowercase defined exceptions. Capitalize within word after '.' and '-'.
93      * Capitalize all others.
94      */
95     public static String intelligentCapitalize(String pText)
96     {
97       boolean lDoCap = false;
98       final StringTokenizer lST = new StringTokenizer(pText, ".- ", true);
99       final StringBuilder lSB = new StringBuilder(50);
100       while(lST.hasMoreTokens())
101       {
102         String lWord = lST.nextToken();
103   
104         if(lWord.equals(".") || lWord.equals("-"))
105         {
106           lDoCap = true;
107           lSB.append(lWord);
108           continue;
109         }
110         if(lWord.equals(" "))
111         {
112           lDoCap = false;
113           lSB.append(lWord);
114           continue;
115         }
116   
117         if(lDoCap || (lWord.length() > 3))
118         {
119           lSB.append(Character.toUpperCase(lWord.charAt(0)));
120           lSB.append(lWord.substring(1).toLowerCase());
121         } else
122         {
123           if(fsNoCap.contains(lWord.toLowerCase()))
124           {
125             lSB.append(lWord.toLowerCase());
126           } else
127           {
128             lSB.append(lWord.toUpperCase());
129           }
130         }
131       }
132   
133       return lSB.toString();
134     }
