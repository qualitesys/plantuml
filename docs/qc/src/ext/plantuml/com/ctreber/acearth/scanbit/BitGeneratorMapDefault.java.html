32     protected ScanBuf scanOutline()
33     {
34       final ScanBuf lScanBuf = new ScanBuf(fImageHeight, fImageWidth);
35   
36       final double lLeft = fProjection.finalizeX(-Math.PI);
37       final double lRight = fProjection.finalizeX(Math.PI);
38       // Will be adjusted to fit height.
39       final double lTop = fProjection.finalizeY(1e6);
40       final double lBottom = fProjection.finalizeY(-1e6);
41   
42       // Top

44       // Left
45       lScanBuf.addLine(lLeft, lTop, lLeft, lBottom);
46       // Bottom
47       lScanBuf.addLine(lLeft, lBottom, lRight, lBottom);
48       // Right
49       lScanBuf.addLine(lRight, lBottom, lRight, lTop);
50

54     /**
55      * <p>Look at 2 neighboring points in polygon and find edge crossings -
56      * edge of the globe?! Edge of image?
57      */
58     protected void scanPolygon(ScanBuf pScanBuf,
59       Point3D[] pPoints3D, Point2D[] pPoints2D, int pIndex)
60     {
61       final Point2D lCurr = pPoints2D[pIndex];
62       final int lIndexPrev = pIndex - 1 >= 0 ? pIndex - 1 : pPoints2D.length - 1;
63       final Point2D lPrev = pPoints2D[lIndexPrev];
64       double dx = lCurr.getX() - lPrev.getX();
65   
66       if(Math.abs(dx) <= Math.PI)
67       {
68         // Perimeter not crossed.
69         pScanBuf.addLine(
70           fProjection.finalizeX(lPrev.getX()), fProjection.finalizeY(lPrev.getY()),
71           fProjection.finalizeX(lCurr.getX()), fProjection.finalizeY(lCurr.getY()));
72         return;
73       }
74   
75       // Perimeter crossed, we need to wrap the line around the edge.
76       int lAngle;
77       double mx;
78       double my = getYMidPoint(pPoints3D[lIndexPrev], pPoints3D[pIndex]);
79       if(dx > 0)
80       {
81         // Curve runs right
82         mx = -Math.PI;
83         lAngle = 2;
84       } else
85       {
86         mx = Math.PI;
87         lAngle = 0;
88       }
89   
90       // From previous point to edge...
91       pScanBuf.addLine(
92         fProjection.finalizeX(lPrev.getX()), fProjection.finalizeY(lPrev.getY()),
93         fProjection.finalizeX(mx), fProjection.finalizeY(my));
94       addEdgeXing(new EdgeCrossing(EdgeCrossing.XingTypeExit, pIndex, mx, my, lAngle));
95   
96       if(dx > 0)
97       {
98         mx = Math.PI;
99         lAngle = 0;
100       } else
101       {
102         mx = -Math.PI;
103         lAngle = 2;
104       }
105   
106       // ...and from edge to current point.
107       pScanBuf.addLine(
108         fProjection.finalizeX(mx), fProjection.finalizeY(my),
109         fProjection.finalizeX(lCurr.getX()), fProjection.finalizeY(lCurr.getY()));
110       addEdgeXing(new EdgeCrossing(EdgeCrossing.XingTypeEntry, pIndex, mx, my, lAngle));
111     }

113     /**
114      * <p>My educated guess is that the mid point between the current and
115      * the previous point is calculated, and - kind of - y of that point
116      * is returned.
117      */
118     private double getYMidPoint(Point3D pPrev, Point3D pCurr)
119     {
120       double lY;
121       final double lZ;
122   
123       if(pCurr.getX() != 0)
124       {
125         // if xPrev is twice xCurr, ratio is 2
126         double ratio = (pPrev.getX() / pCurr.getX());
127         lY = pPrev.getY() - ratio * pCurr.getY();
128         lZ = pPrev.getZ() - ratio * pCurr.getZ();
129       } else
130       {
131         lY = pCurr.getY();
132         lZ = pCurr.getZ();
133       }
134   
135       final double lDistance = Math.sqrt((lY * lY) + (lZ * lZ));
136       lY *= ((lZ > 0) ? -1 : 1) / lDistance;
137   
138       return fProjection.projectY(lY);
139     }

141     /**
142      * <p>Side effect: Creates ScanBuf lines.
143      */
144     protected void handleCrossings(ScanBuf pScanBuf, EdgeCrossing[] xings)
145     {
146       EdgeCrossing from;
147       EdgeCrossing to;
148       int lStart;
149   
150       if(xings[0].getType() == EdgeCrossing.XingTypeExit)
151       {
152         lStart = 0;
153       } else
154       {
155         // Type "entry".
156         from = xings[xings.length - 1];
157         to = xings[0];
158         addEdgeToScanbuf(pScanBuf, from, to);
159         lStart = 1;
160       }
161   
162       for(int i = lStart; i < xings.length - 1; i += 2)
163       {
164         from = xings[i];
165         to = xings[i + 1];
166         addEdgeToScanbuf(pScanBuf, from, to);
167       }
168     }

170     /**
171      * <p>For handleCrossing(). Side effect: Creates ScanBuf lines.
172      *
173      * @param pScanBuf
174      * @param from
175      * @param to
176      */
177     private void addEdgeToScanbuf(ScanBuf pScanBuf, EdgeCrossing from,
178       EdgeCrossing to)
179     {
180       int lAngleFrom = (int)from.getAngle();
181       double lXFrom = fProjection.finalizeX(from.getX());
182       double lYFrom = fProjection.finalizeY(from.getY());
183   
184       // Step around in 90 degree increments until target angle is reached
185       while(lAngleFrom != (int)to.getAngle())
186       {
187         int lAngleNew = 0;
188         double lXNew = 0;
189         double lYNew = 0;
190   
191         switch(lAngleFrom)
192         {
193           case 0:
194             // Top right
195             lXNew = fProjection.finalizeX(Math.PI);
196             lYNew = fProjection.finalizeY(1e6);
197             lAngleNew = 1;
198             break;
199   
200           case 1:
201             // Top left
202             lXNew = fProjection.finalizeX(-Math.PI);
203             lYNew = fProjection.finalizeY(1e6);
204             lAngleNew = 2;
205             break;
206   
207           case 2:
208             // Bottom left
209             lXNew = fProjection.finalizeX(-Math.PI);
210             lYNew = fProjection.finalizeY(-1e6);
211             lAngleNew = 3;
212             break;
213   
214           case 3:
215             // Bottom right
216             lXNew = fProjection.finalizeX(Math.PI);
217             lYNew = fProjection.finalizeY(-1e6);
218             lAngleNew = 0;
219             break;
220         }
221   
222         pScanBuf.addLine(lXFrom, lYFrom, lXNew, lYNew);
223   
224         lAngleFrom = lAngleNew;
225         lXFrom = lXNew;
226         lYFrom = lYNew;
227       }
228   
229       // ...and from last to final.
230       pScanBuf.addLine(lXFrom, lYFrom, fProjection.finalizeX(to.getX()),
231         fProjection.finalizeY(to.getY()));
232     }

234     private static class EdgeXingComparator implements Comparator
235     {
236       public int compare(Object o1, Object o2)
237       {
238         final EdgeCrossing a = (EdgeCrossing)o1;
239         final EdgeCrossing b = (EdgeCrossing)o2;
240   
241         if(a.getAngle() < b.getAngle())
242         {
243           return -1;
244         }
245   
246         if(a.getAngle() > b.getAngle())
247         {
248           return 1;
249         }
250   
251         // Angles are equal.
252         if(a.getAngle() == 0)
253         {
254           return (a.getY() < b.getY()) ? -1 : (a.getY() > b.getY()) ? 1 : 0;
255         }
256   
257         if(a.getAngle() == 2)
258         {
259           return (a.getY() > b.getY()) ? -1 : (a.getY() < b.getY()) ? 1 : 0;
260         }
261   
262         throw new RuntimeException("No result");
263       }
264     }
