6   /**
7    * <p>Renders a row of ScanBits to pixel types.</p>
8    *
9    * <p>&copy; 2002 Christian Treber, ct@ctreber.com (Nov 11, 2002)</p>
10    * @author Christian Treber, ct@ctreber.com
11    *
12    */
13   public class RowTypeRendererScanBit implements RowTypeRenderer
14   {
15     private int fScanBitIndex;
16     private ScanBit[] fScanBits;
17     private final int[] fScanToPixelType = new int[256];
18   
19     public void startNewRun()
20     {
21       fScanBitIndex = 0;
22       generateScanToPixelTypeTable();
23     }
24   
25     public void getPixelTypes(int pRowNo, int[] pPixelTypes)
26     {
27       // For all ScanBits in specified row...
28       while((fScanBitIndex < fScanBits.length) &&
29         (fScanBits[fScanBitIndex].getY() == pRowNo))
30       {
31         for(int i = fScanBits[fScanBitIndex].getlXFrom();
32             i <= fScanBits[fScanBitIndex].getXTo(); i++)
33         {
34           /**
35            * This is weird... why summing up the types? Note the row stays the
36            * same, but it possibly gets paved over a couple of times (There
37            * might be ScanBits painting on the same pixels).
38            *
39            * The polygons specify -1 as water and 1 as land.
40            * The type table says space is 0, Water is 1 to 64, Land is 65+.
41            *
42            * The outline paints the whole world as water (64). Adding a
43            * land pixel (1) creates a value of 65 (land). Adding a water
44            * pixel (-1) changes this back to 64 (water).
45            */
46           pPixelTypes[i] += fScanBits[fScanBitIndex].getType();
47         }
48         fScanBitIndex++;
49       }
50   
51       // Translate generateScanBits values into pixels types.
52       for(int lCol = 0; lCol < pPixelTypes.length; lCol++)
53       {
54         pPixelTypes[lCol] = fScanToPixelType[pPixelTypes[lCol] & 0xff];
55       }
56     }
57   
58     private void generateScanToPixelTypeTable()
59     {
60       for(int i = 0; i < 256; i++)
61       {
62         if(i == 0)
63         {
64           // 0 is Space.
65           fScanToPixelType[i] = BitGeneratorMap.PixTypeSpace;
66         } else if(i > 64)
67         {
68           // Above 64 it's land.
69           fScanToPixelType[i] = BitGeneratorMap.PixTypeLand;
70         } else
71         {
72           // From 1 to 64 incl. it's water
73           fScanToPixelType[i] = BitGeneratorMap.PixTypeWater;
74         }
75       }
76     }
77   
78     public void setScanBits(ScanBit[] pScanBits)
79     {
80       fScanBits = pScanBits;
81     }
82   }
