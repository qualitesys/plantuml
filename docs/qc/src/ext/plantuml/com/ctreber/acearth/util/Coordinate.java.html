17   public class Coordinate {
18       /*
19        * MeanObliquity gives the mean obliquity of the earth's axis at epoch
20        * 1990.0 (computed as 23.440592 degrees according to the method given in
21        * duffett-smith, section 27)
22        */
23       private static final double MEAN_OBLIQUITY = 23.440592 * Toolkit.TWOPI / 360;

40        * @param pLat
41        *            Latitude or DE

55       public Point3D getPoint3D() {
56           final double lLatRad = Toolkit.degsToRads(fLat);
57           final double lLongRad = Toolkit.degsToRads(fLong);
58   
59           final double lX = Math.cos(lLatRad) * Math.sin(lLongRad);
60           final double lY = Math.sin(lLatRad);
61           final double lZ = Math.cos(lLatRad) * Math.cos(lLongRad);
62   
63           return new Point3D(lX, lY, lZ);
64       }

66       /**

68        * Assumes coordinate is not in degrees but rads.
69        * 
70        * @return
71        */
72       public Point3D getPoint3DRads() {
73           final double lX = Math.cos(fLat) * Math.sin(fLong);
74           final double lY = Math.sin(fLat);
75           final double lZ = Math.cos(fLat) * Math.cos(fLong);
76   
77           return new Point3D(lX, lY, lZ);
78       }

80       /**
81        * <p>
82        * Convert from ecliptic to equatorial coordinates (after duffett-smith,
83        * section 27)
84        */
85       public Coordinate eclipticToEquatorial() {
86           final double sin_e = Math.sin(MEAN_OBLIQUITY);
87           final double cos_e = Math.cos(MEAN_OBLIQUITY);
88   
89           final double lRA = Math.atan2(Math.sin(fLong) * cos_e - Math.tan(fLat) * sin_e, Math.cos(fLong));
90           final double lDE = Math.asin(Math.sin(fLat) * cos_e + Math.cos(fLat) * sin_e * Math.sin(fLong));
91   
92           return new Coordinate(lDE, lRA);
93       }

95       /**
96        * <p>
97        * Add position to this position, make sure coordinates are valid.
98        */
99       public void add(Coordinate lOther) {
100           fLat += lOther.fLat;
101           fLong += lOther.fLong;
102           wrap();
103       }

105       /**
106        * <p>
107        * Warp coordinates exceeding valid values. Happens when latitudes and
108        * longitudes are added or substracted.
109        */
110       public void wrap() {
111           if (fLat > 90) {
112               fLat = 180 - fLat;
113               fLong += 180;
114           } else if (fLat < -90) {
115               fLat = -180 - fLat;
116               fLong += 180;
117           }
118   
119           if (fLong > 180) {
120               do {
121                   fLong -= 360;
122               } while (fLong > 180);
123           } else if (fLong < -180) {
124               do {
125                   fLong += 360;
126               } while (fLong < -180);
127           }
128       }
