2   
3   import java.io.IOException;
4   import java.util.ArrayList;
5   import java.util.Iterator;
6   import java.util.List;
7   import java.util.StringTokenizer;
8   
9   import ext.plantuml.com.ctreber.acearth.util.Point3D;
10   import ext.plantuml.com.ctreber.acearth.util.Polygon;
11   
12   /**
13    * The map data file is a big array of short (16-bit) ints, as follows: - it is
14    * a sequence of closed curves - the first value in a curve is the number of
15    * points in the curve - the second value in a curve indicates land/water (1 or
16    * -1, respectively) - this is followed by an [x,y,z] triple that indicates a
17    * point on the unit sphere (each of x, y, and z has been scaled by 30000),
18    * where the x axis points "to the right" (towards 0 N 90 E), the y axis points
19    * "up" (towards the north pole), and the z axis points "out of the screen"
20    * (towards 0 N 0 E). this is the starting point of the curve. - this is
21    * followed by (one less than the number of points in the curve) triples
22    * [dx,dy,dz]; the [x,y,z] triple for each successive point in the curve is
23    * obtained by adding [dx,dy,dz] onto the previous [x,y,z] values. - the curves
24    * are [must be!] non-self-intersecting and traced in a counter-clockwise
25    * direction
26    *
27    * the curves are sampled at a (roughly) a 20 mile resolution.
28    *
29    * <p>
30    * &copy; 2002 Christian Treber, ct@ctreber.com
31    *
32    * @author Christian Treber, ct@ctreber.com
33    *
34    */
35   public class MapDataReader {
36       /** Point value scale (devide value by this number). */
37       private static final double MAP_DATA_SCALE = 30000.0;
38   
39       private static List fData;
40       private static List fPolygons;
41       private static int fIndex;
42   
43       /**
44        * <p>
45        * Read map data.
46        *
47        * @return Array of map polygons.
48        * @throws IOException
49        */
50       public static Polygon[] readMapData() throws IOException {
51           final List lines = new MapData().getLines();
52   
53           fData = new ArrayList();
54           for (Iterator it = lines.iterator(); it.hasNext(); ) {
55               String lLine = (String) it.next();
56               if (lLine.contains("/*")) {
57                   // Filter out comments.
58                   continue;
59               }
60   
61               StringTokenizer lST = new StringTokenizer(lLine, ", ");
62               while (lST.hasMoreTokens()) {
63                   String lToken = lST.nextToken();
64                   final Integer lValue = Integer.valueOf(lToken);
65                   fData.add(lValue);
66               }
67           }
68   
69           fPolygons = new ArrayList();
70           fIndex = 0;
71           while (getValue(fIndex) != 0) {
72               processCurve();
73           }
74   
75           return (Polygon[]) fPolygons.toArray(new Polygon[0]);
76       }
77   
78       private static void processCurve() {
79           final int lNPoint = getValue(fIndex++);
80           final int lType = getValue(fIndex++);
81   
82           final Point3D[] lPoints = new Point3D[lNPoint];
83           final Point3D lPoint3D = new Point3D(getValue(fIndex++) / MAP_DATA_SCALE, getValue(fIndex++) / MAP_DATA_SCALE,
84                   getValue(fIndex++) / MAP_DATA_SCALE);
85   
86           lPoints[0] = lPoint3D;
87           for (int i = 1; i < lNPoint; i++) {
88               lPoints[i] = new Point3D(lPoints[i - 1].getX() + getValue(fIndex++) / MAP_DATA_SCALE, lPoints[i - 1].getY()
89                       + getValue(fIndex++) / MAP_DATA_SCALE, lPoints[i - 1].getZ() + getValue(fIndex++) / MAP_DATA_SCALE);
90           }
91   
92           final Polygon lPolygon = new Polygon(lType, lPoints);
93           fPolygons.add(lPolygon);
94       }
95   
96       /**
97        * <p>
98        * Get value of raw data at specified point.
99        *
100        * @param pIndex
101        *            Index of value.
102        * @return Value of raw data at specified point.
103        */
104       private static int getValue(int pIndex) {
105           return ((Integer) fData.get(pIndex)).intValue();
106       }
107   
108   }
