135           Projection lProjection = null;

137           BitGeneratorMap lScanner = null;

145               lProjection = new ProjectionMerc();
146               lScanner = new BitGeneratorMapDefault(lProjection);
147               lShader = new ShaderDefault();
148           }
149

151               lProjection = new ProjectionOrtho();
152               lScanner = new BitGeneratorMapOrtho(lProjection);
153               lShader = new ShaderOrtho();
154           }

159           lProjection.setShiftX(fConf.getInt("shiftX"));

161           lProjection.setViewMagnification(fConf.getDouble("viewMagnification"));
162           lProjection.setViewPos(fViewPos);
163           lProjection.setViewRotation(fViewRotation);
164   
165           lScanner.setImageWidth(fConf.getInt("imageWidth"));

167           lScanner.setMapData(MapDataReader.readMapData());

227        * <p>
228        * This is repeated when time changes since this influences the position of
229        * Earth.
230        */
231       private void computePositions() {
232           // Determine time for rendering
233           if (fConf.getInt("fixedTime") == 0) {
234               // No fixed time.
235               // final long lTimePassed = System.currentTimeMillis() - fsStartTime;
236               // fCurrentTime = fsStartTime + (long) (fConf.getDouble("timeWarpFactor") * lTimePassed);
237               fCurrentTime = System.currentTimeMillis();
238           } else {
239               // Fixed time.
240               fCurrentTime = fConf.getInt("fixedTime") * 1000L;
241           }
242   
243           if (fConf.getBoolean("sunMovesP")) {
244               fConf.setSunPos(SunPositionCalculator.getSunPositionOnEarth(fCurrentTime));
245           }
246   
247           // Determine viewing position
248           if (fConf.is("viewPositionType", "Fixed")) {
249               fViewPos = fConf.getViewPos();
250           } else if (fConf.is("viewPositionType", "Sun-relative")) {
251               fViewPos = getSunRelativePosition();
252           } else if (fConf.is("viewPositionType", "Orbit")) {
253               fViewPos = getOrbitPosition(fCurrentTime);
254           } else if (fConf.is("viewPositionType", "Random")) {
255               fViewPos = getRandomPosition();
256           } else if (fConf.is("viewPositionType", "Moon")) {
257               fViewPos = SunPositionCalculator.getMoonPositionOnEarth(fCurrentTime);
258           }
259   
260           // for ViewRotGalactic, compute appropriate viewing rotation
261           if (fConf.is("viewRotationType", "Galactic")) {
262               fViewRotation = (Toolkit.degsToRads(fConf.getSunPos().getLat()
263                       * Math.sin((fViewPos.getLong() - fConf.getSunPos().getLong()))));
264           } else {
265               fViewRotation = fConf.getDouble("viewRotation");
266           }
267       }

282       private Coordinate getOrbitPosition(long pTimeMillis) {
283           double x, y, z;
284           double a, c, s;
285           double t1, t2;
286   
287           /* start at 0 N 0 E */
288           x = 0;
289           y = 0;
290           z = 1;
291   
292           /*
293            * rotate in about y axis (from z towards x) according to the number of
294            * orbits we've completed
295            */
296           a = (double) pTimeMillis / (fConf.getDouble("orbitPeriod") * 3600 * 1000) * 2 * Math.PI;
297           c = Math.cos(a);
298           s = Math.sin(a);
299           t1 = c * z - s * x;
300           t2 = s * z + c * x;
301           z = t1;
302           x = t2;
303   
304           /*
305            * rotate about z axis (from x towards y) according to the inclination
306            * of the orbit
307            */
308           a = Toolkit.degsToRads(fConf.getDouble("orbitInclination"));
309           c = Math.cos(a);
310           s = Math.sin(a);
311           t1 = c * x - s * y;
312           t2 = s * x + c * y;
313           x = t1;
314           y = t2;
315   
316           /*
317            * rotate about y axis (from x towards z) according to the number of
318            * rotations the earth has made
319            */
320           a = ((double) pTimeMillis / 86400000) * (2 * Math.PI);
321           c = Math.cos(a);
322           s = Math.sin(a);
323           t1 = c * x - s * z;
324           t2 = s * x + c * z;
325           x = t1;
326           z = t2;
327   
328           return new Coordinate(Toolkit.radsToDegs(Math.asin(y)), Toolkit.radsToDegs(Math.atan2(x, z)));
329       }

331       /**
332        * <p>
333        * Pick a position (lat, lon) at random
334        * 
335        * @return A random position.
336        */
337       private static Coordinate getRandomPosition() {
338   
339           /* select a vector at random */
340           final double[] pos = new double[3];
341           double mag = 0;
342           do {
343               for (int i = 0; i < 3; i++) {
344                   pos[i] = ((Math.random() * 20000) * 1e-4) - 1;
345                   mag += pos[i] * pos[i];
346               }
347           } while ((mag > 1.0) || (mag < 0.01));
348   
349           /* normalize the vector */
350           mag = Math.sqrt(mag);
351           for (int i = 0; i < 3; i++) {
352               pos[i] /= mag;
353           }
354   
355           /* convert to (lat, lon) */
356           final double s_lat = pos[1];
357           final double c_lat = Math.sqrt(1 - s_lat * s_lat);
358           final double s_lon = pos[0] / c_lat;
359           final double c_lon = pos[2] / c_lat;
360   
361           return new Coordinate(Math.atan2(s_lat, c_lat) * (180 / Math.PI), Math.atan2(s_lon, c_lon) * (180 / Math.PI));
362       }
