37     private void addArcToScanbuf(ScanBuf pScanBuf, double pXFrom, double pYFrom,
38       double pAngleFrom, double pXTo, double pYTo, double pAngleTo)
39     {
40       double step = 1 / fProjection.getScale() * 10;
41       if(step > 0.05)
42       {
43         step = 0.05;
44       }
45       final int lAngleFrom = (int)Math.ceil(pAngleFrom / step);
46       final int lAngleTo = (int)Math.floor(pAngleTo / step);
47   
48       double prev_x = fProjection.finalizeX(pXFrom);
49       double prev_y = fProjection.finalizeY(pYFrom);
50       double curr_x;
51       double curr_y;
52       if(lAngleFrom <= lAngleTo)
53       {
54         double c_step = Math.cos(step);
55         double s_step = Math.sin(step);
56   
57         double angle = lAngleFrom * step;
58         double arc_x = Math.cos(angle);
59         double arc_y = Math.sin(angle);
60   
61         for(int i = lAngleFrom; i <= lAngleTo; i++)
62         {
63           curr_x = fProjection.finalizeX(arc_x);
64           curr_y = fProjection.finalizeY(arc_y);
65           pScanBuf.addLine(prev_x, prev_y, curr_x, curr_y);
66   
67           /* instead of repeatedly calling cos() and sin() to get the next
68            * values for arc_x and arc_y, simply rotate the existing values
69            */
70           double tmp = (c_step * arc_x) - (s_step * arc_y);
71           arc_y = (s_step * arc_x) + (c_step * arc_y);
72           arc_x = tmp;
73   
74           prev_x = curr_x;
75           prev_y = curr_y;
76         }
77       }
78   
79       curr_x = fProjection.finalizeX(pXTo);
80       curr_y = fProjection.finalizeY(pYTo);
81       pScanBuf.addLine(prev_x, prev_y, curr_x, curr_y);
82     }

84     protected void scanPolygon(ScanBuf pScanBuf,
85       Point3D[] pPoints3D, Point2D[] pPoints2D, int pIndex)
86     {
87       Point3D extra;
88   
89       Point3D lCurr = pPoints3D[pIndex];
90       final int lIndexPrev = pIndex - 1 >= 0 ? pIndex - 1 : pPoints2D.length - 1;
91       Point3D lPrev = pPoints3D[lIndexPrev];
92   
93       if(lPrev.getZ() <= 0)
94       {
95         if(lCurr.getZ() <= 0)
96         {
97           return;
98         }
99   
100         // Previous point not visible, but current one is: horizon crossed.
101         extra = findEdgeCrossing(lPrev, lCurr);
102         addEdgeXing(new EdgeCrossing(EdgeCrossing.XingTypeEntry, pIndex,
103           extra.getX(), extra.getY(), Math.atan2(extra.getY(), extra.getX())));
104         lPrev = extra;
105       } else
106       {
107         if(lCurr.getZ() <= 0)
108         {
109           // Previous point visible, but current is not: horizon crossed.
110           extra = findEdgeCrossing(lPrev, lCurr);
111           addEdgeXing(new EdgeCrossing(EdgeCrossing.XingTypeExit, pIndex,
112             extra.getX(), extra.getY(), Math.atan2(extra.getY(), extra.getX())));
113           lCurr = extra;
114         }
115       }
116   
117       pScanBuf.addLine(
118         fProjection.finalizeX(lPrev.getX()), fProjection.finalizeY(lPrev.getY()),
119         fProjection.finalizeX(lCurr.getX()), fProjection.finalizeY(lCurr.getY()));
120     }

122     private Point3D findEdgeCrossing(Point3D pPrev, Point3D pCurr)
123     {
124       double tmp = pCurr.getZ() / (pCurr.getZ() - pPrev.getZ());
125       final double r0 = pCurr.getX() - tmp * (pCurr.getX() - pPrev.getX());
126       final double r1 = pCurr.getY() - tmp * (pCurr.getY() - pPrev.getY());
127   
128       tmp = Math.sqrt((r0 * r0) + (r1 * r1));
129   
130       return new Point3D(r0 / tmp, r1 / tmp, 0);
131     }

133     protected void handleCrossings(ScanBuf pScanBuf, EdgeCrossing[] xings)
134     {
135       EdgeCrossing from;
136       EdgeCrossing to;
137       int lStart;
138   
139       if(xings[0].getType() == EdgeCrossing.XingTypeExit)
140       {
141         lStart = 0;
142       } else
143       {
144         from = xings[xings.length - 1];
145         to = xings[0];
146         addArcToScanbuf(pScanBuf, from.getX(), from.getY(), from.getAngle(),
147           to.getX(), to.getY(), to.getAngle() + 2 * Math.PI);
148         lStart = 1;
149       }
150   
151       for(int i = lStart; i < xings.length - 1; i += 2)
152       {
153         from = xings[i];
154         to = xings[i + 1];
155         addArcToScanbuf(pScanBuf, from.getX(), from.getY(), from.getAngle(),
156           to.getX(), to.getY(), to.getAngle());
157       }
158     }

160     private static class EdgeCrossingComparator implements Comparator
161     {
162       public int compare(Object o1, Object o2)
163       {
164         final EdgeCrossing a = (EdgeCrossing)o1;
165         final EdgeCrossing b = (EdgeCrossing)o2;
166   
167         return (a.getAngle() < b.getAngle()) ? -1 : (a.getAngle() > b.getAngle()) ? 1 : 0;
168       }
