9   /**
10    * <p>For each line, the scanbuffer (= a raster divice) records the points hit
11    * I.e., line 5 (y=5) contains the values 2, 6, 40, and 46 (these line have
12    * been crossed). The values always come as pairs because we're dealing with
13    * polygons, which have a left and a right side which consists of a line.
14    * The points in between two values painted as filled.
15    *
16    * <p>&copy; 2002 Christian Treber, ct@ctreber.com
17    * @author Christian Treber, ct@ctreber.com
18    *
19    */
20   class ScanBuf
21   {
22     private List[] fScanbuf;
23     private int fLineMin;
24     private int fLineMax;
25     private final int fLines;
26     private final int fPoints;
27     private boolean fScanBufsAdded;
28   
29     /**
30      * <p>Create a list for each line.
31      *
32      * @param pLines Number of lines aka screen height.
33      * @param pPoints Number of points per line aka screen width.
34      */
35     public ScanBuf(int pLines, int pPoints)
36     {
37       fLines = pLines;
38       fPoints = pPoints;
39   
40       fLineMin = Integer.MAX_VALUE;
41       fLineMax = Integer.MIN_VALUE;
42       fScanBufsAdded = false;
43   
44       fScanbuf = new ArrayList[fLines];
45       for(int i = 0; i < fScanbuf.length; i++)
46       {
47         fScanbuf[i] = new ArrayList();
48       }
49     }
50   
51     /**
52      * <p>Add a line to the generateScanBits buffer.
53      */
54     public void addLine(double pXFrom, double pYFrom, double pXTo, double pYTo)
55     {
56       int lYFrom;
57       int lYTo;
58   
59       // Do some rounding (but not in the way we expect it), limit values
60       if(pYFrom < pYTo)
61       {
62         // Round lYFrom (but .5 is handled oddly)
63         // 1.5001 - 2.5 -> 1.0001 - 2.0 -> 2
64         lYFrom = (int)Math.ceil(pYFrom - 0.5);
65         // Round lYTo, substract 1
66         // 1.5 - 2.4999 -> 1.0 - 1.9999 -> 1
67         lYTo = (int)Math.floor(pYTo - 0.5);
68   
69         /**
70          * I don't know if this is intended, but in Java 3 == 3.001 is false
71          * (the left arg is converted to double), so the expr is true only when
72          * pYTo - 0.5 is exactly lYTo
73          */
74         if(lYTo == pYTo - 0.5)
75         {
76           lYTo--;
77         }
78       } else
79       {
80         lYFrom = (int)Math.ceil(pYTo - 0.5);
81         lYTo = (int)Math.floor(pYFrom - 0.5);
82   
83         if(lYTo == pYFrom - 0.5)
84         {
85           lYTo--;
86         }
87       }
88   
89       // Limit y to size of image
90       if(lYFrom < 0)
91       {
92         lYFrom = 0;
93       }
94       if(lYTo >= fLines)
95       {
96         lYTo = fLines - 1;
97       }
98   
99       if(lYFrom > lYTo)
100       {
101         // No lines crossed.
102         return;
103       }
104   
105       // Note min/max settings so far
106       if(lYFrom < fLineMin)
107       {
108         fLineMin = lYFrom;
109       }
110       if(lYTo > fLineMax)
111       {
112         fLineMax = lYTo;
113       }
114   
115       // todo Curious: What happens if yFrom and yTo are equal? Shit? Or can't they be?
116       double lDx = (pXTo - pXFrom) / (pYTo - pYFrom);
117       double lX = pXFrom + lDx * ((lYFrom + 0.5) - pYFrom);
118   
119       // Record the x value for every line (y).
120       for(int lLineNo = lYFrom; lLineNo <= lYTo; lLineNo++)
121       {
122         fScanbuf[lLineNo].add(Double.valueOf(lX));
123         lX += lDx;
124       }
125       fScanBufsAdded = true;
126     }
127   
128     public boolean containsPoints()
129     {
130       return fScanBufsAdded;
131     }
132   
133     /**
134      * <p>For each line, for each x value pair in line, create one ScanBit.
135      */
136     public List getScanbits(int pCurveType)
137     {
138       final List fScanBits = new ArrayList();
139   
140       // For each generateScanBits line containing points
141       for(int lLineNo = fLineMin; lLineNo <= fLineMax; lLineNo++)
142       {
143         // Sort so that lowest x values come first.
144         Double[] lScanLine = (Double[])fScanbuf[lLineNo].toArray(new Double[0]);
145         CTSort lSort = new QuickSort();
146         lSort.sort(lScanLine);
147   
148         // The length will be divisible by 2 because we render closed polyons,
149         // so every generateScanBits line is crossed twice (left and right edge of polygon,
150         // no intersections allowed!).
151         for(int n = 0; n < lScanLine.length; n += 2)
152         {
153           // Round lLineFrom (but .5 is handled oddly)
154           // 1.5001 - 2.5 -> 1.0001 - 2.0 -> 2
155           int lXLo = (int)Math.ceil(lScanLine[n] - 0.5);
156           // Round lLineTo, substract 1
157           // 1.5 - 2.4999 -> 1.0 - 1.9999 -> 1
158           int lXHi = (int)Math.floor(lScanLine[n + 1] - 0.5);
159   
160           // Limit low and high x to image dimensions
161           if(lXLo < 0)
162           {
163             lXLo = 0;
164           }
165           if(lXHi >= fPoints)
166           {
167             lXHi = fPoints - 1;
168           }
169   
170           if(lXLo <= lXHi)
171           {
172             /**
173              * Shouldn't that always be true since we sorted? "Yes", BUT the
174              * rounding might create lo 3.6 -> 4.0 and hi 3.7 -> 3.0
175              */
176             fScanBits.add(new ScanBit(lLineNo, lXLo, lXHi, pCurveType));
177           }
178         }
179       }
180   
181       return fScanBits;
182     }
