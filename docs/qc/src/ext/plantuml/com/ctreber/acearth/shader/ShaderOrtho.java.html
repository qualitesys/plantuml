12   public class ShaderOrtho extends Shader
13   {
14     private static double[] fXValues;
15   
16     public Color[] getShadedColors(int pRowNo, int[] pRowTypes)
17     {
18       if(pRowNo == 0)
19       {
20         fXValues = computeXValues(pRowTypes.length);
21       }
22   
23       final double y = fProjection.inverseProjectY(fProjection.inverseFinalizeY(pRowNo));
24       final double tmp = 1 - (y * y);
25       final double lYBySunVectorY = y * fLightVector.getY();
26   
27       final Color[] lColors = new Color[pRowTypes.length];
28       for(int lColNo = 0; lColNo < pRowTypes.length; lColNo++)
29       {
30         double x = fXValues[lColNo];
31         double z = Math.sqrt(tmp - (x * x));
32   
33         double lSunValue = (x * fLightVector.getX()) + lYBySunVectorY + (z * fLightVector.getZ());
34         lColors[lColNo] = getShadedColorForType(pRowTypes[lColNo], lSunValue);
35       }
36   
37       return lColors;
38     }
39   
40     /**
41      *
42      * @return X value for each column in image.
43      */
44     private double[] computeXValues(int pWidth)
45     {
46       final double[] lTable = new double[pWidth];
47   
48       for(int lColNo = 0; lColNo < pWidth; lColNo++)
49       {
50         lTable[lColNo] = fProjection.inverseProjectX(fProjection.inverseFinalizeX(lColNo));
51       }
52   
53       return lTable;
54     }
