10   /**
11    * <p>A shader computes Colors for a row of pixel types, depending
12    * on lighting parameters.</p>
13    *
14    * <p>&copy; 2002 Christian Treber, ct@ctreber.com (Nov 11, 2002)</p>
15    * @author Christian Treber, ct@ctreber.com
16    *
17    */
18   abstract public class Shader
19   {
20     private static final Color COLOR_SPACE = Color.black;
21     private static final Color COLOR_STAR = Color.white;
22     private static final Color COLOR_WATER = Color.blue;
23     private static final Color COLOR_LAND = Color.green;
24     // Brown
25     //static final Color COLOR_LAND = new Color(255, 136, 25);
26     private static final Color COLOR_GRID_LAND = Color.white;
27     // Bright blue
28     private static final Color COLOR_GRID_WATER = new Color(128, 128, 255);
29   
30     /** <p>Needed to calculate lighting vectors. */
31     Projection fProjection;
32   
33     // Stuff below only needed when shading.
34     private Coordinate fSunPos;
35     private double fNightSideBrightness;
36     private double fDaySideBrightness;
37     private double fTerminatorDiscontinuity;
38   
39     private double fDaySideValueBase;
40     private double fDaySideValueRange;
41     Point3D fLightVector;
42   
43     abstract public Color[] getShadedColors(int pRowNo, int[] pRowTypes);
44   
45     public void init()
46     {
47       // Precompute shading parameters. I personally find the terminator
48       // stuff is obscure and might as well be left out.
49       final double tmp = fTerminatorDiscontinuity / 100;
50       // 100%: day, 0%: night
51       fDaySideValueBase = (int)(tmp * fDaySideBrightness +
52         (1 - tmp) * fNightSideBrightness);
53       fDaySideValueRange = fDaySideBrightness - fDaySideValueBase;
54       fLightVector = fProjection.rotate(fSunPos.getPoint3D());
55     }
56   
57     Color getShadedColorForType(int pType, double pSunValue)
58     {
59       double lBrightness;
60   
61       if(pSunValue < 0)
62       {
63         // The sun is below the horizon.
64         lBrightness = fNightSideBrightness / 100;
65       } else
66       {
67         // The sun is above the horizon. The brightness will range from
68         // the base to the maximum value.
69         lBrightness = (fDaySideValueBase + pSunValue * fDaySideValueRange) / 100;
70       }
71       if(lBrightness > 1.0)
72       {
73         lBrightness = 1.0;
74       }
75   
76       switch(pType)
77       {
78         case BitGeneratorMap.PixTypeSpace:
79           return COLOR_SPACE;
80   
81         case BitGeneratorMap.PixTypeStar:
82           return COLOR_STAR;
83   
84         case BitGeneratorMap.PixTypeGridLand:
85           return shade(COLOR_GRID_LAND, lBrightness);
86   
87         case BitGeneratorMap.PixTypeGridWater:
88           return shade(COLOR_GRID_WATER, lBrightness);
89   
90         case BitGeneratorMap.PixTypeLand:
91           return shade(COLOR_LAND, lBrightness);
92   
93         case BitGeneratorMap.PixTypeWater:
94           return shade(COLOR_WATER, lBrightness);
95       }
96   
97       return null;
98     }
99   
100     private static Color shade(Color pColor, double pBrightness)
101     {
102       return new Color((int)(pColor.getRed() * pBrightness),
103         (int)(pColor.getGreen() * pBrightness),
104         (int)(pColor.getBlue() * pBrightness));
105     }
106   
107     public void setProjection(Projection pProjection)
108     {
109       fProjection = pProjection;
110     }
111   
112     public void setSunPos(Coordinate pSunPos)
113     {
114       fSunPos = pSunPos;
115     }
116   
117     public void setDaySideBrightness(double pDaySideBrightness)
118     {
119       fDaySideBrightness = pDaySideBrightness;
120     }
121   
122     public void setNightSideBrightness(double pNightSideBrightness)
123     {
124       fNightSideBrightness = pNightSideBrightness;
125     }
126   
127     public void setTerminatorDiscontinuity(double pTerminatorDiscontinuity)
128     {
129       fTerminatorDiscontinuity = pTerminatorDiscontinuity;
130     }
131   }
