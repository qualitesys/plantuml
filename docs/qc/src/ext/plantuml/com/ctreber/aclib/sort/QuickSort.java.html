12     {
13       if(items.length <= 1)
14       {
15         // Nothing to sort t all or only one element.
16         return;
17       }
18   
19       qsort(items, comparator, 0, items.length - 1);

23     private void qsort(Object[] items, Comparator comparator, int l, int r)
24     {
25       final int M = 4;
26       int i;
27       int j;
28       Object v;
29   
30       if((r - l) > M)
31       {
32         i = (r + l) / 2;
33         if(comparator.compare(items[l], items[i]) > 0)
34         {
35           swap(items, l, i);
36         }
37         if(comparator.compare(items[l], items[r]) > 0)
38         {
39           swap(items, l, r);
40         }
41         if(comparator.compare(items[i], items[r]) > 0)
42         {
43           swap(items, i, r);
44         }
45   
46         j = r - 1;
47         swap(items, i, j);
48         i = l;
49         v = items[j];
50         while(true)
51         {
52           while(comparator.compare(items[++i], v) < 0)
53           {
54           }
55           while(comparator.compare(items[--j], v) > 0)
56           {
57           }
58           if(j < i)
59           {
60             break;
61           }
62           swap(items, i, j);
63         }
64         swap(items, i, r - 1);
65         qsort(items, comparator, l, j);
66         qsort(items, comparator, i + 1, r);
67       }
68     }

70     private static void swap(Object[] items, int i, int j)
71     {
72       final Object tmp;
73       tmp = items[i];
74       items[i] = items[j];
75       items[j] = tmp;
76     }

78     private static void insertionSort(Object[] items, Comparator comparator, int lo0, int hi0)
79     {
80       int i;
81       int j;
82       Object v;
83   
84       for(i = lo0 + 1; i <= hi0; i++)
85       {
86         v = items[i];
87         j = i;
88         while((j > lo0) && (comparator.compare(items[j - 1], v) > 0))
89         {
90           items[j] = items[j - 1];
91           j--;
92         }
93         items[j] = v;
94       }
95     }
