160       /**
161        * Adds next GIF frame. The frame is not written immediately, but is actually
162        * deferred until the next frame is received so that timing data can be
163        * inserted. Invoking <code>finish()</code> flushes all frames. If
164        * <code>setSize</code> was not invoked, the size of the first image is used for
165        * all subsequent frames.
166        * 
167        * @param im BufferedImage containing frame to write.
168        * @return true if successful.
169        */
170       public boolean addFrame(BufferedImage im) {
171           if ((im == null) || !started) {
172               return false;
173           }
174           boolean ok = true;
175           try {
176               if (!sizeSet) {
177                   // use first frame's size
178                   setSize(im.getWidth(), im.getHeight());
179               }
180               image = im;
181               getImagePixels(); // convert to correct format if necessary
182               analyzePixels(); // build color table & map pixels
183               if (firstFrame) {
184                   writeLSD(); // logical screen descriptior
185                   writePalette(); // global color table
186                   if (repeat >= 0) {
187                       // use NS app extension to indicate reps
188                       writeNetscapeExt();
189                   }
190               }
191               writeGraphicCtrlExt(); // write graphic control extension
192               writeImageDesc(); // image descriptor
193               if (!firstFrame) {
194                   writePalette(); // local color table
195               }
196               writePixels(); // encode and write pixel data
197               firstFrame = false;
198           } catch (IOException e) {
199               ok = false;
200           }
201   
202           return ok;
203       }

205       /**
206        * Flushes any pending data and closes output file. If writing to an
207        * OutputStream, the stream is not closed.
208        */
209       public boolean finish() {
210           if (!started)
211               return false;
212           boolean ok = true;
213           started = false;
214           try {
215               out.write(0x3b); // gif trailer
216               out.flush();
217               if (closeStream) {
218                   out.close();
219               }
220           } catch (IOException e) {
221               ok = false;
222           }
223   
224           // reset for subsequent use
225           transIndex = 0;
226           out = null;
227           image = null;
228           pixels = null;
229           indexedPixels = null;
230           colorTab = null;
231           closeStream = false;
232           firstFrame = true;
233   
234           return ok;
235       }

322       /**
323        * Analyzes image colors and creates color map.
324        */
325       protected void analyzePixels() {
326           int len = pixels.length;
327           int nPix = len / 3;
328           indexedPixels = new byte[nPix];
329           NeuQuant nq = new NeuQuant(pixels, len, sample);
330           // initialize quantizer
331           colorTab = nq.process(); // create reduced palette
332           // convert map from BGR to RGB
333           for (int i = 0; i < colorTab.length; i += 3) {
334               byte temp = colorTab[i];
335               colorTab[i] = colorTab[i + 2];
336               colorTab[i + 2] = temp;
337               usedEntry[i / 3] = false;
338           }
339           // map image pixels to new palette
340           int k = 0;
341           for (int i = 0; i < nPix; i++) {
342               int index = nq.map(pixels[k++] & 0xff, pixels[k++] & 0xff, pixels[k++] & 0xff);
343               usedEntry[index] = true;
344               indexedPixels[i] = (byte) index;
345           }
346           pixels = null;
347           colorDepth = 8;
348           palSize = 7;
349           // get closest match to transparent color if specified
350           if (transparent != null) {
351               transIndex = findClosest(transparent);
352           }
353       }
354   
355       /**
356        * Returns index of palette color closest to c
357        * 
358        */
359       protected int findClosest(Color c) {
360           if (colorTab == null)
361               return -1;
362           int r = c.getRed();
363           int g = c.getGreen();
364           int b = c.getBlue();
365           int minpos = 0;
366           int dmin = 256 * 256 * 256;
367           int len = colorTab.length;
368           for (int i = 0; i < len;) {
369               int dr = r - (colorTab[i++] & 0xff);
370               int dg = g - (colorTab[i++] & 0xff);
371               int db = b - (colorTab[i] & 0xff);
372               int d = dr * dr + dg * dg + db * db;
373               int index = i / 3;
374               if (usedEntry[index] && (d < dmin)) {
375                   dmin = d;
376                   minpos = index;
377               }
378               i++;
379           }
380           return minpos;
381       }

383       /**
384        * Extracts image pixels into byte array "pixels"
385        */
386       protected void getImagePixels() {
387           int w = image.getWidth();
388           int h = image.getHeight();
389           int type = image.getType();
390           if ((w != width) || (h != height) || (type != BufferedImage.TYPE_3BYTE_BGR)) {
391               // create new image with right size/format
392               BufferedImage temp = new BufferedImage(width, height, BufferedImage.TYPE_3BYTE_BGR);
393               Graphics2D g = temp.createGraphics();
394               g.drawImage(image, 0, 0, null);
395               image = temp;
396           }
397           pixels = ((DataBufferByte) image.getRaster().getDataBuffer()).getData();
398       }
399   
400       /**
401        * Writes Graphic Control Extension
402        */
403       protected void writeGraphicCtrlExt() throws IOException {
404           out.write(0x21); // extension introducer
405           out.write(0xf9); // GCE label
406           out.write(4); // data block size
407           int transp, disp;
408           if (transparent == null) {
409               transp = 0;
410               disp = 0; // dispose = no action
411           } else {
412               transp = 1;
413               disp = 2; // force clear if using transparent color
414           }
415           if (dispose >= 0) {
416               disp = dispose & 7; // user override
417           }
418           disp <<= 2;
419   
420           // packed fields
421           out.write(0 | // 1:3 reserved
422                   disp | // 4:6 disposal
423                   0 | // 7 user input - 0 = none
424                   transp); // 8 transparency flag
425   
426           writeShort(delay); // delay x 1/100 sec
427           out.write(transIndex); // transparent color index
428           out.write(0); // block terminator
429       }
430   
431       /**
432        * Writes Image Descriptor
433        */
434       protected void writeImageDesc() throws IOException {
435           out.write(0x2c); // image separator
436           writeShort(0); // image position x,y = 0,0
437           writeShort(0);
438           writeShort(width); // image size
439           writeShort(height);
440           // packed fields
441           if (firstFrame) {
442               // no LCT - GCT is used for first (or only) frame
443               out.write(0);
444           } else {
445               // specify normal LCT
446               out.write(0x80 | // 1 local color table 1=yes
447                       0 | // 2 interlace - 0=no
448                       0 | // 3 sorted - 0=no
449                       0 | // 4-5 reserved
450                       palSize); // 6-8 size of color table
451           }
452       }
453   
454       /**
455        * Writes Logical Screen Descriptor
456        */
457       protected void writeLSD() throws IOException {
458           // logical screen size
459           writeShort(width);
460           writeShort(height);
461           // packed fields
462           out.write((0x80 | // 1 : global color table flag = 1 (gct used)
463                   0x70 | // 2-4 : color resolution = 7
464                   0x00 | // 5 : gct sort flag = 0
465                   palSize)); // 6-8 : gct size
466   
467           out.write(0); // background color index
468           out.write(0); // pixel aspect ratio - assume 1:1
469       }
470   
471       /**
472        * Writes Netscape application extension to define repeat count.
473        */
474       protected void writeNetscapeExt() throws IOException {
475           out.write(0x21); // extension introducer
476           out.write(0xff); // app extension label
477           out.write(11); // block size
478           writeString("NETSCAPE" + "2.0"); // app id + auth code
479           out.write(3); // sub-block size
480           out.write(1); // loop sub-block id
481           writeShort(repeat); // loop count (extra iterations, 0=repeat forever)
482           out.write(0); // block terminator
483       }
484   
485       /**
486        * Writes color table
487        */
488       protected void writePalette() throws IOException {
489           out.write(colorTab, 0, colorTab.length);
490           int n = (3 * 256) - colorTab.length;
491           for (int i = 0; i < n; i++) {
492               out.write(0);
493           }
494       }

498        */
499       protected void writePixels() throws IOException {
500           LZWEncoder encoder = new LZWEncoder(width, height, indexedPixels, colorDepth);
501           encoder.encode(out);
502       }

512       /**
513        * Writes string to output stream

547   
548       /* four primes near 500 - assume no image has a length so large */

552       protected static final int prime2 = 491;

554       protected static final int prime3 = 487;
555   
556       protected static final int prime4 = 503;

561

563        * Program Skeleton ---------------- [select samplefac in range 1..30] [read

572        */

574       protected static final int maxnetpos = (netsize - 1);
575   
576       protected static final int netbiasshift = 4; /* bias for colour values */
577   
578       protected static final int ncycles = 100; /* no. of learning cycles */
579   
580       /* defs for freq and bias */
581       protected static final int intbiasshift = 16; /* bias for fractions */
582   
583       protected static final int intbias = (((int) 1) << intbiasshift);
584   
585       protected static final int gammashift = 10; /* gamma = 1024 */
586   
587       protected static final int gamma = (((int) 1) << gammashift);
588   
589       protected static final int betashift = 10;
590   
591       protected static final int beta = (intbias >> betashift); /* beta = 1/1024 */
592   
593       protected static final int betagamma = (intbias << (gammashift - betashift));
594   
595       /* defs for decreasing radius factor */
596       protected static final int initrad = (netsize >> 3); /*
597                                                                * for 256 cols, radius starts
598                                                                */
599   
600       protected static final int radiusbiasshift = 6; /* at 32.0 biased by 6 bits */
601   
602       protected static final int radiusbias = (((int) 1) << radiusbiasshift);
603   
604       protected static final int initradius = (initrad * radiusbias); /*
605                                                                        * and decreases by a
606                                                                        */
607   
608       protected static final int radiusdec = 30; /* factor of 1/30 each cycle */
609   
610       /* defs for decreasing alpha factor */
611       protected static final int alphabiasshift = 10; /* alpha starts at 1.0 */
612   
613       protected static final int initalpha = (((int) 1) << alphabiasshift);
614   
615       protected int alphadec; /* biased by 10 bits */
616   
617       /* radbias and alpharadbias used for radpower calculation */
618       protected static final int radbiasshift = 8;
619   
620       protected static final int radbias = (((int) 1) << radbiasshift);
621   
622       protected static final int alpharadbshift = (alphabiasshift + radbiasshift);
623   
624       protected static final int alpharadbias = (((int) 1) << alpharadbshift);
625

627        * Types and Global Variables --------------------------
628        */
629   
630       protected byte[] thepicture; /* the input image itself */
631   
632       protected int lengthcount; /* lengthcount = H*W*3 */
633

635

638

642

644

646       protected int[] freq = new int[netsize];
647

651

653        * Initialise network in range (0,0,0) to (255,255,255) and set parameters
654        * -----------------------------------------------------------------------
655        */
656       public NeuQuant(byte[] thepic, int len, int sample) {

659           int[] p;

663           samplefac = sample;

679               index[network[i][3]] = i;
680           int k = 0;
681           for (int i = 0; i < netsize; i++) {
682               int j = index[i];
683               map[k++] = (byte) (network[j][0]);

690       /*
691        * Insertion sort of network and building of netindex[0..255] (to do after
692        * unbias)
693        * -----------------------------------------------------------------------------
694        * --
695        */
696       public void inxbuild() {
697   
698           int i, j, smallpos, smallval;
699           int[] p;
700           int[] q;
701           int previouscol, startpos;
702   
703           previouscol = 0;
704           startpos = 0;
705           for (i = 0; i < netsize; i++) {
706               p = network[i];
707               smallpos = i;
708               smallval = p[1]; /* index on g */
709               /* find smallest in i..netsize-1 */
710               for (j = i + 1; j < netsize; j++) {
711                   q = network[j];
712                   if (q[1] < smallval) { /* index on g */
713                       smallpos = j;
714                       smallval = q[1]; /* index on g */
715                   }
716               }
717               q = network[smallpos];
718               /* swap p (i) and q (smallpos) entries */
719               if (i != smallpos) {
720                   j = q[0];
721                   q[0] = p[0];
722                   p[0] = j;
723                   j = q[1];
724                   q[1] = p[1];
725                   p[1] = j;
726                   j = q[2];
727                   q[2] = p[2];
728                   p[2] = j;
729                   j = q[3];
730                   q[3] = p[3];
731                   p[3] = j;
732               }
733               /* smallval entry is now in position i */
734               if (smallval != previouscol) {
735                   netindex[previouscol] = (startpos + i) >> 1;
736                   for (j = previouscol + 1; j < smallval; j++)
737                       netindex[j] = i;
738                   previouscol = smallval;
739                   startpos = i;
740               }
741           }
742           netindex[previouscol] = (startpos + maxnetpos) >> 1;
743           for (j = previouscol + 1; j < 256; j++)
744               netindex[j] = maxnetpos; /* really 256 */
745       }
746   
747       /*
748        * Main Learning Loop ------------------
749        */
750       public void learn() {
751   
752           int i, j, b, g, r;
753           int radius, rad, alpha, step, delta, samplepixels;
754           byte[] p;
755           int pix, lim;
756   
757           if (lengthcount < minpicturebytes)
758               samplefac = 1;
759           alphadec = 30 + ((samplefac - 1) / 3);
760           p = thepicture;
761           pix = 0;
762           lim = lengthcount;
763           samplepixels = lengthcount / (3 * samplefac);
764           delta = samplepixels / ncycles;
765           alpha = initalpha;
766           radius = initradius;
767   
768           rad = radius >> radiusbiasshift;
769           if (rad <= 1)
770               rad = 0;
771           for (i = 0; i < rad; i++)
772               radpower[i] = alpha * (((rad * rad - i * i) * radbias) / (rad * rad));
773   
774           // fprintf(stderr,"beginning 1D learning: initial radius=%d\n", rad);
775   
776           if (lengthcount < minpicturebytes)
777               step = 3;
778           else if ((lengthcount % prime1) != 0)
779               step = 3 * prime1;
780           else {
781               if ((lengthcount % prime2) != 0)
782                   step = 3 * prime2;
783               else {
784                   if ((lengthcount % prime3) != 0)
785                       step = 3 * prime3;
786                   else
787                       step = 3 * prime4;
788               }
789           }
790   
791           i = 0;
792           while (i < samplepixels) {
793               b = (p[pix + 0] & 0xff) << netbiasshift;
794               g = (p[pix + 1] & 0xff) << netbiasshift;
795               r = (p[pix + 2] & 0xff) << netbiasshift;
796               j = contest(b, g, r);
797   
798               altersingle(alpha, j, b, g, r);
799               if (rad != 0)
800                   alterneigh(rad, j, b, g, r); /* alter neighbours */
801   
802               pix += step;
803               if (pix >= lim)
804                   pix -= lengthcount;
805   
806               i++;
807               if (delta == 0)
808                   delta = 1;
809               if (i % delta == 0) {
810                   alpha -= alpha / alphadec;
811                   radius -= radius / radiusdec;
812                   rad = radius >> radiusbiasshift;
813                   if (rad <= 1)
814                       rad = 0;
815                   for (j = 0; j < rad; j++)
816                       radpower[j] = alpha * (((rad * rad - j * j) * radbias) / (rad * rad));
817               }
818           }
819           // fprintf(stderr,"finished 1D learning: final alpha=%f
820           // !\n",((float)alpha)/initalpha);
821       }
822   
823       /*
824        * Search for BGR values 0..255 (after net is unbiased) and return colour index
825        * ----------------------------------------------------------------------------
826        */
827       public int map(int b, int g, int r) {
828   
829           int i, j, dist, a, bestd;
830           int[] p;
831           int best;
832   
833           bestd = 1000; /* biggest possible dist is 256*3 */
834           best = -1;
835           i = netindex[g]; /* index on g */
836           j = i - 1; /* start at netindex[g] and work outwards */
837   
838           while ((i < netsize) || (j >= 0)) {
839               if (i < netsize) {
840                   p = network[i];
841                   dist = p[1] - g; /* inx key */
842                   if (dist >= bestd)
843                       i = netsize; /* stop iter */
844                   else {
845                       i++;
846                       if (dist < 0)
847                           dist = -dist;
848                       a = p[0] - b;
849                       if (a < 0)
850                           a = -a;
851                       dist += a;
852                       if (dist < bestd) {
853                           a = p[2] - r;
854                           if (a < 0)
855                               a = -a;
856                           dist += a;
857                           if (dist < bestd) {
858                               bestd = dist;
859                               best = p[3];
860                           }
861                       }
862                   }
863               }
864               if (j >= 0) {
865                   p = network[j];
866                   dist = g - p[1]; /* inx key - reverse dif */
867                   if (dist >= bestd)
868                       j = -1; /* stop iter */
869                   else {
870                       j--;
871                       if (dist < 0)
872                           dist = -dist;
873                       a = p[0] - b;
874                       if (a < 0)
875                           a = -a;
876                       dist += a;
877                       if (dist < bestd) {
878                           a = p[2] - r;
879                           if (a < 0)
880                               a = -a;
881                           dist += a;
882                           if (dist < bestd) {
883                               bestd = dist;
884                               best = p[3];
885                           }
886                       }
887                   }
888               }
889           }
890           return (best);
891       }

893       public byte[] process() {
894           learn();
895           unbiasnet();
896           inxbuild();
897           return colorMap();
898       }
899   
900       /*
901        * Unbias network to give byte values 0..255 and record position i to prepare
902        * for sort
903        * -----------------------------------------------------------------------------
904        * ------
905        */
906       public void unbiasnet() {
907   
908           int i, j;
909   
910           for (i = 0; i < netsize; i++) {
911               network[i][0] >>= netbiasshift;
912               network[i][1] >>= netbiasshift;
913               network[i][2] >>= netbiasshift;
914               network[i][3] = i; /* record colour no */
915           }
916       }
917   
918       /*
919        * Move adjacent neurons by precomputed alpha*(1-((i-j)^2/[r]^2)) in
920        * radpower[|i-j|]
921        * -----------------------------------------------------------------------------
922        * ----
923        */
924       protected void alterneigh(int rad, int i, int b, int g, int r) {
925   
926           int j, k, lo, hi, a, m;
927           int[] p;
928   
929           lo = i - rad;
930           if (lo < -1)
931               lo = -1;
932           hi = i + rad;
933           if (hi > netsize)
934               hi = netsize;
935   
936           j = i + 1;
937           k = i - 1;
938           m = 1;
939           while ((j < hi) || (k > lo)) {
940               a = radpower[m++];
941               if (j < hi) {
942                   p = network[j++];
943                   try {
944                       p[0] -= (a * (p[0] - b)) / alpharadbias;
945                       p[1] -= (a * (p[1] - g)) / alpharadbias;
946                       p[2] -= (a * (p[2] - r)) / alpharadbias;
947                   } catch (Exception e) {
948                   } // prevents 1.3 miscompilation
949               }
950               if (k > lo) {
951                   p = network[k--];
952                   try {
953                       p[0] -= (a * (p[0] - b)) / alpharadbias;
954                       p[1] -= (a * (p[1] - g)) / alpharadbias;
955                       p[2] -= (a * (p[2] - r)) / alpharadbias;
956                   } catch (Exception e) {
957                   }
958               }
959           }
960       }
961   
962       /*
963        * Move neuron i towards biased (b,g,r) by factor alpha
964        * ----------------------------------------------------
965        */
966       protected void altersingle(int alpha, int i, int b, int g, int r) {
967   
968           /* alter hit neuron */
969           int[] n = network[i];
970           n[0] -= (alpha * (n[0] - b)) / initalpha;
971           n[1] -= (alpha * (n[1] - g)) / initalpha;
972           n[2] -= (alpha * (n[2] - r)) / initalpha;
973       }
974   
975       /*
976        * Search for biased BGR values ----------------------------
977        */
978       protected int contest(int b, int g, int r) {
979   
980           /* finds closest neuron (min dist) and updates freq */
981           /* finds best neuron (min dist-bias) and returns position */
982           /*
983            * for frequently chosen neurons, freq[i] is high and bias[i] is negative
984            */
985           /* bias[i] = gamma*((1/netsize)-freq[i]) */
986   
987           int i, dist, a, biasdist, betafreq;
988           int bestpos, bestbiaspos, bestd, bestbiasd;
989           int[] n;
990   
991           bestd = ~(((int) 1) << 31);
992           bestbiasd = bestd;
993           bestpos = -1;
994           bestbiaspos = bestpos;
995   
996           for (i = 0; i < netsize; i++) {
997               n = network[i];
998               dist = n[0] - b;
999               if (dist < 0)

1002               if (a < 0)
1003                   a = -a;
1004               dist += a;
1005               a = n[2] - r;
1006               if (a < 0)
1007                   a = -a;
1008               dist += a;
1009               if (dist < bestd) {
1010                   bestd = dist;
1011                   bestpos = i;
1012               }
1013               biasdist = dist - ((bias[i]) >> (intbiasshift - netbiasshift));
1014               if (biasdist < bestbiasd) {
1015                   bestbiasd = biasdist;
1016                   bestbiaspos = i;
1017               }
1018               betafreq = (freq[i] >> betashift);
1019               freq[i] -= betafreq;
1020               bias[i] += (betafreq << gammashift);
1021           }
1022           freq[bestpos] += beta;
1023           bias[bestpos] -= betagamma;
1024           return (bestbiaspos);
1025       }
1026   }
