72   public class SmetanaForGit {
73   
74       private final UGraphic ug;
75       private final ISkinParam skinParam;
76       private int num;
77       private ST_Agraph_s g;
78       private StringBounder stringBounder;
79   
80       private final Map<GNode, ST_Agnode_s> nodes = new LinkedHashMap<GNode, ST_Agnode_s>();
81       private final List<ST_Agedge_s> edges = new ArrayList<>();
82       private Mirror yMirror;
83   
84       public SmetanaForGit(UGraphic ug, ISkinParam skinParam) {
85           this.stringBounder = ug.getStringBounder();
86           this.skinParam = skinParam;
87           this.ug = getStyle().applyStrokeAndLineColor(ug, skinParam.getIHtmlColorSet());
88       }
89   
90       private Style getStyle() {
91           return StyleSignatureBasic.of(SName.root, SName.element, SName.gitDiagram)
92                   .getMergedStyle(skinParam.getCurrentStyleBuilder());
93       }
94   
95       private HColor arrowColor() {
96           return getStyle().value(PName.LineColor).asColor(skinParam.getIHtmlColorSet());
97       }
98   
99       public void drawMe(Collection<GNode> gnodes) {
100           initGraph(gnodes);
101   
102           for (Entry<GNode, ST_Agnode_s> ent : nodes.entrySet()) {
103               final ST_Agnode_s node = ent.getValue();
104               final UTranslate pos = getPosition(node);
105   
106               final MagicBox magicBox = new MagicBox(skinParam, ent.getKey());
107               magicBox.drawBorder(ug.apply(pos), getSize(node));
108           }
109   
110           for (ST_Agedge_s edge : edges) {
111               final ST_Agedgeinfo_t data = (ST_Agedgeinfo_t) Macro.AGDATA(edge);
112               new GitCurve(data, yMirror).drawCurve(arrowColor(), ug);
113           }
114       }
115   
116       private XDimension2D getSize(ST_Agnode_s node) {
117           final ST_Agnodeinfo_t data = (ST_Agnodeinfo_t) Macro.AGDATA(node);
118           final double width = data.width * 72;
119           final double height = data.height * 72;
120           return new XDimension2D(width, height);
121       }
122   
123       private UTranslate getPosition(ST_Agnode_s node) {
124           final ST_Agnodeinfo_t data = (ST_Agnodeinfo_t) Macro.AGDATA(node);
125           final double width = data.width * 72;
126           final double height = data.height * 72;
127           final double x = data.coord.x;
128           final double y = data.coord.y;
129           return new UTranslate(x - width / 2, yMirror.inv(y + height / 2));
130       }
131   
132       private void initGraph(Collection<GNode> gnodes) {
133           if (g != null)
134               return;
135   
136           Z.open();
137           try {
138               g = agopen(new CString("g"), Z.z().Agdirected, null);
139               agsafeset(g, new CString("ranksep"), new CString("0.35"), new CString(""));
140               for (GNode gnode : gnodes) {
141                   final XDimension2D dim = new MagicBox(skinParam, gnode).getBigDim(stringBounder);
142                   nodes.put(gnode, createNode(dim));
143               }
144   
145               for (GNode gnode : nodes.keySet()) {
146                   for (GNode parent : gnode.getDowns()) {
147                       final ST_Agedge_s edge = createEdge(gnode, parent);
148                       if (edge != null)
149                           edges.add(edge);
150   
151                   }
152               }
153   
154               final ST_GVC_s gvc = gvContext();
155               gvLayoutJobs(gvc, g);
156   
157               double max = 0;
158               for (ST_Agnode_s node : nodes.values()) {
159                   final ST_Agnodeinfo_t data = (ST_Agnodeinfo_t) Macro.AGDATA(node);
160                   final double height = data.height * 72;
161                   final double y = data.coord.y;
162                   max = Math.max(max, y + height);
163               }
164               yMirror = new Mirror(max);
165   
166           } finally {
167               Z.close();
168           }
169   
170       }
171   
172       private ST_Agedge_s createEdge(GNode node0, GNode node1) {
173           final ST_Agnode_s a0 = nodes.get(node0);
174           final ST_Agnode_s a1 = nodes.get(node1);
175           if (a0 == null || a1 == null) {
176               return null;
177           }
178   
179           final ST_Agedge_s edge = agedge(g, a0, a1, null, true);
180           agsafeset(edge, new CString("arrowsize"), new CString(".7"), new CString(""));
181           agsafeset(edge, new CString("arrowtail"), new CString("none"), new CString(""));
182           agsafeset(edge, new CString("arrowhead"), new CString("normal"), new CString(""));
183           return edge;
184       }
185   
186       private ST_Agnode_s createNode(XDimension2D dim) {
187           final String width = "" + (dim.getWidth() / 72);
188           final String height = "" + (dim.getHeight() / 72);
189   
190           final ST_Agnode_s node = agnode(g, new CString("N" + num), true);
191           node.NAME = "N " + num;
192           num++;
193   
194           agsafeset(node, new CString("shape"), new CString("box"), new CString(""));
195           agsafeset(node, new CString("height"), new CString("" + height), new CString(""));
196           agsafeset(node, new CString("width"), new CString("" + width), new CString(""));
197   
198           return node;
199       }
