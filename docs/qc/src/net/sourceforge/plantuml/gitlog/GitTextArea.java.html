36   package net.sourceforge.plantuml.gitlog;
37   
38   import java.util.ArrayList;
39   import java.util.Collections;
40   import java.util.List;
41   
42   public class GitTextArea {
43   
44       private final List<String> lines = new ArrayList<>();
45       private final List<Commit> commits = new ArrayList<>();
46   
47       public void add(String s) {
48           lines.add(s);
49       }
50   
51       public List<Commit> getAllCommits() {
52           if (commits.size() == 0)
53               for (int y = 0; y < lines.size(); y++) {
54                   String s = lines.get(y);
55                   final String name = CursorPosition.getCommitNameInLine(s);
56                   final int x = s.indexOf("*");
57                   assert (name == null) == (x == -1);
58                   if (x == -1) {
59                       continue;
60                   }
61                   commits.add(new Commit(name, new CursorPosition(this, x, y)));
62               }
63   
64           return Collections.unmodifiableList(commits);
65       }
66   
67       public char charAt(int x, int y) {
68           return lines.get(y).charAt(x);
69       }
70   
71       public String getLine(int y) {
72           if (y >= lines.size()) {
73               return "";
74           }
75           return lines.get(y);
76       }
77   
78       public Commit getCommitByName(String name) {
79           for (Commit commit : getAllCommits()) {
80               if (commit.getName().equals(name)) {
81                   return commit;
82               }
83           }
84           return null;
85       }
86   
87   }
