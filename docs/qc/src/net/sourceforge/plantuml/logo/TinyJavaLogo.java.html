39   import java.util.Map;

42   import net.sourceforge.plantuml.ugraphic.color.HColorSet;

44   public class TinyJavaLogo {
45       private final LogoScanner scanner = new LogoScanner();
46   
47       private final TurtleGraphicsPane turtleGraphicsPane;
48   
49       private final Map<String, String> dictionary = new HashMap<String, String>();
50       private HColor penColor;
51   
52       public TinyJavaLogo(TurtleGraphicsPane turtleGraphicsPane) {
53           this.turtleGraphicsPane = turtleGraphicsPane;
54       }
55   
56       private void message(String messageText) {
57           // turtleGraphicsPane.message(messageText);
58       }
59   
60       private void error(String messageText) {
61           turtleGraphicsPane.message("Error: " + messageText);
62       }
63   
64       private void parseCommandBlock(int nestLevel) {
65           String commandName;
66           String code;
67           LogoToken token = scanner.getToken();
68           while (token.kind != LogoToken.END_OF_INPUT && token.kind != LogoToken.INVALID_TOKEN) {
69               switch (token.kind) {
70               case LogoToken.FORWARD:
71                   token = scanner.getToken();
72                   if (token.kind == LogoToken.FLOAT || token.kind == LogoToken.INTEGER) {
73                       turtleGraphicsPane.forward(token.value);
74                       token = scanner.getToken();
75                   } else {
76                       error("FORWARD requires distance");
77                       return;
78                   }
79                   break;
80   
81               case LogoToken.BACK:
82                   token = scanner.getToken();
83                   if (token.kind == LogoToken.FLOAT || token.kind == LogoToken.INTEGER) {
84                       turtleGraphicsPane.back(token.value);
85                       token = scanner.getToken();
86                   } else {
87                       error("BACK requires distance");
88                       return;
89                   }
90                   break;
91   
92               case LogoToken.LEFT:
93                   token = scanner.getToken();
94                   if (token.kind == LogoToken.FLOAT || token.kind == LogoToken.INTEGER) {
95                       turtleGraphicsPane.left(token.value);
96                       token = scanner.getToken();
97                   } else {
98                       error("LEFT requires turn angle");
99                       return;
100                   }
101                   break;
102   
103               case LogoToken.RIGHT:
104                   token = scanner.getToken();
105                   if (token.kind == LogoToken.FLOAT || token.kind == LogoToken.INTEGER) {
106                       turtleGraphicsPane.right(token.value);
107                       token = scanner.getToken();
108                   } else {
109                       error("RIGHT requires turn angle");
110                       return;
111                   }
112                   break;
113   
114               case LogoToken.PENUP:
115                   turtleGraphicsPane.penUp();
116                   token = scanner.getToken();
117                   break;
118   
119               case LogoToken.PENDOWN:
120                   turtleGraphicsPane.penDown();
121                   token = scanner.getToken();
122                   break;
123   
124               case LogoToken.HIDETURTLE:
125                   turtleGraphicsPane.hideTurtle();
126                   token = scanner.getToken();
127                   break;
128   
129               case LogoToken.SHOWTURTLE:
130                   turtleGraphicsPane.showTurtle();
131                   token = scanner.getToken();
132                   break;
133   
134               case LogoToken.CLEARSCREEN:
135                   turtleGraphicsPane.clearScreen();
136                   token = scanner.getToken();
137                   break;
138   
139               case LogoToken.REPEAT:
140                   token = scanner.getToken();
141                   if (token.kind != LogoToken.INTEGER) {
142                       error("REPEAT requires positive integer count");
143                       return;
144                   }
145                   int count = token.intValue;
146                   token = scanner.getToken();
147                   if (token.kind != '[') {
148                       error("REPEAT requires block in []");
149                       return;
150                   }
151                   final int blockStart = scanner.getPosition();
152                   while (count-- > 0) {
153                       scanner.setPosition(blockStart);
154                       parseCommandBlock(nestLevel + 1);
155                   }
156                   token = scanner.getToken();
157                   break;
158   
159               case LogoToken.TO:
160                   token = scanner.getToken();
161                   if (token.kind != LogoToken.IDENTIFIER) {
162                       error("TO requires name for new definition");
163                       return;
164                   }
165                   commandName = token.lexeme;
166                   if (dictionary.get(commandName) == null) {
167                       message("Defining new command " + commandName);
168                   } else {
169                       message("Redefining command " + commandName);
170                   }
171                   code = scanner.getRestAsString();
172                   dictionary.put(commandName, code);
173                   token = scanner.getToken();
174                   break;
175   
176               case LogoToken.IDENTIFIER:
177                   commandName = token.lexeme;
178                   code = dictionary.get(commandName);
179                   if (code == null) {
180                       error("Undefined command " + commandName);
181                       return;
182                   }
183                   final String savedCommand = scanner.getSourceString();
184                   final int savedPosition = scanner.getPosition();
185                   scanner.setSourceString(code);
186                   parseCommandBlock(0);
187                   scanner.setSourceString(savedCommand);
188                   scanner.setPosition(savedPosition);
189                   token = scanner.getToken();
190                   break;
191   
192               case LogoToken.SETPC:
193                   token = scanner.getToken();
194                   String s = token.lexeme;
195                   final HColor newPenColor = s == null ? null : HColorSet.instance().getColorOrWhite(s);
196                   if (newPenColor == null) {
197                       error("Unrecognized color name");
198                       return;
199                   }
200                   penColor = newPenColor;
201                   turtleGraphicsPane.setPenColor(penColor);
202                   token = scanner.getToken();
203                   break;
204   
205               case '[':
206                   token = scanner.getToken();
207                   break;
208   
209               case ']':
210                   if (nestLevel == 0) {
211                       error("] without matching [");
212                       token = scanner.getToken();
213                       return;
214                   }
215                   return;
216   
217               default:
218                   error("Unrecognized symbol in input");
219                   return;
220               }
221           }
222       }
223   
224       public void doCommandLine(String commandLine) {
225           message(commandLine);
226           scanner.setSourceString(commandLine);
227           parseCommandBlock(0);
228       }
229   }
