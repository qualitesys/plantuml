131           public void drawU(UGraphic ug) {

133                   // Matches the adjustment in SvekResult.calculateDimension() except no need to

139               for (Map.Entry<IGroup, ST_Agraph_s> ent : clusters.entrySet())
140                   drawGroup(ug, ymirror, ent.getKey(), ent.getValue());
141   
142               for (Map.Entry<ILeaf, ST_Agnode_s> ent : nodes.entrySet()) {

148                   final IEntityImage image = node.getImage();
149                   image.drawU(ug.apply(new UTranslate(corner)));
150               }
151   
152               for (Map.Entry<Link, ST_Agedge_s> ent : edges.entrySet()) {
153                   final Link link = ent.getKey();
154                   if (link.isInvis())
155                       continue;
156   
157                   final ST_Agedge_s edge = ent.getValue();
158                   new SmetanaPath(link, edge, ymirror, diagram, getLabel(link), getQuantifier(link, 1),
159                           getQuantifier(link, 2)).drawU(ug);
160               }
161           }

199       public void drawGroup(UGraphic ug, YMirror ymirror, IGroup group, ST_Agraph_s gr) {
200           JUtils.LOG2("drawGroup");

203               final ST_Agraphinfo_t data = (ST_Agraphinfo_t) tmp1;
204               final ST_boxf bb = (ST_boxf) data.bb;
205               final double llx = bb.LL.x;
206               double lly = bb.LL.y;
207               final double urx = bb.UR.x;
208               double ury = bb.UR.y;
209               if (ymirror != null) {
210                   final double tmpUry = ury;
211                   ury = ymirror.getMirrored(lly);
212                   lly = ymirror.getMirrored(tmpUry);
213               }
214   
215               final Cluster cluster = dotStringFactory.getBibliotekon().getCluster(group);

219               if (dimTitle != null) {
220                   final double x = (llx + urx) / 2 - dimTitle.getWidth() / 2;
221                   cluster.setTitlePosition(new XPoint2D(x, lly));
222               }
223               JUtils.LOG2("cluster=" + cluster);
224               // ug.apply(new UTranslate(llx, lly)).apply(new
225               // UChangeColor(HtmlColorUtils.BLUE))
226               // .draw(new URectangle(urx - llx, ury - lly));
227               cluster.drawU(ug, diagram.getUmlDiagramType(), diagram.getSkinParam());
228           } catch (Exception e) {
229               System.err.println("CANNOT DRAW GROUP");
