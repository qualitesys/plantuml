89       public void drawU(UGraphic ug) {
90   
91           if (link.isHidden())
92               return;
93   
94           HColor color = StyleSignatureBasic
95                   .of(SName.root, SName.element, diagram.getUmlDiagramType().getStyleName(), SName.arrow)
96                   .getMergedStyle(diagram.getSkinParam().getCurrentStyleBuilder()).value(PName.LineColor)
97                   .asColor(diagram.getSkinParam().getIHtmlColorSet());
98   
99           if (this.link.getColors() != null) {
100               final HColor newColor = this.link.getColors().getColor(ColorType.ARROW, ColorType.LINE);
101               if (newColor != null)
102                   color = newColor;
103           } else if (this.link.getSpecificColor() != null)
104               color = this.link.getSpecificColor();
105   
106           DotPath dotPath = getDotPath(edge);
107           if (ymirror != null && dotPath != null)
108               dotPath = ymirror.getMirrored(dotPath);
109   
110           if (dotPath != null) {
111               final LinkType linkType = link.getType();
112               UStroke stroke = linkType.getStroke3(diagram.getSkinParam().getThickness(LineParam.arrow, null));
113               if (link.getColors() != null && link.getColors().getSpecificLineStroke() != null)
114                   stroke = link.getColors().getSpecificLineStroke();
115   
116               ug.apply(stroke).apply(color).draw(dotPath);
117               printExtremityAtStart(ug.apply(color));
118               printExtremityAtEnd(ug.apply(color));
119           }
120           if (getLabelRectangleTranslate("label") != null)
121               label.drawU(ug.apply(getLabelRectangleTranslate("label")));
122   
123           if (getLabelRectangleTranslate("head_label") != null)
124               headLabel.drawU(ug.apply(getLabelRectangleTranslate("head_label")));
125   
126           if (getLabelRectangleTranslate("tail_label") != null)
127               tailLabel.drawU(ug.apply(getLabelRectangleTranslate("tail_label")));
128   
129           // printDebug(ug);
130   
131       }

133       private void printExtremityAtStart(UGraphic ug) {
134           final ExtremityFactory extremityFactory2 = link.getType().getDecor2()
135                   .getExtremityFactoryComplete(diagram.getSkinParam().getBackgroundColor());
136           if (extremityFactory2 == null)
137               return;
138   
139           final ST_splines splines = getSplines(edge);
140           DotPath s = getDotPath(splines);
141           XPoint2D p0 = s.getStartPoint();
142           double startAngle = s.getStartAngle();
143           if (ymirror != null) {
144               p0 = ymirror.getMirrored(p0);
145               startAngle = -startAngle + Math.PI;
146           }
147           try {
148               final UDrawable extremity2 = extremityFactory2.createUDrawable(p0, startAngle, null);
149               if (extremity2 != null)
150                   extremity2.drawU(ug);
151   
152           } catch (UnsupportedOperationException e) {
153               e.printStackTrace();
154               System.err.println("CANNOT DRAW printExtremityAtStart");
155           }
156       }
157   
158       private void printExtremityAtEnd(UGraphic ug) {
159           final ExtremityFactory extremityFactory1 = link.getType().getDecor1()
160                   .getExtremityFactoryComplete(diagram.getSkinParam().getBackgroundColor());
161           if (extremityFactory1 == null)
162               return;
163   
164           final ST_splines splines = getSplines(edge);
165           DotPath s = getDotPath(splines);
166           XPoint2D p0 = s.getEndPoint();
167           double endAngle = s.getEndAngle();
168           if (ymirror != null) {
169               p0 = ymirror.getMirrored(p0);
170               endAngle = -endAngle;
171           }
172           try {
173               final UDrawable extremity1 = extremityFactory1.createUDrawable(p0, endAngle, null);
174               if (extremity1 != null)
175                   extremity1.drawU(ug);
176   
177           } catch (UnsupportedOperationException e) {
178               e.printStackTrace();
179               System.err.println("CANNOT DRAW printExtremityAtEnd");
180           }
181       }

183       private void printDebug(UGraphic ug) {
184           ug = ug.apply(HColors.BLUE).apply(HColors.BLUE.bg());
185           final ST_splines splines = getSplines(edge);
186           final ST_bezier beziers = splines.list.get__(0);
187           for (int i = 0; i < beziers.size; i++) {
188               XPoint2D pt = getPoint(splines, i);
189               if (ymirror != null)
190                   pt = ymirror.getMirrored(pt);
191   
192               ug.apply(new UTranslate(pt).compose(new UTranslate(-1, -1))).draw(new UEllipse(3, 3));
193           }
194           if (getLabelRectangleTranslate("label") != null && getLabelURectangle() != null) {
195               ug = ug.apply(HColors.BLUE).apply(HColors.none().bg());
196               ug.apply(getLabelRectangleTranslate("label")).draw(getLabelURectangle());
197           }

201       private URectangle getLabelURectangle() {
202           final ST_Agedgeinfo_t data = (ST_Agedgeinfo_t) Macro.AGDATA(edge).castTo(ST_Agedgeinfo_t.class);
203           ST_textlabel_t label = (ST_textlabel_t) data.label;
204           if (label == null)
205               return null;
206   
207           final ST_pointf dimen = (ST_pointf) label.dimen;
208           final ST_pointf space = (ST_pointf) label.space;
209           final ST_pointf pos = (ST_pointf) label.pos;
210           final double x = pos.x;
211           final double y = pos.y;
212           final double width = dimen.x;
213           final double height = dimen.y;
214           return new URectangle(width, height);
215       }

217       private UTranslate getLabelRectangleTranslate(String fieldName) {
218           // final String fieldName = "label";
219           final ST_Agedgeinfo_t data = (ST_Agedgeinfo_t) Macro.AGDATA(edge);
220           ST_textlabel_t label = null;
221           if (fieldName.equals("label"))
222               label = data.label;
223           else if (fieldName.equals("head_label"))
224               label = data.head_label;
225           else if (fieldName.equals("tail_label"))
226               label = data.tail_label;
227   
228           if (label == null)
229               return null;
230   
231           final ST_pointf dimen = (ST_pointf) label.dimen;
232           final ST_pointf space = (ST_pointf) label.space;
233           final ST_pointf pos = (ST_pointf) label.pos;
234           final double x = pos.x;
235           final double y = pos.y;
236           final double width = dimen.x;
237           final double height = dimen.y;
238   
239           if (ymirror == null)
240               return new UTranslate(x - width / 2, y - height / 2);
241   
242           return ymirror.getMirrored(new UTranslate(x - width / 2, y + height / 2));
243       }

256       private DotPath getDotPath(ST_splines splines) {
257           if (splines == null) {
258               System.err.println("ERROR, no splines for getDotPath");
259               return null;
260           }
261           DotPath result = new DotPath();
262           final ST_bezier beziers = (ST_bezier) splines.list.get__(0);
263           final XPoint2D pt1 = getPoint(splines, 0);
264           final XPoint2D pt2 = getPoint(splines, 1);
265           final XPoint2D pt3 = getPoint(splines, 2);
266           final XPoint2D pt4 = getPoint(splines, 3);
267           result = result.addCurve(pt1, pt2, pt3, pt4);
268           final int n = beziers.size;
269           for (int i = 4; i < n; i += 3) {
270               final XPoint2D ppt2 = getPoint(splines, i);
271               final XPoint2D ppt3 = getPoint(splines, i + 1);
272               final XPoint2D ppt4 = getPoint(splines, i + 2);
273               result = result.addCurve(ppt2, ppt3, ppt4);
274           }
275           return result;
276       }
277   
278       private XPoint2D getPoint(ST_splines splines, int i) {
279           final ST_bezier beziers = (ST_bezier) splines.list.get__(0);
280           final ST_pointf pt = beziers.list.get__(i);
281           return new XPoint2D(pt.x, pt.y);
282       }
