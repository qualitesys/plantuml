36   package net.sourceforge.plantuml.png;
37   
38   import java.awt.Graphics2D;
39   import java.awt.image.BufferedImage;
40   import java.io.IOException;
41   import java.util.ArrayList;
42   import java.util.Collections;
43   import java.util.List;
44   
45   import net.sourceforge.plantuml.Log;
46   import net.sourceforge.plantuml.SplitParam;
47   import net.sourceforge.plantuml.SuggestedFile;
48   import net.sourceforge.plantuml.security.SFile;
49   import net.sourceforge.plantuml.security.SImageIO;
50   
51   public class PngSplitter {
52   
53       private final List<SFile> files = new ArrayList<>();
54   
55       public PngSplitter(SuggestedFile pngFile, int horizontalPages, int verticalPages, String metadata, int dpi,
56               SplitParam splitParam) throws IOException {
57           if (horizontalPages == 1 && verticalPages == 1) {
58               this.files.add(pngFile.getFile(0));
59               return;
60           }
61   
62           Log.info("Splitting " + horizontalPages + " x " + verticalPages);
63           final SFile full = pngFile.getTmpFile(); // SecurityUtils.File(pngFile.getParentFile(), pngFile.getName() + ".tmp");
64           // Thread.yield();
65           full.delete();
66           // Thread.yield();
67           final boolean ok = pngFile.getFile(0).renameTo(full);
68           // Thread.yield();
69           if (ok == false) {
70               throw new IOException("Cannot rename");
71           }
72   
73           // Thread.yield();
74           final BufferedImage im = SImageIO.read(full);
75           // Thread.yield();
76           final PngSegment horizontalSegment = new PngSegment(im.getWidth(), horizontalPages);
77           final PngSegment verticalSegment = new PngSegment(im.getHeight(), verticalPages);
78   
79           int x = 0;
80           for (int i = 0; i < horizontalPages; i++) {
81               for (int j = 0; j < verticalPages; j++) {
82                   final SFile f = pngFile.getFile(x++);
83                   this.files.add(f);
84                   final int width = horizontalSegment.getLen(i);
85                   final int height = verticalSegment.getLen(j);
86                   BufferedImage piece = im.getSubimage(horizontalSegment.getStart(i), verticalSegment.getStart(j), width,
87                           height);
88                   if (splitParam.isSet()) {
89                       BufferedImage withMargin = new BufferedImage(width + 2 * splitParam.getExternalMargin(), height + 2
90                               * splitParam.getExternalMargin(), BufferedImage.TYPE_INT_ARGB);
91                       final Graphics2D g2d = withMargin.createGraphics();
92                       if (splitParam.getExternalColor() != null) {
93                           g2d.setColor(splitParam.getExternalColor());
94                           g2d.fillRect(0, 0, withMargin.getWidth(), withMargin.getHeight());
95                       }
96                       g2d.drawImage(piece, splitParam.getExternalMargin(), splitParam.getExternalMargin(), null);
97   
98                       if (splitParam.getBorderColor() != null) {
99                           g2d.setColor(splitParam.getBorderColor());
100                           g2d.drawRect(splitParam.getExternalMargin() - 1, splitParam.getExternalMargin() - 1,
101                                   piece.getWidth() + 1, piece.getHeight() + 1);
102                       }
103   
104                       piece = withMargin;
105   
106                       g2d.dispose();
107                   }
108                   // Thread.yield();
109                   PngIO.write(piece, f, metadata, dpi);
110                   // Thread.yield();
111               }
112           }
113   
114           full.delete();
115           Log.info("End of splitting");
116       }
117   
118       public List<SFile> getFiles() {
119           return Collections.unmodifiableList(files);
120       }
121   
122   }
