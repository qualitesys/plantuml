67   abstract class XmiClassDiagramAbstract implements XmlDiagramTransformer {
68   
69       // https://www.ibm.com/developerworks/library/x-wxxm24/
70       // http://pierre.ree7.fr/blog/?p=5
71   
72       protected final ClassDiagram classDiagram;
73       protected final Document document;
74       protected Element ownedElement;
75   
76       protected final Set<IEntity> done = new HashSet<>();
77   
78       public XmiClassDiagramAbstract(ClassDiagram classDiagram) throws ParserConfigurationException {
79           this.classDiagram = classDiagram;
80   
81           final DocumentBuilder builder = XmlFactories.newDocumentBuilder();
82           this.document = builder.newDocument();
83           document.setXmlVersion("1.0");
84           document.setXmlStandalone(true);
85   
86           final Element xmi = document.createElement("XMI");
87           xmi.setAttribute("xmi.version", "1.1");
88           xmi.setAttribute("xmlns:UML", "href://org.omg/UML/1.3");
89           document.appendChild(xmi);
90   
91           final Element header = document.createElement("XMI.header");
92           xmi.appendChild(header);
93   
94           final Element metamodel = document.createElement("XMI.metamodel");
95           metamodel.setAttribute("xmi.name", "UML");
96           metamodel.setAttribute("xmi.version", "1.3");
97           header.appendChild(metamodel);
98   
99           final Element content = document.createElement("XMI.content");
100           xmi.appendChild(content);
101   
102           final Element model = document.createElement("UML:Model");
103           model.setAttribute("xmi.id", CucaDiagramXmiMaker.getModel(classDiagram));
104           model.setAttribute("name", "PlantUML");
105           content.appendChild(model);
106   
107           // <UML:Namespace.ownedElement>
108           this.ownedElement = document.createElement("UML:Namespace.ownedElement");
109           model.appendChild(ownedElement);
110   
111       }
112   
113       final protected String forXMI(String s) {
114           return s.replace(':', ' ');
115       }
116   
117       final protected String forXMI(Display s) {
118           if (Display.isNull(s)) {
119               return "";
120           }
121           return s.get(0).toString().replace(':', ' ');
122       }
123   
124       final public void transformerXml(OutputStream os) throws TransformerException, ParserConfigurationException {
125           final Source source = new DOMSource(document);
126   
127           final Result resultat = new StreamResult(os);
128   
129           final Transformer transformer = XmlFactories.newTransformer();
130           transformer.setOutputProperty(OutputKeys.INDENT, "yes");
131           transformer.setOutputProperty(OutputKeys.ENCODING, UTF_8.name());
132           // tf.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
133           transformer.transform(source, resultat);
134       }
135   
136       final protected Element createEntityNode(IEntity entity) {
137           final Element cla = document.createElement("UML:Class");
138           if (entity.getLeafType() == LeafType.NOTE)
139               return null;
140   
141           cla.setAttribute("xmi.id", entity.getUid());
142           cla.setAttribute("name", entity.getDisplay().get(0).toString());
143           final String parentCode = entity.getIdent().parent().forXmi();
144   
145           if (parentCode.length() == 0)
146               cla.setAttribute("namespace", CucaDiagramXmiMaker.getModel(classDiagram));
147           else
148               cla.setAttribute("namespace", parentCode);
149   
150           final Stereotype stereotype = entity.getStereotype();
151           if (stereotype != null) {
152               final Element stereo = document.createElement("UML:ModelElement.stereotype");
153               for (String s : stereotype.getMultipleLabels()) {
154                   final Element name = document.createElement("UML:Stereotype");
155                   name.setAttribute("name", s);
156                   stereo.appendChild(name);
157               }
158               cla.appendChild(stereo);
159           }
160   
161           final LeafType type = entity.getLeafType();
162           if (type == LeafType.ABSTRACT_CLASS)
163               cla.setAttribute("isAbstract", "true");
164           else if (type == LeafType.INTERFACE)
165               cla.setAttribute("isInterface", "true");
166   
167           if (((ILeaf) entity).isStatic())
168               cla.setAttribute("isStatic", "true");
169   
170           if (((ILeaf) entity).getVisibilityModifier() == VisibilityModifier.PRIVATE_FIELD
171                   || ((ILeaf) entity).getVisibilityModifier() == VisibilityModifier.PRIVATE_METHOD)
172               cla.setAttribute("visibility", ((ILeaf) entity).getVisibilityModifier().getXmiVisibility());
173   
174           final Element feature = document.createElement("UML:Classifier.feature");
175           cla.appendChild(feature);
176   
177           for (CharSequence cs : entity.getBodier().getFieldsToDisplay()) {
178               final Member m = (Member) cs;
179   
180               final Element attribute = document.createElement("UML:Attribute");
181               attribute.setAttribute("xmi.id", "att" + classDiagram.getUniqueSequence());
182               attribute.setAttribute("name", m.getDisplay(false));
183               final VisibilityModifier visibility = m.getVisibilityModifier();
184               if (visibility != null)
185                   attribute.setAttribute("visibility", visibility.getXmiVisibility());
186               if (m.isStatic())
187                   attribute.setAttribute("isStatic", "true");
188   
189               feature.appendChild(attribute);
190           }
191   
192           for (CharSequence cs : entity.getBodier().getMethodsToDisplay()) {
193               final Member m = (Member) cs;
194               final Element operation = document.createElement("UML:Operation");
195               operation.setAttribute("xmi.id", "att" + classDiagram.getUniqueSequence());
196               operation.setAttribute("name", m.getDisplay(false));
197               final VisibilityModifier visibility = m.getVisibilityModifier();
198               if (visibility != null)
199                   operation.setAttribute("visibility", visibility.getXmiVisibility());
200               if (m.isStatic())
201                   operation.setAttribute("isStatic", "true");
202   
203               feature.appendChild(operation);
204           }
205           return cla;
206       }
