28       private HashMap<Participant, Stack<Element>> activeParticipants;

30       public XmiSequenceDiagramStandard(SequenceDiagram diagram, Document document) {
31           super(diagram, document);
32           
33           covered = new Stack<>();
34           covered.push(new HashSet<String>());
35           
36           activeParticipants = new HashMap<>();
37       }

39       @Override
40       public void build() {
41           Node packagedElement = document.appendChild(createElement("uml:Model", new String[][] {
42               {"xmlns:uml", "http://www.omg.org/spec/UML/20110701"},
43               {"xmlns:xmi", "http://schema.omg.org/spec/XMI/2.1"},
44               {"xmi:version","2.1"},
45               {"xmi:id",getXmiId("uml:Model", diagram)}
46           })).appendChild(createUmlElement(diagram, "packagedElement", "Interaction"));
47           packagedElement.appendChild(createUmlElement(diagram, "nestedClassifier", "Collaboration"));
48           
49           for (Participant participant : diagram.participants()) {
50               packagedElement.appendChild(createElement(participant, "lifeline", new String[][] {
51                   {"name", getDisplayString(participant.getDisplay(false))}
52               }));
53           }
54           
55           Node currentFragment = packagedElement;
56           for (Event event : diagram.events()) {
57               if (event instanceof Note) {
58                   buildNoteEvent(packagedElement, (Note) event);
59               } else if (event instanceof LifeEvent) {
60                   buildLifeEvent(packagedElement, (LifeEvent)event);
61               } else if (event instanceof AbstractMessage) {
62                   buildMessage(packagedElement, currentFragment, (AbstractMessage) event);
63               } else if (event instanceof Grouping) {
64                   currentFragment = buildGrouping(currentFragment, (Grouping) event);
65               }
66           }
67       }

69       private void buildLifeEvent(Node packagedElement, LifeEvent event) {
70           if (event.getType() == LifeEventType.ACTIVATE) {
71               Element execution = createUmlElement(event, "fragment", "BehaviorExecutionSpecification");
72               execution.setAttribute("covered", getXmiId("lifeline", event.getParticipant()));
73               execution.setAttribute("start", getLifeEventOccurrenceId(event));
74               packagedElement.appendChild(execution);
75               activeParticipants.putIfAbsent(event.getParticipant(), new Stack<Element>());
76               activeParticipants.get(event.getParticipant()).push(execution);
77           } else if (event.getType() == LifeEventType.DEACTIVATE) {
78               activeParticipants.get(event.getParticipant()).pop().setAttribute("finish", getLifeEventOccurrenceId(event));
79           }
80       }
81       
82       private Participant getReceiver(AbstractMessage message) {
83           if (message instanceof Message) {
84               return message.getParticipant2();
85           } else if (message instanceof MessageExo) {
86               MessageExo exo = (MessageExo) message;
87               if (exo.getType().toString().startsWith("FROM_")) {
88                   return message.getParticipant1();
89               }
90           }
91           return null;
92       }
93   
94       private String getLifeEventOccurrenceId(LifeEvent event) {
95           if (event.getParticipant() == getReceiver(event.getMessage()))
96           {
97               return getXmiId("receiveEvent", event.getMessage());
98           } else {
99               return getXmiId("sendEvent", event.getMessage());
100           }
101       }
102   
103       private void buildMessage(Node packagedElement, Node currentFragment, AbstractMessage message) {
104           if (message instanceof Message) {
105               buildMessage(packagedElement, currentFragment, (Message) message);
106           }
107           else if (message instanceof MessageExo) {
108               buildMessageExo(packagedElement, currentFragment, (MessageExo) message);
109           }
110           
111           if (message.getParticipant1() != null) {
112               covered.peek().add(getXmiId("lifeline", message.getParticipant1()));
113           }
114           if (message.getParticipant2() != null) {
115               covered.peek().add(getXmiId("lifeline", message.getParticipant2()));
116           }
117       }

119       private void buildNoteEvent(Node packagedElement, Note note) {
120           HashSet<String> annotated = new HashSet<String>();
121           
122           if (note.getParticipant() != null) {
123               annotated.add(getXmiId("lifeline", note.getParticipant()));
124           }
125           
126           if (note.getParticipant2() != null) {
127               annotated.add(getXmiId("lifeline", note.getParticipant2()));
128           }
129           
130           buildNote(packagedElement, note, annotated);
131       }
132   
133       private HashSet<String> getAnnotatedElements(Note note, Message message) {
134           HashSet<String> annotated = new HashSet<String>();
135           int p1 = getParticipantNumber(message.getParticipant1());
136           int p2 = getParticipantNumber(message.getParticipant2());
137           NotePosition senderPosition = p1 < p2 ? NotePosition.LEFT : NotePosition.RIGHT;
138           
139           if (note.getPosition() == senderPosition) {
140               annotated.add(getXmiId("sendEvent", message));
141               annotated.add(getXmiId("lifeline", message.getParticipant1()));
142           } else {
143               annotated.add(getXmiId("receiveEvent", message));
144               annotated.add(getXmiId("lifeline", message.getParticipant2()));
145           }
146           return annotated;
147       }

149       private void buildNote(Node packagedElement, Note note, HashSet<String> annotated) {
150           Element comment = createUmlElement(note, "ownedComment", "Comment");
151           if (!annotated.isEmpty()) {
152               comment.setAttribute("annotatedElement", String.join(" ", annotated));
153           }
154           comment.appendChild(document.createElement("body")).appendChild(
155                   document.createTextNode(getDisplayString(note.getStrings())));
156           packagedElement.appendChild(comment);
157       }

159       private int getParticipantNumber(Participant participant) {
160           return Arrays.asList(diagram.participants().toArray()).indexOf(participant);
161       }

163       private void buildMessageExo(Node packagedElement, Node currentFragment, MessageExo message) {
164           String messageEvent = message.getType().toString().startsWith("TO_") ? "sendEvent" : "receiveEvent";
165           String messageEventId = getXmiId(messageEvent, message);
166           currentFragment.appendChild(createMessageOccurrence(message, messageEvent, message.getParticipant1()));
167           packagedElement.appendChild(
168                   setAttributes(createUmlElement(message, "message", "Message"), new String[][]{
169                       {"name", getDisplayString(message.getLabel())},
170                       {messageEvent, messageEventId}}));
171           
172           HashSet<String> annotated = new HashSet<String>();
173           annotated.add(messageEventId);
174           annotated.add(getXmiId("lifeline", message.getParticipant1()));
175           for (Note note : message.getNoteOnMessages()) {
176               buildNote(packagedElement, note, annotated);
177           }
178       }

180       private void buildMessage(Node packagedElement, Node currentFragment, Message message) {
181           currentFragment.appendChild(createMessageOccurrence(message, "sendEvent", message.getParticipant1()));
182           currentFragment.appendChild(createMessageOccurrence(message, "receiveEvent", message.getParticipant2()));
183           packagedElement.appendChild(
184                   setAttributes(createUmlElement(message, "message", "Message"), new String[][]{
185                       {"name", getDisplayString(message.getLabel())},
186                       {"receiveEvent", getXmiId("receiveEvent", message)},
187                       {"sendEvent", getXmiId("sendEvent", message)}}));
188           
189           for (Note note : message.getNoteOnMessages()) {
190               buildNote(packagedElement, note, getAnnotatedElements(note, message));
191           }
192       }

194       private Node buildGrouping(Node currentFragment, Grouping grouping) {
195           if (grouping.getType() == GroupingType.START)
196           {
197               Element group = createUmlElement(grouping, "fragment", "CombinedFragment");
198               group.setAttribute("interactionOperator", grouping.getTitle());
199               currentFragment.appendChild(group);
200               
201               currentFragment = group.appendChild(createElement(grouping, "operand"));
202               currentFragment.appendChild(createGuardElement(grouping));
203               covered.push(new HashSet<String>());
204           }
205           else if(grouping.getType() == GroupingType.ELSE) {
206               currentFragment = currentFragment.getParentNode().appendChild(createElement(grouping, "operand"));
207               currentFragment.appendChild(createGuardElement(grouping));
208           }
209           else if(grouping.getType() == GroupingType.END){
210               Node coveredAttr = document.createAttribute("covered");
211               HashSet<String> topCovered = covered.pop();
212               coveredAttr.setTextContent(String.join(" ", topCovered));
213               currentFragment.getParentNode().getAttributes().setNamedItem(coveredAttr);
214               covered.peek().addAll(topCovered);
215               currentFragment = currentFragment.getParentNode().getParentNode();
216           }
217           return currentFragment;
218       }

220       private Node createGuardElement(Grouping grouping) {
221           Node guard =  createElement(grouping,"guard");
222           guard.appendChild(setAttribute(createUmlElement(grouping, "specification", "LiteralString"), 
223                           "value", grouping.getComment()));
224           return guard;
225       }
