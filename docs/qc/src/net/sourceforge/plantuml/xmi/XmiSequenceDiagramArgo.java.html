1   package net.sourceforge.plantuml.xmi;
2   
3   import org.w3c.dom.Document;
4   import org.w3c.dom.Element;
5   import org.w3c.dom.Node;
6   
7   import net.sourceforge.plantuml.sequencediagram.Event;
8   import net.sourceforge.plantuml.sequencediagram.Message;
9   import net.sourceforge.plantuml.sequencediagram.Participant;
10   import net.sourceforge.plantuml.sequencediagram.SequenceDiagram;
11   
12   public class XmiSequenceDiagramArgo extends XmiSequenceDiagram {
13       public XmiSequenceDiagramArgo(SequenceDiagram diagram, Document document) {
14           super(diagram, document);
15       }
16   
17       @Override
18       public void build() {
19           final Element xmi = document.createElement("XMI");
20           xmi.setAttribute("xmi.version", "1.2");
21           xmi.setAttribute("xmlns:UML", "href://org.omg/UML/1.3");
22           document.appendChild(xmi);
23   
24           final Element header = document.createElement("XMI.header");
25           xmi.appendChild(header);
26   
27           final Element metamodel = document.createElement("XMI.metamodel");
28           metamodel.setAttribute("xmi.name", "UML");
29           metamodel.setAttribute("xmi.version", "1.4");
30           header.appendChild(metamodel);
31   
32           final Element content = document.createElement("XMI.content");
33           xmi.appendChild(content);
34   
35           final Element model = createElement(diagram, "UML:Model");
36           model.setAttribute("name", "PlantUML");
37           content.appendChild(model);
38   
39           Element ownedElement = document.createElement("UML:Namespace.ownedElement");
40           model.appendChild(ownedElement);
41           ownedElement.appendChild(createCollaborationElement());
42   
43           for (Participant participant : diagram.participants()) {
44               ownedElement.appendChild(createActorElement(participant));
45           }
46       }
47   
48       private Element createActorElement(Participant participant) {
49           Element actor = createElement(participant, "UML:Actor");
50           actor.setAttribute("name", String.join(" ", participant.getDisplay(false).asList()));
51           return actor;
52       }
53   
54       private Node createCollaborationElement() {
55           Element collaboration = document.createElement("UML:Collaboration");
56           Element ownedElement = document.createElement("UML:Namespace.ownedElement");
57   
58           for (Participant participant : diagram.participants()) {
59               ownedElement.appendChild(createClassifierRole(participant));
60           }
61   
62           collaboration.appendChild(ownedElement);
63   
64           Node messages = collaboration.appendChild((document.createElement("UML:Collaboration.interaction")))
65                   .appendChild(document.createElement("UML:Interaction"))
66                   .appendChild(document.createElement("UML:Interaction.message"));
67   
68           Message prevMessage = null;
69           for (Event event : diagram.events()) {
70               if (event instanceof Message) {
71                   Message message = (Message) event;
72                   messages.appendChild(createMessage(message, prevMessage));
73                   ownedElement.appendChild(createSendAction(message));
74                   prevMessage = message;
75               }
76           }
77   
78           return collaboration;
79       }
80   
81       private Node createSendAction(Message message) {
82           Element sendAction = createElement(message, "UML:SendAction");
83           sendAction.appendChild(document.createElement("UML:Action.script"))
84                   .appendChild(createElement("UML:ActionExpression",
85                           new String[][] { { "xmi.id", getXmiId("UML:ActionExpression", message) },
86                                   { "body", getDisplayString(message.getLabel()) } }));
87           return sendAction;
88       }
89   
90       private Element createElement(Object object, String tag) {
91           return createElement(tag, new String[][] { { "xmi.id", getXmiId("UML:ActionExpression", object) } });
92       }
93   
94       private Node createRef(String tag, Object target) {
95           Element role = document.createElement(tag);
96           role.setAttribute("xmi.idref", getXmiId(tag, target));
97           return role;
98       }
99   
100       private Element createClassifierRole(Participant participant) {
101           Element classifierRole = createElement(participant, "UML:ClassifierRole");
102   
103           classifierRole.setAttribute("name", participant.getCode());
104           classifierRole.appendChild(document.createElement("UML:ClassifierRole.base"))
105                   .appendChild(createRef("UML:Actor", participant));
106           return classifierRole;
107       }
108   
109       private Element createMessage(Message message, Message prevMessage) {
110           Element messageElement = createElement(message, "UML:Message");
111           messageElement.appendChild(document.createElement("UML:Message.sender"))
112                   .appendChild(createRef("UML:ClassifierRole", message.getParticipant1()));
113           messageElement.appendChild(document.createElement("UML:Message.receiver"))
114                   .appendChild(createRef("UML:ClassifierRole", message.getParticipant2()));
115           messageElement.appendChild(document.createElement("UML:Message.action"))
116                   .appendChild(createRef("UML:SendAction", message));
117   
118           if (prevMessage != null) {
119               messageElement.appendChild(document.createElement("UML:Message.predecessor"))
120                       .appendChild(createRef("UML:Message", prevMessage));
121           }
122   
123           return messageElement;
124       }
125   
126   }
