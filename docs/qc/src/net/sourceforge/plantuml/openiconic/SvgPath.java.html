48   public class SvgPath {
49   
50       // http://www.w3.org/TR/SVG11/paths.html#PathDataEllipticalArcCommands
51       // https://developer.mozilla.org/en-US/docs/Web/SVG/Tutorial/Paths
52       // http://tutorials.jenkov.com/svg/path-element.html
53   
54       private List<Movement> movements = new ArrayList<>();
55       private List<SvgCommand> commands = new ArrayList<>();
56   
57       public SvgPath(String path) {
58           // System.err.println("before=" + path);
59           path = StringDecipher.decipher(path);
60           // System.err.println("after=" + path);
61   
62           for (final StringTokenizer st = new StringTokenizer(path); st.hasMoreTokens();) {
63               final String token = st.nextToken();
64   
65               if (token.matches("[a-zA-Z]")) {
66                   commands.add(new SvgCommandLetter(token));
67               } else {
68                   commands.add(new SvgCommandNumber(token));
69               }
70           }
71           commands = insertMissingLetter(commands);
72           checkArguments(commands);
73           SvgPosition last = new SvgPosition();
74           SvgPosition lastMove = new SvgPosition();
75           SvgPosition mirrorControlPoint = null;
76           final Iterator<SvgCommand> iterator = commands.iterator();
77           while (iterator.hasNext()) {
78               Movement movement = new Movement(iterator);
79               movement = movement.toAbsoluteUpperCase(last);
80   
81               if (movement.getLetter() == 'Z')
82                   last = lastMove;
83   
84               if (movement.is('S'))
85                   movement = movement.mutoToC(mirrorControlPoint);
86   
87               movements.add(movement);
88   
89               if (movement.getLetter() == 'M')
90                   lastMove = movement.lastPosition();
91   
92               if (movement.lastPosition() != null)
93                   last = movement.lastPosition();
94   
95               mirrorControlPoint = movement.getMirrorControlPoint();
96           }
97       }
98   
99       private List<SvgCommand> insertMissingLetter(List<SvgCommand> commands) {
100           final List<SvgCommand> result = new ArrayList<>();
101           final Iterator<SvgCommand> it = commands.iterator();
102           SvgCommandLetter lastLetter = null;
103           while (it.hasNext()) {
104               final SvgCommand cmd = it.next();
105               // System.err.println("cmd=" + cmd);
106               final int nb;
107               if (cmd instanceof SvgCommandNumber) {
108                   // System.err.println("INSERTING " + lastLetter);
109                   result.add(lastLetter);
110                   result.add(cmd);
111                   nb = lastLetter.argumentNumber() - 1;
112               } else {
113                   result.add(cmd);
114                   lastLetter = ((SvgCommandLetter) cmd).implicit();
115                   nb = lastLetter.argumentNumber();
116               }
117               for (int i = 0; i < nb; i++) {
118                   final SvgCommandNumber number = (SvgCommandNumber) it.next();
119                   result.add(number);
120               }
121           }
122           return result;
123       }
124   
125       private void checkArguments(List<SvgCommand> commands) {
126           final Iterator<SvgCommand> it = commands.iterator();
127           while (it.hasNext()) {
128               final SvgCommandLetter cmd = (SvgCommandLetter) it.next();
129               final int nb = cmd.argumentNumber();
130               for (int i = 0; i < nb; i++) {
131                   final SvgCommandNumber number = (SvgCommandNumber) it.next();
132               }
133           }
134       }

136       public String toSvg() {
137           final StringBuilder result = new StringBuilder("<path d=\"");
138           for (Movement move : movements) {
139               result.append(move.toSvg());
140               result.append(' ');
141           }
142           result.append("\"/>");
143           return result.toString();
144       }

146       private UPath toUPath(double factorx, double factory) {
147           final UPath result = new UPath();
148           for (Movement move : movements) {
149               final char letter = move.getLetter();
150               final SvgPosition lastPosition = move.lastPosition();
151               if (letter == 'M') {
152                   result.moveTo(lastPosition.getXDouble() * factorx, lastPosition.getYDouble() * factory);
153               } else if (letter == 'C') {
154                   final SvgPosition ctl1 = move.getSvgPosition(0);
155                   final SvgPosition ctl2 = move.getSvgPosition(2);
156                   result.cubicTo(ctl1.getXDouble() * factorx, ctl1.getYDouble() * factory, ctl2.getXDouble() * factorx,
157                           ctl2.getYDouble() * factory, lastPosition.getXDouble() * factorx,
158                           lastPosition.getYDouble() * factory);
159               } else if (letter == 'L') {
160                   result.lineTo(lastPosition.getXDouble() * factorx, lastPosition.getYDouble() * factory);
161               } else if (letter == 'A') {
162                   final double rx = move.getArgument(0);
163                   final double ry = move.getArgument(1);
164                   final double x_axis_rotation = move.getArgument(2);
165                   final double large_arc_flag = move.getArgument(3);
166                   final double sweep_flag = move.getArgument(4);
167                   result.arcTo(rx * factorx, ry * factory, x_axis_rotation, large_arc_flag, sweep_flag,
168                           lastPosition.getXDouble() * factorx, lastPosition.getYDouble() * factory);
169               } else if (letter == 'Z') {
170                   result.closePath();
171               } else {
172                   throw new UnsupportedOperationException("letter " + letter);
173               }
174           }
175           result.setOpenIconic(true);
176           return result;
177       }

179       private UPath toUPath(AffineTransform at) {
180           final UPath result = new UPath();
181           for (Movement move : movements) {
182               final char letter = move.getLetter();
183               final SvgPosition lastPosition = move.lastPosition();
184               if (letter == 'M') {
185                   result.moveTo(lastPosition.affine(at));
186               } else if (letter == 'C') {
187                   final SvgPosition ctl1 = move.getSvgPosition(0);
188                   final SvgPosition ctl2 = move.getSvgPosition(2);
189                   result.cubicTo(ctl1.affine(at), ctl2.affine(at), lastPosition.affine(at));
190               } else if (letter == 'L') {
191                   result.lineTo(lastPosition.affine(at));
192               } else if (letter == 'A') {
193                   final double rx = move.getArgument(0);
194                   final double ry = move.getArgument(1);
195                   final double x_axis_rotation = move.getArgument(2);
196                   final double large_arc_flag = move.getArgument(3);
197                   final double sweep_flag = move.getArgument(4);
198                   final XPoint2D tmp = lastPosition.affine(at);
199                   result.arcTo(rx * at.getScaleX(), ry * at.getScaleY(), x_axis_rotation, large_arc_flag, sweep_flag,
200                           tmp.getX(), tmp.getY());
201               } else if (letter == 'Z') {
202                   result.closePath();
203               } else {
204                   throw new UnsupportedOperationException("letter " + letter);
205               }
206           }
207           result.setOpenIconic(true);
208           return result;
209       }

211       public void drawMe(UGraphic ug, double factor) {
212           final UPath path = toUPath(factor, factor);
213           ug.draw(path);
214       }

216       public void drawMe(UGraphic ug, AffineTransform at) {
217           final UPath path = toUPath(at);
218           ug.draw(path);
219       }
