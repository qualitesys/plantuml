57   final public class WElement {
58   
59       private final HColor backColor;
60       private final Display label;
61       private final int level;
62       private final String stereotype;
63       private final WElement parent;
64       private final StyleBuilder styleBuilder;
65       private final List<WElement> childrenLeft = new ArrayList<>();
66       private final List<WElement> childrenRight = new ArrayList<>();
67       private final IdeaShape shape;

69       private StyleSignatureBasic getDefaultStyleDefinitionNode(int level) {
70           final String depth = SName.depth(level);
71           if (level == 0)
72               return StyleSignatureBasic.of(SName.root, SName.element, SName.wbsDiagram, SName.node, SName.rootNode)
73                       .add(stereotype).add(depth);
74   
75           if (shape == IdeaShape.NONE && isLeaf())
76               return StyleSignatureBasic
77                       .of(SName.root, SName.element, SName.wbsDiagram, SName.node, SName.leafNode, SName.boxless)
78                       .add(stereotype).add(depth);
79   
80           if (isLeaf())
81               return StyleSignatureBasic.of(SName.root, SName.element, SName.wbsDiagram, SName.node, SName.leafNode)
82                       .add(stereotype).add(depth);
83   
84           if (shape == IdeaShape.NONE)
85               return StyleSignatureBasic.of(SName.root, SName.element, SName.wbsDiagram, SName.node, SName.boxless)
86                       .add(stereotype).add(depth);
87   
88           return StyleSignatureBasic.of(SName.root, SName.element, SName.wbsDiagram, SName.node).add(stereotype).add(depth);
89       }
90   
91       public ISkinParam withBackColor(ISkinParam skinParam) {
92           if (backColor == null)
93               return skinParam;
94   
95           return new SkinParamColors(skinParam, Colors.empty().add(ColorType.BACK, backColor));
96       }

98       public static final int STEP_BY_PARENT = 1000_1000;
99   
100       public Style getStyle() {
101           int deltaPriority = STEP_BY_PARENT * 1000;
102           Style result = styleBuilder.getMergedStyleSpecial(getDefaultStyleDefinitionNode(level), deltaPriority);
103           for (WElement up = parent; up != null; up = up.parent) {
104               final StyleSignatureBasic ss = up.getDefaultStyleDefinitionNode(level).addStar();
105               deltaPriority -= STEP_BY_PARENT;
106               final Style styleParent = styleBuilder.getMergedStyleSpecial(ss, deltaPriority);
107               result = result.mergeWith(styleParent, MergeStrategy.OVERWRITE_EXISTING_VALUE);
108           }
109           return result;
110       }

112       public WElement(HColor backColor, Display label, String stereotype, StyleBuilder styleBuilder, IdeaShape shape) {
113           this(backColor, 0, label, stereotype, null, shape, styleBuilder);
114       }

116       private WElement(HColor backColor, int level, Display label, String stereotype, WElement parent, IdeaShape shape,
117               StyleBuilder styleBuilder) {
118           this.label = label;
119           this.backColor = backColor;
120           this.level = level;
121           this.parent = parent;
122           this.shape = shape;
123           this.styleBuilder = styleBuilder;
124           this.stereotype = stereotype;
125       }
126   
127       public boolean isLeaf() {
128           return childrenLeft.size() == 0 && childrenRight.size() == 0;
129       }
130   
131       public WElement createElement(HColor backColor, int newLevel, Display newLabel, String stereotype,
132               Direction direction, IdeaShape shape, StyleBuilder styleBuilder) {
133           final WElement result = new WElement(backColor, newLevel, newLabel, stereotype, this, shape, styleBuilder);
134           if (direction == Direction.LEFT && newLevel == 1)
135               this.childrenRight.add(0, result);
136           if (direction == Direction.LEFT)
137               this.childrenLeft.add(result);
138           else
139               this.childrenRight.add(result);
140   
141           return result;
142       }
