25   package net.sourceforge.plantuml.json;
26   
27   import java.io.Writer;
28   
29   
30   /**
31    * Controls the formatting of the JSON output. Use one of the available constants.
32    */
33   public abstract class WriterConfig {
34   
35     /**
36      * Write JSON in its minimal form, without any additional whitespace. This is the default.
37      */
38     public static WriterConfig MINIMAL = new WriterConfig() {
39       @Override
40       JsonWriter createWriter(Writer writer) {
41         return new JsonWriter(writer);
42       }
43     };
44   
45     /**
46      * Write JSON in pretty-print, with each value on a separate line and an indentation of two
47      * spaces.
48      */
49     public static WriterConfig PRETTY_PRINT = PrettyPrint.indentWithSpaces(2);
50   
51     abstract JsonWriter createWriter(Writer writer);
52   
53   }
