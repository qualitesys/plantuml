26   
27   /**
28    * An unchecked exception to indicate that an input does not qualify as valid JSON.
29    */
30   @SuppressWarnings("serial") // use default serial UID
31   public class ParseException extends RuntimeException {
32   
33     private final Location location;
34   
35     ParseException(String message, Location location) {
36       super(message + " at " + location);
37       this.location = location;
38     }
39   
40     /**
41      * Returns the location at which the error occurred.
42      *
43      * @return the error location
44      */
45     public Location getLocation() {
46       return location;
47     }
48   
49     /**
50      * Returns the absolute character index at which the error occurred. The offset of the first
51      * character of a document is 0.
52      *
53      * @return the character offset at which the error occurred, will be &gt;= 0
54      * @deprecated Use {@link #getLocation()} instead
55      */
56     @Deprecated
57     public int getOffset() {
58       return location.offset;
59     }
60   
61     /**
62      * Returns the line number in which the error occurred. The number of the first line is 1.
63      *
64      * @return the line in which the error occurred, will be &gt;= 1
65      * @deprecated Use {@link #getLocation()} instead
66      */
67     @Deprecated
68     public int getLine() {
69       return location.line;
70     }
71   
72     /**
73      * Returns the column number at which the error occurred, i.e. the number of the character in its
74      * line. The number of the first character of a line is 1.
75      *
76      * @return the column in which the error occurred, will be &gt;= 1
77      * @deprecated Use {@link #getLocation()} instead
78      */
79     @Deprecated
80     public int getColumn() {
81       return location.column;
82     }
83   
84   }
