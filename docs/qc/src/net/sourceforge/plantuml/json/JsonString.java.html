25   package net.sourceforge.plantuml.json;
26   
27   import java.io.IOException;
28   
29   
30   @SuppressWarnings("serial") // use default serial UID
31   public class JsonString extends JsonValue {
32   
33     private final String string;
34   
35     JsonString(String string) {
36       if (string == null) {
37         throw new NullPointerException("string is null");
38       }
39       this.string = string;
40     }
41   
42     @Override
43     void write(JsonWriter writer) throws IOException {
44       writer.writeString(string);
45     }
46   
47     @Override
48     public boolean isString() {
49       return true;
50     }
51   
52     @Override
53     public String asString() {
54       return string;
55     }
56   
57     @Override
58     public int hashCode() {
59       return string.hashCode();
60     }
61   
62     @Override
63     public boolean equals(Object object) {
64       if (this == object) {
65         return true;
66       }
67       if (object == null) {
68         return false;
69       }
70       if (getClass() != object.getClass()) {
71         return false;
72       }
73       JsonString other = (JsonString)object;
74       return string.equals(other.string);
75     }
76   
77   }
