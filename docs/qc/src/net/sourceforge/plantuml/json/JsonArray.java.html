35   /**
36    * Represents a JSON array, an ordered collection of JSON values.
37    * <p>
38    * Elements can be added using the <code>add(...)</code> methods which accept instances of
39    * {@link JsonValue}, strings, primitive numbers, and boolean values. To replace an element of an
40    * array, use the <code>set(int, ...)</code> methods.
41    * </p>
42    * <p>
43    * Elements can be accessed by their index using {@link #get(int)}. This class also supports
44    * iterating over the elements in document order using an {@link #iterator()} or an enhanced for
45    * loop:
46    * </p>
47    * <pre>
48    * for (JsonValue value : jsonArray) {
49    *   ...
50    * }
51    * </pre>
52    * <p>
53    * An equivalent {@link List} can be obtained from the method {@link #values()}.
54    * </p>
55    * <p>
56    * Note that this class is <strong>not thread-safe</strong>. If multiple threads access a
57    * <code>JsonArray</code> instance concurrently, while at least one of these threads modifies the
58    * contents of this array, access to the instance must be synchronized externally. Failure to do so
59    * may lead to an inconsistent state.
60    * </p>
61    * <p>
62    * This class is <strong>not supposed to be extended</strong> by clients.
63    * </p>
64    */
