25   package net.sourceforge.plantuml.json;
26   
27   import java.io.IOException;
28   import java.io.Writer;
29   import java.util.Arrays;
30   
31   
32   /**
33    * Enables human readable JSON output by inserting whitespace between values.after commas and
34    * colons. Example:
35    *
36    * <pre>
37    * jsonValue.writeTo(writer, PrettyPrint.singleLine());
38    * </pre>
39    */
40   public class PrettyPrint extends WriterConfig {
41   
42     private final char[] indentChars;
43   
44     protected PrettyPrint(char[] indentChars) {
45       this.indentChars = indentChars;
46     }
47   
48     /**
49      * Print every value on a separate line. Use tabs (<code>\t</code>) for indentation.
50      *
51      * @return A PrettyPrint instance for wrapped mode with tab indentation
52      */
53     public static PrettyPrint singleLine() {
54       return new PrettyPrint(null);
55     }
56   
57     /**
58      * Print every value on a separate line. Use the given number of spaces for indentation.
59      *
60      * @param number
61      *          the number of spaces to use
62      * @return A PrettyPrint instance for wrapped mode with spaces indentation
63      */
64     public static PrettyPrint indentWithSpaces(int number) {
65       if (number < 0) {
66         throw new IllegalArgumentException("number is negative");
67       }
68       char[] chars = new char[number];
69       Arrays.fill(chars, ' ');
70       return new PrettyPrint(chars);
71     }
72   
73     /**
74      * Do not break lines, but still insert whitespace between values.
75      *
76      * @return A PrettyPrint instance for single-line mode
77      */
78     public static PrettyPrint indentWithTabs() {
79       return new PrettyPrint(new char[] {'\t'});
80     }
81   
82     @Override
83     protected JsonWriter createWriter(Writer writer) {
84       return new PrettyPrintWriter(writer, indentChars);
85     }
86   
87     private static class PrettyPrintWriter extends JsonWriter {
88   
89       private final char[] indentChars;
90       private int indent;
91   
92       private PrettyPrintWriter(Writer writer, char[] indentChars) {
93         super(writer);
94         this.indentChars = indentChars;
95       }
96   
97       @Override
98       protected void writeArrayOpen() throws IOException {
99         indent++;
100         writer.write('[');
101         writeNewLine();
102       }
103   
104       @Override
105       protected void writeArrayClose() throws IOException {
106         indent--;
107         writeNewLine();
108         writer.write(']');
109       }
110   
111       @Override
112       protected void writeArraySeparator() throws IOException {
113         writer.write(',');
114         if (!writeNewLine()) {
115           writer.write(' ');
116         }
117       }
118   
119       @Override
120       protected void writeObjectOpen() throws IOException {
121         indent++;
122         writer.write('{');
123         writeNewLine();
124       }
125   
126       @Override
127       protected void writeObjectClose() throws IOException {
128         indent--;
129         writeNewLine();
130         writer.write('}');
131       }
132   
133       @Override
134       protected void writeMemberSeparator() throws IOException {
135         writer.write(':');
136         writer.write(' ');
137       }
138   
139       @Override
140       protected void writeObjectSeparator() throws IOException {
141         writer.write(',');
142         if (!writeNewLine()) {
143           writer.write(' ');
144         }
145       }
146   
147       private boolean writeNewLine() throws IOException {
148         if (indentChars == null) {
149           return false;
150         }
151         writer.write('\n');
152         for (int i = 0; i < indent; i++) {
153           writer.write(indentChars);
154         }
155         return true;
156       }
157   
158     }
159   
160   }
