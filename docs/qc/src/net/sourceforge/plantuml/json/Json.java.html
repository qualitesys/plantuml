31   /**
32    * This class serves as the entry point to the minimal-json API.
33    * <p>
34    * To <strong>parse</strong> a given JSON input, use the <code>parse()</code> methods like in this
35    * example:
36    * </p>
37    * <pre>
38    * JsonObject object = Json.parse(string).asObject();
39    * </pre>
40    * <p>
41    * To <strong>create</strong> a JSON data structure to be serialized, use the methods
42    * <code>value()</code>, <code>array()</code>, and <code>object()</code>. For example, the following
43    * snippet will produce the JSON string <em>{"foo": 23, "bar": true}</em>:
44    * </p>
45    * <pre>
46    * String string = Json.object().add("foo", 23).add("bar", true).toString();
47    * </pre>
48    * <p>
49    * To create a JSON array from a given Java array, you can use one of the <code>array()</code>
50    * methods with varargs parameters:
51    * </p>
52    * <pre>
53    * String[] names = ...
54    * JsonArray array = Json.array(names);
55    * </pre>
56    */

283     /**
284      * Parses the given input string as JSON. The input must contain a valid JSON value, optionally
285      * padded with whitespace.
286      *
287      * @param string
288      *          the input string, must be valid JSON
289      * @return a value that represents the parsed JSON
290      * @throws ParseException
291      *           if the input is not valid JSON
292      */
293     public static JsonValue parse(String string) {
294       if (string == null) {
295         throw new NullPointerException("string is null");
296       }
297       DefaultHandler handler = new DefaultHandler();
298       new JsonParser(handler).parse(string);
299       return handler.getValue();
300     }
301   
302     /**
303      * Reads the entire input from the given reader and parses it as JSON. The input must contain a
304      * valid JSON value, optionally padded with whitespace.
305      * <p>
306      * Characters are read in chunks into an input buffer. Hence, wrapping a reader in an additional
307      * <code>BufferedReader</code> likely won't improve reading performance.
308      * </p>
309      *
310      * @param reader
311      *          the reader to read the JSON value from
312      * @return a value that represents the parsed JSON
313      * @throws IOException
314      *           if an I/O error occurs in the reader
315      * @throws ParseException
316      *           if the input is not valid JSON
317      */
318     public static JsonValue parse(Reader reader) throws IOException {
319       if (reader == null) {
320         throw new NullPointerException("reader is null");
321       }
322       DefaultHandler handler = new DefaultHandler();
323       new JsonParser(handler).parse(reader);
324       return handler.getValue();
325     }

327     private static String cutOffPointZero(String string) {
328       if (string.endsWith(".0")) {
329         return string.substring(0, string.length() - 2);
330       }
331       return string;
332     }

334     public static class DefaultHandler extends JsonHandler<JsonArray, JsonObject> {
335   
336       protected JsonValue value;
337   
338       @Override
339       public JsonArray startArray() {
340         return new JsonArray();
341       }
342   
343       @Override
344       public JsonObject startObject() {
345         return new JsonObject();
346       }
347   
348       @Override
349       public void endNull() {
350         value = NULL;
351       }
352   
353       @Override
354       public void endBoolean(boolean bool) {
355         value = bool ? TRUE : FALSE;
356       }
357   
358       @Override
359       public void endString(String string) {
360         value = new JsonString(string);
361       }
362   
363       @Override
364       public void endNumber(String string) {
365         value = new JsonNumber(string);
366       }
367   
368       @Override
369       public void endArray(JsonArray array) {
370         value = array;
371       }
372   
373       @Override
374       public void endObject(JsonObject object) {
375         value = object;
376       }
377   
378       @Override
379       public void endArrayValue(JsonArray array) {
380         array.add(value);
381       }
382   
383       @Override
384       public void endObjectValue(JsonObject object, String name) {
385         object.add(name, value);
386       }
387   
388       public JsonValue getValue() {
389         return value;
390       }
391   
392     }
