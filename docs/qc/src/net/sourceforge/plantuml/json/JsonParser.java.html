25   package net.sourceforge.plantuml.json;
26   
27   import java.io.IOException;
28   import java.io.Reader;
29   import java.io.StringReader;

32   /**
33    * A streaming parser for JSON text. The parser reports all events to a given handler.
34    */
35   public class JsonParser {
36   
37     private static final int MAX_NESTING_LEVEL = 1000;
38     private static final int MIN_BUFFER_SIZE = 10;
39     private static final int DEFAULT_BUFFER_SIZE = 1024;
40   
41     private final JsonHandler<Object, Object> handler;
42     private Reader reader;
43     private char[] buffer;
44     private int bufferOffset;
45     private int index;
46     private int fill;
47     private int line;
48     private int lineOffset;
49     private int current;
50     private StringBuilder captureBuffer;
51     private int captureStart;
52     private int nestingLevel;
53   
54     /*
55      * |                      bufferOffset
56      *                        v
57      * [a|b|c|d|e|f|g|h|i|j|k|l|m|n|o|p|q|r|s|t]        < input
58      *                       [l|m|n|o|p|q|r|s|t|?|?]    < buffer
59      *                          ^               ^
60      *                       |  index           fill
61      */
62   
63     /**
64      * Creates a new JsonParser with the given handler. The parser will report all parser events to
65      * this handler.
66      *
67      * @param handler
68      *          the handler to process parser events
69      */
70     @SuppressWarnings("unchecked")
71     public JsonParser(JsonHandler<?, ?> handler) {
72       if (handler == null) {
73         throw new NullPointerException("handler is null");
74       }
75       this.handler = (JsonHandler<Object, Object>)handler;
76       handler.parser = this;
77     }
78   
79     /**
80      * Parses the given input string. The input must contain a valid JSON value, optionally padded
81      * with whitespace.
82      *
83      * @param string
84      *          the input string, must be valid JSON
85      * @throws ParseException
86      *           if the input is not valid JSON
87      */
88     public void parse(String string) {
89       if (string == null) {
90         throw new NullPointerException("string is null");
91       }
92       int bufferSize = Math.max(MIN_BUFFER_SIZE, Math.min(DEFAULT_BUFFER_SIZE, string.length()));
93       try {
94         parse(new StringReader(string), bufferSize);
95       } catch (IOException exception) {
96         // StringReader does not throw IOException
97         throw new RuntimeException(exception);
98       }
99     }
100   
101     /**
102      * Reads the entire input from the given reader and parses it as JSON. The input must contain a
103      * valid JSON value, optionally padded with whitespace.
104      * <p>
105      * Characters are read in chunks into a default-sized input buffer. Hence, wrapping a reader in an
106      * additional <code>BufferedReader</code> likely won't improve reading performance.
107      * </p>
108      *
109      * @param reader
110      *          the reader to read the input from
111      * @throws IOException
112      *           if an I/O error occurs in the reader
113      * @throws ParseException
114      *           if the input is not valid JSON
115      */
116     public void parse(Reader reader) throws IOException {
117       parse(reader, DEFAULT_BUFFER_SIZE);
118     }
119   
120     /**
121      * Reads the entire input from the given reader and parses it as JSON. The input must contain a
122      * valid JSON value, optionally padded with whitespace.
123      * <p>
124      * Characters are read in chunks into an input buffer of the given size. Hence, wrapping a reader
125      * in an additional <code>BufferedReader</code> likely won't improve reading performance.
126      * </p>
127      *
128      * @param reader
129      *          the reader to read the input from
130      * @param buffersize
131      *          the size of the input buffer in chars
132      * @throws IOException
133      *           if an I/O error occurs in the reader
134      * @throws ParseException
135      *           if the input is not valid JSON
136      */
137     public void parse(Reader reader, int buffersize) throws IOException {
138       if (reader == null) {
139         throw new NullPointerException("reader is null");
140       }
141       if (buffersize <= 0) {
142         throw new IllegalArgumentException("buffersize is zero or negative");
143       }
144       this.reader = reader;
145       buffer = new char[buffersize];
146       bufferOffset = 0;
147       index = 0;
148       fill = 0;
149       line = 1;
150       lineOffset = 0;
151       current = 0;
152       captureStart = -1;
153       read();
154       skipWhiteSpace();
155       readValue();
156       skipWhiteSpace();
157       if (!isEndOfText()) {
158         throw error("Unexpected character");
159       }
160     }
161   
162     private void readValue() throws IOException {
163       switch (current) {
164         case 'n':
165           readNull();
166           break;
167         case 't':
168           readTrue();
169           break;
170         case 'f':
171           readFalse();
172           break;
173         case '"':
174           readString();
175           break;
176         case '[':
177           readArray();
178           break;
179         case '{':
180           readObject();
181           break;
182         case '-':
183         case '0':
184         case '1':
185         case '2':
186         case '3':
187         case '4':
188         case '5':
189         case '6':
190         case '7':
191         case '8':
192         case '9':
193           readNumber();
194           break;
195         default:
196           throw expected("value");
197       }
198     }
199   
200     private void readArray() throws IOException {
201       Object array = handler.startArray();
202       read();
203       if (++nestingLevel > MAX_NESTING_LEVEL) {
204         throw error("Nesting too deep");
205       }
206       skipWhiteSpace();
207       if (readChar(']')) {
208         nestingLevel--;
209         handler.endArray(array);
210         return;
211       }
212       do {
213         skipWhiteSpace();
214         handler.startArrayValue(array);
215         readValue();
216         handler.endArrayValue(array);
217         skipWhiteSpace();
218       } while (readChar(','));
219       if (!readChar(']')) {
220         throw expected("',' or ']'");
221       }
222       nestingLevel--;
223       handler.endArray(array);
224     }
225   
226     private void readObject() throws IOException {
227       Object object = handler.startObject();
228       read();
229       if (++nestingLevel > MAX_NESTING_LEVEL) {
230         throw error("Nesting too deep");
231       }
232       skipWhiteSpace();
233       if (readChar('}')) {
234         nestingLevel--;
235         handler.endObject(object);
236         return;
237       }
238       do {
239         skipWhiteSpace();
240         handler.startObjectName(object);
241         String name = readName();
242         handler.endObjectName(object, name);
243         skipWhiteSpace();
244         if (!readChar(':')) {
245           throw expected("':'");
246         }
247         skipWhiteSpace();
248         handler.startObjectValue(object, name);
249         readValue();
250         handler.endObjectValue(object, name);
251         skipWhiteSpace();
252       } while (readChar(','));
253       if (!readChar('}')) {
254         throw expected("',' or '}'");
255       }
256       nestingLevel--;
257       handler.endObject(object);
258     }
259   
260     private String readName() throws IOException {
261       if (current != '"') {
262         throw expected("name");
263       }
264       return readStringInternal();
265     }
266   
267     private void readNull() throws IOException {
268       handler.startNull();
269       read();
270       readRequiredChar('u');
271       readRequiredChar('l');
272       readRequiredChar('l');
273       handler.endNull();
274     }
275   
276     private void readTrue() throws IOException {
277       handler.startBoolean();
278       read();
279       readRequiredChar('r');
280       readRequiredChar('u');
281       readRequiredChar('e');
282       handler.endBoolean(true);
283     }
284   
285     private void readFalse() throws IOException {
286       handler.startBoolean();
287       read();
288       readRequiredChar('a');
289       readRequiredChar('l');
290       readRequiredChar('s');
291       readRequiredChar('e');
292       handler.endBoolean(false);
293     }
294   
295     private void readRequiredChar(char ch) throws IOException {
296       if (!readChar(ch)) {
297         throw expected("'" + ch + "'");
298       }
299     }
300   
301     private void readString() throws IOException {
302       handler.startString();
303       handler.endString(readStringInternal());
304     }
305   
306     private String readStringInternal() throws IOException {
307       read();
308       startCapture();
309       while (current != '"') {
310         if (current == '\\') {
311           pauseCapture();
312           readEscape();
313           startCapture();
314         } else if (current < 0x20) {
315           throw expected("valid string character");
316         } else {
317           read();
318         }
319       }
320       String string = endCapture();
321       read();
322       return string;
323     }
324   
325     private void readEscape() throws IOException {
326       read();
327       switch (current) {
328         case '"':
329         case '/':
330         case '\\':
331           captureBuffer.append((char)current);
332           break;
333         case 'b':
334           captureBuffer.append('\b');
335           break;
336         case 'f':
337           captureBuffer.append('\f');
338           break;
339         case 'n':
340           captureBuffer.append('\n');
341           break;
342         case 'r':
343           captureBuffer.append('\r');
344           break;
345         case 't':
346           captureBuffer.append('\t');
347           break;
348         case 'u':
349           char[] hexChars = new char[4];
350           for (int i = 0; i < 4; i++) {
351             read();
352             if (!isHexDigit()) {
353               throw expected("hexadecimal digit");
354             }
355             hexChars[i] = (char)current;
356           }
357           captureBuffer.append((char)Integer.parseInt(new String(hexChars), 16));
358           break;
359         default:
360           throw expected("valid escape sequence");
361       }
362       read();
363     }
364   
365     private void readNumber() throws IOException {
366       handler.startNumber();
367       startCapture();
368       readChar('-');
369       int firstDigit = current;
370       if (!readDigit()) {
371         throw expected("digit");
372       }
373       if (firstDigit != '0') {
374         while (readDigit()) {
375         }
376       }
377       readFraction();
378       readExponent();
379       handler.endNumber(endCapture());
380     }
381   
382     private boolean readFraction() throws IOException {
383       if (!readChar('.')) {
384         return false;
385       }
386       if (!readDigit()) {
387         throw expected("digit");
388       }
389       while (readDigit()) {
390       }
391       return true;
392     }
393   
394     private boolean readExponent() throws IOException {
395       if (!readChar('e') && !readChar('E')) {
396         return false;
397       }
398       if (!readChar('+')) {
399         readChar('-');
400       }
401       if (!readDigit()) {
402         throw expected("digit");
403       }
404       while (readDigit()) {
405       }
406       return true;
407     }
408   
409     private boolean readChar(char ch) throws IOException {
410       if (current != ch) {
411         return false;
412       }
413       read();
414       return true;
415     }
416   
417     private boolean readDigit() throws IOException {
418       if (!isDigit()) {
419         return false;
420       }
421       read();
422       return true;
423     }
424   
425     private void skipWhiteSpace() throws IOException {
426       while (isWhiteSpace()) {
427         read();
428       }
429     }
430   
431     private void read() throws IOException {
432       if (index == fill) {
433         if (captureStart != -1) {
434           captureBuffer.append(buffer, captureStart, fill - captureStart);
435           captureStart = 0;
436         }
437         bufferOffset += fill;
438         fill = reader.read(buffer, 0, buffer.length);
439         index = 0;
440         if (fill == -1) {
441           current = -1;
442           index++;
443           return;
444         }
445       }
446       if (current == '\n') {
447         line++;
448         lineOffset = bufferOffset + index;
449       }
450       current = buffer[index++];
451     }
452   
453     private void startCapture() {
454       if (captureBuffer == null) {
455         captureBuffer = new StringBuilder();
456       }
457       captureStart = index - 1;
458     }
459   
460     private void pauseCapture() {
461       int end = current == -1 ? index : index - 1;
462       captureBuffer.append(buffer, captureStart, end - captureStart);
463       captureStart = -1;
464     }
465   
466     private String endCapture() {
467       int start = captureStart;
468       int end = index - 1;
469       captureStart = -1;
470       if (captureBuffer.length() > 0) {
471         captureBuffer.append(buffer, start, end - start);
472         String captured = captureBuffer.toString();
473         captureBuffer.setLength(0);
474         return captured;
475       }
476       return new String(buffer, start, end - start);
477     }
478   
479     Location getLocation() {
480       int offset = bufferOffset + index - 1;
481       int column = offset - lineOffset + 1;
482       return new Location(offset, line, column);
483     }
484   
485     private ParseException expected(String expected) {
486       if (isEndOfText()) {
487         return error("Unexpected end of input");
488       }
489       return error("Expected " + expected);
490     }
491   
492     private ParseException error(String message) {
493       return new ParseException(message, getLocation());
494     }
495   
496     private boolean isWhiteSpace() {
497       return current == ' ' || current == '\t' || current == '\n' || current == '\r';
498     }
499   
500     private boolean isDigit() {
501       return current >= '0' && current <= '9';
502     }
503   
504     private boolean isHexDigit() {
505       return current >= '0' && current <= '9'
506           || current >= 'a' && current <= 'f'
507           || current >= 'A' && current <= 'F';
508     }
509   
510     private boolean isEndOfText() {
511       return current == -1;
512     }
