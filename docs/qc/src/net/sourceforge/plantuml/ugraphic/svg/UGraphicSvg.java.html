83       private UGraphicSvg(UGraphicSvg other) {
84           super(other);
85           this.textAsPath2 = other.textAsPath2;
86           this.target = other.target;
87           this.interactive = other.interactive;

91       public UGraphicSvg(HColor defaultBackground, boolean svgDimensionStyle, XDimension2D minDim,
92               ColorMapper colorMapper, boolean textAsPath, double scale, String linkTarget, String hover, long seed,
93               String preserveAspectRatio, StringBounder stringBounder, LengthAdjust lengthAdjust, boolean interactive) {
94           this(defaultBackground, minDim, colorMapper,
95                   new SvgGraphics(defaultBackground.toSvg(colorMapper), svgDimensionStyle, minDim, scale, hover, seed,
96                           preserveAspectRatio, lengthAdjust, interactive),
97                   textAsPath, linkTarget, stringBounder, interactive);
98           if (defaultBackground instanceof HColorGradient) {
99               final SvgGraphics svg = getGraphicObject();
100               svg.paintBackcolorGradient(colorMapper, (HColorGradient) defaultBackground);
101           }
102       }

119       private UGraphicSvg(HColor defaultBackground, XDimension2D minDim, ColorMapper colorMapper, SvgGraphics svg,
120               boolean textAsPath, String linkTarget, StringBounder stringBounder, boolean interactive) {
121           super(defaultBackground, colorMapper, stringBounder, svg);
122           this.textAsPath2 = textAsPath;
123           this.target = linkTarget;
124           this.interactive = interactive;
125           register();
126       }

128       private void register() {
129           registerDriver(URectangle.class, new DriverRectangleSvg(this));
130           if (textAsPath2)
131               registerDriver(UText.class, new DriverTextAsPathSvg(this));
132           else
133               registerDriver(UText.class, new DriverTextSvg(getStringBounder(), this));
134   
135           registerDriver(ULine.class, new DriverLineSvg(this));
136           registerDriver(UPixel.class, new DriverPixelSvg());
137           registerDriver(UPolygon.class, new DriverPolygonSvg(this));
138           registerDriver(UEllipse.class, new DriverEllipseSvg(this));
139           registerDriver(UImage.class, new DriverImagePng(this));
140           registerDriver(UImageSvg.class, new DriverImageSvgSvg());
141           registerDriver(UPath.class, new DriverPathSvg(this));
142           registerDriver(DotPath.class, new DriverDotPathSvg());
143           registerDriver(UCenteredCharacter.class, new DriverCenteredCharacterSvg());
144       }

150       @Override
151       public void writeToStream(OutputStream os, String metadata, int dpi) throws IOException {
152           try {
153               if (metadata != null)
154                   getGraphicObject().addComment(metadata);
155   
156               if (interactive) {
157                   // For performance reasons and also because we want the entire graph DOM to be
158                   // create so we can register
159                   // the event handlers on them we will append to the end of the document
160                   getGraphicObject().addStyle("onmouseinteractivefooter.css");
161                   getGraphicObject().addScriptTag("https://cdn.jsdelivr.net/npm/@svgdotjs/svg.js@3.0/dist/svg.min.js");
162                   getGraphicObject().addScript("onmouseinteractivefooter.js");
163               }
164   
165               getGraphicObject().createXml(os);
166           } catch (TransformerException e) {
167               throw new IOException(e.toString());
168           }
169       }
