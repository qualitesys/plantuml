36   package net.sourceforge.plantuml.ugraphic;
37   
38   import java.util.ArrayList;
39   import java.util.Collections;
40   import java.util.List;
41   
42   import net.sourceforge.plantuml.awt.geom.XCubicCurve2D;
43   import net.sourceforge.plantuml.awt.geom.XPoint2D;
44   import net.sourceforge.plantuml.posimo.DotPath;
45   
46   public class UMotif {
47   
48       private final List<XPoint2D> points = new ArrayList<>();
49   
50       public UMotif(int... data) {
51           assert data.length % 2 == 0;
52           for (int i = 0; i < data.length; i += 2) {
53               points.add(new XPoint2D(data[i], data[i + 1]));
54           }
55       }
56   
57       public UMotif(String s) {
58           XPoint2D last = new XPoint2D();
59           for (int i = 0; i < s.length(); i++) {
60               final XPoint2D read = convertPoint(s.charAt(i));
61               last = last.move(read);
62               points.add(last);
63           }
64       }
65   
66       double getLength() {
67           return points.get(0).distance(points.get(points.size() - 1));
68       }
69   
70       List<XPoint2D> getPoints() {
71           return Collections.unmodifiableList(points);
72       }
73   
74       public DotPath getRectangle(double width, double height) {
75           final double len = getLength();
76           final int nb1 = (int) (width / len);
77           DotPath h1 = drawHorizontal(0, 0, nb1);
78           final int nb2 = (int) (height / len);
79           final DotPath v1 = drawVertical(h1.getEndPoint().getX(), h1.getEndPoint().getY(), nb2);
80           h1 = h1.addAfter(v1);
81           return h1;
82       }
83   
84       static XPoint2D convertPoint(char c) {
85           final int v = convertFromChar(c);
86           final int x = v % 7;
87           final int y = v / 7;
88           return new XPoint2D(x - 3, y - 3);
89       }
90   
91       static int convertFromChar(char c) {
92           if (c >= 'A' && c <= 'Z') {
93               return c - 'A';
94           }
95           if (c >= 'a' && c <= 'w') {
96               return c - 'a' + 26;
97           }
98           throw new IllegalArgumentException();
99       }
100   
101       public void drawHorizontal(UGraphic ug, double x, double y, int nb) {
102           final DotPath path = drawHorizontal(x, y, nb);
103           ug.draw(path);
104       }
105   
106       public void drawVertical(UGraphic ug, double x, double y, int nb) {
107           final DotPath path = drawVertical(x, y, nb);
108           ug.draw(path);
109       }
110   
111       DotPath drawHorizontal(double x, double y, int nb) {
112           DotPath path = new DotPath();
113           for (int i = 0; i < nb; i++) {
114               path = addHorizontal(x, y, path);
115               x = path.getEndPoint().getX();
116               y = path.getEndPoint().getY();
117           }
118           return path;
119       }
120   
121       DotPath drawVertical(double x, double y, int nb) {
122           DotPath path = new DotPath();
123           for (int i = 0; i < nb; i++) {
124               path = addVertical(x, y, path);
125               x = path.getEndPoint().getX();
126               y = path.getEndPoint().getY();
127           }
128           return path;
129       }
130   
131       private DotPath addHorizontal(double x, double y, DotPath path) {
132           double lastx = 0;
133           double lasty = 0;
134           for (XPoint2D p : points) {
135               final double x1 = lastx + x;
136               final double y1 = lasty + y;
137               final double x2 = p.getX() + x;
138               final double y2 = p.getY() + y;
139               path = path.addAfter(new XCubicCurve2D(x1, y1, x1, y1, x2, y2, x2, y2));
140               lastx = p.getX();
141               lasty = p.getY();
142           }
143           return path;
144       }
145   
146       private DotPath addVertical(double x, double y, DotPath path) {
147           double lastx = 0;
148           double lasty = 0;
149           for (XPoint2D p : points) {
150               final double x1 = lastx + x;
151               final double y1 = lasty + y;
152               final double x2 = p.getY() + x;
153               final double y2 = p.getX() + y;
154               path = path.addAfter(new XCubicCurve2D(x1, y1, x1, y1, x2, y2, x2, y2));
155               lastx = p.getY();
156               lasty = p.getX();
157           }
158           return path;
159       }
