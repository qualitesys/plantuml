36   package net.sourceforge.plantuml.ugraphic.visio;
37   
38   import java.io.IOException;
39   import java.io.OutputStream;
40   
41   public class VisioText implements VisioShape {
42   
43       private final int id;
44       private final String text;
45       private final int fontSize;
46       private final double x;
47       private final double y;
48       private final double width;
49       private final double height;
50   
51       private final double coefFont = 150.0;
52   
53       public static VisioText createInches(int id, String text, int fontSize, double x, double y, double width,
54               double height) {
55           final double coef = 1.8;
56           return new VisioText(id, text, fontSize, toInches(x), toInches(y + 2.5), toInches(width * coef),
57                   toInches(height * coef));
58       }
59   
60       private static double toInches(double val) {
61           return val / 72.0;
62       }
63   
64       private VisioText(int id, String text, int fontSize, double x, double y, double width, double height) {
65           this.id = id;
66           this.text = text;
67           this.x = x;
68           this.y = y;
69           this.height = height;
70           this.width = width;
71           this.fontSize = fontSize;
72       }
73   
74       public void print(OutputStream os) throws IOException {
75           out(os, "<Shape ID='" + id + "' Type='Shape' LineStyle='1' FillStyle='1' TextStyle='3'>");
76           out(os, "<XForm>");
77           out(os, "<PinX>" + x + "</PinX>");
78           out(os, "<PinY>" + y + "</PinY>");
79           out(os, "<Width>" + width + "</Width>");
80           out(os, "<Height>" + height + "</Height>");
81           // out(os, "<PinX>1.0625</PinX>");
82           // out(os, "<PinY>1.9375</PinY>");
83           out(os, "<LocPinX F='Width*0'>0</LocPinX>");
84           out(os, "<LocPinY F='Height*0'>0</LocPinY>");
85           out(os, "<Angle>0</Angle>");
86           out(os, "<FlipX>0</FlipX>");
87           out(os, "<FlipY>0</FlipY>");
88           out(os, "<ResizeMode>0</ResizeMode>");
89           out(os, "</XForm>");
90           out(os, "<TextBlock>");
91           out(os, "<VerticalAlign>0</VerticalAlign>");
92           out(os, "</TextBlock>");
93           out(os, "<Char IX='" + id + "'>");
94           out(os, "<Font F='Inh'>0</Font>");
95           out(os, "<Color F='Inh'>0</Color>");
96           out(os, "<Style F='Inh'>0</Style>");
97           out(os, "<Case F='Inh'>0</Case>");
98           out(os, "<Pos F='Inh'>0</Pos>");
99           out(os, "<FontScale F='Inh'>1</FontScale>");
100           out(os, "<Locale F='Inh'>0</Locale>");
101           out(os, "<Size Unit='PT'>" + fontSize / coefFont + "</Size>");
102           out(os, "<DblUnderline F='Inh'>0</DblUnderline>");
103           out(os, "<Overline F='Inh'>0</Overline>");
104           out(os, "<Strikethru F='Inh'>0</Strikethru>");
105           out(os, "<Perpendicular F='Inh'>0</Perpendicular>");
106           out(os, "<Letterspace F='Inh'>0</Letterspace>");
107           out(os, "<ColorTrans F='Inh'>0</ColorTrans>");
108           out(os, "</Char>");
109   
110           out(os, "<Para IX='" + id + "'>");
111           out(os, "<HorzAlign>0</HorzAlign>");
112           out(os, "</Para>");
113   
114           out(os, "<Text><cp IX='" + id + "'/><pp IX='" + id + "'/>" + text + "</Text>");
115           out(os, "</Shape>");
116   
117       }
118   
119       public VisioShape yReverse(double maxY) {
120           final double y2 = maxY - y;
121           return new VisioText(id, text, fontSize, x, y2, width, height);
122       }
123   
124       private void out(OutputStream os, String s) throws IOException {
125           os.write(s.getBytes());
126           os.write("\n".getBytes());
127       }
128   
129   }
