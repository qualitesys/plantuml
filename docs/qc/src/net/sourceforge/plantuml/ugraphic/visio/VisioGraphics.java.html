57       public void createVsd(OutputStream os) throws IOException {
58           final double width = toInches(limits.getMaxX());
59           final double height = toInches(limits.getMaxY());
60   
61           out(os, "<?xml version='1.0' encoding='utf-8' ?>");
62           out(os,
63                   "<VisioDocument xml:space='preserve' xmlns='http://schemas.microsoft.com/visio/2003/core' xmlns:vx='http://schemas.microsoft.com/visio/2006/extension' xmlns:v14='http://schemas.microsoft.com/office/visio/2010/extension'>");
64           out(os, "<DocumentProperties>");
65           out(os, "<Creator>PlantUML</Creator>");
66           out(os, "</DocumentProperties>");
67           out(os, "<DocumentSheet NameU='TheDoc' LineStyle='0' FillStyle='0' TextStyle='0'>");
68           out(os, "</DocumentSheet>");
69           out(os, "<Masters/>");
70           out(os, "<Pages>");
71           out(os, "<Page ID='0' NameU='Page-1' Name='Page 1' ViewScale='1' ViewCenterX='" + (width / 2)
72                   + "' ViewCenterY='" + (height / 2) + "'>");
73           out(os, "<PageSheet LineStyle='0' FillStyle='0' TextStyle='0'>");
74           out(os, "<PageProps>");
75           out(os, "<PageWidth Unit='IN_F'>" + width + "</PageWidth>");
76           out(os, "<PageHeight Unit='IN_F'>" + height + "</PageHeight>");
77           out(os, "<PageScale Unit='IN_F'>1</PageScale>");
78           out(os, "<DrawingScale Unit='IN_F'>2</DrawingScale>"); // change for scale
79           out(os, "<DrawingSizeType>3</DrawingSizeType>");
80           out(os, "<DrawingScaleType>0</DrawingScaleType>");
81           out(os, "<InhibitSnap>0</InhibitSnap>");
82           out(os, "</PageProps>");
83           out(os, "</PageSheet>");
84           out(os, "<Shapes>");
85           for (VisioShape sh : shapes) {
86               sh.yReverse(height).print(os);
87               // sh.print(os);
88           }
89           out(os, "</Shapes>");
90           out(os, "</Page>");
91           out(os, "</Pages>");
92           out(os, "</VisioDocument>");
93       }

130       public void line(double x1, double y1, double x2, double y2) {
131           ensureVisible(x1, y1);
132           if (x1 == x2 && y1 == y2) {
133               return;
134           }
135           ensureVisible(x2, y2);
136           final VisioLine line = VisioLine.createInches(shapes.size() + 1, x1, y1, x2, y2);
137           shapes.add(line);
138       }

144       public void upath(final double x, final double y, UPath path) {
145           double lx = x;
146           double ly = y;
147           for (USegment seg : path) {
148               final USegmentType type = seg.getSegmentType();
149               final double coord[] = seg.getCoord();
150               if (type == USegmentType.SEG_MOVETO) {
151                   lx = coord[0] + x;
152                   ly = coord[1] + y;
153               } else if (type == USegmentType.SEG_LINETO) {
154                   line(lx, ly, coord[0] + x, coord[1] + y);
155                   lx = coord[0] + x;
156                   ly = coord[1] + y;
157               } else if (type == USegmentType.SEG_QUADTO) {
158                   line(lx, ly, coord[2] + x, coord[3] + y);
159                   lx = coord[2] + x;
160                   ly = coord[3] + y;
161               } else if (type == USegmentType.SEG_CUBICTO) {
162                   line(lx, ly, coord[4] + x, coord[5] + y);
163                   // linePoint(lx, ly, coord[0] + x, coord[1] + y);
164                   // linePoint(coord[0] + x, coord[1] + y, coord[2] + x, coord[3] + y);
165                   // linePoint(coord[2] + x, coord[3] + y, coord[4] + x, coord[5] + y);
166                   lx = coord[4] + x;
167                   ly = coord[5] + y;
168               } else if (type == USegmentType.SEG_CLOSE) {
169                   // Nothing
170               } else if (type == USegmentType.SEG_ARCTO) {
171                   // Nothing
172               } else {
173                   Log.println("unknown5 " + seg);
174               }
175   
176           }
177   
178       }

180       public void polygon(UPolygon poly) {
181           XPoint2D last = null;
182           for (XPoint2D pt : poly.getPoints()) {
183               if (last != null) {
184                   line(last, pt);
185               }
186               last = pt;
187           }
188       }
