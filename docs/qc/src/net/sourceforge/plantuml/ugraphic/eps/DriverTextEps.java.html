70       public void draw(UText shape, double x, double y, ColorMapper mapper, UParam param, EpsGraphics eps) {
71           final UClip clip = clipContainer.getClip();
72           if (clip != null && clip.isInside(x, y) == false)
73               return;
74   
75           final FontConfiguration fontConfiguration = shape.getFontConfiguration();
76           if (fontConfiguration.getColor().isTransparent())
77               return;
78   
79           if (strategy == EpsStrategy.WITH_MACRO_AND_TEXT) {
80               drawAsText(shape, x, y, param, eps, mapper);
81               return;
82           }
83   
84           final TextLayout textLayout = createTextLayout(shape);
85   
86           MinMax dim = null;
87   
88           if (fontConfiguration.containsStyle(FontStyle.BACKCOLOR)) {
89               final Color extended = fontConfiguration.getExtendedColor().toColor(mapper);
90               if (extended != null) {
91                   eps.setStrokeColor(extended);
92                   eps.setFillColor(extended);
93                   eps.setStrokeWidth(1, 0, 0);
94                   if (dim == null)
95                       dim = getMinMax(x, y, getOutline(textLayout).getPathIterator(null));
96   
97                   eps.epsRectangle(dim.getMinX() - 1, dim.getMinY() - 1, dim.getWidth() + 2, dim.getHeight() + 2, 0, 0);
98               }
99           }
100   
101           eps.setStrokeColor(fontConfiguration.getColor().toColor(mapper));
102           drawPathIterator(eps, x, y, getOutline(textLayout));
103   
104           if (fontConfiguration.containsStyle(FontStyle.UNDERLINE)) {
105               final HColor extended = fontConfiguration.getExtendedColor();
106               if (extended != null)
107                   eps.setStrokeColor(extended.toColor(mapper));
108   
109               if (dim == null)
110                   dim = getMinMax(x, y, getOutline(textLayout).getPathIterator(null));
111   
112               eps.setStrokeWidth(1.1, 0, 0);
113               eps.epsLine(x, y + 1.5, x + dim.getWidth(), y + 1.5);
114               eps.setStrokeWidth(1, 0, 0);
115           }
116           if (fontConfiguration.containsStyle(FontStyle.WAVE)) {
117               if (dim == null)
118                   dim = getMinMax(x, y, getOutline(textLayout).getPathIterator(null));
119   
120               final int ypos = (int) (y + 2.5) - 1;
121               final HColor extended = fontConfiguration.getExtendedColor();
122               if (extended != null)
123                   eps.setStrokeColor(extended.toColor(mapper));
124   
125               eps.setStrokeWidth(1.1, 0, 0);
126               for (int i = (int) x; i < x + dim.getWidth() - 5; i += 6) {
127                   eps.epsLine(i, ypos - 0, i + 3, ypos + 1);
128                   eps.epsLine(i + 3, ypos + 1, i + 6, ypos - 0);
129               }
130               eps.setStrokeWidth(1, 0, 0);
131           }
132           if (fontConfiguration.containsStyle(FontStyle.STRIKE)) {
133               final HColor extended = fontConfiguration.getExtendedColor();
134               if (extended != null)
135                   eps.setStrokeColor(extended.toColor(mapper));
136   
137               if (dim == null)
138                   dim = getMinMax(x, y, getOutline(textLayout).getPathIterator(null));
139   
140               // final FontMetrics fm = font.getFontMetrics();
141               final double ypos = (dim.getMinY() + dim.getMaxY() * 2) / 3;
142               eps.setStrokeWidth(1.3, 0, 0);
143               eps.epsLine(x, ypos, x + dim.getWidth(), ypos);
144               eps.setStrokeWidth(1, 0, 0);
145           }
146   
147       }

154           final FontConfiguration fontConfiguration = shape.getFontConfiguration();
155           // final FontMetrics fm =
156           // g2dummy.getFontMetrics(fontConfiguration.getFont().getFont());
157           // final double ypos = y - fm.getDescent() + 0.5;
158           final double ypos = y - 1;
159   
160           eps.setStrokeColor(fontConfiguration.getColor().toColor(mapper));
161           ((EpsGraphicsMacroAndText) eps).drawText(shape.getText(), fontConfiguration, x, ypos);

165       static void drawPathIterator(EpsGraphics eps, double x, double y, Shape shape) {
166           final List<Integer> breaks = analyze(shape);
167           if (breaks.size() == 0) {
168               final PathIterator path = shape.getPathIterator(null);
169               drawSingle(eps, x, y, path);
170               return;
171           }
172           // System.err.println("breaks=" + breaks);
173           final PathIterator path = new PathIteratorLimited(shape, 0, breaks.get(0));
174           drawSingle(eps, x, y, path);
175           for (int i = 0; i < breaks.size() - 1; i++) {
176               final PathIterator path2 = new PathIteratorLimited(shape, breaks.get(i) + 1, breaks.get(i + 1));
177               drawSingle(eps, x, y, path2);
178           }
179           final PathIterator path3 = new PathIteratorLimited(shape, breaks.get(breaks.size() - 1) + 1, Integer.MAX_VALUE);
180           drawSingle(eps, x, y, path3);
181   
182       }

184       private static void drawSingle(EpsGraphics eps, double x, double y, final PathIterator path) {
185           eps.newpath();
186           final double coord[] = new double[6];
187           while (path.isDone() == false) {
188               final int code = path.currentSegment(coord);
189               if (code == PathIterator.SEG_MOVETO)
190                   eps.moveto(coord[0] + x, coord[1] + y);
191               else if (code == PathIterator.SEG_LINETO)
192                   eps.lineto(coord[0] + x, coord[1] + y);
193               else if (code == PathIterator.SEG_CLOSE)
194                   eps.closepath();
195               else if (code == PathIterator.SEG_CUBICTO)
196                   eps.curveto(coord[0] + x, coord[1] + y, coord[2] + x, coord[3] + y, coord[4] + x, coord[5] + y);
197               else if (code == PathIterator.SEG_QUADTO)
198                   eps.quadto(coord[0] + x, coord[1] + y, coord[2] + x, coord[3] + y);
199               else
200                   throw new UnsupportedOperationException("code=" + code);
201   
202               path.next();
203           }
204   
205           eps.fill(path.getWindingRule());
206       }

208       private static List<Integer> analyze(Shape shape) {
209           int count = PathIteratorLimited.count(shape);
210           final List<Integer> closings = getClosings(shape.getPathIterator(null));
211           final List<Integer> result = new ArrayList<>();
212           for (Integer cl : closings) {
213               if (cl + 2 >= count)
214                   break;
215   
216               final PathIterator path1 = new PathIteratorLimited(shape, 0, cl);
217               final PathIterator path2 = new PathIteratorLimited(shape, cl + 1, Integer.MAX_VALUE);
218               final double max1 = getMinMax(0, 0, path1).getMaxX();
219               final double min2 = getMinMax(0, 0, path2).getMinX();
220               if (min2 > max1)
221                   result.add(cl);
222   
223           }
224           return result;
225       }

227       private static List<Integer> getClosings(PathIterator path) {
228           final List<Integer> result = new ArrayList<>();
229           int current = 0;
230           final double coord[] = new double[6];
231           while (path.isDone() == false) {
232               final int code = path.currentSegment(coord);
233               if (code == PathIterator.SEG_CLOSE)
234                   result.add(current);
235   
236               current++;
237               path.next();
238           }
239           return result;
240       }

242       static private MinMax getMinMax(double x, double y, PathIterator path) {
243   
244           MinMax result = MinMax.getEmpty(false);
245   
246           final double coord[] = new double[6];
247           while (path.isDone() == false) {
248               final int code = path.currentSegment(coord);
249               if (code == PathIterator.SEG_MOVETO) {
250                   result = result.addPoint(coord[0] + x, coord[1] + y);
251               } else if (code == PathIterator.SEG_LINETO) {
252                   result = result.addPoint(coord[0] + x, coord[1] + y);
253               } else if (code == PathIterator.SEG_CLOSE) {
254               } else if (code == PathIterator.SEG_CUBICTO) {
255                   result = result.addPoint(coord[0] + x, coord[1] + y);
256                   result = result.addPoint(coord[2] + x, coord[3] + y);
257                   result = result.addPoint(coord[4] + x, coord[5] + y);
258               } else if (code == PathIterator.SEG_QUADTO) {
259                   result = result.addPoint(coord[0] + x, coord[1] + y);
260                   result = result.addPoint(coord[2] + x, coord[3] + y);
261               } else {
262                   throw new UnsupportedOperationException("code=" + code);
263               }
264               path.next();
265           }
266   
267           return result;
268   
269       }
