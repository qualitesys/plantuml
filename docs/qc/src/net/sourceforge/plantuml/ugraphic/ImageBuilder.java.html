256       private ImageData writeImageInternal(OutputStream os, Animation animationArg) throws IOException {
257           XDimension2D dim = getFinalDimension();
258           double dx = 0;
259           double dy = 0;
260           if (animationArg != null) {
261               final MinMax minmax = animationArg.getMinMax(dim);
262               animationArg.setDimension(dim);
263               dim = minmax.getDimension();
264               dx = -minmax.getMinX();
265               dy = -minmax.getMinY();
266           }
267           final Scale scale = titledDiagram == null ? null : titledDiagram.getScale();
268           final double scaleFactor = (scale == null ? 1 : scale.getScale(dim.getWidth(), dim.getHeight())) * getDpi()
269                   / 96.0;
270           if (scaleFactor <= 0)
271               throw new IllegalStateException("Bad scaleFactor");
272   
273           UGraphic ug = createUGraphic(dim, animationArg, dx, dy, scaleFactor,
274                   titledDiagram == null ? new Pragma() : titledDiagram.getPragma());
275           maybeDrawBorder(ug, dim);
276           if (randomPixel)
277               drawRandomPoint(ug);
278   
279           ug = handwritten(ug.apply(new UTranslate(margin.getLeft(), margin.getTop())));
280           udrawable.drawU(ug);
281           ug.flushUg();
282           ug.writeToStream(os, metadata, 96);
283           os.flush();
284   
285           if (ug instanceof UGraphicG2d) {
286               final Set<Url> urls = ((UGraphicG2d) ug).getAllUrlsEncountered();
287               if (urls.size() > 0) {
288                   final CMapData cmap = CMapData.cmapString(urls, scaleFactor);
289                   return new ImageDataComplex(dim, cmap, warningOrError, status);
290               }
291           }
292           return createImageData(dim);
293       }

295       private void maybeDrawBorder(UGraphic ug, XDimension2D dim) {
296           if (skinParam == null)
297               return;
298   
299           final HColor color = new Rose().getHtmlColor(skinParam, ColorParam.diagramBorder);
300   
301           UStroke stroke = skinParam.getThickness(LineParam.diagramBorder, null);
302           if (stroke == null && color != null)
303               stroke = new UStroke();
304           if (stroke == null)
305               return;
306   
307           final URectangle rectangle = new URectangle(dim.getWidth() - stroke.getThickness(),
308                   dim.getHeight() - stroke.getThickness())
309                           .rounded(skinParam.getRoundCorner(CornerParam.diagramBorder, null));
310   
311           ug.apply(color == null ? HColors.BLACK : color).apply(stroke).draw(rectangle);
312       }

402       private UGraphic createUGraphic(final XDimension2D dim, Animation animationArg, double dx, double dy,
403               double scaleFactor, Pragma pragma) {
404           final ColorMapper colorMapper = fileFormatOption.getColorMapper();
405           switch (fileFormatOption.getFileFormat()) {
406           case PNG:
407               return createUGraphicPNG(scaleFactor, dim, animationArg, dx, dy, fileFormatOption.getWatermark());
408           case SVG:
409               final boolean interactive = "true".equalsIgnoreCase(pragma.getValue("svginteractive"));
410               return createUGraphicSVG(scaleFactor, dim, interactive);
411           case EPS:
412               return new UGraphicEps(backcolor, colorMapper, stringBounder, EpsStrategy.getDefault2());
413           case EPS_TEXT:
414               return new UGraphicEps(backcolor, colorMapper, stringBounder, EpsStrategy.WITH_MACRO_AND_TEXT);
415           case HTML5:
416               return new UGraphicHtml5(backcolor, colorMapper, stringBounder);
417           case VDX:
418               return new UGraphicVdx(backcolor, colorMapper, stringBounder);
419           case LATEX:
420               return new UGraphicTikz(backcolor, colorMapper, stringBounder, scaleFactor, true);
421           case LATEX_NO_PREAMBLE:
422               return new UGraphicTikz(backcolor, colorMapper, stringBounder, scaleFactor, false);
423           case BRAILLE_PNG:
424               return new UGraphicBraille(backcolor, colorMapper, stringBounder);
425           case UTXT:
426           case ATXT:
427               return new UGraphicTxt();
428           case DEBUG:
429               return new UGraphicDebug(scaleFactor, dim, getSvgLinkTarget(), getHoverPathColorRGB(), seed,
430                       getPreserveAspectRatio());
431           default:
432               throw new UnsupportedOperationException(fileFormatOption.getFileFormat().toString());
433           }
434       }

442           final UGraphicSvg ug = new UGraphicSvg(backcolor, svgDimensionStyle, dim, fileFormatOption.getColorMapper(),

449       private UGraphic createUGraphicPNG(double scaleFactor, final XDimension2D dim, Animation affineTransforms,
450               double dx, double dy, String watermark) {
451           Color pngBackColor = new Color(0, 0, 0, 0);
452   
453           if (this.backcolor instanceof HColorSimple)
454               pngBackColor = this.backcolor.toColor(fileFormatOption.getColorMapper());
455   
456           if (OptionFlags.getInstance().isReplaceWhiteBackgroundByTransparent()
457                   && (Color.WHITE.equals(pngBackColor) || Color.BLACK.equals(pngBackColor)))
458               pngBackColor = new Color(0, 0, 0, 0);
459   
460           final EmptyImageBuilder builder = new EmptyImageBuilder(watermark, (int) (dim.getWidth() * scaleFactor),
461                   (int) (dim.getHeight() * scaleFactor), pngBackColor, stringBounder);
