59   public class DriverRectangleG2d extends DriverShadowedG2d implements UDriver<URectangle, Graphics2D> {
60   
61       private final double dpiFactor;
62       private final EnsureVisible visible;
63   
64       public DriverRectangleG2d(double dpiFactor, EnsureVisible visible) {
65           this.dpiFactor = dpiFactor;
66           this.visible = visible;
67       }
68   
69       public void draw(URectangle rect, double x, double y, ColorMapper mapper, UParam param, Graphics2D g2d) {
70           g2d.setStroke(new BasicStroke((float) param.getStroke().getThickness()));
71           final double rx = rect.getRx();
72           final double ry = rect.getRy();
73           final Shape shape;
74           if (rx == 0 && ry == 0)
75               shape = new Rectangle2D.Double(x, y, rect.getWidth(), rect.getHeight());
76           else
77               shape = new RoundRectangle2D.Double(x, y, rect.getWidth(), rect.getHeight(), rx, ry);
78   
79           visible.ensureVisible(x, y);
80           visible.ensureVisible(x + rect.getWidth(), y + rect.getHeight());
81   
82           final HColor back = param.getBackcolor();
83   
84           // Shadow
85           if (rect.getDeltaShadow() != 0) {
86               if (back.isTransparent())
87                   drawOnlyLineShadowSpecial(g2d, shape, rect.getDeltaShadow(), dpiFactor);
88               else
89                   drawShadow(g2d, shape, rect.getDeltaShadow(), dpiFactor);
90           }
91   
92           final HColor color = param.getColor();
93           if (color == null) {
94               param.getColor();
95           }
96           if (back instanceof HColorGradient) {
97               final GradientPaint paint = getPaintGradient(x, y, mapper, rect.getWidth(), rect.getHeight(), back);
98               g2d.setPaint(paint);
99               g2d.fill(shape);
100               drawBorder(param, color, mapper, rect, shape, g2d, x, y);
101           } else {
102               if (param.getBackcolor().isTransparent() == false) {
103                   g2d.setColor(param.getBackcolor().toColor(mapper));
104                   DriverLineG2d.manageStroke(param, g2d);
105                   managePattern(param, g2d);
106                   g2d.fill(shape);
107               }
108               if (color.equals(param.getBackcolor()) == false)
109                   drawBorder(param, color, mapper, rect, shape, g2d, x, y);
110   
111           }
112       }
113   
114       public static void drawBorder(UParam param, HColor color, ColorMapper mapper, UShapeSized sized, Shape shape,
115               Graphics2D g2d, double x, double y) {
116           if (color.isTransparent())
117               return;
118   
119           if (color instanceof HColorGradient) {
120               final GradientPaint paint = getPaintGradient(x, y, mapper, sized.getWidth(), sized.getHeight(), color);
121               g2d.setPaint(paint);
122           } else {
123               g2d.setColor(color.toColor(mapper));
124           }
125           DriverLineG2d.manageStroke(param, g2d);
126           g2d.draw(shape);
127       }
128   
129       public static GradientPaint getPaintGradient(double x, double y, ColorMapper mapper, double width, double height,
130               final HColor back) {
131           final HColorGradient gr = (HColorGradient) back;
132           final char policy = gr.getPolicy();
133           final GradientPaint paint;
134           if (policy == '|')
135               paint = new GradientPaint((float) x, (float) (y + height) / 2, gr.getColor1().toColor(mapper),
136                       (float) (x + width), (float) (y + height) / 2, gr.getColor2().toColor(mapper));
137           else if (policy == '\\')
138               paint = new GradientPaint((float) x, (float) (y + height), gr.getColor1().toColor(mapper),
139                       (float) (x + width), (float) y, gr.getColor2().toColor(mapper));
140           else if (policy == '-')
141               paint = new GradientPaint((float) (x + width) / 2, (float) y, gr.getColor1().toColor(mapper),
142                       (float) (x + width) / 2, (float) (y + height), gr.getColor2().toColor(mapper));
143           else
144               // for /
145               paint = new GradientPaint((float) x, (float) y, gr.getColor1().toColor(mapper), (float) (x + width),
146                       (float) (y + height), gr.getColor2().toColor(mapper));
147   
148           return paint;
149       }

151       public static void managePattern(UParam param, Graphics2D g2d) {
152           final UPattern pattern = param.getPattern();
153           if (pattern == UPattern.VERTICAL_STRIPE) {
154               final BufferedImage bi = new BufferedImage(4, 4, BufferedImage.TYPE_INT_ARGB);
155               final Rectangle r = new Rectangle(0, 0, 4, 4);
156               final int rgb = ((HColorSimple) param.getBackcolor()).getAwtColor().getRGB();
157               for (int i = 0; i < 4; i++)
158                   for (int j = 0; j < 4; j++)
159                       if (i == 0 || i == 1)
160                           bi.setRGB(i, j, rgb);
161   
162               g2d.setPaint(new TexturePaint(bi, r));
163           } else if (pattern == UPattern.HORIZONTAL_STRIPE) {
164               final BufferedImage bi = new BufferedImage(4, 4, BufferedImage.TYPE_INT_ARGB);
165               final Rectangle r = new Rectangle(0, 0, 4, 4);
166               final int rgb = ((HColorSimple) param.getBackcolor()).getAwtColor().getRGB();
167               for (int i = 0; i < 4; i++)
168                   for (int j = 0; j < 4; j++)
169                       if (j == 0 || j == 1)
170                           bi.setRGB(i, j, rgb);
171   
172               g2d.setPaint(new TexturePaint(bi, r));
173           } else if (pattern == UPattern.SMALL_CIRCLE) {
174               final BufferedImage bi = new BufferedImage(4, 4, BufferedImage.TYPE_INT_ARGB);
175               final Rectangle r = new Rectangle(0, 0, 4, 4);
176               final int rgb = ((HColorSimple) param.getBackcolor()).getAwtColor().getRGB();
177               bi.setRGB(0, 1, rgb);
178               bi.setRGB(1, 0, rgb);
179               bi.setRGB(1, 1, rgb);
180               bi.setRGB(1, 2, rgb);
181               bi.setRGB(2, 1, rgb);
182               g2d.setPaint(new TexturePaint(bi, r));
183           }
184       }
