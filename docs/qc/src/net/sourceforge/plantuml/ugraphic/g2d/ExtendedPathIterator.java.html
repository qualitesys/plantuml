57   /**
58    * The <code>ExtendedPathIterator</code> class represents a geometric path constructed from straight lines, quadratic
59    * and cubic (Bezier) curves and elliptical arcs. This interface is identical to that of PathIterator except it can
60    * return SEG_ARCTO from currentSegment, also the array of values passed to currentSegment must be of length 7 or an
61    * error will be thrown.
62    * 
63    * This does not extend PathIterator as it would break the interface contract for that class.
64    * 
65    * @author <a href="mailto:deweese@apache.org">Thomas DeWeese</a>
66    * @version $Id: ExtendedPathIterator.java 475477 2006-11-15 22:44:28Z cam $
67    */
68   public interface ExtendedPathIterator {
69   
70       /**
71        * The segment type constant that specifies that the preceding subpath should be closed by appending a line segment
72        * back to the point corresponding to the most recent SEG_MOVETO.
73        */
74       int SEG_CLOSE = PathIterator.SEG_CLOSE;
75   
76       /**
77        * The segment type constant for a point that specifies the end point of a line to be drawn from the most recently
78        * specified point.
79        */
80       int SEG_MOVETO = PathIterator.SEG_MOVETO;
81   
82       /**
83        * The segment type constant for a point that specifies the end point of a line to be drawn from the most recently
84        * specified point.
85        */
86       int SEG_LINETO = PathIterator.SEG_LINETO;
87   
88       /**
89        * The segment type constant for the pair of points that specify a quadratic parametric curve to be drawn from the
90        * most recently specified point. The curve is interpolated by solving the parametric control equation in the range
91        * (t=[0..1]) using the most recently specified (current) point (CP), the first control point (P1), and the final
92        * interpolated control point (P2).
93        */
94       int SEG_QUADTO = PathIterator.SEG_QUADTO;
95   
96       /**
97        * The segment type constant for the set of 3 points that specify a cubic parametric curve to be drawn from the most
98        * recently specified point. The curve is interpolated by solving the parametric control equation in the range
99        * (t=[0..1]) using the most recently specified (current) point (CP), the first control point (P1), the second
100        * control point (P2), and the final interpolated control point (P3).
101        */
102       int SEG_CUBICTO = PathIterator.SEG_CUBICTO;
103   
104       /**
105        * The segment type constant for an elliptical arc. This consists of Seven values [rx, ry, angle, largeArcFlag,
106        * sweepFlag, x, y]. rx, ry are the radious of the ellipse. angle is angle of the x axis of the ellipse.
107        * largeArcFlag is zero if the smaller of the two arcs are to be used. sweepFlag is zero if the 'left' branch is
108        * taken one otherwise. x and y are the destination for the ellipse.
109        */
110       int SEG_ARCTO = USegmentType.SEG_ARCTO_VALUE;
111   
112       /**
113        * The winding rule constant for specifying an even-odd rule for determining the interior of a path. The even-odd
114        * rule specifies that a point lies inside the path if a ray drawn in any direction from that point to infinity is
115        * crossed by path segments an odd number of times.
116        */
117       int WIND_EVEN_ODD = PathIterator.WIND_EVEN_ODD;
118   
119       /**
120        * The winding rule constant for specifying a non-zero rule for determining the
121        * interior of a path. The non-zero rule specifies that a point lies inside the
122        * path if a ray drawn in any direction from that point to infinity is crossed
123        * by path segments a different number of times in the counter-clockwise
124        * direction than the clockwise direction.
125        */
126       int WIND_NON_ZERO = PathIterator.WIND_NON_ZERO;
127   
128       int currentSegment();
129   
130       int currentSegment(double[] coords);
131   
132       int getWindingRule();
133   
134       boolean isDone();
135   
136       void next();
137   }
