70   /**
71    * The <code>ExtendedGeneralPath</code> class represents a geometric path constructed from straight lines, quadratic and
72    * cubic (Bezier) curves and elliptical arc. This class delegates lines and curves to an enclosed
73    * <code>GeneralPath</code>. Elliptical arc is implemented using an <code>Arc2D</code> in double precision.
74    * 
75    * <p>
76    * <b>Warning</b> : An elliptical arc may be composed of several path segments. For further details, see the SVG
77    * Appendix&nbsp;F.6
78    * 
79    * @author <a href="mailto:Thierry.Kormann@sophia.inria.fr">Thierry Kormann</a>
80    * @version $Id: ExtendedGeneralPath.java 594018 2007-11-12 04:17:41Z cam $
81    */

128       /**
129        * Adds an elliptical arc, defined by two radii, an angle from the x-axis, a flag to choose the large arc or not, a
130        * flag to indicate if we increase or decrease the angles and the final point of the arc.
131        * 
132        * @param rx
133        *            the x radius of the ellipse
134        * @param ry
135        *            the y radius of the ellipse
136        * 
137        * @param angle
138        *            the angle from the x-axis of the current coordinate system to the x-axis of the ellipse in degrees.
139        * 
140        * @param largeArcFlag
141        *            the large arc flag. If true the arc spanning less than or equal to 180 degrees is chosen, otherwise
142        *            the arc spanning greater than 180 degrees is chosen
143        * 
144        * @param sweepFlag
145        *            the sweep flag. If true the line joining center to arc sweeps through decreasing angles otherwise it
146        *            sweeps through increasing angles
147        * 
148        * @param x
149        *            the absolute x coordinate of the final point of the arc.
150        * @param y
151        *            the absolute y coordinate of the final point of the arc.
152        */
153       public void arcTo(double rx, double ry, double angle, boolean largeArcFlag, boolean sweepFlag,
154               double x, double y) {
155   
156           // Ensure radii are valid
157           if (rx == 0 || ry == 0) {
158               lineTo(x, y);
159               return;
160           }
161   
162           checkMoveTo(); // check if prev command was moveto
163   
164           // Get the current (x, y) coordinates of the path
165           final double x0 = cx;
166           final double y0 = cy;
167           if (x0 == x && y0 == y) {
168               // If the endpoints (x, y) and (x0, y0) are identical, then this
169               // is equivalent to omitting the elliptical arc segment entirely.
170               return;
171           }
172   
173           final Arc2D arc = computeArc(x0, y0, rx, ry, angle, largeArcFlag, sweepFlag, x, y);
174           if (arc == null) {
175               return;
176           }
177   
178           final AffineTransform t = AffineTransform.getRotateInstance(Math.toRadians(angle), arc.getCenterX(),
179                   arc.getCenterY());
180           final Shape s = t.createTransformedShape(arc);
181           path.append(s, true);
182   
183           makeRoom(7);
184           types[numSeg++] = ExtendedPathIterator.SEG_ARCTO;
185           values[numVals++] = rx;
186           values[numVals++] = ry;
187           values[numVals++] = angle;
188           values[numVals++] = largeArcFlag ? 1 : 0;
189           values[numVals++] = sweepFlag ? 1 : 0;
190           cx = values[numVals++] = x;
191           cy = values[numVals++] = y;
192       }
193   
194       /**
195        * This constructs an unrotated Arc2D from the SVG specification of an Elliptical arc. To get the final arc you need
196        * to apply a rotation transform such as:
197        * 
198        * AffineTransform.getRotateInstance (angle, arc.getX()+arc.getWidth()/2, arc.getY()+arc.getHeight()/2);
199        */
200       public static Arc2D computeArc(double x0, double y0, double rx, double ry, double angle, boolean largeArcFlag,
201               boolean sweepFlag, double x, double y) {
202           //
203           // Elliptical arc implementation based on the SVG specification notes
204           //
205   
206           // Compute the half distance between the current and the final point
207           final double dx2 = (x0 - x) / 2.0;
208           final double dy2 = (y0 - y) / 2.0;
209           // Convert angle from degrees to radians
210           angle = Math.toRadians(angle % 360.0);
211           final double cosAngle = Math.cos(angle);
212           final double sinAngle = Math.sin(angle);
213   
214           //
215           // Step 1 : Compute (x1, y1)
216           //
217           final double x1 = cosAngle * dx2 + sinAngle * dy2;
218           final double y1 = -sinAngle * dx2 + cosAngle * dy2;
219           // Ensure radii are large enough
220           rx = Math.abs(rx);
221           ry = Math.abs(ry);
222           double prx = rx * rx;
223           double pry = ry * ry;
224           final double px1 = x1 * x1;
225           final double py1 = y1 * y1;
226           // check that radii are large enough
227           final double radiiCheck = px1 / prx + py1 / pry;
228           if (radiiCheck > 1) {
229               rx = Math.sqrt(radiiCheck) * rx;
230               ry = Math.sqrt(radiiCheck) * ry;
231               prx = rx * rx;
232               pry = ry * ry;
233           }
234   
235           //
236           // Step 2 : Compute (cx1, cy1)
237           //
238           double sign = (largeArcFlag == sweepFlag) ? -1 : 1;
239           double sq = ((prx * pry) - (prx * py1) - (pry * px1)) / ((prx * py1) + (pry * px1));
240           sq = (sq < 0) ? 0 : sq;
241           final double coef = sign * Math.sqrt(sq);
242           final double cx1 = coef * ((rx * y1) / ry);
243           final double cy1 = coef * -((ry * x1) / rx);
244   
245           //
246           // Step 3 : Compute (cx, cy) from (cx1, cy1)
247           //
248           final double sx2 = (x0 + x) / 2.0;
249           final double sy2 = (y0 + y) / 2.0;
250           final double cx = sx2 + (cosAngle * cx1 - sinAngle * cy1);
251           final double cy = sy2 + (sinAngle * cx1 + cosAngle * cy1);
252   
253           //
254           // Step 4 : Compute the angleStart (angle1) and the angleExtent (dangle)
255           //
256           final double ux = (x1 - cx1) / rx;
257           final double uy = (y1 - cy1) / ry;
258           final double vx = (-x1 - cx1) / rx;
259           final double vy = (-y1 - cy1) / ry;
260           // Compute the angle start
261           double n = Math.sqrt((ux * ux) + (uy * uy));
262           double p = ux; // (1 * ux) + (0 * uy)
263           sign = (uy < 0) ? -1.0 : 1.0;
264           double angleStart = Math.toDegrees(sign * Math.acos(p / n));
265   
266           // Compute the angle extent
267           n = Math.sqrt((ux * ux + uy * uy) * (vx * vx + vy * vy));
268           p = ux * vx + uy * vy;
269           sign = (ux * vy - uy * vx < 0) ? -1.0 : 1.0;
270           double angleExtent = Math.toDegrees(sign * Math.acos(p / n));
271           if (!sweepFlag && angleExtent > 0) {
272               angleExtent -= 360f;
273           } else if (sweepFlag && angleExtent < 0) {
274               angleExtent += 360f;
275           }
276           angleExtent %= 360f;
277           angleStart %= 360f;
278   
279           //
280           // We can now build the resulting Arc2D in double precision
281           //
282           final Arc2D.Double arc = new Arc2D.Double();
283           arc.x = cx - rx;
284           arc.y = cy - ry;
285           arc.width = rx * 2.0;
286           arc.height = ry * 2.0;
287           arc.start = -angleStart;
288           arc.extent = -angleExtent;
289   
290           return arc;
291       }

350       /**
351        * Delegates to the enclosed <code>GeneralPath</code>.
352        */
353       public void closePath() {
354           // Don't double close path.
355           if (numSeg != 0 && types[numSeg - 1] == PathIterator.SEG_CLOSE) {
356               return;
357           }
358   
359           // Only close path if the previous command wasn't a moveto
360           if (numSeg != 0 && types[numSeg - 1] != PathIterator.SEG_MOVETO) {
361               path.closePath();
362           }
363   
364           makeRoom(0);
365           types[numSeg++] = PathIterator.SEG_CLOSE;
366           cx = mx;
367           cy = my;
368       }
369   
370       /**
371        * Checks if previous command was a moveto command, skipping a close command (if present).
372        */
373       protected void checkMoveTo() {
374           if (numSeg == 0) {
375               return;
376           }
377   
378           switch (types[numSeg - 1]) {
379   
380           case PathIterator.SEG_MOVETO:
381               path.moveTo(values[numVals - 2], values[numVals - 1]);
382               break;
383   
384           case PathIterator.SEG_CLOSE:
385               if (numSeg == 1) {
386                   return;
387               }
388               if (types[numSeg - 2] == PathIterator.SEG_MOVETO) {
389                   path.moveTo(values[numVals - 2], values[numVals - 1]);
390               }
391               break;
392   
393           default:
394               break;
395           }
396       }

405       /**
406        * Delegates to the enclosed <code>GeneralPath</code>.
407        */
408       public void append(PathIterator pi, boolean connect) {
409           final double[] vals = new double[6];
410   
411           while (!pi.isDone()) {
412               Arrays.fill(vals, 0);
413               int type = pi.currentSegment(vals);
414               pi.next();
415               if (connect && numVals != 0) {
416                   if (type == PathIterator.SEG_MOVETO) {
417                       final double x = vals[0];
418                       final double y = vals[1];
419                       if (x != cx || y != cy) {
420                           // Change MOVETO to LINETO.
421                           type = PathIterator.SEG_LINETO;
422                       } else {
423                           // Redundant segment (move to current loc) drop it...
424                           if (pi.isDone()) {
425                               break; // Nothing interesting
426                           }
427                           type = pi.currentSegment(vals);
428                           pi.next();
429                       }
430                   }
431                   connect = false;
432               }
433   
434               switch (type) {
435               case PathIterator.SEG_CLOSE:
436                   closePath();
437                   break;
438               case PathIterator.SEG_MOVETO:
439                   moveTo(vals[0], vals[1]);
440                   break;
441               case PathIterator.SEG_LINETO:
442                   lineTo(vals[0], vals[1]);
443                   break;
444               case PathIterator.SEG_QUADTO:
445                   quadTo(vals[0], vals[1], vals[2], vals[3]);
446                   break;
447               case PathIterator.SEG_CUBICTO:
448                   curveTo(vals[0], vals[1], vals[2], vals[3], vals[4], vals[5]);
449                   break;
450               }
451           }
452       }

454       /**
455        * Delegates to the enclosed <code>GeneralPath</code>.
456        */
457       public void append(ExtendedPathIterator epi, boolean connect) {
458           final double[] vals = new double[7];
459           while (!epi.isDone()) {
460               Arrays.fill(vals, 0);
461               int type = epi.currentSegment(vals);
462               epi.next();
463               if (connect && numVals != 0) {
464                   if (type == PathIterator.SEG_MOVETO) {
465                       final double x = vals[0];
466                       final double y = vals[1];
467                       if ((x != cx) || (y != cy)) {
468                           // Change MOVETO to LINETO.
469                           type = PathIterator.SEG_LINETO;
470                       } else {
471                           // Redundant segment (move to current loc) drop it...
472                           if (epi.isDone()) {
473                               break; // Nothing interesting
474                           }
475                           type = epi.currentSegment(vals);
476                           epi.next();
477                       }
478                   }
479                   connect = false;
480               }
481   
482               switch (type) {
483               case PathIterator.SEG_CLOSE:
484                   closePath();
485                   break;
486               case PathIterator.SEG_MOVETO:
487                   moveTo(vals[0], vals[1]);
488                   break;
489               case PathIterator.SEG_LINETO:
490                   lineTo(vals[0], vals[1]);
491                   break;
492               case PathIterator.SEG_QUADTO:
493                   quadTo(vals[0], vals[1], vals[2], vals[3]);
494                   break;
495               case PathIterator.SEG_CUBICTO:
496                   curveTo(vals[0], vals[1], vals[2], vals[3], vals[4], vals[5]);
497                   break;
498               case ExtendedPathIterator.SEG_ARCTO:
499                   arcTo(vals[0], vals[1], vals[2], vals[3] != 0, vals[4] != 0, vals[5], vals[6]);
500                   break;
501               }
502           }
503       }

639               return types[segNum];
640           }

642           public int currentSegment(double[] coords) {
643               final int ret = types[segNum];
644               switch (ret) {
645               case SEG_CLOSE:
646                   break;
647               case SEG_MOVETO:
648               case SEG_LINETO:
649                   coords[0] = values[valsIdx];
650                   coords[1] = values[valsIdx + 1];
651                   break;
652               case SEG_QUADTO:
653                   coords[0] = values[valsIdx];
654                   coords[1] = values[valsIdx + 1];
655                   coords[2] = values[valsIdx + 2];
656                   coords[3] = values[valsIdx + 3];
657                   break;
658               case SEG_CUBICTO:
659                   coords[0] = values[valsIdx];
660                   coords[1] = values[valsIdx + 1];
661                   coords[2] = values[valsIdx + 2];
662                   coords[3] = values[valsIdx + 3];
663                   coords[4] = values[valsIdx + 4];
664                   coords[5] = values[valsIdx + 5];
665                   break;
666               case SEG_ARCTO:
667                   coords[0] = values[valsIdx];
668                   coords[1] = values[valsIdx + 1];
669                   coords[2] = values[valsIdx + 2];
670                   coords[3] = values[valsIdx + 3];
671                   coords[4] = values[valsIdx + 4];
672                   coords[5] = values[valsIdx + 5];
673                   coords[6] = values[valsIdx + 6];
674                   break;
675               }
676               return ret;
677           }

687           public void next() {
688               final int type = types[segNum++];
689               switch (type) {
690               case SEG_CLOSE:
691                   break;
692               case SEG_MOVETO: // fallthrough is intended

696               case SEG_QUADTO:

705               }

712       public Object clone() {

736       /**
737        * Make sure, that the requested number of slots in vales[] are available. Must be called even for numValues = 0,
738        * because it is also used for initialization of those arrays.
739        * 
740        * @param numValues
741        *            number of requested coordinates
742        */
743       private void makeRoom(int numValues) {
744           if (values == null) {
745               values = new double[2 * numValues];
746               types = new int[2];
747               numVals = 0;
748               numSeg = 0;
749               return;
750           }
751   
752           final int newSize = numVals + numValues;
753           if (newSize > values.length) {
754               int nlen = values.length * 2;
755               if (nlen < newSize) {
756                   nlen = newSize;
757               }
758   
759               final double[] nvals = new double[nlen];
760               System.arraycopy(values, 0, nvals, 0, numVals);
761               values = nvals;
762           }
763   
764           if (numSeg == types.length) {
765               final int[] ntypes = new int[types.length * 2];
766               System.arraycopy(types, 0, ntypes, 0, types.length);
767               types = ntypes;
768           }
769       }
