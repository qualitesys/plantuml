66       public void add(double[] coord, USegmentType pathType) {

68       }

70       public boolean isEmpty() {
71           return segments.size() == 0;
72       }
73   
74       private void addInternal(USegment segment) {
75           segments.add(segment);
76           final double coord[] = segment.getCoord();
77           if (segment.getSegmentType() == USegmentType.SEG_ARCTO) {
78               minmax = minmax.addPoint(coord[5], coord[6]);
79               // minmax = minmax.addPoint(coord[5] + coord[0], coord[6] + coord[1]);
80               // minmax = minmax.addPoint(coord[5] - coord[0], coord[6] - coord[1]);
81           } else {
82               for (int i = 0; i < coord.length; i += 2) {
83                   minmax = minmax.addPoint(coord[i], coord[i + 1]);
84               }
85           }
86       }
87   
88       public UPath translate(double dx, double dy) {
89           final UPath result = new UPath(comment, codeLine);
90           for (USegment seg : segments) {
91               result.addInternal(seg.translate(dx, dy));
92           }
93           return result;
94       }
95   
96       public UPath rotate(double theta) {
97           final UPath result = new UPath(comment, codeLine);
98           for (USegment seg : segments) {
99               result.addInternal(seg.rotate(theta));
100           }
101           return result;
102       }
