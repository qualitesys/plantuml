260               for (String color : colors)
261                   if (isColorValid(color) == false)
262                       return false;
263

265           }

268               if (colors.length == 2)
269                   return new HColorScheme(build(colors[0]), build(colors[1]), null);
270

272           }

276       private Gradient gradientFromString(String s) {
277           final Matcher2 m = MyPattern.cmpile("[-\\\\|/]").matcher(s);
278           if (m.find()) {
279               final char sep = m.group(0).charAt(0);
280               final int idx = s.indexOf(sep);
281               final String s1 = s.substring(0, idx);
282               final String s2 = s.substring(idx + 1);
283               return new Gradient(s1, sep, s2);
284           }
285           return null;
286       }

288       private Automatic automaticFromString(String s) {
289           if (s.startsWith("#"))
290               s = s.substring(1);
291   
292           if (s.startsWith("?") == false)
293               return null;
294   
295           final int idx = s.indexOf(':');
296           if (idx != -1)
297               return new Automatic(s.substring(1).split(":"));
298   
299           return null;
300       }

307               return getColor(s);
308           } catch (NoSuchColorException e) {
309               assert false;

311           }
312       }

318       public HColor getColor(String s) throws NoSuchColorException {
319           if (isColorValid(Objects.requireNonNull(s)) == false)
320               throw new NoSuchColorException();
321   
322           final Automatic automatic = automaticFromString(s);
323           if (automatic != null)
324               return automatic.buildInternal();
325   
326           final Gradient gradient = gradientFromString(s);
327           if (gradient != null)
328               return gradient.buildInternal();
329   
330           if (s.equalsIgnoreCase("#transparent") || s.equalsIgnoreCase("transparent"))
331               s = "#00000000";
332   
333           return build(s);
334       }

336       private boolean isColorValid(String s) {
337           s = removeFirstDieseAndgoLowerCase(s);
338           final Automatic automatic = automaticFromString(s);
339           if (automatic != null)
340               return automatic.isValid();
341   
342           final Gradient gradient = gradientFromString(s);
343           if (gradient != null)
344               return gradient.isValid();
345   
346           if (s.matches("[0-9A-Fa-f]|[0-9A-Fa-f]{3}|[0-9A-Fa-f]{6}|[0-9A-Fa-f]{8}|automatic|transparent"))
347               return true;
348   
349           if (htmlNames.containsKey(s))
350               return true;
351   
352           return false;
353   
354       }

356       private HColor build(String s) {
357           s = removeFirstDieseAndgoLowerCase(s);
358           final Color color;
359           if (s.equalsIgnoreCase("transparent") || s.equalsIgnoreCase("background")) {
360               return HColors.none();
361           } else if (s.equalsIgnoreCase("automatic")) {
362               return new HColorAutomagic();
363           } else if (s.matches("[0-9A-Fa-f]")) {
364               s = "" + s.charAt(0) + s.charAt(0) + s.charAt(0) + s.charAt(0) + s.charAt(0) + s.charAt(0);
365               color = new Color(Integer.parseInt(s, 16));
366           } else if (s.matches("[0-9A-Fa-f]{3}")) {
367               s = "" + s.charAt(0) + s.charAt(0) + s.charAt(1) + s.charAt(1) + s.charAt(2) + s.charAt(2);
368               color = new Color(Integer.parseInt(s, 16));
369           } else if (s.matches("[0-9A-Fa-f]{6}")) {
370               color = new Color(Integer.parseInt(s, 16));
371           } else if (s.matches("[0-9A-Fa-f]{8}")) {
372               color = fromRGBa(s);
373           } else {
374               final String value = Objects.requireNonNull(htmlNames.get(s));
375               color = new Color(Integer.parseInt(value.substring(1), 16));
376           }
377           return HColors.simple(color);
378       }

380       private Color fromRGBa(String s) {
381           // https://forum.plantuml.net/11606/full-opacity-alpha-compositing-support-for-svg-and-png
382           if (s.length() != 8)
383               throw new IllegalArgumentException();
384   
385           final int red = Integer.parseInt(s.substring(0, 2), 16);
386           final int green = Integer.parseInt(s.substring(2, 4), 16);
387           final int blue = Integer.parseInt(s.substring(4, 6), 16);
388           final int alpha = Integer.parseInt(s.substring(6, 8), 16);
389           return new Color(red, green, blue, alpha);
390       }

392       private String removeFirstDieseAndgoLowerCase(String s) {
393           s = StringUtils.goLowerCase(s);
394           if (s.startsWith("#"))
395               s = s.substring(1);
396   
397           return s;
398       }
