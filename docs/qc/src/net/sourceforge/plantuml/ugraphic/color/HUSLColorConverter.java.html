37       private static List<double[]> getBounds(double L) {
38           ArrayList<double[]> result = new ArrayList<double[]>();
39   
40           double sub1 = Math.pow(L + 16, 3) / 1560896;
41           double sub2 = sub1 > epsilon ? sub1 : L / kappa;
42   
43           for (int c = 0; c < 3; ++c) {
44               double m1 = m[c][0];
45               double m2 = m[c][1];
46               double m3 = m[c][2];
47   
48               for (int t = 0; t < 2; ++t) {
49                   double top1 = (284517 * m1 - 94839 * m3) * sub2;
50                   double top2 = (838422 * m3 + 769860 * m2 + 731718 * m1) * L * sub2 - 769860 * t * L;
51                   double bottom = (632260 * m3 - 126452 * m2) * sub2 + 126452 * t;
52   
53                   result.add(new double[] { top1 / bottom, top2 / bottom });
54               }
55           }
56   
57           return result;
58       }

64       private static double distanceFromPole(double[] point) {

68       private static Length lengthOfRayUntilIntersect(double theta, double[] line) {
69           double length = line[1] / (Math.sin(theta) - line[0] * Math.cos(theta));
70   
71           return new Length(length);
72       }

84       private static double maxSafeChromaForL(double L) {
85           List<double[]> bounds = getBounds(L);
86           double min = Double.MAX_VALUE;
87   
88           for (int i = 0; i < 2; ++i) {
89               double m1 = bounds.get(i)[0];
90               double b1 = bounds.get(i)[1];
91               double[] line = new double[] { m1, b1 };
92   
93               double x = intersectLineLine(line, new double[] { -1 / m1, 0 });
94               double length = distanceFromPole(new double[] { x, b1 + x * m1 });
95   
96               min = Math.min(min, length);
97           }
98   
99           return min;
100       }

102       private static double maxChromaForLH(double L, double H) {
103           double hrad = H / 360 * Math.PI * 2;
104   
105           List<double[]> bounds = getBounds(L);
106           double min = Double.MAX_VALUE;
107   
108           for (double[] bound : bounds) {
109               Length length = lengthOfRayUntilIntersect(hrad, bound);
110               if (length.greaterEqualZero) {
111                   min = Math.min(min, length.length);
112               }
113           }
114   
115           return min;
116       }

179       private static double yToL(double Y) {
180           if (Y <= epsilon) {
181               return (Y / refY) * kappa;
182           } else {
183               return 116 * Math.pow(Y / refY, 1.0 / 3.0) - 16;
184           }
185       }

187       private static double lToY(double L) {
188           if (L <= 8) {
189               return refY * L / kappa;
190           } else {
191               return refY * Math.pow((L + 16) / 116, 3);
192           }
193       }

195       public static double[] xyzToLuv(double[] tuple) {
196           double X = tuple[0];
197           double Y = tuple[1];
198           double Z = tuple[2];
199   
200           double varU = (4 * X) / (X + (15 * Y) + (3 * Z));
201           double varV = (9 * Y) / (X + (15 * Y) + (3 * Z));
202   
203           double L = yToL(Y);
204   
205           if (L == 0) {
206               return new double[] { 0, 0, 0 };
207           }
208   
209           double U = 13 * L * (varU - refU);
210           double V = 13 * L * (varV - refV);
211   
212           return new double[] { L, U, V };
213       }

215       public static double[] luvToXyz(double[] tuple) {
216           double L = tuple[0];
217           double U = tuple[1];
218           double V = tuple[2];
219   
220           if (L == 0) {
221               return new double[] { 0, 0, 0 };
222           }
223   
224           double varU = U / (13 * L) + refU;
225           double varV = V / (13 * L) + refV;
226   
227           double Y = lToY(L);
228           double X = 0 - (9 * Y * varU) / ((varU - 4) * varV - varU * varV);
229           double Z = (9 * Y - (15 * varV * Y) - (varV * X)) / (3 * varV);
230   
231           return new double[] { X, Y, Z };
232       }

234       public static double[] luvToLch(double[] tuple) {
235           double L = tuple[0];
236           double U = tuple[1];
237           double V = tuple[2];
238   
239           double C = Math.sqrt(U * U + V * V);
240           double H;
241   
242           if (C < 0.00000001) {
243               H = 0;
244           } else {
245               double Hrad = Math.atan2(V, U);
246   
247               // pi to more digits than they provide it in the stdlib
248               H = (Hrad * 180.0) / 3.1415926535897932;
249   
250               if (H < 0) {
251                   H = 360 + H;
252               }
253           }
254   
255           return new double[] { L, C, H };
256       }
