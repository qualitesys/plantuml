49   public class LimitFinder extends UGraphicNo {
50   
51       @Override
52       public UGraphic apply(UChange change) {
53           final UTranslate tmp = change instanceof UTranslate ? this.getTranslate().compose((UTranslate) change)
54                   : this.getTranslate();
55           final LimitFinder result = new LimitFinder(this.getStringBounder(), tmp, this.minmax);
56           if (instanceOfAny(change, UAntiAliasing.class, UBackground.class, UClip.class, HColor.class, UHidden.class,
57                   UStroke.class, UTranslate.class, CopyForegroundColorToBackgroundColor.class) == false)
58               throw new UnsupportedOperationException(change.getClass().toString());
59           result.clip = change instanceof UClip ? ((UClip) change).translate(result.getTranslate()) : this.clip;
60           return result;
61       }
62   
63       private final MinMaxMutable minmax;
64       private UClip clip;
65   
66       public static LimitFinder create(StringBounder stringBounder, boolean initToZero) {
67           final LimitFinder result = new LimitFinder(stringBounder, new UTranslate(), MinMaxMutable.getEmpty(initToZero));
68           result.clip = null;
69           return result;
70       }
71   
72       private LimitFinder(StringBounder stringBounder, UTranslate translate, MinMaxMutable minmax) {
73           super(stringBounder, translate);
74           this.minmax = minmax;
75       }
76   
77       private void addPoint(double x, double y) {
78           if (clip == null || clip.isInside(x, y))
79               minmax.addPoint(x, y);
80   
81       }
82   
83       public void draw(UShape shape) {
84           final double x = getTranslate().getDx();
85           final double y = getTranslate().getDy();
86           if (shape instanceof UText) {
87               drawText(x, y, (UText) shape);
88           } else if (shape instanceof ULine) {
89               drawULine(x, y, (ULine) shape);
90           } else if (shape instanceof UEllipse) {
91               drawEllipse(x, y, (UEllipse) shape);
92           } else if (shape instanceof UPolygon) {
93               drawUPolygon(x, y, (UPolygon) shape);
94           } else if (shape instanceof UPath) {
95               drawUPath(x, y, (UPath) shape);
96           } else if (shape instanceof URectangle) {
97               drawRectangle(x, y, (URectangle) shape);
98           } else if (shape instanceof DotPath) {
99               drawDotPath(x, y, (DotPath) shape);
100           } else if (shape instanceof UImage) {
101               drawImage(x, y, (UImage) shape);
102           } else if (shape instanceof UImageSvg) {
103               drawImageSvg(x, y, (UImageSvg) shape);
104           } else if (shape instanceof UComment) {
105           } else if (shape instanceof UEmpty) {
106               drawEmpty(x, y, (UEmpty) shape);
107           } else if (shape instanceof TextBlock) {
108               final TextBlock tb = (TextBlock) shape;
109               tb.drawU(this);
110           } else if (shape instanceof UCenteredCharacter) {
111               // To be done
112           } else if (shape instanceof CenteredText) {
113               // Ignored
114           } else if (shape instanceof SpecialText) {
115               // Ignored
116           } else if (shape instanceof CopyForegroundColorToBackgroundColor) {
117               // Ignored
118           } else if (shape instanceof UPixel) {
119               addPoint(x, y);
120           } else {
121               throw new UnsupportedOperationException(shape.getClass().getName());
122           }
123       }
124   
125       private void drawEmpty(double x, double y, UEmpty shape) {
126           addPoint(x, y);
127           addPoint(x + shape.getWidth(), y + shape.getHeight());
128       }
129   
130       private void drawUPath(double x, double y, UPath shape) {
131           addPoint(x + shape.getMinX(), y + shape.getMinY());
132           addPoint(x + shape.getMaxX(), y + shape.getMaxY());
133       }
134   
135       private final static double HACK_X_FOR_POLYGON = 10;
136   
137       private void drawUPolygon(double x, double y, UPolygon shape) {
138           if (shape.getPoints().size() == 0) {
139               return;
140           }
141           addPoint(x + shape.getMinX() - HACK_X_FOR_POLYGON, y + shape.getMinY());
142           addPoint(x + shape.getMaxX() + HACK_X_FOR_POLYGON, y + shape.getMaxY());
143       }
144   
145       private void drawULine(double x, double y, ULine shape) {
146           addPoint(x, y);
147           addPoint(x + shape.getDX(), y + shape.getDY());
148       }
149   
150       private void drawRectangle(double x, double y, URectangle shape) {
151           addPoint(x - 1, y - 1);
152           addPoint(x + shape.getWidth() - 1 + shape.getDeltaShadow() * 2,
153                   y + shape.getHeight() - 1 + shape.getDeltaShadow() * 2);
154       }
155   
156       private void drawDotPath(double x, double y, DotPath shape) {
157           final MinMax shapeMinMax = shape.getMinMax();
158           addPoint(x + shapeMinMax.getMinX(), y + shapeMinMax.getMinY());
159           addPoint(x + shapeMinMax.getMaxX(), y + shapeMinMax.getMaxY());
160       }
161   
162       private void drawImage(double x, double y, UImage shape) {
163           addPoint(x, y);
164           addPoint(x + shape.getWidth() - 1, y + shape.getHeight() - 1);
165       }
166   
167       private void drawImageSvg(double x, double y, UImageSvg shape) {
168           addPoint(x, y);
169           addPoint(x + shape.getWidth() - 1, y + shape.getHeight() - 1);
170       }
171   
172       private void drawEllipse(double x, double y, UEllipse shape) {
173           addPoint(x, y);
174           addPoint(x + shape.getWidth() - 1 + shape.getDeltaShadow() * 2,
175                   y + shape.getHeight() - 1 + shape.getDeltaShadow() * 2);
176       }
177   
178       private void drawText(double x, double y, UText text) {
179           final XDimension2D dim = getStringBounder().calculateDimension(text.getFontConfiguration().getFont(),
180                   text.getText());
181           y -= dim.getHeight() - 1.5;
182           addPoint(x, y);
183           addPoint(x, y + dim.getHeight());
184           addPoint(x + dim.getWidth(), y);
185           addPoint(x + dim.getWidth(), y + dim.getHeight());
186       }
187   
188       public ColorMapper getColorMapper() {
189           return ColorMapper.IDENTITY;
190       }
191   
192       public double getMaxX() {
193           return minmax.getMaxX();
194       }
195   
196       public double getMaxY() {
197           return minmax.getMaxY();
198       }
199   
200       public double getMinX() {
201           return minmax.getMinX();
202       }
203   
204       public double getMinY() {
205           return minmax.getMinY();
206       }
207   
208       public MinMax getMinMax() {
209           if (minmax.isInfinity()) {
210               return MinMax.getEmpty(true);
211           }
212           return MinMax.fromMutable(minmax);
213       }
