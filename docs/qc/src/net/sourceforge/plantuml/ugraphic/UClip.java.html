37   import java.awt.geom.Line2D;
38   import java.awt.geom.Rectangle2D;

40   import net.sourceforge.plantuml.awt.geom.XPoint2D;

42   public class UClip implements UChange {
43   
44       private final double x;
45       private final double y;
46       private final double width;
47       private final double height;
48   
49       public UClip(double x, double y, double width, double height) {
50           this.x = x;
51           this.y = y;
52           this.width = width;
53           this.height = height;
54       }
55   
56       public UClip enlarge(double delta) {
57           return new UClip(x - delta, y - delta, width + 2 * delta, height + 2 * delta);
58       }
59   
60       @Override
61       public String toString() {
62           return "CLIP x=" + x + " y=" + y + " w=" + width + " h=" + height;
63       }
64   
65       public UClip translate(double dx, double dy) {
66           return new UClip(x + dx, y + dy, width, height);
67       }
68   
69       public UClip translate(UTranslate translate) {
70           return translate(translate.getDx(), translate.getDy());
71       }
72   
73       public final double getX() {
74           return x;
75       }
76   
77       public final double getY() {
78           return y;
79       }
80   
81       public final double getWidth() {
82           return width;
83       }
84   
85       public final double getHeight() {
86           return height;
87       }
88   
89       public boolean isInside(XPoint2D pt) {
90           return isInside(pt.getX(), pt.getY());
91       }
92   
93       public boolean isInside(double xp, double yp) {
94           if (xp < x) {
95               assert getClippedX(xp) != xp;
96               return false;
97           }
98           if (xp > x + width) {
99               assert getClippedX(xp) != xp;
100               return false;
101           }
102           if (yp < y) {
103               assert getClippedY(yp) != yp;
104               return false;
105           }
106           if (yp > y + height) {
107               assert getClippedY(yp) != yp;
108               return false;
109           }
110           assert getClippedX(xp) == xp;
111           assert getClippedY(yp) == yp;
112           return true;
113       }
114   
115       public Rectangle2D.Double getClippedRectangle(Rectangle2D.Double r) {
116           return (Rectangle2D.Double) r.createIntersection(new Rectangle2D.Double(x, y, width, height));
117       }
118   
119       public Line2D.Double getClippedLine(Line2D.Double line) {
120           if (isInside(line.x1, line.y1) && isInside(line.x2, line.y2)) {
121               return line;
122           }
123           if (isInside(line.x1, line.y1) == false && isInside(line.x2, line.y2) == false) {
124               if (line.x1 == line.x2) {
125                   final double newy1 = getClippedY(line.y1);
126                   final double newy2 = getClippedY(line.y2);
127                   if (newy1 != newy2) {
128                       return new Line2D.Double(line.x1, newy1, line.x2, newy2);
129                   }
130               }
131               return null;
132           }
133           if (line.x1 != line.x2 && line.y1 != line.y2) {
134               return null;
135           }
136           assert line.x1 == line.x2 || line.y1 == line.y2;
137           if (line.y1 == line.y2) {
138               final double newx1 = getClippedX(line.x1);
139               final double newx2 = getClippedX(line.x2);
140               return new Line2D.Double(newx1, line.y1, newx2, line.y2);
141           }
142           if (line.x1 == line.x2) {
143               final double newy1 = getClippedY(line.y1);
144               final double newy2 = getClippedY(line.y2);
145               return new Line2D.Double(line.x1, newy1, line.x2, newy2);
146           }
147           throw new IllegalStateException();
148       }
149   
150       private double getClippedX(double xp) {
151           if (xp < x) {
152               return x;
153           }
154           if (xp > x + width) {
155               return x + width;
156           }
157           return xp;
158       }
159   
160       private double getClippedY(double yp) {
161           if (yp < y) {
162               return y;
163           }
164           if (yp > y + height) {
165               return y + height;
166           }
167           return yp;
168       }
169   
170       public boolean isInside(double x, double y, UPath shape) {
171           return isInside(x + shape.getMinX(), y + shape.getMinY()) && isInside(x + shape.getMaxX(), y + shape.getMaxY());
172       }

174   }
