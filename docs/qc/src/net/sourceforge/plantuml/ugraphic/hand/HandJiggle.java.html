38   import java.util.Collection;
39   import java.util.Random;

46   public class HandJiggle {
47       private final Collection<XPoint2D> points = new ArrayList<>();
48   
49       private double startX;
50       private double startY;
51       private final double defaultVariation;
52       private final Random rnd;
53   
54       private double randomMe() {
55           return rnd.nextDouble();
56       }
57   
58       public HandJiggle(double startX, double startY, double defaultVariation, Random rnd) {
59           this.startX = startX;
60           this.startY = startY;
61           this.defaultVariation = defaultVariation;
62           this.rnd = rnd;
63           points.add(new XPoint2D(startX, startY));
64       }
65   
66       public static HandJiggle create(XPoint2D start, double defaultVariation, Random rnd) {
67           return new HandJiggle(start.getX(), start.getY(), defaultVariation, rnd);
68       }
69   
70       public void lineTo(XPoint2D end) {
71           lineTo(end.getX(), end.getY());
72       }
73   
74       public void arcTo(double angle0, double angle1, double centerX, double centerY, double rx, double ry) {
75           lineTo(pointOnCircle(centerX, centerY, (angle0 + angle1) / 2, rx, ry));
76           lineTo(pointOnCircle(centerX, centerY, angle1, rx, ry));
77       }
78   
79       private static XPoint2D pointOnCircle(double centerX, double centerY, double angle, double rx, double ry) {
80           final double x = centerX + Math.cos(angle) * rx;
81           final double y = centerY + Math.sin(angle) * ry;
82           return new XPoint2D(x, y);
83   
84       }
85   
86       public void lineTo(final double endX, final double endY) {
87   
88           final double diffX = Math.abs(endX - startX);
89           final double diffY = Math.abs(endY - startY);
90           final double distance = Math.sqrt(diffX * diffX + diffY * diffY);
91           if (distance < 0.001)
92               return;
93   
94           int segments = (int) Math.round(distance / 10);
95           double variation = defaultVariation;
96           if (segments < 5) {
97               segments = 5;
98               variation /= 3;
99           }
100   
101           final double stepX = Math.signum(endX - startX) * diffX / segments;
102           final double stepY = Math.signum(endY - startY) * diffY / segments;
103   
104           final double fx = diffX / distance;
105           final double fy = diffY / distance;
106   
107           for (int s = 0; s < segments; s++) {
108               double x = stepX * s + startX;
109               double y = stepY * s + startY;
110   
111               final double offset = (randomMe() - 0.5) * variation;
112               points.add(new XPoint2D(x - offset * fy, y - offset * fx));
113           }
114           points.add(new XPoint2D(endX, endY));
115   
116           this.startX = endX;
117           this.startY = endY;
118       }
119   
120       public void curveTo(XCubicCurve2D curve) {
121           final double flatness = curve.getFlatness();
122           final double dist = curve.getP1().distance(curve.getP2());
123           if (flatness > 0.1 && dist > 20) {
124               final XCubicCurve2D left = new XCubicCurve2D();
125               final XCubicCurve2D right = new XCubicCurve2D();
126               curve.subdivide(left, right);
127               curveTo(left);
128               curveTo(right);
129               return;
130           }
131           lineTo(curve.getP2());
132       }

134       public UPolygon toUPolygon() {
135           final UPolygon result = new UPolygon();
136           for (XPoint2D p : points)
137               result.addPoint(p.getX(), p.getY());
138   
139           return result;
140       }
141   
142       public UPath toUPath() {
143           UPath path = null;
144           for (XPoint2D p : points) {
145               if (path == null) {
146                   path = new UPath();
147                   path.moveTo(p);
148               } else {
149                   path.lineTo(p);
150               }
151           }
152           if (path == null)
153               throw new IllegalStateException();
154   
155           return path;
156       }
157   
158       public void appendTo(UPath result) {
159           for (XPoint2D p : points)
160               result.lineTo(p);
161   
162       }

164   }
