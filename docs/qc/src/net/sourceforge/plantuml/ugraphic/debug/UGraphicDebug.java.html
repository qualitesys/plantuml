111       public void draw(UShape shape) {
112           if (shape instanceof ULine) {
113               outLine((ULine) shape);
114           } else if (shape instanceof URectangle) {
115               outRectangle((URectangle) shape);
116           } else if (shape instanceof UText) {
117               outText((UText) shape);
118           } else if (shape instanceof UPolygon) {
119               outPolygon((UPolygon) shape);
120           } else if (shape instanceof UEllipse) {
121               outEllipse((UEllipse) shape);
122           } else if (shape instanceof UEmpty) {
123               outEmpty((UEmpty) shape);
124           } else if (shape instanceof UPath) {
125               outPath((UPath) shape);
126           } else if (shape instanceof UComment) {
127               outComment((UComment) shape);
128           } else if (shape instanceof DotPath) {
129               outPath(((DotPath) shape).toUPath());
130           } else if (shape instanceof UCenteredCharacter) {
131               outCenteredCharacter(((UCenteredCharacter) shape));
132           } else {
133               System.err.println("UGraphicDebug " + shape.getClass().getSimpleName());
134               output.add("UGraphicDebug " + shape.getClass().getSimpleName() + " " + new Date());
135           }
136       }

142           output.add("  font: " + shape.getFont().toStringDebug());
143           output.add("  color: " + colorToString(getParam().getColor()));
144           output.add("");
145   
146       }

152       private void outPath(UPath shape) {
153           output.add("PATH:");
154           for (USegment seg : shape) {
155               final USegmentType type = seg.getSegmentType();
156               final double coord[] = seg.getCoord();
157               output.add("   - type: " + type);
158               if (type == USegmentType.SEG_ARCTO) {
159                   output.add("     radius: " + pointd(coord[0], coord[1]));
160                   output.add("     angle: " + coord[2]);
161                   output.add("     largeArcFlag: " + (coord[3] != 0));
162                   output.add("     sweepFlag: " + (coord[4] != 0));
163                   output.add("     dest: " + pointd(coord[5], coord[6]));
164               } else
165                   for (int i = 0; i < type.getNbPoints(); i++) {
166                       final String key = "     pt" + (i + 1) + ": ";
167                       output.add(key + pointd(coord[2 * i], coord[2 * i + 1]));
168                   }
169           }
170   
171           output.add("  stroke: " + getParam().getStroke());
172           output.add("  shadow: " + (int) shape.getDeltaShadow());
173           output.add("  color: " + colorToString(getParam().getColor()));
174           output.add("  backcolor: " + colorToString(getParam().getBackcolor()));
175           output.add("");
176   
177       }

179       private void outPolygon(UPolygon shape) {
180           output.add("POLYGON:");
181           output.add("  points:");
182           for (XPoint2D pt : shape.getPoints()) {
183               final double xp = getTranslateX() + pt.getX();
184               final double yp = getTranslateY() + pt.getY();
185               output.add("   - " + pointd(xp, yp));
186           }
187           output.add("  stroke: " + getParam().getStroke());
188           output.add("  shadow: " + (int) shape.getDeltaShadow());
189           output.add("  color: " + colorToString(getParam().getColor()));
190           output.add("  backcolor: " + colorToString(getParam().getBackcolor()));
191           output.add("");
192   
193       }

253       private String pointd(double x, double y) {
254           return String.format(Locale.US, "[ %.4f ; %.4f ]", x, y);
255       }

257       private String colorToString(HColor color) {
258           if (color == null || color.isTransparent())
259               return "NULL_COLOR";
260   
261           if (color instanceof HColorSimple) {
262               final HColorSimple simple = (HColorSimple) color;
263               final Color internal = simple.getAwtColor();
264   
265               return Integer.toHexString(internal.getRGB());
266           }
267           if (color instanceof HColorMiddle) {
268               final HColorMiddle middle = (HColorMiddle) color;
269               return "middle(" + colorToString(middle.getColor1()) + " & " + colorToString(middle.getColor1()) + " )";
270           }
271           System.err.println("Error colorToString " + color.getClass().getSimpleName());
272           return color.getClass().getSimpleName() + " " + new Date();
273       }

275       @Override
276       public void writeToStream(OutputStream os, String metadata, int dpi) throws IOException {
277           print(os, "DPI: " + dpi);
278           print(os, "dimension: " + pointd(dim.getWidth(), dim.getHeight()));
279           print(os, "scaleFactor: " + String.format(Locale.US, "%.4f", scaleFactor));
280           print(os, "seed: " + seed);
281           print(os, "svgLinkTarget: " + svgLinkTarget);
282           print(os, "hoverPathColorRGB: " + hoverPathColorRGB);
283           print(os, "preserveAspectRatio: " + preserveAspectRatio);
284           print(os, "");
285   
286           for (String s : output)
287               print(os, s);
288   
289           os.flush();
290       }

292       private void print(OutputStream os, String out) throws UnsupportedEncodingException, IOException {
293           os.write(out.getBytes(UTF_8));
294           os.write("\n".getBytes(UTF_8));
295       }
