100       public static FontConfiguration create(ISkinParam skinParam, Style style, Colors colors) {
101           final HColor hyperlinkColor = style.value(PName.HyperLinkColor).asColor(skinParam.getIHtmlColorSet());
102           final UStroke hyperlinkUnderlineStroke = skinParam.useUnderlineForHyperlink();
103           HColor color = colors == null ? null : colors.getColor(ColorType.TEXT);
104           if (color == null)
105               color = style.value(PName.FontColor).asColor(skinParam.getIHtmlColorSet());
106           return create(style.getUFont(), color, hyperlinkColor, hyperlinkUnderlineStroke, skinParam.getTabSize());
107       }

111       private static EnumSet<FontStyle> getStyles(UFont font) {
112           final boolean bold = font.isBold();
113           final boolean italic = font.isItalic();
114           if (bold && italic)
115               return EnumSet.of(FontStyle.ITALIC, FontStyle.BOLD);
116   
117           if (bold)
118               return EnumSet.of(FontStyle.BOLD);
119   
120           if (italic)
121               return EnumSet.of(FontStyle.ITALIC);
122   
123           return EnumSet.noneOf(FontStyle.class);
124       }

147       public FontConfiguration forceFont(UFont newFont, HColor htmlColorForStereotype) {
148           if (newFont == null)
149               return add(FontStyle.ITALIC);
150   
151           FontConfiguration result = new FontConfiguration(styles, newFont, motherColor, newFont, currentColor,
152                   extendedColor, fontPosition, svgAttributes, hyperlinkColor, hyperlinkUnderlineStroke, tabSize);
153           if (htmlColorForStereotype != null)
154               result = result.changeColor(htmlColorForStereotype);
155   
156           return result;
157       }
