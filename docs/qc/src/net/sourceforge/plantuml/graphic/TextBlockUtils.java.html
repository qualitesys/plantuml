151       public static boolean isEmpty(TextBlock text, StringBounder dummyStringBounder) {
152           if (text == null) {
153               return true;
154           }
155           final XDimension2D dim = text.calculateDimension(dummyStringBounder);
156           return dim.getHeight() == 0 && dim.getWidth() == 0;
157       }
