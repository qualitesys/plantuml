59       private Boolean shadowing = null;

75           final Colors result = new Colors();
76           result.map.putAll(this.map);
77           result.lineStyle = this.lineStyle;
78           return result;

84       public Colors(String data, HColorSet set, ColorType mainType) throws NoSuchColorException {
85           data = StringUtils.goLowerCase(data);
86   
87           for (final StringTokenizer st = new StringTokenizer(data, "#;"); st.hasMoreTokens();) {
88               final String s = st.nextToken();
89               final int x = s.indexOf(':');
90               if (x == -1) {
91                   if (s.contains(".") == false)
92                       map.put(mainType, set.getColor(s));
93   
94               } else {
95                   final String name = s.substring(0, x);
96                   final String value = s.substring(x + 1);
97                   if (name.equalsIgnoreCase("shadowing")) {
98                       this.shadowing = value.equalsIgnoreCase("true");
99                   } else {
100                       final ColorType key = ColorType.getType(name);
101                       final HColor color = set.getColor(value);
102                       map.put(key, color);
103                   }
104               }
105           }
106           if (data.contains("line.dashed"))
107               lineStyle = LinkStyle.DASHED();
108           else if (data.contains("line.dotted"))
109               lineStyle = LinkStyle.DOTTED();
110           else if (data.contains("line.bold"))
111               lineStyle = LinkStyle.BOLD();
112   
113       }
114   
115       public HColor getColor(ColorType key) {
116           return map.get(Objects.requireNonNull(key));
117       }
118   
119       public HColor getColor(ColorType key1, ColorType key2) {
120           final HColor result = getColor(key1);
121           if (result != null)
122               return result;
123   
124           return getColor(key2);
125       }

127       public UStroke getSpecificLineStroke() {
128           if (lineStyle == null)
129               return null;
130   
131           return lineStyle.getStroke3();
132       }

134       public Colors add(ColorType type, HColor color) {
135           if (color == null)
136               return this;
137   
138           final Colors result = copy();
139           result.map.put(type, color);
140           return result;
141       }
142   
143       private Colors add(ColorType colorType, Colors other) {
144           final Colors result = copy();
145           result.map.putAll(other.map);
146           if (other.lineStyle != null)
147               result.lineStyle = other.lineStyle;
148   
149           return result;
150       }

154       }

157           return new SkinParamColors(skinParam, this);
158       }
159   
160       public Colors addLegacyStroke(String s) {
161           final Colors result = copy();
162           result.lineStyle = LinkStyle.fromString1(StringUtils.goUpperCase(Objects.requireNonNull(s)));
163           return result;
164   
165       }

167       public static UGraphic applyStroke(UGraphic ug, Colors colors) {
168           if (colors == null) 
169               return ug;
170           
171           if (colors.lineStyle == null) 
172               return ug;
173           
174           return ug.apply(colors.lineStyle.getStroke3());
175       }
176   
177       public Colors applyStereotype(Stereotype stereotype, ISkinParam skinParam, ColorParam param)
178               throws NoSuchColorException {
179           Objects.requireNonNull(stereotype);
180           final ColorType colorType = Objects.requireNonNull(Objects.requireNonNull(param).getColorType());
181           if (getColor(colorType) != null)
182               return this;
183   
184           final Colors colors = skinParam.getColors(param, stereotype);
185           return add(colorType, colors);
186       }
187   
188       public Colors applyStereotypeForNote(Stereotype stereotype, ISkinParam skinParam, ColorParam... params)
189               throws NoSuchColorException {
190           Objects.requireNonNull(stereotype);
191           Colors result = this;
192           for (ColorParam param : Objects.requireNonNull(params))
193               result = result.applyStereotype(stereotype, skinParam, param);
194   
195           result.shadowing = skinParam.shadowingForNote(stereotype);
196           return result;
197       }

199       public Boolean getShadowing() {
200           return shadowing;
201       }

204           if (lineStyle == null)
205               return stroke;
206   
207           return lineStyle.muteStroke(stroke);
208       }
