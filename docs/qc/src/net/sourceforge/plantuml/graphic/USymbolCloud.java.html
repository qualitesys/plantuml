64       private void drawCloud(UGraphic ug, double width, double height, double shadowing) {
65           final UPath shape = getSpecificFrontierForCloud(width, height);
66           shape.setDeltaShadow(shadowing);
67   
68           ug.apply(UTranslate.dy(0)).draw(shape);
69       }

71       private UPath getSpecificFrontierForCloudNew(double width, double height) {
72           final Random rnd = new Random((long) width + 7919L * (long) height);
73           final List<XPoint2D> points = new ArrayList<>();
74   
75           double bubbleSize = 11;
76           if (Math.max(width, height) / bubbleSize > 16) {
77               bubbleSize = Math.max(width, height) / 16;
78           }
79   
80           final double margin1 = 8;
81   
82           final XPoint2D pointA = new XPoint2D(margin1, margin1);
83           final XPoint2D pointB = new XPoint2D(width - margin1, margin1);
84           final XPoint2D pointC = new XPoint2D(width - margin1, height - margin1);
85           final XPoint2D pointD = new XPoint2D(margin1, height - margin1);
86   
87           if (width > 100 && height > 100) {
88               complex(rnd, points, bubbleSize, pointA, pointB, pointC, pointD);
89           } else {
90               simple(rnd, points, bubbleSize, pointA, pointB, pointC, pointD);
91           }
92   
93           points.add(points.get(0));
94   
95           final UPath result = new UPath();
96           result.moveTo(points.get(0));
97           for (int i = 0; i < points.size() - 1; i++) {
98               if (DEBUG) {
99                   result.lineTo(points.get(i + 1));
100               } else {
101                   addCurve(rnd, result, points.get(i), points.get(i + 1));
102               }
103           }
104           return result;
105   
106       }

108       private void complex(final Random rnd, final List<XPoint2D> points, double bubbleSize, final XPoint2D pointA,
109               final XPoint2D pointB, final XPoint2D pointC, final XPoint2D pointD) {
110           final double margin2 = 7;
111           specialLine(bubbleSize, rnd, points, mvX(pointA, margin2), mvX(pointB, -margin2));
112           points.add(mvY(pointB, margin2));
113           specialLine(bubbleSize, rnd, points, mvY(pointB, margin2), mvY(pointC, -margin2));
114           points.add(mvX(pointC, -margin2));
115           specialLine(bubbleSize, rnd, points, mvX(pointC, -margin2), mvX(pointD, margin2));
116           points.add(mvY(pointD, -margin2));
117           specialLine(bubbleSize, rnd, points, mvY(pointD, -margin2), mvY(pointA, margin2));
118           points.add(mvX(pointA, margin2));
119       }

121       private void simple(final Random rnd, final List<XPoint2D> points, double bubbleSize, final XPoint2D pointA,
122               final XPoint2D pointB, final XPoint2D pointC, final XPoint2D pointD) {
123           specialLine(bubbleSize, rnd, points, pointA, pointB);
124           specialLine(bubbleSize, rnd, points, pointB, pointC);
125           specialLine(bubbleSize, rnd, points, pointC, pointD);
126           specialLine(bubbleSize, rnd, points, pointD, pointA);
127       }

129       private static XPoint2D mvX(XPoint2D pt, double dx) {
130           return new XPoint2D(pt.getX() + dx, pt.getY());
131       }

133       private static XPoint2D mvY(XPoint2D pt, double dy) {
134           return new XPoint2D(pt.getX(), pt.getY() + dy);
135       }

137       private void specialLine(double bubbleSize, Random rnd, List<XPoint2D> points, XPoint2D p1, XPoint2D p2) {
138           final CoordinateChange change = CoordinateChange.create(p1, p2);
139           final double length = change.getLength();
140           final XPoint2D middle = change.getTrueCoordinate(length / 2, -rnd(rnd, 1, 1 + Math.min(12, bubbleSize * 0.8)));
141           // final XPoint2D middle = change.getTrueCoordinate(length / 2, -13);
142           if (DEBUG) {
143               points.add(middle);
144               points.add(p2);
145           } else {
146               bubbleLine(rnd, points, p1, middle, bubbleSize);
147               bubbleLine(rnd, points, middle, p2, bubbleSize);
148           }
149       }
150   
151       private void bubbleLine(Random rnd, List<XPoint2D> points, XPoint2D p1, XPoint2D p2, double bubbleSize) {
152           final CoordinateChange change = CoordinateChange.create(p1, p2);
153           final double length = change.getLength();
154           int nb = (int) (length / bubbleSize);
155           if (nb == 0) {
156               bubbleSize = length / 2;
157               nb = (int) (length / bubbleSize);
158           }
159           for (int i = 0; i < nb; i++) {
160               points.add(rnd(rnd, change.getTrueCoordinate(i * length / nb, 0), bubbleSize * .2));
161           }
162       }
163   
164       private void addCurve(Random rnd, UPath path, XPoint2D p1, XPoint2D p2) {
165           final CoordinateChange change = CoordinateChange.create(p1, p2);
166           final double length = change.getLength();
167           final double coef = rnd(rnd, .25, .35);
168           final XPoint2D middle = change.getTrueCoordinate(length * coef, -length * rnd(rnd, .4, .55));
169           final XPoint2D middle2 = change.getTrueCoordinate(length * (1 - coef), -length * rnd(rnd, .4, .55));
170           path.cubicTo(middle, middle2, p2);
171   
172       }
173   
174       static private double rnd(Random rnd, double a, double b) {
175           return rnd.nextDouble() * (b - a) + a;
176       }
177   
178       static private XPoint2D rnd(Random rnd, XPoint2D pt, double v) {
179           final double x = pt.getX() + v * rnd.nextDouble();
180           final double y = pt.getY() + v * rnd.nextDouble();
181           return new XPoint2D(x, y);
182       }

184       private UPath getSpecificFrontierForCloud(double width, double height) {
185           if (NEW) {
186               return getSpecificFrontierForCloudNew(width, height);
187           }
188           final UPath path = new UPath();
189           path.moveTo(0, 10);
190           double x = 0;
191           for (int i = 0; i < width - 9; i += 10) {
192               path.cubicTo(i, -3 + 10, 2 + i, -5 + 10, 5 + i, -5 + 10);
193               path.cubicTo(8 + i, -5 + 10, 10 + i, -3 + 10, 10 + i, 10);
194               x = i + 10;
195           }
196           double y = 0;
197           for (int j = 10; j < height - 9; j += 10) {
198               path.cubicTo(x + 3, j, x + 5, 2 + j, x + 5, 5 + j);
199               path.cubicTo(x + 5, 8 + j, x + 3, 10 + j, x, 10 + j);
200               y = j + 10;
201           }
202           for (int i = 0; i < width - 9; i += 10) {
203               path.cubicTo(x - i, y + 3, x - 3 - i, y + 5, x - 5 - i, y + 5);
204               path.cubicTo(x - 8 - i, y + 5, x - 10 - i, y + 3, x - 10 - i, y);
205           }
206           for (int j = 0; j < height - 9 - 10; j += 10) {
207               path.cubicTo(-3, y - j, -5, y - 2 - j, -5, y - 5 - j);
208               path.cubicTo(-5, y - 8 - j, -3, y - 10 - j, 0, y - 10 - j);
209           }
210           return path;
211       }

221               final SymbolContext symbolContext, final HorizontalAlignment stereoAlignment) {
222           return new AbstractTextBlock() {

224               public void drawU(UGraphic ug) {
225                   final XDimension2D dim = calculateDimension(ug.getStringBounder());
226                   ug = UGraphicStencil.create(ug, dim);
227                   ug = symbolContext.apply(ug);
228                   drawCloud(ug, dim.getWidth(), dim.getHeight(), symbolContext.getDeltaShadow());
229                   final Margin margin = getMargin();
230                   final TextBlock tb = TextBlockUtils.mergeTB(stereotype, label, HorizontalAlignment.CENTER);
