36   package net.sourceforge.plantuml.graphic;
37   
38   import net.sourceforge.plantuml.awt.geom.XDimension2D;
39   import net.sourceforge.plantuml.creole.SheetBlock2;
40   import net.sourceforge.plantuml.style.ClockwiseTopRightBottomLeft;
41   import net.sourceforge.plantuml.ugraphic.Shadowable;
42   import net.sourceforge.plantuml.ugraphic.UGraphic;
43   import net.sourceforge.plantuml.ugraphic.URectangle;
44   import net.sourceforge.plantuml.ugraphic.UStroke;
45   import net.sourceforge.plantuml.ugraphic.UTranslate;
46   import net.sourceforge.plantuml.ugraphic.color.HColor;
47   import net.sourceforge.plantuml.ugraphic.color.HColors;
48   
49   public class TextBlockBordered extends AbstractTextBlock implements TextBlock {
50   
51       private final double cornersize;
52       private final HColor backgroundColor;
53       private final HColor borderColor;
54       private final double top;
55       private final double right;
56       private final double bottom;
57       private final double left;
58       private final UStroke stroke;
59       private final boolean withShadow;
60       private final String id;
61       private final TextBlock textBlock;
62   
63       TextBlockBordered(TextBlock textBlock, UStroke stroke, HColor borderColor, HColor backgroundColor,
64               double cornersize, ClockwiseTopRightBottomLeft margins, String id) {
65           this.top = margins.getTop();
66           this.right = margins.getRight();
67           this.bottom = margins.getBottom();
68           this.left = margins.getLeft();
69           this.cornersize = cornersize;
70           this.textBlock = textBlock;
71           this.withShadow = false;
72           this.stroke = stroke;
73           this.borderColor = borderColor;
74           this.backgroundColor = backgroundColor;
75           this.id = id;
76       }
77   
78       private double getTextHeight(StringBounder stringBounder) {
79           final XDimension2D size = textBlock.calculateDimension(stringBounder);
80           return size.getHeight() + top + bottom;
81       }
82   
83       private double getPureTextWidth(StringBounder stringBounder) {
84           final XDimension2D size = textBlock.calculateDimension(stringBounder);
85           return size.getWidth();
86       }
87   
88       private double getTextWidth(StringBounder stringBounder) {
89           return getPureTextWidth(stringBounder) + left + right;
90       }
91   
92       public XDimension2D calculateDimension(StringBounder stringBounder) {
93           final double height = getTextHeight(stringBounder);
94           final double width = getTextWidth(stringBounder);
95           return new XDimension2D(width + 1, height + 1);
96       }
97   
98       private UGraphic applyStroke(UGraphic ug) {
99           if (stroke == null)
100               return ug;
101   
102           return ug.apply(stroke);
103       }
104   
105       private boolean noBorder() {
106           if (stroke == null)
107               return false;
108   
109           return stroke.getThickness() == 0;
110       }
111   
112       public void drawU(UGraphic ug) {
113           final StringBounder stringBounder = ug.getStringBounder();
114           final Shadowable polygon = getPolygonNormal(stringBounder);
115           final UGraphic ugOriginal = ug;
116           if (withShadow)
117               polygon.setDeltaShadow(4);
118   
119           if (backgroundColor == null)
120               ug = ug.apply(HColors.none().bg());
121           else
122               ug = ug.apply(backgroundColor.bg());
123   
124           HColor color = noBorder() ? backgroundColor : borderColor;
125           if (color == null)
126               color = HColors.none();
127   
128           ug = ug.apply(color);
129           ug = applyStroke(ug);
130           ug.draw(polygon);
131           TextBlock toDraw = textBlock;
132           if (textBlock instanceof SheetBlock2)
133               toDraw = ((SheetBlock2) textBlock).enlargeMe(left, right);
134   
135           toDraw.drawU(ugOriginal.apply(color).apply(new UTranslate(left, top)));
136       }
137   
138       private Shadowable getPolygonNormal(final StringBounder stringBounder) {
139           final double height = getTextHeight(stringBounder);
140           final double width = getTextWidth(stringBounder);
141           return new URectangle(width, height).rounded(cornersize).withCommentAndCodeLine(id, null);
142       }
143   
144   }
