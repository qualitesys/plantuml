38   import java.util.List;
39   
40   import net.sourceforge.plantuml.FileFormat;
41   import net.sourceforge.plantuml.awt.geom.XDimension2D;
42   import net.sourceforge.plantuml.graphic.StringBounder;
43   import net.sourceforge.plantuml.skin.Area;
44   import net.sourceforge.plantuml.skin.ComponentType;
45   import net.sourceforge.plantuml.skin.Context2D;
46   import net.sourceforge.plantuml.ugraphic.UGraphic;
47   import net.sourceforge.plantuml.ugraphic.txt.UGraphicTxt;
48   
49   public class ComponentTextGroupingBody extends AbstractComponentText {
50   
51       private final ComponentType type;
52       private final List<? extends CharSequence> stringsToDisplay;
53       private final FileFormat fileFormat;
54   
55       public ComponentTextGroupingBody(ComponentType type, List<? extends CharSequence> stringsToDisplay,
56               FileFormat fileFormat) {
57           this.type = type;
58           this.stringsToDisplay = stringsToDisplay;
59           this.fileFormat = fileFormat;
60       }
61   
62       public void drawU(UGraphic ug, Area area, Context2D context) {
63           final XDimension2D dimensionToUse = area.getDimensionToUse();
64           final UmlCharArea charArea = ((UGraphicTxt) ug).getCharArea();
65           final int width = (int) dimensionToUse.getWidth();
66           final int height = (int) dimensionToUse.getHeight();
67   
68           if (fileFormat == FileFormat.UTXT) {
69               charArea.drawVLine('\u2551', 0, 0, height);
70               charArea.drawVLine('\u2551', width - 1, 0, height);
71               charArea.drawHLine('\u2550', height - 1, 1, width - 1, '\u2502', '\u256a');
72               charArea.drawChar('\u2560', 0, height - 1);
73               charArea.drawChar('\u2563', width - 1, height - 1);
74           } else {
75               charArea.drawVLine('!', 0, 0, height);
76               charArea.drawVLine('!', width - 1, 0, height);
77               charArea.drawHLine('~', height - 1, 1, width - 1);
78           }
79       }
80   
81       public double getPreferredHeight(StringBounder stringBounder) {
82           return 1;
83       }
84   
85       public double getPreferredWidth(StringBounder stringBounder) {
86           return 1;
87       }
88   
89   }
