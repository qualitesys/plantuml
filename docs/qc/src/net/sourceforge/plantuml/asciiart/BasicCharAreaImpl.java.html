38   import java.io.PrintStream;

45   public class BasicCharAreaImpl implements BasicCharArea {
46   
47       private int charSize1 = 160;
48       private int charSize2 = 160;
49   
50       private int width;
51       private int height;
52   
53       private char chars[][];
54   
55       public BasicCharAreaImpl() {
56           this.chars = new char[charSize1][charSize2];
57           for (int i = 0; i < charSize1; i++) {
58               for (int j = 0; j < charSize2; j++) {
59                   chars[i][j] = ' ';
60               }
61           }
62       }
63   
64       public final int getWidth() {
65           return width;
66       }
67   
68       public final int getHeight() {
69           return height;
70       }
71   
72       public void drawChar(char c, int x, int y) {
73           ensurePossible(x, y);
74           chars[x][y] = c;
75           if (x >= width) {
76               width = x + 1;
77           }
78           if (y >= height) {
79               height = y + 1;
80           }
81       }
82   
83       private boolean isLong(char c) {
84           final int wc = Wcwidth.of(c);
85           if (wc == 1) {
86               return false;
87           }
88           if (wc == 2) {
89               return true;
90           }
91           return false;
92   //        throw new IllegalArgumentException("warning width=" + wc + " char=" + ((int) c));
93       }
94   
95       private void ensurePossible(int x, int y) {
96           int newCharSize1 = charSize1;
97           int newCharSize2 = charSize2;
98           while (x >= newCharSize1) {
99               newCharSize1 *= 2;
100           }
101           while (y >= newCharSize2) {
102               newCharSize2 *= 2;
103           }
104           if (newCharSize1 != charSize1 || newCharSize2 != charSize2) {
105               final char newChars[][] = new char[newCharSize1][newCharSize2];
106               for (int i = 0; i < newCharSize1; i++) {
107                   for (int j = 0; j < newCharSize2; j++) {
108                       char c = ' ';
109                       if (i < charSize1 && j < charSize2) {
110                           c = chars[i][j];
111                       }
112                       newChars[i][j] = c;
113                   }
114               }
115               this.chars = newChars;
116               this.charSize1 = newCharSize1;
117               this.charSize2 = newCharSize2;
118           }
119   
120       }
121   
122       public void drawStringLR(String string, int x, int y) {
123           int pos = x;
124           for (int i = 0; i < string.length(); i++) {
125               final char c = string.charAt(i);
126               drawChar(c, pos, y);
127               pos++;
128               if (isLong(c)) {
129                   drawChar('\0', pos, y);
130                   pos++;
131               }
132           }
133       }
134   
135       public void drawStringTB(String string, int x, int y) {
136           for (int i = 0; i < string.length(); i++) {
137               drawChar(string.charAt(i), x, y + i);
138           }
139       }
140   
141       public String getLine(int line) {
142           final StringBuilder sb = new StringBuilder(charSize1);
143           for (int x = 0; x < width; x++) {
144               final char c = chars[x][line];
145               if (c != '\0') {
146                   StringUtils.appendInternalToRealBoldNumber(sb, c);
147               }
148           }
149           return sb.toString();
150       }
151   
152       public void print(PrintStream ps) {
153           for (String s : getLines()) {
154               ps.println(s);
155           }
156       }
157   
158       public List<String> getLines() {
159           final List<String> result = new ArrayList<>(height);
160           for (int y = 0; y < height; y++) {
161               result.add(getLine(y));
162           }
163           return Collections.unmodifiableList(result);
164       }

166       public void drawHLine(char c, int line, int col1, int col2) {
167           for (int x = col1; x < col2; x++) {
168               this.drawChar(c, x, line);
169           }
170       }
171   
172       public void drawHLine(char c, int line, int col1, int col2, char ifFound, char thenUse) {
173           for (int x = col1; x < col2; x++) {
174               ensurePossible(x, line);
175               if (this.chars[x][line] == ifFound) {
176                   this.drawChar(thenUse, x, line);
177               } else {
178                   this.drawChar(c, x, line);
179               }
180           }
181       }
182   
183       public void drawVLine(char c, int col, int line1, int line2) {
184           for (int y = line1; y < line2; y++) {
185               this.drawChar(c, col, y);
186           }
187       }

189       @Override
190       public String toString() {
191           return getLines().toString();
192       }

194       public void fillRect(char c, int x, int y, int width, int height) {
195           for (int i = 0; i < width; i++) {
196               for (int j = 0; j < height; j++) {
197                   drawChar(c, x + i, y + j);
198               }
199           }
200       }
