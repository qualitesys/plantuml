43   /**
44    * <p>See <a href="http://www.cl.cam.ac.uk/~mgk25/ucs/wcwidth.c">wcwidth.c</a></p>
45    *
46    * <p>This is an implementation of wcwidth() and wcswidth() (defined in
47    * IEEE Std 1002.1-2001) for Unicode.</p>
48    *
49    * http://www.opengroup.org/onlinepubs/007904975/functions/wcwidth.html
50    * http://www.opengroup.org/onlinepubs/007904975/functions/wcswidth.html
51    *
52    * <p>In fixed-width output devices, Latin characters all occupy a single
53    * "cell" position of equal width, whereas ideographic CJK characters
54    * occupy two such cells. Interoperability between terminal-line
55    * applications and (teletype-style) character terminals using the
56    * UTF-8 encoding requires agreement on which character should advance
57    * the cursor by how many cell positions. No established formal
58    * standards exist at present on which Unicode character shall occupy
59    * how many cell positions on character terminals. These routines are
60    * a first attempt of defining such behavior based on simple rules
61    * applied to data provided by the Unicode Consortium.</p>
62    *
63    * <p>For some graphical characters, the Unicode standard explicitly
64    * defines a character-cell width via the definition of the East Asian
65    * FullWidth (F), Wide (W), Half-width (H), and Narrow (Na) classes.
66    * In all these cases, there is no ambiguity about which width a
67    * terminal shall use. For characters in the East Asian Ambiguous (A)
68    * class, the width choice depends purely on a preference of backward
69    * compatibility with either historic CJK or Western practice.
70    * Choosing single-width for these characters is easy to justify as
71    * the appropriate long-term solution, as the CJK practice of
72    * displaying these characters as double-width comes from historic
73    * implementation simplicity (8-bit encoded characters were displayed
74    * single-width and 16-bit ones double-width, even for Greek,
75    * Cyrillic, etc.) and not any typographic considerations.</p>
76    *
77    * <p>Much less clear is the choice of width for the Not East Asian
78    * (Neutral) class. Existing practice does not dictate a width for any
79    * of these characters. It would nevertheless make sense
80    * typographically to allocate two character cells to characters such
81    * as for instance EM SPACE or VOLUME INTEGRAL, which cannot be
82    * represented adequately with a single-width glyph. The following
83    * routines at present merely assign a single-cell width to all
84    * neutral characters, in the interest of simplicity. This is not
85    * entirely satisfactory and should be reconsidered before
86    * establishing a formal standard in this area. At the moment, the
87    * decision which Not East Asian (Neutral) characters should be
88    * represented by double-width glyphs cannot yet be answered by
89    * applying a simple rule from the Unicode database content. Setting
90    * up a proper standard for the behavior of UTF-8 character terminals
91    * will require a careful analysis not only of each Unicode character,
92    * but also of each presentation form, something the author of these
93    * routines has avoided to do so far.</p>
94    *
95    * <p>http://www.unicode.org/unicode/reports/tr11/</p>
96    */

156     static boolean bisearch(int ucs) {
157       int min = 0;
158       int mid;
159       int max = COMBINING.length - 1;
160   
161       if (ucs < COMBINING[0][0] || ucs > COMBINING[max][1]) {
162         return false;
163       }
164       while (max >= min) {
165         mid = (min + max) / 2;
166         if (ucs > COMBINING[mid][1]) {
167           min = mid + 1;
168         } else if (ucs < COMBINING[mid][0]) {
169           max = mid - 1;
170         } else {
171           return true;
172         }
173       }
174   
175       return false;
176     }

178     /**
179      * See : http://www.cl.cam.ac.uk/%7Emgk25/ucs/wcwidth.c
180      *
181      * The following two functions define the column width of an ISO 10646
182      * character as follows:
183      *
184      *    - The null character (U+0000) has a column width of 0.
185      *
186      *    - Other C0/C1 control characters and DEL will lead to a return
187      *      value of -1.
188      *
189      *    - Non-spacing and enclosing combining characters (general
190      *      category code Mn or Me in the Unicode database) have a
191      *      column width of 0.
192      *
193      *    - SOFT HYPHEN (U+00AD) has a column width of 1.
194      *
195      *    - Other format characters (general category code Cf in the Unicode
196      *      database) and ZERO WIDTH SPACE (U+200B) have a column width of 0.
197      *
198      *    - Hangul Jamo medial vowels and final consonants (U+1160-U+11FF)
199      *      have a column width of 0.
200      *
201      *    - Spacing characters in the East Asian Wide (W) or East Asian
202      *      Full-width (F) category as defined in Unicode Technical
203      *      Report #11 have a column width of 2.
204      *
205      *    - All remaining characters (including all printable
206      *      ISO 8859-1 and WGL4 characters, Unicode control characters,
207      *      etc.) have a column width of 1.
208      *
209      * This implementation assumes that wchar_t characters are encoded
210      * in ISO 10646.
211      */
212     public static int of(int codePoint) {
213       // test for 8-bit control characters
214       if (codePoint == 0) {
215         return 0;
216       }
217       if (codePoint < 32 || (codePoint >= 0x7f && codePoint < 0xa0)) {
218         return -1;
219       }
220       // Added by Arnaud : Usual 8 bit char 
221       if (codePoint < 127) {
222         return 1;
223       }
224       // binary search in table of non-spacing characters
225       if (bisearch(codePoint)) {
226         return 0;
227       }
228   
229       // if we arrive here, ucs is not a combining or C0/C1 control character
230       return 1 +
231           ((codePoint >= 0x1100 &&
232               (codePoint <= 0x115f ||                    // Hangul Jamo init. consonants
233                   codePoint == 0x2329 || codePoint == 0x232a ||
234                   (codePoint >= 0x2e80 && codePoint <= 0xa4cf &&
235                       codePoint != 0x303f) ||                  // CJK ... Yi
236                   (codePoint >= 0xac00 && codePoint <= 0xd7a3) || // Hangul Syllables
237                   (codePoint >= 0xf900 && codePoint <= 0xfaff) || // CJK Compatibility Ideographs
238                   (codePoint >= 0xfe10 && codePoint <= 0xfe19) || // Vertical forms
239                   (codePoint >= 0xfe30 && codePoint <= 0xfe6f) || // CJK Compatibility Forms
240                   (codePoint >= 0xff00 && codePoint <= 0xff60) || // Fullwidth Forms
241                   (codePoint >= 0xffe0 && codePoint <= 0xffe6) ||
242                   (codePoint >= 0x20000 && codePoint <= 0x2fffd) ||
243                   (codePoint >= 0x30000 && codePoint <= 0x3fffd))) ? 1 : 0);
244     }
245   
246       public static int length(CharSequence s) {
247           int result = 0;
248           for (int i = 0; i < s.length(); i++) {
249               result += of(s.charAt(i));
250           }
251           return result;
252       }
