254       /**
255        * Check SecurityProfile to see if this file can be open.
256        */
257       private boolean isFileOk() {
258           if (SecurityUtils.getSecurityProfile() == SecurityProfile.SANDBOX)
259               // In SANDBOX, we cannot read any files
260               return false;
261   
262           // In any case SFile should not access the security folders
263           // (the files must be handled internally)
264           try {
265               if (isDenied())
266                   return false;
267           } catch (IOException e) {
268               return false;
269           }
270           // Files in "plantuml.include.path" and "plantuml.allowlist.path" are ok.
271           if (isInAllowList(SecurityUtils.getPath(SecurityUtils.PATHS_INCLUDES)))
272               return true;
273   
274           if (isInAllowList(SecurityUtils.getPath(SecurityUtils.ALLOWLIST_LOCAL_PATHS)))
275               return true;
276   
277           if (SecurityUtils.getSecurityProfile() == SecurityProfile.INTERNET)
278               return false;
279   
280           if (SecurityUtils.getSecurityProfile() == SecurityProfile.ALLOWLIST)
281               return false;
282   
283           if (SecurityUtils.getSecurityProfile() != SecurityProfile.UNSECURE) {
284               // For UNSECURE, we did not do those checks
285               final String path = getCleanPathSecure();
286               if (path.startsWith("/etc/") || path.startsWith("/dev/") || path.startsWith("/boot/")
287                       || path.startsWith("/proc/") || path.startsWith("/sys/"))
288                   return false;
289   
290               if (path.startsWith("//"))
291                   return false;
292   
293           }
294           return true;
295       }
296   
297       private boolean isInAllowList(List<SFile> allowlist) {
298           final String path = getCleanPathSecure();
299           for (SFile allow : allowlist) {
300               if (path.startsWith(allow.getCleanPathSecure())) {
301                   // File directory is in the allowlist
302                   return true;
303               }
304           }
305           return false;
306       }
307   
308       /**
309        * Checks, if the SFile is inside the folder (-structure) of the security area.
310        *
311        * @return true, if the file is not allowed to read/write
312        * @throws IOException If an I/O error occurs, which is possible because the
313        *                     check the pathname may require filesystem queries
314        */
315       private boolean isDenied() throws IOException {
316           SFile securityPath = SecurityUtils.getSecurityPath();
317           if (securityPath == null)
318               return false;
319           return getSanitizedPath().startsWith(securityPath.getSanitizedPath());
320       }
321   
322       /**
323        * Returns a sanitized, canonical and normalized Path to a file.
324        *
325        * @return the Path
326        * @throws IOException If an I/O error occurs, which is possible because the
327        *                     construction of the canonical pathname may require
328        *                     filesystem queries
329        * @see #getCleanPathSecure()
330        * @see File#getCanonicalPath()
331        * @see Path#normalize()
332        */
333       private Path getSanitizedPath() throws IOException {
334           return Paths.get(new File(getCleanPathSecure()).getCanonicalPath()).normalize();
335       }
336   
337       private String getCleanPathSecure() {
338           String result = internal.getAbsolutePath();
339           result = result.replace("\0", "");
340           result = result.replace("\\\\", "/");
341           return result;
342       }
