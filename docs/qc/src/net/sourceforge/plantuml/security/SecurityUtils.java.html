221       public static List<SFile> getPath(String prop) {
222           final List<SFile> result = new ArrayList<>();
223           String paths = getenv(prop);
224           if (paths == null) {
225               return Collections.unmodifiableList(result);
226           }
227           paths = StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(paths);
228           final StringTokenizer st = new StringTokenizer(paths, System.getProperty("path.separator"));
229           while (st.hasMoreTokens()) {
230               final String tmp = st.nextToken();
231               try {
232                   final SFile f = new SFile(tmp).getCanonicalFile();
233                   if (f.isDirectory())
234                       result.add(f);
235               } catch (IOException e) {
236                   Log.info("Cannot access to " + tmp + ". " + e);
237               }
238           }
239           return Collections.unmodifiableList(result);
240       }

269           final Image tmpImage = imageIcon.getImage();

293       /**
294        * Returns the authorize-manager for a security credentials configuration.
295        *
296        * @param credentialConfiguration the credentials
297        * @return the manager.
298        */
299       public static SecurityAuthorizeManager getAuthenticationManager(SecurityCredentials credentialConfiguration) {
300           if (credentialConfiguration == SecurityCredentials.NONE) {
301               return PUBLIC_AUTH_MANAGER;
302           } else if ("tokenauth".equalsIgnoreCase(credentialConfiguration.getType())) {
303               return TOKEN_AUTH_MANAGER;
304           } else if ("basicauth".equalsIgnoreCase(credentialConfiguration.getType())) {
305               return BASICAUTH_AUTH_MANAGER;
306           } else if ("oauth2".equalsIgnoreCase(credentialConfiguration.getType())) {
307               String grantType = credentialConfiguration.getPropertyStr("grantType");
308               if ("client_credentials".equalsIgnoreCase(grantType)) {
309                   return OAUTH2_CLIENT_AUTH_MANAGER;
310               } else if ("password".equalsIgnoreCase(grantType)) {
311                   return OAUTH2_RESOURCEOWNER_AUTH_MANAGER;
312               }
313           }
314           return PUBLIC_AUTH_MANAGER;
315       }

317       /**
318        * Returns the authentication interceptor for a {@link SecurityAuthentication}.
319        *
320        * @param authentication the authentication data
321        * @return the interceptor.
322        */
323       public static SecurityAccessInterceptor getAccessInterceptor(SecurityAuthentication authentication) {
324           if (authentication != null) {
325               String type = authentication.getType();
326               if ("public".equals(type)) {
327                   return PUBLIC_ACCESS_INTERCEPTOR;
328               } else if ("tokenauth".equalsIgnoreCase(type)) {
329                   return TOKEN_ACCESS_INTERCEPTOR;
330               } else if ("basicauth".equalsIgnoreCase(type)) {
331                   return BASICAUTH_ACCESS_INTERCEPTOR;
332               } else if ("oauth2".equalsIgnoreCase(type)) {
333                   return OAUTH2_ACCESS_INTERCEPTOR;
334               }
335           }
336           // Unknown? Fall back to public:
337           return PUBLIC_ACCESS_INTERCEPTOR;
338       }

340       /**
341        * Checks if user credentials existing.
342        *
343        * @param userToken name of the credential file
344        * @return boolean, if exists
345        */
346       public static boolean existsSecurityCredentials(String userToken) {
347           SFile securityPath = getSecurityPath();
348           if (securityPath != null) {
349               // SFile does not allow access to the security path (to hide the credentials in
350               // DSL scripts)
351               File securityFilePath = securityPath.conv();
352               File userCredentials = new File(securityFilePath, userToken + ".credential");
353               return userCredentials.exists() && userCredentials.canRead() && !userCredentials.isDirectory()
354                       && userCredentials.length() > 2;
355   
356           }
357           return false;
358       }

360       /**
361        * Loads the user credentials from the file system.
362        *
363        * @param userToken name of the credential file
364        * @return the credentials or NONE
365        */
366       public static SecurityCredentials loadSecurityCredentials(String userToken) {
367           if (userToken != null && checkFileSystemSaveCharactersStrict(userToken) && !NO_CREDENTIALS.equals(userToken)) {
368               final SFile securityPath = getSecurityPath();
369               if (securityPath != null) {
370                   // SFile does not allow access to the security path (to hide the credentials in
371                   // DSL scripts)
372                   File securityFilePath = securityPath.conv();
373                   File userCredentials = new File(securityFilePath, userToken + ".credential");
374                   JsonValue jsonValue = loadJson(userCredentials);
375                   return SecurityCredentials.fromJson(jsonValue);
376               }
377           }
378           return SecurityCredentials.NONE;
379       }

381       /**
382        * Checks, if the token of a pathname (filename, ext, directory-name) uses only
383        * a very strict set of characters and not longer than 64 characters.
384        * <p>
385        * Only characters from a to Z, Numbers and - are allowed.
386        *
387        * @param pathNameToken filename, ext, directory-name
388        * @return true, if the string fits to the strict allow-list of characters
389        * @see #SECURE_CHARS
390        */
391       private static boolean checkFileSystemSaveCharactersStrict(String pathNameToken) {
392           return StringUtils.isNotEmpty(pathNameToken) && SECURE_CHARS.matcher(pathNameToken).matches()
393                   && pathNameToken.length() <= 64;
394       }

396       /**
397        * Loads the path to the configured security folder, if existing.
398        * <p>
399        * Please note: A SFile referenced to a security folder cannot access the files.
400        * The content of the files in the security path should never have passed to DSL
401        * scripts.
402        *
403        * @return SFile folder or null
404        */
405       public static SFile getSecurityPath() {
406           List<SFile> paths = getPath(PATHS_SECURITY);
407           if (!paths.isEmpty()) {
408               SFile secureSFile = paths.get(0);
409               File securityFolder = secureSFile.conv();
410               if (securityFolder.exists() && securityFolder.isDirectory()) {
411                   return secureSFile;
412               }
413           }
414           return null;
415       }

417       /**
418        * Loads a file as JSON object. If no file exists or the file is not parsable,
419        * the method returns an empty JSON.
420        *
421        * @param jsonFile file path to the JSON file
422        * @return a Json vale (maybe empty)
423        */
424       private static JsonValue loadJson(File jsonFile) {
425           if (jsonFile.exists() && jsonFile.canRead() && jsonFile.length() > 2) {
426               // we have a file with at least two bytes and readable, hopefully it's a JSON
427               try (Reader r = new BufferedReader(new FileReader(jsonFile))) {
428                   return Json.parse(r);
429               } catch (IOException e) {
430                   Logme.error(e);
431               }
432           }
433           return Json.object();
434       }
