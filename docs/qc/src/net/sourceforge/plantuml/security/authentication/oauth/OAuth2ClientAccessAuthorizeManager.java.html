36   package net.sourceforge.plantuml.security.authentication.oauth;
37   
38   import java.util.Arrays;
39   
40   import net.sourceforge.plantuml.StringUtils;
41   import net.sourceforge.plantuml.security.SURL;
42   import net.sourceforge.plantuml.security.authentication.SecurityAuthentication;
43   import net.sourceforge.plantuml.security.authentication.SecurityCredentials;
44   import net.sourceforge.plantuml.security.authentication.basicauth.BasicAuthAuthorizeManager;
45   
46   /**
47    * Authorize the principal (from {@link SecurityCredentials} and creates a {@link SecurityAuthentication} object with a
48    * bearer token secret.
49    *
50    * @author Aljoscha Rittner
51    */
52   public class OAuth2ClientAccessAuthorizeManager extends AbstractOAuth2AccessAuthorizeManager {
53   
54       /**
55        * Basic Auth manager to access the token service with authorization.
56        */
57       private final BasicAuthAuthorizeManager basicAuthManager = new BasicAuthAuthorizeManager();
58   
59       @Override
60       public SecurityAuthentication create(SecurityCredentials credentials) {
61           String grantType = credentials.getPropertyStr("grantType");
62           String requestScope = credentials.getPropertyStr("scope");
63           String accessTokenUri = credentials.getPropertyStr("accessTokenUri");
64           String tokenType = credentials.getPropertyStr("tokenType");
65   
66           // Extra BasicAuth data to access the token service endpoint (if needed)
67           String identifier = credentials.getPropertyStr("credentials.identifier");
68           char[] secret = credentials.getPropertyChars("credentials.secret");
69   
70           try {
71               SURL tokenService = SURL.create(accessTokenUri);
72   
73               StringBuilder content = new StringBuilder()
74                       .append("grant_type=")
75                       .append(urlEncode(grantType));
76               if (StringUtils.isNotEmpty(requestScope)) {
77                   content.append("&scope=").append(urlEncode(requestScope));
78               }
79   
80               SecurityAuthentication basicAuth;
81               if (identifier != null) {
82                   // OAuth2 with extra Endpoint BasicAuth credentials
83                   basicAuth = basicAuthManager.create(
84                           SecurityCredentials.basicAuth(identifier, secret));
85                   // We need to add the principal to the form
86                   content.append("&client_id").append(urlEncode(credentials.getIdentifier()))
87                           .append("&client_secret").append(urlEncode(new String(credentials.getSecret())));
88               } else {
89                   // OAuth2 with BasicAuth via principal (standard)
90                   basicAuth = basicAuthManager.create(
91                           SecurityCredentials.basicAuth(credentials.getIdentifier(), credentials.getSecret()));
92               }
93   
94               return requestAndCreateAuthFromResponse(
95                       credentials.getProxy(), grantType, tokenType, tokenService, content.toString(), basicAuth);
96           } finally {
97               if (secret != null && secret.length > 0) {
98                   Arrays.fill(secret, '*');
99               }
100           }
101       }
102   
103   }
