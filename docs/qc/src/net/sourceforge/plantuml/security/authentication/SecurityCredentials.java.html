50   /**
51    * Defines a configuration for credentials.
52    *
53    * @author Aljoscha Rittner
54    */
55   public class SecurityCredentials implements SecurityCredentialsContainer {
56   
57       private static final Map<String, Object> EMPTY_MAP = Collections.emptyMap();
58   
59       /**
60        * No credentials given.
61        */
62       public static final SecurityCredentials NONE = new SecurityCredentials("<NONE>", "public", null, null);
63   
64       /**
65        * Name of the configuration.
66        */
67       private final String name;
68       /**
69        * The type of authorization and access process (e.g. "basicauth" or "oauth2").
70        */
71       private final String type;
72       /**
73        * Username or client identifier.
74        */
75       private final String identifier;
76       /**
77        * User/Client secret information.
78        */
79       private final char[] secret;
80       /**
81        * Properties defined for a specific authorization and access process.
82        */
83       private final Map<String, Object> properties = new HashMap<String, Object>();
84       /**
85        * Proxy configuration.<p>
86        * <p>
87        * {@link Proxy#NO_PROXY} means, we want direct access. null means, we use the system proxy configuration.
88        */
89       private final Proxy proxy;
90   
91       /**
92        * Creates BasicAuth credentials without a proxy.
93        *
94        * @param name       Name of the credentials
95        * @param type       The type of authentication and access process (e.g. "basicauth" or "oauth2")
96        * @param identifier username, clientId, ...
97        * @param secret     the secret information to authenticate the client or user
98        */
99       public SecurityCredentials(String name, String type, String identifier, char[] secret) {
100           this(name, type, identifier, secret, EMPTY_MAP, null);
101       }
102   
103       /**
104        * Creates BasicAuth credentials with a proxy.
105        *
106        * @param name       Name of the credentials
107        * @param type       The type of authentication and access process (e.g. "basicauth" or "oauth2")
108        * @param identifier username, clientId, ...
109        * @param secret     the secret information to authenticate the client or user
110        * @param proxy      proxy configuration
111        */
112       public SecurityCredentials(String name, String type, String identifier, char[] secret,
113                                  Map<String, Object> properties, Proxy proxy) {
114           if (name == null) {
115               throw new NullPointerException("Credential name should not be null");
116           }
117           this.name = name;
118           this.type = type;
119           this.identifier = identifier;
120           this.secret = secret;
121           this.proxy = proxy;
122           this.properties.putAll(properties);
123       }

125       /**
126        * Creates BasicAuth credentials.
127        *
128        * @param identifier the basic auth user name.
129        * @param secret     password
130        * @return credential object
131        */
132       public static SecurityCredentials basicAuth(String identifier, char[] secret) {
133           return new SecurityCredentials(identifier, "basicauth", identifier, secret);
134       }

136       /**
137        * Creates a SecurityCredentials from a JSON.
138        * <p>
139        * Example:
140        * <pre>
141        *     {
142        *         "name": "jenkins",
143        *         "identifier": "alice",
144        *         "secret": "secret",
145        *         "proxy": {
146        *             "type": "socket",
147        *             "address": "192.168.1.250",
148        *             "port": 8080
149        *         }
150        *     }
151        * </pre>
152        *
153        * @param jsonValue a JSON structure
154        * @return the created SecurityCredentials
155        */
156       public static SecurityCredentials fromJson(JsonValue jsonValue) {
157           try {
158               JsonObject securityObject = jsonValue.asObject();
159               JsonValue name = securityObject.get("name");
160               JsonValue type = securityObject.get("type");
161               JsonValue identifier = securityObject.get("identifier");
162               JsonValue secret = securityObject.get("secret");
163   
164               Map<String, Object> map = new HashMap<String, Object>();
165               buildProperties("", securityObject.get("properties"), map);
166   
167               if (type != null && !type.isNull() && "tokenauth".equals(type.asString())) {
168                   return new SecurityCredentials(name.asString(), "tokenauth",
169                           null, null, map, proxyFromJson(securityObject.get("proxy")));
170               } else if (StringUtils.isNotEmpty(name.asString()) && StringUtils.isNotEmpty(identifier.asString())) {
171                   String authType = type != null && !type.isNull() ? type.asString() : "basicauth";
172                   return new SecurityCredentials(name.asString(), authType,
173                           identifier.asString(), extractSecret(secret),
174                           map, proxyFromJson(securityObject.get("proxy")));
175               }
176   
177           } catch (UnsupportedOperationException use) {
178               // We catch UnsupportedOperationException to stop parsing on unexpected elements
179           }
180           return NONE;
181       }
182   
183       /**
184        * Creates a Proxy object from a JSON value.
185        * <p>
186        * Example:
187        * <pre>
188        *     {
189        *         "type": "socket",
190        *         "address": "192.168.1.250",
191        *         "port": 8080
192        *     }
193        * </pre>
194        *
195        * @param proxyValue JSON, that represents a Proxy object
196        * @return Proxy object or null
197        */
198       private static Proxy proxyFromJson(JsonValue proxyValue) {
199           if (proxyValue != null && !proxyValue.isNull() && proxyValue.isObject()) {
200               Proxy.Type type = Proxy.Type.DIRECT;
201   
202               JsonObject proxyObject = proxyValue.asObject();
203               JsonValue proxyType = proxyObject.get("type");
204               if (proxyType != null && !proxyType.isNull()) {
205                   type = Proxy.Type.valueOf(proxyType.asString().toUpperCase());
206               }
207               if (type == Proxy.Type.DIRECT) {
208                   return Proxy.NO_PROXY;
209               }
210               JsonValue proxyAddress = proxyObject.get("address");
211               JsonValue proxyPort = proxyObject.get("port");
212               if (proxyAddress != null && !proxyAddress.isNull() && !proxyPort.isNull() && proxyPort.isNumber()) {
213                   InetSocketAddress address = new InetSocketAddress(proxyAddress.asString(), proxyPort.asInt());
214                   return new Proxy(type, address);
215               }
216           }
217           return null;
218       }
219   
220       /**
221        * Extracts a password, if it is not empty or null.
222        *
223        * @param pwd password json value
224        * @return password or null
225        */
226       private static char[] extractSecret(JsonValue pwd) {
227           if (pwd == null || pwd.isNull()) {
228               return null;
229           }
230           String pwdStr = pwd.asString();
231           if (StringUtils.isEmpty(pwdStr)) {
232               return null;
233           }
234   
235           return pwdStr.toCharArray();
236       }
237   
238       /**
239        * Creates a properties map from all given key/values.
240        * <p>
241        * Example:<br/>
242        * <pre>
243        *     {
244        *         "grantType": "client_credentials",
245        *         "scope": "read write",
246        *         "accessTokenUri": "https://login-demo.curity.io/oauth/v2/oauth-token"
247        *         "credentials": {
248        *             "identifier": "serviceId",
249        *             "secret": "ServiceSecret"
250        *         }
251        *     }
252        *
253        *     will be transformed to:
254        *
255        *     grantType -> client_credentials
256        *     scope -> read write
257        *     accessTokenUri -> https://login-demo.curity.io/oauth/v2/oauth-token
258        *     credentials.identifier -> serviceId
259        *     credentials.secret -> ServiceSecret
260        * </pre>
261        *
262        * @param prefix    the prefix for the direct children
263        * @param fromValue parent JSON value to read from
264        * @param toMap     map to populate
265        */
266       private static void buildProperties(String prefix, JsonValue fromValue, Map<String, Object> toMap) {
267           if (!isJsonObjectWithMembers(fromValue)) {
268               return;
269           }
270           JsonObject members = fromValue.asObject();
271           for (String name : members.names()) {
272               JsonValue child = members.get(name);
273               if (child.isArray() || child.isNull()) {
274                   // currently, not supported or not needed
275                   continue;
276               }
277               String key = StringUtils.isEmpty(prefix) ? name : prefix + '.' + name;
278               if (child.isObject()) {
279                   buildProperties(key, child, toMap);
280               } else {
281                   if (child.isString()) {
282                       toMap.put(key, child.asString());
283                   } else if (child.isBoolean()) {
284                       toMap.put(key, child.asBoolean());
285                   } else if (child.isNumber()) {
286                       toMap.put(key, child.asDouble());
287                   }
288               }
289           }
290       }
291   
292       /**
293        * Checks, if we have a JSON object with members.
294        *
295        * @param jsonValue the value to check
296        * @return true, if we have members in the JSON object
297        */
298       private static boolean isJsonObjectWithMembers(JsonValue jsonValue) {
299           return jsonValue != null && !jsonValue.isNull() && jsonValue.isObject() && !jsonValue.asObject().isEmpty();
300       }

322       /**
323        * Returns the property as String.
324        *
325        * @param key Name of the property
326        * @return String representation
327        */
328       public String getPropertyStr(String key) {
329           Object value = getProperties().get(key);
330           if (value != null) {
331               return value.toString();
332           }
333           return null;
334       }
335   
336       /**
337        * Returns the property as characters.
338        *
339        * @param key Name of the property
340        * @return char[] representation
341        */
342       public char[] getPropertyChars(String key) {
343           Object value = getProperties().get(key);
344           if (value != null) {
345               return value.toString().toCharArray();
346           }
347           return null;
348       }
349   
350       /**
351        * Returns the property as boolean.
352        *
353        * @param key Name of the property
354        * @return boolean representation
355        */
356       public boolean getPropertyBool(String key) {
357           Object value = getProperties().get(key);
358           if (value != null) {
359               if (value instanceof Boolean) {
360                   return (Boolean) value;
361               } else if (value instanceof String) {
362                   return Boolean.parseBoolean((String) value);
363               }
364           }
365           return false;
366       }
367   
368       /**
369        * Returns the property as Number.
370        *
371        * @param key Name of the property
372        * @return boolean representation
373        */
374       public Number getPropertyNum(String key) {
375           Object value = getProperties().get(key);
376           if (value != null) {
377               if (value instanceof Number) {
378                   return (Number) value;
379               } else if (value instanceof String) {
380                   return Double.parseDouble((String) value);
381               }
382           }
383           return null;
384       }

391       public void eraseCredentials() {
392           if (secret != null && secret.length > 0) {
393               Arrays.fill(secret, '*');
394           }
