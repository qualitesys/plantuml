36   package net.sourceforge.plantuml.security;
37   
38   /**
39    * There are 4 different security profile defined.
40    * <p>
41    * The security profile to be used is set at the launch of PlantUML and cannot
42    * be changed by users. The security profile defines what an instance of
43    * PlantUML is allowed to do:<br>
44    * - access some local file <br>
45    * - connection to some remote URL <br>
46    * - print some technical information to the users.
47    * <p>
48    * <p>
49    * The security profile is defined: <br>
50    * - either by an environment variable<br>
51    * - or an option at command line
52    * <p>
53    * There is also a default value, which is LEGACY in this current
54    * implementation.
55    * 
56    */
57   public enum SecurityProfile {
58   
59       /**
60        * Running in SANDBOX mode is completely secure. No local file can be read
61        * (except dot executable) No remote URL access can be used No technical
62        * information are print to users.
63        * <p>
64        * This mode is defined for test and debug, since it's not very useful for
65        * users. However, you can use it if you need to.
66        */
67       SANDBOX,
68   
69       /**
70        * 
71        */
72       ALLOWLIST,
73   
74       /**
75        * This mode is designed for PlantUML running in a web server.
76        * 
77        */
78       INTERNET,
79   
80       /**
81        * This mode reproduce old PlantUML version behaviour.
82        * <p>
83        * Right now, this is the default Security Profile but this will be removed from
84        * future version because it is now full secure, especially on Internet server.
85        */
86       LEGACY,
87   
88       /**
89        * Running in UNSECURE mode means that PlantUML can access to any local file and
90        * can connect to any URL.
91        * <p>
92        * Some technical information (file path, Java version) are also printed in some
93        * error messages. This is not an issue if you are running PlantUML locally. But
94        * you should not use this mode if PlantUML is running on some server,
95        * especially if the server is accessible from Internet.
96        */
97       UNSECURE;
98   
99       /**
100        * Initialize the default value.
101        * <p>
102        * It search in some config variable if the user has defined a some default
103        * value.
104        * 
105        * @return the value
106        */
107       static SecurityProfile init() {
108           final String env = SecurityUtils.getenv("PLANTUML_SECURITY_PROFILE");
109           if ("SANDBOX".equalsIgnoreCase(env)) {
110               return SANDBOX;
111           } else if ("ALLOWLIST".equalsIgnoreCase(env)) {
112               return ALLOWLIST;
113           } else if ("INTERNET".equalsIgnoreCase(env)) {
114               return INTERNET;
115           } else if ("UNSECURE".equalsIgnoreCase(env)) {
116               return UNSECURE;
117           }
118           return LEGACY;
119       }
120   
121       /**
122        * A Human understandable description.
123        */
124       public String longDescription() {
125           switch (this) {
126           case SANDBOX:
127               return "This is completely safe: no access to local files or to distant URL.";
128           case ALLOWLIST:
129               return "Some local resource may be accessible.";
130           case INTERNET:
131               return "<i>Mode designed for server connected to Internet.";
132           case LEGACY:
133               return "<b>Warning: this mode will be removed in future version";
134           case UNSECURE:
135               return "<b>Make sure that this server is not accessible from Internet";
136           }
137           return "<i>This is completely safe: no access on local files or on distant URL.";
138       }
139   
140       /**
141        * Retrieve the timeout for URL.
142        */
143       public long getTimeout() {
144           switch (this) {
145           case SANDBOX:
146               return 1000L;
147           case ALLOWLIST:
148               return 1000L * 60 * 5;
149           case INTERNET:
150               return 1000L * 10;
151           case LEGACY:
152               return 1000L * 60;
153           case UNSECURE:
154               return 1000L * 60 * 5;
155           }
156           throw new AssertionError();
157       }
