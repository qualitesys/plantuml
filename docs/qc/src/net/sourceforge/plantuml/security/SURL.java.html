76   /**
77    * Secure replacement for java.net.URL.
78    * <p>
79    * This class should be used instead of java.net.URL.
80    * <p>
81    * This class does some control access and manages access-tokens via URL. If a
82    * URL contains a access-token, similar to a user prefix, SURL loads the
83    * authorization config for this user-token and passes the credentials to the
84    * host.
85    * <p>
86    * Example:<br/>
87    * 
88    * <pre>
89    *     SURL url = SURL.create ("https://jenkins-access@jenkins.mycompany.com/api/json")
90    * </pre>
91    * 
92    * The {@code jenkins-access} will checked against the Security context access
93    * token configuration. If a configuration exists for this token name, the token
94    * will be removed from the URL and the credentials will be added to the
95    * headers. If the token is not found, the URL remains as it is and no separate
96    * authentication will be performed.
97    * <p>
98    * TODO: Some methods should be moved to a HttpClient implementation, because
99    * SURL is not the valid class to manage it. <br/>
100    * TODO: BAD_HOSTS implementation should be reviewed and moved to HttpClient
101    * implementation with a circuit-breaker. <br/>
102    * TODO: Token expiration with refresh should be implemented in future. <br/>
103    */
104   public class SURL {

141       /**
142        * Create a secure URL from a String.
143        * <p>
144        * The url must be http or https. Return null in case of error or if
145        * <code>url</code> is null
146        * 
147        * @param url plain url starting by http:// or https//
148        * @return the secure URL or null
149        */
150       public static SURL create(String url) {
151           if (url == null)
152               return null;
153   
154           if (url.startsWith("http://") || url.startsWith("https://"))
155               try {
156                   return create(new URL(url));
157               } catch (MalformedURLException e) {
158                   Logme.error(e);
159               }
160           return null;
161       }

163       /**
164        * Create a secure URL from a <code>java.net.URL</code> object.
165        * <p>
166        * It takes into account credentials.
167        * 
168        * @param url
169        * @return the secure URL
170        * @throws MalformedURLException if <code>url</code> is null
171        */
172       public static SURL create(URL url) throws MalformedURLException {
173           if (url == null)
174               throw new MalformedURLException("URL cannot be null");
175   
176           final String credentialId = url.getUserInfo();
177   
178           if (credentialId == null || credentialId.indexOf(':') > 0)
179               // No user info at all, or a user with password (This is a legacy BasicAuth
180               // access, and we bypass it):
181               return new SURL(url, WITHOUT_AUTHENTICATION);
182           else if (SecurityUtils.existsSecurityCredentials(credentialId))
183               // Given userInfo, but without a password. We try to find SecurityCredentials
184               return new SURL(removeUserInfo(url), credentialId);
185           else
186               return new SURL(url, WITHOUT_AUTHENTICATION);
187       }
188   
189       /**
190        * Creates a URL without UserInfo part and without SecurityCredentials.
191        *
192        * @param url plain URL
193        * @return SURL without any user credential information.
194        * @throws MalformedURLException
195        */
196       static SURL createWithoutUser(URL url) throws MalformedURLException {
197           return new SURL(removeUserInfo(url), WITHOUT_AUTHENTICATION);
198       }

218       /**
219        * Check SecurityProfile to see if this URL can be opened.
220        */
221       private boolean isUrlOk() {
222           if (SecurityUtils.getSecurityProfile() == SecurityProfile.SANDBOX)
223               // In SANDBOX, we cannot read any URL
224               return false;
225   
226           if (SecurityUtils.getSecurityProfile() == SecurityProfile.LEGACY)
227               return true;
228   
229           if (SecurityUtils.getSecurityProfile() == SecurityProfile.UNSECURE)
230               // We are UNSECURE anyway
231               return true;
232   
233           if (isInUrlAllowList())
234               return true;
235   
236           if (SecurityUtils.getSecurityProfile() == SecurityProfile.INTERNET) {
237               if (forbiddenURL(cleanPath(internal.toString())))
238                   return false;
239   
240               final int port = internal.getPort();
241               // Using INTERNET profile, port 80 and 443 are ok
242               return port == 80 || port == 443 || port == -1;
243           }
244           return false;
245       }
246   
247       private boolean forbiddenURL(String full) {
248           if (full.matches("^https?://[-#.0-9:\\[\\]+]+/.*"))
249               return true;
250           if (full.matches("^https?://[^.]+/.*"))
251               return true;
252           return false;
253       }
254   
255       private boolean isInUrlAllowList() {
256           final String full = cleanPath(internal.toString());
257           for (String allow : getUrlAllowList())
258               if (full.startsWith(cleanPath(allow)))
259                   return true;
260   
261           return false;
262       }
263   
264       private String cleanPath(String path) {
265           // Remove user information, because we don't like to store user/password or
266           // userTokens in allow-list
267           path = removeUserInfoFromUrlPath(path);
268           path = path.trim().toLowerCase(Locale.US);
269           // We simplify/normalize the url, removing default ports
270           path = path.replace(":80/", "");
271           path = path.replace(":443/", "");
272           return path;
273       }
274   
275       private List<String> getUrlAllowList() {
276           final String env = SecurityUtils.getenv(SecurityUtils.ALLOWLIST_URL);
277           if (env == null)
278               return Collections.emptyList();
279   
280           return Arrays.asList(StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(env).split(";"));
281       }

283       /**
284        * Reads from an endpoint (with configured credentials and proxy) the response
285        * as blob.
286        * <p>
287        * This method allows access to an endpoint, with a configured
288        * SecurityCredentials object. The credentials will load on the fly and
289        * authentication fetched from an authentication-manager. Caching of tokens is
290        * not supported.
291        * <p>
292        * authors: Alain Corbiere, Aljoscha Rittner
293        *
294        * @return data loaded data from endpoint
295        */
296       public byte[] getBytes() {
297           if (isUrlOk() == false)
298               return null;
299   
300           final SecurityCredentials credentials = SecurityUtils.loadSecurityCredentials(securityIdentifier);
301           final SecurityAuthentication authentication = SecurityUtils.getAuthenticationManager(credentials)
302                   .create(credentials);
303           try {
304               final String host = internal.getHost();
305               final Long bad = BAD_HOSTS.get(host);
306               if (bad != null) {
307                   if ((System.currentTimeMillis() - bad) < 1000L * 60)
308                       return null;
309                   BAD_HOSTS.remove(host);
310               }
311   
312               try {
313                   final Future<byte[]> result = EXE
314                           .submit(requestWithGetAndResponse(internal, credentials.getProxy(), authentication, null));
315                   final byte[] data = result.get(SecurityUtils.getSecurityProfile().getTimeout(), TimeUnit.MILLISECONDS);
316                   if (data != null)
317                       return data;
318   
319               } catch (Exception e) {
320                   System.err.println("issue " + host + " " + e);
321               }
322   
323               BAD_HOSTS.put(host, System.currentTimeMillis());
324               return null;
325           } finally {
326               // clean up. We don't cache tokens, no expire handling. All time a re-request.
327               credentials.eraseCredentials();
328               authentication.eraseCredentials();
329           }
330       }

332       /**
333        * Reads from an endpoint with a given authentication and proxy the response as
334        * blob.
335        * <p>
336        * This method allows a parametrized access to an endpoint, without a configured
337        * SecurityCredentials object. This is useful to access internally identity
338        * providers (IDP), or authorization servers (to request access tokens).
339        * <p>
340        * This method don't use the "bad-host" functionality, because the access to
341        * infrastructure services should not be obfuscated by some internal management.
342        * <p>
343        * <strong>Please don't use this method directly from DSL scripts.</strong>
344        *
345        * @param authentication authentication object data. Caller is responsible to
346        *                       erase credentials
347        * @param proxy          proxy configuration
348        * @param headers        additional headers, if needed
349        * @return loaded data from endpoint
350        */
351       private byte[] getBytes(Proxy proxy, SecurityAuthentication authentication, Map<String, Object> headers) {
352           if (isUrlOk() == false)
353               return null;
354   
355           final Future<byte[]> result = EXE.submit(requestWithGetAndResponse(internal, proxy, authentication, headers));
356   
357           try {
358               return result.get(SecurityUtils.getSecurityProfile().getTimeout(), TimeUnit.MILLISECONDS);
359           } catch (Exception e) {
360               System.err.println("SURL response issue to " + internal.getHost() + " " + e);
361               return null;
362           }
363       }

401       /**
402        * Creates a GET request and response handler
403        * 
404        * @param url            URL to request
405        * @param proxy          proxy to apply
406        * @param authentication the authentication to use
407        * @param headers        additional headers, if needed
408        * @return the callable handler.
409        */
410       private static Callable<byte[]> requestWithGetAndResponse(final URL url, final Proxy proxy,
411               final SecurityAuthentication authentication, final Map<String, Object> headers) {
412           return new Callable<byte[]>() {
413   
414               private HttpURLConnection openConnection(final URL url) throws IOException {
415                   // Add proxy, if passed throw parameters
416                   final URLConnection connection = proxy == null ? url.openConnection() : url.openConnection(proxy);
417                   if (connection == null)
418                       return null;
419   
420                   final HttpURLConnection http = (HttpURLConnection) connection;
421   
422                   applyEndpointAccessAuthentication(http, authentication);
423                   applyAdditionalHeaders(http, headers);
424                   return http;
425               }
426   
427               public byte[] call() throws IOException {
428                   HttpURLConnection http = openConnection(url);
429                   final int responseCode = http.getResponseCode();
430   
431                   if (responseCode == HttpURLConnection.HTTP_MOVED_TEMP
432                           || responseCode == HttpURLConnection.HTTP_MOVED_PERM) {
433                       final String newUrl = http.getHeaderField("Location");
434                       http = openConnection(new URL(newUrl));
435                   }
436   
437                   return retrieveResponseAsBytes(http);
438               }
439           };
440       }

442       /**
443        * Creates a POST request and response handler with a simple String content. The
444        * content will be identified as form or JSON data. The charset encoding can be
445        * set by header parameters or will be set to UTF-8. The method to some fancy
446        * logic to simplify it for the user.
447        * 
448        * @param url            URL to request via POST method
449        * @param proxy          proxy to apply
450        * @param authentication the authentication to use
451        * @param headers        additional headers, if needed
452        * @return the callable handler.
453        */
454       private static Callable<byte[]> requestWithPostAndResponse(final URL url, final Proxy proxy,
455               final SecurityAuthentication authentication, final String data, final Map<String, Object> headers) {
456           return new Callable<byte[]>() {
457               public byte[] call() throws IOException {
458                   // Add proxy, if passed throw parameters
459                   final URLConnection connection = proxy == null ? url.openConnection() : url.openConnection(proxy);
460                   if (connection == null)
461                       return null;
462   
463                   final boolean withContent = StringUtils.isNotEmpty(data);
464   
465                   final HttpURLConnection http = (HttpURLConnection) connection;
466                   http.setRequestMethod("POST");
467                   if (withContent)
468                       http.setDoOutput(true);
469   
470                   applyEndpointAccessAuthentication(http, authentication);
471                   applyAdditionalHeaders(http, headers);
472   
473                   final Charset charSet = extractCharset(http.getRequestProperty("Content-Type"));
474   
475                   if (withContent)
476                       sendRequestAsBytes(http, data.getBytes(charSet != null ? charSet : StandardCharsets.UTF_8));
477   
478                   return retrieveResponseAsBytes(http);
479               }
480           };

506       private static byte[] retrieveResponseAsBytes(HttpURLConnection connection) throws IOException {

517               }

523        *

526        * @throws IOException if something went wrong
527        */
528       private static byte[] retrieveData(InputStream input) throws IOException {
529           final ByteArrayOutputStream out = new ByteArrayOutputStream();

595       private static void applyEndpointAccessAuthentication(URLConnection http, SecurityAuthentication authentication) {
596           if (authentication.isPublic())
597               // Shortcut: No need to apply authentication.

617           if (headers == null || headers.isEmpty())
618               return;

621               final Object value = header.getValue();

623                   http.setRequestProperty(header.getKey(), (String) value);
624               else if (value instanceof List)
625                   for (Object item : (List<?>) value)
626                       if (item != null)
627                           http.addRequestProperty(header.getKey(), item.toString());
628   
629           }
630       }
