75       public FtpLoop(Socket socket, FtpServer ftpServer) throws IOException {
76           this.incoming = socket;
77           this.ftpServer = ftpServer;
78           this.br = new BufferedReader(new InputStreamReader(incoming.getInputStream(), ftpServer.getCharset()));
79           this.pw = SecurityUtils.createPrintWriter(incoming.getOutputStream(), true);

90       private void runInternal() throws IOException, InterruptedException {
91           localLog("Starting Loop");
92           myOut("220 PlantUML");
93           while (true) {
94               final String s = br.readLine();
95               localLog("s=" + s);
96               if (s == null) {
97                   pw.close();
98                   br.close();
99                   return;
100               }
101               final boolean finish = manage(s);
102               if (finish) {
103                   return;
104               }
105           }
106       }

108       private boolean manage(final String cmd) throws UnknownHostException, IOException, InterruptedException {
109           final String upper = StringUtils.goUpperCase(cmd);
110           if (upper.startsWith("USER")) {
111               myOut("331 Password required");
112               final String user = cmd.substring("USER ".length());
113               connexion = ftpServer.getFtpConnexion(user);
114           } else if (upper.startsWith("PASS")) {
115               myOut("230 Logged in.");
116           } else if (upper.startsWith("PWD")) {
117               // myOut("/");
118               // myOut("200 OK /");
119               myOut("257 \"/\" is current directory.");
120           } else if (upper.startsWith("CWD")) {
121               final String dir = cmd.substring("CWD ".length());
122               myOut("250 \"" + dir + "\" is new working directory..");
123           } else if (upper.startsWith("TYPE")) {
124               myOut("200 Command okay.");
125               // localLog("type=" + s);
126           } else if (upper.startsWith("PORT")) {
127               mode = Mode.ACTIF;
128               final StringTokenizer st = new StringTokenizer(cmd, " ,");
129               st.nextToken();
130               ipClient = st.nextToken() + "." + st.nextToken() + "." + st.nextToken() + "." + st.nextToken();
131               port = Integer.parseInt(st.nextToken()) * 256 + Integer.parseInt(st.nextToken());
132               // localLog("ipClient=" + ipClient);
133               // localLog("port=" + port);
134   
135               myOut("200 Command okay.");
136           } else if (upper.startsWith("LIST")) {
137               if (mode == Mode.ACTIF) {
138                   listActif();
139               } else {
140                   listPassif();
141               }
142           } else if (upper.startsWith("STOR")) {
143               if (mode == Mode.ACTIF) {
144                   storActif(cmd);
145               } else {
146                   storPassif(cmd);
147               }
148           } else if (upper.startsWith("PASV")) {
149               mode = Mode.PASSIF;
150               port = ftpServer.getFreePort();
151               final int p1 = port / 256;
152               final int p2 = port % 256;
153               assert port == p1 * 256 + p2;
154               localLog("adr=" + incoming.getInetAddress().getHostAddress());
155               final String ipServer = ftpServer.getIpServer();
156               localLog("server=" + ipServer);
157               myOut("227 Entering Passive Mode (" + ipServer.replace('.', ',') + "," + p1 + "," + p2 + ").");
158               ipClient = ipServer;
159           } else if (upper.startsWith("RETR")) {
160               if (mode == Mode.ACTIF) {
161                   retrActif(cmd);
162               } else {
163                   retrPassif(cmd);
164               }
165           } else if (upper.startsWith("DELE")) {
166               final String file = cmd.substring("DELE ".length());
167               connexion.delete(file);
168               myOut("200 Command okay.");
169           } else if (upper.startsWith("QUIT")) {
170               myOut("221 Goodbye.");
171               return true;
172           } else if (upper.startsWith("SYST")) {
173               myOut("215 UNIX Type: L8.");
174           } else {
175               myOut("502 Command not implemented.");
176           }
177           return false;
178       }

183       private void retr(final String fileName, Socket soc)
184               throws UnknownHostException, IOException, InterruptedException {
185           final OutputStream os = soc.getOutputStream();
186           final byte[] data = connexion.getData(fileName);
187   
188           if (data != null) {
189               os.write(data);
190           }
191           os.flush();
192           os.close();
193           soc.close();
194           myOut("226 Transfer complete.");
195       }

197       private void retrPassif(final String s) throws UnknownHostException, IOException, InterruptedException {
198           String fileName = s.substring("STOR ".length());
199           fileName = removeStartingsSlash(fileName);
200           if (connexion.willExist(fileName) == false) {
201               myOut("550 No such file.");
202               return;
203           }
204           myOut("150 Opening");
205           waitForMe(fileName);
206           final ServerSocket ss = new ServerSocket(port);
207           final Socket incoming = ss.accept();
208           retr(fileName, incoming);
209           ss.close();
210       }

212       private void waitForMe(String fileName) throws InterruptedException {
213           do {
214               if (connexion.doesExist(fileName)) {
215                   return;
216               }
217               Thread.sleep(200L);
218           } while (true);
219       }

221       private void retrActif(final String s) throws UnknownHostException, IOException, InterruptedException {
222           String fileName = s.substring("STOR ".length());
223           fileName = removeStartingsSlash(fileName);
224           if (connexion.willExist(fileName) == false) {
225               myOut("550 No such file.");
226               return;
227           }
228           myOut("150 Opening");
229           waitForMe(fileName);
230           final Socket soc = new Socket(ipClient, port);
231           retr(fileName, soc);
232       }

250       private String removeStartingsSlash(String fileName) {
251           while (fileName.startsWith("/")) {
252               fileName = fileName.substring(1);
253           }
254           return fileName;
255       }

257       private void stor(String fileName, Socket socket) throws UnknownHostException, IOException {
258           final InputStream is = socket.getInputStream();
259           final ByteArrayOutputStream baos = new ByteArrayOutputStream();
260           FileUtils.copyToStream(is, baos);
261   
262           myOut("226 Transfer complete.");
263   
264           if ("png".equalsIgnoreCase(fileName)) {
265               connexion.setFileFormat(FileFormat.PNG);
266           } else if ("svg".equalsIgnoreCase(fileName)) {
267               connexion.setFileFormat(FileFormat.SVG);
268           } else if ("eps".equalsIgnoreCase(fileName)) {
269               connexion.setFileFormat(FileFormat.EPS);
270           }
271   
272           if (fileName.length() > 3) {
273               final String data = new String(baos.toByteArray(), ftpServer.getCharset());
274               final String pngFileName = connexion.getFutureFileName(fileName);
275               connexion.futureOutgoing(pngFileName);
276               connexion.addIncoming(fileName, data);
277   
278               ftpServer.processImage(connexion, fileName);
279           }
280       }

296       private void list(final Socket soc) throws IOException {
297           final PrintWriter listing = SecurityUtils.createPrintWriter(soc.getOutputStream(), true);
298           final Collection<String> files = connexion.getFiles();
299           if (files.size() > 0) {
300               int total = 0;
301               for (String n : files) {
302                   total += (connexion.getSize(n) + 511) / 512;
303               }
304               listing.println("total " + total);
305               // localLog(total);
306               for (String n : files) {
307                   final String ls = String.format("%10s %4d %-8s %-8s %8d %3s %2s %5s %s", "-rw-rw-r--", 1, "plantuml",
308                           "plantuml", connexion.getSize(n), "Sep", 28, 2006, n);
309                   listing.println(ls);
310                   // localLog(ls);
311               }
312           }
313           listing.flush();
314           listing.close();
315           soc.close();
316           myOut("226 Listing completed.");
317       }

319       private void myOut(String s) {
320           if (s.indexOf('\t') != -1) {
321               throw new IllegalArgumentException();
322           }
323           pw.println(s);
324           pw.flush();
325       }
