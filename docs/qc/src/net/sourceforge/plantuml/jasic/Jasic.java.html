34   /*
35    * This defines a single class that contains an entire interpreter for a language very similar to the original BASIC.
36    * Everything is here (albeit in very simplified form): tokenizing, parsing, and interpretation. The file is organized
37    * in phases, with each appearing roughly in the order that they occur when a program is run. You should be able to read
38    * this top-down to walk through the entire process of loading and running a program.
39    * 
40    * Jasic language syntax ---------------------
41    * 
42    * Comments start with ' and proceed to the end of the line:
43    * 
44    * print "hi there" ' this is a comment
45    * 
46    * Numbers and strings are supported. Strings should be in "double quotes", and only positive integers can be parsed
47    * (though numbers are double internally).
48    * 
49    * Variables are identified by name which must start with a letter and can contain letters or numbers. Case is
50    * significant for names and keywords.
51    * 
52    * Each statement is on its own line. Optionally, a line may have a label before the statement. A label is a name that
53    * ends with a colon:
54    * 
55    * foo:
56    * 
57    * 
58    * The following statements are supported:
59    * 
60    * <name> = <expression> Evaluates the expression and assigns the result to the given named variable. All variables are
61    * globally scoped.
62    * 
63    * pi = (314159 / 10000)
64    * 
65    * print <expression> Evaluates the expression and prints the result.
66    * 
67    * print "hello, " + "world"
68    * 
69    * input <name> Reads in a line of input from the user and stores it in the variable with the given name.
70    * 
71    * input guess
72    * 
73    * goto <label> Jumps to the statement after the label with the given name.
74    * 
75    * goto loop
76    * 
77    * if <expression> then <label> Evaluates the expression. If it evaluates to a non-zero number, then jumps to the
78    * statement after the given label.
79    * 
80    * if a < b then dosomething
81    * 
82    * 
83    * The following expressions are supported:
84    * 
85    * <expression> = <expression> Evaluates to 1 if the two expressions are equal, 0 otherwise.
86    * 
87    * <expression> + <expression> If the left-hand expression is a number, then adds the two expressions, otherwise
88    * concatenates the two strings.
89    * 
90    * <expression> - <expression> <expression> * <expression> <expression> / <expression> <expression> < <expression>
91    * <expression> > <expression> You can figure it out.
92    * 
93    * <name> A name in an expression simply returns the value of the variable with that name. If the variable was never
94    * set, it defaults to 0.
95    * 
96    * All binary operators have the same precedence. Sorry, I had to cut corners somewhere.
97    * 
98    * To keep things simple, I've omitted some stuff or hacked things a bit. When possible, I'll leave a "HACK" note there
99    * explaining what and why. If you make your own interpreter, you'll want to address those.
100    * 
101    * @author Bob Nystrom
102    */
103   public class Jasic {
104   
105       // Tokenizing (lexing) -----------------------------------------------------
106   
107       /**
108        * This function takes a script as a string of characters and chunks it into a sequence of tokens. Each token is a
109        * meaningful unit of program, like a variable name, a number, a string, or an operator.
110        */
111       private static List<Token> tokenize(String source) {
112           List<Token> tokens = new ArrayList<>();
113   
114           String token = "";
115           TokenizeState state = TokenizeState.DEFAULT;
116   
117           // Many tokens are a single character, like operators and ().
118           String charTokens = "\n=+-*/<>()";
119           TokenType[] tokenTypes = { TokenType.LINE, TokenType.EQUALS, TokenType.OPERATOR, TokenType.OPERATOR,
120                   TokenType.OPERATOR, TokenType.OPERATOR, TokenType.OPERATOR, TokenType.OPERATOR, TokenType.LEFT_PAREN,
121                   TokenType.RIGHT_PAREN };
122   
123           // Scan through the code one character at a time, building up the list
124           // of tokens.
125           for (int i = 0; i < source.length(); i++) {
126               char c = source.charAt(i);
127               switch (state) {
128               case DEFAULT:
129                   if (charTokens.indexOf(c) != -1) {
130                       tokens.add(new Token(Character.toString(c), tokenTypes[charTokens.indexOf(c)]));
131                   } else if (Character.isLetter(c)) {
132                       token += c;
133                       state = TokenizeState.WORD;
134                   } else if (Character.isDigit(c)) {
135                       token += c;
136                       state = TokenizeState.NUMBER;
137                   } else if (c == '"') {
138                       state = TokenizeState.STRING;
139                   } else if (c == '\'') {
140                       state = TokenizeState.COMMENT;
141                   }
142                   break;
143   
144               case WORD:
145                   if (Character.isLetterOrDigit(c)) {
146                       token += c;
147                   } else if (c == ':') {
148                       tokens.add(new Token(token, TokenType.LABEL));
149                       token = "";
150                       state = TokenizeState.DEFAULT;
151                   } else {
152                       tokens.add(new Token(token, TokenType.WORD));
153                       token = "";
154                       state = TokenizeState.DEFAULT;
155                       i--; // Reprocess this character in the default state.
156                   }
157                   break;
158   
159               case NUMBER:
160                   // HACK: Negative numbers and floating points aren't supported.
161                   // To get a negative number, just do 0 - <your number>.
162                   // To get a floating point, divide.
163                   if (Character.isDigit(c)) {
164                       token += c;
165                   } else {
166                       tokens.add(new Token(token, TokenType.NUMBER));
167                       token = "";
168                       state = TokenizeState.DEFAULT;
169                       i--; // Reprocess this character in the default state.
170                   }
171                   break;
172   
173               case STRING:
174                   if (c == '"') {
175                       tokens.add(new Token(token, TokenType.STRING));
176                       token = "";
177                       state = TokenizeState.DEFAULT;
178                   } else {
179                       token += c;
180                   }
181                   break;
182   
183               case COMMENT:
184                   if (c == '\n') {
185                       state = TokenizeState.DEFAULT;
186                   }
187                   break;
188               }
189           }
190   
191           // HACK: Silently ignore any in-progress token when we run out of
192           // characters. This means that, for example, if a script has a string
193           // that's missing the closing ", it will just ditch it.
194           return tokens;
195       }
196   
197       // Token data --------------------------------------------------------------
198   
199       /**
200        * This defines the different kinds of tokens or meaningful chunks of code that the parser knows how to consume.
201        * These let us distinguish, for example, between a string "foo" and a variable named "foo".
202        * 
203        * HACK: A typical tokenizer would actually have unique token types for each keyword (print, goto, etc.) so that the
204        * parser doesn't have to look at the names, but Jasic is a little more crude.
205        */
206       private enum TokenType {
207           WORD, NUMBER, STRING, LABEL, LINE, EQUALS, OPERATOR, LEFT_PAREN, RIGHT_PAREN, EOF
208       }
209   
210       /**
211        * This is a single meaningful chunk of code. It is created by the tokenizer and consumed by the parser.
212        */
213       private static class Token {
214           public Token(String text, TokenType type) {
215               this.text = text;
216               this.type = type;
217           }
218   
219           public final String text;
220           public final TokenType type;
221       }
222   
223       /**
224        * This defines the different states the tokenizer can be in while it's scanning through the source code. Tokenizers
225        * are state machines, which means the only data they need to store is where they are in the source code and this
226        * one "state" or mode value.
227        * 
228        * One of the main differences between tokenizing and parsing is this regularity. Because the tokenizer stores only
229        * this one state value, it can't handle nesting (which would require also storing a number to identify how deeply
230        * nested you are). The parser is able to handle that.
231        */
232       private enum TokenizeState {
233           DEFAULT, WORD, NUMBER, STRING, COMMENT
234       }
235   
236       // Parsing -----------------------------------------------------------------
237   
238       /**
239        * This defines the Jasic parser. The parser takes in a sequence of tokens and generates an abstract syntax tree.
240        * This is the nested data structure that represents the series of statements, and the expressions (which can nest
241        * arbitrarily deeply) that they evaluate. In technical terms, what we have is a recursive descent parser, the
242        * simplest kind to hand-write.
243        *
244        * As a side-effect, this phase also stores off the line numbers for each label in the program. It's a bit gross,
245        * but it works.
246        */
247       private class Parser {
248           public Parser(List<Token> tokens) {
249               this.tokens = tokens;
250               position = 0;
251           }
252   
253           /**
254            * The top-level function to start parsing. This will keep consuming tokens and routing to the other parse
255            * functions for the different grammar syntax until we run out of code to parse.
256            * 
257            * @param labels
258            *            A map of label names to statement indexes. The parser will fill this in as it scans the code.
259            * @return The list of parsed statements.
260            */
261           public List<Statement> parse(Map<String, Integer> labels) {
262               List<Statement> statements = new ArrayList<>();
263   
264               while (true) {
265                   // Ignore empty lines.
266                   while (match(TokenType.LINE))
267                       ;
268   
269                   if (match(TokenType.LABEL)) {
270                       // Mark the index of the statement after the label.
271                       labels.put(last(1).text, statements.size());
272                   } else if (match(TokenType.WORD, TokenType.EQUALS)) {
273                       String name = last(2).text;
274                       Expression value = expression();
275                       statements.add(new AssignStatement(name, value));
276                   } else if (match("print")) {
277                       statements.add(new PrintStatement(expression()));
278                   } else if (match("input")) {
279                       statements.add(new InputStatement(consume(TokenType.WORD).text));
280                   } else if (match("goto")) {
281                       statements.add(new GotoStatement(consume(TokenType.WORD).text));
282                   } else if (match("if")) {
283                       Expression condition = expression();
284                       consume("then");
285                       String label = consume(TokenType.WORD).text;
286                       statements.add(new IfThenStatement(condition, label));
287                   } else
288                       break; // Unexpected token (likely EOF), so end.
289               }
290   
291               return statements;
292           }
293   
294           // The following functions each represent one grammatical part of the
295           // language. If this parsed English, these functions would be named like
296           // noun() and verb().
297   
298           /**
299            * Parses a single expression. Recursive descent parsers start with the lowest-precedent term and moves towards
300            * higher precedence. For Jasic, binary operators (+, -, etc.) are the lowest.
301            * 
302            * @return The parsed expression.
303            */
304           private Expression expression() {
305               return operator();
306           }
307   
308           /**
309            * Parses a series of binary operator expressions into a single expression. In Jasic, all operators have the
310            * same predecence and associate left-to-right. That means it will interpret: 1 + 2 * 3 - 4 / 5 like: ((((1 + 2)
311            * * 3) - 4) / 5)
312            * 
313            * It works by building the expression tree one at a time. So, given this code: 1 + 2 * 3, this will:
314            * 
315            * 1. Parse (1) as an atomic expression. 2. See the (+) and start a new operator expression. 3. Parse (2) as an
316            * atomic expression. 4. Build a (1 + 2) expression and replace (1) with it. 5. See the (*) and start a new
317            * operator expression. 6. Parse (3) as an atomic expression. 7. Build a ((1 + 2) * 3) expression and replace (1
318            * + 2) with it. 8. Return the last expression built.
319            * 
320            * @return The parsed expression.
321            */
322           private Expression operator() {
323               Expression expression = atomic();
324   
325               // Keep building operator expressions as long as we have operators.
326               while (match(TokenType.OPERATOR) || match(TokenType.EQUALS)) {
327                   char operator = last(1).text.charAt(0);
328                   Expression right = atomic();
329                   expression = new OperatorExpression(expression, operator, right);
330               }
331   
332               return expression;
333           }
334   
335           /**
336            * Parses an "atomic" expression. This is the highest level of precedence and contains single literal tokens
337            * like 123 and "foo", as well as parenthesized expressions.
338            * 
339            * @return The parsed expression.
340            */
341           private Expression atomic() {
342               if (match(TokenType.WORD)) {
343                   // A word is a reference to a variable.
344                   return new VariableExpression(last(1).text);
345               } else if (match(TokenType.NUMBER)) {
346                   return new NumberValue(Double.parseDouble(last(1).text));
347               } else if (match(TokenType.STRING)) {
348                   return new StringValue(last(1).text);
349               } else if (match(TokenType.LEFT_PAREN)) {
350                   // The contents of a parenthesized expression can be any
351                   // expression. This lets us "restart" the precedence cascade
352                   // so that you can have a lower precedence expression inside
353                   // the parentheses.
354                   Expression expression = expression();
355                   consume(TokenType.RIGHT_PAREN);
356                   return expression;
357               }
358               throw new Error("Couldn't parse :(");
359           }
360   
361           // The following functions are the core low-level operations that the
362           // grammar parser is built in terms of. They match and consume tokens in
363           // the token stream.
364   
365           /**
366            * Consumes the next two tokens if they are the given type (in order). Consumes no tokens if either check fais.
367            * 
368            * @param type1
369            *            Expected type of the next token.
370            * @param type2
371            *            Expected type of the subsequent token.
372            * @return True if tokens were consumed.
373            */
374           private boolean match(TokenType type1, TokenType type2) {
375               if (get(0).type != type1)
376                   return false;
377               if (get(1).type != type2)
378                   return false;
379               position += 2;
380               return true;
381           }
382   
383           /**
384            * Consumes the next token if it's the given type.
385            * 
386            * @param type
387            *            Expected type of the next token.
388            * @return True if the token was consumed.
389            */
390           private boolean match(TokenType type) {
391               if (get(0).type != type)
392                   return false;
393               position++;
394               return true;
395           }
396   
397           /**
398            * Consumes the next token if it's a word token with the given name.
399            * 
400            * @param name
401            *            Expected name of the next word token.
402            * @return True if the token was consumed.
403            */
404           private boolean match(String name) {
405               if (get(0).type != TokenType.WORD)
406                   return false;
407               if (!get(0).text.equals(name))
408                   return false;
409               position++;
410               return true;
411           }
412   
413           /**
414            * Consumes the next token if it's the given type. If not, throws an exception. This is for cases where the
415            * parser demands a token of a certain type in a certain position, for example a matching ) after an opening (.
416            * 
417            * @param type
418            *            Expected type of the next token.
419            * @return The consumed token.
420            */
421           private Token consume(TokenType type) {
422               if (get(0).type != type)
423                   throw new Error("Expected " + type + ".");
424               return tokens.get(position++);
425           }
426   
427           /**
428            * Consumes the next token if it's a word with the given name. If not, throws an exception.
429            * 
430            * @param name
431            *            Expected name of the next word token.
432            * @return The consumed token.
433            */
434           private Token consume(String name) {
435               if (!match(name))
436                   throw new Error("Expected " + name + ".");
437               return last(1);
438           }
439   
440           /**
441            * Gets a previously consumed token, indexing backwards. last(1) will be the token just consumed, last(2) the
442            * one before that, etc.
443            * 
444            * @param offset
445            *            How far back in the token stream to look.
446            * @return The consumed token.
447            */
448           private Token last(int offset) {
449               return tokens.get(position - offset);
450           }
451   
452           /**
453            * Gets an unconsumed token, indexing forward. get(0) will be the next token to be consumed, get(1) the one
454            * after that, etc.
455            * 
456            * @param offset
457            *            How far forward in the token stream to look.
458            * @return The yet-to-be-consumed token.
459            */
460           private Token get(int offset) {
461               if (position + offset >= tokens.size()) {
462                   return new Token("", TokenType.EOF);
463               }
464               return tokens.get(position + offset);
465           }
466   
467           private final List<Token> tokens;
468           private int position;
469       }

624       /**
625        * An operator expression evaluates two expressions and then performs some arithmetic operation on the results.
626        */
627       public class OperatorExpression implements Expression {
628           public OperatorExpression(Expression left, char operator, Expression right) {
629               this.left = left;
630               this.operator = operator;
631               this.right = right;
632           }
633   
634           public Value evaluate() {
635               Value leftVal = left.evaluate();
636               Value rightVal = right.evaluate();
637   
638               switch (operator) {
639               case '=':
640                   // Coerce to the left argument's type, then compare.
641                   if (leftVal instanceof NumberValue) {
642                       return new NumberValue((leftVal.toNumber() == rightVal.toNumber()) ? 1 : 0);
643                   } else {
644                       return new NumberValue(leftVal.toString().equals(rightVal.toString()) ? 1 : 0);
645                   }
646               case '+':
647                   // Addition if the left argument is a number, otherwise do
648                   // string concatenation.
649                   if (leftVal instanceof NumberValue) {
650                       return new NumberValue(leftVal.toNumber() + rightVal.toNumber());
651                   } else {
652                       return new StringValue(leftVal.toString() + rightVal.toString());
653                   }
654               case '-':
655                   return new NumberValue(leftVal.toNumber() - rightVal.toNumber());
656               case '*':
657                   return new NumberValue(leftVal.toNumber() * rightVal.toNumber());
658               case '/':
659                   return new NumberValue(leftVal.toNumber() / rightVal.toNumber());
660               case '<':
661                   // Coerce to the left argument's type, then compare.
662                   if (leftVal instanceof NumberValue) {
663                       return new NumberValue((leftVal.toNumber() < rightVal.toNumber()) ? 1 : 0);
664                   } else {
665                       return new NumberValue((leftVal.toString().compareTo(rightVal.toString()) < 0) ? 1 : 0);
666                   }
667               case '>':
668                   // Coerce to the left argument's type, then compare.
669                   if (leftVal instanceof NumberValue) {
670                       return new NumberValue((leftVal.toNumber() > rightVal.toNumber()) ? 1 : 0);
671                   } else {
672                       return new NumberValue((leftVal.toString().compareTo(rightVal.toString()) > 0) ? 1 : 0);
673                   }
674               }
675               throw new Error("Unknown operator.");
676           }
677   
678           private final Expression left;
679           private final char operator;
680           private final Expression right;
681       }

685       /**
686        * This is the base interface for a value. Values are the data that the interpreter processes. They are what gets
687        * stored in variables, printed, and operated on.
688        * 
689        * There is an implementation of this interface for each of the different primitive types (really just double and
690        * string) that Jasic supports. Wrapping them in a single Value interface lets Jasic be dynamically-typed and
691        * convert between different representations as needed.
692        * 
693        * Note that Value extends Expression. This is a bit of a hack, but it lets us use values (which are typically only
694        * ever seen by the interpreter and not the parser) as both runtime values, and as object representing literals in
695        * code.
696        */

699            * Value types override this to convert themselves to a string representation.
700            */
701           String toString();
702   
703           /**
704            * Value types override this to convert themselves to a numeric representation.
705            */
706           double toNumber();
707       }

728           }

757       // Interpreter -------------------------------------------------------------
758   
759       /**
760        * Constructs a new Jasic instance. The instance stores the global state of the interpreter such as the values of
761        * all of the variables and the current statement.
762        */
763       public Jasic() {
764           variables = new HashMap<String, Value>();
765           labels = new HashMap<String, Integer>();
766   
767           InputStreamReader converter = new InputStreamReader(System.in);
768           lineIn = new BufferedReader(converter);
769       }
770   
771       /**
772        * This is where the magic happens. This runs the code through the parsing pipeline to generate the AST. Then it
773        * executes each statement. It keeps track of the current line in a member variable that the statement objects have
774        * access to. This lets "goto" and "if then" do flow control by simply setting the index of the current statement.
775        *
776        * In an interpreter that didn't mix the interpretation logic in with the AST node classes, this would be doing a
777        * lot more work.
778        * 
779        * @param source
780        *            A string containing the source code of a .jas script to interpret.
781        */
782       public void interpret(String source) {
783           // Tokenize.
784           List<Token> tokens = tokenize(source);
785   
786           // Parse.
787           Parser parser = new Parser(tokens);
788           List<Statement> statements = parser.parse(labels);
789   
790           // Interpret until we're done.
791           currentStatement = 0;
792           while (currentStatement < statements.size()) {
793               int thisStatement = currentStatement;
794               currentStatement++;
795               statements.get(thisStatement).execute();
796           }
797       }
798   
799       private final Map<String, Value> variables;
800       private final Map<String, Integer> labels;
801   
802       private final BufferedReader lineIn;
803   
804       private int currentStatement;
805   
806   }
