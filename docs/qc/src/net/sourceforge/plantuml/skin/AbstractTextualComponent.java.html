56   public abstract class AbstractTextualComponent extends AbstractComponent {
57   
58       private final int marginX1;
59       private final int marginX2;
60       private final int marginY;
61   
62       private final TextBlock textBlock;
63       private final ISkinSimple spriteContainer;
64   
65       private final UFont font;
66       private final HColor fontColor;
67       private final HorizontalAlignment alignment;
68   
69       public AbstractTextualComponent(Style style, LineBreakStrategy maxMessageSize, int marginX1, int marginX2,
70               int marginY, ISkinSimple spriteContainer, CharSequence label) {
71           this(style, style, maxMessageSize, marginX1, marginX2, marginY, spriteContainer,
72                   Display.getWithNewlines(label == null ? "" : label.toString()), false);
73       }
74   
75       public AbstractTextualComponent(Style style, LineBreakStrategy maxMessageSize, int marginX1, int marginX2,
76               int marginY, ISkinSimple spriteContainer, Display display, boolean enhanced) {
77           this(style, style, maxMessageSize, marginX1, marginX2, marginY, spriteContainer, display, enhanced);
78       }
79   
80       public AbstractTextualComponent(Style style, Style stereo, LineBreakStrategy maxMessageSize, int marginX1,
81               int marginX2, int marginY, ISkinSimple spriteContainer, Display display, boolean enhanced) {
82           super(style);
83           this.spriteContainer = spriteContainer;
84   
85           final FontConfiguration fc = style.getFontConfiguration(getIHtmlColorSet());
86           this.font = style.getUFont();
87           this.fontColor = style.value(PName.FontColor).asColor(getIHtmlColorSet());
88           final HorizontalAlignment horizontalAlignment = style.getHorizontalAlignment();
89           final UFont fontForStereotype = stereo.getUFont();
90           final HColor htmlColorForStereotype = stereo.value(PName.FontColor).asColor(getIHtmlColorSet());
91           display = display.withoutStereotypeIfNeeded(style);
92   
93           this.marginX1 = marginX1;
94           this.marginX2 = marginX2;
95           this.marginY = marginY;
96   
97           if (display.size() == 1 && display.get(0).length() == 0)
98               textBlock = new TextBlockEmpty();
99           else if (enhanced)
100               textBlock = BodyFactory.create3(display, spriteContainer, horizontalAlignment, fc, maxMessageSize, style);
101           else
102               textBlock = display.create0(fc, horizontalAlignment, spriteContainer, maxMessageSize, CreoleMode.FULL,
103                       fontForStereotype, htmlColorForStereotype, marginX1, marginX2);
104   
105           this.alignment = horizontalAlignment;
106       }
107   
108       protected HColorSet getIHtmlColorSet() {
109           return ((ISkinParam) spriteContainer).getIHtmlColorSet();
110       }
111   
112       protected TextBlock getTextBlock() {
113           return textBlock;
114       }
115   
116       protected double getPureTextWidth(StringBounder stringBounder) {
117           final TextBlock textBlock = getTextBlock();
118           final XDimension2D size = textBlock.calculateDimension(stringBounder);
119           return size.getWidth();
120       }
121   
122       final public double getTextWidth(StringBounder stringBounder) {
123           return getPureTextWidth(stringBounder) + marginX1 + marginX2;
124       }
125   
126       final protected double getTextHeight(StringBounder stringBounder) {
127           final TextBlock textBlock = getTextBlock();
128           final XDimension2D size = textBlock.calculateDimension(stringBounder);
129           return size.getHeight() + 2 * marginY;
130       }
131   
132       final protected int getMarginX1() {
133           return marginX1;
134       }
135   
136       final protected int getMarginX2() {
137           return marginX2;
138       }
139   
140       final protected int getMarginY() {
141           return marginY;
142       }
143   
144       final protected UFont getFont() {
145           return font;
146       }
147   
148       protected HColor getFontColor() {
149           return fontColor;
150       }
151   
152       protected final ISkinSimple getISkinSimple() {
153           return spriteContainer;
154       }
155   
156       public final HorizontalAlignment getHorizontalAlignment() {
157           return alignment;
158       }
159   
160   }
