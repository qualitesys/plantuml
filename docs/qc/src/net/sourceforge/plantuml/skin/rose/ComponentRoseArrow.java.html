68       public ComponentRoseArrow(Style style, Display stringsToDisplay, ArrowConfiguration arrowConfiguration,
69               HorizontalAlignment messagePosition, ISkinSimple spriteContainer, LineBreakStrategy maxMessageSize,

71           super(style, stringsToDisplay, arrowConfiguration, spriteContainer, maxMessageSize);

73           this.niceArrow = niceArrow;
74           this.belowForResponse = belowForResponse;
75           this.inclination1 = arrowConfiguration.getInclination1();
76           this.inclination2 = arrowConfiguration.getInclination2();
77       }

79       public static final double spaceCrossX = 6;
80       public static final double diamCircle = 8;
81       public static final double thinCircle = 1.5;
82   
83       @Override
84       public void drawInternalU(UGraphic ug, Area area) {
85           final ArrowConfiguration arrowConfiguration = getArrowConfiguration();
86           if (arrowConfiguration.isHidden())
87               return;
88   
89           final XDimension2D dimensionToUse = area.getDimensionToUse();
90           final StringBounder stringBounder = ug.getStringBounder();
91           ug = ug.apply(getForegroundColor());
92   
93           final ArrowDressing dressing1 = arrowConfiguration.getDressing1();
94           final ArrowDressing dressing2 = arrowConfiguration.getDressing2();
95   
96           double start = 0;
97           double len = dimensionToUse.getWidth() - 1;
98           final double lenFull = dimensionToUse.getWidth();
99   
100           final double pos1 = start + 1;
101           final double pos2 = len - 1;
102   
103           if (arrowConfiguration.getDecoration2() == ArrowDecoration.CIRCLE && dressing2.getHead() == ArrowHead.NONE)
104               len -= diamCircle / 2;
105   
106           if (arrowConfiguration.getDecoration2() == ArrowDecoration.CIRCLE && dressing2.getHead() != ArrowHead.NONE)
107               len -= diamCircle / 2 + thinCircle;
108   
109           if (arrowConfiguration.getDecoration1() == ArrowDecoration.CIRCLE
110                   && dressing1.getHead() == ArrowHead.NONE) {
111               start += diamCircle / 2;
112               len -= diamCircle / 2;
113           }
114           if (arrowConfiguration.getDecoration1() == ArrowDecoration.CIRCLE
115                   && dressing1.getHead() == ArrowHead.NORMAL) {
116               start += diamCircle + thinCircle;
117               len -= diamCircle + thinCircle;
118           }
119   
120           if (dressing2.getPart() == ArrowPart.FULL && dressing2.getHead() == ArrowHead.NORMAL)
121               len -= getArrowDeltaX() / 2;
122   
123           if (dressing1.getPart() == ArrowPart.FULL && dressing1.getHead() == ArrowHead.NORMAL) {
124               start += getArrowDeltaX() / 2;
125               len -= getArrowDeltaX() / 2;
126           }
127   
128           if (dressing2.getHead() == ArrowHead.CROSSX)
129               len -= 2 * spaceCrossX;
130   
131           if (dressing1.getHead() == ArrowHead.CROSSX) {
132               start += 2 * spaceCrossX;
133               len -= 2 * spaceCrossX;
134           }
135   
136           final double posArrow;
137           final double yText;
138           if (isBelowForResponse()) {
139               posArrow = 0;
140               yText = getMarginY();
141           } else {
142               posArrow = getTextHeight(stringBounder);
143               yText = 0;
144           }
145   
146           drawDressing1(ug.apply(new UTranslate(pos1, posArrow + inclination1)), dressing1,
147                   arrowConfiguration.getDecoration1(), lenFull);
148           drawDressing2(ug.apply(new UTranslate(pos2, posArrow + inclination2)), dressing2,
149                   arrowConfiguration.getDecoration2(), lenFull);
150   
151           if (inclination1 == 0 && inclination2 == 0)
152               arrowConfiguration.applyStroke(ug).apply(new UTranslate(start, posArrow)).draw(new ULine(len, 0));
153           else if (inclination1 != 0)
154               drawLine(arrowConfiguration.applyStroke(ug), start + len, posArrow, 0, posArrow + inclination1);
155           else if (inclination2 != 0)
156               drawLine(arrowConfiguration.applyStroke(ug), start, posArrow, pos2, posArrow + inclination2);
157   
158           final ArrowDirection direction2 = getDirection2();
159           final double textPos;
160           if (messagePosition == HorizontalAlignment.CENTER) {
161               final double textWidth = getTextBlock().calculateDimension(stringBounder).getWidth();
162               textPos = (dimensionToUse.getWidth() - textWidth) / 2;
163           } else if (messagePosition == HorizontalAlignment.RIGHT) {
164               final double textWidth = getTextBlock().calculateDimension(stringBounder).getWidth();
165               textPos = dimensionToUse.getWidth() - textWidth - getMarginX2()
166                       - (direction2 == ArrowDirection.LEFT_TO_RIGHT_NORMAL ? getArrowDeltaX() : 0);
167           } else {
168               textPos = getMarginX1()
169                       + (direction2 == ArrowDirection.RIGHT_TO_LEFT_REVERSE || direction2 == ArrowDirection.BOTH_DIRECTION
170                               ? getArrowDeltaX()
171                               : 0);
172           }
173           getTextBlock().drawU(ug.apply(new UTranslate(textPos, yText)));
174       }

176       private void drawLine(UGraphic ug, double x1, double y1, double x2, double y2) {
177           ug = ug.apply(new UTranslate(x1, y1));
178           ug.draw(new ULine(x2 - x1, y2 - y1));
179   
180       }

182       public double getPosArrow(StringBounder stringBounder) {
183           if (isBelowForResponse())
184               return 0;
185   
186           return getTextHeight(stringBounder) - 2 * getMarginY();
187       }

189       private boolean isBelowForResponse() {
190           return belowForResponse && getDirection2() == ArrowDirection.RIGHT_TO_LEFT_REVERSE;
191       }

193       private void drawDressing1(UGraphic ug, ArrowDressing dressing, ArrowDecoration decoration, double lenFull) {
194   
195           if (decoration == ArrowDecoration.CIRCLE) {
196               final UEllipse circle = new UEllipse(diamCircle, diamCircle);
197               ug.apply(new UStroke(thinCircle)).apply(getForegroundColor())
198                       .apply(new UTranslate(-diamCircle / 2 - thinCircle, -diamCircle / 2 - thinCircle / 2)).draw(circle);
199               if (dressing.getHead() != ArrowHead.CROSSX)
200                   ug = ug.apply(UTranslate.dx(diamCircle / 2 + thinCircle));
201           }
202   
203           if (dressing.getHead() == ArrowHead.ASYNC) {
204               if (dressing.getPart() != ArrowPart.BOTTOM_PART)
205                   getArrowConfiguration().applyThicknessOnly(ug).draw(
206                           new ULine(getArrowDeltaX(), -getArrowDeltaY()).rotate(Math.atan2(-inclination1, lenFull)));
207   
208               if (dressing.getPart() != ArrowPart.TOP_PART)
209                   getArrowConfiguration().applyThicknessOnly(ug)
210                           .draw(new ULine(getArrowDeltaX(), getArrowDeltaY()).rotate(Math.atan2(-inclination1, lenFull)));
211   
212           } else if (dressing.getHead() == ArrowHead.CROSSX) {
213               ug = ug.apply(new UStroke(2));
214               ug.apply(new UTranslate(spaceCrossX, -getArrowDeltaX() / 2))
215                       .draw(new ULine(getArrowDeltaX(), getArrowDeltaX()));
216               ug.apply(new UTranslate(spaceCrossX, getArrowDeltaX() / 2))
217                       .draw(new ULine(getArrowDeltaX(), -getArrowDeltaX()));
218           } else if (dressing.getHead() == ArrowHead.NORMAL) {
219               final UPolygon polygon = getPolygonReverse(dressing.getPart());
220   
221               if (inclination1 != 0)
222                   polygon.rotate(Math.atan2(-inclination1, lenFull));
223   
224               ug.apply(getForegroundColor().bg()).draw(polygon);
225           }
226   
227       }

229       private void drawDressing2(UGraphic ug, ArrowDressing dressing, ArrowDecoration decoration, double lenFull) {
230   
231           if (decoration == ArrowDecoration.CIRCLE) {
232               ug = ug.apply(new UStroke(thinCircle)).apply(getForegroundColor());
233               final UEllipse circle = new UEllipse(diamCircle, diamCircle);
234               ug.apply(new UTranslate(-diamCircle / 2 + thinCircle, -diamCircle / 2 - thinCircle / 2)).draw(circle);
235               ug = ug.apply(new UStroke());
236               ug = ug.apply(UTranslate.dx(-diamCircle / 2 - thinCircle));
237           }
238   
239           if (dressing.getHead() == ArrowHead.ASYNC) {
240               if (dressing.getPart() != ArrowPart.BOTTOM_PART)
241                   getArrowConfiguration().applyThicknessOnly(ug).draw(
242                           new ULine(-getArrowDeltaX(), -getArrowDeltaY()).rotate(Math.atan2(inclination2, lenFull)));
243   
244               if (dressing.getPart() != ArrowPart.TOP_PART)
245                   getArrowConfiguration().applyThicknessOnly(ug)
246                           .draw(new ULine(-getArrowDeltaX(), getArrowDeltaY()).rotate(Math.atan2(inclination2, lenFull)));
247   
248           } else if (dressing.getHead() == ArrowHead.CROSSX) {
249               ug = ug.apply(new UStroke(2));
250               ug.apply(new UTranslate(-spaceCrossX - getArrowDeltaX(), -getArrowDeltaX() / 2))
251                       .draw(new ULine(getArrowDeltaX(), getArrowDeltaX()));
252               ug.apply(new UTranslate(-spaceCrossX - getArrowDeltaX(), getArrowDeltaX() / 2))
253                       .draw(new ULine(getArrowDeltaX(), -getArrowDeltaX()));
254           } else if (dressing.getHead() == ArrowHead.NORMAL) {
255               final UPolygon polygon = getPolygonNormal(dressing.getPart());
256   
257               if (inclination2 != 0)
258                   polygon.rotate(Math.atan2(inclination2, lenFull));
259   
260               ug.apply(getForegroundColor().bg()).draw(polygon);
261           }
262   
263       }
264   
265       private UPolygon getPolygonNormal(ArrowPart part) {
266           UPolygon polygon = new UPolygon();
267           if (part == ArrowPart.TOP_PART) {
268               polygon.addPoint(-getArrowDeltaX(), -getArrowDeltaY());
269               polygon.addPoint(0, 0);
270               polygon.addPoint(-getArrowDeltaX(), 0);
271           } else if (part == ArrowPart.BOTTOM_PART) {
272               polygon.addPoint(-getArrowDeltaX(), 1);
273               polygon.addPoint(0, 1);
274               polygon.addPoint(-getArrowDeltaX(), getArrowDeltaY() + 1);
275           } else {
276               polygon.addPoint(-getArrowDeltaX(), -getArrowDeltaY());
277               polygon.addPoint(0, 0);
278               polygon.addPoint(-getArrowDeltaX(), +getArrowDeltaY());
279               if (niceArrow)
280                   polygon.addPoint(-getArrowDeltaX() + 4, 0);
281           }
282           return polygon;
283       }
284   
285       private UPolygon getPolygonReverse(ArrowPart part) {
286           final UPolygon polygon = new UPolygon();
287           if (part == ArrowPart.TOP_PART) {
288               polygon.addPoint(getArrowDeltaX(), -getArrowDeltaY());
289               polygon.addPoint(0, 0);
290               polygon.addPoint(getArrowDeltaX(), 0);
291           } else if (part == ArrowPart.BOTTOM_PART) {
292               polygon.addPoint(getArrowDeltaX(), 1);
293               polygon.addPoint(0, 1);
294               polygon.addPoint(getArrowDeltaX(), getArrowDeltaY() + 1);
295           } else {
296               polygon.addPoint(getArrowDeltaX(), -getArrowDeltaY());
297               polygon.addPoint(0, 0);
298               polygon.addPoint(getArrowDeltaX(), getArrowDeltaY());
299               if (niceArrow)
300                   polygon.addPoint(getArrowDeltaX() - 4, 0);
301   
302           }
303           return polygon;
304       }

306       public XPoint2D getStartPoint(StringBounder stringBounder, XDimension2D dimensionToUse) {
307           final double y = getYPoint(stringBounder);
308           if (getDirection2() == ArrowDirection.LEFT_TO_RIGHT_NORMAL)
309               return new XPoint2D(getPaddingX(), y + inclination2);
310   
311           return new XPoint2D(dimensionToUse.getWidth() + getPaddingX(), y + inclination2);
312       }
313   
314       public XPoint2D getEndPoint(StringBounder stringBounder, XDimension2D dimensionToUse) {
315           final double y = getYPoint(stringBounder);
316           if (getDirection2() == ArrowDirection.LEFT_TO_RIGHT_NORMAL)
317               return new XPoint2D(dimensionToUse.getWidth() + getPaddingX(), y);
318   
319           return new XPoint2D(getPaddingX(), y);
320       }

322       @Override
323       public double getYPoint(StringBounder stringBounder) {
324           if (isBelowForResponse())
325               return getPaddingY();
326   
327           return getTextHeight(stringBounder) + getPaddingY();
328       }

337       }

340       public double getPreferredWidth(StringBounder stringBounder) {
