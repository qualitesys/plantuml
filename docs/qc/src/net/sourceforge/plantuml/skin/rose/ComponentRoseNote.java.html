36   package net.sourceforge.plantuml.skin.rose;
37   
38   import net.sourceforge.plantuml.ISkinSimple;
39   import net.sourceforge.plantuml.creole.Stencil;
40   import net.sourceforge.plantuml.cucadiagram.Display;
41   import net.sourceforge.plantuml.graphic.HorizontalAlignment;
42   import net.sourceforge.plantuml.graphic.StringBounder;
43   import net.sourceforge.plantuml.graphic.SymbolContext;
44   import net.sourceforge.plantuml.graphic.color.Colors;
45   import net.sourceforge.plantuml.skin.AbstractTextualComponent;
46   import net.sourceforge.plantuml.skin.Area;
47   import net.sourceforge.plantuml.style.PName;
48   import net.sourceforge.plantuml.style.Style;
49   import net.sourceforge.plantuml.svek.image.Opale;
50   import net.sourceforge.plantuml.ugraphic.UGraphic;
51   import net.sourceforge.plantuml.ugraphic.UGraphicStencil;
52   import net.sourceforge.plantuml.ugraphic.UPath;
53   import net.sourceforge.plantuml.ugraphic.UStroke;
54   import net.sourceforge.plantuml.ugraphic.UTranslate;
55   
56   final public class ComponentRoseNote extends AbstractTextualComponent implements Stencil {
57   
58       private final double paddingX;
59       private final double paddingY;
60       private final SymbolContext symbolContext;
61       private final double roundCorner;
62       private final HorizontalAlignment position;
63   
64       public ComponentRoseNote(Style style, Display strings, double paddingX, double paddingY,
65               ISkinSimple spriteContainer, HorizontalAlignment textAlignment, HorizontalAlignment position,
66               Colors colors) {
67           super(style, spriteContainer.wrapWidth(), textAlignment == HorizontalAlignment.CENTER ? 15 : 6, 15, 5,
68                   spriteContainer, strings, true);
69           this.paddingX = paddingX;
70           this.paddingY = paddingY;
71           this.position = position;
72           this.symbolContext = style.getSymbolContext(getIHtmlColorSet(), colors);
73           this.roundCorner = style.value(PName.RoundCorner).asInt();
74   
75       }
76   
77       @Override
78       final public double getPreferredWidth(StringBounder stringBounder) {
79           final double result = getTextWidth(stringBounder) + 2 * getPaddingX() + symbolContext.getDeltaShadow();
80           return result;
81       }
82   
83       @Override
84       final public double getPreferredHeight(StringBounder stringBounder) {
85           return getTextHeight(stringBounder) + 2 * getPaddingY() + symbolContext.getDeltaShadow();
86       }
87   
88       @Override
89       public double getPaddingX() {
90           return paddingX;
91       }
92   
93       @Override
94       public double getPaddingY() {
95           return paddingY;
96       }
97   
98       @Override
99       protected void drawInternalU(UGraphic ug, Area area) {
100   
101           final StringBounder stringBounder = ug.getStringBounder();
102           final int textHeight = (int) getTextHeight(stringBounder);
103   
104           int x2 = (int) getTextWidth(stringBounder);
105           final double diffX = area.getDimensionToUse().getWidth() - getPreferredWidth(stringBounder);
106           if (diffX < 0) {
107               throw new IllegalArgumentException();
108           }
109           if (area.getDimensionToUse().getWidth() > getPreferredWidth(stringBounder)) {
110               x2 = (int) (area.getDimensionToUse().getWidth() - 2 * getPaddingX());
111           }
112   
113           final UPath polygon = Opale.getPolygonNormal(x2, textHeight, roundCorner);
114           polygon.setDeltaShadow(symbolContext.getDeltaShadow());
115   
116           ug = symbolContext.apply(ug);
117           ug.draw(polygon);
118   
119           ug.draw(Opale.getCorner(x2, roundCorner));
120           UGraphic ug2 = UGraphicStencil.create(ug, this, new UStroke());
121   
122           if (position == HorizontalAlignment.LEFT) {
123               ug2 = ug2.apply(new UTranslate(getMarginX1(), getMarginY()));
124           } else if (position == HorizontalAlignment.RIGHT) {
125               ug2 = ug2.apply(
126                       new UTranslate(area.getDimensionToUse().getWidth() - getTextWidth(stringBounder), getMarginY()));
127           } else {
128               ug2 = ug2.apply(new UTranslate(getMarginX1() + diffX / 2, getMarginY()));
129           }
130   
131           getTextBlock().drawU(ug2);
132   
133       }
134   
135       public double getStartingX(StringBounder stringBounder, double y) {
136           return 0;
137       }
138   
139       public double getEndingX(StringBounder stringBounder, double y) {
140           return getTextWidth(stringBounder);
141       }
142   
143   }
