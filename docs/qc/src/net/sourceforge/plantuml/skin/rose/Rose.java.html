92       public Component createComponentNote(Style[] styles, ComponentType type, ISkinParam param, Display stringsToDisplay,
93               Colors colors, NotePosition notePosition) {
94           checkRose();
95           final HorizontalAlignment textAlign;
96           final HorizontalAlignment position;
97           if (notePosition == NotePosition.OVER_SEVERAL) {
98               textAlign = param.getHorizontalAlignment(AlignmentParam.noteTextAlignment, null, false,
99                       HorizontalAlignment.LEFT);
100               if (textAlign == param.getHorizontalAlignment(AlignmentParam.noteTextAlignment, null, false,
101                       HorizontalAlignment.CENTER))
102                   // Which means we use default
103                   position = textAlign;
104               else
105                   position = HorizontalAlignment.CENTER;
106           } else {
107               textAlign = param.getHorizontalAlignment(AlignmentParam.noteTextAlignment, null, false, null);
108               position = textAlign;
109           }
110   
111           if (type == ComponentType.NOTE)
112               return new ComponentRoseNote(styles[0], stringsToDisplay, paddingX, paddingY, param, textAlign, position,
113                       colors);
114   
115           if (type == ComponentType.NOTE_HEXAGONAL)
116               return new ComponentRoseNoteHexagonal(styles[0], stringsToDisplay, param, colors);
117   
118           if (type == ComponentType.NOTE_BOX)
119               return new ComponentRoseNoteBox(styles[0], stringsToDisplay, param, colors);
120   
121           throw new UnsupportedOperationException(type.toString());
122       }

270       public ArrowComponent createComponentArrow(Style[] styles, ArrowConfiguration config, ISkinParam param,
271               Display stringsToDisplay) {
272           checkRose();
273   
274           if (config.getArrowDirection() == ArrowDirection.SELF)
275               return new ComponentRoseSelfArrow(styles[0], stringsToDisplay, config, param, param.maxMessageSize(),
276                       param.strictUmlStyle() == false);
277   
278           final ArrowDirection arrowDirection = config.getArrowDirection();
279   
280           final StyleSignatureBasic signature = StyleSignatureBasic.of(SName.root, SName.element, SName.sequenceDiagram,
281                   SName.arrow);
282           final Style textStyle = signature.getMergedStyle(param.getCurrentStyleBuilder());
283           final String value = textStyle.value(PName.HorizontalAlignment).asString();
284           HorizontalAlignment messageHorizontalAlignment = textStyle.getHorizontalAlignment();
285   
286           if ("first".equalsIgnoreCase(value)) {
287               final boolean isReverseDefine = config.isReverseDefine();
288               if (arrowDirection == ArrowDirection.RIGHT_TO_LEFT_REVERSE) {
289                   if (isReverseDefine)
290                       messageHorizontalAlignment = HorizontalAlignment.LEFT;
291                   else
292                       messageHorizontalAlignment = HorizontalAlignment.RIGHT;
293   
294               } else {
295                   if (isReverseDefine)
296                       messageHorizontalAlignment = HorizontalAlignment.RIGHT;
297                   else
298                       messageHorizontalAlignment = HorizontalAlignment.LEFT;
299   
300               }
301           } else if ("direction".equalsIgnoreCase(value)) {
302               if (arrowDirection == ArrowDirection.LEFT_TO_RIGHT_NORMAL)
303                   messageHorizontalAlignment = HorizontalAlignment.LEFT;
304               else if (arrowDirection == ArrowDirection.RIGHT_TO_LEFT_REVERSE)
305                   messageHorizontalAlignment = HorizontalAlignment.RIGHT;
306               else if (arrowDirection == ArrowDirection.BOTH_DIRECTION)
307                   messageHorizontalAlignment = HorizontalAlignment.CENTER;
308   
309           } else if ("reversedirection".equalsIgnoreCase(value)) {
310               if (arrowDirection == ArrowDirection.LEFT_TO_RIGHT_NORMAL)
311                   messageHorizontalAlignment = HorizontalAlignment.RIGHT;
312               else if (arrowDirection == ArrowDirection.RIGHT_TO_LEFT_REVERSE)
313                   messageHorizontalAlignment = HorizontalAlignment.LEFT;
314               else if (arrowDirection == ArrowDirection.BOTH_DIRECTION)
315                   messageHorizontalAlignment = HorizontalAlignment.CENTER;
316   
317           }
318   
319           return new ComponentRoseArrow(styles[0], stringsToDisplay, config, messageHorizontalAlignment, param,
320                   param.maxMessageSize(), param.strictUmlStyle() == false, param.responseMessageBelowArrow());
321       }
