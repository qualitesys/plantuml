36   package net.sourceforge.plantuml.skin.rose;
37   
38   import net.sourceforge.plantuml.ISkinSimple;
39   import net.sourceforge.plantuml.LineBreakStrategy;
40   import net.sourceforge.plantuml.cucadiagram.Display;
41   import net.sourceforge.plantuml.graphic.StringBounder;
42   import net.sourceforge.plantuml.graphic.SymbolContext;
43   import net.sourceforge.plantuml.graphic.TextBlock;
44   import net.sourceforge.plantuml.skin.AbstractTextualComponent;
45   import net.sourceforge.plantuml.skin.Area;
46   import net.sourceforge.plantuml.style.PName;
47   import net.sourceforge.plantuml.style.Style;
48   import net.sourceforge.plantuml.ugraphic.Shadowable;
49   import net.sourceforge.plantuml.ugraphic.UGraphic;
50   import net.sourceforge.plantuml.ugraphic.URectangle;
51   import net.sourceforge.plantuml.ugraphic.UStroke;
52   import net.sourceforge.plantuml.ugraphic.UTranslate;
53   import net.sourceforge.plantuml.ugraphic.color.HColor;
54   
55   public class ComponentRoseParticipant extends AbstractTextualComponent {
56   
57       private final HColor back;
58       private final HColor foregroundColor;
59       private final double deltaShadow;
60       private final double roundCorner;
61       private final double diagonalCorner;
62       private final UStroke stroke;
63       private final double minWidth;
64       private final boolean collections;
65       private final double padding;
66   
67       public ComponentRoseParticipant(Style style, Style stereo, Display stringsToDisplay, ISkinSimple spriteContainer,
68               double minWidth, boolean collections, double padding) {
69           super(style, stereo, LineBreakStrategy.NONE, 7, 7, 7, spriteContainer, stringsToDisplay, false);
70   
71           this.roundCorner = style.value(PName.RoundCorner).asInt();
72           this.diagonalCorner = style.value(PName.DiagonalCorner).asInt();
73           final SymbolContext biColor = style.getSymbolContext(getIHtmlColorSet());
74           this.stroke = style.getStroke();
75   
76           this.padding = padding;
77           this.minWidth = minWidth;
78           this.collections = collections;
79           this.back = biColor.getBackColor();
80           this.deltaShadow = biColor.getDeltaShadow();
81           this.foregroundColor = biColor.getForeColor();
82       }
83   
84       @Override
85       protected void drawInternalU(UGraphic ug, Area area) {
86           final StringBounder stringBounder = ug.getStringBounder();
87           ug = ug.apply(UTranslate.dx(padding));
88           if (foregroundColor != null)
89               ug = ug.apply(foregroundColor);
90           if (back != null)
91               ug = ug.apply(back.bg());
92           ug = ug.apply(stroke);
93           final Shadowable rect = new URectangle(getTextWidth(stringBounder), getTextHeight(stringBounder))
94                   .rounded(roundCorner).diagonalCorner(diagonalCorner);
95           rect.setDeltaShadow(deltaShadow);
96           if (collections) {
97               ug.apply(UTranslate.dx(getDeltaCollection())).draw(rect);
98               ug = ug.apply(UTranslate.dy(getDeltaCollection()));
99           }
100           ug.draw(rect);
101           ug = ug.apply(new UStroke());
102           final TextBlock textBlock = getTextBlock();
103           textBlock.drawU(ug.apply(new UTranslate(getMarginX1() + suppWidth(stringBounder) / 2, getMarginY())));
104       }
105   
106       private double getDeltaCollection() {
107           if (collections)
108               return 4;
109   
110           return 0;
111       }
112   
113       @Override
114       public double getPreferredHeight(StringBounder stringBounder) {
115           return getTextHeight(stringBounder) + deltaShadow + 1 + getDeltaCollection();
116       }
117   
118       @Override
119       public double getPreferredWidth(StringBounder stringBounder) {
120           return getTextWidth(stringBounder) + deltaShadow + getDeltaCollection() + 2 * padding;
121       }
122   
123       @Override
124       protected double getPureTextWidth(StringBounder stringBounder) {
125           return Math.max(super.getPureTextWidth(stringBounder), minWidth);
126       }
127   
128       private final double suppWidth(StringBounder stringBounder) {
129           return getPureTextWidth(stringBounder) - super.getPureTextWidth(stringBounder);
130       }
131   
132   }
