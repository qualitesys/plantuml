62       private ArrowConfiguration(ArrowBody body, ArrowDressing dressing1, ArrowDressing dressing2,
63               ArrowDecoration decoration1, ArrowDecoration decoration2, HColor color, boolean isSelf, double thickness,
64               boolean reverseDefine, int inclination) {
65           this.reverseDefine = reverseDefine;
66           this.thickness = thickness;
67           this.body = Objects.requireNonNull(body);
68           this.dressing1 = Objects.requireNonNull(dressing1);
69           this.dressing2 = Objects.requireNonNull(dressing2);
70           this.decoration1 = decoration1;
71           this.decoration2 = decoration2;
72           this.color = color;
73           this.isSelf = isSelf;
74           this.inclination = inclination;
75       }

180       public final ArrowDirection getArrowDirection() {
181           if (isSelf)
182               return ArrowDirection.SELF;
183   
184           if (this.dressing1.getHead() == ArrowHead.NONE && this.dressing2.getHead() != ArrowHead.NONE)
185               return ArrowDirection.LEFT_TO_RIGHT_NORMAL;
186   
187           if (this.dressing1.getHead() != ArrowHead.NONE && this.dressing2.getHead() == ArrowHead.NONE)
188               return ArrowDirection.RIGHT_TO_LEFT_REVERSE;
189   
190           return ArrowDirection.BOTH_DIRECTION;
191       }
