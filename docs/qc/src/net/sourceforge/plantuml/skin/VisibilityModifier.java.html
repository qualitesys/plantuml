116       private void drawInternal(UGraphic ug, int size, final HColor foregroundColor, final HColor backgoundColor,
117               double x, double y) {
118           if (backgoundColor == null)
119               ug = ug.apply(HColors.none().bg());
120           else
121               ug = ug.apply(backgoundColor.bg());
122   
123           ug = ug.apply(foregroundColor);
124           size = ensureEven(size);
125           switch (this) {
126           case PACKAGE_PRIVATE_FIELD:
127               drawTriangle(ug, false, size, x, y);
128               break;
129   
130           case PRIVATE_FIELD:
131               drawSquare(ug, false, size, x, y);
132               break;
133   
134           case PROTECTED_FIELD:
135               drawDiamond(ug, false, size, x, y);
136               break;
137   
138           case PUBLIC_FIELD:
139               drawCircle(ug, false, size, x, y);
140               break;
141   
142           case PACKAGE_PRIVATE_METHOD:
143               drawTriangle(ug, true, size, x, y);
144               break;
145   
146           case PRIVATE_METHOD:
147               drawSquare(ug, true, size, x, y);
148               break;
149   
150           case PROTECTED_METHOD:
151               drawDiamond(ug, true, size, x, y);
152               break;
153   
154           case PUBLIC_METHOD:
155               drawCircle(ug, true, size, x, y);
156               break;
157   
158           case IE_MANDATORY:
159               drawCircle(ug, true, size, x, y);
160               break;
161   
162           default:
163               throw new IllegalStateException();
164           }
165       }

175       static private int ensureEven(int n) {
176           if (n % 2 == 1)
177               n--;
178           return n;
179       }

181       private void drawDiamond(UGraphic ug, boolean filled, int size, double x, double y) {
182           final UPolygon poly = new UPolygon();
183           size -= 2;
184           poly.addPoint(size / 2.0, 0);
185           poly.addPoint(size, size / 2.0);
186           poly.addPoint(size / 2.0, size);
187           poly.addPoint(0, size / 2.0);
188           ug.apply(new UTranslate(x + 1, y)).draw(poly);
189       }
190   
191       private void drawTriangle(UGraphic ug, boolean filled, int size, double x, double y) {
192           final UPolygon poly = new UPolygon();
193           size -= 2;
194           poly.addPoint(size / 2.0, 1);
195           poly.addPoint(0, size - 1);
196           poly.addPoint(size, size - 1);
197           ug.apply(new UTranslate(x + 1, y)).draw(poly);
198       }

200       public static boolean isVisibilityCharacter(CharSequence s) {
201           if (s.length() <= 2)
202               return false;
203   
204           final char c = s.charAt(0);
205           if (s.charAt(1) == c)
206               return false;
207   
208           if (c == '-')
209               return true;
210   
211           if (c == '#')
212               return true;
213   
214           if (c == '+')
215               return true;
216   
217           if (c == '~')
218               return true;
219   
220           if (c == '*')
221               return true;
222   
223           return false;
224       }

226       public static VisibilityModifier getByUnicode(char c) {
227           for (VisibilityModifier modifier : VisibilityModifier.values())
228               if (modifier.unicode == c)
229                   return modifier;
230   
231           return null;
232       }

234       public static String replaceVisibilityModifierByUnicodeChar(String s, boolean isField) {
235           final VisibilityModifier modifier = getVisibilityModifier(s, isField);
236           if (modifier == null)
237               return s;
238   
239           return "" + modifier.unicode + s.substring(1);
240       }

242       public static VisibilityModifier getVisibilityModifier(CharSequence s, boolean isField) {
243           if (s.length() <= 2)
244               return null;
245   
246           final char c = s.charAt(0);
247           if (s.charAt(1) == c)
248               return null;
249   
250           if (isField)
251               return getVisibilityModifierForField(c);
252   
253           return getVisibilityModifierForMethod(c);
254       }

256       private static VisibilityModifier getVisibilityModifierForField(char c) {
257           if (c == '-')
258               return VisibilityModifier.PRIVATE_FIELD;
259   
260           if (c == '#')
261               return VisibilityModifier.PROTECTED_FIELD;
262   
263           if (c == '+')
264               return VisibilityModifier.PUBLIC_FIELD;
265   
266           if (c == '~')
267               return VisibilityModifier.PACKAGE_PRIVATE_FIELD;
268   
269           if (c == '*')
270               return VisibilityModifier.IE_MANDATORY;
271   
272           return null;
273       }
274   
275       private static VisibilityModifier getVisibilityModifierForMethod(char c) {
276           if (c == '-')
277               return VisibilityModifier.PRIVATE_METHOD;
278   
279           if (c == '#')
280               return VisibilityModifier.PROTECTED_METHOD;
281   
282           if (c == '+')
283               return VisibilityModifier.PUBLIC_METHOD;
284   
285           if (c == '~')
286               return VisibilityModifier.PACKAGE_PRIVATE_METHOD;
287   
288           if (c == '*')
289               return VisibilityModifier.IE_MANDATORY;
290   
291           return null;
292       }

306           if (this == PRIVATE_FIELD || this == PRIVATE_METHOD)

310               return "protected";
