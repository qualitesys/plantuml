80           return RegexConcat.build(CommandLinkActivity.class.getName(), RegexLeaf.start(), //
81                   new RegexOptional(//

86                                   new RegexLeaf("QUOTED", "[%g]([^%g]+)[%g](?:[%s]+as[%s]+([%pLN_.]+))?"))), //
87                   RegexLeaf.spaceZeroOrMore(), //
88                   new RegexLeaf("STEREOTYPE", "(\\<\\<.*\\>\\>)?"), //

125       @Override
126       protected CommandExecutionResult executeArg(ActivityDiagram diagram, LineLocation location, RegexResult arg)
127               throws NoSuchColorException {
128           final IEntity entity1 = getEntity(diagram, arg, true);
129           if (entity1 == null)
130               return CommandExecutionResult.error("No such activity");
131   
132           if (arg.get("STEREOTYPE", 0) != null)
133               entity1.setStereotype(Stereotype.build(arg.get("STEREOTYPE", 0)));
134   
135           if (arg.get("BACKCOLOR", 0) != null) {
136               String s = arg.get("BACKCOLOR", 0);
137               entity1.setSpecificColorTOBEREMOVED(ColorType.BACK,
138                       diagram.getSkinParam().getIHtmlColorSet().getColor(s));
139           }
140   
141           final IEntity entity2 = getEntity(diagram, arg, false);
142           if (entity2 == null)
143               return CommandExecutionResult.error("No such activity");
144   
145           if (arg.get("BACKCOLOR2", 0) != null) {
146               String s = arg.get("BACKCOLOR2", 0);
147               entity2.setSpecificColorTOBEREMOVED(ColorType.BACK,
148                       diagram.getSkinParam().getIHtmlColorSet().getColor(s));
149           }
150           if (arg.get("STEREOTYPE2", 0) != null)
151               entity2.setStereotype(Stereotype.build(arg.get("STEREOTYPE2", 0)));
152   
153           final Display linkLabel = Display.getWithNewlines(arg.get("BRACKET", 0));
154   
155           final String arrowBody1 = CommandLinkClass.notNull(arg.get("ARROW_BODY1", 0));
156           final String arrowBody2 = CommandLinkClass.notNull(arg.get("ARROW_BODY2", 0));
157           final String arrowDirection = CommandLinkClass.notNull(arg.get("ARROW_DIRECTION", 0));
158   
159           final String arrow = StringUtils.manageArrowForCuca(arrowBody1 + arrowDirection + arrowBody2 + ">");
160           int lenght = arrow.length() - 1;
161           if (arrowDirection.contains("*"))
162               lenght = 2;
163   
164           LinkType type = new LinkType(LinkDecor.ARROW, LinkDecor.NONE);
165           if ((arrowBody1 + arrowBody2).contains("."))
166               type = type.goDotted();
167   
168           final LinkArg linkArg = LinkArg.build(linkLabel, lenght, diagram.getSkinParam().classAttributeIconSize() > 0);
169           Link link = new Link(diagram.getSkinParam().getCurrentStyleBuilder(), entity1, entity2, type, linkArg);
170           if (arrowDirection.contains("*"))
171               link.setConstraint(false);
172   
173           final Direction direction = StringUtils.getArrowDirection(arrowBody1 + arrowDirection + arrowBody2 + ">");
174           if (direction == Direction.LEFT || direction == Direction.UP)
175               link = link.getInv();
176   
177           if (arg.get("URL", 0) != null) {
178               final UrlBuilder urlBuilder = new UrlBuilder(diagram.getSkinParam().getValue("topurl"), UrlMode.STRICT);
179               final Url urlLink = urlBuilder.getUrl(arg.get("URL", 0));
180               link.setUrl(urlLink);
181           }
182   
183           link.applyStyle(arg.getLazzy("ARROW_STYLE", 0));
184           diagram.addLink(link);
185   
186           return CommandExecutionResult.ok();
187   
188       }

190       static IEntity getEntity(ActivityDiagram diagram, RegexResult arg, final boolean start) {
191           final String suf = start ? "" : "2";
192   
193           final String openBracket2 = arg.get("OPENBRACKET" + suf, 0);
194           if (openBracket2 != null)
195               return diagram.createInnerActivity();
196   
197           if (arg.get("STAR" + suf, 0) != null) {
198               final String suppId = arg.get("STAR" + suf, 1);
199               if (start) {
200   //                if (suppId != null)
201   //                    diagram.getStart().setTop(true);
202                   return diagram.getStart();
203               }
204               return diagram.getEnd(suppId);
205           }
206           String partition = arg.get("PARTITION" + suf, 0);
207           if (partition != null)
208               partition = StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(partition);
209   
210           final String idShort = arg.get("CODE" + suf, 0);
211           if (idShort != null) {
212               if (partition != null) {
213                   final Ident idNewLong = diagram.buildLeafIdent(partition);
214                   final Code codeP = diagram.V1972() ? idNewLong : diagram.buildCode(partition);
215                   diagram.gotoGroup(idNewLong, codeP, Display.getWithNewlines(partition), GroupType.PACKAGE,
216                           diagram.getRootGroup(), NamespaceStrategy.SINGLE);
217               }
218               final Ident ident = diagram.buildLeafIdent(idShort);
219               final Code code = diagram.V1972() ? ident : diagram.buildCode(idShort);
220               final LeafType type = diagram.V1972() ? getTypeIfExistingSmart(diagram, ident)
221                       : getTypeIfExisting(diagram, code);
222               IEntity result;
223               if (diagram.V1972()) {
224                   result = diagram.getLeafVerySmart(ident);
225                   if (result == null)
226                       result = diagram.getOrCreate(ident, code, Display.getWithNewlines(code), type);
227               } else
228                   result = diagram.getOrCreate(ident, code, Display.getWithNewlines(code), type);
229               if (partition != null)
230                   diagram.endGroup();
231   
232               return result;
233           }
234           final String bar = arg.get("BAR" + suf, 0);
235           if (bar != null) {
236               final Ident identBar = diagram.buildLeafIdent(bar);
237               final Code codeBar = diagram.V1972() ? identBar : diagram.buildCode(bar);
238               if (diagram.V1972()) {
239                   final ILeaf result = diagram.getLeafVerySmart(identBar);
240                   if (result != null)
241                       return result;
242   
243               }
244               return diagram.getOrCreate(identBar, codeBar, Display.getWithNewlines(bar), LeafType.SYNCHRO_BAR);
245           }
246           final RegexPartialMatch quoted = arg.get("QUOTED" + suf);
247           if (quoted.get(0) != null) {
248               final String quotedString = quoted.get(1) == null ? quoted.get(0) : quoted.get(1);
249               if (partition != null) {
250                   final Ident idNewLong = diagram.buildLeafIdent(partition);
251                   final Code codeP = diagram.V1972() ? idNewLong : diagram.buildCode(partition);
252                   diagram.gotoGroup(idNewLong, codeP, Display.getWithNewlines(partition), GroupType.PACKAGE,
253                           diagram.getRootGroup(), NamespaceStrategy.SINGLE);
254               }
255               final Ident quotedIdent = diagram.buildLeafIdent(quotedString);
256               final Code quotedCode = diagram.V1972() ? quotedIdent : diagram.buildCode(quotedString);
257               final LeafType type = diagram.V1972() ? getTypeIfExistingSmart(diagram, quotedIdent)
258                       : getTypeIfExisting(diagram, quotedCode);
259               final IEntity result = diagram.getOrCreate(quotedIdent, quotedCode, Display.getWithNewlines(quoted.get(0)),
260                       type);
261               if (partition != null)
262                   diagram.endGroup();
263   
264               return result;
265           }
266           final String quoteInvisibleString = arg.get("QUOTED_INVISIBLE" + suf, 0);
267           if (quoteInvisibleString != null) {
268               if (partition != null) {
269                   final Ident idNewLong = diagram.buildLeafIdent(partition);
270                   final Code codeP = diagram.V1972() ? idNewLong : diagram.buildCode(partition);
271                   diagram.gotoGroup(idNewLong, codeP, Display.getWithNewlines(partition), GroupType.PACKAGE,
272                           diagram.getRootGroup(), NamespaceStrategy.SINGLE);
273               }
274               final Ident identInvisible = diagram.buildLeafIdent(quoteInvisibleString);
275               final Code quotedInvisible = diagram.V1972() ? identInvisible : diagram.buildCode(quoteInvisibleString);
276               final IEntity result = diagram.getOrCreate(identInvisible, quotedInvisible,
277                       Display.getWithNewlines(quotedInvisible), LeafType.ACTIVITY);
278               if (partition != null)
279                   diagram.endGroup();
280   
281               return result;
282           }
283           final String first = arg.get("FIRST" + suf, 0);
284           if (first == null)
285               return diagram.getLastEntityConsulted();
286   
287           return null;
288       }

290       private static LeafType getTypeIfExistingSmart(ActivityDiagram system, Ident ident) {
291           final IEntity ent = system.getLeafSmart(ident);
292           if (ent != null)
293               if (ent.getLeafType() == LeafType.BRANCH)
294                   return LeafType.BRANCH;
295   
296           return LeafType.ACTIVITY;
297       }
298   
299       private static LeafType getTypeIfExisting(ActivityDiagram system, Code code) {
300           if (system.leafExist(code)) {
301               final IEntity ent = system.getLeaf(code);
302               if (ent.getLeafType() == LeafType.BRANCH)
303                   return LeafType.BRANCH;
304   
305           }
306           return LeafType.ACTIVITY;
307       }
