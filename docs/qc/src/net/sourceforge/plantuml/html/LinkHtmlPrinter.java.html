38   import java.io.PrintWriter;

40   import net.sourceforge.plantuml.StringUtils;

46   public final class LinkHtmlPrinter {
47   
48       private final Link link;
49       // private final Entity entity;
50       private final boolean chiral;
51   
52       public LinkHtmlPrinter(Link link, IEntity entity) {
53           this.link = link;
54           if (link.getEntity1() == entity) {
55               this.chiral = false;
56           } else if (link.getEntity2() == entity) {
57               this.chiral = true;
58           } else {
59               throw new IllegalArgumentException();
60           }
61       }
62   
63       void printLink(PrintWriter pw) {
64           final String ent1h;
65           final String ent2h;
66           if (chiral) {
67               ent1h = htmlLink(link.getEntity1());
68               ent2h = "<i>" + StringUtils.unicodeForHtml(link.getEntity2().getCodeGetName()) + "</i>";
69           } else {
70               ent1h = "<i>" + StringUtils.unicodeForHtml(link.getEntity1().getCodeGetName()) + "</i>";
71               ent2h = htmlLink(link.getEntity2());
72           }
73           String label = link.getLabel() == null ? null : StringUtils.unicodeForHtml(link.getLabel());
74           String ent1 = ent1h;
75           String ent2 = ent2h;
76           if (link.getQuantifier1() != null) {
77               ent1 = ent1 + " (" + link.getQuantifier1() + ")";
78               if (label != null) {
79                   label = "(" + link.getQuantifier1() + " " + ent1h + ") " + label;
80               }
81           }
82           if (link.getQuantifier2() != null) {
83               ent2 = ent2 + " (" + link.getQuantifier2() + ")";
84               if (label != null) {
85                   label = label + " (" + link.getQuantifier2() + " " + ent2h + ")";
86               }
87           }
88           if (chiral) {
89               pw.println(getHtmlChiral(ent1, ent2));
90           } else {
91               pw.println(getHtml(ent1, ent2));
92           }
93           if (label != null) {
94               pw.println("&nbsp;:&nbsp;");
95               pw.println(label);
96           }
97       }
98   
99       private String getHtml(String ent1, String ent2) {
100           final LinkDecor decor1 = link.getType().getDecor1();
101           final LinkDecor decor2 = link.getType().getDecor2();
102   
103           if (decor1 == LinkDecor.NONE && decor2 == LinkDecor.NONE) {
104               return ent1 + " is linked to " + ent2;
105           }
106           if (decor1 == LinkDecor.NONE && decor2 == LinkDecor.EXTENDS) {
107               return ent1 + " is extended by " + ent2;
108           }
109           if (decor1 == LinkDecor.EXTENDS && decor2 == LinkDecor.NONE) {
110               return ent1 + " extends " + ent2;
111           }
112           if (decor2 == LinkDecor.AGREGATION) {
113               return ent1 + " is aggregated by  " + ent2;
114           }
115           if (decor1 == LinkDecor.AGREGATION) {
116               return ent1 + " aggregates " + ent2;
117           }
118           if (decor2 == LinkDecor.COMPOSITION) {
119               return ent1 + " is composed by " + ent2;
120           }
121           if (decor1 == LinkDecor.COMPOSITION) {
122               return ent1 + " composes " + ent2;
123           }
124           if (decor1 == LinkDecor.NONE && decor2 == LinkDecor.ARROW) {
125               return ent1 + " is navigable from  " + ent2;
126           }
127           if (decor1 == LinkDecor.ARROW && decor2 == LinkDecor.NONE) {
128               return ent1 + " navigates to  " + ent2;
129           }
130           return ent1 + " " + decor1 + "-" + decor2 + " " + ent2;
131   
132       }
133   
134       private String getHtmlChiral(String ent1, String ent2) {
135           final LinkDecor decor1 = link.getType().getDecor1();
136           final LinkDecor decor2 = link.getType().getDecor2();
137   
138           if (decor1 == LinkDecor.NONE && decor2 == LinkDecor.NONE) {
139               return ent2 + " is linked to " + ent1;
140           }
141           if (decor1 == LinkDecor.NONE && decor2 == LinkDecor.EXTENDS) {
142               return ent2 + " extends " + ent1;
143           }
144           if (decor1 == LinkDecor.EXTENDS && decor2 == LinkDecor.NONE) {
145               return ent2 + " is extended by " + ent1;
146           }
147           if (decor2 == LinkDecor.AGREGATION) {
148               return ent2 + " aggregates " + ent1;
149           }
150           if (decor1 == LinkDecor.AGREGATION) {
151               return ent2 + " is aggregated by " + ent1;
152           }
153           if (decor2 == LinkDecor.COMPOSITION) {
154               return ent2 + " composes " + ent1;
155           }
156           if (decor1 == LinkDecor.COMPOSITION) {
157               return ent2 + " is composed by " + ent1;
158           }
159           if (decor1 == LinkDecor.NONE && decor2 == LinkDecor.ARROW) {
160               return ent2 + " navigates to  " + ent1;
161           }
162           if (decor1 == LinkDecor.ARROW && decor2 == LinkDecor.NONE) {
163               return ent2 + " is navigable from  " + ent1;
164           }
165           return ent1 + " " + decor1 + "-" + decor2 + " " + ent2;
166       }

168       static String htmlLink(IEntity ent) {
169           final StringBuilder sb = new StringBuilder();
170           sb.append("<a href=\"");
171           sb.append(urlOf(ent));
172           sb.append("\">");
173           sb.append(StringUtils.unicodeForHtml(ent.getCodeGetName()));
174           sb.append("</a>");
175           return sb.toString();
176       }
177   
178       static String urlOf(IEntity ent) {
179           if (ent.getLeafType() == LeafType.NOTE) {
180               throw new IllegalArgumentException();
181           }
182           if (ent.getCodeGetName().matches("[-\\w_ .]+")) {
183               return StringUtils.unicodeForHtml(ent.getCodeGetName()) + ".html";
184           }
185           return StringUtils.unicodeForHtml(ent.getUid()) + ".html";
186       }
