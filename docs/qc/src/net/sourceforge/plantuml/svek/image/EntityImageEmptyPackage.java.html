71   public class EntityImageEmptyPackage extends AbstractEntityImage {
72   
73       private final TextBlock desc;
74       private final static int MARGIN = 10;
75   
76       private final Stereotype stereotype;
77       private final TextBlock stereoBlock;
78       private final Url url;
79       private final SName sname;
80       private final double shadowing;
81       private final HColor borderColor;
82       private final UStroke stroke;
83       private final double roundCorner;
84       private final double diagonalCorner;
85       private final HColor back;
86   
87       private Style getStyle() {
88           return StyleSignatureBasic.of(SName.root, SName.element, sname, SName.package_, SName.title)
89                   .withTOBECHANGED(stereotype).getMergedStyle(getSkinParam().getCurrentStyleBuilder());
90       }
91   
92       public EntityImageEmptyPackage(ILeaf entity, ISkinParam skinParam, PortionShower portionShower, SName sname) {
93           super(entity, skinParam);
94           this.sname = sname;
95   
96           final Colors colors = entity.getColors();
97           final HColor specificBackColor = colors.getColor(ColorType.BACK);
98           this.stereotype = entity.getStereotype();
99           this.url = entity.getUrl99();
100   
101           Style style = getStyle();
102           style = style.eventuallyOverride(colors);
103           this.borderColor = style.value(PName.LineColor).asColor(getSkinParam().getIHtmlColorSet());
104           this.shadowing = style.value(PName.Shadowing).asDouble();
105           this.stroke = style.getStroke(colors);
106           this.roundCorner = style.value(PName.RoundCorner).asDouble();
107           this.diagonalCorner = style.value(PName.DiagonalCorner).asDouble();
108           if (specificBackColor == null)
109               this.back = style.value(PName.BackGroundColor).asColor(getSkinParam().getIHtmlColorSet());
110           else
111               this.back = specificBackColor;
112   
113           final FontConfiguration titleFontConfiguration = style.getFontConfiguration(getSkinParam().getIHtmlColorSet());
114           final HorizontalAlignment titleHorizontalAlignment = style.getHorizontalAlignment();
115   
116           this.desc = entity.getDisplay().create(titleFontConfiguration, titleHorizontalAlignment, skinParam);
117   
118           final DisplayPositioned legend = ((EntityImpl) entity).getLegend();
119           if (legend != null) {
120               final TextBlock legendBlock = EntityImageLegend.create(legend.getDisplay(), skinParam);
121               stereoBlock = legendBlock;
122           } else {
123               if (stereotype == null || stereotype.getLabel(Guillemet.DOUBLE_COMPARATOR) == null
124                       || portionShower.showPortion(EntityPortion.STEREOTYPE, entity) == false)
125                   stereoBlock = TextBlockUtils.empty(0, 0);
126               else
127                   stereoBlock = TextBlockUtils.withMargin(Display.create(stereotype.getLabels(skinParam.guillemet()))
128                           .create(FontConfiguration.create(getSkinParam(), FontParam.PACKAGE_STEREOTYPE, stereotype),
129                                   titleHorizontalAlignment, skinParam),
130                           1, 0);
131           }
132   
133       }
134   
135       public XDimension2D calculateDimension(StringBounder stringBounder) {
136           final XDimension2D dimDesc = desc.calculateDimension(stringBounder);
137           XDimension2D dim = TextBlockUtils.mergeTB(desc, stereoBlock, HorizontalAlignment.LEFT)
138                   .calculateDimension(stringBounder);
139           dim = XDimension2D.atLeast(dim, 0, 2 * dimDesc.getHeight());
140           return XDimension2D.delta(dim, MARGIN * 2, MARGIN * 2);
141       }
142   
143       final public void drawU(UGraphic ug) {
144           if (url != null)
145               ug.startUrl(url);
146   
147           final StringBounder stringBounder = ug.getStringBounder();
148           final XDimension2D dimTotal = calculateDimension(stringBounder);
149   
150           final double widthTotal = dimTotal.getWidth();
151           final double heightTotal = dimTotal.getHeight();
152   
153           final ClusterPosition clusterPosition = new ClusterPosition(0, 0, widthTotal, heightTotal);
154           final ClusterDecoration decoration = new ClusterDecoration(getSkinParam().packageStyle(), null, desc,
155                   stereoBlock, clusterPosition, stroke);
156   
157           final HorizontalAlignment horizontalAlignment = getSkinParam()
158                   .getHorizontalAlignment(AlignmentParam.packageTitleAlignment, null, false, null);
159           final HorizontalAlignment stereotypeAlignment = getSkinParam().getStereotypeAlignment();
160   
161           decoration.drawU(ug, back, borderColor, shadowing, roundCorner, horizontalAlignment, stereotypeAlignment,
162                   diagonalCorner);
163   
164           if (url != null)
165               ug.closeUrl();
166   
167       }
168   
169       public ShapeType getShapeType() {
170           return ShapeType.RECTANGLE;
171       }
