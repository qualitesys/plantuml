96       }

98       public XDimension2D calculateDimension(StringBounder stringBounder) {
99           final double height = getHeight(stringBounder);
100           final double width = getWidth(stringBounder);
101           return new XDimension2D(width, height);
102       }

104       final public void drawU(UGraphic ug) {
105           final StringBounder stringBounder = ug.getStringBounder();
106           ug = ug.apply(noteBackgroundColor.bg()).apply(borderColor);
107           final UPath polygon;
108           if (withLink == false)
109               polygon = getPolygonNormal(stringBounder);
110           else if (strategy == Direction.LEFT)
111               polygon = getPolygonLeft(stringBounder, pp1, pp2);
112           else if (strategy == Direction.RIGHT)
113               polygon = getPolygonRight(stringBounder, pp1, pp2);
114           else if (strategy == Direction.UP)
115               polygon = getPolygonUp(stringBounder, pp1, pp2);
116           else if (strategy == Direction.DOWN)
117               polygon = getPolygonDown(stringBounder, pp1, pp2);
118           else
119               throw new IllegalArgumentException();
120   
121           polygon.setDeltaShadow(shadowing2);
122           if (stroke != null)
123               ug = ug.apply(stroke);
124           ug.draw(polygon);
125           ug.draw(getCorner(getWidth(stringBounder), roundCorner));
126           textBlock.drawU(ug.apply(new UTranslate(marginX1, marginY)));
127       }

129       private UPath getPolygonNormal(final StringBounder stringBounder) {
130           return getPolygonNormal(getWidth(stringBounder), getHeight(stringBounder), roundCorner);
131       }

133       public static UPath getCorner(double width, double roundCorner) {
134           final UPath path = new UPath();
135           path.moveTo(width - cornersize, 0);
136           if (roundCorner == 0) {
137               path.lineTo(width - cornersize, cornersize);
138           } else {
139               path.lineTo(width - cornersize, cornersize - roundCorner / 4);
140               path.arcTo(new XPoint2D(width - cornersize + roundCorner / 4, cornersize), roundCorner / 4, 0, 0);
141           }
142           path.lineTo(width, cornersize);
143           path.lineTo(width - cornersize, 0);
144           path.closePath();
145           return path;
146       }
147   
148       public static UPath getPolygonNormal(double width, double height, double roundCorner) {
149           final UPath polygon = new UPath();
150           if (roundCorner == 0) {
151               polygon.moveTo(0, 0);
152               polygon.lineTo(0, height);
153               polygon.lineTo(width, height);
154               polygon.lineTo(width, cornersize);
155               polygon.lineTo(width - cornersize, 0);
156               polygon.lineTo(0, 0);
157           } else {
158               polygon.moveTo(0, roundCorner / 2);
159               polygon.lineTo(0, height - roundCorner / 2);
160               polygon.arcTo(new XPoint2D(roundCorner / 2, height), roundCorner / 2, 0, 0);
161               polygon.lineTo(width - roundCorner / 2, height);
162               polygon.arcTo(new XPoint2D(width, height - roundCorner / 2), roundCorner / 2, 0, 0);
163               polygon.lineTo(width, cornersize);
164               polygon.lineTo(width - cornersize, 0);
165               polygon.lineTo(roundCorner / 2, 0);
166               polygon.arcTo(new XPoint2D(0, roundCorner / 2), roundCorner / 2, 0, 0);
167           }
168           polygon.closePath();
169           return polygon;
170       }

174       private UPath getPolygonLeft(final StringBounder stringBounder, final XPoint2D pp1, final XPoint2D pp2) {
175           final UPath polygon = new UPath();
176           polygon.moveTo(0, roundCorner / 2);
177   
178           double y1 = pp1.getY() - delta;
179           y1 = MathUtils.limitation(y1, 0, getHeight(stringBounder) - 2 * delta);
180           polygon.lineTo(0, y1);
181           polygon.lineTo(pp2.getX(), pp2.getY());
182           polygon.lineTo(0, y1 + 2 * delta);
183   
184           polygon.lineTo(0, getHeight(stringBounder) - roundCorner / 2);
185           polygon.arcTo(new XPoint2D(roundCorner / 2, getHeight(stringBounder)), roundCorner / 2, 0, 0);
186           polygon.lineTo(getWidth(stringBounder) - roundCorner / 2, getHeight(stringBounder));
187           polygon.arcTo(new XPoint2D(getWidth(stringBounder), getHeight(stringBounder) - roundCorner / 2),
188                   roundCorner / 2, 0, 0);
189           polygon.lineTo(getWidth(stringBounder), cornersize);
190           polygon.lineTo(getWidth(stringBounder) - cornersize, 0);
191           polygon.lineTo(roundCorner / 2, 0);
192           polygon.arcTo(new XPoint2D(0, roundCorner / 2), roundCorner / 2, 0, 0);
193           polygon.closePath();
194           return polygon;
195       }

197       private UPath getPolygonRight(final StringBounder stringBounder, final XPoint2D pp1, final XPoint2D pp2) {
198           final UPath polygon = new UPath();
199           polygon.moveTo(0, roundCorner / 2);
200           polygon.lineTo(0, getHeight(stringBounder) - roundCorner / 2);
201           polygon.arcTo(new XPoint2D(roundCorner / 2, getHeight(stringBounder)), roundCorner / 2, 0, 0);
202           polygon.lineTo(getWidth(stringBounder) - roundCorner / 2, getHeight(stringBounder));
203           polygon.arcTo(new XPoint2D(getWidth(stringBounder), getHeight(stringBounder) - roundCorner / 2),
204                   roundCorner / 2, 0, 0);
205   
206           double y1 = pp1.getY() - delta;
207           y1 = MathUtils.limitation(y1, cornersize, getHeight(stringBounder) - 2 * delta);
208           polygon.lineTo(getWidth(stringBounder), y1 + 2 * delta);
209           polygon.lineTo(pp2.getX(), pp2.getY());
210           polygon.lineTo(getWidth(stringBounder), y1);
211   
212           polygon.lineTo(getWidth(stringBounder), cornersize);
213           polygon.lineTo(getWidth(stringBounder) - cornersize, 0);
214           polygon.lineTo(roundCorner / 2, 0);
215           polygon.arcTo(new XPoint2D(0, roundCorner / 2), roundCorner / 2, 0, 0);
216           polygon.closePath();
217           return polygon;
218       }

220       private UPath getPolygonUp(final StringBounder stringBounder, final XPoint2D pp1, final XPoint2D pp2) {
221           final UPath polygon = new UPath();
222           polygon.moveTo(0, roundCorner / 2);
223           polygon.lineTo(0, getHeight(stringBounder) - roundCorner / 2);
224           polygon.arcTo(new XPoint2D(roundCorner / 2, getHeight(stringBounder)), roundCorner / 2, 0, 0);
225           polygon.lineTo(getWidth(stringBounder) - roundCorner / 2, getHeight(stringBounder));
226           polygon.arcTo(new XPoint2D(getWidth(stringBounder), getHeight(stringBounder) - roundCorner / 2),
227                   roundCorner / 2, 0, 0);
228           polygon.lineTo(getWidth(stringBounder), cornersize);
229           polygon.lineTo(getWidth(stringBounder) - cornersize, 0);
230   
231           double x1 = pp1.getX() - delta;
232           x1 = MathUtils.limitation(x1, 0, getWidth(stringBounder) - cornersize);
233           polygon.lineTo(x1 + 2 * delta, 0);
234           polygon.lineTo(pp2.getX(), pp2.getY());
235   
236           polygon.lineTo(x1, 0);
237           polygon.lineTo(roundCorner / 2, 0);
238           polygon.arcTo(new XPoint2D(0, roundCorner / 2), roundCorner / 2, 0, 0);
239           polygon.closePath();
240           return polygon;
241       }

243       private UPath getPolygonDown(final StringBounder stringBounder, final XPoint2D pp1, final XPoint2D pp2) {
244           final UPath polygon = new UPath();
245           polygon.moveTo(0, roundCorner / 2);
246           polygon.lineTo(0, getHeight(stringBounder) - roundCorner / 2);
247           polygon.arcTo(new XPoint2D(roundCorner / 2, getHeight(stringBounder)), roundCorner / 2, 0, 0);
248   
249           double x1 = pp1.getX() - delta;
250           x1 = MathUtils.limitation(x1, 0, getWidth(stringBounder));
251           polygon.lineTo(x1, getHeight(stringBounder));
252           polygon.lineTo(pp2.getX(), pp2.getY());
253           polygon.lineTo(x1 + 2 * delta, getHeight(stringBounder));
254   
255           polygon.lineTo(getWidth(stringBounder) - roundCorner / 2, getHeight(stringBounder));
256           polygon.arcTo(new XPoint2D(getWidth(stringBounder), getHeight(stringBounder) - roundCorner / 2),
257                   roundCorner / 2, 0, 0);
258           polygon.lineTo(getWidth(stringBounder), cornersize);
259           polygon.lineTo(getWidth(stringBounder) - cornersize, 0);
260           polygon.lineTo(roundCorner / 2, 0);
261           polygon.arcTo(new XPoint2D(0, roundCorner / 2), roundCorner / 2, 0, 0);
262           polygon.closePath();
263           return polygon;
264       }
