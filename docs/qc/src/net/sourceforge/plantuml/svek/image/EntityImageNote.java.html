193       final public void drawU(UGraphic ug) {
194           final Url url = getEntity().getUrl99();
195   
196           final Map<UGroupType, String> typeIDent = new EnumMap<>(UGroupType.class);
197           typeIDent.put(UGroupType.CLASS, "elem " + getEntity().getCode() + " selected");
198           typeIDent.put(UGroupType.ID, "elem_" + getEntity().getCode());
199           ug.startGroup(typeIDent);
200   
201           if (url != null)
202               ug.startUrl(url);
203   
204           final UGraphic ug2 = UGraphicStencil.create(ug, this, new UStroke());
205           if (opaleLine == null || opaleLine.isOpale() == false) {
206               drawNormal(ug2);
207           } else {
208               final StringBounder stringBounder = ug.getStringBounder();
209               DotPath path = opaleLine.getDotPath();
210               path.moveSvek(-node.getMinX(), -node.getMinY());
211               XPoint2D p1 = path.getStartPoint();
212               XPoint2D p2 = path.getEndPoint();
213               final double textWidth = getTextWidth(stringBounder);
214               final double textHeight = getTextHeight(stringBounder);
215               final XPoint2D center = new XPoint2D(textWidth / 2, textHeight / 2);
216               if (p1.distance(center) > p2.distance(center)) {
217                   path = path.reverse();
218                   p1 = path.getStartPoint();
219                   // p2 = path.getEndPoint();
220               }
221               final Direction strategy = getOpaleStrategy(textWidth, textHeight, p1);
222               final XPoint2D pp1 = path.getStartPoint();
223               final XPoint2D pp2 = path.getEndPoint();
224               final XPoint2D newRefpp2 = move(pp2, node.getMinX(), node.getMinY());
225               final XPoint2D projection = move(other.projection(newRefpp2, stringBounder), -node.getMinX(),
226                       -node.getMinY());
227               final Opale opale = new Opale(shadowing, borderColor, noteBackgroundColor, textBlock, true, getStroke());
228               opale.setRoundCorner(getRoundCorner());
229               opale.setOpale(strategy, pp1, projection);
230               final UGraphic stroked = applyStroke(ug2);
231               opale.drawU(Colors.applyStroke(stroked, getEntity().getColors()));
232           }
233           if (url != null)
234               ug.closeUrl();
235   
236           ug.closeGroup();
237       }

271       private Direction getOpaleStrategy(double width, double height, XPoint2D pt) {
272           final double d1 = getOrthoDistance(new XLine2D(width, 0, width, height), pt);
273           final double d2 = getOrthoDistance(new XLine2D(0, height, width, height), pt);
274           final double d3 = getOrthoDistance(new XLine2D(0, 0, 0, height), pt);
275           final double d4 = getOrthoDistance(new XLine2D(0, 0, width, 0), pt);
276           if (d3 <= d1 && d3 <= d2 && d3 <= d4)
277               return Direction.LEFT;
278   
279           if (d1 <= d2 && d1 <= d3 && d1 <= d4)
280               return Direction.RIGHT;
281   
282           if (d4 <= d1 && d4 <= d2 && d4 <= d3)
283               return Direction.UP;
284   
285           if (d2 <= d1 && d2 <= d3 && d2 <= d4)
286               return Direction.DOWN;
287   
288           return null;
289   
290       }

292       private static double getOrthoDistance(XLine2D seg, XPoint2D pt) {
293           if (isHorizontal(seg))
294               return Math.abs(seg.getP1().getY() - pt.getY());
295   
296           if (isVertical(seg))
297               return Math.abs(seg.getP1().getX() - pt.getX());
298   
299           throw new IllegalArgumentException();
300       }
301   
302       private static boolean isHorizontal(XLine2D seg) {
303           return seg.getP1().getY() == seg.getP2().getY();
304       }
305   
306       private static boolean isVertical(XLine2D seg) {
307           return seg.getP1().getX() == seg.getP2().getX();
308       }
