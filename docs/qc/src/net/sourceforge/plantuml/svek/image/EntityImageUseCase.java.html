86       public EntityImageUseCase(ILeaf entity, ISkinParam skinParam2, PortionShower portionShower) {
87           super(entity, entity.getColors().mute(skinParam2));
88           final Stereotype stereotype = entity.getStereotype();
89   
90           final Style style = getStyle();
91           final HorizontalAlignment align = style.getHorizontalAlignment();
92   
93           final TextBlock tmp = BodyFactory.create2(getSkinParam().getDefaultTextAlignment(align), entity.getDisplay(),
94                   getSkinParam(), stereotype, entity, getStyle());
95   
96           if (stereotype == null || stereotype.getLabel(Guillemet.DOUBLE_COMPARATOR) == null
97                   || portionShower.showPortion(EntityPortion.STEREOTYPE, entity) == false) {
98               this.desc = tmp;
99           } else {
100               final TextBlock stereo;
101               if (stereotype.getSprite(getSkinParam()) != null) {
102                   stereo = stereotype.getSprite(getSkinParam());
103               } else {
104                   stereo = Display.getWithNewlines(stereotype.getLabel(getSkinParam().guillemet())).create(
105                           FontConfiguration.create(getSkinParam(), FontParam.USECASE_STEREOTYPE, stereotype),
106                           HorizontalAlignment.CENTER, getSkinParam());
107               }
108               this.desc = TextBlockUtils.mergeTB(stereo, tmp, HorizontalAlignment.CENTER);
109           }
110           this.url = entity.getUrl99();
111   
112       }

118       final public void drawU(UGraphic ug) {
119           final StringBounder stringBounder = ug.getStringBounder();
120   
121           final Style style = getStyle();
122           final double shadow = style.value(PName.Shadowing).asDouble();
123   
124           final TextBlockInEllipse ellipse = new TextBlockInEllipse(desc, stringBounder);
125           ellipse.setDeltaShadow(shadow);
126   
127           if (url != null)
128               ug.startUrl(url);
129   
130           ug = ug.apply(getStyle().getStroke(getEntity().getColors()));
131           final HColor linecolor = getLineColor();
132           ug = ug.apply(linecolor);
133           final HColor backcolor = getBackColor();
134           ug = ug.apply(backcolor.bg());
135           final UGraphic ug2 = new MyUGraphicEllipse(ug, 0, 0, ellipse.getUEllipse());
136   
137           final Map<UGroupType, String> typeIDent = new EnumMap<>(UGroupType.class);
138           typeIDent.put(UGroupType.CLASS, "elem " + getEntity().getCode() + " selected");
139           typeIDent.put(UGroupType.ID, "elem_" + getEntity().getCode());
140           ug.startGroup(typeIDent);
141           ellipse.drawU(ug2);
142           ug2.closeGroup();
143   
144           if (getEntity().getLeafType() == LeafType.USECASE_BUSINESS)
145               specialBusiness(ug, ellipse.getUEllipse());
146   
147           if (url != null)
148               ug.closeUrl();
149   
150       }

152       private void specialBusiness(UGraphic ug, UEllipse frontier) {
153           final RotatedEllipse rotatedEllipse = new RotatedEllipse(frontier, Math.PI / 4);
154   
155           final double theta1 = 20.0 * Math.PI / 180;
156           final double theta2 = rotatedEllipse.getOtherTheta(theta1);
157   
158           final UEllipse frontier2 = frontier.scale(0.99);
159           final XPoint2D p1 = frontier2.getPointAtAngle(-theta1);
160           final XPoint2D p2 = frontier2.getPointAtAngle(-theta2);
161           drawLine(ug, p1, p2);
162       }
163   
164       private void specialBusiness0(UGraphic ug, UEllipse frontier) {
165           final double c = frontier.getWidth() / frontier.getHeight();
166           final double ouverture = Math.PI / 2;
167           final XPoint2D p1 = frontier.getPointAtAngle(getTrueAngle(c, Math.PI / 4 - ouverture));
168           final XPoint2D p2 = frontier.getPointAtAngle(getTrueAngle(c, Math.PI / 4 + ouverture));
169           drawLine(ug, p1, p2);
170       }

172       private void drawLine(UGraphic ug, final XPoint2D p1, final XPoint2D p2) {
173           ug = ug.apply(new UTranslate(p1));
174           ug.draw(new ULine(p2.getX() - p1.getX(), p2.getY() - p1.getY()));
175       }

177       private double getTrueAngle(final double c, final double gamma) {
178           return Math.atan2(Math.sin(gamma), Math.cos(gamma) / c);
179       }

225           MyUGraphicEllipse(UGraphic ug, double startingX, double yTheoricalPosition, UEllipse ellipse) {
226               super(ug);
227               this.startingX = startingX;
228               this.ellipse = ellipse;
229               this.yTheoricalPosition = yTheoricalPosition;
230           }

232           private double getNormalized(double y) {
233               if (y < yTheoricalPosition)
234                   throw new IllegalArgumentException();
235   
236               y = y - yTheoricalPosition;
237               if (y > ellipse.getHeight())
238                   throw new IllegalArgumentException();
239   
240               return y;
241           }
242   
243           private double getStartingXInternal(double y) {
244               return startingX + ellipse.getStartingX(getNormalized(y));
245           }
246   
247           private double getEndingXInternal(double y) {
248               return startingX + ellipse.getEndingX(getNormalized(y));
249           }

251           private Stencil getStencil2(UTranslate translate) {
252               final double dy = translate.getDy();
253               return new Stencil() {
254   
255                   public double getStartingX(StringBounder stringBounder, double y) {
256                       return getStartingXInternal(y + dy);
257                   }
258   
259                   public double getEndingX(StringBounder stringBounder, double y) {
260                       return getEndingXInternal(y + dy);
261                   }
262               };
263           }
