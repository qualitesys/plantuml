55   public final class CucaDiagramFileMakerSvek2InternalImage extends AbstractTextBlock implements IEntityImage {
56   
57       private final List<IEntityImage> inners;
58       private final Separator separator;
59       private final ISkinParam skinParam;
60       private final Stereotype stereotype;
61   
62       static enum Separator {
63           VERTICAL, HORIZONTAL;
64   
65           static Separator fromChar(char sep) {
66               if (sep == '|')
67                   return VERTICAL;
68   
69               if (sep == '-')
70                   return HORIZONTAL;
71   
72               throw new IllegalArgumentException();
73           }
74   
75           UTranslate move(XDimension2D dim) {
76               if (this == VERTICAL)
77                   return UTranslate.dx(dim.getWidth());
78   
79               return UTranslate.dy(dim.getHeight());
80           }
81   
82           XDimension2D add(XDimension2D orig, XDimension2D other) {
83               if (this == VERTICAL)
84                   return new XDimension2D(orig.getWidth() + other.getWidth(),
85                           Math.max(orig.getHeight(), other.getHeight()));
86   
87               return new XDimension2D(Math.max(orig.getWidth(), other.getWidth()),
88                       orig.getHeight() + other.getHeight());
89           }
90   
91           void drawSeparator(UGraphic ug, XDimension2D dimTotal) {
92               final double THICKNESS_BORDER = 1.5;
93               final int DASH = 8;
94               ug = ug.apply(new UStroke(DASH, 10, THICKNESS_BORDER));
95               if (this == VERTICAL)
96                   ug.draw(ULine.vline(dimTotal.getHeight() + DASH));
97               else
98                   ug.draw(ULine.hline(dimTotal.getWidth() + DASH));
99   
100           }
101       }
102   
103       public CucaDiagramFileMakerSvek2InternalImage(List<IEntityImage> inners, char concurrentSeparator,
104               ISkinParam skinParam, Stereotype stereotype) {
105           this.separator = Separator.fromChar(concurrentSeparator);
106           this.skinParam = skinParam;
107           this.stereotype = stereotype;
108           this.inners = inners;
109       }
110   
111       private Style getStyle() {
112           return getStyleSignature().getMergedStyle(skinParam.getCurrentStyleBuilder());
113       }
114   
115       private StyleSignatureBasic getStyleSignature() {
116           return StyleSignatureBasic.of(SName.root, SName.element, SName.stateDiagram, SName.state);
117       }
118   
119       public void drawU(UGraphic ug) {
120           final HColor borderColor = getStyle().value(PName.LineColor).asColor(skinParam.getIHtmlColorSet());
121           final StringBounder stringBounder = ug.getStringBounder();
122           final XDimension2D dimTotal = calculateDimension(stringBounder);
123   
124           for (int i = 0; i < inners.size(); i++) {
125               final IEntityImage inner = inners.get(i);
126               inner.drawU(ug);
127               final XDimension2D dim = inner.calculateDimension(stringBounder);
128               ug = ug.apply(separator.move(dim));
129               if (i < inners.size() - 1)
130                   separator.drawSeparator(ug.apply(borderColor), dimTotal);
131   
132           }
133   
134       }
135   
136       public XDimension2D calculateDimension(StringBounder stringBounder) {
137           XDimension2D result = new XDimension2D(0, 0);
138           for (IEntityImage inner : inners) {
139               final XDimension2D dim = inner.calculateDimension(stringBounder);
140               result = separator.add(result, dim);
141           }
142           return result;
143       }
144   
145       public HColor getBackcolor() {
146           return skinParam.getBackgroundColor();
147       }
148   
149       public double getOverscanX(StringBounder stringBounder) {
150           return 0;
151       }
152   
153       public boolean isHidden() {
154           return false;
155       }
156   
157       public Margins getShield(StringBounder stringBounder) {
158           return Margins.NONE;
159       }
160   
161       public ShapeType getShapeType() {
162           return ShapeType.RECTANGLE;
163       }
