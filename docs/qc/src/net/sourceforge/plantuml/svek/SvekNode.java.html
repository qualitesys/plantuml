322       private void appendShapeInternal(StringBuilder sb) {
323           if (type == ShapeType.RECTANGLE && shield().isZero() == false)
324               throw new UnsupportedOperationException();
325           else if (type == ShapeType.RECTANGLE || type == ShapeType.RECTANGLE_WITH_CIRCLE_INSIDE
326                   || type == ShapeType.FOLDER)
327               sb.append("shape=rect");
328           else if (type == ShapeType.RECTANGLE_HTML_FOR_PORTS)
329               throw new UnsupportedOperationException();
330           else if (type == ShapeType.OCTAGON)
331               sb.append("shape=octagon");
332           else if (type == ShapeType.HEXAGON)
333               sb.append("shape=hexagon");
334           else if (type == ShapeType.DIAMOND)
335               sb.append("shape=diamond");
336           else if (type == ShapeType.CIRCLE)
337               sb.append("shape=circle");
338           else if (type == ShapeType.OVAL)
339               sb.append("shape=ellipse");
340           else if (type == ShapeType.ROUND_RECTANGLE)
341               sb.append("shape=rect,style=rounded");
342           else
343               throw new IllegalStateException(type.toString());
344   
345       }

439       public void fixOverlap() {
440           if (leaf instanceof EntityImpl == false)
441               return;
442   
443           fixHoverlap(((EntityImpl) leaf).getKals(Direction.DOWN));
444           fixHoverlap(((EntityImpl) leaf).getKals(Direction.UP));
445       }

447       private void fixHoverlap(final List<Kal> list) {
448           final LineOfSegments los = new LineOfSegments();
449           for (Kal kal : list)
450               los.addSegment(kal.getX1(), kal.getX2());
451   
452           final double[] res = los.solveOverlaps();
453           for (int i = 0; i < list.size(); i++) {
454               final Kal kal = list.get(i);
455               final double diff = res[i] - kal.getX1();
456               kal.moveX(diff);
457           }
458       }
