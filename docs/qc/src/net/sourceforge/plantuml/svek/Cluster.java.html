186       private List<SvekNode> getNodesOrderedTop(Collection<SvekLine> lines) {
187           final List<SvekNode> firsts = new ArrayList<>();
188           final Map<String, SvekNode> shs = new HashMap<String, SvekNode>();
189   
190           for (final Iterator<SvekNode> it = nodes.iterator(); it.hasNext();) {
191               final SvekNode node = it.next();
192               shs.put(node.getUid(), node);
193           }
194   
195           for (SvekLine l : lines) {
196               if (l.isInverted()) {
197                   final SvekNode sh = shs.get(l.getStartUidPrefix());
198                   if (sh != null && sh.getEntityPosition() == EntityPosition.NORMAL)
199                       firsts.add(0, sh);
200               }
201           }
202   
203           return firsts;
204       }

206       private List<SvekNode> getNodesOrderedWithoutTop(Collection<SvekLine> lines) {
207           final List<SvekNode> all = new ArrayList<>(nodes);
208           final Map<String, SvekNode> shs = new HashMap<String, SvekNode>();
209   
210           for (final Iterator<SvekNode> it = all.iterator(); it.hasNext();) {
211               final SvekNode sh = it.next();
212               if (sh.getEntityPosition() != EntityPosition.NORMAL) {
213                   it.remove();
214                   continue;
215               }
216               shs.put(sh.getUid(), sh);
217           }
218   
219           for (SvekLine l : lines) {
220               if (l.isInverted()) {
221                   final SvekNode sh = shs.get(l.getStartUidPrefix());
222                   if (sh != null)
223                       all.remove(sh);
224               }
225           }
226   
227           return all;
228       }

282       public void drawU(UGraphic ug, UmlDiagramType umlDiagramType, ISkinParam skinParam2unused) {
283           if (group.isHidden())
284               return;
285   
286           if (diagram.getPragma().useKermor()) {
287               if (xyNoteTop != null)
288                   getCucaNote(Position.TOP).drawU(ug.apply(new UTranslate(xyNoteTop)));
289               if (xyNoteBottom != null)
290                   getCucaNote(Position.BOTTOM).drawU(ug.apply(new UTranslate(xyNoteBottom)));
291           }
292   
293           final String fullName = group.getCodeGetName();
294           if (fullName.startsWith("##") == false)
295               ug.draw(new UComment("cluster " + fullName));
296   
297           final USymbol uSymbol = group.getUSymbol() == null ? USymbols.PACKAGE : group.getUSymbol();
298           Style style = getDefaultStyleDefinition(umlDiagramType.getStyleName(), uSymbol)
299                   .withTOBECHANGED(group.getStereotype()).getMergedStyle(skinParam.getCurrentStyleBuilder());
300           final double shadowing = style.value(PName.Shadowing).asDouble();
301           HColor borderColor;
302           if (group.getColors().getColor(ColorType.LINE) != null)
303               borderColor = group.getColors().getColor(ColorType.LINE);
304           else
305               borderColor = style.value(PName.LineColor).asColor(skinParam.getIHtmlColorSet());
306           final double rounded = style.value(PName.RoundCorner).asDouble();
307           final double diagonalCorner = style.value(PName.DiagonalCorner).asDouble();
308   
309           ug.startGroup(Collections.singletonMap(UGroupType.ID, "cluster_" + fullName));
310   
311           final Url url = group.getUrl99();
312           if (url != null)
313               ug.startUrl(url);
314   
315           try {
316               if (entityPositionsExceptNormal().size() > 0)
317                   manageEntryExitPoint(ug.getStringBounder());
318   
319               if (skinParam.useSwimlanes(umlDiagramType)) {
320                   drawSwinLinesState(ug, borderColor);
321                   return;
322               }
323               final boolean isState = umlDiagramType == UmlDiagramType.STATE;
324   
325               if (isState && group.getUSymbol() == null) {
326                   drawUState(ug, umlDiagramType, rounded, shadowing);
327                   return;
328               }
329               PackageStyle packageStyle = group.getPackageStyle();
330               if (packageStyle == null)
331                   packageStyle = skinParam.packageStyle();
332   
333               final UStroke stroke = getStrokeInternal(group, style);
334   
335               HColor backColor = getBackColor(umlDiagramType, style);
336               backColor = getBackColor(backColor, group.getStereotype(), umlDiagramType.getStyleName(),
337                       group.getUSymbol(), skinParam.getCurrentStyleBuilder(), skinParam.getIHtmlColorSet());
338   
339               final ClusterDecoration decoration = new ClusterDecoration(packageStyle, group.getUSymbol(),
340                       clusterHeader.getTitle(), clusterHeader.getStereo(), clusterPosition, stroke);
341               decoration.drawU(ug, backColor, borderColor, shadowing, rounded,
342                       skinParam.getHorizontalAlignment(AlignmentParam.packageTitleAlignment, null, false, null),
343                       skinParam.getStereotypeAlignment(), diagonalCorner);
344   
345           } catch (Exception e) {
346               e.printStackTrace();
347           } finally {
348               if (url != null)
349                   ug.closeUrl();
350               ug.closeGroup();
351           }
352       }

372           final Collection<ClusterPosition> insides = new ArrayList<>();
373           final List<XPoint2D> points = new ArrayList<>();

377               else

379

382

385               frontierCalculator.ensureMinWidth(getTitleAndAttributeWidth() + 10);

415                       .asColor(skinParam.getIHtmlColorSet());

420                       .asColor(skinParam.getIHtmlColorSet());
421

423                   .value(PName.BackGroundColor).asColor(skinParam.getIHtmlColorSet());
424   
425           final TextBlock attribute = ((EntityImpl) group).getStateHeader(skinParam);
426           final double attributeHeight = attribute.calculateDimension(ug.getStringBounder()).getHeight();
427           if (total.getWidth() == 0) {
428               System.err.println("Cluster::drawUState issue");
429               return;
430           }
431   
432           UStroke stroke = group.getColors().getSpecificLineStroke();
433           if (stroke == null)
434               stroke = EntityImageStateCommon.getStyleState(group, skinParam).getStroke();
435   
436           final RoundedContainer r = new RoundedContainer(total, suppY,
437                   attributeHeight + (attributeHeight > 0 ? IEntityImage.MARGIN : 0), borderColor, backColor, imgBackcolor,
438                   stroke, rounded, shadowing);
439           r.drawU(ug.apply(clusterPosition.getPosition()));
440   
441           clusterHeader.getTitle().drawU(ug.apply(new UTranslate(xyTitle)));
442   
443           if (attributeHeight > 0)
444               attribute.drawU(ug.apply(new UTranslate(clusterPosition.getMinX() + IEntityImage.MARGIN,
445                       clusterPosition.getMinY() + suppY + IEntityImage.MARGIN / 2.0)));
446   
447           final Stereotype stereotype = group.getStereotype();
448           final boolean withSymbol = stereotype != null && stereotype.isWithOOSymbol();
449           if (withSymbol)
450               EntityImageState.drawSymbol(ug.apply(borderColor), clusterPosition.getMaxX(), clusterPosition.getMaxY());
451   
452       }
