142       private static IEntityImage createEntityImageBlockInternal(ILeaf leaf, ISkinParam skinParam,
143               boolean isHideEmptyDescriptionForState, PortionShower portionShower, Bibliotekon bibliotekon,
144               GraphvizVersion graphvizVersion, UmlDiagramType umlDiagramType, Collection<Link> links) {
145           if (leaf.isRemoved())
146               throw new IllegalStateException();
147   
148           if (leaf.getLeafType().isLikeClass()) {
149               final EntityImageClass entityImageClass = new EntityImageClass((ILeaf) leaf, skinParam, portionShower);
150               final Neighborhood neighborhood = leaf.getNeighborhood();
151               if (neighborhood != null)
152                   return new EntityImageProtected(entityImageClass, 20, neighborhood, bibliotekon);
153   
154               return entityImageClass;
155           }
156           if (leaf.getLeafType() == LeafType.NOTE)
157               return new EntityImageNote(leaf, skinParam, umlDiagramType);
158   
159           if (leaf.getLeafType() == LeafType.ACTIVITY)
160               return new EntityImageActivity(leaf, skinParam, bibliotekon);
161   
162           if (/* (leaf.getLeafType() == LeafType.PORT) || */leaf.getLeafType() == LeafType.PORTIN
163                   || leaf.getLeafType() == LeafType.PORTOUT) {
164               final Cluster parent = bibliotekon.getCluster(leaf.getParentContainer());
165               return new EntityImagePort(leaf, skinParam, parent, bibliotekon, umlDiagramType.getStyleName());
166           }
167   
168           if (leaf.getLeafType() == LeafType.STATE) {
169               if (leaf.getEntityPosition() != EntityPosition.NORMAL) {
170                   final Cluster stateParent = bibliotekon.getCluster(leaf.getParentContainer());
171                   return new EntityImageStateBorder(leaf, skinParam, stateParent, bibliotekon,
172                           umlDiagramType.getStyleName());
173               }
174               if (isHideEmptyDescriptionForState && leaf.getBodier().getRawBody().size() == 0)
175                   return new EntityImageStateEmptyDescription(leaf, skinParam);
176   
177               if (leaf.getStereotype() != null
178                       && "<<sdlreceive>>".equals(leaf.getStereotype().getLabel(Guillemet.DOUBLE_COMPARATOR)))
179                   return new EntityImageState2(leaf, skinParam, umlDiagramType.getStyleName());
180   
181               return new EntityImageState(leaf, skinParam);
182   
183           }
184           if (leaf.getLeafType() == LeafType.CIRCLE_START)
185               return new EntityImageCircleStart(leaf, skinParam);
186   
187           if (leaf.getLeafType() == LeafType.CIRCLE_END)
188               return new EntityImageCircleEnd(leaf, skinParam);
189   
190           if (leaf.getLeafType() == LeafType.BRANCH || leaf.getLeafType() == LeafType.STATE_CHOICE)
191               return new EntityImageBranch(leaf, skinParam);
192   
193           if (leaf.getLeafType() == LeafType.LOLLIPOP_FULL || leaf.getLeafType() == LeafType.LOLLIPOP_HALF)
194               return new EntityImageLollipopInterface(leaf, skinParam, umlDiagramType.getStyleName());
195   
196           if (leaf.getLeafType() == LeafType.CIRCLE)
197               return new EntityImageDescription(leaf, skinParam, portionShower, links, umlDiagramType.getStyleName(),
198                       bibliotekon);
199   
200           if (leaf.getLeafType() == LeafType.DESCRIPTION) {
201               if (OptionFlags.USE_INTERFACE_EYE1 && leaf.getUSymbol() instanceof USymbolInterface) {
202                   return new EntityImageLollipopInterfaceEye1(leaf, skinParam, bibliotekon);
203               } else if (OptionFlags.USE_INTERFACE_EYE2 && leaf.getUSymbol() instanceof USymbolInterface) {
204                   return new EntityImageLollipopInterfaceEye2(leaf, skinParam, portionShower);
205               } else {
206                   return new EntityImageDescription(leaf, skinParam, portionShower, links, umlDiagramType.getStyleName(),
207                           bibliotekon);
208               }
209           }
210           if (leaf.getLeafType() == LeafType.USECASE)
211               return new EntityImageUseCase(leaf, skinParam, portionShower);
212   
213           if (leaf.getLeafType() == LeafType.USECASE_BUSINESS)
214               return new EntityImageUseCase(leaf, skinParam, portionShower);
215   
216           // if (leaf.getEntityType() == LeafType.CIRCLE_INTERFACE) {
217           // return new EntityImageCircleInterface(leaf, skinParam);
218           // }
219           if (leaf.getLeafType() == LeafType.OBJECT)
220               return new EntityImageObject(leaf, skinParam, portionShower);
221   
222           if (leaf.getLeafType() == LeafType.MAP)
223               return new EntityImageMap(leaf, skinParam, portionShower);
224   
225           if (leaf.getLeafType() == LeafType.JSON)
226               return new EntityImageJson(leaf, skinParam, portionShower);
227   
228           if (leaf.getLeafType() == LeafType.SYNCHRO_BAR || leaf.getLeafType() == LeafType.STATE_FORK_JOIN)
229               return new EntityImageSynchroBar(leaf, skinParam, umlDiagramType.getStyleName());
230   
231           if (leaf.getLeafType() == LeafType.ARC_CIRCLE)
232               return new EntityImageArcCircle(leaf, skinParam);
233   
234           if (leaf.getLeafType() == LeafType.POINT_FOR_ASSOCIATION)
235               return new EntityImageAssociationPoint(leaf, skinParam);
236   
237           if (leaf.isGroup())
238               return new EntityImageGroup(leaf, skinParam);
239   
240           if (leaf.getLeafType() == LeafType.EMPTY_PACKAGE) {
241               if (leaf.getUSymbol() != null)
242                   return new EntityImageDescription(leaf, skinParam, portionShower, links, umlDiagramType.getStyleName(),
243                           bibliotekon);
244   
245               return new EntityImageEmptyPackage(leaf, skinParam, portionShower, umlDiagramType.getStyleName());
246           }
247           if (leaf.getLeafType() == LeafType.ASSOCIATION)
248               return new EntityImageAssociation(leaf, skinParam, umlDiagramType.getStyleName());
249   
250           if (leaf.getLeafType() == LeafType.PSEUDO_STATE)
251               return new EntityImagePseudoState(leaf, skinParam, umlDiagramType.getStyleName());
252   
253           if (leaf.getLeafType() == LeafType.DEEP_HISTORY)
254               return new EntityImageDeepHistory(leaf, skinParam, umlDiagramType.getStyleName());
255   
256           if (leaf.getLeafType() == LeafType.TIPS)
257               return new EntityImageTips(leaf, skinParam, bibliotekon, umlDiagramType);
258   
259           // TODO Clean
260           if (leaf.getLeafType() == LeafType.DOMAIN && leaf.getStereotype() != null
261                   && leaf.getStereotype().isMachineOrSpecification())
262               return new EntityImageMachine(leaf, skinParam);
263           else if (leaf.getLeafType() == LeafType.DOMAIN && leaf.getStereotype() != null
264                   && leaf.getStereotype().isDesignedOrSolved())
265               return new EntityImageDesignedDomain(leaf, skinParam);
266           else if (leaf.getLeafType() == LeafType.REQUIREMENT)
267               return new EntityImageRequirement(leaf, skinParam);
268           else if (leaf.getLeafType() == LeafType.DOMAIN && leaf.getStereotype() != null
269                   && leaf.getStereotype().isLexicalOrGiven())
270               return new EntityImageDomain(leaf, skinParam, 'X');
271           else if (leaf.getLeafType() == LeafType.DOMAIN && leaf.getStereotype() != null
272                   && leaf.getStereotype().isCausal())
273               return new EntityImageDomain(leaf, skinParam, 'C');
274           else if (leaf.getLeafType() == LeafType.DOMAIN && leaf.getStereotype() != null
275                   && leaf.getStereotype().isBiddableOrUncertain())
276               return new EntityImageDomain(leaf, skinParam, 'B');
277           else if (leaf.getLeafType() == LeafType.DOMAIN)
278               return new EntityImageDomain(leaf, skinParam, 'P');
279           else
280               throw new UnsupportedOperationException(leaf.getLeafType().toString());
281       }

382           if (dotData.isDegeneratedWithFewEntities(0))
383               return new EntityImageSimpleEmpty(dotData.getSkinParam().getBackgroundColor());
384   
385           if (dotData.isDegeneratedWithFewEntities(1) && dotData.getUmlDiagramType() != UmlDiagramType.STATE) {
386               final ILeaf single = dotData.getLeafs().iterator().next();
387               final IGroup group = single.getParentContainer();
388               if (group instanceof GroupRoot && single.getUSymbol() instanceof USymbolHexagon == false) {
389                   final IEntityImage tmp = GeneralImageBuilder.createEntityImageBlock(single, dotData.getSkinParam(),
390                           dotData.isHideEmptyDescriptionForState(), dotData, null, null, dotData.getUmlDiagramType(),
391                           dotData.getLinks());
392                   return new EntityImageDegenerated(tmp, getBackcolor());
393               }
394           }
395           dotData.removeIrrelevantSametail();
396           final DotStringFactory dotStringFactory = new DotStringFactory(stringBounder, dotData);
397   
398           printGroups(dotStringFactory, dotData.getRootGroup());
399           printEntities(dotStringFactory, getUnpackagedEntities());
400   
401           for (Link link : dotData.getLinks()) {
402               if (link.isRemoved())
403                   continue;
404
