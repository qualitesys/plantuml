172       private Direction getArrowDirectionInternal() {
173           if (isAutolink()) {
174               final double startAngle = dotPath.getStartAngle();
175               return Direction.LEFT;
176           }
177           final XPoint2D start = dotPath.getStartPoint();
178           final XPoint2D end = dotPath.getEndPoint();
179           final double ang = Math.atan2(end.getX() - start.getX(), end.getY() - start.getY());
180           if (ang > -Math.PI / 4 && ang < Math.PI / 4)
181               return Direction.DOWN;
182   
183           if (ang > Math.PI * 3 / 4 || ang < -Math.PI * 3 / 4)
184               return Direction.UP;
185   
186           return end.getX() > start.getX() ? Direction.RIGHT : Direction.LEFT;
187       }

361       public void appendLine(GraphvizVersion graphvizVersion, StringBuilder sb, DotMode dotMode, DotSplines dotSplines) {
362           // Log.println("inverted=" + isInverted());
363           // if (isInverted()) {
364           // sb.append(endUid);
365           // sb.append("->");
366           // sb.append(startUid);
367           // } else {
368           sb.append(startUid.getFullString());
369           sb.append("->");
370           sb.append(endUid.getFullString());
371           // }
372           sb.append("[");
373           final LinkType linkType = link.getTypePatchCluster();
374           String decoration = linkType.getSpecificDecorationSvek();
375           if (decoration.length() > 0 && decoration.endsWith(",") == false)
376               decoration += ",";
377   
378           sb.append(decoration);
379   
380           int length = link.getLength();
381           if (graphvizVersion.ignoreHorizontalLinks() && length == 1)
382               length = 2;
383   
384           if (useRankSame) {
385               if (pragma.horizontalLineBetweenDifferentPackageAllowed() || link.isInvis() || length != 1) {
386                   // if (graphvizVersion.isJs() == false) {
387                   sb.append("minlen=" + (length - 1));
388                   sb.append(",");
389                   // }
390               }
391           } else {
392               sb.append("minlen=" + (length - 1));
393               sb.append(",");
394           }
395           sb.append("color=\"" + StringUtils.sharp000000(lineColor) + "\"");
396           if (hasNoteLabelText() || link.getLinkConstraint() != null) {
397               sb.append(",");
398               if (graphvizVersion.useXLabelInsteadOfLabel() || dotMode == DotMode.NO_LEFT_RIGHT_AND_XLABEL
399                       || dotSplines == DotSplines.ORTHO) {
400                   sb.append("xlabel=<");
401               } else {
402                   sb.append("label=<");
403               }
404               XDimension2D dimNote = hasNoteLabelText() ? labelText.calculateDimension(stringBounder) : CONSTRAINT_SPOT;
405               dimNote = XDimension2D.delta(dimNote, 2 * labelShield);
406   
407               appendTable(sb, eventuallyDivideByTwo(dimNote), noteLabelColor, graphvizVersion);
408               sb.append(">");
409           }
410   
411           if (startTailText != null) {
412               sb.append(",");
413               sb.append("taillabel=<");
414               appendTable(sb, startTailText.calculateDimension(stringBounder), startTailColor, graphvizVersion);
415               sb.append(">");
416           }
417           if (endHeadText != null) {
418               sb.append(",");
419               sb.append("headlabel=<");
420               appendTable(sb, endHeadText.calculateDimension(stringBounder), endHeadColor, graphvizVersion);
421               sb.append(">");
422           }
423   
424           if (link.isInvis()) {
425               sb.append(",");
426               sb.append("style=invis");
427           }
428   
429           if (link.isConstraint() == false || link.hasTwoEntryPointsSameContainer())
430               sb.append(",constraint=false");
431   
432           if (link.getSametail() != null)
433               sb.append(",sametail=" + link.getSametail());
434   
435           sb.append("];");
436           SvekUtils.println(sb);
437       }
