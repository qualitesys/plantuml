68       void printInternal(StringBuilder sb, Collection<SvekLine> lines, StringBounder stringBounder, DotMode dotMode,
69               GraphvizVersion graphvizVersion, UmlDiagramType type) {
70           if (cluster.diagram.getPragma().useKermor()) {
71               new ClusterDotStringKermor(cluster, skinParam).printInternal(sb, lines, stringBounder, dotMode,
72                       graphvizVersion, type);
73               return;
74           }
75           final boolean thereALinkFromOrToGroup2 = isThereALinkFromOrToGroup(lines);
76           boolean thereALinkFromOrToGroup1 = thereALinkFromOrToGroup2;
77           final boolean useProtectionWhenThereALinkFromOrToGroup = graphvizVersion
78                   .useProtectionWhenThereALinkFromOrToGroup();
79           if (useProtectionWhenThereALinkFromOrToGroup == false)
80               thereALinkFromOrToGroup1 = false;
81   
82           if (thereALinkFromOrToGroup1)
83               subgraphClusterNoLabel(sb, "a");
84   
85           final Set<EntityPosition> entityPositionsExceptNormal = entityPositionsExceptNormal();
86           if (entityPositionsExceptNormal.size() > 0)
87               for (SvekLine line : lines)
88                   if (line.isLinkFromOrTo(cluster.getGroup()))
89                       line.setProjectionCluster(cluster);
90   
91           boolean protection0 = protection0(type);
92           boolean protection1 = protection1(type);
93           if (entityPositionsExceptNormal.size() > 0 || useProtectionWhenThereALinkFromOrToGroup == false) {
94               protection0 = false;
95               protection1 = false;
96           }
97   
98           if (protection0)
99               subgraphClusterNoLabel(sb, "p0");
100   
101           sb.append("subgraph " + cluster.getClusterId() + " {");
102           sb.append("style=solid;");
103           sb.append("color=\"" + StringUtils.sharp000000(cluster.getColor()) + "\";");
104   
105           final String label;
106           if (cluster.isLabel()) {
107               final StringBuilder sblabel = new StringBuilder("<");
108               SvekLine.appendTable(sblabel, cluster.getTitleAndAttributeWidth(), cluster.getTitleAndAttributeHeight() - 5,
109                       cluster.getTitleColor());
110               sblabel.append(">");
111               label = sblabel.toString();
112               final HorizontalAlignment align = skinParam.getHorizontalAlignment(AlignmentParam.packageTitleAlignment,
113                       null, false, null);
114               sb.append("labeljust=\"" + align.getGraphVizValue() + "\";");
115           } else {
116               label = "\"\"";
117           }
118   
119           if (entityPositionsExceptNormal.size() > 0) {
120               printRanks(Cluster.RANK_SOURCE, withPosition(EntityPosition.getInputs()), sb, stringBounder);
121               printRanks(Cluster.RANK_SINK, withPosition(EntityPosition.getOutputs()), sb, stringBounder);
122               if (hasPort())
123                   subgraphClusterNoLabel(sb, ID_EE);
124               else
125                   subgraphClusterWithLabel(sb, ID_EE, label);
126   
127           } else {
128               sb.append("label=" + label + ";");
129               SvekUtils.println(sb);
130           }
131   
132           if (thereALinkFromOrToGroup2)
133               sb.append(Cluster.getSpecialPointId(cluster.getGroup()) + " [shape=point,width=.01,label=\"\"];");
134   
135           if (thereALinkFromOrToGroup1)
136               subgraphClusterNoLabel(sb, "i");
137   
138           if (protection1)
139               subgraphClusterNoLabel(sb, "p1");
140   
141           if (skinParam.useSwimlanes(type)) {
142               sb.append("{rank = source; ");
143               sb.append(getSourceInPoint(type));
144               sb.append(" [shape=point,width=.01,label=\"\"];");
145               sb.append(cluster.getMinPoint(type) + "->" + getSourceInPoint(type) + "  [weight=999];");
146               sb.append("}");
147               SvekUtils.println(sb);
148               sb.append("{rank = sink; ");
149               sb.append(getSinkInPoint(type));
150               sb.append(" [shape=point,width=.01,label=\"\"];");
151               sb.append("}");
152               sb.append(getSinkInPoint(type) + "->" + cluster.getMaxPoint(type) + "  [weight=999];");
153               SvekUtils.println(sb);
154           }
155           SvekUtils.println(sb);
156           cluster.printCluster1(sb, lines, stringBounder);
157   
158           final SvekNode added = cluster.printCluster2(sb, lines, stringBounder, dotMode, graphvizVersion, type);
159           if (entityPositionsExceptNormal.size() > 0)
160               if (hasPort()) {
161                   sb.append(empty() + " [shape=rect,width=.01,height=.01,label=");
162                   sb.append(label);
163                   sb.append("];");
164               } else if (added == null) {
165                   sb.append(empty() + " [shape=point,width=.01,label=\"\"];");
166               }
167           SvekUtils.println(sb);
168   
169           sb.append("}");
170           if (protection1)
171               sb.append("}");
172   
173           if (thereALinkFromOrToGroup1) {
174               sb.append("}");
175               sb.append("}");
176           }
177           if (entityPositionsExceptNormal.size() > 0)
178               sb.append("}");
179   
180           if (protection0)
181               sb.append("}");
182   
183           SvekUtils.println(sb);
184       }

186       private String getSourceInPoint(UmlDiagramType type) {
187           if (skinParam.useSwimlanes(type))
188               return "sourceIn" + cluster.getColor();
189   
190           return null;
191       }

193       private String getSinkInPoint(UmlDiagramType type) {
194           if (skinParam.useSwimlanes(type))
195               return "sinkIn" + cluster.getColor();
196   
197           return null;
198       }

200       private String empty() {
201           // return "empty" + color;
202           // We use the same node with one for thereALinkFromOrToGroup2 as an empty
203           // because we cannot put a new node in the nested inside of the cluster
204           // if thereALinkFromOrToGroup2 is enabled.
205           return Cluster.getSpecialPointId(cluster.getGroup());
206       }

208       private boolean hasPort() {
209           for (EntityPosition pos : entityPositionsExceptNormal())
210               if (pos.isPort())
211                   return true;
212   
213           return false;
214       }

216       private Set<EntityPosition> entityPositionsExceptNormal() {
217           final Set<EntityPosition> result = EnumSet.<EntityPosition>noneOf(EntityPosition.class);
218           for (SvekNode sh : cluster.getNodes())
219               if (sh.getEntityPosition() != EntityPosition.NORMAL)
220                   result.add(sh.getEntityPosition());
221   
222           return Collections.unmodifiableSet(result);
223       }

225       private void subgraphClusterNoLabel(StringBuilder sb, String id) {
226           subgraphClusterWithLabel(sb, id, "\"\"");
227       }

229       private void subgraphClusterWithLabel(StringBuilder sb, String id, String label) {
230           sb.append("subgraph " + cluster.getClusterId() + id + " {");
231           sb.append("label=" + label + ";");
232       }

234       private void printRanks(String rank, List<? extends SvekNode> entries, StringBuilder sb,
235               StringBounder stringBounder) {
236           if (entries.size() > 0) {
237               sb.append("{rank=" + rank + ";");
238               for (SvekNode sh1 : entries)
239                   sb.append(sh1.getUid() + ";");
240   
241               sb.append("}");
242               SvekUtils.println(sb);
243               for (SvekNode sh2 : entries)
244                   sh2.appendShape(sb, stringBounder);
245   
246               SvekUtils.println(sb);
247               if (hasPort()) {
248                   boolean arrow = false;
249                   String node = null;
250                   for (SvekNode sh : entries) {
251                       if (arrow)
252                           sb.append("->");
253   
254                       arrow = true;
255                       node = sh.getUid();
256                       sb.append(node);
257                   }
258                   if (arrow)
259                       sb.append(" [arrowhead=none]");
260   
261                   sb.append(';');
262                   SvekUtils.println(sb);
263                   sb.append(node + "->" + empty() + ";");
264                   SvekUtils.println(sb);
265               }
266           }
267       }

269       private List<SvekNode> withPosition(Set<EntityPosition> positions) {
270           final List<SvekNode> result = new ArrayList<>();
271           for (final Iterator<SvekNode> it = cluster.getNodes().iterator(); it.hasNext();) {
272               final SvekNode sh = it.next();
273               if (positions.contains(sh.getEntityPosition()))
274                   result.add(sh);
275   
276           }
277           return result;
278       }

280       private boolean protection0(UmlDiagramType type) {
281           if (skinParam.useSwimlanes(type))
282               return false;
283   
284           return true;
285       }

287       private boolean protection1(UmlDiagramType type) {
288           if (cluster.getGroup().getUSymbol() == USymbols.NODE)
289               return true;
290   
291           if (skinParam.useSwimlanes(type))
292               return false;
293   
294           return true;
295       }

297       private boolean isThereALinkFromOrToGroup(Collection<SvekLine> lines) {
298           for (SvekLine line : lines)
299               if (line.isLinkFromOrTo(cluster.getGroup()))
300                   return true;
301   
302           return false;
303       }
