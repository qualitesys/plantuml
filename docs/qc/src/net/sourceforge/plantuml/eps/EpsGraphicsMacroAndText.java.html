73       private String getTextAsEps(String text) {
74           final StringBuilder sb = new StringBuilder();
75           for (int i = 0; i < text.length(); i++) {
76               final char c = text.charAt(i);
77               if (c == '\\') {
78                   sb.append("\\\\");
79               } else if (c == '(') {
80                   sb.append("\\(");
81               } else if (c == ')') {
82                   sb.append("\\)");
83               } else if (c < ' ') {
84                   sb.append("?");
85               } else if (c >= ' ' && c <= 127) {
86                   sb.append(c);
87               } else {
88                   final String s = "" + c;
89                   try {
90                       final byte b[] = s.getBytes("ISO-8859-1");
91                       if (b.length == 1) {
92                           final int code = b[0] & 0xFF;
93                           sb.append("\\" + Integer.toOctalString(code));
94                       } else {
95                           sb.append('?');
96                       }
97                   } catch (UnsupportedEncodingException e) {
98                       sb.append('?');
99                   }
100               }
101           }
102           return sb.toString();
103       }
