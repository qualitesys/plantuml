29   package net.sourceforge.plantuml.api;
30   
31   import java.util.StringTokenizer;
32   import java.util.concurrent.locks.Lock;
33   import java.util.concurrent.locks.ReentrantLock;
34   import java.util.prefs.Preferences;
35   
36   import net.sourceforge.plantuml.Log;
37   import net.sourceforge.plantuml.annotation.HaxeIgnored;
38   import net.sourceforge.plantuml.log.Logme;
39   @HaxeIgnored
40   public class NumberAnalyzed implements INumberAnalyzed {
41   
42       private static final int SLIDING_WINDOW = 1024;
43   
44       private long nb;
45       private long sum;
46       private long min;
47       private long max;
48       private long sumOfSquare;
49       // See https://fossies.org/linux/haproxy/include/proto/freq_ctr.h
50       private long sliddingSum;
51       private final String name;
52   
53       private final Lock saveLock = new ReentrantLock();
54   
55       public NumberAnalyzed(String name) {
56           this.name = name;
57       }
58   
59       public synchronized void reset() {
60           this.nb = 0;
61           this.sum = 0;
62           this.min = 0;
63           this.max = 0;
64           this.sumOfSquare = 0;
65           this.sliddingSum = 0;
66       }
67   
68       public NumberAnalyzed() {
69           this("");
70       }
71   
72       public final void save(Preferences prefs) {
73           if (name.length() == 0) {
74               throw new UnsupportedOperationException();
75           }
76           if (saveLock.tryLock())
77               try {
78                   prefs.put(name + ".saved", getSavedString());
79               } finally {
80                   saveLock.unlock();
81               }
82       }
83   
84       private String getSavedString() {
85           final long nb1;
86           final long sum1;
87           final long min1;
88           final long max1;
89           final long sumOfSquare1;
90           final long sliddingSum1;
91           final String supp1;
92           synchronized (this) {
93               nb1 = nb;
94               sum1 = sum;
95               min1 = min;
96               max1 = max;
97               sumOfSquare1 = sumOfSquare;
98               sliddingSum1 = sliddingSum;
99           }
100           supp1 = getSavedSupplementatyData();
101           return longToString(nb1) + ";" + longToString(sum1) + ";" + longToString(min1) + ";" + longToString(max1) + ";"
102                   + longToString(sumOfSquare1) + ";" + longToString(sliddingSum1) + ";" + supp1 + ";";
103       }
104   
105       protected String getSavedSupplementatyData() {
106           return "";
107       }
108   
109       protected final String longToString(long val) {
110           return Long.toString(val, 36);
111       }
112   
113       public static NumberAnalyzed load(String name, Preferences prefs) {
114           final String value = prefs.get(name + ".saved", "");
115           if (value.length() == 0) {
116               Log.info("Cannot load " + name);
117               return null;
118           }
119           try {
120               final StringTokenizer st = new StringTokenizer(value, ";");
121               return new NumberAnalyzed(name, Long.parseLong(st.nextToken(), 36), Long.parseLong(st.nextToken(), 36),
122                       Long.parseLong(st.nextToken(), 36), Long.parseLong(st.nextToken(), 36), Long.parseLong(
123                               st.nextToken(), 36), Long.parseLong(st.nextToken(), 36));
124           } catch (Exception e) {
125               Logme.error(e);
126               Log.info("Error reading " + value);
127               return null;
128           }
129       }
130   
131       @Override
132       public synchronized String toString() {
133           return "sum=" + sum + " nb=" + nb + " min=" + min + " max=" + max + " mean=" + getMean();
134       }
135   
136       protected NumberAnalyzed(String name, long nb, long sum, long min, long max, long sumOfSquare, long sliddingSum) {
137           this(name);
138           this.nb = nb;
139           this.sum = sum;
140           this.min = min;
141           this.max = max;
142           this.sumOfSquare = sumOfSquare;
143           this.sliddingSum = sliddingSum;
144       }
145   
146       public synchronized INumberAnalyzed getCopyImmutable() {
147           final NumberAnalyzed copy = new NumberAnalyzed(name, nb, sum, min, max, sumOfSquare, sliddingSum);
148           return copy;
149       }
150   
151       public synchronized void addValue(long v) {
152           nb++;
153           if (nb == 1) {
154               min = v;
155               max = v;
156           } else if (v > max) {
157               max = v;
158           } else if (v < min) {
159               min = v;
160           }
161           sum += v;
162           sumOfSquare += v * v;
163           sliddingSum = sliddingSum * (SLIDING_WINDOW - 1) / SLIDING_WINDOW + v;
164       }
165   
166       public void add(NumberAnalyzed other) {
167           final long nb1;
168           final long sum1;
169           final long min1;
170           final long max1;
171           final long sumOfSquare1;
172           final long sliddingSum1;
173           synchronized (other) {
174               nb1 = other.nb;
175               sum1 = other.sum;
176               min1 = other.min;
177               max1 = other.max;
178               sumOfSquare1 = other.sumOfSquare;
179               sliddingSum1 = other.sliddingSum;
180           }
181           synchronized (this) {
182               this.sum += sum1;
183               this.nb += nb1;
184               this.min = Math.min(this.min, min1);
185               this.max = Math.max(this.max, max1);
186               this.sumOfSquare += sumOfSquare1;
187               // Not good!
188               this.sliddingSum += sliddingSum1;
189           }
190       }
191   
192       synchronized public final long getNb() {
193           return nb;
194       }
195   
196       synchronized public final long getSum() {
197           return sum;
198       }
199   
200       synchronized public final long getMin() {
201           return min;
202       }
203   
204       synchronized public final long getMax() {
205           return max;
206       }
207   
208       synchronized public final long getMean() {
209           if (nb == 0) {
210               return 0;
211           }
212           return sum / nb;
213       }
214   
215       synchronized public final long getSliddingMean() {
216           if (nb == 0) {
217               return 0;
218           }
219           if (nb < SLIDING_WINDOW) {
220               return sum / nb;
221           }
222           return sliddingSum / nb;
223       }
224   
225       public final long getStandardDeviation() {
226           final long sum1;
227           final long sumOfSquare1;
228           final long nb1;
229           synchronized (this) {
230               sum1 = this.sum;
231               sumOfSquare1 = this.sumOfSquare;
232               nb1 = this.nb;
233           }
234           if (nb1 == 0) {
235               return 0;
236           }
237           final long mean = sum1 / nb1;
238           return Math.round(Math.sqrt(sumOfSquare1 / nb1 - mean * mean));
239       }
240   
241       final public String getName() {
242           return name;
243       }
244   
245   }
