95       public void lineSimple(TileGeometry orientation, String idDest, String label) {
96           final Tile newTile;
97           if (field == null) {
98               field = new TilesField();
99               tilesBoxes.clear();
100               newTile = field.getRoot();
101           } else {
102               newTile = field.createTile(lastTile, orientation);
103           }
104           final ActivityBox box = new ActivityBox(newTile, idDest, label);
105           tilesBoxes.put(newTile, box);
106           lastTile = newTile;
107           return;
108       }
109   
110       public void linkSimple(TileGeometry orientation, String idDest) {
111           final Tile tile = getTileById(idDest);
112           field.addPath(lastTile, tile, orientation);
113       }

115       private Tile getTileById(String id) {
116           for (Map.Entry<Tile, ActivityBox> ent : tilesBoxes.entrySet()) {
117               if (ent.getValue().getId().equals(id)) {
118                   return ent.getKey();
119               }
120           }
121           throw new IllegalArgumentException(id);
122       }

139       public void drawU(UGraphic ug) {
140           double x = 0;
141           double y = 0;
142           final MinMaxGolem minMax = getMinMax();
143           x -= minMax.getMinX() * SINGLE_SIZE_X;
144           y -= minMax.getMinY() * SINGLE_SIZE_Y;
145           final StringBounder stringBounder = ug.getStringBounder();
146           for (Map.Entry<Tile, ActivityBox> ent : tilesBoxes.entrySet()) {
147               final Tile tile = ent.getKey();
148               final Position pos = field.getPosition(tile);
149               final int xmin = pos.getXmin();
150               final int ymin = pos.getYmin();
151               final ActivityBox box = ent.getValue();
152               final XDimension2D dimBox = box.calculateDimension(stringBounder);
153               final double deltaX = SINGLE_SIZE_X * 2 - dimBox.getWidth();
154               final double deltaY = SINGLE_SIZE_Y * 2 - dimBox.getHeight();
155               box.drawU(ug.apply(
156                       new UTranslate((x + xmin * SINGLE_SIZE_X + deltaX / 2), (y + ymin * SINGLE_SIZE_Y + deltaY / 2))));
157           }
158           ug = ug.apply(HColors.MY_RED);
159           ug = ug.apply(HColors.MY_RED.bg());
160           final UShape arrow = new UEllipse(7, 7);
161           for (Path p : field.getPaths()) {
162               final TileArea start = p.getStart();
163               final TileArea dest = p.getDest();
164               final XPoint2D pStart = movePoint(getCenter(start), start.getTile(), start.getGeometry(), stringBounder);
165               final XPoint2D pDest = movePoint(getCenter(dest), dest.getTile(), dest.getGeometry(), stringBounder);
166               final ULine line = new ULine(pDest.getX() - pStart.getX(), pDest.getY() - pStart.getY());
167               ug.apply(new UTranslate(x + pStart.getX(), y + pStart.getY())).draw(line);
168               ug.apply(new UTranslate(x + pDest.getX() - 3, y + pDest.getY() - 3)).draw(arrow);
169   
170           }
171       }

173       private XPoint2D getCenter(TileArea area) {
174           final Tile tile = area.getTile();
175           final Position position = field.getPosition(tile);
176           final double x = position.getCenterX();
177           final double y = position.getCenterY();
178           return new XPoint2D(x * SINGLE_SIZE_X, y * SINGLE_SIZE_Y);
179       }
180   
181       private XPoint2D movePoint(XPoint2D pt, Tile tile, TileGeometry tileGeometry, StringBounder stringBounder) {
182           final XDimension2D dim = tilesBoxes.get(tile).calculateDimension(stringBounder);
183           final double width = dim.getWidth();
184           final double height = dim.getHeight();
185           double x = pt.getX();
186           double y = pt.getY();
187           switch (tileGeometry) {
188           case SOUTH:
189               y += height / 2;
190               break;
191           case NORTH:
192               y -= height / 2;
193               break;
194           case EAST:
195               x += width / 2;
196               break;
197           case WEST:
198               x -= width / 2;
199               break;
200           default:
201               throw new IllegalStateException();
202           }
203           return new XPoint2D(x, y);
204       }

206       private MinMaxGolem getMinMax() {
207           final MinMaxGolem minMax = new MinMaxGolem();
208           for (Tile tile : tilesBoxes.keySet()) {
209               minMax.manage(field.getPosition(tile));
210           }
211           return minMax;
212       }

214       public XDimension2D calculateDimension(StringBounder stringBounder) {
215           final MinMaxGolem minMax = getMinMax();
216           return new XDimension2D(minMax.getWidth() * SINGLE_SIZE_X, minMax.getHeight() * SINGLE_SIZE_Y);
217       }
