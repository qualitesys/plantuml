301       public String activate(Participant p, LifeEventType lifeEventType, HColor backcolor, HColor linecolor) {
302           if (lastDelay != null)
303               return "You cannot Activate/Deactivate just after a ...";
304   
305           final LifeEvent lifeEvent = new LifeEvent(p, lifeEventType, new SymbolContext(backcolor, linecolor));
306           events.add(lifeEvent);
307           if (lifeEventType == LifeEventType.CREATE) {
308               pendingCreate = lifeEvent;
309               return null;
310           }
311           if (lastEventWithDeactivate == null) {
312               if (lifeEventType == LifeEventType.ACTIVATE) {
313                   p.incInitialLife(new SymbolContext(backcolor, linecolor));
314                   return null;
315               }
316               if (p.getInitialLife() == 0)
317                   return "You cannot deactivate here";
318   
319               return null;
320           }
321           if (lifeEventType == LifeEventType.ACTIVATE && lastEventWithDeactivate instanceof AbstractMessage)
322               activationState.push((AbstractMessage) lastEventWithDeactivate);
323           else if (lifeEventType == LifeEventType.DEACTIVATE && activationState.empty() == false)
324               activationState.pop();
325   
326           final boolean ok = lastEventWithDeactivate.addLifeEvent(lifeEvent);
327           if (lastEventWithDeactivate instanceof AbstractMessage) {
328               final AbstractMessage lastMessage = (AbstractMessage) lastEventWithDeactivate;
329               lifeEvent.setMessage(lastMessage);
330           }
331           if (ok)
332               return null;
333   
334           return "Activate/Deactivate already done on " + p.getCode();
335       }

339       public boolean grouping(String title, String comment, GroupingType type, HColor backColorGeneral,
340               HColor backColorElement, boolean parallel) {
341           if (type != GroupingType.START && openGroupings.size() == 0)
342               return false;
343   
344           if (backColorGeneral == null)
345               backColorGeneral = getSkinParam().getHtmlColor(ColorParam.sequenceGroupBodyBackground, null, false);
346   
347           final GroupingStart top = openGroupings.size() > 0 ? openGroupings.get(0) : null;
348   
349           final Grouping g = type == GroupingType.START
350                   ? new GroupingStart(title, comment, backColorGeneral, backColorElement, top,
351                           getSkinParam().getCurrentStyleBuilder())
352                   : new GroupingLeaf(title, comment, type, backColorGeneral, backColorElement, top,
353                           getSkinParam().getCurrentStyleBuilder());
354           events.add(g);
355   
356           if (type == GroupingType.START) {
357               if (parallel)
358                   ((GroupingStart) g).goParallel();
359   
360               openGroupings.add(0, (GroupingStart) g);
361           } else if (type == GroupingType.END) {
362               openGroupings.remove(0);
363               lastEventWithDeactivate = (GroupingLeaf) g;
364           }
365   
366           return true;
367       }
