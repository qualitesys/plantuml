35   package net.sourceforge.plantuml.sequencediagram;
36   
37   import java.text.DecimalFormat;
38   import java.util.ArrayList;
39   import java.util.List;
40   import java.util.regex.Matcher;
41   import java.util.regex.Pattern;
42   
43   public class DottedNumber {
44   
45       private final List<Integer> nums;
46       private final List<String> separators;
47   
48       private DottedNumber(List<Integer> nums, List<String> separators) {
49           this.nums = nums;
50           this.separators = separators;
51       }
52   
53       public static DottedNumber create(String value) {
54           final Pattern p = Pattern.compile("(\\d+)|(\\D+)");
55           final Matcher m = p.matcher(value);
56           final List<Integer> nums = new ArrayList<>();
57           final List<String> separators = new ArrayList<>();
58           while (m.find()) {
59               final String part = m.group();
60               if (isDigit(part.charAt(0))) {
61                   nums.add(Integer.parseInt(part));
62               } else {
63                   separators.add(part);
64               }
65           }
66           return new DottedNumber(nums, separators);
67       }
68   
69       private static boolean isDigit(final char c) {
70           return c >= '0' && c <= '9';
71       }
72   
73       @Override
74       public String toString() {
75           final StringBuilder sb = new StringBuilder();
76           for (int i = 0; i < nums.size(); i++) {
77               sb.append(nums.get(i));
78               if (i < separators.size()) {
79                   sb.append(separators.get(i));
80               }
81           }
82           return sb.toString();
83       }
84   
85       public void incrementMinor(int step) {
86           final int last = nums.size() - 1;
87           final int newValue = nums.get(last) + step;
88           nums.set(last, newValue);
89       }
90   
91       public void incrementIntermediate() {
92           final int intermediate = nums.size() == 1 ? 0 : nums.size() - 2;
93           incrementIntermediate(intermediate);
94       }
95   
96       public void incrementIntermediate(int position) {
97           final int intermediate = position;
98           final int newValue = nums.get(intermediate) + 1;
99           for (int i = intermediate + 1; i < nums.size(); i++) {
100               nums.set(i, 1);
101           }
102           nums.set(intermediate, newValue);
103       }
104   
105       public String format(DecimalFormat format) {
106           if (nums.size() == 1 && separators.size() == 0) {
107               return format.format(nums.get(0));
108           }
109           return "<b>" + toString() + "</b>";
110       }
111   
112   }
