36   package net.sourceforge.plantuml.sequencediagram.command;
37   
38   import java.util.regex.Matcher;
39   import java.util.regex.Pattern;
40   
41   import net.sourceforge.plantuml.LineLocation;
42   import net.sourceforge.plantuml.StringUtils;
43   import net.sourceforge.plantuml.command.CommandExecutionResult;
44   import net.sourceforge.plantuml.command.SingleLineCommand2;
45   import net.sourceforge.plantuml.command.regex.IRegex;
46   import net.sourceforge.plantuml.command.regex.RegexConcat;
47   import net.sourceforge.plantuml.command.regex.RegexLeaf;
48   import net.sourceforge.plantuml.command.regex.RegexOptional;
49   import net.sourceforge.plantuml.command.regex.RegexResult;
50   import net.sourceforge.plantuml.sequencediagram.GroupingType;
51   import net.sourceforge.plantuml.sequencediagram.SequenceDiagram;
52   import net.sourceforge.plantuml.ugraphic.color.HColor;
53   import net.sourceforge.plantuml.ugraphic.color.HColorSet;
54   import net.sourceforge.plantuml.ugraphic.color.NoSuchColorException;
55   
56   public class CommandGrouping extends SingleLineCommand2<SequenceDiagram> {
57   
58       public CommandGrouping() {
59           super(getRegexConcat());
60       }
61   
62       static IRegex getRegexConcat() {
63           return RegexConcat.build(CommandGrouping.class.getName(), RegexLeaf.start(), //
64                   new RegexLeaf("PARALLEL", "(&[%s]*)?"), //
65                   new RegexLeaf("TYPE", "(opt|alt|loop|par|par2|break|critical|else|end|also|group)"), //
66                   new RegexLeaf("COLORS", "((?<!else)(?<!also)(?<!end)#\\w+)?(?:[%s]+(#\\w+))?"), //
67                   new RegexOptional(//
68                           new RegexConcat( //
69                                   RegexLeaf.spaceOneOrMore(), //
70                                   new RegexLeaf("COMMENT", "(.*?)") //
71                           )), RegexLeaf.end());
72       }
73   
74       @Override
75       protected CommandExecutionResult executeArg(SequenceDiagram diagram, LineLocation location, RegexResult arg)
76               throws NoSuchColorException {
77           String type = StringUtils.goLowerCase(arg.get("TYPE", 0));
78           final String s = arg.get("COLORS", 0);
79           final HColorSet colorSet = diagram.getSkinParam().getIHtmlColorSet();
80           HColor backColorElement = null;
81           if (s != null) {
82               backColorElement = colorSet.getColor(s);
83           }
84           final String s2 = arg.get("COLORS", 1);
85           HColor backColorGeneral = null;
86           if (s2 != null) {
87               backColorGeneral = colorSet.getColor(s2);
88           }
89           String comment = arg.get("COMMENT", 0);
90           final GroupingType groupingType = GroupingType.getType(type);
91           if ("group".equals(type)) {
92               if (StringUtils.isEmpty(comment)) {
93                   comment = "group";
94               } else {
95                   final Pattern p = Pattern.compile("^(.*\\[\\[.*\\]\\].*?|.*?)\\[(.*)\\]$");
96                   final Matcher m = p.matcher(comment);
97                   if (m.find()) {
98                       type = m.group(1);
99                       comment = m.group(2);
100                   }
101               }
102           }
103   
104           final boolean parallel = arg.get("PARALLEL", 0) != null;
105           final boolean result = diagram.grouping(type, comment, groupingType, backColorGeneral, backColorElement,
106                   parallel);
107           if (result == false) {
108               return CommandExecutionResult.error("Cannot create group");
109           }
110           return CommandExecutionResult.ok();
111       }
112   }
