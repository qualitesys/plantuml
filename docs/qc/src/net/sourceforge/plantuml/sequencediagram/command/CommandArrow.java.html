135       private List<Participant> getMulticasts(SequenceDiagram system, RegexResult arg2) {
136           final String multicast = arg2.get("MULTICAST", 0);
137           if (multicast != null) {
138               final List<Participant> result = new ArrayList<>();
139               for (String s : multicast.split("&")) {
140                   s = s.trim();
141                   if (s.length() == 0)
142                       continue;
143   
144                   final Participant participant = system.getOrCreateParticipant(s);
145                   if (participant != null)
146                       result.add(participant);
147   
148               }
149               return Collections.unmodifiableList(result);
150           }
151           return Collections.emptyList();
152       }

154       private Participant getOrCreateParticipant(SequenceDiagram system, RegexResult arg2, String n) {
155           final String code;
156           final Display display;
157           if (arg2.get(n + "CODE", 0) != null) {
158               code = arg2.get(n + "CODE", 0);
159               display = Display.getWithNewlines(code);
160           } else if (arg2.get(n + "LONG", 0) != null) {
161               code = arg2.get(n + "LONG", 0);
162               display = Display.getWithNewlines(code);
163           } else if (arg2.get(n + "LONGCODE", 0) != null) {
164               display = Display.getWithNewlines(arg2.get(n + "LONGCODE", 0));
165               code = arg2.get(n + "LONGCODE", 1);
166           } else if (arg2.get(n + "CODELONG", 0) != null) {
167               code = arg2.get(n + "CODELONG", 0);
168               display = Display.getWithNewlines(arg2.get(n + "CODELONG", 1));
169               return system.getOrCreateParticipant(code, display);
170           } else {
171               throw new IllegalStateException();
172           }
173           return system.getOrCreateParticipant(code, display);
174       }

184       private String getDressing(RegexResult arg, String key) {
185           String value = arg.get(key, 0);
186           value = CommandLinkClass.notNull(value);
187           value = value.replace("_", "");
188           return StringUtils.goLowerCase(value);
189       }

191       private int getInclination(String key) {
192           if (key == null)
193               return 0;
194           final int x1 = key.indexOf('(');
195           if (x1 == -1)
196               return 0;
197           final int x2 = key.indexOf(')');
198           if (x2 == -1)
199               return 0;
200           return Integer.parseInt(key.substring(x1 + 1, x2));
201       }

203       @Override
204       protected CommandExecutionResult executeArg(SequenceDiagram diagram, LineLocation location, RegexResult arg)
205               throws NoSuchColorException {
206   
207           Participant p1;
208           Participant p2;
209   
210           final String dressing1 = getDressing(arg, "ARROW_DRESSING1");
211           final String dressing2 = getDressing(arg, "ARROW_DRESSING2");
212           final int inclination1 = getInclination(arg.get("ARROW_DRESSING1", 0));
213           final int inclination2 = getInclination(arg.get("ARROW_DRESSING2", 0));
214   
215           final boolean circleAtStart;
216           final boolean circleAtEnd;
217   
218           final boolean hasDressing2 = contains(dressing2, ">", "\\", "/", "x");
219           final boolean hasDressing1 = contains(dressing1, "x", "<", "\\", "/");
220           final boolean reverseDefine;
221           if (hasDressing2) {
222               p1 = getOrCreateParticipant(diagram, arg, "PART1");
223               p2 = getOrCreateParticipant(diagram, arg, "PART2");
224               circleAtStart = dressing1.contains("o");
225               circleAtEnd = dressing2.contains("o");
226               reverseDefine = false;
227           } else if (hasDressing1) {
228               p2 = getOrCreateParticipant(diagram, arg, "PART1");
229               p1 = getOrCreateParticipant(diagram, arg, "PART2");
230               circleAtStart = dressing2.contains("o");
231               circleAtEnd = dressing1.contains("o");
232               reverseDefine = true;
233           } else {
234               return CommandExecutionResult.error("Illegal sequence arrow");
235   
236           }
237   
238           final boolean sync = contains(dressing1, "<<", "\\\\", "//") || contains(dressing2, ">>", "\\\\", "//");
239   
240           final boolean dotted = getLength(arg) > 1;
241   
242           final Display labels;
243           if (arg.get("MESSAGE", 0) == null) {
244               labels = Display.create("");
245           } else {
246               // final String message = UrlBuilder.multilineTooltip(arg.get("MESSAGE", 0));
247               final String message = arg.get("MESSAGE", 0);
248               labels = Display.getWithNewlines(message);
249           }
250   
251           ArrowConfiguration config = hasDressing1 && hasDressing2 ? ArrowConfiguration.withDirectionBoth()
252                   : ArrowConfiguration.withDirectionNormal();
253           if (dotted)
254               config = config.withBody(ArrowBody.DOTTED);
255   
256           if (sync)
257               config = config.withHead(ArrowHead.ASYNC);
258   
259           if (dressing2.contains("\\") || dressing1.contains("/"))
260               config = config.withPart(ArrowPart.TOP_PART);
261   
262           if (dressing2.contains("/") || dressing1.contains("\\"))
263               config = config.withPart(ArrowPart.BOTTOM_PART);
264   
265           if (circleAtEnd)
266               config = config.withDecoration2(ArrowDecoration.CIRCLE);
267   
268           if (circleAtStart)
269               config = config.withDecoration1(ArrowDecoration.CIRCLE);
270   
271           if (reverseDefine) {
272               if (dressing1.contains("x"))
273                   config = config.withHead2(ArrowHead.CROSSX);
274   
275               if (dressing2.contains("x"))
276                   config = config.withHead1(ArrowHead.CROSSX);
277   
278           } else {
279               if (dressing1.contains("x"))
280                   config = config.withHead1(ArrowHead.CROSSX);
281   
282               if (dressing2.contains("x"))
283                   config = config.withHead2(ArrowHead.CROSSX);
284   
285           }
286           if (reverseDefine)
287               config = config.reverseDefine();
288   
289           config = applyStyle(arg.getLazzy("ARROW_STYLE", 0), config);
290   
291           config = config.withInclination(inclination1 + inclination2);
292   
293           final String activationSpec = arg.get("ACTIVATION", 0);
294   
295           if (activationSpec != null && activationSpec.charAt(0) == '*')
296               diagram.activate(p2, LifeEventType.CREATE, null);
297   
298           final String messageNumber = diagram.getNextMessageNumber();
299           final Message msg = new Message(diagram.getSkinParam().getCurrentStyleBuilder(), p1, p2,
300                   diagram.manageVariable(labels), config, messageNumber);
301           msg.setMulticast(getMulticasts(diagram, arg));
302           final String url = arg.get("URL", 0);
303           if (url != null) {
304               final UrlBuilder urlBuilder = new UrlBuilder(diagram.getSkinParam().getValue("topurl"), UrlMode.STRICT);
305               final Url urlLink = urlBuilder.getUrl(url);
306               msg.setUrl(urlLink);
307           }
308   
309           if (arg.get("STEREOTYPE", 0) != null) {
310               final Stereotype stereotype = Stereotype.build(arg.get("STEREOTYPE", 0));
311               msg.getStereotype(stereotype);
312           }
313   
314           final boolean parallel = arg.get("PARALLEL", 0) != null;
315           if (parallel)
316               msg.goParallel();
317   
318           msg.setAnchor(arg.get("ANCHOR", 1));
319           msg.setPart1Anchor(arg.get("PART1ANCHOR", 1));
320           msg.setPart2Anchor(arg.get("PART2ANCHOR", 1));
321   
322           final CommandExecutionResult status = diagram.addMessage(msg);
323           if (status.isOk() == false)
324               return status;
325   
326           final String s = arg.get("LIFECOLOR", 0);
327   
328           final HColor activationColor = s == null ? null
329                   : diagram.getSkinParam().getIHtmlColorSet().getColor(s);
330   
331           if (activationSpec != null)
332               return manageActivations(activationSpec, diagram, p1, p2, activationColor);
333   
334           if (diagram.isAutoactivate() && (config.getHead() == ArrowHead.NORMAL || config.getHead() == ArrowHead.ASYNC))
335               if (config.isDotted())
336                   diagram.activate(p1, LifeEventType.DEACTIVATE, null);
337               else
338                   diagram.activate(p2, LifeEventType.ACTIVATE, activationColor);
339   
340           return CommandExecutionResult.ok();
341       }

343       private CommandExecutionResult manageActivations(String spec, SequenceDiagram diagram, Participant p1,
344               Participant p2, HColor activationColor) {
345           switch (spec.charAt(0)) {
346           case '+':
347               diagram.activate(p2, LifeEventType.ACTIVATE, activationColor);
348               break;
349           case '-':
350               diagram.activate(p1, LifeEventType.DEACTIVATE, null);
351               break;
352           case '!':
353               diagram.activate(p2, LifeEventType.DESTROY, null);
354               break;
355           }
356           if (spec.length() == 4) {
357               switch (spec.charAt(2)) {
358               case '+':
359                   diagram.activate(p2, LifeEventType.ACTIVATE, activationColor);
360                   break;
361               case '-':
362                   diagram.activate(p1, LifeEventType.DEACTIVATE, null);
363                   break;
364               }
365           }
366           return CommandExecutionResult.ok();
367       }

369       private int getLength(RegexResult arg2) {
370           String sa = arg2.getLazzy("ARROW_BODYA", 0);
371           if (sa == null)
372               sa = "";
373   
374           String sb = arg2.getLazzy("ARROW_BODYB", 0);
375           if (sb == null)
376               sb = "";
377   
378           return sa.length() + sb.length();
379       }

381       public static ArrowConfiguration applyStyle(String arrowStyle, ArrowConfiguration config)
382               throws NoSuchColorException {
383           if (arrowStyle == null)
384               return config;
385   
386           final StringTokenizer st = new StringTokenizer(arrowStyle, ",");
387           while (st.hasMoreTokens()) {
388               final String s = st.nextToken();
389               if (s.equalsIgnoreCase("dashed")) {
390                   config = config.withBody(ArrowBody.DOTTED);
391               } else if (s.equalsIgnoreCase("bold")) {
392               } else if (s.equalsIgnoreCase("dotted")) {
393                   config = config.withBody(ArrowBody.DOTTED);
394               } else if (s.equalsIgnoreCase("hidden")) {
395                   config = config.withBody(ArrowBody.HIDDEN);
396               } else {
397                   config = config.withColor(HColorSet.instance().getColor(s));
398               }
399           }
400           return config;
401       }
