79       public CommunicationTile(StringBounder stringBounder, LivingSpaces livingSpaces, Message message, Rose skin,
80               ISkinParam skinParam, YGauge currentY) {
81           super(stringBounder, currentY);
82           this.livingSpace1 = livingSpaces.get(message.getParticipant1());
83           this.livingSpace2 = livingSpaces.get(message.getParticipant2());
84           this.livingSpaces = livingSpaces;
85           this.message = message;
86           this.skin = skin;
87           this.skinParam = skinParam;
88   
89           if (message.isCreate())
90               livingSpace2.goCreate();
91   
92           final ArrowComponent comp = getComponent(getStringBounder());
93           final XDimension2D dim = comp.getPreferredDimension(getStringBounder());
94   
95           if (message.isParallel())
96               this.yGauge = YGauge.create(currentY.getMin(), dim.getHeight());
97           else
98               this.yGauge = YGauge.create(currentY.getMax(), dim.getHeight());
99       }

106       public boolean isReverse(StringBounder stringBounder) {
107           final Real point1 = livingSpace1.getPosC(stringBounder);
108           final Real point2 = livingSpace2.getPosC(stringBounder);
109           if (point1.getCurrentValue() > point2.getCurrentValue())
110               return true;
111   
112           return false;
113       }

127       private ArrowComponent getComponent(StringBounder stringBounder) {
128           ArrowConfiguration arrowConfiguration = message.getArrowConfiguration();
129           if (isReverse(stringBounder))
130               arrowConfiguration = arrowConfiguration.reverse();
131

133   
134           return skin.createComponentArrow(message.getUsedStyles(), arrowConfiguration, skinParam,
135                   message.getLabelNumbered());
136       }

138       private ArrowComponent getComponentMulticast(StringBounder stringBounder, boolean reverse) {
139           ArrowConfiguration arrowConfiguration = message.getArrowConfiguration();
140           if (reverse)
141               arrowConfiguration = arrowConfiguration.reverse();
142   
143           arrowConfiguration = arrowConfiguration.withThickness(getArrowThickness());
144   
145           return skin.createComponentArrow(message.getUsedStyles(), arrowConfiguration, skinParam, Display.NULL);
146       }

153       public static final double LIVE_DELTA_SIZE = 5;
154   
155       @Override
156       final protected void callbackY_internal(TimeHook y) {
157           super.callbackY_internal(y);
158           if (message.isCreate())
159               livingSpace2.goCreate(y.getValue());
160   
161           final AbstractComponentRoseArrow comp = (AbstractComponentRoseArrow) getComponent(getStringBounder());
162           final XDimension2D dim = comp.getPreferredDimension(getStringBounder());
163   
164           final double arrowY = comp.getStartPoint(getStringBounder(), dim).getY();
165   
166           livingSpace1.addStepForLivebox(getEvent(), y.getValue() + arrowY);
167           livingSpace2.addStepForLivebox(getEvent(), y.getValue() + arrowY);
168       }

170       public void drawU(UGraphic ug) {
171           if (YGauge.USE_ME)
172               ug = ug.apply(UTranslate.dy(getYGauge().getMin().getCurrentValue()));
173           final String anchor1 = message.getPart1Anchor();
174           final String anchor2 = message.getPart2Anchor();
175           if (anchor1 != null || anchor2 != null)
176               return;
177   
178           final StringBounder stringBounder = ug.getStringBounder();
179           final ArrowComponent comp = getComponent(stringBounder);
180           final XDimension2D dim = comp.getPreferredDimension(stringBounder);
181           double x1 = getPoint1(stringBounder).getCurrentValue();
182           double x2 = getPoint2(stringBounder).getCurrentValue();
183           drawMulticast(ug.apply(UTranslate.dy(comp.getPosArrow(stringBounder))));
184   
185           final Area area;
186           if (isReverse(stringBounder)) {
187               final int level1 = livingSpace1.getLevelAt(this, EventsHistoryMode.IGNORE_FUTURE_DEACTIVATE);
188               final int level2 = livingSpace2.getLevelAt(this, EventsHistoryMode.IGNORE_FUTURE_DEACTIVATE);
189               if (level1 > 0)
190                   x1 -= LIVE_DELTA_SIZE;
191   
192               x2 += LIVE_DELTA_SIZE * level2;
193               area = Area.create(x1 - x2, dim.getHeight());
194               ug = ug.apply(UTranslate.dx(x2));
195               if (isCreate())
196                   livingSpace2.drawHead(ug, (Context2D) ug, VerticalAlignment.TOP, HorizontalAlignment.RIGHT);
197   
198           } else {
199               final int level1 = livingSpace1.getLevelAt(this, EventsHistoryMode.IGNORE_FUTURE_DEACTIVATE);
200               int level2 = livingSpace2.getLevelAt(this, EventsHistoryMode.IGNORE_FUTURE_DEACTIVATE);
201               if (level2 > 0)
202                   level2 = level2 - 2;
203   
204               x1 += LIVE_DELTA_SIZE * level1;
205               x2 += LIVE_DELTA_SIZE * level2;
206               area = Area.create(x2 - x1, dim.getHeight());
207               ug = ug.apply(UTranslate.dx(x1));
208               if (isCreate())
209                   livingSpace2.drawHead(ug.apply(UTranslate.dx(area.getDimensionToUse().getWidth())), (Context2D) ug,
210                           VerticalAlignment.TOP, HorizontalAlignment.LEFT);
211   
212           }
213           comp.drawU(ug, area, (Context2D) ug);
214       }

216       private void drawMulticast(final UGraphic ug) {
217           if (message.getMulticast().size() == 0)
218               return;
219   
220           final StringBounder stringBounder = ug.getStringBounder();
221   
222           final double x1 = getPoint1(stringBounder).getCurrentValue();
223           double dy = 2;
224           for (Participant participant : message.getMulticast()) {
225               final double x2 = livingSpaces.get(participant).getPosC(stringBounder).getCurrentValue();
226               final boolean reverse = x2 < x1;
227               final ArrowComponent comp = getComponentMulticast(stringBounder, reverse);
228               final XDimension2D dim = comp.getPreferredDimension(stringBounder);
229               final Area area = Area.create(Math.abs(x2 - x1), dim.getHeight());
230               final UGraphic ug2 = ug.apply(UTranslate.dx(Math.min(x1, x2))).apply(UTranslate.dy(dy));
231               dy += 2;
232               comp.drawU(ug2, area, (Context2D) ug2);
233           }

235       }

237       public double getPreferredHeight() {
238           final Component comp = getComponent(getStringBounder());
239           final XDimension2D dim = comp.getPreferredDimension(getStringBounder());
240           double height = dim.getHeight();
241           if (isCreate())
242               height = Math.max(height, livingSpace2.getHeadPreferredDimension(getStringBounder()).getHeight());
243   
244           return height;
245       }

247       public void addConstraints() {
248           final Component comp = getComponent(getStringBounder());
249           final XDimension2D dim = comp.getPreferredDimension(getStringBounder());
250           final double width = dim.getWidth();
251   
252           Real point1 = getPoint1(getStringBounder());
253           Real point2 = getPoint2(getStringBounder());
254           if (isReverse(getStringBounder())) {
255               final int level1 = livingSpace1.getLevelAt(this, EventsHistoryMode.IGNORE_FUTURE_DEACTIVATE);
256               final int level2 = livingSpace2.getLevelAt(this, EventsHistoryMode.IGNORE_FUTURE_DEACTIVATE);
257               if (level1 > 0)
258                   point1 = point1.addFixed(-LIVE_DELTA_SIZE);
259   
260               point2 = point2.addFixed(level2 * LIVE_DELTA_SIZE);
261               point1.ensureBiggerThan(point2.addFixed(width));
262           } else {
263               final int level2 = livingSpace2.getLevelAt(this, EventsHistoryMode.IGNORE_FUTURE_DEACTIVATE);
264               if (level2 > 0)
265                   point2 = point2.addFixed(-LIVE_DELTA_SIZE);
266   
267               point2.ensureBiggerThan(point1.addFixed(width));
268           }
269       }
