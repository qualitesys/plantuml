51   public class DelayTile extends AbstractTile implements Tile {
52   
53       private final Delay delay;
54       private final TileArguments tileArguments;
55       // private Real first;
56       // private Real last;
57       private Real middle;
58       private final YGauge yGauge;
59   
60       public Event getEvent() {
61           return delay;
62       }
63   
64       public DelayTile(Delay delay, TileArguments tileArguments, YGauge currentY) {
65           super(tileArguments.getStringBounder(), currentY);
66           this.delay = delay;
67           this.tileArguments = tileArguments;
68           this.yGauge = YGauge.create(currentY.getMax(), getPreferredHeight());
69       }
70   
71       @Override
72       public YGauge getYGauge() {
73           return yGauge;
74       }
75   
76       private void init(StringBounder stringBounder) {
77           if (middle != null) {
78               return;
79           }
80           final Real first = tileArguments.getFirstLivingSpace().getPosC(stringBounder);
81           final Component comp = getComponent(stringBounder);
82           final Real last = tileArguments.getLastLivingSpace().getPosC(stringBounder);
83           this.middle = RealUtils.middle(first, last);
84   
85       }
86   
87       private Component getComponent(StringBounder stringBounder) {
88           final Component comp = tileArguments.getSkin().createComponent(delay.getUsedStyles(), ComponentType.DELAY_TEXT,
89                   null, tileArguments.getSkinParam(), delay.getText());
90           return comp;
91       }
92   
93       private double getPreferredWidth(StringBounder stringBounder) {
94           final Component comp = getComponent(stringBounder);
95           final XDimension2D dim = comp.getPreferredDimension(stringBounder);
96           return dim.getWidth();
97       }
98   
99       public void drawU(UGraphic ug) {
100           final StringBounder stringBounder = ug.getStringBounder();
101           init(stringBounder);
102           final Component comp = getComponent(stringBounder);
103           final XDimension2D dim = comp.getPreferredDimension(stringBounder);
104           final Area area = Area.create(getPreferredWidth(stringBounder), dim.getHeight());
105           final double ypos;
106           if (YGauge.USE_ME)
107               ypos = getYGauge().getMin().getCurrentValue();
108           else
109               ypos = getTimeHook().getValue();
110           tileArguments.getLivingSpaces().delayOn(ypos, dim.getHeight());
111   
112           ug = ug.apply(UTranslate.dx(getMinX().getCurrentValue()));
113           comp.drawU(ug, area, (Context2D) ug);
114       }
115   
116       public double getPreferredHeight() {
117           final Component comp = getComponent(getStringBounder());
118           final XDimension2D dim = comp.getPreferredDimension(getStringBounder());
119           return dim.getHeight();
120       }
121   
122       public void addConstraints() {
123       }
124   
125       public Real getMinX() {
126           init(getStringBounder());
127           return this.middle.addFixed(-getPreferredWidth(getStringBounder()) / 2);
128       }
129   
130       public Real getMaxX() {
131           init(getStringBounder());
132           return this.middle.addFixed(getPreferredWidth(getStringBounder()) / 2);
133       }
