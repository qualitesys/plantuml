81       public SequenceDiagramFileMakerTeoz(SequenceDiagram diagram, Rose skin, FileFormatOption fileFormatOption,
82               int index) {
83           this.index = index;
84           this.stringBounder = fileFormatOption.getDefaultStringBounder(diagram.getSkinParam());
85           this.diagram = diagram;
86           this.fileFormatOption = fileFormatOption;
87           this.skin = skin;
88           this.body = new PlayingSpaceWithParticipants(createMainTile());
89           this.footer = getFooterOrHeader(FontParam.FOOTER);
90           this.header = getFooterOrHeader(FontParam.HEADER);
91           this.annotatedBuilder = new AnnotatedBuilder(diagram, diagram.getSkinParam(), stringBounder);
92   
93           this.min1 = body.getMinX(stringBounder);
94   
95           this.title = getTitle();
96           this.legend = getLegend();
97           this.caption = annotatedBuilder.getCaption();
98   
99           this.heightEnglober1 = dolls.getOffsetForEnglobers(stringBounder);
100           this.heightEnglober2 = heightEnglober1 == 0 ? 0 : 10;
101   
102           final double totalWidth = MathUtils.max(body.calculateDimension(stringBounder).getWidth(),
103                   title.calculateDimension(stringBounder).getWidth(), footer.calculateDimension(stringBounder).getWidth(),
104                   header.calculateDimension(stringBounder).getWidth(),
105                   legend.calculateDimension(stringBounder).getWidth());
106           final double totalHeight = body.calculateDimension(stringBounder).getHeight() + heightEnglober1
107                   + heightEnglober2 + title.calculateDimension(stringBounder).getHeight()
108                   + header.calculateDimension(stringBounder).getHeight()
109                   + legend.calculateDimension(stringBounder).getHeight()
110                   + caption.calculateDimension(stringBounder).getHeight()
111                   + footer.calculateDimension(stringBounder).getHeight() + (annotatedBuilder.hasMainFrame() ? 10 : 0);
112           this.dimTotal = new XDimension2D(totalWidth, totalHeight);
113       }

154       private UGraphic goDownAndCenterForEnglobers(UGraphic ug) {
155           ug = goDown(ug, title);
156           ug = goDown(ug, header);
157           if (diagram.getLegend().getVerticalAlignment() == VerticalAlignment.TOP)
158               ug = goDown(ug, legend);
159   
160           final double dx = (dimTotal.getWidth() - body.calculateDimension(stringBounder).getWidth()) / 2;
161           return ug.apply(UTranslate.dx(dx));
162       }
163   
164       private UGraphic goDown(UGraphic ug, TextBlock size) {
165           return ug.apply(UTranslate.dy(size.calculateDimension(stringBounder).getHeight()));
166       }

168       public void printAligned(UGraphic ug, HorizontalAlignment align, final TextBlock layer) {
169           double dx = 0;
170           if (align == HorizontalAlignment.RIGHT)
171               dx = dimTotal.getWidth() - layer.calculateDimension(stringBounder).getWidth();
172           else if (align == HorizontalAlignment.CENTER)
173               dx = (dimTotal.getWidth() - layer.calculateDimension(stringBounder).getWidth()) / 2;
174   
175           layer.drawU(ug.apply(UTranslate.dx(dx)));
176       }

178       private PlayingSpace createMainTile() {
179           final RealOrigin xorigin = RealUtils.createOrigin();
180           Real xcurrent = xorigin.addAtLeast(0);
181           final RealOrigin yorigin = RealUtils.createOrigin();
182           for (Participant p : diagram.participants()) {
183               final LivingSpace livingSpace = new LivingSpace(p, diagram.getEnglober(p), skin, getSkinParam(), xcurrent,
184                       diagram.events());
185               livingSpaces.put(p, livingSpace);
186               xcurrent = livingSpace.getPosD(stringBounder).addAtLeast(0);
187           }
188   
189           final TileArguments tileArguments = new TileArguments(stringBounder, livingSpaces, skin, diagram.getSkinParam(),
190                   xorigin, yorigin);
191   
192           this.dolls = new Dolls(tileArguments);
193           final PlayingSpace mainTile = new PlayingSpace(diagram, dolls, tileArguments);
194           this.livingSpaces.addConstraints(stringBounder);
195           mainTile.addConstraints();
196           this.dolls.addConstraints(stringBounder);
197           xorigin.compileNow();
198           if (YGauge.USE_ME)
199               System.err.println("COMPILING Y");
200           yorigin.compileNow();
201           tileArguments.setBordered(mainTile);
202           return mainTile;
203       }

214                   .getMergedStyle(diagram.getSkinParam().getCurrentStyleBuilder());
215           final TextBlock compTitle = style.createTextBlockBordered(diagram.getTitle().getDisplay(),
216                   diagram.getSkinParam().getIHtmlColorSet(), diagram.getSkinParam(), Style.ID_TITLE);
217           return compTitle;

221       private TextBlock getLegend() {
222           final Display legend = diagram.getLegend().getDisplay();
223           if (Display.isNull(legend))
224               return TextBlockUtils.empty(0, 0);
225

229       public TextBlock getFooterOrHeader(final FontParam param) {
230           if (diagram.getFooterOrHeaderTeoz(param).isNull())
231               return new TeozLayer(null, stringBounder, param);
232   
233           final DisplaySection display = diagram.getFooterOrHeaderTeoz(param).withPage(index + 1, getNbPages());
234           final ISkinParam skinParam = diagram.getSkinParam();
235   
236           final StyleSignatureBasic def = param.getStyleDefinition(null);
237           final Style style = def.getMergedStyle(skinParam.getCurrentStyleBuilder());
238   
239           final PngTitler pngTitler = new PngTitler(display, style, skinParam.getIHtmlColorSet(), skinParam);
240           return new TeozLayer(pngTitler, stringBounder, param);
241       }

247       private void drawInternal(UGraphic ug, int index) {
248           body.setIndex(index);
249           final UTranslate min1translate = UTranslate.dx(-min1.getCurrentValue());
250           ug = ug.apply(min1translate);
251   
252           dolls.drawEnglobers(goDownAndCenterForEnglobers(ug),
253                   body.calculateDimension(stringBounder).getHeight() + heightEnglober1 + heightEnglober2 / 2,
254                   new SimpleContext2D(true));
255   
256           printAligned(ug, diagram.getFooterOrHeaderTeoz(FontParam.HEADER).getHorizontalAlignment(), header);
257           ug = goDown(ug, header);
258   
259           final StyleSignatureBasic def = FontParam.TITLE.getStyleDefinition(null);
260           final HorizontalAlignment titleAlignment = def.getMergedStyle(diagram.getSkinParam().getCurrentStyleBuilder())
261                   .getHorizontalAlignment();
262   
263           printAligned(ug, titleAlignment, title);
264           ug = goDown(ug, title);
265   
266           if (diagram.getLegend().getVerticalAlignment() == VerticalAlignment.TOP) {
267               printAligned(ug, diagram.getLegend().getHorizontalAlignment(), legend);
268               ug = goDown(ug, legend);
269           }
270   
271           ug = ug.apply(UTranslate.dy(heightEnglober1));
272           final TextBlock bodyFramed = annotatedBuilder.decoreWithFrame(body);
273           printAligned(ug.apply(UTranslate.dx((annotatedBuilder.hasMainFrame() ? 4 : 0))), HorizontalAlignment.CENTER,
274                   bodyFramed);
275           ug = goDown(ug, bodyFramed);
276           ug = ug.apply(UTranslate.dy(heightEnglober2));
277   
278           if (diagram.getLegend().getVerticalAlignment() == VerticalAlignment.BOTTOM) {
279               printAligned(ug, diagram.getLegend().getHorizontalAlignment(), legend);
280               ug = goDown(ug, legend);
281           }
282           printAligned(ug, HorizontalAlignment.CENTER, caption);
283           ug = goDown(ug, caption);
284   
285           printAligned(ug, diagram.getFooterOrHeaderTeoz(FontParam.FOOTER).getHorizontalAlignment(), footer);
286       }
