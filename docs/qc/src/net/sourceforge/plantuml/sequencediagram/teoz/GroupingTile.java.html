96       public GroupingTile(Iterator<Event> it, GroupingStart start, TileArguments tileArgumentsBackColorChanged,
97               TileArguments tileArgumentsOriginal, YGauge currentY) {
98           super(tileArgumentsBackColorChanged.getStringBounder(), currentY);
99           final Real firstY = currentY.getMax();
100           final StringBounder stringBounder = tileArgumentsOriginal.getStringBounder();
101           this.start = start;
102           this.display = start.getTitle().equals("group") ? Display.create(start.getComment())
103                   : Display.create(start.getTitle(), start.getComment());
104           this.skin = tileArgumentsOriginal.getSkin();
105           // this.skinParam = tileArgumentsOriginal.getSkinParam();
106           this.skinParam = tileArgumentsBackColorChanged.getSkinParam();
107   
108           final List<Real> min2 = new ArrayList<>();
109           final List<Real> max2 = new ArrayList<>();
110   
111           final List<Tile> allElses = new ArrayList<>();
112           final XDimension2D dim1 = getPreferredDimensionIfEmpty(stringBounder);
113   
114           final double h = dim1.getHeight() + MARGINY_MAGIC / 2;
115           currentY = YGauge.create(currentY.getMax().addAtLeast(h), 0);
116   
117           while (it.hasNext()) {
118               final Event ev = it.next();
119               if (ev instanceof GroupingLeaf && ((Grouping) ev).getType() == GroupingType.END)
120                   break;
121   
122               for (Tile tile : TileBuilder.buildOne(it, tileArgumentsOriginal, ev, this, currentY)) {
123                   tiles.add(tile);
124                   if (YGauge.USE_ME)
125                       currentY = tile.getYGauge();
126               }
127   
128           }
129   
130           tiles = mergeParallel(getStringBounder(), tiles);
131   
132           for (Tile tile : tiles) {
133               bodyHeight += tile.getPreferredHeight();
134               final Event ev = tile.getEvent();
135               if (ev instanceof GroupingLeaf && ((Grouping) ev).getType() == GroupingType.ELSE) {
136                   allElses.add(tile);
137                   continue;
138               }
139               min2.add(tile.getMinX().addFixed(-MARGINX));
140               final Real m = tile.getMaxX();
141               // max2.add(m == tileArgumentsOriginal.getOmega() ? m : m.addFixed(MARGINX));
142               max2.add(m.addFixed(MARGINX));
143           }
144           final double width = dim1.getWidth();
145           if (min2.size() == 0)
146               min2.add(tileArgumentsOriginal.getXOrigin());
147   
148           this.min = RealUtils.min(min2);
149           for (Tile anElse : allElses)
150               max2.add(anElse.getMaxX());
151   
152           max2.add(this.min.addFixed(width + 16));
153           this.max = RealUtils.max(max2);
154           this.yGauge = YGauge.create(firstY, getPreferredHeight());
155   
156       }

168       public void drawU(UGraphic ug) {
169           final StringBounder stringBounder = ug.getStringBounder();
170   
171           final Component comp = getComponent(stringBounder);
172           final XDimension2D dim1 = getPreferredDimensionIfEmpty(stringBounder);
173           final Area area = Area.create(max.getCurrentValue() - min.getCurrentValue(), getTotalHeight(stringBounder));
174   
175           if (YGauge.USE_ME) {
176               comp.drawU(ug.apply(new UTranslate(min.getCurrentValue(), getYGauge().getMin().getCurrentValue())), area,
177                       (Context2D) ug);
178           } else {
179               comp.drawU(ug.apply(UTranslate.dx(min.getCurrentValue())), area, (Context2D) ug);
180               drawAllElses(ug);
181           }
182   
183           double h = dim1.getHeight() + MARGINY_MAGIC / 2;
184           for (Tile tile : tiles) {
185               if (YGauge.USE_ME)
186                   ((UDrawable) tile).drawU(ug);
187               else
188                   ((UDrawable) tile).drawU(ug.apply(UTranslate.dy(h)));
189               final double preferredHeight = tile.getPreferredHeight();
190               h += preferredHeight;
191           }
192       }

195           final XDimension2D dimIfEmpty = getPreferredDimensionIfEmpty(stringBounder);

197       }

199       private void drawAllElses(UGraphic ug) {
200           final StringBounder stringBounder = ug.getStringBounder();
201           final double totalHeight = getTotalHeight(stringBounder);
202   
203           final List<Double> ys = new ArrayList<>();
204           for (Tile tile : tiles) {
205               if (tile instanceof ElseTile) {
206                   final ElseTile elseTile = (ElseTile) tile;
207                   final double ypos;
208                   if (YGauge.USE_ME)
209                       ypos = elseTile.getYGauge().getMin().getCurrentValue() - getYGauge().getMin().getCurrentValue()
210                               + MARGINY_MAGIC / 2;
211                   else
212                       ypos = elseTile.getTimeHook().getValue() - getTimeHook().getValue() + MARGINY_MAGIC / 2;
213                   ys.add(ypos);
214               }
215           }
216           ys.add(totalHeight);
217           int i = 0;
218           for (Tile tile : tiles) {
219               if (tile instanceof ElseTile) {
220                   final ElseTile elseTile = (ElseTile) tile;
221                   final Component comp = elseTile.getComponent(stringBounder);
222                   final Area area = Area.create(max.getCurrentValue() - min.getCurrentValue(), ys.get(i + 1) - ys.get(i));
223                   comp.drawU(ug.apply(new UTranslate(min.getCurrentValue(), ys.get(i))), area, (Context2D) ug);
224                   i++;
225               }
226           }
227       }

249       public static TimeHook fillPositionelTiles(StringBounder stringBounder, TimeHook y, List<Tile> tiles,
250               final List<CommonTile> local, List<CommonTile> full) {
251           for (Tile tile : mergeParallel(stringBounder, tiles)) {
252               tile.callbackY(y);
253               local.add((CommonTile) tile);
254               full.add((CommonTile) tile);
255               if (tile instanceof GroupingTile) {
256                   final GroupingTile groupingTile = (GroupingTile) tile;
257                   final double headerHeight = groupingTile.getHeaderHeight(stringBounder);
258                   final ArrayList<CommonTile> local2 = new ArrayList<>();
259                   fillPositionelTiles(stringBounder, new TimeHook(y.getValue() + headerHeight), groupingTile.tiles,
260                           local2, full);
261               }
262               y = new TimeHook(y.getValue() + tile.getPreferredHeight());
263           }
264           return y;
265   
266       }

272       private static List<Tile> mergeParallel(StringBounder stringBounder, List<Tile> tiles) {
273           if (YGauge.USE_ME)
274               return tiles;
275   
276           TileParallel pending = null;
277           tiles = removeEmptyCloseToParallel(tiles);
278           final List<Tile> result = new ArrayList<>();
279           for (Tile tile : tiles) {
280               if (result.size() > 0 && isParallel(tile)) {
281                   if (pending == null) {
282                       pending = new TileParallel(stringBounder, null);
283                       final Tile tmp = result.get(result.size() - 1);
284                       if (tmp instanceof LifeEventTile) {
285                           pending.add(result.get(result.size() - 2));
286                           pending.add(tmp);
287                           // result.set(result.size() - 1, pending);
288                           result.set(result.size() - 2, pending);
289                           result.remove(result.size() - 1);
290                       } else {
291                           pending.add(tmp);
292                           result.set(result.size() - 1, pending);
293                       }
294                   }
295                   pending.add(tile);
296               } else {
297                   result.add(tile);
298                   pending = null;
299               }
300           }
301           return result;
302       }
303   
304       private static List<Tile> removeEmptyCloseToParallel(List<Tile> tiles) {
305           final List<Tile> result = new ArrayList<>();
306           for (Tile tile : tiles) {
307               if (isParallel(tile))
308                   removeHeadEmpty(result);
309   
310               result.add(tile);
311           }
312           return result;
313   
314       }
315   
316       private static void removeHeadEmpty(List<Tile> tiles) {
317           while (tiles.size() > 0 && tiles.get(tiles.size() - 1) instanceof EmptyTile)
318               tiles.remove(tiles.size() - 1);
319   
320       }

322       public static boolean isParallel(Tile tile) {
323           return tile instanceof TileParallel == false && tile.getEvent().isParallel();
324       }
