65       public static List<Tile> buildSeveral(Iterator<Event> it, TileArguments tileArguments, Tile parent,
66               YGauge currentY) {
67           final List<Tile> tiles = new ArrayList<>();
68           while (it.hasNext()) {
69               final Event ev = it.next();
70               for (Tile tile : TileBuilder.buildOne(it, tileArguments, ev, parent, currentY)) {
71                   tiles.add(tile);
72                   final Real tmpMax = tile.getMaxX();
73                   if (YGauge.USE_ME) {
74                       final YGauge gauge = tile.getYGauge();
75                       currentY = gauge;
76                   }
77               }
78           }
79           return Collections.unmodifiableList(tiles);
80       }
81   
82       public static List<Tile> buildOne(Iterator<Event> it, TileArguments tileArguments, final Event ev, Tile parent,
83               YGauge currentY) {
84   
85           final StringBounder stringBounder = tileArguments.getStringBounder();
86           final Rose skin = tileArguments.getSkin();
87           final ISkinParam skinParam = tileArguments.getSkinParam();
88           final LivingSpaces livingSpaces = tileArguments.getLivingSpaces();
89   
90           final List<Tile> tiles = new ArrayList<>();
91           // System.err.println("TileBuilder::buildOne " + ev);
92           if (ev instanceof Message) {
93               final Message msg = (Message) ev;
94               final LivingSpace livingSpace1 = livingSpaces.get(msg.getParticipant1());
95               final LivingSpace livingSpace2 = livingSpaces.get(msg.getParticipant2());
96               boolean reverse = false;
97               Tile result = null;
98               if (msg.isSelfMessage()) {
99                   result = new CommunicationTileSelf(tileArguments.getStringBounder(), livingSpace1, msg, skin, skinParam,
100                           livingSpaces, currentY);
101               } else {
102                   result = new CommunicationTile(tileArguments.getStringBounder(), livingSpaces, msg, skin, skinParam,
103                           currentY);
104                   reverse = ((CommunicationTile) result).isReverse(stringBounder);
105               }
106               for (Note noteOnMessage : msg.getNoteOnMessages()) {
107                   final NotePosition notePosition = noteOnMessage.getPosition();
108                   if (notePosition == NotePosition.LEFT)
109                       result = new CommunicationTileNoteLeft(result, msg, skin, skinParam,
110                               reverse ? livingSpace2 : livingSpace1, noteOnMessage, currentY);
111                   else if (notePosition == NotePosition.RIGHT && msg.isSelfMessage())
112                       result = new CommunicationTileSelfNoteRight((CommunicationTileSelf) result, msg, skin, skinParam,
113                               noteOnMessage, currentY);
114                   else if (notePosition == NotePosition.RIGHT)
115                       result = new CommunicationTileNoteRight(result, msg, skin, skinParam,
116                               reverse ? livingSpace1 : livingSpace2, noteOnMessage, currentY);
117                   else if (notePosition == NotePosition.BOTTOM)
118                       result = new CommunicationTileNoteBottom(result, msg, skin, skinParam, noteOnMessage, currentY);
119                   else if (notePosition == NotePosition.TOP)
120                       result = new CommunicationTileNoteTop(result, msg, skin, skinParam, noteOnMessage, currentY);
121   
122               }
123               tiles.add(result);
124           } else if (ev instanceof MessageExo) {
125               final MessageExo exo = (MessageExo) ev;
126               final LivingSpace livingSpace1 = livingSpaces.get(exo.getParticipant());
127               Tile result = null;
128               result = new CommunicationExoTile(livingSpace1, exo, skin, skinParam, tileArguments, currentY);
129               for (Note noteOnMessage : exo.getNoteOnMessages()) {
130                   final NotePosition notePosition = exo.getNoteOnMessages().get(0).getPosition();
131                   if (notePosition == NotePosition.LEFT)
132                       result = new CommunicationTileNoteLeft(result, exo, skin, skinParam, livingSpace1, noteOnMessage,
133                               currentY);
134                   else if (notePosition == NotePosition.RIGHT)
135                       result = new CommunicationTileNoteRight(result, exo, skin, skinParam, livingSpace1, noteOnMessage,
136                               currentY);
137               }
138               tiles.add(result);
139           } else if (ev instanceof Note) {
140               final Note note = (Note) ev;
141               LivingSpace livingSpace1 = livingSpaces.get(note.getParticipant());
142               LivingSpace livingSpace2 = note.getParticipant2() == null ? null : livingSpaces.get(note.getParticipant2());
143               if (livingSpace1 == null && livingSpace2 == null) {
144                   livingSpace1 = tileArguments.getFirstLivingSpace();
145                   livingSpace2 = tileArguments.getLastLivingSpace();
146               }
147               tiles.add(new NoteTile(tileArguments.getStringBounder(), livingSpace1, livingSpace2, note, skin, skinParam,
148                       currentY));
149           } else if (ev instanceof Notes) {
150               final Notes notes = (Notes) ev;
151               tiles.add(new NotesTile(tileArguments.getStringBounder(), livingSpaces, notes, skin, skinParam, currentY));
152           } else if (ev instanceof Divider) {
153               final Divider divider = (Divider) ev;
154               tiles.add(new DividerTile(divider, tileArguments, currentY));
155           } else if (ev instanceof GroupingStart) {
156               final GroupingStart start = (GroupingStart) ev;
157               final GroupingTile groupingTile = new GroupingTile(it, start,
158                       tileArguments.withBackColorGeneral(start.getBackColorElement(), start.getBackColorGeneral()),
159                       tileArguments, currentY);
160               if (YGauge.USE_ME == false)
161                   tiles.add(new EmptyTile(4, groupingTile, currentY));
162               tiles.add(groupingTile);
163               if (YGauge.USE_ME == false)
164                   tiles.add(new EmptyTile(4, groupingTile, currentY));
165   
166           } else if (ev instanceof GroupingLeaf && ((GroupingLeaf) ev).getType() == GroupingType.ELSE) {
167               final GroupingLeaf anElse = (GroupingLeaf) ev;
168               tiles.add(new ElseTile(anElse, skin, skinParam, parent, currentY));
169           } else if (ev instanceof Reference) {
170               final Reference ref = (Reference) ev;
171               tiles.add(new ReferenceTile(ref, tileArguments.withBackColor(ref), currentY));
172           } else if (ev instanceof Delay) {
173               final Delay delay = (Delay) ev;
174               tiles.add(new DelayTile(delay, tileArguments, currentY));
175           } else if (ev instanceof HSpace) {
176               final HSpace hspace = (HSpace) ev;
177               tiles.add(new HSpaceTile(hspace, tileArguments, currentY));
178           } else if (ev instanceof LifeEvent) {
179               final LifeEvent lifeEvent = (LifeEvent) ev;
180               final LivingSpace livingSpace = livingSpaces.get(lifeEvent.getParticipant());
181               tiles.add(new LifeEventTile(lifeEvent, tileArguments, livingSpace, skin, skinParam, currentY));
182           } else if (ev instanceof Newpage) {
183               final Newpage newpage = (Newpage) ev;
184               tiles.add(new NewpageTile(newpage, tileArguments, currentY));
185           } else {
186               System.err.println("TileBuilder::Ignoring " + ev.getClass());
187           }
188           return tiles;
189       }
190   }
