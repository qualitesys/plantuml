36   package net.sourceforge.plantuml.sequencediagram.graphic;
37   
38   import java.util.Objects;
39   
40   import net.sourceforge.plantuml.Url;
41   import net.sourceforge.plantuml.awt.geom.XDimension2D;
42   import net.sourceforge.plantuml.graphic.StringBounder;
43   import net.sourceforge.plantuml.sequencediagram.NotePosition;
44   import net.sourceforge.plantuml.skin.Area;
45   import net.sourceforge.plantuml.skin.ArrowComponent;
46   import net.sourceforge.plantuml.skin.Component;
47   import net.sourceforge.plantuml.skin.Context2D;
48   import net.sourceforge.plantuml.skin.rose.Rose;
49   import net.sourceforge.plantuml.ugraphic.UGraphic;
50   import net.sourceforge.plantuml.ugraphic.UTranslate;
51   
52   class MessageArrow extends Arrow {
53   
54       private final LivingParticipantBox p1;
55       private final LivingParticipantBox p2;
56       private final Component compAliveBox;
57   
58       public MessageArrow(double startingY, Rose skin, ArrowComponent arrow, LivingParticipantBox p1, LivingParticipantBox p2,
59               Url url, Component compAliveBox) {
60           super(startingY, skin, arrow, url);
61   
62           if (p1 == p2) {
63               throw new IllegalArgumentException();
64           }
65           this.p1 = Objects.requireNonNull(p1);
66           this.p2 = Objects.requireNonNull(p2);
67           this.compAliveBox = compAliveBox;
68       }
69   
70       @Override
71       public double getActualWidth(StringBounder stringBounder) {
72           final double r = getRightEndInternal(stringBounder) - getLeftStartInternal(stringBounder);
73           assert r > 0;
74           return r;
75       }
76   
77       private double getLeftStartInternal(StringBounder stringBounder) {
78           return getParticipantAt(stringBounder, NotePosition.LEFT)
79                   .getLiveThicknessAt(stringBounder, getArrowYStartLevel(stringBounder)).getSegment().getPos2();
80       }
81   
82       private double getRightEndInternal(StringBounder stringBounder) {
83           final Segment segment = getParticipantAt(stringBounder, NotePosition.RIGHT).getLiveThicknessAt(stringBounder,
84                   getArrowYStartLevel(stringBounder)).getSegment();
85           if (segment.getLength() == 0) {
86               return segment.getPos1();
87           }
88           final double rectWidth = compAliveBox.getPreferredWidth(stringBounder);
89           return segment.getPos2() - rectWidth;
90       }
91   
92       @Override
93       public double getPreferredHeight(StringBounder stringBounder) {
94           return getArrowComponent().getPreferredHeight(stringBounder);
95       }
96   
97       @Override
98       public double getStartingX(StringBounder stringBounder) {
99           return getLeftStartInternal(stringBounder);
100       }
101   
102       @Override
103       public int getDirection(StringBounder stringBounder) {
104           final double x1 = p1.getParticipantBox().getCenterX(stringBounder);
105           final double x2 = p2.getParticipantBox().getCenterX(stringBounder);
106           if (x1 < x2) {
107               return 1;
108           }
109           return -1;
110       }
111   
112       public LivingParticipantBox getParticipantAt(StringBounder stringBounder, NotePosition position) {
113           final int direction = getDirection(stringBounder);
114           if (direction == 1 && position == NotePosition.RIGHT) {
115               return p2;
116           }
117           if (direction == 1 && position == NotePosition.LEFT) {
118               return p1;
119           }
120           if (direction == -1 && position == NotePosition.RIGHT) {
121               return p1;
122           }
123           if (direction == -1 && position == NotePosition.LEFT) {
124               return p2;
125           }
126           throw new IllegalArgumentException();
127       }
128   
129       @Override
130       public double getPreferredWidth(StringBounder stringBounder) {
131           return getArrowComponent().getPreferredWidth(stringBounder);
132       }
133   
134       @Override
135       protected void drawInternalU(UGraphic ug, double maxX, Context2D context) {
136           final StringBounder stringBounder = ug.getStringBounder();
137           ug = ug.apply(new UTranslate(getStartingX(stringBounder), getStartingY()));
138           startUrl(ug);
139           getArrowComponent().drawU(ug, new Area(getActualDimension(stringBounder)), context);
140           endUrl(ug);
141       }
142   
143       private XDimension2D getActualDimension(StringBounder stringBounder) {
144           return new XDimension2D(getActualWidth(stringBounder) - getPaddingArrowHead(), getArrowComponent()
145                   .getPreferredHeight(stringBounder));
146       }
147   
148       @Override
149       public double getArrowYStartLevel(StringBounder stringBounder) {
150           if (getArrowComponent() instanceof ArrowComponent) {
151               final ArrowComponent arrowComponent = (ArrowComponent) getArrowComponent();
152               final XDimension2D dim = new XDimension2D(arrowComponent.getPreferredWidth(stringBounder),
153                       arrowComponent.getPreferredHeight(stringBounder));
154               return getStartingY() + arrowComponent.getStartPoint(stringBounder, dim).getY();
155           }
156           return getStartingY();
157       }
158   
159       @Override
160       public double getArrowYEndLevel(StringBounder stringBounder) {
161           if (getArrowComponent() instanceof ArrowComponent) {
162               final ArrowComponent arrowComponent = (ArrowComponent) getArrowComponent();
163               final XDimension2D dim = new XDimension2D(arrowComponent.getPreferredWidth(stringBounder),
164                       arrowComponent.getPreferredHeight(stringBounder));
165               return getStartingY() + arrowComponent.getEndPoint(stringBounder, dim).getY();
166           }
167           return getStartingY() + getArrowComponent().getPreferredHeight(stringBounder);
168       }
169   
170       public double getMaxX(StringBounder stringBounder) {
171           return getRightEndInternal(stringBounder);
172       }
173   
174       public double getMinX(StringBounder stringBounder) {
175           return getLeftStartInternal(stringBounder);
176       }
177   
178       public String toString(StringBounder stringBounder) {
179           return getMinX(stringBounder) + "-" + getMaxX(stringBounder);
180       }
181   
182   }
