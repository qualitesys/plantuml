131       public DrawableSet createDrawableSet(StringBounder stringBounder) {
132           if (freeY2 != null)
133               throw new IllegalStateException();
134   
135           this.defaultLineType = useContinueLineBecauseOfDelay() ? ComponentType.CONTINUE_LINE
136                   : ComponentType.PARTICIPANT_LINE;
137   
138           for (Participant p : drawableSet.getAllParticipants())
139               prepareParticipant(stringBounder, p);
140   
141           this.freeY2 = new FrontierStackImpl(drawableSet.getHeadHeight(stringBounder),
142                   drawableSet.getAllParticipants().size());
143   
144           this.lastFreeY2 = this.freeY2;
145   
146           drawableSet.setTopStartingY(this.freeY2.getFreeY(getFullParticipantRange()));
147   
148           for (Participant p : drawableSet.getAllParticipants()) {
149               final LivingParticipantBox living = drawableSet.getLivingParticipantBox(p);
150               for (int i = 0; i < p.getInitialLife(); i++)
151                   living.getLifeLine().addSegmentVariation(LifeSegmentVariation.LARGER,
152                           freeY2.getFreeY(getFullParticipantRange()), p.getLiveSpecificBackColors());
153   
154           }
155   
156           final List<ParticipantBox> col = new ArrayList<>();
157           for (LivingParticipantBox livingParticipantBox : drawableSet.getAllLivingParticipantBox())
158               col.add(livingParticipantBox.getParticipantBox());
159   
160           constraintSet = new ConstraintSet(col, freeX);
161   
162           for (Event ev : new ArrayList<>(drawableSet.getAllEvents())) {
163               final ParticipantRange range = getParticipantRange(ev);
164               final double diffY = freeY2.getFreeY(range) - lastFreeY2.getFreeY(range);
165               // final double diffY = freeY2.diff(lastFreeY2);
166               if (autonewpage > 0 && diffY > 0 && diffY + getTotalHeight(0, stringBounder) > autonewpage)
167                   prepareNewpageSpecial(stringBounder, new Newpage(null), ev, range);
168   
169               if (ev instanceof MessageExo)
170                   prepareMessageExo(stringBounder, (MessageExo) ev, range);
171               else if (ev instanceof Message)
172                   prepareMessage(stringBounder, (Message) ev, range);
173               else if (ev instanceof Note)
174                   prepareNote(stringBounder, (Note) ev, range);
175               else if (ev instanceof Notes)
176                   prepareNotes(stringBounder, (Notes) ev, range);
177               else if (ev instanceof LifeEvent)
178                   prepareLiveEvent(stringBounder, (LifeEvent) ev, range);
179               else if (ev instanceof GroupingLeaf)
180                   prepareGroupingLeaf(stringBounder, (GroupingLeaf) ev, range);
181               else if (ev instanceof GroupingStart)
182                   prepareGroupingStart(stringBounder, (GroupingStart) ev, range);
183               else if (ev instanceof Newpage)
184                   prepareNewpage(stringBounder, (Newpage) ev, range);
185               else if (ev instanceof Divider)
186                   prepareDivider(stringBounder, (Divider) ev, range);
187               else if (ev instanceof HSpace)
188                   prepareHSpace(stringBounder, (HSpace) ev, range);
189               else if (ev instanceof Delay)
190                   prepareDelay(stringBounder, (Delay) ev, col, range);
191               else if (ev instanceof Reference)
192                   prepareReference(stringBounder, (Reference) ev, range);
193               else
194                   throw new IllegalStateException();
195   
196           }
197   
198           takeParticipantEngloberPadding(stringBounder);
199           constraintSet.takeConstraintIntoAccount(stringBounder);
200           takeParticipantEngloberTitleWidth(stringBounder);
201   
202           prepareMissingSpace(stringBounder);
203   
204           drawableSet.setDimension(new XDimension2D(freeX,
205                   getTotalHeight(freeY2.getFreeY(getFullParticipantRange()), stringBounder)));
206           return drawableSet;
207       }

210           final double padding = drawableSet.getSkinParam().getPadding(PaddingParam.BOX);
211           if (padding == 0)
212               return;
213   
214           for (Doll pe : drawableSet.getExistingParticipantEnglober(stringBounder)) {
215               final ParticipantBox first = drawableSet.getLivingParticipantBox(pe.getFirst2TOBEPRIVATE())
216                       .getParticipantBox();
217               final ParticipantBox last = drawableSet.getLivingParticipantBox(pe.getLast2TOBEPRIVATE())
218                       .getParticipantBox();
219               constraintSet.pushToLeftParticipantBox(padding, first, true);
220               constraintSet.pushToLeftParticipantBox(padding, last, false);
221           }
222       }

252       private void prepareMissingSpace(StringBounder stringBounder) {
253           freeX = constraintSet.getMaxX();
254   
255           double missingSpace1 = 0;
256           double missingSpace2 = 0;
257   
258           for (GraphicalElement ev : drawableSet.getAllGraphicalElements()) {
259               if (ev instanceof GraphicalDelayText) {
260                   final double missing = ev.getPreferredWidth(stringBounder) - freeX;
261                   if (missing > 0) {
262                       missingSpace1 = Math.max(missingSpace1, missing / 2);
263                       missingSpace2 = Math.max(missingSpace2, missing / 2);
264                   }
265                   continue;
266               }
267               final double startX = ev.getStartingX(stringBounder);
268               final double delta1 = -startX;
269               if (delta1 > missingSpace1)
270                   missingSpace1 = delta1;
271   
272               if (ev instanceof Arrow) {
273                   final Arrow a = (Arrow) ev;
274                   a.setMaxX(freeX);
275               }
276               double width = ev.getPreferredWidth(stringBounder);
277               if (ev instanceof Arrow) {
278                   final Arrow a = (Arrow) ev;
279                   if (width < a.getActualWidth(stringBounder))
280                       width = a.getActualWidth(stringBounder);
281   
282               }
283               if (ev instanceof GroupingGraphicalElementHeader) {
284                   final GroupingGraphicalElementHeader gh = (GroupingGraphicalElementHeader) ev;
285                   if (width < gh.getActualWidth(stringBounder))
286                       width = gh.getActualWidth(stringBounder);
287   
288               }
289               final double endX = startX + width;
290               final double delta2 = endX - freeX;
291               if (delta2 > missingSpace2)
292                   missingSpace2 = delta2;
293           }
294   
295           if (missingSpace1 > 0)
296               constraintSet.pushToLeft(missingSpace1);
297   
298           freeX = constraintSet.getMaxX() + missingSpace2;
299       }

379       private void prepareGroupingLeaf(StringBounder stringBounder, final GroupingLeaf m, ParticipantRange range) {
380           final GraphicalElement element;
381           final ISkinParam skinParam = new SkinParamBackcolored(drawableSet.getSkinParam(), null,
382                   m.getBackColorGeneral());
383           if (m.getType() == GroupingType.ELSE) {
384               if (m.isParallel())
385                   freeY2 = ((FrontierStack) freeY2).restore();
386   
387               final Component compElse = drawableSet.getSkin().createComponent(m.getUsedStyles(),
388                       ComponentType.GROUPING_ELSE_LEGACY, null, skinParam, Display.create(m.getComment()));
389               final Lazy lazy = new Lazy() {
390                   public double getNow() {
391                       final GraphicalElement after = drawableSet.getEvent(m.getJustAfter());
392                       if (after == null)
393                           return 0;
394   
395                       return after.getStartingY();
396                   }
397               };
398               element = new GroupingGraphicalElementElse(freeY2.getFreeY(range), compElse,
399                       inGroupableStack.getTopGroupingStructure(), m.isParallel(), lazy);
400               final double preferredHeight = element.getPreferredHeight(stringBounder);
401               freeY2 = freeY2.add(preferredHeight, range);
402               // MODIF42
403               inGroupableStack.addElement((GroupingGraphicalElementElse) element);
404           } else if (m.getType() == GroupingType.END) {
405               final List<Component> notes = new ArrayList<>();
406               for (Note noteOnMessage : m.getNoteOnMessages()) {
407                   final ISkinParam sk = noteOnMessage.getSkinParamBackcolored(drawableSet.getSkinParam());
408                   final Component note = drawableSet.getSkin().createComponentNote(noteOnMessage.getUsedStyles(),
409                           noteOnMessage.getNoteStyle().getNoteComponentType(), sk, noteOnMessage.getStrings(),
410                           noteOnMessage.getColors());
411                   notes.add(note);
412               }
413               if (m.isParallel())
414                   freeY2 = ((FrontierStack) freeY2).closeBar();
415   
416               final GroupingGraphicalElementHeader groupingHeaderStart = (GroupingGraphicalElementHeader) drawableSet
417                       .getEvent(m.getGroupingStart());
418               if (groupingHeaderStart != null) {
419                   groupingHeaderStart.setEndY(freeY2.getFreeY(range));
420                   groupingHeaderStart.addNotes(stringBounder, notes);
421               }
422               element = new GroupingGraphicalElementTail(freeY2.getFreeY(range),
423                       inGroupableStack.getTopGroupingStructure(), m.isParallel());
424               final Component comp = drawableSet.getSkin().createComponent(null, ComponentType.GROUPING_SPACE, null,

489       private void prepareLiveEvent(StringBounder stringBounder, LifeEvent lifeEvent, ParticipantRange range) {
490           final double y = freeY2.getFreeY(range);
491           final AbstractMessage message = lifeEvent.getMessage();
492           if (lifeEvent.getType() == LifeEventType.ACTIVATE) {
493               double pos = 0;
494               if (message != null) {
495                   int delta1 = 0;
496                   if (message.isCreate())
497                       delta1 += 10;
498                   else if (OptionFlags.STRICT_SELFMESSAGE_POSITION && message.isSelfMessage())
499                       delta1 += 8;
500   
501                   pos = message.getPosYstartLevel() + delta1;
502               }
503               final LifeLine line1 = drawableSet.getLivingParticipantBox(lifeEvent.getParticipant()).getLifeLine();
504               line1.addSegmentVariation(LifeSegmentVariation.LARGER, pos, lifeEvent.getSpecificColors());
505           } else if (lifeEvent.getType() == LifeEventType.DESTROY || lifeEvent.getType() == LifeEventType.DEACTIVATE) {
506               double delta = 0;
507               if (OptionFlags.STRICT_SELFMESSAGE_POSITION && message != null && message.isSelfMessage())
508                   delta += 7;
509   
510               final Participant p = lifeEvent.getParticipant();
511               final LifeLine line = drawableSet.getLivingParticipantBox(p).getLifeLine();
512               double pos2 = y;
513               if (message != null)
514                   pos2 = message.getPosYendLevel() - delta;
515   
516               line.addSegmentVariation(LifeSegmentVariation.SMALLER, pos2, lifeEvent.getSpecificColors());
517           }
518   
519           if (lifeEvent.getType() == LifeEventType.DESTROY) {
520               final Component comp = drawableSet.getSkin().createComponent(null, ComponentType.DESTROY, null,
521                       drawableSet.getSkinParam(), null);
522               final double delta = comp.getPreferredHeight(stringBounder) / 2;
523               final LivingParticipantBox livingParticipantBox = drawableSet
524                       .getLivingParticipantBox(lifeEvent.getParticipant());
525               double pos2 = y;
526               if (message == null) {
527                   pos2 = y;
528                   freeY2 = freeY2.add(comp.getPreferredHeight(stringBounder), range);
529               } else
530                   pos2 = message.getPosYendLevel() - delta;
531   
532               final LifeDestroy destroy = new LifeDestroy(pos2, livingParticipantBox.getParticipantBox(), comp);
533               drawableSet.addEvent(lifeEvent, destroy);
534           } else
535               drawableSet.addEvent(lifeEvent, new GraphicalElementLiveEvent(y));
536   
537       }

587       private void prepareParticipant(StringBounder stringBounder, Participant p) {
