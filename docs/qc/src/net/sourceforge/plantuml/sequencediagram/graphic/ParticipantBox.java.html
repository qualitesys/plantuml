113       public void drawHeadTailU(UGraphic ug, double topStartingY, boolean showHead, double positionTail) {
114           if (topStartingY == 0) {
115               throw new IllegalStateException("setTopStartingY cannot be zero");

117   
118           // final double atX = ug.getTranslateX();

121   
122           if (showHead) {
123               final double y1 = topStartingY - head.getPreferredHeight(stringBounder)
124                       - line.getPreferredHeight(stringBounder) / 2;
125               head.drawU(

129               // ug.setTranslate(atX, atY);

133               // final double y2 = positionTail - topStartingY +
134               // line.getPreferredHeight(stringBounder) / 2 - 1;
135               positionTail += line.getPreferredHeight(stringBounder) / 2 - 1;
136               // if (y2 != y22) {
137               // throw new IllegalStateException();
138               // }

158       public void drawLineU22(UGraphic ug, double startingY, final double endingY, boolean showTail, double myDelta) {
159           ug = ug.apply(UTranslate.dx(startingX));
160           if (delays.size() > 0) {
161               final StringBounder stringBounder = ug.getStringBounder();
162               for (GraphicalDelayText delay : delays) 
163                   if (delay.getStartingY() - myDelta >= startingY) {
164                       drawLineIfLowerThan(ug, startingY, delay.getStartingY() - myDelta, line, endingY);
165                       drawLineIfLowerThan(ug, delay.getStartingY() - myDelta, delay.getEndingY(stringBounder) - myDelta,
166                               delayLine, endingY);
167                       startingY = delay.getEndingY(stringBounder) - myDelta;
168                   }
169               
170               if (delays.get(delays.size() - 1).getEndingY(stringBounder) - myDelta > startingY) 
171                   startingY = delays.get(delays.size() - 1).getEndingY(stringBounder) - myDelta;
172               
173           }
174           drawLineIfLowerThan(ug, startingY, endingY, line, endingY);
175       }

177       private void drawLineIfLowerThan(UGraphic ug, double startingY, double endingY, Component comp, double limitY) {
178           startingY = Math.min(startingY, limitY);
179           endingY = Math.min(endingY, limitY);
180           if (startingY < limitY || endingY < limitY) 
181               drawLine(ug, startingY, endingY, comp);
182           
183   
184       }
185   
186       private void drawLine(UGraphic ug, double startingY, double endingY, Component comp) {
187           final StringBounder stringBounder = ug.getStringBounder();
188           comp.drawU(ug.apply(UTranslate.dy(startingY)),
189                   new Area(new XDimension2D(head.getPreferredWidth(stringBounder) + outMargin * 2, endingY
190                           - startingY)), new SimpleContext2D(false));
191       }
