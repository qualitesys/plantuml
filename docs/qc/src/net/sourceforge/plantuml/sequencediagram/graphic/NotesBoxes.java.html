46   final class NotesBoxes extends GraphicalElement implements InGroupable {
47   
48       private final List<NoteBox> notes = new ArrayList<>();
49       private final List<ParticipantBox> participants1 = new ArrayList<>();
50       private final List<ParticipantBox> participants2 = new ArrayList<>();
51   
52       NotesBoxes(double startingY) {
53           super(startingY);
54       }

56       public void add(NoteBox noteBox, ParticipantBox participantBox1, ParticipantBox participantBox2) {
57           notes.add(noteBox);
58           participants1.add(participantBox1);
59           if (participantBox2 == null)
60               participants2.add(participantBox1);
61           else
62               participants2.add(participantBox2);
63   
64       }
65   
66       public void ensureConstraints(StringBounder stringBounder, ConstraintSet constraintSet) {
67           for (int i = 0; i < notes.size(); i++) {
68               final NoteBox noteBox = notes.get(i);
69               final ParticipantBox participantBox1 = participants1.get(i);
70               final ParticipantBox participantBox2 = participants2.get(i);
71               final double width = noteBox.getPreferredWidth(stringBounder);
72               // System.err.println("i=" + i);
73               // System.err.println("width=" + width);
74               // System.err.println("participantBox1=" + participantBox1);
75               // System.err.println("participantBox2=" + participantBox2);
76               constraintSet.getConstraintBefore(participantBox1).ensureValue(width / 2);
77               constraintSet.getConstraintAfter(participantBox2).ensureValue(width / 2);
78               for (int j = i + 1; j < notes.size(); j++) {
79                   final NoteBox noteBox2 = notes.get(j);
80                   final ParticipantBox otherParticipantBox1 = participants1.get(j);
81                   final double width2 = noteBox2.getPreferredWidth(stringBounder);
82                   if (participantBox2 != otherParticipantBox1)
83                       constraintSet.getConstraint(participantBox2, otherParticipantBox1)
84                               .ensureValue((width + width2) / 2);
85               }
86           }
87       }

93               if (m < result)
94                   result = m;

109       }
