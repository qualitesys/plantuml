83           this.fileFormatOption = fileFormatOption;

89   
90           for (Event ev : diagram.events()) {
91               initializer.addEvent(ev);
92               // if (ev instanceof Message) {

97               // /*

99               // */) {
100               // initializer.addEvent(lifeEvent);
101               // }
102               // }
103               // }
104           }
105           drawableSet = initializer.createDrawableSet(stringBounder);
106           final List<Newpage> newpages = new ArrayList<>();
107           for (Event ev : drawableSet.getAllEvents())
108               if (ev instanceof Newpage)

123       private PageSplitter create(DrawableSet drawableSet, Map<Newpage, Double> positions, boolean showFootbox,
124               Display title) {
125   
126           final double headerHeight = drawableSet.getHeadHeight(stringBounder);
127           final double tailHeight = drawableSet.getTailHeight(stringBounder, showFootbox);
128           final double signatureHeight = 0;
129           final double newpageHeight = drawableSet.getSkin().createComponentNewPage(drawableSet.getSkinParam())
130                   .getPreferredHeight(stringBounder);
131   
132           return new PageSplitter(fullDimension.getHeight(), headerHeight, positions, tailHeight, signatureHeight,
133                   newpageHeight, title);
134       }

136       public ImageData createOne(OutputStream os, final int index, boolean isWithMetadata) throws IOException {
137   
138           final Page page = pages.get(index);
139           final AnnotatedBuilder builder = new AnnotatedBuilder(diagram, diagram.getSkinParam(), stringBounder);
140           double pageHeight = page.getHeight();
141           if (builder.hasMainFrame())
142               pageHeight += builder.mainFrameSuppHeight();
143   
144           final SequenceDiagramArea area = new SequenceDiagramArea(fullDimension.getWidth(), pageHeight);
145   
146           final TextBlock compTitle;
147           final TextBlock caption = builder.getCaption();
148           area.setCaptionArea(caption.calculateDimension(stringBounder));
149   
150           if (Display.isNull(page.getTitle())) {
151               compTitle = null;
152           } else {
153               final Style style = StyleSignatureBasic.of(SName.root, SName.document, SName.title)
154                       .getMergedStyle(diagram.getSkinParam().getCurrentStyleBuilder());
155               compTitle = style.createTextBlockBordered(page.getTitle(), diagram.getSkinParam().getIHtmlColorSet(),
156                       diagram.getSkinParam(), Style.ID_TITLE);
157               final XDimension2D dimTitle = compTitle.calculateDimension(stringBounder);
158               area.setTitleArea(dimTitle.getWidth(), dimTitle.getHeight());
159           }
160           area.initFooter(getPngTitler(FontParam.FOOTER, index), stringBounder);
161           area.initHeader(getPngTitler(FontParam.HEADER, index), stringBounder);
162   
163           final TextBlock legendBlock;
164           if (diagram.getLegend().isNull()) {
165               legendBlock = TextBlockUtils.empty(0, 0);
166           } else {
167               final Style style = StyleSignatureBasic.of(SName.root, SName.document, SName.legend)
168                       .getMergedStyle(diagram.getSkinParam().getCurrentStyleBuilder());
169               legendBlock = style.createTextBlockBordered(diagram.getLegend().getDisplay(),
170                       diagram.getSkinParam().getIHtmlColorSet(), diagram.getSkinParam(), Style.ID_LEGEND);
171           }
172           final XDimension2D dimLegend = legendBlock.calculateDimension(stringBounder);
173           area.setLegend(dimLegend, isLegendTop(), diagram.getLegend().getHorizontalAlignment());
174   
175           final UDrawable drawable = new UDrawable() {
176               public void drawU(UGraphic ug) {
177   
178                   double delta = 0;
179                   if (index > 0)
180                       delta = page.getNewpage1() - page.getHeaderHeight();
181   
182                   if (delta < 0)
183                       delta = 0;
184   
185                   if (compTitle != null) {
186                       final HColor back = diagram.calculateBackColor();
187                       compTitle.drawU(ug.apply(back.bg()).apply(new UTranslate(area.getTitleX(), area.getTitleY())));
188                   }
189                   caption.drawU(ug.apply(new UTranslate(area.getCaptionX(), area.getCaptionY())));
190   
191                   final double delta1 = Math.max(0, area.getLegendWidth() - area.getWidth());
192   
193                   final UTranslate forCore = new UTranslate(area.getSequenceAreaX() + delta1 / 2,
194                           area.getSequenceAreaY());
195                   TextBlock core = drawableSet.asTextBlock(delta, fullDimension.getWidth(), page,
196                           diagram.isShowFootbox());
197                   core = builder.decoreWithFrame(core);
198                   core.drawU(ug.apply(forCore));
199   
200                   drawHeader(area, ug, index);
201                   drawFooter(area, ug, index);
202   
203                   if (area.hasLegend())
204                       legendBlock.drawU(ug.apply(new UTranslate(area.getLegendX(), area.getLegendY())));
205   
206               }
207   
208           };
209           return diagram.createImageBuilder(fileFormatOption).drawable(drawable).write(os);
210       }
211   
212       private void drawFooter(SequenceDiagramArea area, UGraphic ug, int page) {
213           final PngTitler pngTitler = getPngTitler(FontParam.FOOTER, page);
214           final TextBlock text = pngTitler.getRibbonBlock();
215           if (text == null)
216               return;
217   
218           text.drawU(ug.apply(
219                   new UTranslate(area.getFooterX(diagram.getFooter().getHorizontalAlignment()), area.getFooterY())));
220       }
221   
222       private void drawHeader(SequenceDiagramArea area, UGraphic ug, int page) {
223           final PngTitler pngTitler = getPngTitler(FontParam.HEADER, page);
224           final TextBlock text = pngTitler.getRibbonBlock();
225           if (text == null)
226               return;
227   
228           text.drawU(ug.apply(
229                   new UTranslate(area.getHeaderX(diagram.getHeader().getHorizontalAlignment()), area.getHeaderY())));
230       }
231   
232       private PngTitler getPngTitler(final FontParam fontParam, int page) {
233           final ISkinParam skinParam = diagram.getSkinParam();
234           final DisplaySection display = diagram.getFooterOrHeaderTeoz(fontParam).withPage(page + 1, pages.size());
235           final StyleSignatureBasic def = fontParam.getStyleDefinition(null);
236           final Style style = def.getMergedStyle(skinParam.getCurrentStyleBuilder());
237   
238           return new PngTitler(display, style, skinParam.getIHtmlColorSet(), skinParam);
239       }
240   
241       private boolean isLegendTop() {
242           return diagram.getLegend().isNull() == false
243                   && diagram.getLegend().getVerticalAlignment() == VerticalAlignment.TOP;
244       }
