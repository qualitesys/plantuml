62       Step1Message(ParticipantRange range, StringBounder stringBounder, Message message, DrawableSet drawingSet,
63               Frontier freeY) {
64           super(range, stringBounder, message, drawingSet, freeY);
65   
66           final double x1 = getParticipantBox1().getCenterX(stringBounder);
67           final double x2 = getParticipantBox2().getCenterX(stringBounder);
68   
69           this.setConfig(isSelfMessage() ? getSelfArrowType(message) : getArrowType(message, x1, x2));
70   
71           if (isSelfMessage()) {
72               this.messageArrow = null;
73           } else {
74               final ArrowComponent comp = drawingSet.getSkin().createComponentArrow(message.getUsedStyles(), getConfig(),
75                       drawingSet.getSkinParam(), message.getLabelNumbered());
76               final Component compAliveBox = drawingSet.getSkin().createComponent(
77                       new Style[] { ComponentType.ALIVE_BOX_OPEN_OPEN.getStyleSignature()
78                               .getMergedStyle(drawingSet.getSkinParam().getCurrentStyleBuilder()) },
79                       ComponentType.ALIVE_BOX_OPEN_OPEN, null, drawingSet.getSkinParam(), null);
80   
81               this.messageArrow = new MessageArrow(freeY.getFreeY(range), drawingSet.getSkin(), comp,
82                       getLivingParticipantBox1(), getLivingParticipantBox2(), message.getUrl(), compAliveBox);
83           }
84   
85           final List<Note> noteOnMessages = message.getNoteOnMessages();
86           for (Note noteOnMessage : noteOnMessages) {
87               final ISkinParam skinParam = noteOnMessage.getSkinParamBackcolored(drawingSet.getSkinParam());
88               final Component note = drawingSet.getSkin().createComponentNote(noteOnMessage.getUsedStyles(),
89                       noteOnMessage.getNoteStyle().getNoteComponentType(), skinParam, noteOnMessage.getStrings(),
90                       noteOnMessage.getColors());
91               addNote(note);
92           }
93   
94       }

96       Frontier prepareMessage(ConstraintSet constraintSet, InGroupablesStack inGroupablesStack) {
97           final Arrow graphic = createArrow();
98           final double arrowYStartLevel = graphic.getArrowYStartLevel(getStringBounder());
99           final double arrowYEndLevel = graphic.getArrowYEndLevel(getStringBounder());
100   
101           // final double delta1 = isSelfMessage() ? 4 : 0;
102           final double delta1 = 0;
103           getMessage().setPosYstartLevel(arrowYStartLevel + delta1);
104   
105           final double length;
106           if (isSelfMessage()) {
107               length = graphic.getArrowOnlyWidth(getStringBounder()) + getLivingParticipantBox1()
108                       .getLiveThicknessAt(getStringBounder(), arrowYStartLevel).getSegment().getLength();
109           } else {
110               length = graphic.getArrowOnlyWidth(getStringBounder())
111                       + getLivingParticipantBox(NotePosition.LEFT).getLifeLine().getRightShift(arrowYStartLevel)
112                       + getLivingParticipantBox(NotePosition.RIGHT).getLifeLine().getLeftShift(arrowYStartLevel);
113           }
114   
115           incFreeY(graphic.getPreferredHeight(getStringBounder()));
116           double marginActivateAndDeactive = 0;
117           if (getMessage().isActivateAndDeactive()) {
118               marginActivateAndDeactive = 30;
119               incFreeY(marginActivateAndDeactive);
120           }
121           getDrawingSet().addEvent(getMessage(), graphic);
122   
123           if (isSelfMessage()) {
124               if (this.getConfig().isReverseDefine()) {
125                   constraintSet.getConstraintBefore(getParticipantBox1()).ensureValue(length);
126               } else {
127                   constraintSet.getConstraintAfter(getParticipantBox1()).ensureValue(length);
128               }
129           } else {
130               constraintSet.getConstraint(getParticipantBox1(), getParticipantBox2()).ensureValue(length);
131           }
132   
133           final double posYendLevel = arrowYEndLevel + marginActivateAndDeactive - delta1;
134           getMessage().setPosYendLevel(posYendLevel);
135   
136           assert graphic instanceof InGroupable;
137           if (graphic instanceof InGroupable) {
138               inGroupablesStack.addElement((InGroupable) graphic);
139               inGroupablesStack.addElement(getLivingParticipantBox1());
140               if (isSelfMessage() == false) {
141                   inGroupablesStack.addElement(getLivingParticipantBox2());
142               }
143           }
144   
145           return getFreeY();
146       }

168       private LivingParticipantBox getLivingParticipantBox(NotePosition position) {
169           if (isSelfMessage()) {
170               throw new IllegalStateException();
171           }
172           return messageArrow.getParticipantAt(getStringBounder(), position);
173       }

175       private Arrow createArrow() {
176           if (getMessage().isCreate()) {
177               return createArrowCreate();
178           }
179           if (getMessage().getNoteOnMessages().size() > 0 && isSelfMessage()) {
180               final MessageSelfArrow messageSelfArrow = createMessageSelfArrow();
181               final List<NoteBox> noteBoxes = new ArrayList<>();
182               for (int i = 0; i < getNotes().size(); i++) {
183                   final Component note = getNotes().get(i);
184                   final Note noteOnMessage = getMessage().getNoteOnMessages().get(i);
185                   noteOnMessage.temporaryProtectedUntilTeozIsStandard();
186                   noteBoxes.add(createNoteBox(getStringBounder(), messageSelfArrow, note, noteOnMessage));
187               }
188               return new ArrowAndNoteBox(getStringBounder(), messageSelfArrow, noteBoxes);
189           } else if (getMessage().getNoteOnMessages().size() > 0) {
190               final List<NoteBox> noteBoxes = new ArrayList<>();
191               for (int i = 0; i < getNotes().size(); i++) {
192                   final Component note = getNotes().get(i);
193                   final Note noteOnMessage = getMessage().getNoteOnMessages().get(i);
194                   noteOnMessage.temporaryProtectedUntilTeozIsStandard();
195                   noteBoxes.add(createNoteBox(getStringBounder(), messageArrow, note, noteOnMessage));
196               }
197               return new ArrowAndNoteBox(getStringBounder(), messageArrow, noteBoxes);
198           } else if (isSelfMessage()) {
199               return createMessageSelfArrow();
200           } else {
201               return messageArrow;
202           }
203       }
204   
205       private MessageSelfArrow createMessageSelfArrow() {
206           final double posY = getFreeY().getFreeY(getParticipantRange());
207           double deltaY = 0;
208           double deltaX = 0;
209           if (getMessage().isActivate()) {
210               deltaY -= getHalfLifeWidth();
211               if (OptionFlags.STRICT_SELFMESSAGE_POSITION) {
212                   deltaX += 5;
213               }
214           }
215           if (getMessage().isDeactivate()) {
216               deltaY += getHalfLifeWidth();
217           }
218   
219           final Style[] styles = getMessage().getUsedStyles();
220           final ArrowComponent comp = getDrawingSet().getSkin().createComponentArrow(styles, getConfig(),
221                   getDrawingSet().getSkinParam(), getMessage().getLabelNumbered());
222           return new MessageSelfArrow(posY, getDrawingSet().getSkin(), comp, getLivingParticipantBox1(), deltaY,
223                   getMessage().getUrl(), deltaX);
224       }

226       private double getHalfLifeWidth() {
227           return getDrawingSet().getSkin()
228                   .createComponent(
229                           new Style[] { ComponentType.ALIVE_BOX_OPEN_OPEN.getStyleSignature()
230                                   .getMergedStyle(getDrawingSet().getSkinParam().getCurrentStyleBuilder()) },
231                           ComponentType.ALIVE_BOX_OPEN_OPEN, null, getDrawingSet().getSkinParam(), Display.create(""))
232                   .getPreferredWidth(null) / 2;
233       }

235       private Arrow createArrowCreate() {
236           if (messageArrow == null) {
237               throw new IllegalStateException();
238           }
239           Arrow result = new ArrowAndParticipant(getStringBounder(), messageArrow, getParticipantBox2(),
240                   getDrawingSet().getSkinParam().getPadding(PaddingParam.PARTICIPANT));
241           if (getMessage().getNoteOnMessages().size() > 0) {
242               final List<NoteBox> noteBoxes = new ArrayList<>();
243               for (int i = 0; i < getNotes().size(); i++) {
244                   final Component note = getNotes().get(i);
245                   final Note noteOnMessage = getMessage().getNoteOnMessages().get(i);
246                   final NoteBox noteBox = createNoteBox(getStringBounder(), result, note, noteOnMessage);
247                   if (noteOnMessage.getPosition() == NotePosition.RIGHT) {
248                       noteBox.pushToRight(getParticipantBox2().getPreferredWidth(getStringBounder()) / 2);
249                   }
250                   noteBoxes.add(noteBox);
251               }
252               result = new ArrowAndNoteBox(getStringBounder(), result, noteBoxes);
253           }
254           getLivingParticipantBox2()
255                   .create(getFreeY().getFreeY(getParticipantRange()) + result.getPreferredHeight(getStringBounder()) / 2);
256           return result;
257       }

259       private ArrowConfiguration getSelfArrowType(Message m) {
260           ArrowConfiguration result = ArrowConfiguration.withDirectionSelf(m.getArrowConfiguration().isReverseDefine());
261           if (m.getArrowConfiguration().isDotted()) {
262               result = result.withBody(ArrowBody.DOTTED);
263           }
264           if (m.getArrowConfiguration().isHidden()) {
265               result = result.withBody(ArrowBody.HIDDEN);
266           }
267           if (m.getArrowConfiguration().isAsync()) {
268               result = result.withHead(ArrowHead.ASYNC);
269           }
270           result = result.withHead1(m.getArrowConfiguration().getDressing1().getHead());
271           result = result.withHead2(m.getArrowConfiguration().getDressing2().getHead());
272           result = result.withPart(m.getArrowConfiguration().getPart());
273           result = result.withColor(m.getArrowConfiguration().getColor());
274           result = result.withDecoration1(m.getArrowConfiguration().getDecoration1());
275           result = result.withDecoration2(m.getArrowConfiguration().getDecoration2());
276   
277           return result;
278       }
279   
280       private ArrowConfiguration getArrowType(Message m, final double x1, final double x2) {
281           if (x2 > x1) {
282               return m.getArrowConfiguration();
283           }
284           return m.getArrowConfiguration().reverse();
285       }
