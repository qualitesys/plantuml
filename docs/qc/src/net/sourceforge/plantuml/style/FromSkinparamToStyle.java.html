253       private final List<Style> styles = new ArrayList<>();
254       private final String stereo;
255       private final String key;
256   
257       public FromSkinparamToStyle(String key) {
258   
259           if (key.contains("<<")) {
260               final StringTokenizer st = new StringTokenizer(key, "<>");
261               this.key = st.nextToken();
262               this.stereo = st.hasMoreTokens() ? st.nextToken().trim() : null;
263           } else {
264               this.key = key;
265               this.stereo = null;
266           }
267   
268       }
269   
270       public void convertNow(String value, final AutomaticCounter counter) {
271           if (key.endsWith("shadowing")) {
272               if (value.equalsIgnoreCase("false"))
273                   value = "0";
274               else if (value.equalsIgnoreCase("true"))
275                   value = "3";
276           }
277           if (value.equalsIgnoreCase("right:right"))
278               value = "right";
279           if (value.equalsIgnoreCase("dotted"))
280               value = "1;3";
281           if (value.equalsIgnoreCase("dashed"))
282               value = "7;7";
283   
284           final List<Data> datas = knowlegde.get(key.toLowerCase());
285   
286           if (datas == null) {
287               if (key.equalsIgnoreCase("shadowing"))
288                   addStyle(PName.Shadowing, getShadowingValue(value, counter), SName.root);
289               else if (key.equalsIgnoreCase("noteshadowing"))
290                   addStyle(PName.Shadowing, getShadowingValue(value, counter), SName.root, SName.note);
291               return;
292           }
293   
294           final boolean complex = isComplexValue(value);
295           if (complex) {
296               if (value.contains(";")) {
297                   if (value.startsWith(";"))
298                       value = " " + value;
299                   final StringTokenizer st = new StringTokenizer(value, ";");
300                   value = st.nextToken();
301                   while (st.hasMoreTokens()) {
302                       final String read = st.nextToken();
303                       readValue(read, datas, counter);
304                   }
305               } else {
306                   readValue(value, datas, counter);
307                   return;
308               }
309           }
310   
311           if (" ".equals(value) == false)
312               for (Data data : datas)
313                   addStyle(data.propertyName, ValueImpl.regular(value, counter), data.styleNames);
314   
315       }
316   
317       private void readValue(final String read, final List<Data> datas, final AutomaticCounter counter) {
318           if (read.startsWith("text:")) {
319               final String value2 = read.split(":")[1];
320               for (Data data : datas)
321                   addStyle(PName.FontColor, ValueImpl.regular(value2, counter), data.styleNames);
322           } else if (read.startsWith("line.dotted")) {
323               for (Data data : datas)
324                   addStyle(PName.LineStyle, ValueImpl.regular("1;3", counter), data.styleNames);
325           } else if (read.startsWith("line.dashed")) {
326               for (Data data : datas)
327                   addStyle(PName.LineStyle, ValueImpl.regular("7;7", counter), data.styleNames);
328           } else if (read.toLowerCase().contains("bold")) {
329               for (Data data : datas)
330                   addStyle(PName.LineThickness, ValueImpl.regular("2", counter), data.styleNames);
331           }
332       }
333   
334       private boolean isComplexValue(String value) {
335           if (value.contains(";"))
336               return true;
337           if (value.startsWith("text:"))
338               return true;
339           return false;
340       }
341   
342       private ValueImpl getShadowingValue(final String value, final AutomaticCounter counter) {
343           if (value.equalsIgnoreCase("false") || value.equalsIgnoreCase("no"))
344               return ValueImpl.regular("0", counter);
345   
346           if (value.equalsIgnoreCase("true") || value.equalsIgnoreCase("yes"))
347               return ValueImpl.regular("3", counter);
348   
349           return ValueImpl.regular(value, counter);
350       }
351   
352       private void addStyle(PName propertyName, Value value, SName... styleNames) {
353           Map<PName, Value> map = new EnumMap<PName, Value>(PName.class);
354           map.put(propertyName, value);
355           StyleSignatureBasic sig = StyleSignatureBasic.of(styleNames);
356           if (stereo != null) {
357               map = StyleLoader.addPriorityForStereotype(map);
358               for (String s : stereo.split("\\&"))
359                   sig = sig.add(s);
360           }
361   
362           final Style style = new Style(sig, map);
363           styles.add(style);
364       }
365   
366       public List<Style> getStyles() {
367           return Collections.unmodifiableList(styles);
368       }

370       private static void addConvert(String skinparam, PName propertyName, SName... styleNames) {
371           skinparam = skinparam.toLowerCase();
372           List<Data> datas = knowlegde.get(skinparam);
373           if (datas == null) {
374               datas = new ArrayList<>();
375               knowlegde.put(skinparam, datas);
376           }
377           datas.add(new Data(styleNames, propertyName));
378       }

380       private static void addConFont(String skinparam, SName... styleNames) {
381           addConvert(skinparam + "FontSize", PName.FontSize, styleNames);
382           addConvert(skinparam + "FontStyle", PName.FontStyle, styleNames);
383           addConvert(skinparam + "FontColor", PName.FontColor, styleNames);
384           addConvert(skinparam + "FontName", PName.FontName, styleNames);
385       }
