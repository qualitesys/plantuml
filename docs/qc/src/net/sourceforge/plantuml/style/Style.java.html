70       public Style deltaPriority(int delta) {
71           if (signature.isStarred() == false)
72               throw new UnsupportedOperationException();
73   
74           final EnumMap<PName, Value> copy = new EnumMap<PName, Value>(PName.class);
75           for (Entry<PName, Value> ent : this.map.entrySet())
76               copy.put(ent.getKey(), ((ValueImpl) ent.getValue()).addPriority(delta));
77   
78           return new Style(this.signature, copy);
79   
80       }

111       public Style mergeWith(Style other, MergeStrategy strategy) {
112           if (other == null)
113               return this;
114   
115           final EnumMap<PName, Value> both = new EnumMap<PName, Value>(this.map);
116           for (Entry<PName, Value> ent : other.map.entrySet()) {
117               final Value previous = this.map.get(ent.getKey());
118               if (previous != null && previous.getPriority() > StyleLoader.DELTA_PRIORITY_FOR_STEREOTYPE
119                       && strategy == MergeStrategy.KEEP_EXISTING_VALUE_OF_STEREOTYPE)
120                   continue;
121               final PName key = ent.getKey();
122               both.put(key, ((ValueImpl) ent.getValue()).mergeWith(previous));
123           }
124           return new Style(this.signature.mergeWith(other.getSignature()), both);
125       }

127       public Style eventuallyOverride(PName param, HColor color) {
128           if (color == null)
129               return this;
130   
131           final EnumMap<PName, Value> result = new EnumMap<PName, Value>(this.map);
132           final Value old = result.get(param);
133           result.put(param, new ValueColor(color, old.getPriority()));
134           return new Style(this.signature, result);
135       }

147       public Style eventuallyOverride(Colors colors) {
148           Style result = this;
149           if (colors != null) {
150               final HColor back = colors.getColor(ColorType.BACK);
151               if (back != null)
152                   result = result.eventuallyOverride(PName.BackGroundColor, back);
153   
154               final HColor line = colors.getColor(ColorType.LINE);
155               if (line != null)
156                   result = result.eventuallyOverride(PName.LineColor, line);
157   
158               final HColor text = colors.getColor(ColorType.TEXT);
159               if (text != null)
160                   result = result.eventuallyOverride(PName.FontColor, text);
161   
162           }
163           return result;
164       }

203       public SymbolContext getSymbolContext(HColorSet set, Colors colors) {
204           HColor backColor = colors == null ? null : colors.getColor(ColorType.BACK);
205           if (backColor == null)
206               backColor = value(PName.BackGroundColor).asColor(set);
207           HColor foreColor = colors == null ? null : colors.getColor(ColorType.LINE);
208           if (foreColor == null)
209               foreColor = value(PName.LineColor).asColor(set);
210           final double deltaShadowing = value(PName.Shadowing).asDouble();
211           final double roundCorner = value(PName.RoundCorner).asDouble();
212           final double diagonalCorner = value(PName.DiagonalCorner).asDouble();
213           return new SymbolContext(backColor, foreColor).withStroke(getStroke()).withDeltaShadow(deltaShadowing)
214                   .withCorner(roundCorner, diagonalCorner);
215       }

236       private UStroke getStroke(final PName thicknessParam, final PName styleParam) {
237           final double thickness = value(thicknessParam).asDouble();
238           final String dash = value(styleParam).asString();
239           if (dash.length() == 0)
240               return new UStroke(thickness);
241   
242           try {
243               final StringTokenizer st = new StringTokenizer(dash, "-;,");
244               final double dashVisible = Double.parseDouble(st.nextToken().trim());
245               double dashSpace = dashVisible;
246               if (st.hasMoreTokens())
247                   dashSpace = Double.parseDouble(st.nextToken().trim());
248   
249               return new UStroke(dashVisible, dashSpace, thickness);
250           } catch (Exception e) {
251               return new UStroke(thickness);
252           }
253       }

292       public TextBlock createTextBlockBordered(Display note, HColorSet set, ISkinSimple spriteContainer, String id) {
293           final HorizontalAlignment alignment = this.getHorizontalAlignment();
294           final TextBlock textBlock = this.createTextBlockInternal(note, set, spriteContainer, alignment);

296           final HColor backgroundColor = this.value(PName.BackGroundColor).asColor(set);
297           final HColor lineColor = this.value(PName.LineColor).asColor(set);
298           final UStroke stroke = this.getStroke();
299           final int cornersize = this.value(PName.RoundCorner).asInt();
300           final ClockwiseTopRightBottomLeft margin = this.getMargin();
301           final ClockwiseTopRightBottomLeft padding = this.getPadding();

303                   padding, id);
304           return TextBlockUtils.withMargin(result, margin);
