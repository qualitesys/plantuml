70       public StyleBuilder loadSkin(String filename) throws IOException {
71           this.styleBuilder = new StyleBuilder(skinParam);
72   
73           final InputStream internalIs = getInputStreamForStyle(filename);
74           if (internalIs == null) {
75               Log.error("No .skin file seems to be available");
76               throw new NoStyleAvailableException();
77           }
78           final BlocLines lines2 = BlocLines.load(internalIs, new LineLocationImpl(filename, null));
79           loadSkinInternal(lines2);
80           if (this.styleBuilder == null) {
81               Log.error("No .skin file seems to be available");
82               throw new NoStyleAvailableException();
83           }
84           return this.styleBuilder;
85       }

87       public static InputStream getInputStreamForStyle(String filename) throws IOException {
88           InputStream internalIs = null;
89           SFile localFile = new SFile(filename);
90           Log.info("Trying to load style " + filename);
91           try {
92               if (localFile.exists() == false)
93                   localFile = FileSystem.getInstance().getFile(filename);
94           } catch (IOException e) {
95               Log.info("Cannot open file. " + e);
96           }
97   
98           if (localFile.exists()) {
99               Log.info("File found : " + localFile.getPrintablePath());
100               internalIs = localFile.openFile();
101           } else {
102               Log.info("File not found : " + localFile.getPrintablePath());
103               final String res = "/skin/" + filename;
104               internalIs = StyleLoader.class.getResourceAsStream(res);
105               if (internalIs != null)
106                   Log.info("... but " + filename + " found inside the .jar");
107   
108           }
109           return internalIs;
110       }

112       private void loadSkinInternal(final BlocLines lines) {
113           for (Style newStyle : getDeclaredStyles(lines, styleBuilder))
114               this.styleBuilder.loadInternal(newStyle.getSignature(), newStyle);
115   
116       }

118       private final static String KEYNAMES = "[-.\\w(), ]+?";
119       private final static Pattern2 keyName = MyPattern.cmpile("^[:]?(" + KEYNAMES + ")([%s]+\\*)?[%s]*\\{$");
120       private final static Pattern2 propertyAndValue = MyPattern.cmpile("^([\\w]+):?[%s]+(.*?);?$");
121       private final static Pattern2 closeBracket = MyPattern.cmpile("^\\}$");
122   
123       public static Collection<Style> getDeclaredStyles(BlocLines lines, AutomaticCounter counter) {
124           lines = lines.eventuallyMoveAllEmptyBracket();
125           final List<Style> result = new ArrayList<>();
126           final CssVariables variables = new CssVariables();
127           StyleScheme scheme = StyleScheme.REGULAR;
128   
129           Context context = new Context();
130           final List<Map<PName, Value>> maps = new ArrayList<Map<PName, Value>>();
131           boolean inComment = false;
132           for (StringLocated s : lines) {
133               String trimmed = s.getTrimmed().getString();
134   
135               if (trimmed.startsWith("/*") || trimmed.endsWith("*/"))
136                   continue;
137               if (trimmed.startsWith("/'") || trimmed.endsWith("'/"))
138                   continue;
139   
140               if (trimmed.startsWith("/*") || trimmed.startsWith("/'")) {
141                   inComment = true;
142                   continue;
143               }
144               if (trimmed.endsWith("*/") || trimmed.endsWith("'/")) {
145                   inComment = false;
146                   continue;
147               }
148               if (inComment)
149                   continue;
150   
151               if (trimmed.matches("@media.*dark.*\\{")) {
152                   scheme = StyleScheme.DARK;
153                   continue;
154               }
155   
156               if (trimmed.startsWith("--")) {
157                   variables.learn(trimmed);
158                   continue;
159               }
160   
161               final int x = trimmed.lastIndexOf("//");
162               if (x != -1)
163                   trimmed = trimmed.substring(0, x).trim();
164   
165               final Matcher2 mKeyNames = keyName.matcher(trimmed);
166               if (mKeyNames.find()) {
167                   String names = mKeyNames.group(1);
168                   final boolean isRecurse = mKeyNames.group(2) != null;
169                   if (isRecurse)
170                       names += "*";
171   
172                   context = context.push(names);
173                   maps.add(new EnumMap<PName, Value>(PName.class));
174                   continue;
175               }
176               final Matcher2 mPropertyAndValue = propertyAndValue.matcher(trimmed);
177               if (mPropertyAndValue.find()) {
178                   final PName key = PName.getFromName(mPropertyAndValue.group(1), scheme);
179                   final String value = variables.value(mPropertyAndValue.group(2));
180                   if (key != null && maps.size() > 0)
181                       maps.get(maps.size() - 1).put(key, //
182                               scheme == StyleScheme.REGULAR ? //
183                                       ValueImpl.regular(value, counter) : ValueImpl.dark(value, counter));
184   
185                   continue;
186               }
187               final Matcher2 mCloseBracket = closeBracket.matcher(trimmed);
188               if (mCloseBracket.find()) {
189                   if (context.size() > 0) {
190                       final Collection<StyleSignatureBasic> signatures = context.toSignatures();
191                       for (StyleSignatureBasic signature : signatures) {
192                           Map<PName, Value> tmp = maps.get(maps.size() - 1);
193                           if (signature.isWithDot())
194                               tmp = addPriorityForStereotype(tmp);
195                           if (tmp.size() > 0) {
196                               final Style style = new Style(signature, tmp);
197                               result.add(style);
198                           }
199                       }
200                       context = context.pop();
201                       maps.remove(maps.size() - 1);
202                   } else {
203                       scheme = StyleScheme.REGULAR;
204                   }
205               }
206           }
207   
208           return Collections.unmodifiableList(result);
209   
210       }
211   
212       public static Map<PName, Value> addPriorityForStereotype(Map<PName, Value> tmp) {
213           final Map<PName, Value> result = new EnumMap<>(PName.class);
214           for (Entry<PName, Value> ent : tmp.entrySet())
215               result.put(ent.getKey(), ((ValueImpl) ent.getValue()).addPriority(DELTA_PRIORITY_FOR_STEREOTYPE));
216   
217           return result;
218       }
