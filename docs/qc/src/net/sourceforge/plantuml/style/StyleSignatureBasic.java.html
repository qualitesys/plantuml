49   public class StyleSignatureBasic implements StyleSignature {
50   
51       private final Set<String> names = new LinkedHashSet<>();
52       private final boolean withDot;
53   
54       public StyleSignatureBasic(String s) {
55           if (s.contains("*") || s.contains("&") || s.contains("-"))
56               throw new IllegalArgumentException();
57   
58           this.withDot = s.contains(".");
59           this.names.add(clean(s));
60       }
61   
62       public static StyleSignatureBasic empty() {
63           return new StyleSignatureBasic(false);
64       }
65   
66       private StyleSignatureBasic(boolean withDot) {
67           this.withDot = withDot;
68       }
69   
70       private StyleSignatureBasic(boolean withDot, Collection<String> copy) {
71           this.names.addAll(copy);
72           this.withDot = withDot;
73       }
74   
75       public StyleSignatureBasic addClickable(Url url) {
76           if (url == null)
77               return this;
78   
79           final Set<String> result = new LinkedHashSet<>(names);
80           result.add(SName.clickable.name());
81           return new StyleSignatureBasic(withDot, result);
82   
83       }
84   
85       public StyleSignatureBasic add(String s) {
86           if (s == null)
87               return this;
88   
89           if (s.contains("&"))
90               throw new IllegalArgumentException();
91   
92           final Set<String> result = new LinkedHashSet<>(names);
93           result.add(clean(s));
94           return new StyleSignatureBasic(withDot || s.contains("."), result);
95       }
96   
97       public StyleSignatureBasic add(SName name) {
98           return add(name.name().toLowerCase().replace("_", ""));
99       }
100   
101       public StyleSignatureBasic addStar() {
102           final Set<String> result = new LinkedHashSet<>(names);
103           result.add("*");
104           return new StyleSignatureBasic(withDot, result);
105       }
106   
107       public boolean isStarred() {
108           return names.contains("*");
109       }
110   
111       @Override
112       public boolean equals(Object arg) {
113           final StyleSignatureBasic other = (StyleSignatureBasic) arg;
114           return this.names.equals(other.names);
115       }
116   
117       @Override
118       public int hashCode() {
119           return names.hashCode();
120       }
121   
122       @Override
123       public String toString() {
124           final StringBuilder result = new StringBuilder();
125           for (String n : names) {
126               if (result.length() > 0)
127                   result.append('.');
128   
129               result.append(n);
130           }
131           return result.toString() + " " + withDot;
132       }
133   
134       public boolean matchAll(StyleSignatureBasic other) {
135           final boolean namesContainsStar = names.contains("*");
136           if (other.isStarred() && namesContainsStar == false)
137               return false;
138   
139           final int depthInNames = depthFromTokens(other.names);
140   
141           for (String token : names) {
142               if (token.equals("*"))
143                   continue;
144   
145               if (namesContainsStar && depthInNames != -1 && depthFromToken(token) != -1) {
146                   // depth comparaison
147                   if (depthInNames < depthFromToken(token))
148                       return false;
149               } else {
150                   if (other.names.contains(token) == false)
151                       return false;
152               }
153   
154           }
155           return true;
156       }
157   
158       private static int depthFromToken(String token) {
159           if (token.startsWith("depth("))
160               return Integer.parseInt(token.substring("depth(".length(), token.length() - 1));
161           return -1;
162       }
163   
164       private static int depthFromTokens(Collection<String> tokens) {
165           for (String token : tokens) {
166               final int depth = depthFromToken(token);
167               if (depth != -1)
168                   return depth;
169           }
170           return -1;
171   
172       }
173   
174       public final Set<String> getNames() {
175           return Collections.unmodifiableSet(names);
176       }
177   
178       public static StyleSignatureBasic of(SName... names) {
179           final List<String> result = new ArrayList<>();
180           for (SName name : names)
181               result.add(name.name().toLowerCase().replace("_", ""));
182   
183           return new StyleSignatureBasic(false, result);
184       }
185   
186       public StyleSignature forStereotypeItself(Stereotype stereotype) {
187           if (stereotype == null || stereotype.getStyleNames().size() == 0)
188               return this;
189   
190           final StyleSignatures result = new StyleSignatures();
191           for (String name : stereotype.getStyleNames()) {
192               final List<String> tmp = new ArrayList<>(names);
193               tmp.add(SName.stereotype.name().toLowerCase().replace("_", ""));
194               tmp.add(clean(name));
195               result.add(new StyleSignatureBasic(false, tmp));
196           }
197           return result;
198   
199       }
200   
201       @Override
202       public StyleSignature withTOBECHANGED(Stereotype stereotype) {
203           if (stereotype == null || stereotype.getStyleNames().size() == 0)
204               return this;
205   
206           final StyleSignatures result = new StyleSignatures();
207           for (String name : stereotype.getStyleNames()) {
208               final List<String> tmp = new ArrayList<>(names);
209               tmp.add(clean(name));
210               result.add(new StyleSignatureBasic(true, tmp));
211           }
212           return result;
213       }
214   
215       @Override
216       public StyleSignature with(Stereostyles stereostyles) {
217           if (stereostyles.isEmpty())
218               return this;
219           final List<String> result = new ArrayList<>(names);
220           for (String name : stereostyles.getStyleNames())
221               result.add(clean(name));
222   
223           return new StyleSignatureBasic(true, result);
224       }
225   
226       private String clean(String name) {
227           return name.toLowerCase().replace("_", "").replace(".", "");
228       }
229   
230       public StyleSignatureBasic mergeWith(List<Style> others) {
231           final List<String> copy = new ArrayList<>(names);
232           for (Style other : others)
233               for (String s : other.getSignature().getNames())
234                   copy.add(s);
235   
236           return new StyleSignatureBasic(withDot, copy);
237       }
238   
239       public StyleSignatureBasic mergeWith(StyleSignatureBasic other) {
240           final List<String> copy = new ArrayList<>(names);
241           copy.addAll(other.names);
242           return new StyleSignatureBasic(withDot || other.withDot, copy);
243       }
244   
245       @Override
246       public Style getMergedStyle(StyleBuilder styleBuilder) {
247           if (styleBuilder == null)
248               return null;
249   
250           return styleBuilder.getMergedStyle(this);
251       }
252   
253       public boolean match(Stereotype stereotype) {
254           for (String s : stereotype.getMultipleLabels())
255               if (names.contains(clean(s)))
256                   return true;
257   
258           return false;
259       }
260   
261       public final boolean isWithDot() {
262           return withDot;
263       }
