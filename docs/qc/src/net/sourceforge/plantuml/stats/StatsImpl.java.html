74       private StatsLine createDataLineSession(final ParsedGenerated data) {
75           final Map<StatsColumn, Object> result = new EnumMap<StatsColumn, Object>(StatsColumn.class);
76           final long id = data.getId();
77           if (id != -1) {
78               result.put(StatsColumn.SESSION_ID, id);
79           }
80           result.put(StatsColumn.VERSION, data.getVersion());
81           final long creationTime = data.parsedDated().getCreationTime();
82           final long modificationTime = data.parsedDated().getModificationTime();
83           result.put(StatsColumn.STARTING, new Date(creationTime));
84           result.put(StatsColumn.LAST, new Date(modificationTime));
85           result.put(StatsColumn.DURATION_STRING, new HumanDuration(modificationTime - creationTime));
86           result.put(StatsColumn.PARSED_COUNT, data.parsedDated().getNb());
87           result.put(StatsColumn.PARSED_MEAN_TIME, data.parsedDated().getMean());
88           result.put(StatsColumn.PARSED_STANDARD_DEVIATION, data.parsedDated().getStandardDeviation());
89           result.put(StatsColumn.PARSED_MAX_TIME, data.parsedDated().getMax());
90           result.put(StatsColumn.GENERATED_COUNT, data.generatedDated().getNb());
91           result.put(StatsColumn.GENERATED_MEAN_TIME, data.generatedDated().getMean());
92           result.put(StatsColumn.GENERATED_STANDARD_DEVIATION, data.generatedDated().getStandardDeviation());
93           result.put(StatsColumn.GENERATED_MAX_TIME, data.generatedDated().getMax());
94           return new StatsLineImpl(result);
95       }

97       private StatsLine createLineByDiagramType(String key, NumberAnalyzed parse, NumberAnalyzed generate) {
98           final Map<StatsColumn, Object> result = new EnumMap<StatsColumn, Object>(StatsColumn.class);
99           result.put(StatsColumn.DIAGRAM_TYPE, key);
100           result.put(StatsColumn.PARSED_COUNT, parse.getNb());
101           result.put(StatsColumn.PARSED_MEAN_TIME, parse.getMean());
102           result.put(StatsColumn.PARSED_STANDARD_DEVIATION, parse.getStandardDeviation());
103           result.put(StatsColumn.PARSED_MAX_TIME, parse.getMax());
104           result.put(StatsColumn.GENERATED_COUNT, generate.getNb());
105           result.put(StatsColumn.GENERATED_MEAN_TIME, generate.getMean());
106           result.put(StatsColumn.GENERATED_STANDARD_DEVIATION, generate.getStandardDeviation());
107           result.put(StatsColumn.GENERATED_MAX_TIME, generate.getMax());
108           return new StatsLineImpl(result);
109       }

111       public StatsTable getLastSessions() {
112           final StatsTableImpl result = new StatsTableImpl("Statistics");
113           for (ParsedGenerated histo : historicalData.getHistorical()) {
114               result.addLine(createDataLineSession(histo));
115           }
116           if (historicalData.current() != null) {
117               result.addLine(createDataLineSession(historicalData.current()));
118           }
119           result.addLine(createDataLineSession(fullEver));
120           return result;
121       }

139       private StatsTable getByDiagramType(String name, Map<String, ParsedGenerated> data) {
140           final StatsTableImpl result = new StatsTableImpl(name);
141           final TreeSet<String> keys = new TreeSet<>(data.keySet());
142           final NumberAnalyzed totalParsing = new NumberAnalyzed();
143           final NumberAnalyzed totalGenerating = new NumberAnalyzed();
144           for (String key : keys) {
145               final NumberAnalyzed parse = data.get(key).parsed();
146               final NumberAnalyzed generate = data.get(key).generated();
147               totalParsing.add(parse);
148               totalGenerating.add(generate);
149               result.addLine(createLineByDiagramType(key, parse, generate));
150           }
151           result.addLine(createLineByDiagramType("Total", totalParsing, totalGenerating));
152           return result;
153       }
