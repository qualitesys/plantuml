64       private final DateFormat formatter = DateFormat.getDateTimeInstance(DateFormat.SHORT, DateFormat.MEDIUM);

68       public XmlConverter(Stats stats) {
69           this.stats = stats;
70       }

72       private Document getDocument() throws ParserConfigurationException {
73           final DocumentBuilder builder = XmlFactories.newDocumentBuilder();
74           final Document document = builder.newDocument();
75           document.setXmlStandalone(true);
76   
77           final Element root = (Element) document.createElement("plantuml".toUpperCase());
78           document.appendChild(root);
79   
80           // final Element elt1 = (Element) document.createElement("totalLaunch".toUpperCase());
81           // elt1.setTextContent("" + stats.totalLaunch());
82           // root.appendChild(elt1);
83   
84           addNode(root, document, stats.getLastSessions());
85           addNode(root, document, stats.getCurrentSessionByDiagramType());
86           addNode(root, document, stats.getCurrentSessionByFormat());
87           addNode(root, document, stats.getAllByDiagramType());
88           addNode(root, document, stats.getAllByFormat());
89   
90           return document;
91       }
92   
93       private void addNode(Element root, Document document, StatsTable table) {
94           final Element elt = (Element) document.createElement(toXmlName(table.getName()).toUpperCase());
95   
96           for (StatsLine statsLine : table.getLines()) {
97               final Element line = (Element) document.createElement("line".toUpperCase());
98               for (StatsColumn col : table.getColumnHeaders()) {
99                   final Element value = (Element) document.createElement(col.name());
100                   // value.setAttribute("value", toText(statsLine.getValue(col)));
101                   value.setTextContent(toText(statsLine.getValue(col)));
102                   line.appendChild(value);
103               }
104               elt.appendChild(line);
105           }
106   
107           root.appendChild(elt);
108       }
109   
110       private String toXmlName(String name) {
111           return name.replaceAll("\\W+", "_");
112       }
113   
114       private String toText(Object tmp) {
115           if (tmp instanceof Date) {
116               return "" + ((Date) tmp).getTime();
117           }
118           if (tmp == null) {
119               return "";
120           }
121           return tmp.toString();
122       }
123   
124       private Transformer getTransformer() throws TransformerException {
125           final Transformer transformer = XmlFactories.newTransformer();
126           transformer.setOutputProperty(OutputKeys.STANDALONE, "yes");
127           return transformer;
128       }
129   
130       public void createXml(OutputStream os) throws TransformerException, ParserConfigurationException {
131           final DOMSource source = new DOMSource(getDocument());
132           final StreamResult scrResult = new StreamResult(os);
133           getTransformer().transform(source, scrResult);
134   
135       }

137       public String toHtml() {
138           final StringBuilder result = new StringBuilder();
139           result.append("<html>");
140           result.append("<style type=\"text/css\">");
141           result.append("body { font-family: arial, helvetica, sans-serif; font-size: 12px; font-weight: normal; color: black; background: white;}");
142           result.append("th,td { font-size: 12px;}");
143           result.append("table { border-collapse: collapse; border-style: none;}");
144           result.append("</style>");
145           result.append("<h2>Statistics</h2>");
146           printTableHtml(result, stats.getLastSessions());
147           result.append("<h2>Current session statistics</h2>");
148           printTableHtml(result, stats.getCurrentSessionByDiagramType());
149           result.append("<p>");
150           printTableHtml(result, stats.getCurrentSessionByFormat());
151           result.append("<h2>General statistics since ever</h2>");
152           printTableHtml(result, stats.getAllByDiagramType());
153           result.append("<p>");
154           printTableHtml(result, stats.getAllByFormat());
155           result.append("</html>");
156           return result.toString();
157       }

159       private void printTableHtml(StringBuilder result, StatsTable table) {
160           final Collection<StatsColumn> headers = table.getColumnHeaders();
161           result.append("<table border=1 cellspacing=0 cellpadding=2>");
162           result.append(getHtmlHeader(headers));
163           final List<StatsLine> lines = table.getLines();
164           for (int i = 0; i < lines.size(); i++) {
165               final StatsLine line = lines.get(i);
166               final boolean bold = i == lines.size() - 1;
167               result.append(getCreoleLine(headers, line, bold));
168   
169           }
170           result.append("</table>");
171       }
172   
173       private String getCreoleLine(Collection<StatsColumn> headers, StatsLine line, boolean bold) {
174           final StringBuilder result = new StringBuilder();
175           if (bold) {
176               result.append("<tr bgcolor=#f0f0f0>");
177           } else {
178               result.append("<tr bgcolor=#fcfcfc>");
179           }
180           for (StatsColumn col : headers) {
181               final Object v = line.getValue(col);
182               if (v instanceof Long || v instanceof HumanDuration) {
183                   result.append("<td align=right>");
184               } else {
185                   result.append("<td>");
186               }
187               if (bold) {
188                   result.append("<b>");
189               }
190               if (v instanceof Long) {
191                   result.append(String.format("%,d", v));
192               } else if (v instanceof Date) {
193                   result.append(formatter.format(v));
194               } else if (v == null || v.toString().length() == 0) {
195                   result.append(" ");
196               } else {
197                   result.append(v.toString());
198               }
199               if (bold) {
200                   result.append("</b>");
201               }
202               result.append("</td>");
203           }
204           result.append("</tr>");
205           return result.toString();
206       }
207   
208       private String getHtmlHeader(Collection<StatsColumn> headers) {
209           final StringBuilder sb = new StringBuilder();
210           sb.append("<tr bgcolor=#e0e0e0>");
211           for (StatsColumn col : headers) {
212               sb.append("<td><b>");
213               sb.append(col.getTitle().replace(BackSlash.BS_BS_N, "<br>"));
214               sb.append("</b></td>");
215           }
216           sb.append("</tr>");
217           return sb.toString();
218       }
