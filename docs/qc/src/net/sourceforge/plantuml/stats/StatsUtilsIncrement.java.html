54   public class StatsUtilsIncrement {
55   
56       final private static Preferences prefs = StatsUtils.prefs;
57   
58       final private static ConcurrentMap<String, ParsedGenerated> byTypeEver = StatsUtils.byTypeEver;
59       final private static ConcurrentMap<String, ParsedGenerated> byTypeCurrent = StatsUtils.byTypeCurrent;
60   
61       final private static FormatCounter formatCounterCurrent = StatsUtils.formatCounterCurrent;
62       final private static FormatCounter formatCounterEver = StatsUtils.formatCounterEver;
63   
64       public static void onceMoreParse(long duration, Class<? extends Diagram> type) {
65           if (StatsUtils.fullEver == null || StatsUtils.historicalData == null) {
66               return;
67           }
68           getByTypeCurrent(type).parsed().addValue(duration);
69           final ParsedGenerated byTypeEver = getByTypeEver(type);
70           byTypeEver.parsed().addValue(duration);
71           StatsUtils.fullEver.parsed().addValue(duration);
72           StatsUtils.historicalData.current().parsed().addValue(duration);
73   
74           StatsUtils.historicalData.current().parsed().save(prefs);
75           StatsUtils.fullEver.parsed().save(prefs);
76           byTypeEver.parsed().save(prefs);
77           realTimeExport();
78       }
79   
80       public static void onceMoreGenerate(long duration, Class<? extends Diagram> type, FileFormat fileFormat) {
81           if (StatsUtils.fullEver == null || StatsUtils.historicalData == null) {
82               return;
83           }
84           if (formatCounterCurrent == null || formatCounterEver == null) {
85               return;
86           }
87           getByTypeCurrent(type).generated().addValue(duration);
88           final ParsedGenerated byTypeEver = getByTypeEver(type);
89           byTypeEver.generated().addValue(duration);
90           StatsUtils.fullEver.generated().addValue(duration);
91           StatsUtils.historicalData.current().generated().addValue(duration);
92           formatCounterCurrent.plusOne(fileFormat, duration);
93           formatCounterEver.plusOne(fileFormat, duration);
94   
95           formatCounterEver.save(prefs, fileFormat);
96           StatsUtils.historicalData.current().generated().save(prefs);
97           StatsUtils.fullEver.generated().save(prefs);
98           byTypeEver.generated().save(prefs);
99           realTimeExport();
100       }
101   
102       private static ParsedGenerated getByTypeCurrent(Class<? extends Diagram> type) {
103           final String name = name(type);
104           ParsedGenerated n = byTypeCurrent.get(name);
105           if (n == null) {
106               byTypeCurrent.putIfAbsent(name, ParsedGenerated.createVolatileDated());
107               n = byTypeCurrent.get(name);
108           }
109           return n;
110       }
111   
112       private static ParsedGenerated getByTypeEver(Class<? extends Diagram> type) {
113           final String name = name(type);
114           ParsedGenerated n = byTypeEver.get(name);
115           if (n == null) {
116               byTypeEver.putIfAbsent(name, ParsedGenerated.loadDated(prefs, "type." + name));
117               n = byTypeEver.get(name);
118           }
119           return n;
120       }
121   
122       private static String name(Class<? extends Diagram> type) {
123           if (PSystemErrorUtils.isDiagramError(type)) {
124               return "Error";
125           }
126           if (type == ActivityDiagram3.class) {
127               return "ActivityDiagramBeta";
128           }
129           if (type == PSystemSalt.class) {
130               return "Salt";
131           }
132           if (type.getSimpleName().equals("PSystemSudoku")) {
133               return "Sudoku";
134           }
135           if (type == PSystemDot.class) {
136               return "Dot";
137           }
138           if (type == PSystemWelcome.class) {
139               return "Welcome";
140           }
141           if (type.getSimpleName().equals("PSystemDitaa")) {
142               return "Ditaa";
143           }
144           if (type.getSimpleName().equals("PSystemJcckit")) {
145               return "Jcckit";
146           }
147           if (type == PSystemMath.class) {
148               return "Math";
149           }
150           final String name = type.getSimpleName();
151           if (name.endsWith("Diagram")) {
152               return name;
153           }
154           // return "Other " + name;
155           return "Other";
156       }

158       private static final Lock lockXml = new ReentrantLock();
159       private static final Lock lockHtml = new ReentrantLock();
160   
161       private static void realTimeExport() {
162           if (StatsUtils.realTimeStats) {
163               final Stats stats = StatsUtils.getStatsLazzy();
164               try {
165                   if (StatsUtils.xmlStats && lockXml.tryLock())
166                       try {
167                           StatsUtils.xmlOutput(stats);
168                       } finally {
169                           lockXml.unlock();
170                       }
171                   if (StatsUtils.htmlStats && lockHtml.tryLock())
172                       try {
173                           StatsUtils.htmlOutput(stats);
174                       } finally {
175                           lockHtml.unlock();
176                       }
177               } catch (Exception e) {
178                   Logme.error(e);
179               }
180   
181           }
182       }
