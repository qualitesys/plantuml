76       private static byte[] load(String name) throws IOException {
77           final InputStream tmp = PSystemDedication.class.getResourceAsStream(name + ".png");
78           final ByteArrayOutputStream baos = new ByteArrayOutputStream();

87           final TinyHashableString sentence = new TinyHashableString(line);
88           for (Dedication dedication : all) {
89               final BufferedImage image = dedication.getImage(sentence);
90               if (image != null) {

93           }
94           return null;
95       }
