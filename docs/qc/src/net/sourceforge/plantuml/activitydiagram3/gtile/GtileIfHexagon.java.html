76       private GtileIfHexagon(Swimlane swimlane, List<Gtile> gtiles, List<Branch> branches) {
77           super(gtiles, swimlane, 20);
78   
79           this.branches = branches;
80   
81           this.shape1 = getShape1(swimlane);
82           this.shape2 = Gtiles.diamondEmpty(swimlane, getStringBounder(), skinParam(), getDefaultStyleDefinitionDiamond(),
83                   branches.get(0).getColor());
84   
85           final XDimension2D dimShape1 = shape1.calculateDimension(stringBounder);
86           this.pushDown(dimShape1.getHeight());
87   
88           final double posShape;
89           final XDimension2D totalDim;
90           if (branches.size() == 2) {
91               final Gtile tile0 = branches.get(0).getGtile();
92               final Gtile tile1 = branches.get(1).getGtile();
93   
94               final double northHook0 = tile0.getGPoint(GPoint.NORTH_HOOK).getCoord().getDx();
95               final double width0 = tile0.calculateDimension(getStringBounder()).getWidth();
96               final double northHook1 = tile1.getGPoint(GPoint.NORTH_HOOK).getCoord().getDx();
97               final double width1 = tile1.calculateDimension(getStringBounder()).getWidth();
98   
99               final double shlen1 = shape1.calculateDimension(getStringBounder()).getWidth();
100               final double shlen = shape1.getGPoint(GPoint.EAST_HOOK).getCoord().getDx()
101                       - shape1.getGPoint(GPoint.WEST_HOOK).getCoord().getDx();
102   
103               System.err.println("shlen=" + shlen);
104   
105               System.err.println("northHook0=" + northHook0);
106               System.err.println("width0=" + width0);
107   
108               System.err.println("northHook1=" + northHook1);
109               System.err.println("width1=" + width1);
110   
111               final double alpha = northHook0;
112               final double beta = width1 - northHook1;
113               System.err.println("alpha=" + alpha);
114               System.err.println("beta=" + beta);
115   
116               final double wantedWidth = alpha + shlen + beta;
117               System.err.println("wantedWidth=" + wantedWidth);
118               double diff = wantedWidth - width0 - width1;
119               if (diff < 0)
120                   diff = 0;
121               System.err.println("diff=" + diff);
122               this.setMargin(diff + 10);
123               totalDim = calculateDimensionRaw(stringBounder);
124               final double p2 = totalDim.getWidth() - beta;
125               posShape = (alpha + p2) / 2;
126           } else {
127               totalDim = calculateDimensionRaw(stringBounder);
128               posShape = totalDim.getWidth() / 2;
129           }
130   
131           this.positionShape1 = UTranslate.dx(posShape).compose(shape1.getCoord(GPoint.NORTH_BORDER).reverse());
132           this.positionShape2 = new UTranslate(posShape, totalDim.getHeight())
133                   .compose(shape2.getCoord(GPoint.SOUTH_BORDER).reverse());
134   
135       }

138       protected UTranslate getCoordImpl(String name) {
139           if (name.equals(GPoint.NORTH_HOOK))
140               return shape1.getCoord(name).compose(positionShape1);
141           if (name.equals(GPoint.SOUTH_HOOK))
142               return shape2.getCoord(name).compose(positionShape2);
143           return super.getCoordImpl(name);
144       }

146       private Gtile getShape1(Swimlane swimlane) {
147           GtileHexagonInside tmp = Gtiles.hexagonInside(swimlane, getStringBounder(), skinParam(),
148                   getDefaultStyleDefinitionDiamond(), branches.get(0).getColor(), branches.get(0).getLabelTest());
149   
150           final TextBlock tmp0 = branches.get(0).getTextBlockPositive();
151           if (branches.size() == 1) {
152               return Gtiles.withSouthMargin(tmp.withSouthLabel(tmp0), 10);
153           }
154   
155           final TextBlock tmp1 = branches.get(1).getTextBlockPositive();
156           return Gtiles.withSouthMargin(tmp.withWestLabel(tmp0).withEastLabel(tmp1), 10);
157       }

159       @Override
160       public XDimension2D calculateDimension(StringBounder stringBounder) {
161           final XDimension2D rawDim = calculateDimensionRaw(stringBounder);
162           if (branches.size() == 2) {
163               final XDimension2D shape1Dim = shape1.calculateDimension(stringBounder);
164               final double shape1max = this.positionShape1.getDx() + shape1Dim.getWidth();
165               return XDimension2D.atLeast(rawDim, shape1max, 0);
166   
167           }
168           // return MathUtils.max(rawDim, shape1Dim);
169           return rawDim;
170       }

172       private XDimension2D calculateDimensionRaw(StringBounder stringBounder) {
173           final double height2 = shape2.calculateDimension(stringBounder).getHeight();
174           final XDimension2D nude = super.calculateDimension(stringBounder);
175           // +30 to be done only when branches.size()==1 ?
176           return XDimension2D.delta(nude, 0, height2 + 30);
177       }

214       @Override
215       public Collection<GConnection> getInnerConnections() {
216           if (branches.size() == 1) {
217               final GConnection arrow1 = new GConnectionVerticalDown(positionShape1, shape1.getGPoint(GPoint.SOUTH_HOOK),
218                       getPosition(0), gtiles.get(0).getGPoint(GPoint.NORTH_HOOK), TextBlockUtils.EMPTY_TEXT_BLOCK);
219               final GConnection arrow2 = new GConnectionVerticalDown(getPosition(0),
220                       gtiles.get(0).getGPoint(GPoint.SOUTH_HOOK), positionShape2, shape2.getGPoint(GPoint.NORTH_HOOK),
221                       TextBlockUtils.EMPTY_TEXT_BLOCK);
222               return Arrays.asList(arrow1, arrow2);
223           } else if (branches.size() == 2) {
224               final GConnection arrow1 = new GConnectionHorizontalThenVerticalDown(positionShape1,
225                       shape1.getGPoint(GPoint.WEST_HOOK), getPosition(0), gtiles.get(0).getGPoint(GPoint.NORTH_HOOK),
226                       TextBlockUtils.EMPTY_TEXT_BLOCK);
227               final GConnection arrow2 = new GConnectionHorizontalThenVerticalDown(positionShape1,
228                       shape1.getGPoint(GPoint.EAST_HOOK), getPosition(1), gtiles.get(1).getGPoint(GPoint.NORTH_HOOK),
229                       TextBlockUtils.EMPTY_TEXT_BLOCK);
230   
231               final GConnection arrow3 = new GConnectionVerticalDownThenHorizontal(getPosition(0),
232                       gtiles.get(0).getGPoint(GPoint.SOUTH_HOOK), positionShape2, shape2.getGPoint(GPoint.WEST_HOOK),
233                       TextBlockUtils.EMPTY_TEXT_BLOCK);
234               final GConnection arrow4 = new GConnectionVerticalDownThenHorizontal(getPosition(1),
235                       gtiles.get(1).getGPoint(GPoint.SOUTH_HOOK), positionShape2, shape2.getGPoint(GPoint.EAST_HOOK),
236                       TextBlockUtils.EMPTY_TEXT_BLOCK);
237   
238               return Arrays.asList(arrow1, arrow2, arrow3, arrow4);
239           }
240           return super.getInnerConnections();
241       }
