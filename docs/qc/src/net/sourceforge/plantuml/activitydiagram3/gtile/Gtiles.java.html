61       static public GtileHexagonInside hexagonInside(Swimlane swimlane, StringBounder stringBounder, ISkinParam skinParam,
62               StyleSignatureBasic styleSignature, HColor color, Display label) {
63   
64           final Style style = styleSignature.getMergedStyle(skinParam.getCurrentStyleBuilder());
65           final HColor borderColor = style.value(PName.LineColor).asColor(skinParam.getIHtmlColorSet());
66           final HColor backColor = color == null
67                   ? style.value(PName.BackGroundColor).asColor(skinParam.getIHtmlColorSet())
68                   : color;
69           final FontConfiguration fcTest = style.getFontConfiguration(skinParam.getIHtmlColorSet());
70   
71           final Sheet sheet = Parser
72                   .build(fcTest, skinParam.getDefaultTextAlignment(HorizontalAlignment.LEFT), skinParam, CreoleMode.FULL)
73                   .createSheet(label);
74           final SheetBlock1 sheetBlock1 = new SheetBlock1(sheet, LineBreakStrategy.NONE, skinParam.getPadding());
75           final TextBlock tbTest = new SheetBlock2(sheetBlock1, Hexagon.asStencil(sheetBlock1), new UStroke());
76   
77           return new GtileHexagonInside(stringBounder, tbTest, skinParam, backColor, borderColor, swimlane);
78       }
