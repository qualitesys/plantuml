141       static public Ftile create(Swimlane swimlane, HColor borderColor, HColor backColor, Rainbow arrowColor,
142               FtileFactory ftileFactory, ConditionStyle conditionStyle, ConditionEndStyle conditionEndStyle,
143               Branch branch1, Branch branch2, ISkinParam skinParam, StringBounder stringBounder,
144               FontConfiguration fcArrow, FontConfiguration fcTest, Url url) {
145           final ConditionalBuilder builder = new ConditionalBuilder(swimlane, borderColor, backColor, arrowColor,
146                   ftileFactory, conditionStyle, conditionEndStyle, branch1, branch2, skinParam, stringBounder, fcArrow,
147                   fcTest, url);
148           if (isEmptyOrOnlySingleStopOrSpot(branch2) && isEmptyOrOnlySingleStopOrSpot(branch1) == false)
149               return builder.createDown(builder.branch1, builder.branch2);
150   
151           if (branch1.isEmpty() && branch2.isOnlySingleStopOrSpot())
152               return builder.createDown(builder.branch1, builder.branch2);
153   
154           if (isEmptyOrOnlySingleStopOrSpot(branch1) && isEmptyOrOnlySingleStopOrSpot(branch2) == false)
155               return builder.createDown(builder.branch2, builder.branch1);
156   
157           if (branch2.isEmpty() && branch1.isOnlySingleStopOrSpot())
158               return builder.createDown(builder.branch2, builder.branch1);
159   
160           return builder.createWithLinks();
161           // return builder.createWithDiamonds();
162           // return builder.createNude();
163       }

165       private static boolean isEmptyOrOnlySingleStopOrSpot(Branch branch) {
166           return branch.isEmpty() || branch.isOnlySingleStopOrSpot();
167       }

169       private Ftile createDown(Branch branch1, Branch branch2) {
170           final Ftile tile1 = new FtileMinWidthCentered(branch1.getFtile(), 30);
171           final Ftile tile2 = new FtileMinWidthCentered(branch2.getFtile(), 30);
172           final TextBlock tb1 = getLabelPositive(branch1);
173           final TextBlock tb2 = getLabelPositive(branch2);
174           final Ftile diamond1 = getShape1(false, tb1, tb2);
175           final Ftile diamond2 = getShape2(branch1, branch2, true);
176           if (branch2.isOnlySingleStopOrSpot())
177               return FtileIfDown.create(diamond1, diamond2, swimlane, FtileUtils.addHorizontalMargin(tile1, 10),
178                       arrowColor, conditionEndStyle, ftileFactory, branch2.getFtile(), branch2.getOut());
179   
180           if (branch1.isOnlySingleStopOrSpot())
181               return FtileIfDown.create(diamond1, diamond2, swimlane, FtileUtils.addHorizontalMargin(tile2, 10),
182                       arrowColor, conditionEndStyle, ftileFactory, branch1.getFtile(), branch1.getOut());
183   
184           if (branch1.isEmpty())
185               return FtileIfDown.create(diamond1, diamond2, swimlane, FtileUtils.addHorizontalMargin(tile2, 10),
186                       arrowColor, conditionEndStyle, ftileFactory, null, null);
187   
188           return FtileIfDown.create(diamond1, diamond2, swimlane, FtileUtils.addHorizontalMargin(tile1, 10), arrowColor,
189                   conditionEndStyle, ftileFactory, null, branch2.getOut());
190       }

197       private Ftile createWithDiamonds() {
198           final Ftile diamond1 = getDiamond1(true);
199           final Ftile diamond2 = getShape2(branch1, branch2, false);
200           final FtileIfWithDiamonds ftile = new FtileIfWithDiamonds(diamond1, tile1, tile2, diamond2, swimlane,
201                   stringBounder);
202           final XDimension2D label1 = getLabelPositive(branch1).calculateDimension(stringBounder);
203           final XDimension2D label2 = getLabelPositive(branch2).calculateDimension(stringBounder);
204           final double diff1 = ftile.computeMarginNeedForBranchLabe1(stringBounder, label1);
205           final double diff2 = ftile.computeMarginNeedForBranchLabe2(stringBounder, label2);
206           Ftile result = FtileUtils.addHorizontalMargin(ftile, diff1, diff2);
207           final double suppHeight = ftile.computeVerticalMarginNeedForBranchs(stringBounder, label1, label2);
208           result = FtileUtils.addVerticalMargin(result, suppHeight, 0);
209           return result;
210       }

212       private Ftile createWithLinks() {
213           Ftile diamond1 = getDiamond1(true);
214           if (url != null)
215               diamond1 = new FtileWithUrl(diamond1, url);
216   
217           final Ftile diamond2 = getShape2(branch1, branch2, false);
218           final Ftile tmp1 = FtileUtils.addHorizontalMargin(tile1, 10);
219           final Ftile tmp2 = FtileUtils.addHorizontalMargin(tile2, 10);
220           final FtileIfWithLinks ftile = new FtileIfWithLinks(diamond1, tmp1, tmp2, diamond2, swimlane, arrowColor,
221                   conditionEndStyle, stringBounder);
222           final XDimension2D label1 = getLabelPositive(branch1).calculateDimension(stringBounder);
223           final XDimension2D label2 = getLabelPositive(branch2).calculateDimension(stringBounder);
224           final double diff1 = ftile.computeMarginNeedForBranchLabe1(stringBounder, label1);
225           final double diff2 = ftile.computeMarginNeedForBranchLabe2(stringBounder, label2);
226           final double suppHeight = ftile.computeVerticalMarginNeedForBranchs(stringBounder, label1, label2);
227           Ftile result = ftile.addLinks(branch1, branch2, stringBounder);
228           result = FtileUtils.addHorizontalMargin(result, diff1, diff2);
229           result = FtileUtils.addVerticalMargin(result, suppHeight, 0);
230           return result;
231       }

237       private Ftile getShape1(boolean eastWest, TextBlock tb1, TextBlock tb2) {
238           final Display labelTest = branch1.getLabelTest();
239   
240           final Sheet sheet = Parser.build(fontTest, skinParam.getDefaultTextAlignment(HorizontalAlignment.LEFT),
241                   skinParam, CreoleMode.FULL).createSheet(labelTest);
242           final SheetBlock1 sheetBlock1 = new SheetBlock1(sheet, diamondLineBreak, skinParam.getPadding());
243   
244           final UStroke thickness = tile1
245                   .getThickness(getStyleSignatureDiamond().getMergedStyle(skinParam.getCurrentStyleBuilder()));
246           final TextBlock tbTest = new SheetBlock2(sheetBlock1, Hexagon.asStencil(sheetBlock1), thickness);
247   
248           final Ftile shape1;
249           if (conditionStyle == ConditionStyle.INSIDE_HEXAGON) {
250               if (eastWest)
251                   shape1 = new FtileDiamondInside(tbTest, tile1.skinParam(), backColor, borderColor, swimlane)
252                           .withWestAndEast(tb1, tb2);
253               else
254                   shape1 = new FtileDiamondInside(tbTest, tile1.skinParam(), backColor, borderColor, swimlane)
255                           .withSouth(tb1).withEast(tb2);
256   
257           } else if (conditionStyle == ConditionStyle.EMPTY_DIAMOND) {
258               if (eastWest)
259                   shape1 = new FtileDiamond(tile1.skinParam(), backColor, borderColor, swimlane).withNorth(tbTest)
260                           .withWestAndEast(tb1, tb2);
261               else
262                   shape1 = new FtileDiamond(tile1.skinParam(), backColor, borderColor, swimlane).withNorth(tbTest)
263                           .withSouth(tb1).withEast(tb2);
264   
265           } else if (conditionStyle == ConditionStyle.INSIDE_DIAMOND) {
266               if (eastWest)
267                   shape1 = new FtileDiamondSquare(tbTest, tile1.skinParam(), backColor, borderColor, swimlane)
268                           .withWestAndEast(tb1, tb2);
269               else
270                   shape1 = new FtileDiamondSquare(tbTest, tile1.skinParam(), backColor, borderColor, swimlane)
271                           .withSouth(tb1).withEast(tb2);
272           } else {
273               throw new IllegalStateException();
274           }
275           return shape1;
276       }

283       private Ftile getShape2(Branch branch1, Branch branch2, boolean useNorth) {
284           final Ftile shape2;
285           if (conditionEndStyle == ConditionEndStyle.HLINE)
286               return new FtileEmpty(tile1.skinParam(), 0, Hexagon.hexagonHalfSize, swimlane);
287   
288           // else use default ConditionEndStyle.DIAMOND
289           if (hasTwoBranches()) {
290               final Display out1 = branch1.getFtile().getOutLinkRendering().getDisplay();
291               final TextBlock tbout1 = out1 == null ? null
292                       : out1.create7(fontArrow, HorizontalAlignment.LEFT, ftileFactory.skinParam(),
293                               CreoleMode.SIMPLE_LINE);
294               final Display out2 = branch2.getFtile().getOutLinkRendering().getDisplay();
295               final TextBlock tbout2 = out2 == null ? null
296                       : out2.create7(fontArrow, HorizontalAlignment.LEFT, ftileFactory.skinParam(),
297                               CreoleMode.SIMPLE_LINE);
298               FtileDiamond tmp = new FtileDiamond(tile1.skinParam(), backColor, borderColor, swimlane);
299               tmp = useNorth ? tmp.withNorth(tbout1) : tmp.withWest(tbout1);
300               tmp = tmp.withEast(tbout2);
301               shape2 = tmp;
302           } else {
303               shape2 = new FtileEmpty(tile1.skinParam(), 0, Hexagon.hexagonHalfSize / 2, swimlane);
304           }
305           return shape2;
306       }

308       public boolean hasTwoBranches() {
309           return tile1.calculateDimension(stringBounder).hasPointOut()
310                   && tile2.calculateDimension(stringBounder).hasPointOut();
311       }
