289       private void drawWhenSwimlanes(UGraphic ug, TextBlock full) {
290           final StringBounder stringBounder = ug.getStringBounder();
291           final UTranslate titleHeightTranslate = getTitleHeightTranslate(stringBounder);
292   
293           drawTitlesBackground(ug);
294   
295           final XDimension2D dimensionFull = full.calculateDimension(stringBounder);
296           int i = 0;
297           assert dividers.size() == swimlanes().size() + 1;
298           for (Swimlane swimlane : swimlanesSpecial()) {
299               final LaneDivider divider1 = dividers.get(i);
300   
301               final double xpos = swimlane.getTranslate().getDx() + swimlane.getMinMax().getMinX();
302               final HColor back = swimlane.getColors().getColor(ColorType.BACK);
303               if (back != null && back.isTransparent() == false) {
304                   final LaneDivider divider2 = dividers.get(i + 1);
305                   final UGraphic background = ug.apply(back.bg()).apply(back)
306                           .apply(UTranslate.dx(xpos - divider1.getX2()));
307                   final double width = swimlane.getActualWidth() + divider1.getX2() + divider2.getX1();
308                   final double height = dimensionFull.getHeight() + titleHeightTranslate.getDy();
309                   background.draw(new URectangle(width, height).ignoreForCompressionOnX().ignoreForCompressionOnY());
310               }
311   
312               full.drawU(new UGraphicInterceptorOneSwimlane(ug, swimlane, swimlanes()).apply(swimlane.getTranslate())
313                       .apply(getTitleHeightTranslate(stringBounder)));
314   
315               final double dividerWith = divider1.calculateDimension(stringBounder).getWidth();
316               divider1.drawU(ug.apply(UTranslate.dx(xpos - dividerWith)));
317               i++;
318           }
319   
320           final Cross cross = new Cross(ug.apply(getTitleHeightTranslate(stringBounder)));
321           full.drawU(cross);
322           cross.flushUg();
323   
324           drawTitles(ug);
325   
326       }

362       private void computeSizeInternal(UGraphic ug, TextBlock full) {
363           computeDrawingWidths(ug, full);
364   
365           double min = skinParam.swimlaneWidth();
366   
367           if (min == ISkinParam.SWIMLANE_WIDTH_SAME) {
368               for (Swimlane swimlane : swimlanes()) {
369                   min = Math.max(min, getWidthWithoutTitle(swimlane));
370               }
371           }
372           final StringBounder stringBounder = ug.getStringBounder();
373   
374           for (int i = 0; i < swimlanesSpecial().size(); i++) {
375               final Swimlane swimlane = swimlanesSpecial().get(i);
376               final double swimlaneActualWidth = MathUtils.max(min, getWidthWithoutTitle(swimlane));
377               swimlane.setWidth(swimlaneActualWidth);
378           }
379   
380           final UTranslate titleHeightTranslate = getTitleHeightTranslate(stringBounder);
381           final XDimension2D dimensionFull = full.calculateDimension(stringBounder);
382   
383           dividers.clear();
384           double xpos = 0;
385           for (int i = 0; i < swimlanesSpecial().size(); i++) {
386               final Swimlane swimlane = swimlanesSpecial().get(i);
387               double x1 = getHalfMissingSpace(stringBounder, i, min);
388               double x2 = getHalfMissingSpace(stringBounder, i + 1, min);
389               final LaneDivider laneDivider = new LaneDivider(skinParam, x1, x2,
390                       dimensionFull.getHeight() + titleHeightTranslate.getDy());
391               dividers.add(laneDivider);
392   
393               final double xx = xpos + laneDivider.getWidth() - swimlane.getMinMax().getMinX()
394                       + (swimlane.getActualWidth() - getWidthWithoutTitle(swimlane)) / 2.0;
395               swimlane.setTranslate(UTranslate.dx(xx));
396   
397               xpos += swimlane.getActualWidth() + laneDivider.getWidth();
398           }
399           assert dividers.size() == swimlanes().size() + 1;
400       }

402       public double getHalfMissingSpace(StringBounder stringBounder, int i, double min) {
403           if (i == 0 || i > swimlanesSpecial().size()) {
404               return 5;
405           }
406           final Swimlane swimlane = swimlanesSpecial().get(i - 1);
407           final double swimlaneActualWidth = Math.max(min, getWidthWithoutTitle(swimlane));
408           final double titleWidth = getTitle(swimlane).calculateDimension(stringBounder).getWidth();
409           if (titleWidth <= swimlaneActualWidth) {
410               return 5;
411           }
412           assert titleWidth > swimlaneActualWidth;
413           return Math.max(5, 5 + (titleWidth - swimlaneActualWidth) / 2);
414       }
