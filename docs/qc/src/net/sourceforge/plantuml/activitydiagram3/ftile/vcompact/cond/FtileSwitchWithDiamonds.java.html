38   import java.util.ArrayList;
39   import java.util.Collection;
40   import java.util.Collections;
41   import java.util.List;
42   
43   import net.sourceforge.plantuml.activitydiagram3.Branch;
44   import net.sourceforge.plantuml.activitydiagram3.ftile.Ftile;
45   import net.sourceforge.plantuml.activitydiagram3.ftile.FtileGeometry;
46   import net.sourceforge.plantuml.activitydiagram3.ftile.Swimlane;
47   import net.sourceforge.plantuml.graphic.StringBounder;
48   import net.sourceforge.plantuml.ugraphic.UGraphic;
49   import net.sourceforge.plantuml.ugraphic.UTranslate;
50   
51   public class FtileSwitchWithDiamonds extends FtileSwitchNude {
52   
53       private static final double SUPP15 = 15;
54   
55       static enum Mode {
56           BIG_DIAMOND, SMALL_DIAMOND
57       };
58   
59       protected final Ftile diamond1;
60       protected final Ftile diamond2;
61       protected final List<Branch> branches;
62       protected final Mode mode;
63       private final double w13;
64       private final double w9;
65   
66       public FtileSwitchWithDiamonds(List<Ftile> tiles, List<Branch> branches, Swimlane in, Ftile diamond1,
67               Ftile diamond2, StringBounder stringBounder) {
68           super(tiles, in);
69           this.branches = branches;
70           this.diamond1 = diamond1;
71           this.diamond2 = diamond2;
72   
73           w13 = diamond1.calculateDimension(stringBounder).getWidth()
74                   - tiles.get(0).calculateDimension(stringBounder).getRight()
75                   - tiles.get(tiles.size() - 1).calculateDimension(stringBounder).getLeft();
76           w9 = getW9(stringBounder);
77           if (w13 > w9)
78               mode = Mode.BIG_DIAMOND;
79           else
80               mode = Mode.SMALL_DIAMOND;
81   
82       }
83   
84       private double getW9(StringBounder stringBounder) {
85           double result = 0;
86           for (int i = 1; i < tiles.size() - 1; i++)
87               result += tiles.get(i).calculateDimension(stringBounder).getWidth();
88   
89           return result;
90       }
91   
92       @Override
93       final public Collection<Ftile> getMyChildren() {
94           final Collection<Ftile> result = new ArrayList<>(super.getMyChildren());
95           result.add(diamond1);
96           result.add(diamond2);
97           return Collections.unmodifiableCollection(result);
98       }
99   
100       protected double getYdelta1a(StringBounder stringBounder) {
101           return 20;
102       }
103   
104       final protected double getYdelta1b(StringBounder stringBounder) {
105           return 10;
106       }
107   
108       @Override
109       final protected FtileGeometry calculateDimensionInternalSlow(StringBounder stringBounder) {
110           final FtileGeometry dim1 = diamond1.calculateDimension(stringBounder);
111           final FtileGeometry dim2 = diamond2.calculateDimension(stringBounder);
112   
113           final double nudeHeight = super.calculateDimensionInternalSlow(stringBounder).getHeight();
114   
115           if (mode == Mode.BIG_DIAMOND) {
116               final double height = dim1.getHeight() + nudeHeight + dim2.getHeight() + getYdelta1a(stringBounder)
117                       + getYdelta1b(stringBounder);
118               final FtileGeometry tile0 = tiles.get(0).calculateDimension(stringBounder);
119               final double width = tile0.getWidth() + SUPP15 + w13 + SUPP15
120                       + +tiles.get(tiles.size() - 1).calculateDimension(stringBounder).getWidth();
121               return new FtileGeometry(width, height, tile0.getLeft() + SUPP15 + dim1.getLeft(), 0, height);
122   
123           } else {
124               final FtileGeometry dimNude = super.calculateDimensionInternalSlow(stringBounder);
125               final FtileGeometry all = dim1.appendBottom(dimNude).appendBottom(dim2);
126               return all.addDim(0, getYdelta1a(stringBounder) + getYdelta1b(stringBounder));
127           }
128   
129       }
130   
131       @Override
132       final public void drawU(UGraphic ug) {
133           final StringBounder stringBounder = ug.getStringBounder();
134   
135           ug.apply(getTranslateDiamond1(stringBounder)).draw(diamond1);
136           if (mode == Mode.BIG_DIAMOND)
137               for (Ftile tile : tiles)
138                   tile.drawU(ug.apply(getTranslateOf(tile, stringBounder)));
139           else
140               super.drawU(ug.apply(getTranslateMain(stringBounder)));
141   
142           if (calculateDimension(stringBounder).hasPointOut())
143               ug.apply(getTranslateDiamond2(stringBounder)).draw(diamond2);
144   
145       }
146   
147       final protected UTranslate getTranslateOf(Ftile tile, StringBounder stringBounder) {
148           final UTranslate main = getTranslateMain(stringBounder);
149           if (mode == Mode.BIG_DIAMOND) {
150               double dx = 0;
151               final double suppx = (w13 - w9) / (tiles.size() - 1);
152               for (int i = 0; i < tiles.size() - 1; i++) {
153                   if (tile == tiles.get(i))
154                       return main.compose(UTranslate.dx(dx));
155   
156                   dx += tiles.get(i).calculateDimension(stringBounder).getWidth() + suppx;
157               }
158               if (tile == tiles.get(tiles.size() - 1)) {
159                   final double dx9 = tiles.get(0).calculateDimension(stringBounder).getWidth() + w13 + SUPP15 + SUPP15;
160                   return main.compose(UTranslate.dx(dx9));
161               }
162               throw new IllegalArgumentException();
163   
164           }
165           return getTranslateNude(tile, stringBounder).compose(main);
166       }
167   
168       final protected UTranslate getTranslateMain(StringBounder stringBounder) {
169           final FtileGeometry dimDiamond1 = diamond1.calculateDimension(stringBounder);
170           final double dy1 = dimDiamond1.getHeight() + getYdelta1a(stringBounder);
171           return UTranslate.dy(dy1);
172       }
173   
174       final protected UTranslate getTranslateDiamond1(StringBounder stringBounder) {
175           final double y1 = 0;
176           final FtileGeometry dimTotal = calculateDimensionInternal(stringBounder);
177           final FtileGeometry dimDiamond1 = diamond1.calculateDimension(stringBounder);
178           final double x1 = dimTotal.getLeft() - dimDiamond1.getLeft();
179           return new UTranslate(x1, y1);
180       }
181   
182       final protected UTranslate getTranslateDiamond2(StringBounder stringBounder) {
183           final FtileGeometry dimTotal = calculateDimensionInternal(stringBounder);
184           final FtileGeometry dimDiamond2 = diamond2.calculateDimension(stringBounder);
185           final double y2 = dimTotal.getHeight() - dimDiamond2.getHeight();
186           final double x2 = dimTotal.getLeft() - dimDiamond2.getWidth() / 2;
187           return new UTranslate(x2, y2);
188       }

190   }
