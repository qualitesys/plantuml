65   public class ParallelBuilderMerge extends AbstractParallelFtilesBuilder {
66   
67       public ParallelBuilderMerge(ISkinParam skinParam, StringBounder stringBounder, List<Ftile> all) {
68           super(skinParam, stringBounder, all);
69       }
70   
71       @Override
72       protected Ftile doStep1(Ftile inner) {
73           Ftile result = inner;
74           final List<Connection> conns = new ArrayList<>();
75   
76           final Ftile black = new FtileBlackBlock(skinParam(), list99.get(0).getSwimlaneIn());
77           double x = 0;
78           for (Ftile tmp : list99) {
79               final XDimension2D dim = tmp.calculateDimension(getStringBounder());
80               Style style = getStyleSignature().getMergedStyle(skinParam().getCurrentStyleBuilder());
81               final Rainbow def = Rainbow.build(style, skinParam().getIHtmlColorSet());
82               final Rainbow rainbow = tmp.getInLinkRendering().getRainbow(def);
83               conns.add(new ConnectionIn(black, tmp, x, rainbow));
84               x += dim.getWidth();
85           }
86   
87           result = FtileUtils.addConnection(result, conns);
88           ((FtileBlackBlock) black).setBlackBlockDimension(result.calculateDimension(getStringBounder()).getWidth(),
89                   barHeight);
90   
91           return new FtileAssemblySimple(black, result);
92       }
93   
94       @Override
95       protected Ftile doStep2(Ftile inner, Ftile result) {
96           final Style style = getStyleSignature().getMergedStyle(skinParam().getCurrentStyleBuilder());
97   
98           final HColor borderColor = style.value(PName.LineColor).asColor(skinParam().getIHtmlColorSet());
99           final HColor backColor = style.value(PName.BackGroundColor).asColor(skinParam().getIHtmlColorSet());
100   
101           final Ftile out = new FtileDiamond(skinParam(), backColor, borderColor, swimlaneOutForStep2());
102           result = new FtileAssemblySimple(result, out);
103           final List<Connection> conns = new ArrayList<>();
104           final UTranslate diamondTranslate = result.getTranslateFor(out, getStringBounder());
105   
106           double x = 0;
107           for (Ftile tmp : list99) {
108               final XDimension2D dim = tmp.calculateDimension(getStringBounder());
109               final UTranslate translate0 = new UTranslate(x, barHeight);
110               final Rainbow def = Rainbow.build(style, skinParam().getIHtmlColorSet());
111               final Rainbow rainbow = tmp.getOutLinkRendering().getRainbow(def);
112               if (tmp.calculateDimension(getStringBounder()).hasPointOut())
113                   conns.add(new ConnectionHorizontalThenVertical(tmp, out, rainbow, translate0, diamondTranslate));
114   
115               x += dim.getWidth();
116   
117           }
118           return FtileUtils.addConnection(result, conns);
119       }

121       class ConnectionHorizontalThenVertical extends AbstractConnection /* implements ConnectionTranslatable */ {
122   
123           private final Rainbow arrowColor;
124           private final UTranslate diamondTranslate;
125           private final UTranslate translate0;
126   
127           public ConnectionHorizontalThenVertical(Ftile tile, Ftile diamond, Rainbow arrowColor, UTranslate translate0,
128                   UTranslate diamondTranslate) {
129               super(tile, diamond);
130               this.arrowColor = arrowColor;
131               this.diamondTranslate = diamondTranslate;
132               this.translate0 = translate0;
133           }
134   
135           public void drawU(UGraphic ug) {
136               final StringBounder stringBounder = ug.getStringBounder();
137               final XPoint2D p1 = getP1(stringBounder);
138               final XPoint2D p2 = getP2(stringBounder, p1.getX());
139               final double x1 = p1.getX();
140               final double y1 = p1.getY();
141               final double x2 = p2.getX();
142               final double y2 = p2.getY();
143   
144               final UTranslate arrival = arrivalOnDiamond(stringBounder, p1.getX());
145               final UPolygon endDecoration;
146               if (arrival.getDx() < 0)
147                   endDecoration = Arrows.asToRight();
148               else if (arrival.getDx() > 0)
149                   endDecoration = Arrows.asToLeft();
150               else
151                   endDecoration = Arrows.asToDown();
152   
153               final Snake snake = Snake.create(skinParam(), arrowColor, endDecoration);
154               snake.addPoint(x1, y1);
155               snake.addPoint(x1, y2);
156               snake.addPoint(x2, y2);
157   
158               ug.draw(snake);
159           }
160   
161           private XPoint2D getP1(StringBounder stringBounder) {
162               return translate0.getTranslated(getFtile1().calculateDimension(stringBounder).getPointOut());
163           }
164   
165           private XPoint2D getP2(StringBounder stringBounder, double startX) {
166               final UTranslate arrival = arrivalOnDiamond(stringBounder, startX);
167               return arrival.getTranslated(getDiamondOut(stringBounder));
168           }
169   
170           public XPoint2D getDiamondOut(StringBounder stringBounder) {
171               return diamondTranslate.getTranslated(getFtile2().calculateDimension(stringBounder).getPointOut());
172           }
173   
174           public UTranslate arrivalOnDiamond(StringBounder stringBounder, double startX) {
175               final XPoint2D result = getDiamondOut(stringBounder);
176               final XDimension2D dim = getFtile2().calculateDimension(stringBounder);
177               final double a = result.getX() - dim.getWidth() / 2;
178               final double b = result.getX() + dim.getWidth() / 2;
179   
180               final UTranslate arrival;
181               if (startX < a)
182                   arrival = new UTranslate(-dim.getWidth() / 2, -dim.getHeight() / 2);
183               else if (startX > b)
184                   arrival = new UTranslate(dim.getWidth() / 2, -dim.getHeight() / 2);
185               else
186                   arrival = new UTranslate(0, -dim.getHeight());
187   
188               return arrival;
189           }

193       class ConnectionIn extends AbstractConnection implements ConnectionTranslatable {
194   
195           private final double x;
196           private final Rainbow arrowColor;
197           private final Display label;
198   
199           public ConnectionIn(Ftile ftile1, Ftile ftile2, double x, Rainbow arrowColor) {
200               super(ftile1, ftile2);
201               label = ftile2.getInLinkRendering().getDisplay();
202               this.x = x;
203               this.arrowColor = arrowColor;
204           }
205   
206           public void drawU(UGraphic ug) {
207               ug = ug.apply(UTranslate.dx(x));
208               final FtileGeometry geo = getFtile2().calculateDimension(getStringBounder());
209               Snake snake = Snake.create(skinParam(), arrowColor, Arrows.asToDown());
210               if (Display.isNull(label) == false)
211                   snake = snake.withLabel(getTextBlock(label), arrowHorizontalAlignment());
212   
213               snake.addPoint(geo.getLeft(), 0);
214               snake.addPoint(geo.getLeft(), geo.getInY());
215               ug.draw(snake);
216           }
217   
218           @Override
219           public void drawTranslate(UGraphic ug, UTranslate translate1, UTranslate translate2) {
220               ug = ug.apply(UTranslate.dx(x));
221               final FtileGeometry geo = getFtile2().calculateDimension(getStringBounder());
222               final XPoint2D p1 = new XPoint2D(geo.getLeft(), 0);
223               final XPoint2D p2 = new XPoint2D(geo.getLeft(), geo.getInY());
224   
225               Snake snake = Snake.create(skinParam(), arrowColor, Arrows.asToDown());
226               if (Display.isNull(label) == false)
227                   snake = snake.withLabel(getTextBlock(label), arrowHorizontalAlignment());
228   
229               final XPoint2D mp1a = translate1.getTranslated(p1);
230               final XPoint2D mp2b = translate2.getTranslated(p2);
231               final double middle = mp1a.getY() + 4;
232               snake.addPoint(mp1a);
233               snake.addPoint(mp1a.getX(), middle);
234               snake.addPoint(mp2b.getX(), middle);
235               snake.addPoint(mp2b);
236               ug.draw(snake);
237           }
238       }
