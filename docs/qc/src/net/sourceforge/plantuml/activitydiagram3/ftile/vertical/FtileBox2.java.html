146       private FtileBox2(ISkinParam skinParam, Display label, Swimlane swimlane, BoxStyle boxStyle, Style style,
147               Style styleArrow) {
148           super(skinParam);
149           Colors specBack = null;
150           if (skinParam instanceof SkinParamColors)
151               specBack = ((SkinParamColors) skinParam).getColors();
152           style = style.eventuallyOverride(specBack);
153   
154           this.style = style;
155           this.boxStyle = boxStyle;
156           this.swimlane = swimlane;
157   
158           this.inRendering = LinkRendering
159                   .create(Rainbow.build(styleArrow, getIHtmlColorSet()));
160           this.borderColor = style.value(PName.LineColor).asColor(getIHtmlColorSet());
161           this.backColor = style.value(PName.BackGroundColor).asColor(getIHtmlColorSet());
162   
163           final FontConfiguration fc = style.getFontConfiguration(getIHtmlColorSet());
164   
165           this.horizontalAlignment = style.getHorizontalAlignment();
166           this.padding = style.getPadding();
167           this.margin = style.getMargin();
168           this.roundCorner = style.value(PName.RoundCorner).asDouble();
169           this.shadowing = style.value(PName.Shadowing).asDouble();
170   
171           final LineBreakStrategy wrapWidth = style.wrapWidth();
172   
173           this.minimumWidth = style.value(PName.MinimumWidth).asDouble();
174   
175           final Sheet sheet = Parser
176                   .build(fc, skinParam.getDefaultTextAlignment(horizontalAlignment), skinParam, CreoleMode.FULL)
177                   .createSheet(label);
178           this.tb = new SheetBlock2(new SheetBlock1(sheet, wrapWidth, skinParam.getPadding()), new MyStencil(),
179                   new UStroke(1));
180           this.print = label.toString();
181   
182       }

191       public void drawU(UGraphic ug) {
192           final XDimension2D dimRaw = getDimRaw(ug.getStringBounder());
193           final double widthTotal = dimRaw.getWidth();
194           final double heightTotal = dimRaw.getHeight();
195           final UDrawable shape = boxStyle.getUDrawable(widthTotal, heightTotal, shadowing, roundCorner);
196   
197           final XDimension2D dimTotal = calculateDimension(ug.getStringBounder());
198           ug.draw(new UEmpty(dimTotal));
199   
200           final UStroke thickness = style.getStroke();
201   
202           if (borderColor == null)
203               ug = ug.apply(HColors.none());
204           else
205               ug = ug.apply(borderColor);
206   
207           if (backColor == null)
208               ug = ug.apply(HColors.none().bg());
209           else
210               ug = ug.apply(backColor.bg());
211   
212           ug = ug.apply(thickness);
213           ug = ug.apply(new UTranslate(margin.getLeft(), margin.getTop()));
214           shape.drawU(ug);
215   
216           if (horizontalAlignment == HorizontalAlignment.LEFT) {
217               tb.drawU(ug.apply(new UTranslate(padding.getLeft(), padding.getTop())));
218           } else if (horizontalAlignment == HorizontalAlignment.RIGHT) {
219               final XDimension2D dimTb = tb.calculateDimension(ug.getStringBounder());
220               tb.drawU(ug.apply(
221                       new UTranslate(dimRaw.getWidth() - dimTb.getWidth() - padding.getRight(), padding.getBottom())));
222           } else if (horizontalAlignment == HorizontalAlignment.CENTER) {
223               final XDimension2D dimTb = tb.calculateDimension(ug.getStringBounder());
224               tb.drawU(ug.apply(new UTranslate((dimRaw.getWidth() - dimTb.getWidth()) / 2, padding.getBottom())));
225           }
226       }

228       @Override
229       protected FtileGeometry calculateDimensionFtile(StringBounder stringBounder) {
230           XDimension2D dimRaw = getDimRaw(stringBounder);
231           return new FtileGeometry(dimRaw.getWidth() + margin.getLeft() + margin.getRight(),
232                   dimRaw.getHeight() + margin.getTop() + margin.getBottom(), margin.getLeft() + dimRaw.getWidth() / 2,
233                   margin.getTop(), margin.getTop() + dimRaw.getHeight());
234       }
235   
236       private XDimension2D getDimRaw(StringBounder stringBounder) {
237           XDimension2D dimRaw = tb.calculateDimension(stringBounder);
238           dimRaw = XDimension2D.delta(dimRaw, padding.getLeft() + padding.getRight() + boxStyle.getShield(),
239                   padding.getBottom() + padding.getTop());
240           dimRaw = XDimension2D.atLeast(dimRaw, minimumWidth, 0);
241           return dimRaw;
242       }
