212       private void drawRainbow(UGraphic ug) {
213           List<HtmlColorAndStyle> colors = color.getColors();
214           final int colorArrowSeparationSpace = color.getColorArrowSeparationSpace();
215           final double move = 2 + colorArrowSeparationSpace;
216           final WormMutation mutation = WormMutation.create(worm, move);
217           if (mutation.isDxNegative()) {
218               colors = new ArrayList<>(colors);
219               Collections.reverse(colors);
220           }
221           final double globalMove = -1.0 * (colors.size() - 1) / 2.0;
222           Worm current = worm.moveFirstPoint(mutation.getFirst().multiplyBy(globalMove));
223           if (mutation.size() > 2)
224               current = current.moveLastPoint(mutation.getLast().multiplyBy(globalMove));
225   
226           for (int i = 0; i < colors.size(); i++) {
227               double stroke = 1.5;
228               if (colorArrowSeparationSpace == 0)
229                   stroke = i == colors.size() - 1 ? 2.0 : 3.0;
230   
231               current.drawInternalOneColor(startDecoration, ug, colors.get(i), stroke, emphasizeDirection, endDecoration);
232               current = mutation.mute(current);
233           }
234           final UTranslate textTranslate = mutation.getTextTranslate(colors.size());
235           drawInternalLabel(ug.apply(textTranslate));
236       }

238       private void drawInternalLabel(UGraphic ug) {
239           for (Text text : texts)
240               if (text.hasText(ug.getStringBounder())) {
241                   final XPoint2D position = getTextBlockPosition(ug.getStringBounder(), text);
242                   text.textBlock.drawU(ug.apply(new UTranslate(position)));
243               }

246       public double getMaxX(StringBounder stringBounder) {
247           double result = -Double.MAX_VALUE;
248           for (XPoint2D pt : worm)
249               result = Math.max(result, pt.getX());
250   
251           for (Text text : texts) {
252               final XPoint2D position = getTextBlockPosition(stringBounder, text);
253               final XDimension2D dim = text.textBlock.calculateDimension(stringBounder);
254               result = Math.max(result, position.getX() + dim.getWidth());
255           }
256           return result;
257       }

259       private XPoint2D getTextBlockPosition(StringBounder stringBounder, Text text) {
260           final XPoint2D pt1 = worm.get(0);
261           final XPoint2D pt2 = worm.get(1);
262           final XDimension2D dim = text.textBlock.calculateDimension(stringBounder);
263           double x = Math.max(pt1.getX(), pt2.getX()) + 4;
264           final boolean zigzag = worm.getDirectionsCode().startsWith("DLD") || worm.getDirectionsCode().startsWith("DRD");
265           double y = (pt1.getY() + pt2.getY()) / 2 - dim.getHeight() / 2;
266           if (text.verticalAlignment == VerticalAlignment.BOTTOM) {
267               x = worm.getLast().getX();
268               throw new AssertionError();
269           } else if (text.verticalAlignment == VerticalAlignment.CENTER) {
270               x = worm.getMinX();
271               y = (worm.getFirst().getY() + worm.getLast().getY() - 10) / 2 - dim.getHeight() / 2;
272           } else if (text.horizontalAlignment == HorizontalAlignment.CENTER && zigzag) {
273               final XPoint2D pt3 = worm.get(2);
274               x = (pt2.getX() + pt3.getX()) / 2 - dim.getWidth() / 2;
275           } else if (text.horizontalAlignment == HorizontalAlignment.RIGHT && zigzag) {
276               x = Math.max(pt1.getX(), pt2.getX()) - dim.getWidth() - 4;
277           } else if (worm.getDirectionsCode().equals("RD")) {
278               x = Math.max(pt1.getX(), pt2.getX());
279               y = (pt1.getY() + worm.get(2).getY()) / 2 - dim.getHeight() / 2;
280           } else if (worm.getDirectionsCode().equals("LD")) {
281               x = Math.min(pt1.getX(), pt2.getX());
282               y = (pt1.getY() + worm.get(2).getY()) / 2 - dim.getHeight() / 2;
283           }
284           return new XPoint2D(x, y);
285       }

313       public Snake merge(Snake other, StringBounder stringBounder) {
314           final MergeStrategy strategy = this.mergeable.max(other.mergeable);
315           if (strategy == MergeStrategy.NONE)
316               return null;
317   
318           for (Text text : other.texts)
319               if (text.hasText(stringBounder))
320                   return null;
321   
322           if (same(this.getLast(), other.getFirst())) {
323               final UPolygon oneOf = other.endDecoration == null ? endDecoration : other.endDecoration;
324               if (this.startDecoration != null || other.startDecoration != null)
325                   throw new UnsupportedOperationException("Not yet coded: to be done");
326   
327               final ArrayList<Text> mergeTexts = new ArrayList<Text>(this.texts);
328               mergeTexts.addAll(other.texts);
329               final Snake result = new Snake(skinParam, null, color, oneOf, this.worm.merge(other.worm, strategy),
330                       strategy, emphasizeDirection == null ? other.emphasizeDirection : emphasizeDirection, mergeTexts);
331               return result;
332           }
333           if (same(this.getFirst(), other.getLast()))
334               return other.merge(this, stringBounder);
335   
336           return null;
337       }

339       public boolean touches(Snake other) {
340           if (other.mergeable != MergeStrategy.FULL)
341               return false;
342   
343           if (other.worm.isPureHorizontal())
344               return false;
345   
346           return same(this.getLast(), other.getFirst());
347       }
