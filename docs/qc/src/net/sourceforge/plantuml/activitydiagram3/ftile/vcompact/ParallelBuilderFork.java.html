79       protected List<Ftile> getFoo2(List<Ftile> all) {
80           final double maxHeight = computeMaxHeight(all);
81           final double ymargin1 = getSuppSpace1(all, getStringBounder());
82           final double ymargin2 = getSuppSpace2(all, getStringBounder());
83           final List<Ftile> result = new ArrayList<>();
84           for (Ftile ftile : all) {
85               final Ftile newFtile = computeNewFtile(ftile, maxHeight, ymargin1, ymargin2);
86               result.add(newFtile);
87           }
88           return result;
89       }

91       private Ftile computeNewFtile(Ftile ftile, double maxHeight, double ymargin1, double ymargin2) {
92           final double spaceArroundBlackBar = 20;
93           final double xMargin = 14;
94           Ftile tmp;
95           tmp = FtileUtils.addHorizontalMargin(ftile, xMargin, xMargin + getSuppForIncomingArrow(ftile));
96           tmp = new FtileHeightFixedCentered(tmp, maxHeight + 2 * spaceArroundBlackBar);
97           tmp = new FtileHeightFixedMarged(ymargin1, tmp, ymargin2);
98           return tmp;
99       }

101       private double getSuppForIncomingArrow(Ftile ftile) {
102           final double x1 = getXSuppForDisplay(ftile, ftile.getInLinkRendering().getDisplay());
103           final double x2 = getXSuppForDisplay(ftile, ftile.getOutLinkRendering().getDisplay());
104           return Math.max(x1, x2);
105       }

107       private double getXSuppForDisplay(Ftile ftile, Display label) {
108           final TextBlock text = getTextBlock(label);
109           if (text == null)
110               return 0;
111   
112           final double textWidth = text.calculateDimension(getStringBounder()).getWidth();
113           final FtileGeometry ftileDim = ftile.calculateDimension(getStringBounder());
114           final double pos2 = ftileDim.getLeft() + textWidth;
115           if (pos2 > ftileDim.getWidth())
116               return pos2 - ftileDim.getWidth();
117   
118           return 0;
119       }

128           Ftile result = middle;

130           final Swimlane swimlaneBlack = in;
131           final Style style = getStyleSignature().getMergedStyle(skinParam().getCurrentStyleBuilder());
132           final Ftile black = new FtileBlackBlock(skinParam(), swimlaneBlack);
133           double x = 0;
134           for (Ftile tmp : list99) {
135               final XDimension2D dim = tmp.calculateDimension(getStringBounder());
136               final Rainbow def = Rainbow.build(style, skinParam().getIHtmlColorSet());

138               conns.add(new ConnectionIn(black, tmp, x, rainbow));
139               x += dim.getWidth();
140           }
141   
142           result = FtileUtils.addConnection(result, conns);
143           ((FtileBlackBlock) black).setBlackBlockDimension(result.calculateDimension(getStringBounder()).getWidth(),
144                   barHeight);
145   
146           return new FtileAssemblySimple(black, result);
147       }

149       private double getSuppSpace1(List<Ftile> all, StringBounder stringBounder) {
150           double result = 0;
151           for (Ftile child : all) {
152               final TextBlock text = getTextBlock(child.getInLinkRendering().getDisplay());
153               if (text == null)
154                   continue;
155   
156               final XDimension2D dim = text.calculateDimension(stringBounder);
157               result = Math.max(result, dim.getHeight());
158   
159           }
160           return result;
161       }

163       private double getSuppSpace2(List<Ftile> all, StringBounder stringBounder) {
164           double result = 0;
165           for (Ftile child : all) {
166               final TextBlock text = getTextBlock(child.getOutLinkRendering().getDisplay());
167               if (text == null)
168                   continue;

171               result = Math.max(result, dim.getHeight());
172           }
173           return result;
174       }

181       protected Ftile doStep2(Ftile middle, Ftile result) {
182           final Swimlane swimlaneBlack = out;
183           final Ftile out = new FtileBlackBlock(skinParam(), swimlaneBlack);
184           ((FtileBlackBlock) out).setBlackBlockDimension(result.calculateDimension(getStringBounder()).getWidth(),
185                   barHeight);
186           if (label != null)
187               ((FtileBlackBlock) out).setLabel(getTextBlock(Display.getWithNewlines(label)));
188   
189           result = new FtileAssemblySimple(result, out);
190           final List<Connection> conns = new ArrayList<>();
191           final Style style = getStyleSignatureArrow().getMergedStyle(skinParam().getCurrentStyleBuilder());
192           double x = 0;
193           for (Ftile tmp : list99) {
194               final XDimension2D dim = tmp.calculateDimension(getStringBounder());
195               final Rainbow def = Rainbow.build(style, skinParam().getIHtmlColorSet());
196               final Rainbow rainbow = tmp.getOutLinkRendering().getRainbow(def);
197               if (tmp.calculateDimension(getStringBounder()).hasPointOut())
198                   conns.add(new ConnectionOut(tmp, out, x, rainbow, getJustBeforeBar2(middle, getStringBounder())));
199   
200               x += dim.getWidth();
201           }
202           result = FtileUtils.addConnection(result, conns);
203           return result;
204       }

210       class ConnectionIn extends AbstractConnection implements ConnectionTranslatable {

212           private final double x;
213           private final Rainbow arrowColor;
214           private final Display label;

216           public ConnectionIn(Ftile ftile1, Ftile ftile2, double x, Rainbow arrowColor) {
217               super(ftile1, ftile2);
218               this.label = ftile2.getInLinkRendering().getDisplay();
219               this.x = x;
220               this.arrowColor = arrowColor;
221           }

223           public void drawU(UGraphic ug) {
224               ug = ug.apply(UTranslate.dx(x));
225               final FtileGeometry geo2 = getFtile2().calculateDimension(getStringBounder());
226               Snake snake = Snake.create(skinParam(), arrowColor, Arrows.asToDown());
227               if (Display.isNull(label) == false)
228                   snake = snake.withLabel(getTextBlock(label), arrowHorizontalAlignment());
229   
230               final XPoint2D p1 = new XPoint2D(geo2.getLeft(), 0);
231               final XPoint2D p2 = new XPoint2D(geo2.getLeft(), geo2.getInY());
232               snake.addPoint(p1);
233               snake.addPoint(p2);
234               ug.draw(snake);
235           }

237           @Override
238           public void drawTranslate(UGraphic ug, UTranslate translate1, UTranslate translate2) {
239               ug = ug.apply(UTranslate.dx(x));
240               final FtileGeometry geo2 = getFtile2().calculateDimension(getStringBounder());
241               final XPoint2D p1 = new XPoint2D(geo2.getLeft(), 0);
242               final XPoint2D p2 = new XPoint2D(geo2.getLeft(), geo2.getInY());
243   
244               Snake snake = Snake.create(skinParam(), arrowColor, Arrows.asToDown()).ignoreForCompression();
245               if (Display.isNull(label) == false)
246                   snake = snake.withLabel(getTextBlock(label), arrowHorizontalAlignment());
247   
248               final XPoint2D mp1a = translate1.getTranslated(p1);
249               final XPoint2D mp2b = translate2.getTranslated(p2);
250               final double middle = mp1a.getY() + 4;
251               snake.addPoint(mp1a);
252               snake.addPoint(mp1a.getX(), middle);
253               snake.addPoint(mp2b.getX(), middle);
254               snake.addPoint(mp2b);
255               ug.draw(snake);
256           }
257       }
