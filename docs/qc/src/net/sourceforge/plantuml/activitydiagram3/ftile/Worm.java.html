88       public void drawInternalOneColor(UPolygon startDecoration, UGraphic ug, HtmlColorAndStyle colorAndStyle,
89               double strokeValue, Direction emphasizeDirection, UPolygon endDecoration) {
90           final HColor arrowColor = Objects.requireNonNull(colorAndStyle.getArrowColor());
91           final LinkStyle linkStyle = colorAndStyle.getStyle();
92           if (linkStyle.isInvisible())
93               return;
94   
95           ug = ug.apply(arrowColor);
96           ug = ug.apply(arrowColor.bg());
97           if (linkStyle.isNormal())
98               ug = ug.apply(style.getStroke());
99           else
100               ug = ug.apply(linkStyle.goThickness(strokeValue).getStroke3());
101   
102           boolean drawn = false;
103           for (int i = 0; i < points.size() - 1; i++) {
104               final XPoint2D p1 = points.get(i);
105               final XPoint2D p2 = points.get(i + 1);
106               final XLine2D line = new XLine2D(p1, p2);
107               if (drawn == false && emphasizeDirection != null && Direction.fromVector(p1, p2) == emphasizeDirection) {
108                   drawLine(ug, line, emphasizeDirection);
109                   drawn = true;
110               } else {
111                   drawLine(ug, line, null);
112               }
113           }
114   
115           final HColor arrowHeadColor = colorAndStyle.getArrowHeadColor();
116           if (arrowHeadColor == null)
117               throw new IllegalStateException();
118   
119   //        if (arrowHeadColor == null || arrowHeadColor.equals(HColorUtils.transparent())) {
120   //            return;
121   //        }
122           ug = ug.apply(arrowHeadColor);
123           ug = ug.apply(arrowHeadColor.bg());
124   
125           if (startDecoration != null) {
126               ug = ug.apply(new UStroke(1.5));
127               final XPoint2D start = points.get(0);
128               if (ignoreForCompression)
129                   startDecoration.setCompressionMode(CompressionMode.ON_X);
130   
131               ug.apply(new UTranslate(start)).apply(new UStroke()).draw(startDecoration);
132           }
133           if (endDecoration != null) {
134               ug = ug.apply(new UStroke(1.5));
135               final XPoint2D end = points.get(points.size() - 1);
136               if (ignoreForCompression)
137                   endDecoration.setCompressionMode(CompressionMode.ON_X);
138   
139               ug.apply(new UTranslate(end)).apply(new UStroke()).draw(endDecoration);
140           }
141       }

163       public Worm moveFirstPoint(UTranslate move) {
164           final double dx = move.getDx();
165           final double dy = move.getDy();
166           if (dx != 0 && dy != 0)
167               throw new IllegalArgumentException("move=" + move);
168   
169           final Worm result = new Worm(style);
170           double x0 = this.points.get(0).getX();
171           double y0 = this.points.get(0).getY();
172           double x1 = this.points.get(1).getX();
173           double y1 = this.points.get(1).getY();
174   
175           if (dx != 0 && x0 == x1)
176               x1 += dx;
177   
178           if (dy != 0 && y0 == y1)
179               y1 += dy;
180   
181           x0 += dx;
182           y0 += dy;
183   
184           result.addPoint(x0, y0);
185           result.addPoint(x1, y1);
186           for (int i = 2; i < this.points.size(); i++)
187               result.addPoint(this.points.get(i));
188   
189           return result;
190       }

344       private void mergeMe(MergeStrategy merge) {
345           boolean change = false;
346           do {
347               change = false;
348               change = change || removeNullVector();
349               change = change || removeRedondantDirection();
350               change = change || removePattern1();
351               change = change || removePattern2();
352               change = change || removePattern3();
353               change = change || removePattern4();
354               change = change || removePattern5();
355               change = change || removePattern6();
356               change = change || removePattern7();
357               if (merge == MergeStrategy.FULL)
358                   change = change || removePattern8();
359   
360           } while (change);
361       }
