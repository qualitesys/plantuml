48   // Created from Luc Trudeau original work
49   public enum BoxStyle {
50       PLAIN('\0', 0) {
51           @Override
52           protected Shadowable getShape(double width, double height, double roundCorner) {
53               return new URectangle(width, height).rounded(roundCorner);
54           }
55       },
56       SDL_INPUT('<', 10) {
57           @Override
58           protected Shadowable getShape(double width, double height, double roundCorner) {
59               final UPolygon result = new UPolygon();
60               result.addPoint(0, 0);
61               result.addPoint(width + DELTA_INPUT_OUTPUT, 0);
62               result.addPoint(width, height / 2);
63               result.addPoint(width + DELTA_INPUT_OUTPUT, height);
64               result.addPoint(0, height);
65               return result;
66           }
67       },
68       SDL_OUTPUT('>', 10) {
69           @Override
70           protected Shadowable getShape(double width, double height, double roundCorner) {
71               final UPolygon result = new UPolygon();
72               result.addPoint(0.0, 0.0);
73               result.addPoint(width, 0.0);
74               result.addPoint(width + DELTA_INPUT_OUTPUT, height / 2);
75               result.addPoint(width, height);
76               result.addPoint(0.0, height);
77               return result;
78           }
79       },
80       SDL_PROCEDURE('|', 0) {
81           @Override
82           protected void drawInternal(UGraphic ug, double width, double height, double shadowing, double roundCorner) {
83               final URectangle rect = new URectangle(width, height);
84               rect.setDeltaShadow(shadowing);
85               ug.draw(rect);
86               final ULine vline = ULine.vline(height);
87               ug.apply(UTranslate.dx(PADDING)).draw(vline);
88               ug.apply(UTranslate.dx(width - PADDING)).draw(vline);
89           }
90       },
91       SDL_SAVE('\\', 0) {
92           @Override
93           protected Shadowable getShape(double width, double height, double roundCorner) {
94               final UPolygon result = new UPolygon();
95               result.addPoint(0.0, 0.0);
96               result.addPoint(width - DELTA_INPUT_OUTPUT, 0.0);
97               result.addPoint(width, height);
98               result.addPoint(DELTA_INPUT_OUTPUT, height);
99               return result;
100           }
101       },
102       SDL_ANTISAVE('/', 0) {
103           @Override
104           protected Shadowable getShape(double width, double height, double roundCorner) {
105               final UPolygon result = new UPolygon();
106               result.addPoint(DELTA_INPUT_OUTPUT, 0.0);
107               result.addPoint(width, 0.0);
108               result.addPoint(width - DELTA_INPUT_OUTPUT, height);
109               result.addPoint(0, height);
110               return result;
111           }
112       },
113       SDL_CONTINUOUS('}', 0) {
114           @Override
115           protected Shadowable getShape(double width, double height, double roundCorner) {
116               final UPath result = new UPath();
117               final double c1[] = { DELTA_CONTINUOUS, 0 };
118               final double c2[] = { 0, height / 2 };
119               final double c3[] = { DELTA_CONTINUOUS, height };
120   
121               result.add(c1, USegmentType.SEG_MOVETO);
122               result.add(c2, USegmentType.SEG_LINETO);
123               result.add(c3, USegmentType.SEG_LINETO);
124   
125               final double c4[] = { width - DELTA_CONTINUOUS, 0 };
126               final double c5[] = { width, height / 2 };
127               final double c6[] = { width - DELTA_CONTINUOUS, height };
128   
129               result.add(c4, USegmentType.SEG_MOVETO);
130               result.add(c5, USegmentType.SEG_LINETO);
131               result.add(c6, USegmentType.SEG_LINETO);
132               return result;
133           }
134       },
135       SDL_TASK(']', 0) {
136           @Override
137           protected Shadowable getShape(double width, double height, double roundCorner) {
138               return new URectangle(width, height);
139           }
140       };
141   
142       private final char style;
143       private final double shield;
144   
145       private static int DELTA_INPUT_OUTPUT = 10;
146       private static double DELTA_CONTINUOUS = 5.0;
147       private static int PADDING = 5;
148   
149       private BoxStyle(char style, double shield) {
150           this.style = style;
151           this.shield = shield;
152       }
153   
154       public static BoxStyle fromChar(char style) {
155           for (BoxStyle bs : BoxStyle.values()) {
156               if (bs.style == style) {
157                   return bs;
158               }
159           }
160           return PLAIN;
161       }
162   
163       public final UDrawable getUDrawable(final double width, final double height, final double shadowing,
164               final double roundCorner) {
165           return new UDrawable() {
166               public void drawU(UGraphic ug) {
167                   drawInternal(ug, width - getShield(), height, shadowing, roundCorner);
168               }
169           };
170       }
171   
172       protected Shadowable getShape(double width, double height, double roundCorner) {
173           return null;
174       }
175   
176       protected void drawInternal(UGraphic ug, double width, double height, double shadowing, double roundCorner) {
177           final Shadowable s = getShape(width, height, roundCorner);
178           s.setDeltaShadow(shadowing);
179           ug.draw(s);
180   
181       }
182   
183       public final double getShield() {
184           return shield;
185       }

187   }
