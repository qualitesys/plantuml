36   package net.sourceforge.plantuml.activitydiagram3.ftile;
37   
38   import net.sourceforge.plantuml.awt.geom.XDimension2D;
39   import net.sourceforge.plantuml.awt.geom.XPoint2D;
40   import net.sourceforge.plantuml.ugraphic.UTranslate;
41   
42   public class FtileGeometry extends XDimension2D {
43   
44       private final double width;
45       private final double height;
46       private final double left;
47       private final double inY;
48       private final double outY;
49   
50       public XPoint2D getPointA() {
51           return new XPoint2D(left, inY);
52       }
53   
54       public XPoint2D getPointIn() {
55           return new XPoint2D(left, inY);
56       }
57   
58       public XPoint2D getPointB() {
59           if (outY == Double.MIN_NORMAL) {
60               throw new UnsupportedOperationException();
61           }
62           return new XPoint2D(width, (inY + outY) / 2);
63       }
64   
65       public XPoint2D getPointC() {
66           if (outY == Double.MIN_NORMAL) {
67               throw new UnsupportedOperationException();
68           }
69           return new XPoint2D(left, outY);
70       }
71   
72       public XPoint2D getPointD() {
73           if (outY == Double.MIN_NORMAL) {
74               throw new UnsupportedOperationException();
75           }
76           return new XPoint2D(0, (inY + outY) / 2);
77       }
78   
79       public XPoint2D getPointOut() {
80           if (outY == Double.MIN_NORMAL) {
81               throw new UnsupportedOperationException();
82           }
83           return new XPoint2D(left, outY);
84       }
85   
86       public FtileGeometry(XDimension2D dim, double left, double inY) {
87           this(dim.getWidth(), dim.getHeight(), left, inY);
88       }
89   
90       public FtileGeometry(double width, double height, double left, double inY) {
91           this(width, height, left, inY, Double.MIN_NORMAL);
92       }
93   
94       @Override
95       public String toString() {
96           return "[" + width + "x" + height + " left=" + left + "]";
97       }
98   
99       public FtileGeometry(double width, double height, double left, double inY, double outY) {
100           this.left = left;
101           this.inY = inY;
102           this.outY = outY;
103           this.width = width;
104           this.height = height;
105       }
106   
107       public FtileGeometry incHeight(double northHeight) {
108           return new FtileGeometry(width, height + northHeight, left, inY, outY);
109       }
110   
111       public FtileGeometry addTop(double northHeight) {
112           if (hasPointOut())
113               return new FtileGeometry(width, height + northHeight, left, inY + northHeight, outY + northHeight);
114           return new FtileGeometry(width, height + northHeight, left, inY + northHeight, Double.MIN_NORMAL);
115       }
116   
117       public FtileGeometry addBottom(double southHeight) {
118           if (hasPointOut())
119               return new FtileGeometry(width, height + southHeight, left, inY, outY);
120           return new FtileGeometry(width, height + southHeight, left, inY, Double.MIN_NORMAL);
121       }
122   
123       public FtileGeometry incRight(double missing) {
124           return new FtileGeometry(width + missing, height, left, inY, outY);
125       }
126   
127       public FtileGeometry incLeft(double missing) {
128           return new FtileGeometry(width + missing, height, left + missing, inY, outY);
129       }
130   
131       public FtileGeometry incVertically(double missing1, double missing2) {
132           return new FtileGeometry(width, height + missing1 + missing2, left, inY + missing1,
133                   hasPointOut() ? outY + missing1 : outY);
134       }
135   
136       public FtileGeometry(XDimension2D dim, double left, double inY, double outY) {
137           this(dim.getWidth(), dim.getHeight(), left, inY, outY);
138       }
139   
140       public boolean hasPointOut() {
141           return outY != Double.MIN_NORMAL;
142       }
143   
144       public FtileGeometry withoutPointOut() {
145           return new FtileGeometry(width, height, left, inY);
146       }
147   
148       public FtileGeometry translate(UTranslate translate) {
149           final double dx = translate.getDx();
150           final double dy = translate.getDy();
151           if (this.outY == Double.MIN_NORMAL) {
152               return new FtileGeometry(width, height, left + dx, inY + dy);
153           }
154           return new FtileGeometry(width, height, left + dx, inY + dy, outY + dy);
155       }
156   
157       public final double getInY() {
158           return inY;
159       }
160   
161       public final double getLeft() {
162           return left;
163       }
164   
165       public final double getRight() {
166           return width - left;
167       }
168   
169       public double getOutY() {
170           return outY;
171       }
172   
173       public final double getWidth() {
174           return width;
175       }
176   
177       public final double getHeight() {
178           return height;
179       }
180   
181       public FtileGeometry addDim(double deltaWidth, double deltaHeight) {
182           return new FtileGeometry(width + deltaWidth, height + deltaHeight, left, inY, outY + deltaHeight);
183       }
184   
185       public FtileGeometry addMarginX(double marginx) {
186           return new FtileGeometry(width + 2 * marginx, height, left + marginx, inY, outY);
187       }
188   
189       public FtileGeometry addMarginX(double margin1, double margin2) {
190           return new FtileGeometry(width + margin1 + margin2, height, left + margin1, inY, outY);
191       }
192   
193       public FtileGeometry fixedHeight(double fixedHeight) {
194           return new FtileGeometry(width, fixedHeight, left, inY, outY);
195       }
196   
197       public FtileGeometry appendBottom(FtileGeometry other) {
198           return new FtileGeometryMerger(this, other).getResult();
199       }
200   
201       public FtileGeometry ensureHeight(double newHeight) {
202           if (this.height > newHeight) {
203               return this;
204           }
205           return fixedHeight(newHeight);
206       }
207   
208       private FtileGeometry ensureRightStrange(double newRight) {
209           final double right = this.width - this.left;
210           if (right > newRight) {
211               return this;
212           }
213           // return addMarginX(0, newRight - right);
214           return addMarginX(0, newRight);
215       }

217   }
