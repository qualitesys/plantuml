62   public class InstructionRepeat extends AbstractInstruction implements Instruction {
63   
64       private final InstructionList repeatList;
65       private final Instruction parent;
66       private final LinkRendering nextLinkRenderer;
67       private final Swimlane swimlane;
68       private final Swimlanes swimlanes;
69       private Swimlane swimlaneOut;
70       private Swimlane swimlaneBackward;
71       private BoxStyle boxStyle;
72       private boolean killed = false;
73       private final BoxStyle boxStyleIn;
74   
75       private Display backward = Display.NULL;
76   
77       private LinkRendering incoming1 = LinkRendering.none();
78       private LinkRendering incoming2 = LinkRendering.none();
79       private List<PositionedNote> backwardNotes = new ArrayList<>();
80       private Display test = Display.NULL;
81       private Display yes = Display.NULL;
82       private Display out = Display.NULL;
83       private final Display startLabel;
84       private boolean testCalled = false;
85       private LinkRendering endRepeatLinkRendering = LinkRendering.none();
86   
87       private final Colors colors;
88   
89       public boolean containsBreak() {
90           return repeatList.containsBreak();
91       }
92   
93       public InstructionRepeat(Swimlanes swimlanes, Instruction parent, LinkRendering nextLinkRenderer, HColor color,
94               Display startLabel, BoxStyle boxStyleIn, Colors colors) {
95           this.swimlanes = swimlanes;
96           this.swimlane = swimlanes.getCurrentSwimlane();
97           this.repeatList = new InstructionList(this.swimlane);
98           this.boxStyleIn = boxStyleIn;
99           this.startLabel = startLabel;
100           this.parent = parent;
101           this.nextLinkRenderer = Objects.requireNonNull(nextLinkRenderer);
102           this.colors = colors;
103       }
104   
105       private boolean isLastOfTheParent() {
106           if (parent instanceof InstructionList)
107               return ((InstructionList) parent).getLast() == this;
108   
109           return false;
110       }

112       public void setBackward(Display label, Swimlane swimlaneBackward, BoxStyle boxStyle, LinkRendering incoming1,
113               LinkRendering incoming2) {
114           this.backward = label;
115           this.swimlaneBackward = swimlaneBackward;
116           this.boxStyle = boxStyle;
117           this.incoming1 = incoming1;
118           this.incoming2 = incoming2;
119       }
120   
121       public boolean hasBackward() {
122           return this.backward != Display.NULL;
123       }
124   
125       @Override
126       public CommandExecutionResult add(Instruction ins) {
127           return repeatList.add(ins);
128       }

130       @Override
131       public Gtile createGtile(ISkinParam skinParam, StringBounder stringBounder) {
132   
133           final Gtile tile = repeatList.createGtile(skinParam, stringBounder);
134           final Gtile backward = getGtileBackward(skinParam, stringBounder);
135   
136           return new GtileRepeat(swimlane, tile, null, test, backward);
137       }
138   
139       private Gtile getGtileBackward(ISkinParam skinParam, StringBounder stringBounder) {
140           if (Display.isNull(backward))
141               return null;
142   
143           GtileBox result = GtileBox.create(stringBounder, skinParam, backward, getSwimlaneIn(), boxStyle, null);
144   //        if (backwardNotes.size() > 0) {
145   //            result = factory.addNote(result, swimlaneOut, backwardNotes);
146   //        }
147           return result;
148       }
149   
150       public Ftile createFtile(FtileFactory factory) {
151           final Ftile back = getFtileBackward(factory);
152           final Ftile decorateOut = factory.decorateOut(repeatList.createFtile(factory), endRepeatLinkRendering);
153           if (this.testCalled == false && incoming1.isNone())
154               incoming1 = swimlanes.nextLinkRenderer();
155           final Ftile result = factory.repeat(boxStyleIn, swimlane, swimlaneOut, startLabel, decorateOut, test, yes, out,
156                   colors, back, isLastOfTheParent(), incoming1, incoming2);
157           if (killed)
158               return new FtileKilled(result);
159   
160           return result;
161       }
162   
163       private Ftile getFtileBackward(FtileFactory factory) {
164           if (Display.isNull(backward))
165               return null;
166   
167           Ftile result = factory.activity(backward, swimlaneBackward, boxStyle, Colors.empty(), null);
168           if (backwardNotes.size() > 0)
169               result = factory.addNote(result, swimlaneBackward, backwardNotes, VerticalAlignment.CENTER);
170   
171           return result;
172       }
173   
174       public Instruction getParent() {
175           return parent;
176       }
177   
178       public void setTest(Display test, Display yes, Display out, LinkRendering endRepeatLinkRendering,
179               LinkRendering back, Swimlane swimlaneOut) {
180           this.swimlaneOut = swimlaneOut;
181           this.test = Objects.requireNonNull(test);
182           this.yes = Objects.requireNonNull(yes);
183           this.out = Objects.requireNonNull(out);
184           this.endRepeatLinkRendering = endRepeatLinkRendering;
185           if (back.isNone() == false)
186               this.incoming1 = back;
187           this.testCalled = true;
188       }
189   
190       final public boolean kill() {
191           if (testCalled) {
192               this.killed = true;
193               return true;
194           }
195           return repeatList.kill();
196       }

203       @Override
204       public boolean addNote(Display note, NotePosition position, NoteType type, Colors colors, Swimlane swimlaneNote) {
205           if (Display.isNull(backward))
206               return repeatList.addNote(note, position, type, colors, swimlaneNote);
207   
208           this.backwardNotes.add(new PositionedNote(note, position, type, swimlaneNote, colors));
209           return true;
210   
211       }
