175       private boolean manageSpecialStopEndAfterEndWhile(Instruction special) {
176           if (current() instanceof InstructionList == false)
177               return false;
178   
179           final InstructionList current = (InstructionList) current();
180           final Instruction last = current.getLast();
181           if (last instanceof InstructionWhile == false)
182               return false;
183   
184           final InstructionWhile instructionWhile = (InstructionWhile) last;
185           if (instructionWhile.containsBreak())
186               return false;
187   
188           instructionWhile.setSpecial(special);
189           return true;
190       }
