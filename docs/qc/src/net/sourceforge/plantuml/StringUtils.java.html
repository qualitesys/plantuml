288       public static String eventuallyRemoveStartingAndEndingDoubleQuote(String s, String format) {
289           if (s == null)
290               return null;
291   
292           if (format.contains("\"") && s.length() > 1 && isDoubleQuote(s.charAt(0))
293                   && isDoubleQuote(s.charAt(s.length() - 1)))
294               return s.substring(1, s.length() - 1);
295   
296           if (format.contains("(") && s.startsWith("(") && s.endsWith(")"))
297               return s.substring(1, s.length() - 1);
298   
299           if (format.contains("[") && s.startsWith("[") && s.endsWith("]"))
300               return s.substring(1, s.length() - 1);
301   
302           if (format.contains(":") && s.startsWith(":") && s.endsWith(":"))
303               return s.substring(1, s.length() - 1);
304   
305           return s;
306       }
