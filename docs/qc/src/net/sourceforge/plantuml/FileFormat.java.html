105       public String getFileSuffix() {
106           if (name().startsWith("XMI"))
107               return ".xmi";
108   
109           if (this == MJPEG)
110               return ".avi";
111   
112           if (this == LATEX_NO_PREAMBLE)
113               return ".latex";
114   
115           if (this == ANIMATED_GIF)
116               return ".gif";
117   
118           if (this == BRAILLE_PNG)
119               return ".braille.png";
120   
121           if (this == EPS_TEXT)
122               return EPS.getFileSuffix();
123   
124           return "." + StringUtils.goLowerCase(name());
125       }

182       static private XDimension2D getJavaDimension(UFont font, String text) {
183           final Font javaFont = font.getUnderlayingFont();
184           final FontMetrics fm = gg.getFontMetrics(javaFont);
185           final Rectangle2D rect = fm.getStringBounds(text, gg);
186           return new XDimension2D(rect.getWidth(), rect.getHeight());
187       }

196               protected XDimension2D calculateDimensionInternal(UFont font, String text) {
197                   final int nb = BrailleCharFactory.build(text).size();
198                   final double quanta = UGraphicBraille.QUANTA;
199                   final double height = 5 * quanta;

201                   return new XDimension2D(width, height);
202               }

211       private StringBounder getTikzStringBounder(final TikzFontDistortion tikzFontDistortion) {
212           return new StringBounderRaw() {
213               @Override
214               public String toString() {
215                   return "FileFormat::getTikzStringBounder";
216               }
217   
218               protected XDimension2D calculateDimensionInternal(UFont font, String text) {
219                   text = text.replace("\t", "    ");
220                   final XDimension2D w1 = getJavaDimension(font.goTikz(-1), text);
221                   final XDimension2D w2 = getJavaDimension(font.goTikz(0), text);
222                   final XDimension2D w3 = getJavaDimension(font.goTikz(1), text);
223                   final double factor = (w3.getWidth() - w1.getWidth()) / w2.getWidth();
224                   final double distortion = tikzFontDistortion.getDistortion();
225                   final double magnify = tikzFontDistortion.getMagnify();
226                   final double delta = (w2.getWidth() - w1.getWidth()) * factor * distortion;
227                   return w2.withWidth(Math.max(w1.getWidth(), magnify * w2.getWidth() - delta));
228               }
229           };
230       }

247       public String changeName(String fileName, int cpt) {
248           if (cpt == 0)
249               return changeName(fileName, getFileSuffix());
250   
251           return changeName(fileName,
252                   OptionFlags.getInstance().getFileSeparator() + String.format("%03d", cpt) + getFileSuffix());
253       }

255       private SFile computeFilename(SFile pngFile, int i) {
256           if (i == 0)
257               return pngFile;
258   
259           final SFile dir = pngFile.getParentFile();
260           return dir.file(computeFilenameInternal(pngFile.getName(), i));
261       }
262   
263       private String changeName(String fileName, String replacement) {
264           String result = fileName.replaceAll("\\.\\w+$", replacement);
265           if (result.equals(fileName))
266               result = fileName + replacement;
267   
268           return result;
269       }
270   
271       private String computeFilenameInternal(String name, int i) {
272           if (i == 0)
273               return name;
274   
275           return name.replaceAll("\\" + getFileSuffix() + "$",
276                   OptionFlags.getInstance().getFileSeparator() + String.format("%03d", i) + getFileSuffix());
277       }

283       public boolean equalsMetadata(String currentMetadata, SFile existingFile) {
284           try {
285               if (this == PNG) {
286                   final MetadataTag tag = new MetadataTag(existingFile, "plantuml");
287                   final String previousMetadata = tag.getData();
288                   final boolean sameMetadata = currentMetadata.equals(previousMetadata);
289                   return sameMetadata;
290               }
291               if (this == SVG) {
292                   final String svg = FileUtils.readSvg(existingFile);
293                   if (svg == null)
294                       return false;
295   
296                   final String currentSignature = SvgGraphics.getMD5Hex(currentMetadata);
297                   final int idx = svg.lastIndexOf(SvgGraphics.MD5_HEADER);
298                   if (idx != -1) {
299                       final String part = svg.substring(idx + SvgGraphics.MD5_HEADER.length());
300                       return part.startsWith(currentSignature);
301                   }
302   
303               }
304           } catch (IOException e) {
305               Logme.error(e);
306           }
307           return false;
308       }
