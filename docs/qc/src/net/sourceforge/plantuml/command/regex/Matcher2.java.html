45   public class Matcher2 {
46   
47       private final static boolean INSTRUMENT = false;
48       private final Matcher matcher;
49       private final String id;
50   
51       private Matcher2(Matcher matcher, String id) {
52           this.matcher = matcher;
53           this.id = id;
54       }

56       public static Matcher2 build(Pattern pattern, CharSequence input) {
57           final long now = System.currentTimeMillis();
58           final String id = pattern.pattern();
59           try {
60               final Matcher matcher2 = pattern.matcher(input);
61               return new Matcher2(matcher2, id);
62           } finally {
63               if (INSTRUMENT) {
64                   addTime(id, System.currentTimeMillis() - now);
65               }
66           }
67       }

70           final long now = System.currentTimeMillis();

73           } finally {
74               addTime(System.currentTimeMillis() - now);
75           }
76       }

78       private void addTime(long duration) {
79           if (INSTRUMENT == false) {
80               return;
81           }
82           addTime(id, duration);
83       }
84   
85       private static final Map<String, Long> durations = new HashMap<String, Long>();
86       private static long printed;
87   
88       private static synchronized void addTime(String id, long duration) {
89           Long total = durations.get(id);
90           if (total == null) {
91               total = 0L;
92           }
93           total += duration;
94           durations.put(id, total);
95           final String longest = getLongest();
96           if (longest == null) {
97               return;
98           }
99           if (durations.get(longest) > printed) {
100               Log.info("---------- Regex " + longest + " " + durations.get(longest) + "ms (" + durations.size() + ")");
101               printed = durations.get(longest);
102           }
103   
104       }
105   
106       private static String getLongest() {
107           long max = 0;
108           String result = null;
109           for (Map.Entry<String, Long> ent : durations.entrySet()) {
110               if (ent.getValue() > max) {
111                   max = ent.getValue();
112                   result = ent.getKey();
113               }
114           }
115           return result;
116       }
