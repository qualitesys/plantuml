63   public final class CommandFactoryNoteOnLink implements SingleMultiFactoryCommand<CucaDiagram> {
64   
65       private IRegex getRegexConcatSingleLine() {
66           return RegexConcat.build(CommandFactoryNoteOnLink.class.getName() + "single", RegexLeaf.start(), //
67                   new RegexLeaf("note"), //
68                   RegexLeaf.spaceOneOrMore(), //
69                   new RegexLeaf("POSITION", "(right|left|top|bottom)?"), //
70                   RegexLeaf.spaceZeroOrMore(), //
71                   new RegexLeaf("o[nf]"), //
72                   RegexLeaf.spaceOneOrMore(), //
73                   new RegexLeaf("link"), //
74                   RegexLeaf.spaceZeroOrMore(), //
75                   color().getRegex(), //
76                   RegexLeaf.spaceZeroOrMore(), //
77                   new RegexLeaf(":"), //
78                   RegexLeaf.spaceZeroOrMore(), //
79                   new RegexLeaf("NOTE", "(.*)"), RegexLeaf.end());
80       }
81   
82       private IRegex getRegexConcatMultiLine() {
83           return RegexConcat.build(CommandFactoryNoteOnLink.class.getName() + "multi", RegexLeaf.start(), //
84                   new RegexLeaf("note"), //
85                   RegexLeaf.spaceOneOrMore(), //
86                   new RegexLeaf("POSITION", "(right|left|top|bottom)?"), //
87                   RegexLeaf.spaceZeroOrMore(), //
88                   new RegexLeaf("o[nf]"), //
89                   RegexLeaf.spaceOneOrMore(), //
90                   new RegexLeaf("link"), //
91                   RegexLeaf.spaceZeroOrMore(), //
92                   color().getRegex(), RegexLeaf.end());
93       }
94   
95       private static ColorParser color() {
96           return ColorParser.simpleColor(ColorType.BACK);
97       }
98   
99       public Command<CucaDiagram> createMultiLine(boolean withBracket) {
100           return new CommandMultilines2<CucaDiagram>(getRegexConcatMultiLine(), MultilinesStrategy.KEEP_STARTING_QUOTE, Trim.BOTH) {
101   
102               @Override
103               public String getPatternEnd() {
104                   return "^end[%s]?note$";
105               }
106   
107               protected CommandExecutionResult executeNow(final CucaDiagram system, BlocLines lines)
108                       throws NoSuchColorException {
109                   final String line0 = lines.getFirst().getTrimmed().getString();
110                   lines = lines.subExtract(1, 1);
111                   lines = lines.removeEmptyColumns();
112                   if (lines.size() > 0) {
113                       final RegexResult arg = getStartingPattern().matcher(line0);
114                       return executeInternal(system, lines, arg);
115                   }
116                   return CommandExecutionResult.error("No note defined");
117               }
118   
119           };
120       }
121   
122       public Command<CucaDiagram> createSingleLine() {
123           return new SingleLineCommand2<CucaDiagram>(getRegexConcatSingleLine()) {
124   
125               @Override
126               protected CommandExecutionResult executeArg(final CucaDiagram system, LineLocation location,
127                       RegexResult arg) throws NoSuchColorException {
128                   final BlocLines note = BlocLines.getWithNewlines(arg.get("NOTE", 0));
129                   return executeInternal(system, note, arg);
130               }
131           };
132       }
133   
134       private CommandExecutionResult executeInternal(CucaDiagram diagram, BlocLines note, final RegexResult arg)
135               throws NoSuchColorException {
136           final Link link = diagram.getLastLink();
137           if (link == null)
138               return CommandExecutionResult.error("No link defined");
139   
140           Position position = Position.BOTTOM;
141           if (arg.get("POSITION", 0) != null)
142               position = Position.valueOf(StringUtils.goUpperCase(arg.get("POSITION", 0)));
143   
144           Url url = null;
145           if (arg.get("URL", 0) != null) {
146               final UrlBuilder urlBuilder = new UrlBuilder(diagram.getSkinParam().getValue("topurl"), UrlMode.STRICT);
147               url = urlBuilder.getUrl(arg.get("URL", 0));
148           }
149           final Colors colors = color().getColor(arg, diagram.getSkinParam().getIHtmlColorSet());
150           link.addNote(CucaNote.build(note.toDisplay(), position, colors));
151           return CommandExecutionResult.ok();
152       }
