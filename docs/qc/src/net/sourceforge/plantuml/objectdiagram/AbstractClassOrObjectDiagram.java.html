94       private Link foundLink(IEntity entity1, IEntity entity2) {
95           final List<Link> links = getLinks();
96           for (int i = links.size() - 1; i >= 0; i--) {
97               final Link l = links.get(i);
98               if (l.isBetween(entity1, entity2)) {
99                   return l;
100               }
101           }
102           return null;
103       }

122       public CommandExecutionResult associationClass(String name1A, String name1B, String name2A, String name2B,
123               LinkType linkType, Display label) {
124           final Ident ident1A = buildLeafIdent(name1A);
125           final Ident ident1B = buildLeafIdent(name1B);
126           final Ident ident2A = buildLeafIdent(name2A);
127           final Ident ident2B = buildLeafIdent(name2B);
128           final Code code1A = this.V1972() ? ident1A : buildCode(name1A);
129           final Code code1B = this.V1972() ? ident1B : buildCode(name1B);
130           final Code code2A = this.V1972() ? ident2A : buildCode(name2A);
131           final Code code2B = this.V1972() ? ident2B : buildCode(name2B);
132           final IEntity entity1A = getOrCreateLeaf(ident1A, code1A, null, null);
133           final IEntity entity1B = getOrCreateLeaf(ident1B, code1B, null, null);
134           final IEntity entity2A = getOrCreateLeaf(ident2A, code2A, null, null);
135           final IEntity entity2B = getOrCreateLeaf(ident2B, code2B, null, null);
136           final List<Association> same1 = getExistingAssociatedPoints(entity1A, entity1B);
137           final List<Association> same2 = getExistingAssociatedPoints(entity2A, entity2B);
138           if (same1.size() == 0 && same2.size() == 0) {
139               final String tmp1 = this.getUniqueSequence("apoint");
140               final String tmp2 = this.getUniqueSequence("apoint");
141               final Ident ident1 = buildLeafIdent(tmp1);
142               final Ident ident2 = buildLeafIdent(tmp2);
143               final Code code1 = this.V1972() ? ident1 : buildCode(tmp1);
144               final Code code2 = this.V1972() ? ident2 : buildCode(tmp2);
145               final IEntity point1 = getOrCreateLeaf(ident1, code1, LeafType.POINT_FOR_ASSOCIATION, null);
146               final IEntity point2 = getOrCreateLeaf(ident2, code2, LeafType.POINT_FOR_ASSOCIATION, null);
147   
148               insertPointBetween(entity1A, entity1B, point1);
149               insertPointBetween(entity2A, entity2B, point2);
150   
151               final int length = 1;
152               final Link point1ToPoint2 = new Link(getSkinParam().getCurrentStyleBuilder(), point1, point2, linkType,
153                       LinkArg.build(label, length));
154               addLink(point1ToPoint2);
155   
156               return CommandExecutionResult.ok();
157           }
158           return CommandExecutionResult.error("Cannot link two associations points");
159       }

199       public boolean associationClass(int mode, String name1, String name2, IEntity associed, LinkType linkType,
200               Display label) {
201           final Ident ident1 = buildLeafIdent(name1);
202           final Ident ident2 = buildLeafIdent(name2);
203           final Code code1 = this.V1972() ? ident1 : buildCode(name1);
204           final Code code2 = this.V1972() ? ident2 : buildCode(name2);
205           final IEntity entity1 = getOrCreateLeaf(ident1, code1, null, null);
206           final IEntity entity2 = getOrCreateLeaf(ident2, code2, null, null);
207           final List<Association> same = getExistingAssociatedPoints(entity1, entity2);
208           if (same.size() > 1) {
209               return false;
210           } else if (same.size() == 0) {
211               final Association association = new Association(mode, entity1, entity2, associed);
212               association.createNew(mode, linkType, label);
213   
214               this.associations.add(association);
215               return true;
216           }
217           assert same.size() == 1;
218           final Association association = same.get(0).createSecondAssociation(mode, associed, label);
219           association.createInSecond(linkType, label);
220   
221           this.associations.add(association);
222           return true;
223       }

225       private List<Association> getExistingAssociatedPoints(final IEntity entity1, final IEntity entity2) {
226           final List<Association> same = new ArrayList<>();
227           for (Association existing : associations) {
228               if (existing.sameCouple(entity1, entity2)) {
229                   same.add(existing);
230               }
231           }
232           return same;
233       }
