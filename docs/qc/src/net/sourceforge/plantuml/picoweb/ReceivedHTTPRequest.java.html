32           final String requestLine = readLine(in);
33

40           request.path = tokenizer.nextToken();

44   
45           while (true) {

58       private static int parseContentLengthHeader(String line) throws IOException {
59           int contentLength;

61           try {

63           } catch (NumberFormatException e) {
64               throw new BadRequest400("Invalid content length");
65           }
66   
67           if (contentLength < 0) {

74       private static byte[] readBody(InputStream in, int contentLength) throws IOException {
75           if (contentLength == 0) {
76               return new byte[0];

78   
79           final byte[] body = new byte[contentLength];
80           int n = 0;
81           int offset = 0;
82   
83           // java.io.InputStream.readNBytes() can replace this from Java 9
84           while (n < contentLength) {
85               int count = in.read(body, offset + n, contentLength - n);
86               if (count < 0) {
87                   throw new BadRequest400("Body too short");
88               }
89               n += count;
90           }
91           return body;
92       }

104   
105           if (builder.length() > 0 && builder.charAt(builder.length() - 1) == '\r') {
106               builder.deleteCharAt(builder.length() - 1);
107           }
