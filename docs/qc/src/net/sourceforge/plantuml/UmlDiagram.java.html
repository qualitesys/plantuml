125       @Override

127               throws IOException {
128   
129           fileFormatOption = fileFormatOption.withTikzFontDistortion(getSkinParam().getTikzFontDistortion());
130   
131           if (fileFormatOption.getFileFormat() == FileFormat.PDF)
132               return exportDiagramInternalPdf(os, index);

134           try {
135               final ImageData imageData = exportDiagramInternal(os, index, fileFormatOption);

137               return imageData;
138           } catch (NoStyleAvailableException e) {

145               // Logme.error(e);

157       public static void exportDiagramError(OutputStream os, Throwable exception, FileFormatOption fileFormat, long seed,
158               String metadata, String flash, List<String> strings) throws IOException {
159   
160           if (fileFormat.getFileFormat() == FileFormat.ATXT || fileFormat.getFileFormat() == FileFormat.UTXT) {
161               exportDiagramErrorText(os, exception, strings);
162               return;
163           }
164   
165           strings.addAll(CommandExecutionResult.getStackTrace(exception));
166   
167           BufferedImage im2 = null;
168           if (flash != null) {
169               final FlashCodeUtils utils = FlashCodeFactory.getFlashCodeUtils();
170               try {
171                   im2 = utils.exportFlashcode(flash, Color.BLACK, Color.WHITE);
172               } catch (Throwable e) {
173                   Log.error("Issue in flashcode generation " + e);
174                   // Logme.error(e);
175               }
176               if (im2 != null)
177                   GraphvizCrash.addDecodeHint(strings);
178   
179           }
180           final BufferedImage im = im2;
181           final TextBlockBackcolored graphicStrings = GraphicStrings.createBlackOnWhite(strings, IconLoader.getRandom(),
182                   GraphicPosition.BACKGROUND_CORNER_TOP_RIGHT);
183   
184           final UDrawable drawable = (im == null) ? graphicStrings : new UDrawable() {
185               public void drawU(UGraphic ug) {
186                   graphicStrings.drawU(ug);
187                   final double height = graphicStrings.calculateDimension(ug.getStringBounder()).getHeight();
188                   ug = ug.apply(UTranslate.dy(height));
189                   ug.draw(new UImage(new PixelImage(im, AffineTransformType.TYPE_NEAREST_NEIGHBOR)).scale(3));
190               }
191           };
192   
193           plainImageBuilder(drawable, fileFormat).metadata(metadata).seed(seed).write(os);
194       }

216       static private List<String> getFailureText1(Throwable exception, String graphvizVersion, String textDiagram) {
217           final List<String> strings = GraphvizCrash.anErrorHasOccured(exception, textDiagram);
218           strings.add("PlantUML (" + Version.versionString() + ") cannot parse result from dot/GraphViz.");
219           if (exception instanceof EmptySvgException)
220               strings.add("Because dot/GraphViz returns an empty string.");
221   
222           GraphvizCrash.checkOldVersionWarning(strings);
223           if (graphvizVersion != null) {
224               strings.add(" ");
225               strings.add("GraphViz version used : " + graphvizVersion);
226           }
227           GraphvizCrash.pleaseGoTo(strings);
228           GraphvizCrash.addProperties(strings);
229           strings.add(" ");
230           GraphvizCrash.thisMayBeCaused(strings);
231           strings.add(" ");
232           GraphvizCrash.youShouldSendThisDiagram(strings);
233           strings.add(" ");
234           return strings;
235       }

301       @Override
302       public String getWarningOrError() {
303           if (lastInfo == null)
304               return null;
305   
306           final double actualWidth = lastInfo.getWidth();
307           if (actualWidth == 0)
308               return null;
309   
310           final String value = getSkinParam().getValue("widthwarning");
311           if (value == null)
312               return null;
313   
314           if (value.matches("\\d+") == false)
315               return null;
316   
317           final int widthwarning = Integer.parseInt(value);
318           if (actualWidth > widthwarning)
319               return "The image is " + ((int) actualWidth) + " pixel width. (Warning limit is " + widthwarning + ")";
320   
321           return null;
322       }
