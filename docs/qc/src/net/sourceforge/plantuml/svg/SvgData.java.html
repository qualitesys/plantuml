36   package net.sourceforge.plantuml.svg;

38   import java.util.StringTokenizer;
39   import java.util.regex.Matcher;
40   import java.util.regex.Pattern;
41   
42   import net.sourceforge.plantuml.StringUtils;
43   
44   public final class SvgData {
45   
46       private String svg;
47       private double startX;
48       private double startY;
49       private double width;
50       private double height;
51   
52       public static SvgData fromGraphViz(String s) {
53           return new SvgData(removeSvgXmlHeader1(s));
54       }
55   
56       private SvgData(String svg) {
57           this.svg = svg;
58           init();
59       }
60   
61       private SvgData() {
62       }
63   
64       private static String removeSvgXmlHeader1(String svg) {
65           final String newString = "<svg xmlns=\"http://www.w3.org/2000/svg\" xmlns:xlink=\"http://www.w3.org/1999/xlink\">";
66           svg = svg.replaceFirst("(?i)<svg[^>]*>", newString);
67           return svg;
68       }
69   
70       public SvgData mutateFromSvgTitler(String newSvg, double suppHeight, double suppWidth) {
71           final SvgData result = new SvgData();
72           result.svg = newSvg;
73           result.startX = this.startX - suppWidth / 2;
74           result.startY = this.startY;
75           result.width = this.width + suppWidth;
76           result.height = this.height + suppHeight;
77           return result;
78       }
79   
80       public final String getSvg() {
81           return modifiedSvgXmlHeader();
82       }
83   
84       private void init() {
85           final Pattern p = Pattern.compile("(?i)<polygon\\s+[^>]*points=\"([^\"]+)\"");
86           final Matcher m = p.matcher(svg);
87           double minX = Double.MAX_VALUE;
88           double minY = Double.MAX_VALUE;
89           double maxX = -Double.MAX_VALUE;
90           double maxY = -Double.MAX_VALUE;
91           if (m.find() == false) {
92               return;
93           }
94           final String points = m.group(1);
95           final StringTokenizer st = new StringTokenizer(points, " ");
96           while (st.hasMoreTokens()) {
97               final String token = st.nextToken();
98               final StringTokenizer st2 = new StringTokenizer(token, ",");
99               final double x = Double.parseDouble(StringUtils.trin(st2.nextToken()));
100               final double y = Double.parseDouble(StringUtils.trin(st2.nextToken()));
101               if (x < minX) {
102                   minX = x;
103               }
104               if (y < minY) {
105                   minY = y;
106               }
107               if (x > maxX) {
108                   maxX = x;
109               }
110               if (y > maxY) {
111                   maxY = y;
112               }
113           }
114   
115           this.width = maxX - minX;
116           this.height = maxY - minY;
117       }
118   
119       private String modifiedSvgXmlHeader() {
120           final StringBuilder newString = new StringBuilder(
121                   "<svg xmlns=\"http://www.w3.org/2000/svg\" xmlns:xlink=\"http://www.w3.org/1999/xlink\" ");
122           newString.append("style=\"width:");
123           newString.append(Math.round(getWidth()));
124           newString.append(";height:");
125           newString.append(Math.round(getHeight()));
126           newString.append(";\" ");
127           newString.append("width=\"" + Math.round(getWidth()) + "pt\" ");
128           newString.append("height=\"" + Math.round(getHeight()) + "pt\" ");
129           newString.append("viewBox=\"" + Math.round(startX) + " " + Math.round(startY) + " " + Math.round(getWidth()) + " "
130                   + Math.round(getHeight()) + "\"");
131           // newString.append("viewBox=\"" + Math.round(minX) + " " +
132           // Math.round(minY) + " " + Math.round(getWidth()) + " "
133           // + Math.round(getHeight()) + "\"");
134           newString.append(">");
135           return svg.replaceFirst("(?i)<svg[^>]*>", newString.toString());
136       }
137   
138       public double getWidth() {
139           return width;
140       }
141   
142       public double getHeight() {
143           return height;
144       }
