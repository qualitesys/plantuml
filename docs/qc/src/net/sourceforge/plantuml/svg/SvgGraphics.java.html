331       public String createSvgGradient(String color1, String color2, char policy) {
332           final List<Object> key = Arrays.asList((Object) color1, color2, policy);
333           String id = gradients.get(key);
334           if (id == null) {
335               final Element elt = (Element) document.createElement("linearGradient");
336               if (policy == '|') {
337                   elt.setAttribute("x1", "0%");
338                   elt.setAttribute("y1", "50%");
339                   elt.setAttribute("x2", "100%");
340                   elt.setAttribute("y2", "50%");
341               } else if (policy == '\\') {
342                   elt.setAttribute("x1", "0%");
343                   elt.setAttribute("y1", "100%");
344                   elt.setAttribute("x2", "100%");
345                   elt.setAttribute("y2", "0%");
346               } else if (policy == '-') {
347                   elt.setAttribute("x1", "50%");
348                   elt.setAttribute("y1", "0%");
349                   elt.setAttribute("x2", "50%");
350                   elt.setAttribute("y2", "100%");
351               } else {
352                   elt.setAttribute("x1", "0%");
353                   elt.setAttribute("y1", "0%");
354                   elt.setAttribute("x2", "100%");
355                   elt.setAttribute("y2", "100%");
356               }
357               id = gradientId + gradients.size();
358               gradients.put(key, id);
359               elt.setAttribute("id", id);
360   
361               final Element stop1 = (Element) document.createElement("stop");
362               stop1.setAttribute("stop-color", color1);
363               stop1.setAttribute("offset", "0%");
364               final Element stop2 = (Element) document.createElement("stop");
365               stop2.setAttribute("stop-color", color2);
366               stop2.setAttribute("offset", "100%");
367   
368               elt.appendChild(stop1);
369               elt.appendChild(stop2);
370               defs.appendChild(elt);
371           }
372           return id;
373       }
