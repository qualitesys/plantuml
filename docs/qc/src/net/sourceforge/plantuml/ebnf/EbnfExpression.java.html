58   public class EbnfExpression implements TextBlockable {

60       private final List<Token> tokens = new ArrayList<>();
61       private final boolean isCompact;
62       private final String commentAbove;
63       private final String commentBelow;

65       public static EbnfExpression create(CharIterator it, boolean isCompact, String commentAbove, String commentBelow) {
66           return new EbnfExpression(it, isCompact, commentAbove, commentBelow);
67       }

69       private EbnfExpression(CharIterator it, boolean isCompact, String commentAbove, String commentBelow) {
70           this.isCompact = isCompact;
71           this.commentAbove = commentAbove;
72           this.commentBelow = commentBelow;
73           while (true) {
74               final char ch = it.peek(0);
75               if (Character.isWhitespace(ch)) {
76               } else if (isLetterOrDigit(ch)) {
77                   final String litteral = readLitteral(it);
78                   tokens.add(new Token(Symbol.LITTERAL, litteral));
79                   continue;
80               } else if (ch == '*') {
81                   tokens.add(new Token(Symbol.REPETITION_SYMBOL, null));
82               } else if (ch == '(' && it.peek(1) == '*') {
83                   final String comment = readComment(it);
84                   if (comment.trim().length() > 0)
85                       tokens.add(new Token(Symbol.COMMENT_TOKEN, comment));
86                   continue;
87               } else if (ch == ',') {
88                   tokens.add(new Token(Symbol.CONCATENATION, null));
89               } else if (ch == '|') {
90                   tokens.add(new Token(Symbol.ALTERNATION, null));
91               } else if (ch == '=') {
92                   tokens.add(new Token(Symbol.DEFINITION, null));
93               } else if (ch == '(') {
94                   tokens.add(new Token(Symbol.GROUPING_OPEN, null));
95               } else if (ch == ')') {
96                   tokens.add(new Token(Symbol.GROUPING_CLOSE, null));
97               } else if (ch == '[') {
98                   tokens.add(new Token(Symbol.OPTIONAL_OPEN, null));
99               } else if (ch == ']') {
100                   tokens.add(new Token(Symbol.OPTIONAL_CLOSE, null));
101               } else if (ch == '{') {
102                   tokens.add(new Token(Symbol.REPETITION_OPEN, null));
103               } else if (ch == '}' && it.peek(1) == '-') {
104                   tokens.add(new Token(Symbol.REPETITION_MINUS_CLOSE, null));
105                   it.next();
106               } else if (ch == '}') {
107                   tokens.add(new Token(Symbol.REPETITION_CLOSE, null));
108               } else if (ch == ';' || ch == 0) {
109                   // it.next();
110                   break;
111               } else if (ch == '\"') {
112                   final String litteral = readString(it);
113                   tokens.add(new Token(Symbol.TERMINAL_STRING1, litteral));
114               } else if (ch == '\'') {
115                   final String litteral = readString(it);
116                   tokens.add(new Token(Symbol.TERMINAL_STRING2, litteral));
117               } else {
118                   tokens.clear();
119                   return;
120               }
121               it.next();
122               continue;
123           }
124       }

126       public TextBlock getUDrawable(ISkinParam skinParam) {
127           final Style style = ETile.getStyleSignature().getMergedStyle(skinParam.getCurrentStyleBuilder());
128           final FontConfiguration fc = style.getFontConfiguration(skinParam.getIHtmlColorSet());
129   
130           if (tokens.size() == 0)
131               return EbnfEngine.syntaxError(fc, skinParam);
132   
133           try {
134               final Iterator<Token> iterator = tokens.iterator();
135               final Token name = iterator.next();
136               final Token definition = iterator.next();
137               if (definition.getSymbol() != Symbol.DEFINITION)
138                   return EbnfEngine.syntaxError(fc, skinParam);
139   
140               final TextBlock main;
141               if (iterator.hasNext()) {
142                   final List<Token> full = new ShuntingYard(iterator).getOuputQueue();
143                   if (full.size() == 0)
144                       return EbnfEngine.syntaxError(fc, skinParam);
145   
146                   main = getMainDrawing(skinParam, full.iterator());
147               } else {
148                   final HColor lineColor = style.value(PName.LineColor).asColor(skinParam.getIHtmlColorSet());
149                   main = new ETileWithCircles(new ETileEmpty(), lineColor);
150               }
151   
152               TextBlock titleBox = new TitleBox(name.getData(), fc);
153               if (commentAbove != null)
154                   titleBox = TextBlockUtils.mergeTB(getNoteAbove(skinParam), titleBox, HorizontalAlignment.CENTER);
155               if (commentBelow != null)
156                   titleBox = TextBlockUtils.mergeTB(titleBox, getNoteBelow(skinParam), HorizontalAlignment.CENTER);
157   
158               return TextBlockUtils.mergeTB(titleBox, TextBlockUtils.withMargin(main, 0, 0, 10, 15),
159                       HorizontalAlignment.LEFT);
160           } catch (Exception e) {
161               e.printStackTrace();
162               return EbnfEngine.syntaxError(fc, skinParam);
163           }
164       }

166       private TextBlock getNoteAbove(ISkinParam skinParam) {
167           if (commentAbove == null)
168               return null;
169           final FloatingNote note = FloatingNote.create(Display.getWithNewlines(commentAbove), skinParam, SName.ebnf);
170           return note;
171       }
172   
173       private TextBlock getNoteBelow(ISkinParam skinParam) {
174           if (commentBelow == null)
175               return null;
176           final FloatingNote note = FloatingNote.create(Display.getWithNewlines(commentBelow), skinParam, SName.ebnf);
177           return note;
178       }

180       private TextBlock getMainDrawing(ISkinParam skinParam, Iterator<Token> it) {
181           final EbnfEngine engine = new EbnfEngine(skinParam);
182           while (it.hasNext()) {
183               final Token element = it.next();
184               if (element.getSymbol() == Symbol.TERMINAL_STRING1 || element.getSymbol() == Symbol.TERMINAL_STRING2
185                       || element.getSymbol() == Symbol.LITTERAL)
186                   engine.push(element);
187               else if (element.getSymbol() == Symbol.COMMENT_ABOVE)
188                   engine.commentAbove(element.getData());
189               else if (element.getSymbol() == Symbol.COMMENT_BELOW)
190                   engine.commentBelow(element.getData());
191               else if (element.getSymbol() == Symbol.ALTERNATION)
192                   engine.alternation();
193               else if (element.getSymbol() == Symbol.CONCATENATION)
194                   engine.concatenation();
195               else if (element.getSymbol() == Symbol.OPTIONAL)
196                   engine.optional();
197               else if (element.getSymbol() == Symbol.REPETITION_ZERO_OR_MORE)
198                   engine.repetitionZeroOrMore(isCompact);
199               else if (element.getSymbol() == Symbol.REPETITION_ONE_OR_MORE)
200                   engine.repetitionOneOrMore();
201               else if (element.getSymbol() == Symbol.REPETITION_SYMBOL)
202                   engine.repetitionSymbol();
203               else
204                   throw new UnsupportedOperationException(element.toString());
205           }
206   
207           return engine.getTextBlock();
208       }

210       private String readString(CharIterator it) {
211           final char separator = it.peek(0);
212           it.next();
213           final StringBuilder sb = new StringBuilder();
214           while (true) {
215               final char ch = it.peek(0);
216               if (ch == separator)
217                   return sb.toString();
218               sb.append(ch);
219               it.next();
220           }
221       }
222   
223       private String readLitteral(CharIterator it) {
224           final StringBuilder sb = new StringBuilder();
225           while (true) {
226               final char ch = it.peek(0);
227               if (isLetterOrDigit(ch) == false)
228                   return sb.toString();
229               sb.append(ch);
230               it.next();
231           }
232       }
233   
234       private String readComment(CharIterator it) {
235           final StringBuilder sb = new StringBuilder();
236           it.next();
237           it.next();
238           while (true) {
239               final char ch = it.peek(0);
240               if (ch == '\0')
241                   return sb.toString();
242               if (ch == '*' && it.peek(1) == ')') {
243                   it.next();
244                   it.next();
245                   return sb.toString();
246               }
247               sb.append(ch);
248               it.next();
249           }
250       }

252       private boolean isLetterOrDigit(char ch) {
253           return ch == '-' || ch == '_' || Character.isLetterOrDigit(ch);
254       }

256       public boolean isEmpty() {
257           return tokens.size() == 0;
258       }
