36   package net.sourceforge.plantuml.ebnf;
37   
38   import net.sourceforge.plantuml.ISkinParam;
39   import net.sourceforge.plantuml.activitydiagram3.ftile.vcompact.FloatingNote;
40   import net.sourceforge.plantuml.awt.geom.XDimension2D;
41   import net.sourceforge.plantuml.cucadiagram.Display;
42   import net.sourceforge.plantuml.graphic.StringBounder;
43   import net.sourceforge.plantuml.graphic.TextBlock;
44   import net.sourceforge.plantuml.graphic.TextBlockUtils;
45   import net.sourceforge.plantuml.style.SName;
46   import net.sourceforge.plantuml.ugraphic.UGraphic;
47   import net.sourceforge.plantuml.ugraphic.URectangle;
48   import net.sourceforge.plantuml.ugraphic.UTranslate;
49   import net.sourceforge.plantuml.ugraphic.color.HColors;
50   
51   public class ETileOptional extends ETile {
52   
53       private final double deltax;
54       private final double deltay = 16;
55       private final ETile orig;
56       private final ISkinParam skinParam;
57       private String commentAbove;
58       private String commentBelow;
59   
60       private final boolean specialForAlternate;
61   
62       public ETileOptional(ETile orig, ISkinParam skinParam) {
63           this.skinParam = skinParam;
64           this.orig = orig;
65           this.specialForAlternate = orig instanceof ETileAlternation;
66           this.deltax = this.specialForAlternate ? 0 : 15;
67       }
68   
69       @Override
70       public double getH1(StringBounder stringBounder) {
71           final TextBlock note = getNoteAbove(stringBounder);
72           return deltay + orig.getH1(stringBounder) + note.calculateDimension(stringBounder).getHeight();
73       }
74   
75       private double getDeltaY(StringBounder stringBounder) {
76           final TextBlock note = getNoteAbove(stringBounder);
77           return deltay + note.calculateDimension(stringBounder).getHeight();
78       }
79   
80       @Override
81       public double getH2(StringBounder stringBounder) {
82           final TextBlock note = getNoteBelow(stringBounder);
83           return orig.getH2(stringBounder) + note.calculateDimension(stringBounder).getHeight();
84       }
85   
86       @Override
87       public double getWidth(StringBounder stringBounder) {
88           return orig.getWidth(stringBounder) + 2 * deltax;
89       }
90   
91       @Override
92       public void drawU(UGraphic ug) {
93           final StringBounder stringBounder = ug.getStringBounder();
94           final XDimension2D dim = calculateDimension(stringBounder);
95           if (TRACE)
96               ug.apply(HColors.BLUE).draw(new URectangle(dim));
97   
98           final double linePos = getH1(stringBounder);
99   
100           final double posA = specialForAlternate ? 12 : 8;
101           final double corner = specialForAlternate ? 12 : 8;
102           final double posB = dim.getWidth() - posA;
103   
104           final double posYoptionalLine = getDeltaY(stringBounder) - 11;
105   
106           CornerCurved.createSE(corner).drawU(ug.apply(new UTranslate(posA, linePos)));
107           drawVline(ug, posA, corner + posYoptionalLine, linePos - corner);
108           CornerCurved.createNW(corner).drawU(ug.apply(new UTranslate(posA, posYoptionalLine)));
109   
110           drawHlineDirected(ug, posYoptionalLine, posA + corner, posB - corner, 0.4);
111   
112           CornerCurved.createSW(corner).drawU(ug.apply(new UTranslate(posB, linePos)));
113           drawVline(ug, posB, corner + posYoptionalLine, linePos - corner);
114           CornerCurved.createNE(corner).drawU(ug.apply(new UTranslate(posB, posYoptionalLine)));
115   
116           drawHline(ug, linePos, 0, deltax);
117           drawHline(ug, linePos, dim.getWidth() - deltax, dim.getWidth());
118   
119           orig.drawU(ug.apply(new UTranslate(deltax, getDeltaY(stringBounder))));
120   
121           final TextBlock noteAbove = getNoteAbove(stringBounder);
122           if (noteAbove != TextBlockUtils.EMPTY_TEXT_BLOCK) {
123               final double pos2 = (dim.getWidth() - noteAbove.calculateDimension(stringBounder).getWidth()) / 2;
124               noteAbove.drawU(ug.apply(UTranslate.dx(pos2)));
125           }
126   
127           final TextBlock noteBelow = getNoteBelow(stringBounder);
128           if (noteBelow != TextBlockUtils.EMPTY_TEXT_BLOCK) {
129               final XDimension2D dimBelow = noteBelow.calculateDimension(stringBounder);
130               final double pos2 = (dim.getWidth() - dimBelow.getWidth()) / 2;
131               noteBelow.drawU(ug.apply(new UTranslate(pos2, dim.getHeight() - dimBelow.getHeight())));
132           }
133   
134       }
135   
136       @Override
137       public void push(ETile tile) {
138           throw new UnsupportedOperationException();
139       }
140   
141       @Override
142       protected void addCommentAbove(String comment) {
143           this.commentAbove = comment;
144       }
145   
146       @Override
147       protected void addCommentBelow(String comment) {
148           this.commentBelow = comment;
149       }
150   
151       private TextBlock getNoteAbove(StringBounder stringBounder) {
152           if (commentAbove == null)
153               return TextBlockUtils.EMPTY_TEXT_BLOCK;
154           final FloatingNote note = FloatingNote.create(Display.getWithNewlines(commentAbove), skinParam, SName.ebnf);
155           return TextBlockUtils.withMargin(note, 0, 0, 0, 10);
156       }
157   
158       private TextBlock getNoteBelow(StringBounder stringBounder) {
159           if (commentBelow == null)
160               return TextBlockUtils.EMPTY_TEXT_BLOCK;
161           final FloatingNote note = FloatingNote.create(Display.getWithNewlines(commentBelow), skinParam, SName.ebnf);
162           return TextBlockUtils.withMargin(note, 0, 0, 10, 0);
163       }
164   
165   }
