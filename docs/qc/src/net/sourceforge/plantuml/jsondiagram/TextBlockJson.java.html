95           double getHeightOfRow(StringBounder stringBounder) {
96               final double height = b1.calculateDimension(stringBounder).getHeight();

105       private HColor getBackColor() {

109       TextBlockJson(ISkinParam skinParam, JsonValue root, List<String> allHighlighteds, Style styleNode,
110               Style styleNodeHightlight, Style styleNodeHeader, Style styleNodeHeaderHighlight) {
111           this.skinParam = skinParam;
112           this.styleNode = styleNode;
113           this.styleNodeHeaderHighlight = styleNodeHeaderHighlight;
114           this.styleNodeHightlight = styleNodeHightlight;
115           this.styleNodeHeader = styleNodeHeader;
116           this.root = root;
117           if (root instanceof JsonObject)
118               for (Member member : (JsonObject) root) {
119                   final String key = member.getName();
120                   final String value = getShortString(member.getValue());
121   
122                   final boolean highlighted = isHighlighted(key, allHighlighteds);
123                   final TextBlock block1 = getTextBlock(getStyleToUse(true, highlighted), key);
124                   final TextBlock block2 = getTextBlock(getStyleToUse(false, highlighted), value);
125                   this.lines.add(new Line(block1, block2, highlighted));
126               }
127           if (root instanceof JsonArray) {
128               int i = 0;
129               for (JsonValue value : (JsonArray) root) {
130                   final boolean highlighted = isHighlighted("" + i, allHighlighteds);
131                   final TextBlock block2 = getTextBlock(getStyleToUse(false, highlighted), getShortString(value));
132                   this.lines.add(new Line(block2, highlighted));
133                   i++;
134               }
135           }
136       }

138       private Style getStyleToUse(boolean header, boolean highlighted) {
139           if (header && highlighted)
140               return styleNodeHeaderHighlight;
141   
142           if (highlighted)
143               return styleNodeHightlight;
144   
145           if (header)
146               return styleNodeHeader;
147   
148           return styleNode;
149       }

151       private boolean isHighlighted(String key, List<String> highlighted) {
152           for (String tmp : highlighted)
153               if (tmp.trim().equals("\"" + key + "\""))
154                   return true;
155   
156           return false;
157       }

159       public int size() {
160           int size = 0;

163                   size++;
164   
165           if (root instanceof JsonArray)
166               for (JsonValue value : (JsonArray) root)
167                   size++;
168

170

173       private String getShortString(JsonValue value) {
174           if (value.isString())
175               return value.asString();
176   
177           if (value.isNull())
178               return "<U+2400>";
179           // return "<U+2205> null";
180   
181           if (value.isNumber())
182               return value.toString();
183   
184           if (value.isBoolean())
185               if (value.isTrue())
186                   return "<U+2611> true";
187               else
188                   return "<U+2610> false";
189   
190           return "   ";
191       }

193       public List<JsonValue> children() {
194           final List<JsonValue> result = new ArrayList<>();
195           if (root instanceof JsonObject)
196               for (Member member : (JsonObject) root) {
197                   final JsonValue value = member.getValue();
198                   if (value instanceof JsonObject || value instanceof JsonArray)
199                       result.add(value);
200                   else
201                       result.add(null);
202               }
203   
204           if (root instanceof JsonArray)
205               for (JsonValue value : (JsonArray) root)
206                   if (value instanceof JsonObject || value instanceof JsonArray)
207                       result.add(value);
208                   else
209                       result.add(null);
210   
211           return Collections.unmodifiableList(result);
212       }

214       public List<String> keys() {
215           final List<String> result = new ArrayList<>();
216           if (root instanceof JsonObject)
217               for (Member member : (JsonObject) root) {
218                   final String key = member.getName();
219                   result.add(key);
220               }
221   
222           if (root instanceof JsonArray) {
223               int i = 0;
224               for (JsonValue value : (JsonArray) root) {
225                   result.add("" + i);
226                   i++;
227               }
228           }
229           return Collections.unmodifiableList(result);
230       }

238           double width = 0;
239           for (Line line : lines)
240               width = Math.max(width, line.b1.calculateDimension(stringBounder).getWidth());
241   
242           return width;
243       }

245       public double getWidthColB(StringBounder stringBounder) {
246           double width = 0;
247           for (Line line : lines)
248               if (line.b2 != null)
249                   width = Math.max(width, line.b2.calculateDimension(stringBounder).getWidth());
250   
251           return width;
252       }

254       public void drawU(final UGraphic ug) {
255           final StringBounder stringBounder = ug.getStringBounder();
256   
257           final XDimension2D fullDim = calculateDimension(stringBounder);
258           double trueWidth = Math.max(fullDim.getWidth(), totalWidth);
259           final double widthColA = getWidthColA(stringBounder);
260           final double widthColB = getWidthColB(stringBounder);
261   
262           double y = 0;
263           final UGraphic ugNode = styleNode.applyStrokeAndLineColor(ug, skinParam.getIHtmlColorSet());
264           for (Line line : lines) {
265               final double heightOfRow = line.getHeightOfRow(stringBounder);
266               y += heightOfRow;
267           }
268           if (y == 0)
269               y = 15;
270           if (trueWidth == 0)
271               trueWidth = 30;
272   
273           final double round = styleNode.value(PName.RoundCorner).asDouble();
274           final URectangle fullNodeRectangle = new URectangle(trueWidth, y).rounded(round);
275           final HColor backColor = styleNode.value(PName.BackGroundColor).asColor(skinParam.getIHtmlColorSet());
276           ugNode.apply(backColor.bg()).apply(backColor).draw(fullNodeRectangle);
277   
278           final Style styleSeparator = styleNode.getSignature().add(SName.separator)
279                   .getMergedStyle(skinParam.getCurrentStyleBuilder());
280           final UGraphic ugSeparator = styleSeparator.applyStrokeAndLineColor(ug, skinParam.getIHtmlColorSet());
281   
282           y = 0;
283           for (Line line : lines) {
284               final UGraphic ugline = ugSeparator.apply(UTranslate.dy(y));
285               final double heightOfRow = line.getHeightOfRow(stringBounder);
286               if (line.highlighted) {
287                   final URectangle back = new URectangle(trueWidth - 2, heightOfRow).rounded(4);
288                   ugline.apply(getBackColor()).apply(getBackColor().bg()).apply(new UTranslate(1.5, 0)).draw(back);
289               }
290   
291               if (y > 0)
292                   ugline.draw(ULine.hline(trueWidth));
293   
294               final HorizontalAlignment horizontalAlignment = styleNode.getHorizontalAlignment();
295               horizontalAlignment.draw(ugline, line.b1, 0, widthColA);
296   
297               if (line.b2 != null) {
298                   final UGraphic uglineColB = ugline.apply(UTranslate.dx(widthColA));
299                   horizontalAlignment.draw(uglineColB, line.b2, 0, widthColB);
300                   uglineColB.draw(ULine.vline(heightOfRow));
301               }
302   
303               y += heightOfRow;
304           }
305           ugNode.draw(fullNodeRectangle);
306   
307       }

309       private double getTotalHeight(StringBounder stringBounder) {
310           double height = 0;
311           for (Line line : lines)
312               height += line.getHeightOfRow(stringBounder);
313   
314           return height;
315       }

317       private TextBlock getTextBlock(Style style, String key) {
318           final Display display = Display.getWithNewlines(key);
319           final FontConfiguration fontConfiguration = style.getFontConfiguration(skinParam.getIHtmlColorSet());
320           final LineBreakStrategy wrap = style.wrapWidth();
321           final HorizontalAlignment horizontalAlignment = style.getHorizontalAlignment();
322           TextBlock result = display.create0(fontConfiguration, horizontalAlignment, skinParam, wrap,
323                   CreoleMode.NO_CREOLE, null, null);
324           result = TextBlockUtils.withMargin(result, 5, 2);
325           return result;
326       }
