36   package net.sourceforge.plantuml;
37   
38   import java.util.ArrayList;
39   import java.util.Collections;
40   import java.util.List;
41   import java.util.Set;
42   
43   public class CMapData {
44   
45       private final StringBuilder stringBuilder = new StringBuilder();
46   
47       public String asString(String nameId) {
48           return "<map id=\"" + nameId + "_map\" name=\"" + nameId + "_map\">\n" + stringBuilder.toString() + "</map>\n";
49       }
50   
51       public boolean containsData() {
52           return stringBuilder.length() > 0;
53       }
54   
55       public void appendString(String s) {
56           stringBuilder.append(s);
57       }
58   
59       public void appendLong(long s) {
60           stringBuilder.append(s);
61       }
62   
63       private void appendUrl(int seq, Url url, double scale) {
64           appendString("<area shape=\"rect\" id=\"id");
65           appendLong(seq);
66           appendString("\" href=\"");
67           appendString(url.getUrl());
68           appendString("\" title=\"");
69           final String tooltip = url.getTooltip().replaceAll("\\\\n", BackSlash.NEWLINE).replaceAll("&", "&#38;")
70                   .replaceAll("\"", "&#34;").replaceAll("\'", "&#39;");
71           appendString(tooltip);
72           appendString("\" alt=\"\" coords=\"");
73           appendString(url.getCoords(scale));
74           appendString("\"/>");
75   
76           appendString(BackSlash.NEWLINE);
77       }
78   
79       public static CMapData cmapString(Set<Url> allUrlEncountered, double scale) {
80           final CMapData cmapdata = new CMapData();
81   
82           final List<Url> all = new ArrayList<>(allUrlEncountered);
83           Collections.sort(all, Url.SURFACE_COMPARATOR);
84   
85           int seq = 1;
86           for (Url u : all) {
87               if (u.hasData() == false) {
88                   continue;
89               }
90               cmapdata.appendUrl(seq, u, scale);
91               seq++;
92           }
93           return cmapdata;
94       }
95   
96   }
