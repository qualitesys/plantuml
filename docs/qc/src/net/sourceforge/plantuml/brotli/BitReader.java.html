9   package net.sourceforge.plantuml.brotli;
10   
11   /**
12    * Bit reading helpers.
13    */
14   final class BitReader {
15   
16     // Added by Arnaud... not very beautifull
17     private static final boolean CHECK_UNUSED_BYTES_AFTER_END = false;
18   
19     // Possible values: {5, 6}.  5 corresponds to 32-bit build, 6 to 64-bit. This value is used for
20     // conditional compilation -> produced artifacts might be binary INCOMPATIBLE (JLS 13.2).
21     private static final int LOG_BITNESS = 6;
22     private static final int BITNESS = 1 << LOG_BITNESS;
23   
24     private static final int BYTENESS = BITNESS / 8;
25     private static final int CAPACITY = 4096;
26     // After encountering the end of the input stream, this amount of zero bytes will be appended.
27     private static final int SLACK = 64;
28     private static final int BUFFER_SIZE = CAPACITY + SLACK;
29     // Don't bother to replenish the buffer while this number of bytes is available.
30     private static final int SAFEGUARD = 36;
31     private static final int WATERLINE = CAPACITY - SAFEGUARD;
32   
33     // "Half" refers to "half of native integer type", i.e. on 64-bit machines it is 32-bit type,
34     // on 32-bit machines it is 16-bit.
35     private static final int HALF_BITNESS = BITNESS / 2;
36     private static final int HALF_SIZE = BYTENESS / 2;
37     private static final int HALVES_CAPACITY = CAPACITY / HALF_SIZE;
38     private static final int HALF_BUFFER_SIZE = BUFFER_SIZE / HALF_SIZE;
39     private static final int HALF_WATERLINE = WATERLINE / HALF_SIZE;
40   
41     private static final int LOG_HALF_SIZE = LOG_BITNESS - 4;
42   
43     /**
44      * Fills up the input buffer.
45      *
46      * <p> No-op if there are at least 36 bytes present after current position.
47      *
48      * <p> After encountering the end of the input stream, 64 additional zero bytes are copied to the
49      * buffer.
50      */
51     static void readMoreInput(State s) {
52       if (s.halfOffset > HALF_WATERLINE) {
53         doReadMoreInput(s);
54       }
55     }
56   
57     static void doReadMoreInput(State s) {
58       if (s.endOfStreamReached != 0) {
59         if (halfAvailable(s) >= -2) {
60           return;
61         }
62         throw new BrotliRuntimeException("No more input");
63       }
64       int readOffset = s.halfOffset << LOG_HALF_SIZE;
65       int bytesInBuffer = CAPACITY - readOffset;
66       // Move unused bytes to the head of the buffer.
67       Utils.copyBytesWithin(s.byteBuffer, 0, readOffset, CAPACITY);
68       s.halfOffset = 0;
69       while (bytesInBuffer < CAPACITY) {
70         int spaceLeft = CAPACITY - bytesInBuffer;
71         int len = Utils.readInput(s.input, s.byteBuffer, bytesInBuffer, spaceLeft);
72         // EOF is -1 in Java, but 0 in C#.
73         if (len <= 0) {
74           s.endOfStreamReached = 1;
75           s.tailBytes = bytesInBuffer;
76           bytesInBuffer += HALF_SIZE - 1;
77           break;
78         }
79         bytesInBuffer += len;
80       }
81       bytesToNibbles(s, bytesInBuffer);
82     }
83   
84     static void checkHealth(State s, int endOfStream) {
85       if (s.endOfStreamReached == 0) {
86         return;
87       }
88       int byteOffset = (s.halfOffset << LOG_HALF_SIZE) + ((s.bitOffset + 7) >> 3) - BYTENESS;
89       if (byteOffset > s.tailBytes) {
90         throw new BrotliRuntimeException("Read after end");
91       }
92       if (CHECK_UNUSED_BYTES_AFTER_END && (endOfStream != 0) && (byteOffset != s.tailBytes)) {
93         throw new BrotliRuntimeException("Unused bytes after end");
94       }
95     }
96   
97     static void fillBitWindow(State s) {
98       if (s.bitOffset >= HALF_BITNESS) {
99         // Same as doFillBitWindow. JVM fails to inline it.
100         if (BITNESS == 64) {
101           s.accumulator64 = ((long) s.intBuffer[s.halfOffset++] << HALF_BITNESS)
102               | (s.accumulator64 >>> HALF_BITNESS);
103         } else {
104           s.accumulator32 = ((int) s.shortBuffer[s.halfOffset++] << HALF_BITNESS)
105               | (s.accumulator32 >>> HALF_BITNESS);
106         }
107         s.bitOffset -= HALF_BITNESS;
108       }
109     }
110   
111     private static void doFillBitWindow(State s) {
112       if (BITNESS == 64) {
113         s.accumulator64 = ((long) s.intBuffer[s.halfOffset++] << HALF_BITNESS)
114             | (s.accumulator64 >>> HALF_BITNESS);
115       } else {
116         s.accumulator32 = ((int) s.shortBuffer[s.halfOffset++] << HALF_BITNESS)
117             | (s.accumulator32 >>> HALF_BITNESS);
118       }
119       s.bitOffset -= HALF_BITNESS;
120     }
121   
122     static int peekBits(State s) {
123       if (BITNESS == 64) {
124         return (int) (s.accumulator64 >>> s.bitOffset);
125       } else {
126         return s.accumulator32 >>> s.bitOffset;
127       }
128     }
129   
130     static int readFewBits(State s, int n) {
131       int val = peekBits(s) & ((1 << n) - 1);
132       s.bitOffset += n;
133       return val;
134     }
135   
136     static int readBits(State s, int n) {
137       if (HALF_BITNESS >= 24) {
138         return readFewBits(s, n);
139       } else {
140         return (n <= 16) ? readFewBits(s, n) : readManyBits(s, n);
141       }
142     }
143   
144     private static int readManyBits(State s, int n) {
145       int low = readFewBits(s, 16);
146       doFillBitWindow(s);
147       return low | (readFewBits(s, n - 16) << 16);
148     }
149   
150     static void initBitReader(State s) {
151       s.byteBuffer = new byte[BUFFER_SIZE];
152       if (BITNESS == 64) {
153         s.accumulator64 = 0;
154         s.intBuffer = new int[HALF_BUFFER_SIZE];
155       } else {
156         s.accumulator32 = 0;
157         s.shortBuffer = new short[HALF_BUFFER_SIZE];
158       }
159       s.bitOffset = BITNESS;
160       s.halfOffset = HALVES_CAPACITY;
161       s.endOfStreamReached = 0;
162       prepare(s);
163     }
164   
165     private static void prepare(State s) {
166       readMoreInput(s);
167       checkHealth(s, 0);
168       doFillBitWindow(s);
169       doFillBitWindow(s);
170     }
171   
172     static void reload(State s) {
173       if (s.bitOffset == BITNESS) {
174         prepare(s);
175       }
176     }
177   
178     static void jumpToByteBoundary(State s) {
179       int padding = (BITNESS - s.bitOffset) & 7;
180       if (padding != 0) {
181         int paddingBits = readFewBits(s, padding);
182         if (paddingBits != 0) {
183           throw new BrotliRuntimeException("Corrupted padding bits");
184         }
185       }
186     }
187   
188     static int halfAvailable(State s) {
189       int limit = HALVES_CAPACITY;
190       if (s.endOfStreamReached != 0) {
191         limit = (s.tailBytes + (HALF_SIZE - 1)) >> LOG_HALF_SIZE;
192       }
193       return limit - s.halfOffset;
194     }
195   
196     static void copyBytes(State s, byte[] data, int offset, int length) {
197       if ((s.bitOffset & 7) != 0) {
198         throw new BrotliRuntimeException("Unaligned copyBytes");
199       }
200   
201       // Drain accumulator.
202       while ((s.bitOffset != BITNESS) && (length != 0)) {
203         data[offset++] = (byte) peekBits(s);
204         s.bitOffset += 8;
205         length--;
206       }
207       if (length == 0) {
208         return;
209       }
210   
211       // Get data from shadow buffer with "sizeof(int)" granularity.
212       int copyNibbles = Math.min(halfAvailable(s), length >> LOG_HALF_SIZE);
213       if (copyNibbles > 0) {
214         int readOffset = s.halfOffset << LOG_HALF_SIZE;
215         int delta = copyNibbles << LOG_HALF_SIZE;
216         System.arraycopy(s.byteBuffer, readOffset, data, offset, delta);
217         offset += delta;
218         length -= delta;
219         s.halfOffset += copyNibbles;
220       }
221       if (length == 0) {
222         return;
223       }
224   
225       // Read tail bytes.
226       if (halfAvailable(s) > 0) {
227         // length = 1..3
228         fillBitWindow(s);
229         while (length != 0) {
230           data[offset++] = (byte) peekBits(s);
231           s.bitOffset += 8;
232           length--;
233         }
234         checkHealth(s, 0);
235         return;
236       }
237   
238       // Now it is possible to copy bytes directly.
239       while (length > 0) {
240         int len = Utils.readInput(s.input, data, offset, length);
241         if (len == -1) {
242           throw new BrotliRuntimeException("Unexpected end of input");
243         }
244         offset += len;
245         length -= len;
246       }
247     }
248   
249     /**
250      * Translates bytes to halves (int/short).
251      */
252     static void bytesToNibbles(State s, int byteLen) {
253       byte[] byteBuffer = s.byteBuffer;
254       int halfLen = byteLen >> LOG_HALF_SIZE;
255       if (BITNESS == 64) {
256         int[] intBuffer = s.intBuffer;
257         for (int i = 0; i < halfLen; ++i) {
258           intBuffer[i] = ((byteBuffer[i * 4] & 0xFF))
259               | ((byteBuffer[(i * 4) + 1] & 0xFF) << 8)
260               | ((byteBuffer[(i * 4) + 2] & 0xFF) << 16)
261               | ((byteBuffer[(i * 4) + 3] & 0xFF) << 24);
262         }
263       } else {
264         short[] shortBuffer = s.shortBuffer;
265         for (int i = 0; i < halfLen; ++i) {
266           shortBuffer[i] = (short) ((byteBuffer[i * 2] & 0xFF)
267               | ((byteBuffer[(i * 2) + 1] & 0xFF) << 8));
268         }
269       }
270     }
271   }
