11   import java.nio.ByteBuffer;

13   /**
14    * Transformations on dictionary words.
15    */
16   final class Transform {
17   
18     static final int NUM_TRANSFORMS = 121;
19     private static final int[] TRANSFORMS = new int[NUM_TRANSFORMS * 3];
20     private static final byte[] PREFIX_SUFFIX = new byte[217];
21     private static final int[] PREFIX_SUFFIX_HEADS = new int[51];
22   
23     // Bundle of 0-terminated strings.
24     private static final String PREFIX_SUFFIX_SRC = "# #s #, #e #.# the #.com/#\u00C2\u00A0# of # and"
25         + " # in # to #\"#\">#\n#]# for # a # that #. # with #'# from # by #. The # on # as # is #ing"
26         + " #\n\t#:#ed #(# at #ly #=\"# of the #. This #,# not #er #al #='#ful #ive #less #est #ize #"
27         + "ous #";
28     private static final String TRANSFORMS_SRC = "     !! ! ,  *!  &!  \" !  ) *   * -  ! # !  #!*!  "
29         + "+  ,$ !  -  %  .  / #   0  1 .  \"   2  3!*   4%  ! # /   5  6  7  8 0  1 &   $   9 +   : "
30         + " ;  < '  !=  >  ?! 4  @ 4  2  &   A *# (   B  C& ) %  ) !*# *-% A +! *.  D! %'  & E *6  F "
31         + " G% ! *A *%  H! D  I!+!  J!+   K +- *4! A  L!*4  M  N +6  O!*% +.! K *G  P +%(  ! G *D +D "
32         + " Q +# *K!*G!+D!+# +G +A +4!+% +K!+4!*D!+K!*K";
33   
34     private static void unpackTransforms(byte[] prefixSuffix, int[] prefixSuffixHeads,
35         int[] transforms, String prefixSuffixSrc, String transformsSrc) {
36       int n = prefixSuffixSrc.length();
37       int index = 1;
38       for (int i = 0; i < n; ++i) {
39         char c = prefixSuffixSrc.charAt(i);
40         prefixSuffix[i] = (byte) c;
41         if (c == 35) { // == #
42           prefixSuffixHeads[index++] = i + 1;
43           prefixSuffix[i] = 0;
44         }
45       }
46   
47       for (int i = 0; i < NUM_TRANSFORMS * 3; ++i) {
48         transforms[i] = transformsSrc.charAt(i) - 32;
49       }
50     }
51   
52     static {
53       unpackTransforms(PREFIX_SUFFIX, PREFIX_SUFFIX_HEADS, TRANSFORMS, PREFIX_SUFFIX_SRC,
54           TRANSFORMS_SRC);
55     }
56   
57     static int transformDictionaryWord(byte[] dst, int dstOffset, ByteBuffer data, int wordOffset,
58         int len, int transformIndex) {
59       int offset = dstOffset;
60       int transformOffset = 3 * transformIndex;
61       int transformPrefix = PREFIX_SUFFIX_HEADS[TRANSFORMS[transformOffset]];
62       int transformType = TRANSFORMS[transformOffset + 1];
63       int transformSuffix = PREFIX_SUFFIX_HEADS[TRANSFORMS[transformOffset + 2]];
64   
65       // Copy prefix.
66       while (PREFIX_SUFFIX[transformPrefix] != 0) {
67         dst[offset++] = PREFIX_SUFFIX[transformPrefix++];
68       }
69   
70       // Copy trimmed word.
71       int omitFirst = transformType >= 12 ? (transformType - 11) : 0;
72       if (omitFirst > len) {
73         omitFirst = len;
74       }
75       wordOffset += omitFirst;
76       len -= omitFirst;
77       len -= transformType <= 9 ? transformType : 0;  // Omit last.
78       int i = len;
79       while (i > 0) {
80         dst[offset++] = data.get(wordOffset++);
81         i--;
82       }
83   
84       // Ferment.
85       if (transformType == 11 || transformType == 10) {
86         int uppercaseOffset = offset - len;
87         if (transformType == 10) {
88           len = 1;
89         }
90         while (len > 0) {
91           int tmp = dst[uppercaseOffset] & 0xFF;
92           if (tmp < 0xc0) {
93             if (tmp >= 97 && tmp <= 122) { // in [a..z] range
94               dst[uppercaseOffset] ^= (byte) 32;
95             }
96             uppercaseOffset += 1;
97             len -= 1;
98           } else if (tmp < 0xe0) {
99             dst[uppercaseOffset + 1] ^= (byte) 32;
100             uppercaseOffset += 2;
101             len -= 2;
102           } else {
103             dst[uppercaseOffset + 2] ^= (byte) 5;
104             uppercaseOffset += 3;
105             len -= 3;
106           }
107         }
108       }
109   
110       // Copy suffix.
111       while (PREFIX_SUFFIX[transformSuffix] != 0) {
112         dst[offset++] = PREFIX_SUFFIX[transformSuffix++];
113       }
114   
115       return offset - dstOffset;
116     }
117   }
