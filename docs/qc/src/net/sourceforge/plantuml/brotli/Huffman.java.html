9   package net.sourceforge.plantuml.brotli;
10   
11   /**
12    * Utilities for building Huffman decoding tables.
13    */
14   final class Huffman {
15   
16     private static final int MAX_LENGTH = 15;
17   
18     /**
19      * Returns reverse(reverse(key, len) + 1, len).
20      *
21      * <p> reverse(key, len) is the bit-wise reversal of the len least significant bits of key.
22      */
23     private static int getNextKey(int key, int len) {
24       int step = 1 << (len - 1);
25       while ((key & step) != 0) {
26         step >>= 1;
27       }
28       return (key & (step - 1)) + step;
29     }
30   
31     /**
32      * Stores {@code item} in {@code table[0], table[step], table[2 * step] .., table[end]}.
33      *
34      * <p> Assumes that end is an integer multiple of step.
35      */
36     private static void replicateValue(int[] table, int offset, int step, int end, int item) {
37       do {
38         end -= step;
39         table[offset + end] = item;
40       } while (end > 0);
41     }
42   
43     /**
44      * @param count histogram of bit lengths for the remaining symbols,
45      * @param len code length of the next processed symbol.
46      * @return table width of the next 2nd level table.
47      */
48     private static int nextTableBitSize(int[] count, int len, int rootBits) {
49       int left = 1 << (len - rootBits);
50       while (len < MAX_LENGTH) {
51         left -= count[len];
52         if (left <= 0) {
53           break;
54         }
55         len++;
56         left <<= 1;
57       }
58       return len - rootBits;
59     }
60   
61     /**
62      * Builds Huffman lookup table assuming code lengths are in symbol order.
63      */
64     static void buildHuffmanTable(int[] rootTable, int tableOffset, int rootBits, int[] codeLengths,
65         int codeLengthsSize) {
66       int key; // Reversed prefix code.
67       int[] sorted = new int[codeLengthsSize]; // Symbols sorted by code length.
68       // TODO: fill with zeroes?
69       int[] count = new int[MAX_LENGTH + 1]; // Number of codes of each length.
70       int[] offset = new int[MAX_LENGTH + 1]; // Offsets in sorted table for each length.
71       int symbol;
72   
73       // Build histogram of code lengths.
74       for (symbol = 0; symbol < codeLengthsSize; symbol++) {
75         count[codeLengths[symbol]]++;
76       }
77   
78       // Generate offsets into sorted symbol table by code length.
79       offset[1] = 0;
80       for (int len = 1; len < MAX_LENGTH; len++) {
81         offset[len + 1] = offset[len] + count[len];
82       }
83   
84       // Sort symbols by length, by symbol order within each length.
85       for (symbol = 0; symbol < codeLengthsSize; symbol++) {
86         if (codeLengths[symbol] != 0) {
87           sorted[offset[codeLengths[symbol]]++] = symbol;
88         }
89       }
90   
91       int tableBits = rootBits;
92       int tableSize = 1 << tableBits;
93       int totalSize = tableSize;
94   
95       // Special case code with only one value.
96       if (offset[MAX_LENGTH] == 1) {
97         for (key = 0; key < totalSize; key++) {
98           rootTable[tableOffset + key] = sorted[0];
99         }
100         return;
101       }
102   
103       // Fill in root table.
104       key = 0;
105       symbol = 0;
106       for (int len = 1, step = 2; len <= rootBits; len++, step <<= 1) {
107         for (; count[len] > 0; count[len]--) {
108           replicateValue(rootTable, tableOffset + key, step, tableSize, len << 16 | sorted[symbol++]);
109           key = getNextKey(key, len);
110         }
111       }
112   
113       // Fill in 2nd level tables and add pointers to root table.
114       int mask = totalSize - 1;
115       int low = -1;
116       int currentOffset = tableOffset;
117       for (int len = rootBits + 1, step = 2; len <= MAX_LENGTH; len++, step <<= 1) {
118         for (; count[len] > 0; count[len]--) {
119           if ((key & mask) != low) {
120             currentOffset += tableSize;
121             tableBits = nextTableBitSize(count, len, rootBits);
122             tableSize = 1 << tableBits;
123             totalSize += tableSize;
124             low = key & mask;
125             rootTable[tableOffset + low] =
126                 (tableBits + rootBits) << 16 | (currentOffset - tableOffset - low);
127           }
128           replicateValue(rootTable, currentOffset + (key >> rootBits), step, tableSize,
129               (len - rootBits) << 16 | sorted[symbol++]);
130           key = getNextKey(key, len);
131         }
132       }
133     }
134   }
