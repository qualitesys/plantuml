9   package net.sourceforge.plantuml.brotli;
10   
11   import java.nio.ByteBuffer;
12   
13   /**
14    * Collection of static dictionary words.
15    *
16    * <p>Dictionary content is loaded from binary resource when {@link #getData()} is executed for the
17    * first time. Consequently, it saves memory and CPU in case dictionary is not required.
18    *
19    * <p>One possible drawback is that multiple threads that need dictionary data may be blocked (only
20    * once in each classworld). To avoid this, it is enough to call {@link #getData()} proactively.
21    */
22   public final class Dictionary {
23     private static volatile ByteBuffer data;
24   
25     private static class DataLoader {
26       static final boolean OK;
27   
28       static {
29         boolean ok = true;
30         try {
31           Class.forName(Dictionary.class.getPackage().getName() + ".DictionaryData");
32         } catch (Throwable ex) {
33           ok = false;
34         }
35         OK = ok;
36       }
37     }
38   
39     public static void setData(ByteBuffer data) {
40       if (!data.isDirect() || !data.isReadOnly()) {
41         throw new BrotliRuntimeException("data must be a direct read-only byte buffer");
42       }
43       Dictionary.data = data;
44     }
45   
46     public static ByteBuffer getData() {
47       if (data != null) {
48         return data;
49       }
50       if (!DataLoader.OK) {
51         throw new BrotliRuntimeException("brotli dictionary is not set");
52       }
53       /* Might have been set when {@link DictionaryData} was loaded.*/
54       return data;
55     }
56   }
