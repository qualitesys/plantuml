128     private static int decodeWindowBits(State s) {
129       BitReader.fillBitWindow(s);
130       if (BitReader.readFewBits(s, 1) == 0) {
131         return 16;
132       }
133       int n = BitReader.readFewBits(s, 3);
134       if (n != 0) {
135         return 17 + n;
136       }
137       n = BitReader.readFewBits(s, 3);
138       if (n != 0) {
139         return 8 + n;
140       }
141       return 17;
142     }

146      *
147      * @param s uninitialized state without associated input
148      * @param input compressed data source
149      */
150     static void initState(State s, InputStream input) {
151       if (s.runningState != UNINITIALIZED) {
152         throw new IllegalStateException("State MUST be uninitialized");
153       }

159         throw new BrotliRuntimeException("Invalid 'windowBits' code");
160       }
161       s.maxRingBufferSize = 1 << windowBits;
162       s.maxBackwardDistance = s.maxRingBufferSize - 16;
163       s.runningState = BLOCK_START;
164     }

183     private static int decodeVarLenUnsignedByte(State s) {
184       BitReader.fillBitWindow(s);
185       if (BitReader.readFewBits(s, 1) != 0) {
186         int n = BitReader.readFewBits(s, 3);
187         if (n == 0) {
188           return 1;
189         } else {
190           return BitReader.readFewBits(s, n) + (1 << n);
191         }
192       }
193       return 0;
194     }

196     private static void decodeMetaBlockLength(State s) {
197       BitReader.fillBitWindow(s);
198       s.inputEnd = BitReader.readFewBits(s, 1);
199       s.metaBlockLength = 0;
200       s.isUncompressed = 0;
201       s.isMetadata = 0;
202       if ((s.inputEnd != 0) && BitReader.readFewBits(s, 1) != 0) {
203         return;
204       }
205       int sizeNibbles = BitReader.readFewBits(s, 2) + 4;
206       if (sizeNibbles == 7) {
207         s.isMetadata = 1;
208         if (BitReader.readFewBits(s, 1) != 0) {
209           throw new BrotliRuntimeException("Corrupted reserved bit");
210         }
211         int sizeBytes = BitReader.readFewBits(s, 2);
212         if (sizeBytes == 0) {
213           return;
214         }
215         for (int i = 0; i < sizeBytes; i++) {
216           BitReader.fillBitWindow(s);
217           int bits = BitReader.readFewBits(s, 8);
218           if (bits == 0 && i + 1 == sizeBytes && sizeBytes > 1) {
219             throw new BrotliRuntimeException("Exuberant nibble");
220           }
221           s.metaBlockLength |= bits << (i * 8);
222         }
223       } else {
224         for (int i = 0; i < sizeNibbles; i++) {
225           BitReader.fillBitWindow(s);
226           int bits = BitReader.readFewBits(s, 4);
227           if (bits == 0 && i + 1 == sizeNibbles && sizeNibbles > 4) {
228             throw new BrotliRuntimeException("Exuberant nibble");
229           }
230           s.metaBlockLength |= bits << (i * 4);
231         }
232       }
233       s.metaBlockLength++;
234       if (s.inputEnd == 0) {
235         s.isUncompressed = BitReader.readFewBits(s, 1);
236       }
237     }

239     /**
240      * Decodes the next Huffman code from bit-stream.
241      */
242     private static int readSymbol(int[] table, int offset, State s) {
243       int val = BitReader.peekBits(s);
244       offset += val & HUFFMAN_TABLE_MASK;
245       int bits = table[offset] >> 16;
246       int sym = table[offset] & 0xFFFF;
247       if (bits <= HUFFMAN_TABLE_BITS) {
248         s.bitOffset += bits;
249         return sym;
250       }
251       offset += sym;
252       int mask = (1 << bits) - 1;
253       offset += (val & mask) >>> HUFFMAN_TABLE_BITS;
254       s.bitOffset += ((table[offset] >> 16) + HUFFMAN_TABLE_BITS);
255       return table[offset] & 0xFFFF;
256     }

258     private static int readBlockLength(int[] table, int offset, State s) {
259       BitReader.fillBitWindow(s);
260       int code = readSymbol(table, offset, s);
261       int n = BLOCK_LENGTH_N_BITS[code];
262       BitReader.fillBitWindow(s);
263       return BLOCK_LENGTH_OFFSET[code] + BitReader.readBits(s, n);
264     }

275     private static void moveToFront(int[] v, int index) {
276       int value = v[index];
277       for (; index > 0; index--) {
278         v[index] = v[index - 1];
279       }
280       v[0] = value;
281     }

283     private static void inverseMoveToFrontTransform(byte[] v, int vLen) {
284       int[] mtf = new int[256];
285       for (int i = 0; i < 256; i++) {
286         mtf[i] = i;
287       }
288       for (int i = 0; i < vLen; i++) {
289         int index = v[i] & 0xFF;
290         v[i] = (byte) mtf[index];
291         if (index != 0) {
292           moveToFront(mtf, index);
293         }
294       }

297     private static void readHuffmanCodeLengths(
298         int[] codeLengthCodeLengths, int numSymbols, int[] codeLengths, State s) {
299       int symbol = 0;
300       int prevCodeLen = DEFAULT_CODE_LENGTH;
301       int repeat = 0;
302       int repeatCodeLen = 0;
303       int space = 32768;
304       int[] table = new int[32];
305   
306       Huffman.buildHuffmanTable(table, 0, 5, codeLengthCodeLengths, CODE_LENGTH_CODES);
307   
308       while (symbol < numSymbols && space > 0) {
309         BitReader.readMoreInput(s);
310         BitReader.fillBitWindow(s);
311         int p = BitReader.peekBits(s) & 31;
312         s.bitOffset += table[p] >> 16;
313         int codeLen = table[p] & 0xFFFF;
314         if (codeLen < CODE_LENGTH_REPEAT_CODE) {
315           repeat = 0;
316           codeLengths[symbol++] = codeLen;
317           if (codeLen != 0) {
318             prevCodeLen = codeLen;
319             space -= 32768 >> codeLen;
320           }
321         } else {
322           int extraBits = codeLen - 14;
323           int newLen = 0;
324           if (codeLen == CODE_LENGTH_REPEAT_CODE) {
325             newLen = prevCodeLen;
326           }
327           if (repeatCodeLen != newLen) {
328             repeat = 0;
329             repeatCodeLen = newLen;
330           }
331           int oldRepeat = repeat;
332           if (repeat > 0) {
333             repeat -= 2;
334             repeat <<= extraBits;
335           }
336           BitReader.fillBitWindow(s);
337           repeat += BitReader.readFewBits(s, extraBits) + 3;
338           int repeatDelta = repeat - oldRepeat;
339           if (symbol + repeatDelta > numSymbols) {
340             throw new BrotliRuntimeException("symbol + repeatDelta > numSymbols"); // COV_NF_LINE
341           }
342           for (int i = 0; i < repeatDelta; i++) {
343             codeLengths[symbol++] = repeatCodeLen;
344           }
345           if (repeatCodeLen != 0) {
346             space -= repeatDelta << (15 - repeatCodeLen);
347           }
348         }
349       }
350       if (space != 0) {
351         throw new BrotliRuntimeException("Unused space"); // COV_NF_LINE
352       }
353       // TODO: Pass max_symbol to Huffman table builder instead?
354       Utils.fillIntsWithZeroes(codeLengths, symbol, numSymbols);
355     }

358       for (int i = 0; i < length - 1; ++i) {
359         for (int j = i + 1; j < length; ++j) {
360           if (symbols[i] == symbols[j]) {

362           }
363         }
364       }
365       return 1;
366     }

368     // TODO: Use specialized versions for smaller tables.
369     static void readHuffmanCode(int alphabetSize, int[] table, int offset, State s) {
370       int ok = 1;
371       int simpleCodeOrSkip;
372       BitReader.readMoreInput(s);
373       // TODO: Avoid allocation.
374       int[] codeLengths = new int[alphabetSize];
375       BitReader.fillBitWindow(s);
376       simpleCodeOrSkip = BitReader.readFewBits(s, 2);
377       if (simpleCodeOrSkip == 1) { // Read symbols, codes & code lengths directly.
378         int maxBitsCounter = alphabetSize - 1;
379         int maxBits = 0;
380         int[] symbols = new int[4];
381         int numSymbols = BitReader.readFewBits(s, 2) + 1;
382         while (maxBitsCounter != 0) {
383           maxBitsCounter >>= 1;
384           maxBits++;
385         }
386         // TODO: uncomment when codeLengths is reused.
387         // Utils.fillWithZeroes(codeLengths, 0, alphabetSize);
388         for (int i = 0; i < numSymbols; i++) {
389           BitReader.fillBitWindow(s);
390           symbols[i] = BitReader.readFewBits(s, maxBits) % alphabetSize;
391           codeLengths[symbols[i]] = 2;
392         }
393         codeLengths[symbols[0]] = 1;
394         switch (numSymbols) {
395           case 2:
396             codeLengths[symbols[1]] = 1;
397             break;
398           case 4:
399             if (BitReader.readFewBits(s, 1) == 1) {
400               codeLengths[symbols[2]] = 3;
401               codeLengths[symbols[3]] = 3;
402             } else {
403               codeLengths[symbols[0]] = 2;
404             }
405             break;
406           default:
407             break;
408         }
409         ok = checkDupes(symbols, numSymbols);
410       } else { // Decode Huffman-coded code lengths.
411         int[] codeLengthCodeLengths = new int[CODE_LENGTH_CODES];
412         int space = 32;
413         int numCodes = 0;
414         for (int i = simpleCodeOrSkip; i < CODE_LENGTH_CODES && space > 0; i++) {
415           int codeLenIdx = CODE_LENGTH_CODE_ORDER[i];
416           BitReader.fillBitWindow(s);
417           int p = BitReader.peekBits(s) & 15;
418           // TODO: Demultiplex FIXED_TABLE.
419           s.bitOffset += FIXED_TABLE[p] >> 16;
420           int v = FIXED_TABLE[p] & 0xFFFF;
421           codeLengthCodeLengths[codeLenIdx] = v;
422           if (v != 0) {
423             space -= (32 >> v);
424             numCodes++;
425           }
426         }
427         if (space != 0 && numCodes != 1) {
428           ok = 0;
429         }
430         readHuffmanCodeLengths(codeLengthCodeLengths, alphabetSize, codeLengths, s);
431       }
432       if (ok == 0) {
433         throw new BrotliRuntimeException("Can't readHuffmanCode"); // COV_NF_LINE
434       }
435       Huffman.buildHuffmanTable(table, offset, HUFFMAN_TABLE_BITS, codeLengths, alphabetSize);
436     }

453       int[] table = new int[HUFFMAN_TABLE_SIZE];
454       readHuffmanCode(numTrees + maxRunLengthPrefix, table, 0, s);
455       for (int i = 0; i < contextMapSize; ) {
456         BitReader.readMoreInput(s);
457         BitReader.fillBitWindow(s);
458         int code = readSymbol(table, 0, s);
459         if (code == 0) {
460           contextMap[i] = 0;
461           i++;
462         } else if (code <= maxRunLengthPrefix) {
463           BitReader.fillBitWindow(s);
464           int reps = (1 << code) + BitReader.readFewBits(s, code);
465           while (reps != 0) {
466             if (i >= contextMapSize) {
467               throw new BrotliRuntimeException("Corrupted context map"); // COV_NF_LINE
468             }
469             contextMap[i] = 0;
470             i++;
471             reps--;
472           }
473         } else {
474           contextMap[i] = (byte) (code - maxRunLengthPrefix);
475           i++;
476         }
477       }
478       BitReader.fillBitWindow(s);
479       if (BitReader.readFewBits(s, 1) == 1) {
480         inverseMoveToFrontTransform(contextMap, contextMapSize);
481       }
482       return numTrees;
483     }

485     private static int decodeBlockTypeAndLength(State s, int treeType, int numBlockTypes) {
486       final int[] ringBuffers = s.rings;

498       }

500         blockType -= numBlockTypes;
501       }
502       ringBuffers[offset] = ringBuffers[offset + 1];
503       ringBuffers[offset + 1] = blockType;
504       return result;
505     }

599       s.numLiteralBlockTypes = decodeVarLenUnsignedByte(s) + 1;
600       s.literalBlockLength = readMetablockPartition(s, 0, s.numLiteralBlockTypes);
601       s.numCommandBlockTypes = decodeVarLenUnsignedByte(s) + 1;
602       s.commandBlockLength = readMetablockPartition(s, 1, s.numCommandBlockTypes);
603       s.numDistanceBlockTypes = decodeVarLenUnsignedByte(s) + 1;

605   
606       BitReader.readMoreInput(s);
607       BitReader.fillBitWindow(s);
608       s.distancePostfixBits = BitReader.readFewBits(s, 2);
609       s.numDirectDistanceCodes =
610           NUM_DISTANCE_SHORT_CODES + (BitReader.readFewBits(s, 4) << s.distancePostfixBits);
611       s.distancePostfixMask = (1 << s.distancePostfixBits) - 1;
612       int numDistanceCodes = s.numDirectDistanceCodes + (48 << s.distancePostfixBits);
613       // TODO: Reuse?

615       for (int i = 0; i < s.numLiteralBlockTypes;) {
616         /* Ensure that less than 256 bits read between readMoreInput. */
617         int limit = Math.min(i + 96, s.numLiteralBlockTypes);
618         for (; i < limit; ++i) {
619           BitReader.fillBitWindow(s);
620           s.contextModes[i] = (byte) (BitReader.readFewBits(s, 2));
621         }
622         BitReader.readMoreInput(s);
623       }
624   
625       // TODO: Reuse?
626       s.contextMap = new byte[s.numLiteralBlockTypes << LITERAL_CONTEXT_BITS];
627       int numLiteralTrees = decodeContextMap(s.numLiteralBlockTypes << LITERAL_CONTEXT_BITS,
628           s.contextMap, s);
629       s.trivialLiteralContext = 1;
630       for (int j = 0; j < s.numLiteralBlockTypes << LITERAL_CONTEXT_BITS; j++) {
631         if (s.contextMap[j] != j >> LITERAL_CONTEXT_BITS) {
632           s.trivialLiteralContext = 0;
633           break;
634         }
635       }
636   
637       // TODO: Reuse?
638       s.distContextMap = new byte[s.numDistanceBlockTypes << DISTANCE_CONTEXT_BITS];
639       int numDistTrees = decodeContextMap(s.numDistanceBlockTypes << DISTANCE_CONTEXT_BITS,
640           s.distContextMap, s);
641   
642       s.hGroup0 = decodeHuffmanTreeGroup(NUM_LITERAL_CODES, numLiteralTrees, s);
643       s.hGroup1 =
644           decodeHuffmanTreeGroup(NUM_INSERT_AND_COPY_CODES, s.numCommandBlockTypes, s);
645       s.hGroup2 = decodeHuffmanTreeGroup(numDistanceCodes, numDistTrees, s);
646   
647       s.contextMapSlice = 0;
648       s.distContextMapSlice = 0;
649       s.contextLookupOffset1 = (int) (s.contextModes[0]) << 9;
650       s.contextLookupOffset2 = s.contextLookupOffset1 + 256;
651       s.literalTreeIndex = 0;
652       s.literalTree = s.hGroup0[0];
653       s.treeCommandOffset = s.hGroup1[0];
654   
655       s.rings[4] = 1;
656       s.rings[5] = 0;
657       s.rings[6] = 1;
658       s.rings[7] = 0;
659       s.rings[8] = 1;
660       s.rings[9] = 0;
661     }

717     /**
718      * Actual decompress implementation.
719      */
720     static void decompress(State s) {
721       if (s.runningState == UNINITIALIZED) {
722         throw new IllegalStateException("Can't decompress until initialized");
723       }
724       if (s.runningState == CLOSED) {
725         throw new IllegalStateException("Can't decompress after close");
726       }
727       int ringBufferMask = s.ringBufferSize - 1;
728       byte[] ringBuffer = s.ringBuffer;
729   
730       while (s.runningState != FINISHED) {
731         // TODO: extract cases to methods for the better readability.
732         switch (s.runningState) {
733           case BLOCK_START:
734             if (s.metaBlockLength < 0) {
735               throw new BrotliRuntimeException("Invalid metablock length");
736             }
737             readNextMetablockHeader(s);
738             /* Ring-buffer would be reallocated here. */
739             ringBufferMask = s.ringBufferSize - 1;
740             ringBuffer = s.ringBuffer;
741             continue;
742   
743           case COMPRESSED_BLOCK_START:
744             readMetablockHuffmanCodesAndContextMaps(s);
745             s.runningState = MAIN_LOOP;
746             // Fall through
747   
748           case MAIN_LOOP:
749             if (s.metaBlockLength <= 0) {
750               s.runningState = BLOCK_START;
751               continue;
752             }
753             BitReader.readMoreInput(s);
754             if (s.commandBlockLength == 0) {
755               decodeCommandBlockSwitch(s);
756             }
757             s.commandBlockLength--;
758             BitReader.fillBitWindow(s);
759             int cmdCode = readSymbol(s.hGroup1, s.treeCommandOffset, s);
760             int rangeIdx = cmdCode >>> 6;
761             s.distanceCode = 0;
762             if (rangeIdx >= 2) {
763               rangeIdx -= 2;
764               s.distanceCode = -1;
765             }
766             int insertCode = INSERT_RANGE_LUT[rangeIdx] + ((cmdCode >>> 3) & 7);
767             BitReader.fillBitWindow(s);
768             int insertBits = INSERT_LENGTH_N_BITS[insertCode];
769             int insertExtra = BitReader.readBits(s, insertBits);
770             s.insertLength = INSERT_LENGTH_OFFSET[insertCode] + insertExtra;
771             int copyCode = COPY_RANGE_LUT[rangeIdx] + (cmdCode & 7);
772             BitReader.fillBitWindow(s);
773             int copyBits = COPY_LENGTH_N_BITS[copyCode];
774             int copyExtra = BitReader.readBits(s, copyBits);
775             s.copyLength = COPY_LENGTH_OFFSET[copyCode] + copyExtra;
776   
777             s.j = 0;
778             s.runningState = INSERT_LOOP;
779   
780             // Fall through
781           case INSERT_LOOP:
782             if (s.trivialLiteralContext != 0) {
783               while (s.j < s.insertLength) {
784                 BitReader.readMoreInput(s);
785                 if (s.literalBlockLength == 0) {
786                   decodeLiteralBlockSwitch(s);
787                 }
788                 s.literalBlockLength--;
789                 BitReader.fillBitWindow(s);
790                 ringBuffer[s.pos] =
791                     (byte) readSymbol(s.hGroup0, s.literalTree, s);
792                 s.j++;
793                 if (s.pos++ == ringBufferMask) {
794                   s.nextRunningState = INSERT_LOOP;
795                   s.bytesToWrite = s.ringBufferSize;
796                   s.bytesWritten = 0;
797                   s.runningState = WRITE;
798                   break;
799                 }
800               }
801             } else {
802               int prevByte1 = ringBuffer[(s.pos - 1) & ringBufferMask] & 0xFF;
803               int prevByte2 = ringBuffer[(s.pos - 2) & ringBufferMask] & 0xFF;
804               while (s.j < s.insertLength) {
805                 BitReader.readMoreInput(s);
806                 if (s.literalBlockLength == 0) {
807                   decodeLiteralBlockSwitch(s);
808                 }
809                 int literalTreeIndex = s.contextMap[s.contextMapSlice
810                   + (Context.LOOKUP[s.contextLookupOffset1 + prevByte1]
811                       | Context.LOOKUP[s.contextLookupOffset2 + prevByte2])] & 0xFF;
812                 s.literalBlockLength--;
813                 prevByte2 = prevByte1;
814                 BitReader.fillBitWindow(s);
815                 prevByte1 = readSymbol(
816                     s.hGroup0, s.hGroup0[literalTreeIndex], s);
817                 ringBuffer[s.pos] = (byte) prevByte1;
818                 s.j++;
819                 if (s.pos++ == ringBufferMask) {
820                   s.nextRunningState = INSERT_LOOP;
821                   s.bytesToWrite = s.ringBufferSize;
822                   s.bytesWritten = 0;
823                   s.runningState = WRITE;
824                   break;
825                 }
826               }
827             }
828             if (s.runningState != INSERT_LOOP) {
829               continue;
830             }
831             s.metaBlockLength -= s.insertLength;
832             if (s.metaBlockLength <= 0) {
833               s.runningState = MAIN_LOOP;
834               continue;
835             }
836             if (s.distanceCode < 0) {
837               BitReader.readMoreInput(s);
838               if (s.distanceBlockLength == 0) {
839                 decodeDistanceBlockSwitch(s);
840               }
841               s.distanceBlockLength--;
842               BitReader.fillBitWindow(s);
843               s.distanceCode = readSymbol(s.hGroup2, s.hGroup2[
844                   s.distContextMap[s.distContextMapSlice
845                       + (s.copyLength > 4 ? 3 : s.copyLength - 2)] & 0xFF], s);
846               if (s.distanceCode >= s.numDirectDistanceCodes) {
847                 s.distanceCode -= s.numDirectDistanceCodes;
848                 int postfix = s.distanceCode & s.distancePostfixMask;
849                 s.distanceCode >>>= s.distancePostfixBits;
850                 int n = (s.distanceCode >>> 1) + 1;
851                 int offset = ((2 + (s.distanceCode & 1)) << n) - 4;
852                 BitReader.fillBitWindow(s);
853                 int distanceExtra = BitReader.readBits(s, n);
854                 s.distanceCode = s.numDirectDistanceCodes + postfix
855                     + ((offset + distanceExtra) << s.distancePostfixBits);
856               }
857             }
858   
859             // Convert the distance code to the actual distance by possibly looking up past distances
860             // from the ringBuffer.
861             s.distance = translateShortCodes(s.distanceCode, s.rings, s.distRbIdx);
862             if (s.distance < 0) {
863               throw new BrotliRuntimeException("Negative distance"); // COV_NF_LINE
864             }
865   
866             if (s.maxDistance != s.maxBackwardDistance
867                 && s.pos < s.maxBackwardDistance) {
868               s.maxDistance = s.pos;
869             } else {
870               s.maxDistance = s.maxBackwardDistance;
871             }
872   
873             s.copyDst = s.pos;
874             if (s.distance > s.maxDistance) {
875               s.runningState = TRANSFORM;
876               continue;
877             }
878   
879             if (s.distanceCode > 0) {
880               s.rings[s.distRbIdx & 3] = s.distance;
881               s.distRbIdx++;
882             }
883   
884             if (s.copyLength > s.metaBlockLength) {
885               throw new BrotliRuntimeException("Invalid backward reference"); // COV_NF_LINE
886             }
887             s.j = 0;
888             s.runningState = COPY_LOOP;
889             // fall through
890           case COPY_LOOP:
891             int src = (s.pos - s.distance) & ringBufferMask;
892             int dst = s.pos;
893             int copyLength = s.copyLength - s.j;
894             int srcEnd = src + copyLength;
895             int dstEnd = dst + copyLength;
896             if ((srcEnd < ringBufferMask) && (dstEnd < ringBufferMask)) {
897               if (copyLength < 12 || (srcEnd > dst && dstEnd > src)) {
898                 for (int k = 0; k < copyLength; ++k) {
899                   ringBuffer[dst++] = ringBuffer[src++];
900                 }
901               } else {
902                 Utils.copyBytesWithin(ringBuffer, dst, src, srcEnd);
903               }
904               s.j += copyLength;
905               s.metaBlockLength -= copyLength;
906               s.pos += copyLength;
907             } else {
908               for (; s.j < s.copyLength;) {
909                 ringBuffer[s.pos] =
910                     ringBuffer[(s.pos - s.distance) & ringBufferMask];
911                 s.metaBlockLength--;
912                 s.j++;
913                 if (s.pos++ == ringBufferMask) {
914                   s.nextRunningState = COPY_LOOP;
915                   s.bytesToWrite = s.ringBufferSize;
916                   s.bytesWritten = 0;
917                   s.runningState = WRITE;
918                   break;
919                 }
920               }
921             }
922             if (s.runningState == COPY_LOOP) {
923               s.runningState = MAIN_LOOP;
924             }
925             continue;
926   
927           case TRANSFORM:
928             if (s.copyLength >= MIN_WORD_LENGTH
929                 && s.copyLength <= MAX_WORD_LENGTH) {
930               int offset = DICTIONARY_OFFSETS_BY_LENGTH[s.copyLength];
931               int wordId = s.distance - s.maxDistance - 1;
932               int shift = DICTIONARY_SIZE_BITS_BY_LENGTH[s.copyLength];
933               int mask = (1 << shift) - 1;
934               int wordIdx = wordId & mask;
935               int transformIdx = wordId >>> shift;
936               offset += wordIdx * s.copyLength;
937               if (transformIdx < Transform.NUM_TRANSFORMS) {
938                 int len = Transform.transformDictionaryWord(ringBuffer, s.copyDst,
939                     Dictionary.getData(), offset, s.copyLength, transformIdx);
940                 s.copyDst += len;
941                 s.pos += len;
942                 s.metaBlockLength -= len;
943                 if (s.copyDst >= s.ringBufferSize) {
944                   s.nextRunningState = COPY_WRAP_BUFFER;
945                   s.bytesToWrite = s.ringBufferSize;
946                   s.bytesWritten = 0;
947                   s.runningState = WRITE;
948                   continue;
949                 }
950               } else {
951                 throw new BrotliRuntimeException("Invalid backward reference"); // COV_NF_LINE
952               }
953             } else {
954               throw new BrotliRuntimeException("Invalid backward reference"); // COV_NF_LINE
955             }
956             s.runningState = MAIN_LOOP;
957             continue;
958   
959           case COPY_WRAP_BUFFER:
960             Utils.copyBytesWithin(ringBuffer, 0, s.ringBufferSize, s.copyDst);
961             s.runningState = MAIN_LOOP;
962             continue;
963   
964           case READ_METADATA:
965             while (s.metaBlockLength > 0) {
966               BitReader.readMoreInput(s);
967               // Optimize
968               BitReader.fillBitWindow(s);
969               BitReader.readFewBits(s, 8);
970               s.metaBlockLength--;
971             }
972             s.runningState = BLOCK_START;
973             continue;
974   
975   
976           case COPY_UNCOMPRESSED:
977             copyUncompressedData(s);
978             continue;
979   
980           case WRITE:
981             if (writeRingBuffer(s) == 0) {
982               // Output buffer is full.
983               return;
984             }
985             if (s.pos >= s.maxBackwardDistance) {
986               s.maxDistance = s.maxBackwardDistance;
987             }
988             s.pos &= ringBufferMask;
989             s.runningState = s.nextRunningState;
990             continue;
991   
992           default:
993             throw new BrotliRuntimeException("Unexpected state " + s.runningState);
994         }
995       }
996       if (s.runningState == FINISHED) {
997         if (s.metaBlockLength < 0) {
998           throw new BrotliRuntimeException("Invalid metablock length");
999         }
1000         BitReader.jumpToByteBoundary(s);
1001         BitReader.checkHealth(s, 1);
1002       }
1003     }
1004   }
