11   /**
12    * Common context lookup table for all context modes.
13    */
14   final class Context {
15   
16     static final int[] LOOKUP = new int[2048];
17   
18     private static final String UTF_MAP = "         !!  !                  \"#$##%#$&'##(#)#+++++++++"
19         + "+((&*'##,---,---,-----,-----,-----&#'###.///.///./////./////./////&#'# ";
20     private static final String UTF_RLE = "A/*  ':  & : $  \u0081 @";
21   
22     private static void unpackLookupTable(int[] lookup, String map, String rle) {
23       // LSB6, MSB6, SIGNED
24       for (int i = 0; i < 256; ++i) {
25         lookup[i] = i & 0x3F;
26         lookup[512 + i] = i >> 2;
27         lookup[1792 + i] = 2 + (i >> 6);
28       }
29       // UTF8
30       for (int i = 0; i < 128; ++i) {
31         lookup[1024 + i] = 4 * (map.charAt(i) - 32);
32       }
33       for (int i = 0; i < 64; ++i) {
34         lookup[1152 + i] = i & 1;
35         lookup[1216 + i] = 2 + (i & 1);
36       }
37       int offset = 1280;
38       for (int k = 0; k < 19; ++k) {
39         int value = k & 3;
40         int rep = rle.charAt(k) - 32;
41         for (int i = 0; i < rep; ++i) {
42           lookup[offset++] = value;
43         }
44       }
45       // SIGNED
46       for (int i = 0; i < 16; ++i) {
47         lookup[1792 + i] = 1;
48         lookup[2032 + i] = 6;
49       }
50       lookup[1792] = 0;
51       lookup[2047] = 7;
52       for (int i = 0; i < 256; ++i) {
53         lookup[1536 + i] = lookup[1792 + i] << 3;
54       }
55     }
56   
57     static {
58       unpackLookupTable(LOOKUP, UTF_MAP, UTF_RLE);
59     }
60   }
