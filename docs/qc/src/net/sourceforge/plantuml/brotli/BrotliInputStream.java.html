9   package net.sourceforge.plantuml.brotli;
10   
11   import java.io.IOException;
12   import java.io.InputStream;
13   
14   /**
15    * {@link InputStream} decorator that decompresses brotli data.
16    *
17    * <p> Not thread-safe.
18    */
19   public class BrotliInputStream extends InputStream {
20   
21     public static final int DEFAULT_INTERNAL_BUFFER_SIZE = 16384;
22   
23     /**
24      * Internal buffer used for efficient byte-by-byte reading.
25      */
26     private byte[] buffer;
27   
28     /**
29      * Number of decoded but still unused bytes in internal buffer.
30      */
31     private int remainingBufferBytes;
32   
33     /**
34      * Next unused byte offset.
35      */
36     private int bufferOffset;
37   
38     /**
39      * Decoder state.
40      */
41     private final State state = new State();
42   
43     /**
44      * Creates a {@link InputStream} wrapper that decompresses brotli data.
45      *
46      * <p> For byte-by-byte reading ({@link #read()}) internal buffer with
47      * {@link #DEFAULT_INTERNAL_BUFFER_SIZE} size is allocated and used.
48      *
49      * <p> Will block the thread until first kilobyte of data of source is available.
50      *
51      * @param source underlying data source
52      * @throws IOException in case of corrupted data or source stream problems
53      */
54     public BrotliInputStream(InputStream source) throws IOException {
55       this(source, DEFAULT_INTERNAL_BUFFER_SIZE);
56     }
57   
58     /**
59      * Creates a {@link InputStream} wrapper that decompresses brotli data.
60      *
61      * <p> For byte-by-byte reading ({@link #read()}) internal buffer of specified size is
62      * allocated and used.
63      *
64      * <p> Will block the thread until first kilobyte of data of source is available.
65      *
66      * @param source compressed data source
67      * @param byteReadBufferSize size of internal buffer used in case of
68      *        byte-by-byte reading
69      * @throws IOException in case of corrupted data or source stream problems
70      */
71     public BrotliInputStream(InputStream source, int byteReadBufferSize) throws IOException {
72       if (byteReadBufferSize <= 0) {
73         throw new IllegalArgumentException("Bad buffer size:" + byteReadBufferSize);
74       } else if (source == null) {
75         throw new IllegalArgumentException("source is null");
76       }
77       this.buffer = new byte[byteReadBufferSize];
78       this.remainingBufferBytes = 0;
79       this.bufferOffset = 0;
80       try {
81         Decode.initState(state, source);
82       } catch (BrotliRuntimeException ex) {
83         throw new IOException("Brotli decoder initialization failed", ex);
84       }
85     }
86   
87     /**
88      * {@inheritDoc}
89      */
90     @Override
91     public void close() throws IOException {
92       Decode.close(state);
93     }
94   
95     /**
96      * {@inheritDoc}
97      */
98     @Override
99     public int read() throws IOException {
100       if (bufferOffset >= remainingBufferBytes) {
101         remainingBufferBytes = read(buffer, 0, buffer.length);
102         bufferOffset = 0;
103         if (remainingBufferBytes == -1) {
104           return -1;
105         }
106       }
107       return buffer[bufferOffset++] & 0xFF;
108     }
109   
110     /**
111      * {@inheritDoc}
112      */
113     @Override
114     public int read(byte[] destBuffer, int destOffset, int destLen) throws IOException {
115       if (destOffset < 0) {
116         throw new IllegalArgumentException("Bad offset: " + destOffset);
117       } else if (destLen < 0) {
118         throw new IllegalArgumentException("Bad length: " + destLen);
119       } else if (destOffset + destLen > destBuffer.length) {
120         throw new IllegalArgumentException(
121             "Buffer overflow: " + (destOffset + destLen) + " > " + destBuffer.length);
122       } else if (destLen == 0) {
123         return 0;
124       }
125       int copyLen = Math.max(remainingBufferBytes - bufferOffset, 0);
126       if (copyLen != 0) {
127         copyLen = Math.min(copyLen, destLen);
128         System.arraycopy(buffer, bufferOffset, destBuffer, destOffset, copyLen);
129         bufferOffset += copyLen;
130         destOffset += copyLen;
131         destLen -= copyLen;
132         if (destLen == 0) {
133           return copyLen;
134         }
135       }
136       try {
137         state.output = destBuffer;
138         state.outputOffset = destOffset;
139         state.outputLength = destLen;
140         state.outputUsed = 0;
141         Decode.decompress(state);
142         if (state.outputUsed == 0) {
143           return -1;
144         }
145         return state.outputUsed + copyLen;
146       } catch (BrotliRuntimeException ex) {
147         throw new IOException("Brotli stream decoding failed", ex);
148       }
149   
150       // <{[INJECTED CODE]}>
151     }
152   }
