70       public Tile createTile(Tile start, TileGeometry position) {
71           final Tile result = new Tile(size++);
72           final Position p = getFreePosition(start, position);
73           positions.put(result, p);
74           paths.add(buildPath(start.getArea(position), result.getArea(position.opposite())));
75           return result;
76       }

82       private Path buildPath(TileArea tileArea1, TileArea tileArea2) {
83           if (isAdjoining(tileArea1, tileArea2)) {
84               return Path.build(tileArea1, tileArea2);
85           }
86           final Tile tile1 = tileArea1.getTile();
87           final Tile tile2 = tileArea2.getTile();
88           final Position pos1 = getPosition(tile1);
89           final Position pos2 = getPosition(tile2);
90           final TileGeometry geom1 = tileArea1.getGeometry();
91           final TileGeometry geom2 = tileArea2.getGeometry();
92           if (pos1.getYmin() == pos2.getYmin() && pos1.getYmax() == pos2.getYmax() && geom1 == TileGeometry.WEST
93                   && geom2 == TileGeometry.EAST) {
94               return Path.build(tileArea1, tileArea2);
95   
96           }
97           throw new IllegalArgumentException();
98       }

100       private boolean isAdjoining(TileArea tileArea1, TileArea tileArea2) {
101           final Tile tile1 = tileArea1.getTile();
102           final Tile tile2 = tileArea2.getTile();
103           final Position pos1 = getPosition(tile1);
104           final Position pos2 = getPosition(tile2);
105           final TileGeometry geom1 = tileArea1.getGeometry();
106           final TileGeometry geom2 = tileArea2.getGeometry();
107           if (pos1.equals(pos2)) {
108               assert tile1 == tile2;
109               if (geom1 == geom2) {
110                   throw new IllegalArgumentException();
111               }
112               return true;
113           }
114           if (geom1.equals(geom2.opposite()) == false) {
115               return false;
116           }
117           switch (geom1) {
118           case EAST:
119               return pos1.getYmin() == pos2.getYmin() && pos1.getYmax() == pos2.getYmax()
120                       && pos1.getXmax() + 1 == pos2.getXmin();
121           case WEST:
122               return pos1.getYmin() == pos2.getYmin() && pos1.getYmax() == pos2.getYmax()
123                       && pos1.getXmin() == pos2.getXmax() + 1;
124           case SOUTH:
125               return pos1.getXmin() == pos2.getXmin() && pos1.getXmax() == pos2.getXmax()
126                       && pos1.getYmax() + 1 == pos2.getYmin();
127           case NORTH:
128               return pos1.getXmin() == pos2.getXmin() && pos1.getXmax() == pos2.getXmax()
129                       && pos1.getYmin() == pos2.getYmax() + 1;
130           case CENTER:
131               return false;
132   
133           default:
134               throw new IllegalStateException();
135           }
136   
137       }

139       private Tile getTileAt(Position p) {
140           for (Map.Entry<Tile, Position> ent : positions.entrySet()) {
141               if (p.equals(ent.getValue())) {
142                   return ent.getKey();
143               }
144           }
145           return null;
146       }

148       private Position getFreePosition(Tile start, TileGeometry position) {
149           final Position p = getPosition(start).move(position, 2);
150           while (isOccuped(p)) {
151               // p = p.move(TileGeometry.EAST, 2);
152               moveAllToEast(p);
153           }
154           return p;
155       }

157       private void moveAllToEast(Position startingPosition) {
158           final List<Position> toMove = new ArrayList<>();
159           for (Position p : positions.values()) {
160               if (p.getXmax() < startingPosition.getXmin()) {
161                   continue;
162               }
163               if (p.getYmax() < startingPosition.getYmin()) {
164                   continue;
165               }
166               toMove.add(p);
167           }
168           for (Position p : toMove) {
169               positions.put(getTileAt(p), p.move(TileGeometry.EAST, 2));
170           }
171   
172       }

174       private boolean isOccuped(Position test) {
175           for (Position p : positions.values()) {
176               if (p.equals(test)) {
177                   return true;
178               }
179           }
180           return false;
181       }

237       public void drawU(UGraphic ug) {
238           double x = 0;
239           double y = 0;
240           final int xmin = getXmin();
241           final int ymin = getYmin();
242           final XDimension2D dimSingle = root.calculateDimension(ug.getStringBounder());
243           x -= xmin * dimSingle.getWidth() / 2;
244           y -= ymin * dimSingle.getHeight() / 2;
245           for (Map.Entry<Tile, Position> ent : positions.entrySet()) {
246               final Position p = ent.getValue();
247               final Tile t = ent.getKey();
248               final double xt = p.getXmin() * dimSingle.getWidth() / 2;
249               final double yt = p.getYmin() * dimSingle.getHeight() / 2;
250               t.drawU(ug.apply(new UTranslate((x + xt), (y + yt))));
251           }
252           ug = ug.apply(HColors.RED);
253           for (Path p : paths) {
254               final TileArea start = p.getStart();
255               final TileArea dest = p.getDest();
256               final XPoint2D pstart = getPoint2D(dimSingle, start);
257               final XPoint2D pdest = getPoint2D(dimSingle, dest);
258               ug.apply(new UTranslate(x + pstart.getX(), y + pstart.getY())).draw(new ULine(pdest.getX() - pstart.getX(), pdest.getY()
259               - pstart.getY()));
260           }
261       }

263       private XPoint2D getPoint2D(XDimension2D dimSingle, TileArea area) {
264           final Position p = getPosition(area.getTile());
265           double xt = p.getXmin() * dimSingle.getWidth() / 2;
266           double yt = p.getYmin() * dimSingle.getHeight() / 2;
267           xt += dimSingle.getWidth() / 2;
268           yt += dimSingle.getHeight() / 2;
269           final double coef = 0.33;
270           switch (area.getGeometry()) {
271           case NORTH:
272               yt -= dimSingle.getHeight() * coef;
273               break;
274           case SOUTH:
275               yt += dimSingle.getHeight() * coef;
276               break;
277           case EAST:
278               xt += dimSingle.getWidth() * coef;
279               break;
280           case WEST:
281               xt -= dimSingle.getWidth() * coef;
282               break;
283           default:
284               throw new IllegalStateException();
285           }
286           return new XPoint2D(xt, yt);
287       }

289       public XDimension2D calculateDimension(StringBounder stringBounder) {
290           final int xmin = getXmin();
291           final int xmax = getXmax();
292           final int ymin = getYmin();
293           final int ymax = getYmax();
294           final int width = (xmax - xmin) / 2 + 1;
295           final int height = (ymax - ymin) / 2 + 1;
296           final XDimension2D dimSingle = root.calculateDimension(stringBounder);
297           return new XDimension2D(width * dimSingle.getWidth(), height * dimSingle.getHeight());
298       }
