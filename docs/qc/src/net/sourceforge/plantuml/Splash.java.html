57   public class Splash extends Window implements MouseListener, MouseMotionListener {
58   
59       private static final Color LINK_NORMAL = Color.BLUE;
60       private static final Color LINK_HOVER = new Color(127, 0, 127);
61   
62       private static Splash singleton;
63   
64       private final int width = 280;
65       private final int height = 80;
66       private final BufferedImage logo;
67       private final AtomicInteger total = new AtomicInteger();
68       private final AtomicInteger done = new AtomicInteger();
69       private final AtomicInteger errors = new AtomicInteger();
70       private final Font font = new Font("SansSerif", Font.BOLD, 12);
71   
72       public static void main(String[] args) throws Exception {
73           Splash.createSplash();
74           Splash.incTotal(30);
75           for (int i = 0; i < 20; i++) {
76               Splash.incDone(i > 3);
77               Thread.sleep(1000);
78           }
79       }
80   
81       private Splash() {
82           super(null);
83           this.logo = PSystemVersion.getPlantumlImage();
84           final Dimension dimension = Toolkit.getDefaultToolkit().getScreenSize();
85           final int x = (int) ((dimension.getWidth() - width)) / 2;
86           final int y = (int) ((dimension.getHeight() - height)) / 2;
87   
88           this.setBounds(x, y, width, height);
89           this.setVisible(true);
90           this.setAlwaysOnTop(true);
91           this.setFocusable(false);
92           this.setFocusableWindowState(false);
93   
94           addMouseListener(this);
95           addMouseMotionListener(this);
96       }
97   
98       private int xClicked;
99       private int yClicked;
100       private int limY;
101       private int limX;
102   
103       public void mouseDragged(MouseEvent event) {
104           int x = event.getXOnScreen();
105           int y = event.getYOnScreen();
106           this.setBounds(x - xClicked, y - yClicked, width, height);
107   
108       }
109   
110       private Color link = LINK_NORMAL;
111   
112       public void mouseMoved(MouseEvent event) {
113           if (event.getY() > limY && event.getX() < limX) {
114               updateLinkColor(LINK_HOVER);
115           } else {
116               updateLinkColor(LINK_NORMAL);
117           }
118       }
119   
120       private void updateLinkColor(final Color newLink) {
121           if (link != newLink) {
122               link = newLink;
123               this.setCursor(
124                       link == LINK_NORMAL ? Cursor.getDefaultCursor() : Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
125               repaint();
126           }
127       }
128   
129       public void mouseReleased(MouseEvent event) {
130           // System.err.println("mouseReleased " + event);
131       }
132   
133       public void mousePressed(MouseEvent event) {
134           this.xClicked = event.getX();
135           this.yClicked = event.getY();
136       }
137   
138       public void mouseClicked(MouseEvent event) {
139           if (link != LINK_NORMAL) {
140               try {
141                   Desktop.getDesktop().browse(new URL("https://plantuml.com").toURI());
142               } catch (Exception e) {
143                   Logme.error(e);
144               }
145               return;
146           }
147           this.xClicked = event.getX();
148           this.yClicked = event.getY();
149       }
150   
151       public void mouseExited(MouseEvent event) {
152           updateLinkColor(LINK_NORMAL);
153       }
154   
155       public void mouseEntered(MouseEvent event) {
156           // System.err.println("mouseEntered " + event);
157       }
158   
159       private void incTotalInternal(int nb) {
160           this.total.addAndGet(nb);
161           // invalidate();
162           repaint();
163           // System.err.println("INC TOTAL=" + done + "/" + total);
164       }
165   
166       private void incDoneInternal(boolean error) {
167           this.done.incrementAndGet();
168           if (error) {
169               this.errors.incrementAndGet();
170           }
171           // invalidate();
172           repaint();
173           // System.err.println("INC TOTAL=" + done + "/" + total);
174       }
175   
176       @Override
177       public void paint(Graphics g) {
178           super.paint(g);
179           g.setColor(Color.WHITE);
180           g.fillRect(0, 0, width, height);
181           g.setColor(Color.BLACK);
182           g.drawRect(0, 0, width - 1, height - 1);
183           g.drawRect(1, 1, width - 3, height - 3);
184           // g.setColor(Color.RED);
185           // final String status = done + "/" + total;
186           // g.drawString(status, width / 2, height / 2);
187           g.drawImage(logo, width - logo.getWidth() - 4, height - logo.getHeight() - 4, null);
188           drawProgessBar(g, done.intValue(), total.intValue());
189           final int nbErrors = errors.get();
190           if (nbErrors > 0) {
191               g.setColor(Color.RED);
192               final String message = "" + nbErrors + (nbErrors > 1 ? " diagrams" : " diagram") + " contains errors";
193               g.drawString(message, 10, 20);
194           }
195           g.setColor(link);
196           final String urllink = "http://plantuml.com";
197           final Rectangle2D rect = getUsed(g, urllink);
198           g.drawString(urllink, 10, (int) (height - rect.getMaxY()));
199           limY = (int) (height - rect.getMaxY() + rect.getMinY());
200           limX = (int) (10 + rect.getMaxX());
201   
202       }
203   
204       private void drawProgessBar(Graphics g, int intValue, int totalValue) {
205           if (totalValue == 0) {
206               return;
207           }
208           g.setFont(font);
209           final String label = "" + intValue + "/" + totalValue;
210           final Rectangle2D rect = getUsed(g, label);
211   
212           final int x = 10;
213           final int y = 33;
214           final int barWidth = 170;
215           final int barHeight = (int) (rect.getHeight() + 2);
216           final int gray = 230;
217           g.setColor(new Color(gray, gray, gray));
218           final int value = barWidth * intValue / totalValue;
219           g.fillRect(x, y, value, barHeight);
220           g.setColor(Color.BLACK);
221           g.drawRect(x, y, barWidth, barHeight);
222   
223           final double xLabel = x + (barWidth - rect.getWidth()) / 2;
224           final double yLabel = y - rect.getY() + (barHeight - rect.getHeight()) / 2;
225           g.drawString(label, (int) xLabel, (int) yLabel);
226       }
227   
228       private Rectangle2D getUsed(Graphics g, final String label) {
229           return g.getFontMetrics().getStringBounds(label, g);
230       }

232       // Static
233   
234       public synchronized static void createSplash() {
235           if (singleton == null) {
236               singleton = new Splash();
237           }
238       }
239   
240       public static void incTotal(int nb) {
241           if (singleton != null) {
242               singleton.incTotalInternal(nb);
243           }
244       }
245   
246       public static void incDone(boolean error) {
247           if (singleton != null) {
248               singleton.incDoneInternal(error);
249           }
250       }
251   
252       public synchronized static void disposeSplash() {
253           if (singleton != null) {
254               singleton.setVisible(false);
255               singleton.dispose();
256               singleton = null;
257           }
258       }
