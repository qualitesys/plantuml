64       public static List<FileImageData> exportDiagrams(Diagram system, SuggestedFile suggestedFile,
65               FileFormatOption fileFormatOption, boolean checkMetadata) throws IOException {
66   
67           final SFile existingFile = suggestedFile.getFile(0);
68           if (checkMetadata && fileFormatOption.getFileFormat().doesSupportMetadata() && existingFile.exists()
69                   && system.getNbImages() == 1) {
70               // final String version = Version.versionString();
71               // System.out.println(system.getMetadata());
72               // System.out.println(data);
73               // System.out.println(version);
74               // System.out.println(data.contains(version));
75               final boolean sameMetadata = fileFormatOption.getFileFormat().equalsMetadata(system.getMetadata(),
76                       existingFile);
77               if (sameMetadata) {
78                   Log.info("Skipping " + existingFile.getPrintablePath() + " because metadata has not changed.");
79                   return Arrays.asList(new FileImageData(existingFile, null));
80               }
81           }
82   
83           if (system instanceof NewpagedDiagram) {
84               return exportDiagramsNewpaged((NewpagedDiagram) system, suggestedFile, fileFormatOption);
85           }
86           if (system instanceof SequenceDiagram) {
87               return exportDiagramsSequence((SequenceDiagram) system, suggestedFile, fileFormatOption);
88           }
89           if (system instanceof CucaDiagram && fileFormatOption.getFileFormat() == FileFormat.HTML) {
90               return createFilesHtml((CucaDiagram) system, suggestedFile);
91           }
92   
93           return exportDiagramsDefault(system, suggestedFile, fileFormatOption);
94       }

96       private static List<FileImageData> exportDiagramsNewpaged(NewpagedDiagram system, SuggestedFile suggestedFile,
97               FileFormatOption fileFormat) throws IOException {
98           final List<FileImageData> result = new ArrayList<>();
99           final int nbImages = system.getNbImages();
100           for (int i = 0; i < nbImages; i++) {
101   
102               final SFile f = suggestedFile.getFile(i);
103               if (canFileBeWritten(f) == false) {
104                   return result;
105               }
106               final OutputStream fos = f.createBufferedOutputStream();
107               ImageData cmap = null;
108               try {
109                   system.exportDiagram(fos, i, fileFormat);
110               } finally {
111                   fos.close();
112               }
113               // if (system.hasUrl() && cmap != null && cmap.containsCMapData()) {
114               // system.exportCmap(suggestedFile, cmap);
115               // }
116               Log.info("File size : " + f.length());
117               result.add(new FileImageData(f, cmap));
118           }
119           return result;
120       }

122       public static boolean canFileBeWritten(final SFile f) {
123           Log.info("Creating file: " + f.getAbsolutePath());
124           if (f.exists() && f.canWrite() == false) {
125               if (OptionFlags.getInstance().isOverwrite()) {
126                   Log.info("Overwrite " + f);
127                   f.setWritable(true);
128                   f.delete();
129                   return true;
130               }
131               Log.error("Cannot write to file " + f.getAbsolutePath());
132               return false;
133           }
134           return true;
135       }

137       private static List<FileImageData> exportDiagramsSequence(SequenceDiagram system, SuggestedFile suggestedFile,
138               FileFormatOption fileFormat) throws IOException {
139           final List<FileImageData> result = new ArrayList<>();
140           final int nbImages = system.getNbImages();
141           for (int i = 0; i < nbImages; i++) {
142   
143               final SFile f = suggestedFile.getFile(i);
144               if (PSystemUtils.canFileBeWritten(suggestedFile.getFile(i)) == false) {
145                   return result;
146               }
147               final OutputStream fos = f.createBufferedOutputStream();
148               ImageData cmap = null;
149               try {
150                   cmap = system.exportDiagram(fos, i, fileFormat);
151               } finally {
152                   fos.close();
153               }
154               if (cmap != null && cmap.containsCMapData()) {
155                   system.exportCmap(suggestedFile, i, cmap);
156               }
157               Log.info("File size : " + f.length());
158               result.add(new FileImageData(f, cmap));
159           }
160           return result;
161       }

163       private static List<FileImageData> createFilesHtml(CucaDiagram system, SuggestedFile suggestedFile)
164               throws IOException {
165           final String name = suggestedFile.getName();
166           final int idx = name.lastIndexOf('.');
167           final SFile dir = suggestedFile.getParentFile().file(name.substring(0, idx));
168           final CucaDiagramHtmlMaker maker = new CucaDiagramHtmlMaker(system, dir);
169           return maker.create();
170       }

172       private static List<FileImageData> splitPng(TitledDiagram diagram, SuggestedFile pngFile, ImageData imageData, FileFormatOption fileFormatOption)
173               throws IOException {
174   
175           final List<SFile> files = new PngSplitter(
176                   pngFile,
177                   diagram.getSplitPagesHorizontal(),
178                   diagram.getSplitPagesVertical(),
179                   fileFormatOption.isWithMetadata() ? diagram.getMetadata() : null,
180                   diagram.getSkinParam().getDpi(),
181                   diagram instanceof GanttDiagram
182                           ? new SplitParam(HColors.BLACK, null, 5)  // for backwards compatibility
183                           : diagram.getSkinParam().getSplitParam()
184           ).getFiles();
185   
186           final List<FileImageData> result = new ArrayList<>();
187           for (SFile f : files) {
188               result.add(new FileImageData(f, imageData));
189           }
190           return result;
191       }

193       private static List<FileImageData> exportDiagramsDefault(Diagram system, SuggestedFile suggestedFile,
194               FileFormatOption fileFormatOption) throws IOException {
195   
196           final SFile outputFile = suggestedFile.getFile(0);
197   
198           if (outputFile.isDirectory()) {
199               throw new IllegalArgumentException("File is a directory " + suggestedFile);
200           }
201   
202           if (!canFileBeWritten(outputFile)) {
203               return emptyList();
204           }
205   
206           final ImageData imageData;
207   
208           try (OutputStream os = outputFile.createBufferedOutputStream()) {
209               imageData = system.exportDiagram(os, 0, fileFormatOption);
210           }
211   
212           if (imageData == null) {
213               return emptyList();
214           }
215   
216           if (imageData.containsCMapData() && system instanceof UmlDiagram) {
217               ((UmlDiagram) system).exportCmap(suggestedFile, 0, imageData);
218           }
219   
220           if (system instanceof TitledDiagram && fileFormatOption.getFileFormat() == FileFormat.PNG) {
221               return splitPng((TitledDiagram) system, suggestedFile, imageData, fileFormatOption);
222           }
223   
224           return singletonList(new FileImageData(outputFile, imageData));
225       }
