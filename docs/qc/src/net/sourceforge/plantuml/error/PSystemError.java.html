148       private TextBlockBackcolored getGraphicalFormatted() {

164           result = TextBlockUtils.mergeTB(result, result3, HorizontalAlignment.LEFT);
165           result = TextBlockUtils.mergeTB(result4, result, HorizontalAlignment.LEFT);
166           result = TextBlockUtils.withMargin(result, 5, 5);
167           return TextBlockUtils.addBackcolor(result, HColors.BLACK);

185       private List<String> getTextFromStack() {
186           LineLocation lineLocation = getLineLocation();
187           final List<String> result = new ArrayList<>();
188           if (lineLocation != null) {
189               append(result, lineLocation);
190               while (lineLocation.getParent() != null) {
191                   lineLocation = lineLocation.getParent();
192                   append(result, lineLocation);
193               }
194           }
195           return result;
196       }

198       protected List<String> getTextFullBody() {
199           final List<String> result = new ArrayList<>();
200           result.add(" ");
201           final int traceSize = trace.size();
202           if (traceSize > 40) {
203               for (StringLocated s : trace.subList(0, 5)) {
204                   addToResult(result, s);
205               }
206               result.add("...");
207               final int skipped = traceSize - 5 - 20;
208               result.add("... ( skipping " + skipped + " lines )");
209               result.add("...");
210               for (StringLocated s : trace.subList(traceSize - 20, traceSize)) {
211                   addToResult(result, s);
212               }
213           } else {
214               for (StringLocated s : trace) {
215                   addToResult(result, s);
216               }
217           }
218           return result;
219       }

221       private void addToResult(final List<String> result, StringLocated s) {
222           String tmp = s.getString();
223           if (tmp.length() > 120) {
224               tmp = tmp.substring(0, 120) + " ...";
225           }
226           result.add(tmp);
227       }

247       @Override
248       protected UDrawable getRootDrawable(FileFormatOption fileFormatOption) throws IOException {
249           final TextBlockBackcolored result = getGraphicalFormatted();
250   
251           TextBlock udrawable;
252           if (getSource().getTotalLineCountLessThan5()) {
253               udrawable = addWelcome(result);
254           } else {
255               udrawable = result;
256           }
257           final int min = (int) (System.currentTimeMillis() / 60000L) % 60;
258           // udrawable = addMessageAdopt(udrawable);
259           if (disableTimeBasedErrorDecorations) {
260               // do nothing
261           } else if (min == 1 || min == 8 || min == 13 || min == 55) {
262               udrawable = addMessagePatreon(udrawable);
263           } else if (min == 15) {
264               udrawable = addMessageLiberapay(udrawable);
265           } else if (min == 30 || min == 39 || min == 48) {
266               udrawable = addMessageDedication(udrawable);
267           } else if (getSource().containsIgnoreCase("arecibo")) {
268               udrawable = addMessageArecibo(udrawable);
269           }
270           return udrawable;
271       }
