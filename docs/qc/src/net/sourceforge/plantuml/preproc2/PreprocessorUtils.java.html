62       public static String withEnvironmentVariable(String s) {
63           final Pattern p = Pattern.compile("%(\\w+)%");
64   
65           final Matcher m = p.matcher(s);
66           final StringBuffer sb = new StringBuffer(); // Can't be switched to StringBuilder in order to support Java 8
67           while (m.find()) {
68               final String var = m.group(1);
69               final String value = getenv(var);
70               if (value != null)
71                   m.appendReplacement(sb, Matcher.quoteReplacement(value));
72   
73           }
74           m.appendTail(sb);
75           s = sb.toString();
76           return s;
77       }

79       public static String getenv(String var) {
80           final String env = System.getProperty(var);
81           if (StringUtils.isNotEmpty(env))
82               return StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(env);
83   
84           final String getenv = System.getenv(var);
85           if (StringUtils.isNotEmpty(getenv))
86               return StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(getenv);
87   
88           return null;
89       }

91       private static InputStream getStdlibInputStream(String filename) {
92           final InputStream result = Stdlib.getResourceAsStream(filename);
93           // Log.info("Loading sdlib " + filename + " ok");
94           return result;
95       }

97       public static ReadLine getReaderStdlibInclude(StringLocated s, String filename) {
98           Log.info("Loading sdlib " + filename);
99           InputStream is = getStdlibInputStream(filename);
100           if (is == null)
101               return null;
102   
103           final String description = "<" + filename + ">";
104           try {
105               if (StartDiagramExtractReader.containsStartDiagram(is, s, description)) {
106                   is = getStdlibInputStream(filename);
107                   return StartDiagramExtractReader.build(is, s, description);
108               }
109               is = getStdlibInputStream(filename);
110               if (is == null)
111                   return null;
112   
113               return ReadLineReader.create(new InputStreamReader(is), description);
114           } catch (IOException e) {
115               Logme.error(e);
116               return new ReadLineSimple(s, e.toString());
117           }
118       }

120       public static ReadLine getReaderIncludeUrl(final SURL url, StringLocated s, String suf, Charset charset)
121               throws EaterException {
122           try {
123               if (StartDiagramExtractReader.containsStartDiagram(url, s, charset))
124                   return StartDiagramExtractReader.build(url, s, suf, charset);
125   
126               return getReaderInclude(url, s.getLocation(), charset);
127           } catch (IOException e) {
128               Logme.error(e);
129               throw EaterException.located("Cannot open URL " + e.getMessage());
130           }
131   
132       }

134       public static ReadLine getReaderInclude(SURL url, LineLocation lineLocation, Charset charset)
135               throws EaterException, UnsupportedEncodingException {
136           final InputStream is = url.openStream();
137           if (is == null)
138               throw EaterException.located("Cannot open URL");
139   
140           return ReadLineReader.create(new InputStreamReader(is, charset), url.toString(), lineLocation);
141       }
