96       private List<String> encode16(BufferedImage img) {
97           final int width = img.getWidth();
98           final int height = img.getHeight();
99           // final int type = img.getType();
100   
101           final List<String> result = new ArrayList<>();
102   
103           for (int y = 0; y < height; y++) {
104               final StringBuilder sb = new StringBuilder();
105               for (int x = 0; x < width; x++) {
106                   final int level = getGrayOn16(img, x, y);
107                   final char code = "0123456789ABCDEF".charAt(level);
108                   sb.append(code);
109               }
110               result.add(sb.toString());
111           }
112           return Collections.unmodifiableList(result);
113       }

115       private List<String> encode8(BufferedImage img) {
116           final int width = img.getWidth();
117           final int height = img.getHeight();
118           // final int type = img.getType();
119   
120           final List<String> result = new ArrayList<>();
121   
122           for (int y = 0; y < height; y += 2) {
123               final StringBuilder sb = new StringBuilder();
124               for (int x = 0; x < width; x++) {
125                   final int level1 = getGrayOn16(img, x, y) / 2;
126                   assert level1 >= 0 && level1 <= 7;
127                   final int level2 = getGrayOn16(img, x, y + 1) / 2;
128                   assert level2 >= 0 && level2 <= 7;
129                   final int v = level1 * 8 + level2;
130                   sb.append(AsciiEncoder.encode6bit((byte) v));
131               }
132               result.add(sb.toString());
133           }
134           return Collections.unmodifiableList(result);
135       }

137       private List<String> encode4(BufferedImage img) {
138           final int width = img.getWidth();
139           final int height = img.getHeight();
140           // final int type = img.getType();
141   
142           final List<String> result = new ArrayList<>();
143   
144           for (int y = 0; y < height; y += 3) {
145               final StringBuilder sb = new StringBuilder();
146               for (int x = 0; x < width; x++) {
147                   final int level1 = getGrayOn16(img, x, y) / 4;
148                   assert level1 >= 0 && level1 <= 3;
149                   final int level2 = getGrayOn16(img, x, y + 1) / 4;
150                   assert level2 >= 0 && level2 <= 3;
151                   final int level3 = getGrayOn16(img, x, y + 2) / 4;
152                   assert level3 >= 0 && level3 <= 3;
153                   final int v = level1 * 16 + level2 * 4 + level3;
154                   sb.append(AsciiEncoder.encode6bit((byte) v));
155               }
156               result.add(sb.toString());
157           }
158           return Collections.unmodifiableList(result);
159       }

161       private int getGrayOn16(BufferedImage img, int x, int y) {
162           if (x >= img.getWidth()) {
163               return 0;
164           }
165           if (y >= img.getHeight()) {
166               return 0;
167           }
168           final Color g = ColorUtils.getGrayScaleColor(new Color(img.getRGB(x, y)));
169           final int gray = 255 - g.getRed();
170           return gray / 16;
171       }

173       public Sprite buildSprite(int width, int height, List<String> strings) {
174           if (this == SpriteGrayLevel.GRAY_16) {
175               return buildSprite16(strings);
176           }
177           if (this == SpriteGrayLevel.GRAY_8) {
178               return buildSprite8(width, height, strings);
179           }
180           if (this == SpriteGrayLevel.GRAY_4) {
181               return buildSprite4(width, height, strings);
182           }
183           throw new UnsupportedOperationException(toString());
184       }

186       private Sprite buildSprite16(List<String> strings) {
187           final SpriteMonochrome result = new SpriteMonochrome(strings.get(0).length(), strings.size(), 16);
188           for (int col = 0; col < result.getWidth(); col++) {
189               for (int line = 0; line < result.getHeight(); line++) {
190                   if (col >= strings.get(line).length()) {
191                       continue;
192                   }
193                   if (strings.get(line).charAt(col) != '0') {
194                       final String s = "" + strings.get(line).charAt(col);
195                       final int x = Integer.parseInt(StringUtils.goUpperCase(s), 16);
196                       result.setGray(col, line, x);
197                   }
198               }
199           }
200           return result;
201       }

203       private Sprite buildSprite8(int width, int height, List<String> strings) {
204           final SpriteMonochrome result = new SpriteMonochrome(width, height, 8);
205           for (int col = 0; col < result.getWidth(); col++) {
206               for (int line = 0; line < strings.size(); line++) {
207                   if (col >= strings.get(line).length()) {
208                       continue;
209                   }
210                   final int v = AsciiEncoder.decode6bit(strings.get(line).charAt(col));
211                   final int w1 = v / 8;
212                   final int w2 = v % 8;
213                   result.setGray(col, line * 2, w1);
214                   result.setGray(col, line * 2 + 1, w2);
215   
216               }
217           }
218           return result;
219       }

221       private Sprite buildSprite4(int width, int height, List<String> strings) {
222           final SpriteMonochrome result = new SpriteMonochrome(width, height, 4);
223           for (int col = 0; col < result.getWidth(); col++) {
224               for (int line = 0; line < strings.size(); line++) {
225                   if (col >= strings.get(line).length()) {
226                       continue;
227                   }
228                   int v = AsciiEncoder.decode6bit(strings.get(line).charAt(col));
229                   final int w1 = v / 16;
230                   v = v % 16;
231                   final int w2 = v / 4;
232                   final int w3 = v % 4;
233                   result.setGray(col, line * 3, w1);
234                   result.setGray(col, line * 3 + 1, w2);
235                   result.setGray(col, line * 3 + 2, w3);
236   
237               }
238           }
239           return result;
240       }

242       public List<String> encodeZ(BufferedImage img) {
243           final int width = img.getWidth();
244           final int height = img.getHeight();
245           final byte raw[] = new byte[width * height];
246           int cpt = 0;
247           final int coef = 16 / nbColor;
248           for (int y = 0; y < height; y++) {
249               for (int x = 0; x < width; x++) {
250                   final int color = getGrayOn16(img, x, y) / coef;
251                   raw[cpt++] = (byte) color;
252               }
253           }
254           // final byte[] comp = new CompressionZlib().compress(raw);
255           final byte[] comp = new CompressionZopfliZlib().compress(raw);
256           return cut(new AsciiEncoderFinalZeros().encode(comp));
257       }

259       private List<String> encodeZSpiral(BufferedImage img) {
260           final int width = img.getWidth();
261           final int height = img.getHeight();
262           final byte raw[] = new byte[width * height];
263           final int coef = 16 / nbColor;
264           final SpiralOnRectangle spiral = new SpiralOnRectangle(width, height);
265           for (int cpt = 0; cpt < width * height; cpt++) {
266               final PairInt pt = spiral.nextPoint();
267               final int color = getGrayOn16(img, pt.getX(), pt.getY()) / coef;
268               raw[cpt] = (byte) color;
269           }
270           // final byte[] comp = new CompressionZlib().compress(raw);
271           final byte[] comp = new CompressionZopfliZlib().compress(raw);
272           return cut(new AsciiEncoderFinalZeros().encode(comp));
273       }

275       private List<String> cut(String s) {
276           final List<String> result = new ArrayList<>();
277           for (int i = 0; i < s.length(); i += 120) {
278               final int j = Math.min(i + 120, s.length());
279               result.add(s.substring(i, j));
280   
281           }
282           return Collections.unmodifiableList(result);
283       }

285       public Sprite buildSpriteZ(int width, int height, String compressed) {
286           final byte comp[] = new AsciiEncoder().decode(compressed);
287           try {
288               final ByteArray img = new CompressionZlib().decompress(comp);
289               final SpriteMonochrome result = new SpriteMonochrome(width, height, nbColor);
290               int cpt = 0;
291               for (int line = 0; line < result.getHeight(); line++) {
292                   for (int col = 0; col < result.getWidth(); col++) {
293                       result.setGray(col, line, img.getByteAt(cpt++));
294   
295                   }
296               }
297               return result;
298           } catch (NoPlantumlCompressionException e) {
299               Logme.error(e);
300               return null;
301           }
302       }
