67   public final class PlayerRobustConcise extends Player {
68   
69       private final Set<ChangeState> changes = new TreeSet<>();
70       private final List<TimeConstraint> constraints = new ArrayList<>();
71       private final List<TimingNote> notes = new ArrayList<>();
72       private final Map<String, String> statesLabel = new LinkedHashMap<String, String>();
73       private final TimingStyle type;
74   
75       private String initialState;
76       private PDrawing cached;
77       private Colors initialColors;
78   
79       public PlayerRobustConcise(TimingStyle type, String full, ISkinParam skinParam, TimingRuler ruler, boolean compact,
80               Stereotype stereotype) {
81           super(full, skinParam, ruler, compact, stereotype);
82           this.type = type;
83           this.suggestedHeight = 0;
84       }
85   
86       @Override
87       protected StyleSignature getStyleSignature() {
88           if (type == TimingStyle.CONCISE)
89               return StyleSignatureBasic.of(SName.root, SName.element, SName.timingDiagram, SName.concise)
90                       .withTOBECHANGED(stereotype);
91           if (type == TimingStyle.ROBUST)
92               return StyleSignatureBasic.of(SName.root, SName.element, SName.timingDiagram, SName.robust)
93                       .withTOBECHANGED(stereotype);
94           throw new IllegalStateException();
95       }
96   
97       private PDrawing buildPDrawing() {
98           final Style style = getStyleSignature().getMergedStyle(skinParam.getCurrentStyleBuilder());
99           if (type == TimingStyle.CONCISE)
100               return new Ribbon(ruler, skinParam, notes, isCompact(), getTitle(), suggestedHeight, style);
101   
102           if (type == TimingStyle.ROBUST) {
103               final Style style0 = StyleSignatureBasic.of(SName.root, SName.element, SName.timingDiagram)
104                       .getMergedStyle(skinParam.getCurrentStyleBuilder());
105               return new Histogram(ruler, skinParam, statesLabel.values(), isCompact(), getTitle(), suggestedHeight,
106                       style, style0);
107           }
108   
109           throw new IllegalStateException();
110       }
111   
112       public final TextBlock getPart1(final double fullAvailableWidth, final double specialVSpace) {
113           return new AbstractTextBlock() {
114   
115               public void drawU(UGraphic ug) {
116                   if (isCompact() == false)
117                       new PlayerFrame(getTitle(), skinParam).drawFrameTitle(ug);
118   
119                   ug = ug.apply(getTranslateForTimeDrawing(ug.getStringBounder())).apply(UTranslate.dy(specialVSpace));
120                   getTimeDrawing().getPart1(fullAvailableWidth).drawU(ug);
121               }
122   
123               public XDimension2D calculateDimension(StringBounder stringBounder) {
124                   return getTimeDrawing().getPart1(fullAvailableWidth).calculateDimension(stringBounder);
125               }
126           };
127       }
128   
129       public UDrawable getPart2() {
130           return new UDrawable() {
131               public void drawU(UGraphic ug) {
132                   ug = ug.apply(getTranslateForTimeDrawing(ug.getStringBounder()));
133                   getTimeDrawing().getPart2().drawU(ug);
134               }
135           };
136       }
137   
138       private UTranslate getTranslateForTimeDrawing(StringBounder stringBounder) {
139           return UTranslate.dy(getTitleHeight(stringBounder));
140       }
141   
142       public final double getFullHeight(StringBounder stringBounder) {
143           return getTitleHeight(stringBounder) + getZoneHeight(stringBounder);
144       }
145   
146       private double getTitleHeight(StringBounder stringBounder) {
147           if (isCompact())
148               return 6;
149   
150           return getTitle().calculateDimension(stringBounder).getHeight() + 6;
151       }
152   
153       private PDrawing getTimeDrawing() {
154           if (cached == null)
155               cached = computeTimeDrawing();
156   
157           return cached;
158       }
159   
160       private PDrawing computeTimeDrawing() {
161           final PDrawing result = buildPDrawing();
162           result.setInitialState(initialState, initialColors);
163           for (ChangeState change : changes)
164               result.addChange(change);
165   
166           for (TimeConstraint constraint : constraints)
167               result.addConstraint(constraint);
168   
169           return result;
170       }
171   
172       private double getZoneHeight(StringBounder stringBounder) {
173           return getTimeDrawing().getFullHeight(stringBounder);
174       }
175   
176       public final void setState(TimeTick now, String comment, Colors color, String... states) {
177           for (int i = 0; i < states.length; i++)
178               states[i] = decodeState(states[i]);
179   
180           if (now == null) {
181               this.initialState = states[0];
182               this.initialColors = color;
183           } else {
184               this.changes.add(new ChangeState(now, comment, color, states));
185           }
186   
187       }
188   
189       private String decodeState(String code) {
190           final String label = statesLabel.get(code);
191           if (label == null)
192               return code;
193   
194           return label;
195       }
196   
197       public final IntricatedPoint getTimeProjection(StringBounder stringBounder, TimeTick tick) {
198           if (tick == null)
199               return null;
200           final IntricatedPoint point = getTimeDrawing().getTimeProjection(stringBounder, tick);
201           if (point == null)
202               return null;
203   
204           final UTranslate translation = getTranslateForTimeDrawing(stringBounder);
205           return point.translated(translation);
206       }

209       public final void createConstraint(TimeTick tick1, TimeTick tick2, String message, ArrowConfiguration config) {
210           this.constraints.add(new TimeConstraint(tick1, tick2, message, skinParam, config));
211       }

213       public final void addNote(TimeTick now, Display note, Position position) {
214   
215           final StyleSignatureBasic signature = StyleSignatureBasic.of(SName.root, SName.element, SName.timingDiagram,
216                   SName.note);
217           final Style style = signature.getMergedStyle(skinParam.getCurrentStyleBuilder());
218   
219           this.notes.add(new TimingNote(now, this, note, position, skinParam, style));
220       }

222       public final void defineState(String stateCode, String label) {
223           statesLabel.put(stateCode, label);
224       }
