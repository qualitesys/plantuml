66   public class PlayerBinary extends Player {
67   
68       private static final String LOW_STRING = "0";
69       private static final String HIGH_STRING = "1";
70   
71       private final List<TimeConstraint> constraints = new ArrayList<>();
72       private final SortedMap<TimeTick, ChangeState> values = new TreeMap<>();
73       private ChangeState initialState;
74   
75       public PlayerBinary(String code, ISkinParam skinParam, TimingRuler ruler, boolean compact, Stereotype stereotype) {
76           super(code, skinParam, ruler, compact, stereotype);
77           this.suggestedHeight = 30;
78       }
79   
80       private double getHeightForConstraints(StringBounder stringBounder) {
81           return TimeConstraint.getHeightForConstraints(stringBounder, constraints);
82       }
83   
84       public double getFullHeight(StringBounder stringBounder) {
85           return getHeightForConstraints(stringBounder) + suggestedHeight;
86       }
87   
88       @Override
89       protected StyleSignature getStyleSignature() {
90           return StyleSignatureBasic.of(SName.root, SName.element, SName.timingDiagram, SName.binary)
91                   .withTOBECHANGED(stereotype);
92       }

94       public IntricatedPoint getTimeProjection(StringBounder stringBounder, TimeTick tick) {
95           final double x = ruler.getPosInPixel(tick);
96           return new IntricatedPoint(new XPoint2D(x, getYpos(stringBounder, HIGH_STRING)),
97                   new XPoint2D(x, getYpos(stringBounder, HIGH_STRING)));
98       }

100       public void addNote(TimeTick now, Display note, Position position) {

105           throw new UnsupportedOperationException();
106       }

108       public void setState(TimeTick now, String comment, Colors color, String... states) {
109           final ChangeState cs = new ChangeState(now, comment, color, convert(states));
110           if (now == null)
111               this.initialState = cs;
112           else
113               this.values.put(now, cs);
114   
115       }

117       private String[] convert(String[] states) {
118           if (states.length == 1)
119               return new String[] { convert(states[0]) };
120           return new String[] { convert(states[0]), convert(states[1]) };
121       }
122   
123       private String convert(String value) {
124           if ("1".equals(value) || "high".equalsIgnoreCase(value))
125               return HIGH_STRING;
126           return LOW_STRING;
127       }
128   
129       @Override
130       public void createConstraint(TimeTick tick1, TimeTick tick2, String message, ArrowConfiguration config) {
131           this.constraints.add(new TimeConstraint(tick1, tick2, message, skinParam, config));
132       }
133   
134       private final double ymargin = 8;
135   
136       private double getYpos(StringBounder stringBounder, String state) {
137           if (state.equalsIgnoreCase(LOW_STRING))
138               return getYlow(stringBounder);
139           return getYhigh(stringBounder);
140       }
141   
142       private double getYlow(StringBounder stringBounder) {
143           return getFullHeight(stringBounder) - ymargin;
144       }
145   
146       private double getYhigh(StringBounder stringBounder) {
147           return ymargin + getHeightForConstraints(stringBounder);
148       }

150       public TextBlock getPart1(double fullAvailableWidth, double specialVSpace) {
151           return new AbstractTextBlock() {
152   
153               public void drawU(UGraphic ug) {
154                   final StringBounder stringBounder = ug.getStringBounder();
155                   final TextBlock title = getTitle();
156                   final XDimension2D dim = title.calculateDimension(stringBounder);
157                   final double y = (getFullHeight(stringBounder) - dim.getHeight()) / 2;
158                   title.drawU(ug.apply(UTranslate.dy(y)));
159               }
160   
161               public XDimension2D calculateDimension(StringBounder stringBounder) {
162                   final XDimension2D dim = getTitle().calculateDimension(stringBounder);
163                   return XDimension2D.delta(dim, 5, 0);
164               }
165           };
166       }

168       public UDrawable getPart2() {
169           return new UDrawable() {
170               public void drawU(UGraphic ug) {
171                   ug = getContext().apply(ug);
172                   double lastx = 0;
173                   List<String> lastValues = initialState == null ? Collections.singletonList(LOW_STRING)
174                           : initialState.getStates();
175                   final StringBounder stringBounder = ug.getStringBounder();
176                   final double yhigh = getYhigh(stringBounder);
177                   final double ylow = getYlow(stringBounder);
178                   final ULine vline = ULine.vline(ylow - yhigh);
179                   for (Map.Entry<TimeTick, ChangeState> ent : values.entrySet()) {
180                       final ChangeState value = ent.getValue();
181   
182                       final double x = ruler.getPosInPixel(ent.getKey());
183   
184                       if (lastValues.size() == 1)
185                           ug.apply(new UTranslate(lastx, getYpos(stringBounder, lastValues.get(0))))
186                                   .draw(ULine.hline(x - lastx));
187                       else
188                           for (double tmpx = lastx; tmpx < x; tmpx += 5)
189                               ug.apply(new UTranslate(tmpx, yhigh)).draw(vline);
190   
191                       if (lastValues.equals(value.getStates()) == false)
192                           ug.apply(new UTranslate(x, yhigh)).draw(vline);
193   
194                       if (value.getComment() != null) {
195                           final TextBlock label = getTextBlock(value.getComment());
196                           label.drawU(ug.apply(new UTranslate(x + 2, yhigh)));
197                       }
198   
199                       lastx = x;
200                       lastValues = value.getStates();
201                   }
202                   ug.apply(new UTranslate(lastx, getYpos(stringBounder, lastValues.get(0))))
203                           .draw(ULine.hline(ruler.getWidth() - lastx));
204   
205                   drawConstraints(ug.apply(UTranslate.dy(getHeightForConstraints(ug.getStringBounder()))));
206   
207               }
208           };
209       }

216       private TextBlock getTextBlock(String value) {
217           final Display display = Display.getWithNewlines(value);
218           return display.create(getCommentFontConfiguration(), HorizontalAlignment.LEFT, skinParam);
219       }

221       private void drawConstraints(final UGraphic ug) {
222           for (TimeConstraint constraint : constraints) {
223               constraint.drawU(ug, ruler);
224           }
225       }
