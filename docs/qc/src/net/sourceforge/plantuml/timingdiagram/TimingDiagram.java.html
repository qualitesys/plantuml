145       private void drawInternal(UGraphic ug) {
146           ruler.ensureNotEmpty();
147           final StringBounder stringBounder = ug.getStringBounder();
148           final double part1MaxWidth = getPart1MaxWidth(stringBounder);
149           final UTranslate widthPart1 = UTranslate.dx(part1MaxWidth);
150           if (compactByDefault == false)
151               drawBorder(ug);
152   
153           ug = ug.apply(UTranslate.dx(marginX1));
154   
155           drawHighlightsBack(ug.apply(widthPart1));
156           ruler.drawVlines(ug.apply(widthPart1), getHeightInner(stringBounder));
157           boolean first = true;
158   
159           for (Player player : players.values()) {
160               final UGraphic ugPlayer = ug.apply(getUTranslateForPlayer(player, stringBounder));
161               final double caption = getHeightForCaptions(stringBounder);
162               if (first) {
163                   if (player.isCompact() == false)
164                       drawHorizontalSeparator(ugPlayer);
165   
166                   player.getPart1(part1MaxWidth, caption).drawU(ugPlayer);
167                   player.getPart2().drawU(ugPlayer.apply(widthPart1).apply(UTranslate.dy(caption)));
168               } else {
169                   if (player.isCompact() == false)
170                       drawHorizontalSeparator(ugPlayer.apply(UTranslate.dy(caption)));
171   
172                   player.getPart1(part1MaxWidth, 0).drawU(ugPlayer.apply(UTranslate.dy(caption)));
173                   player.getPart2().drawU(ugPlayer.apply(widthPart1).apply(UTranslate.dy(caption)));
174               }
175               first = false;
176           }
177           ug = ug.apply(widthPart1);
178           ruler.drawTimeAxis(ug.apply(getLastTranslate(stringBounder)), this.timeAxisStategy, codes);
179   
180           for (TimeMessage timeMessage : messages)
181               drawMessages(ug, timeMessage);
182   
183           drawHighlightsLines(ug);
184       }

186       private void drawHorizontalSeparator(UGraphic ug) {
187           final StringBounder stringBounder = ug.getStringBounder();
188           ug = ug.apply(black());
189           ug = ug.apply(getBorderStroke());
190           ug = ug.apply(UTranslate.dx(-marginX1));
191           ug.draw(ULine.hline(getWidthTotal(stringBounder)));
192       }

197           ug = ug.apply(black()).apply(getBorderStroke());
198           ug.draw(border);
199           ug.apply(UTranslate.dx(getWidthTotal(stringBounder))).draw(border);
200       }

217       private void drawHighlightsLines(UGraphic ug) {

226       private double getHeightTotal(StringBounder stringBounder) {

234       private double getHeightForCaptions(StringBounder stringBounder) {
235           double result = 0;
236           for (Highlight highlight : highlights) {
237               final TextBlock caption = highlight.getCaption(getSkinParam());
238               result = Math.max(result, caption.calculateDimension(stringBounder).getHeight());
239           }
240           return result;
241       }

243       private double getWidthTotal(final StringBounder stringBounder) {
244           return getPart1MaxWidth(stringBounder) + ruler.getWidth() + marginX1 + marginX2;
245       }
246   
247       private double getPart1MaxWidth(StringBounder stringBounder) {
248           double width = 0;
249           for (Player player : players.values())
250               width = Math.max(width, player.getPart1(0, 0).calculateDimension(stringBounder).getWidth());
251   
252           return width;
253       }

255       private void drawMessages(UGraphic ug, TimeMessage message) {
256           final Player player1 = message.getPlayer1();
257           final Player player2 = message.getPlayer2();
258   
259           final StringBounder stringBounder = ug.getStringBounder();
260           final UTranslate translate1 = getUTranslateForPlayer(player1, stringBounder)
261                   .compose(UTranslate.dy(getHeightForCaptions(stringBounder)));
262           final UTranslate translate2 = getUTranslateForPlayer(player2, stringBounder)
263                   .compose(UTranslate.dy(getHeightForCaptions(stringBounder)));
264   
265           final IntricatedPoint pt1 = player1.getTimeProjection(stringBounder, message.getTick1());
266           final IntricatedPoint pt2 = player2.getTimeProjection(stringBounder, message.getTick2());
267   
268           if (pt1 == null || pt2 == null)
269               return;
270   
271           final TimeArrow timeArrow = TimeArrow.create(pt1.translated(translate1), pt2.translated(translate2),
272                   message.getLabel(), getSkinParam(), message);
273           timeArrow.drawU(ug);
274   
275       }

277       private UTranslate getUTranslateForPlayer(Player candidat, StringBounder stringBounder) {
278           double y = 0;
279           for (Player player : players.values()) {
280               if (candidat == player)
281                   return UTranslate.dy(y);
282   
283   //            if (y == 0) {
284   //                y += getHeightHighlights(stringBounder);
285   //            }
286               y += player.getFullHeight(stringBounder);
287           }
288           if (candidat == null)
289               return UTranslate.dy(y);
290   
291           throw new IllegalArgumentException();
292       }
