97       public IntricatedPoint getTimeProjection(StringBounder stringBounder, TimeTick tick) {
98           final double x = ruler.getPosInPixel(tick);
99           final List<String> states = getStatesAt(tick);
100           if (states.size() == 0)
101               return null;
102   
103           final double heightForConstraints = getHeightForConstraints(stringBounder);
104           if (states.size() == 1) {
105               final double y = yOfState(states.get(0)) + heightForConstraints;
106               return new IntricatedPoint(new XPoint2D(x, y), new XPoint2D(x, y));
107           }
108           assert states.size() == 2;
109           final double y1 = yOfState(states.get(0)) + heightForConstraints;
110           final double y2 = yOfState(states.get(1)) + heightForConstraints;
111           assert y1 != y2;
112           return new IntricatedPoint(new XPoint2D(x, y1), new XPoint2D(x, y2));
113       }

115       private List<String> getStatesAt(TimeTick tick) {
116           if (changes.size() == 0)
117               return Collections.emptyList();
118   
119           for (int i = 0; i < changes.size(); i++) {
120               final int tickWithCurrentChangeTimeComparisonResult = changes.get(i).getWhen().compareTo(tick);
121               if (tickWithCurrentChangeTimeComparisonResult == 0) {
122                   if (i == 0 && initialState == null)
123                       return Arrays.asList(changes.get(i).getState());
124   
125                   if (i == 0 && initialState != null)
126                       return Arrays.asList(initialState, changes.get(i).getState());
127   
128                   return Arrays.asList(changes.get(i - 1).getState(), changes.get(i).getState());
129               }
130               if (tickWithCurrentChangeTimeComparisonResult > 0) {
131                   final int changeIndex;
132                   if (i == 0) {
133                       // if this time tick was not yet defined in any place, and is less then the
134                       // first one,
135                       // assume it's the leftmost
136                       changeIndex = 0;
137                   } else {
138                       changeIndex = i - 1;
139                   }
140                   return Collections.singletonList(changes.get(changeIndex).getState());
141               }
142           }
143           return Collections.singletonList(changes.get(changes.size() - 1).getState());
144       }

146       public void addChange(ChangeState change) {
147           changes.add(change);
148           if (change.isCompletelyHidden())
149               return;
150   
151           final List<String> states = change.getStates();
152           for (String state : states)
153               if (allStates.contains(state) == false)
154                   allStates.add(state);
155   
156       }

158       private XPoint2D[] getPoints(int n) {
159           final ChangeState change = changes.get(n);
160           final double x = ruler.getPosInPixel(change.getWhen());
161           final List<String> states = change.getStates();
162           if (states.size() == 2)
163               return new XPoint2D[] { new XPoint2D(x, yOfState(states.get(0))),
164                       new XPoint2D(x, yOfState(states.get(1))) };
165   
166           return new XPoint2D[] { new XPoint2D(x, yOfState(states.get(0))) };
167       }

172       }

225       private XDimension2D calculateDimensionPart1(StringBounder stringBounder) {
226           double width = 0;
227           for (String state : allStates) {
228               final TextBlock label = getTextBlock(state);
229               final XDimension2D dim = label.calculateDimension(stringBounder);
230               width = Math.max(width, dim.getWidth());
231           }
232           if (initialState != null)
233               width += getInitialWidth();
234   
235           if (compact)
236               width += title.calculateDimension(stringBounder).getWidth() + 15;
237   
238           return new XDimension2D(width, getFullHeight(stringBounder));
239       }

241       private void drawPart1(UGraphic ug, double fullAvailableWidth) {
242           final StringBounder stringBounder = ug.getStringBounder();
243           ug = ug.apply(UTranslate.dy(getHeightForConstraints(stringBounder)));
244   
245           if (compact) {
246               final double titleHeight = title.calculateDimension(stringBounder).getHeight();
247               final double dy = (getFullHeight(stringBounder) - titleHeight) / 2;
248               title.drawU(ug.apply(UTranslate.dy(dy)));
249           }
250           double width = getStatesWidth(stringBounder);
251           if (initialState != null)
252               width += getInitialWidth();
253   
254           if (fullAvailableWidth > width + 5)
255               ug = ug.apply(UTranslate.dx(fullAvailableWidth - width - 5));
256           else
257               ug = ug.apply(UTranslate.dx(fullAvailableWidth - width));
258           for (String state : allStates) {
259               final TextBlock label = getTextBlock(state);
260               final XDimension2D dim = label.calculateDimension(stringBounder);
261               label.drawU(ug.apply(UTranslate.dy(yOfState(state) - dim.getHeight() / 2 + 1)));
262           }
263       }

265       private double getStatesWidth(StringBounder stringBounder) {
266           double result = 0;
267           for (String state : allStates) {
268               final TextBlock label = getTextBlock(state);
269               final XDimension2D dim = label.calculateDimension(stringBounder);
270               result = Math.max(result, dim.getWidth());
271           }
272           return result;
273       }

275       private void drawPart2(UGraphic ug) {
276           if (changes.size() == 0)
277               return;
278   
279           ug = getContext().apply(ug);
280           ug = ug.apply(UTranslate.dy(getHeightForConstraints(ug.getStringBounder())));
281           drawHlines(ug);
282           drawVlines(ug);
283           drawLabels(ug);
284           drawConstraints(ug.apply(UTranslate.dy(-TimeConstraint.getTopMargin())));
285       }

287       private void drawHlines(UGraphic ug) {
288           if (initialState != null)
289               for (XPoint2D pt : getPoints(0))
290                   drawHLine(ug, getInitialPoint(), getInitialWidth() + pt.getX());
291   
292           for (int i = 0; i < changes.size(); i++) {
293               if (changes.get(i).isCompletelyHidden())
294                   continue;
295   
296               final double x2 = i < changes.size() - 1 ? getPointx(i + 1) : ruler.getWidth();
297               final double len = x2 - getPointx(i);
298               final XPoint2D[] points = getPoints(i);
299               if (points.length == 2)
300                   drawHBlock(ug.apply(changes.get(i).getBackColor(skinParam, style).bg()), points[0], points[1], len);
301   
302               if (i < changes.size() - 1)
303                   for (XPoint2D pt : points)
304                       drawHLine(ug, pt, len);
305   
306           }
307   
308           if (changes.get(changes.size() - 1).isCompletelyHidden() == false) {
309               for (XPoint2D pt : getPoints(changes.size() - 1)) {
310                   final double len = ruler.getWidth() - pt.getX();
311                   drawHLine(ug, pt, len);
312               }
313           }
314       }

316       private void drawHBlock(UGraphic ug, XPoint2D pt1, XPoint2D pt2, double len) {
317           final double minY = Math.min(pt1.getY(), pt2.getY());
318           final double maxY = Math.max(pt1.getY(), pt2.getY());
319           final XPoint2D pt = new XPoint2D(pt1.getX(), minY);
320           ug = ug.apply(new UTranslate(pt));
321           ug.draw(new URectangle(len, maxY - minY));
322           for (double x = 0; x < len; x += 5)
323               ug.apply(UTranslate.dx(x)).draw(ULine.vline(maxY - minY));
324   
325       }

328           ug.apply(new UTranslate(pt)).draw(ULine.hline(len));
329       }

331       private void drawVlines(UGraphic ug) {
332           if (initialState != null) {
333               final XPoint2D before = getInitialPoint();
334               final XPoint2D current = getPoints(0)[0];
335               ug.apply(new UTranslate(current)).draw(ULine.vline(before.getY() - current.getY()));
336           }
337           for (int i = 1; i < changes.size(); i++) {
338               if (changes.get(i - 1).isCompletelyHidden() || changes.get(i).isCompletelyHidden())
339                   continue;
340   
341               final double minY = Math.min(getPointMinY(i), getPointMinY(i - 1));
342               final double maxY = Math.max(getPointMaxY(i), getPointMaxY(i - 1));
343               ug.apply(new UTranslate(getPointx(i), minY)).draw(ULine.vline(maxY - minY));
344           }
345       }

347       private void drawLabels(UGraphic ug) {
348           for (int i = 0; i < changes.size(); i++) {
349               final XPoint2D ptLabel = getPoints(i)[0];
350               final String comment = changes.get(i).getComment();
351               if (comment == null)
352                   continue;
353   
354               final TextBlock label = getTextBlock(comment);
355               final XDimension2D dim = label.calculateDimension(ug.getStringBounder());
356               label.drawU(ug.apply(new UTranslate(ptLabel).compose(new UTranslate(2, -dim.getHeight()))));
357           }
358       }

360       private void drawConstraints(UGraphic ug) {
361           for (TimeConstraint constraint : constraints) {
362               double y = yOfState(constraint.getTick1());
363               for (ChangeState change : changes)
364                   if (constraint.containsStrict(change.getWhen()))
365                       y = Math.min(y, yOfState(change.getWhen()));
366   
367               constraint.drawU(ug.apply(UTranslate.dy(y)), ruler);
368           }
369       }
