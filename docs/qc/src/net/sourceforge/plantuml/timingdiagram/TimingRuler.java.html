94       public void scaleInPixels(long tick, long pixel) {
95           if (pixel <= 0 || tick <= 0)
96               throw new IllegalArgumentException();
97           this.tickIntervalInPixels = pixel;

99       }

101       private long tickUnitary() {
102           if (tickUnitary == 0)
103               return highestCommonFactor();
104   
105           return tickUnitary;
106   
107       }

109       private long highestCommonFactorInternal = -1;
110   
111       private long highestCommonFactor() {
112           if (highestCommonFactorInternal == -1) {
113               for (long tick : getAbsolutesTicks()) {
114                   if (highestCommonFactorInternal == -1) {
115                       highestCommonFactorInternal = tick;
116                   } else {
117                       final long candidate = computeHighestCommonFactor(highestCommonFactorInternal, tick);
118                       final double size = (getMax().getTime().doubleValue() - getMin().getTime().doubleValue())
119                               / candidate;
120                       if (size > 200)
121                           return highestCommonFactorInternal;
122   
123                       highestCommonFactorInternal = candidate;
124                   }
125               }
126           }
127           return highestCommonFactorInternal;
128       }
129   
130       private Set<Long> getAbsolutesTicks() {
131           final Set<Long> result = new TreeSet<>(new Comparator<Long>() {
132               public int compare(Long o1, Long o2) {
133                   return o2.compareTo(o1);
134               }
135           });
136           for (TimeTick time : times) {
137               final long value = Math.abs(time.getTime().longValue());
138               if (value > 0)
139                   result.add(value);
140   
141           }
142           return result;
143       }

145       private int getNbTick() {
146           if (times.size() == 0)
147               return 1;
148   
149           final long delta = getMax().getTime().longValue() - getMin().getTime().longValue();
150           return Math.min(1000, (int) (1 + delta / tickUnitary()));
151       }

153       public double getWidth() {
154           final double delta = getMax().getTime().doubleValue() - getMin().getTime().doubleValue();
155   
156           return (delta / tickUnitary() + 1) * tickIntervalInPixels;
157       }

163       private double getPosInPixelInternal(double time) {

165           return time / tickUnitary() * tickIntervalInPixels;
166       }

198       public void drawTimeAxis(UGraphic ug, TimeAxisStategy timeAxisStategy, Map<String, TimeTick> codes) {
199           if (timeAxisStategy == TimeAxisStategy.HIDDEN)
200               return;
201   
202           final Style style = StyleSignatureBasic.of(SName.root, SName.timingDiagram, SName.timeline)
203                   .getMergedStyle(skinParam.getCurrentStyleBuilder());
204   
205           final HColor color = style.value(PName.LineColor).asColor(skinParam.getIHtmlColorSet());
206           final UStroke stroke = style.getStroke();
207   
208           ug = ug.apply(stroke).apply(color);
209   
210           if (timeAxisStategy == TimeAxisStategy.AUTOMATIC)
211               drawTimeAxisAutomatic(ug);
212           else
213               drawTimeAxisManual(ug, codes);
214   
215       }

217       private void drawTimeAxisManual(UGraphic ug, Map<String, TimeTick> codes) {
218           final double tickHeight = 5;
219           final ULine line = ULine.vline(tickHeight);
220           final double firstTickPosition = getPosInPixelInternal(getFirstPositiveOrZeroValue().doubleValue());
221           int nb = 0;
222           while (firstTickPosition + nb * tickIntervalInPixels <= getWidth())
223               nb++;
224   
225           ug.apply(UTranslate.dx(firstTickPosition)).draw(ULine.hline((nb - 1) * tickIntervalInPixels));
226   
227           for (TimeTick tick : times) {
228               ug.apply(UTranslate.dx(getPosInPixel(tick))).draw(line);
229               final String label = getLabel(tick, codes);
230               if (label.length() == 0)
231                   continue;
232               final TextBlock text = getTimeTextBlock(label);
233               final XDimension2D dim = text.calculateDimension(ug.getStringBounder());
234               text.drawU(ug.apply(new UTranslate(getPosInPixel(tick) - dim.getWidth() / 2, tickHeight + 1)));
235   
236           }
237       }

239       private void drawTimeAxisAutomatic(UGraphic ug) {
240           final double tickHeight = 5;
241           final ULine line = ULine.vline(tickHeight);
242           final double firstTickPosition = getPosInPixelInternal(getFirstPositiveOrZeroValue().doubleValue());
243           int nb = 0;
244           while (firstTickPosition + nb * tickIntervalInPixels <= getWidth()) {
245               ug.apply(UTranslate.dx(firstTickPosition + nb * tickIntervalInPixels)).draw(line);
246               nb++;
247           }
248           ug.apply(UTranslate.dx(firstTickPosition)).draw(ULine.hline((nb - 1) * tickIntervalInPixels));
249   
250           for (long round : roundValues()) {
251               final TextBlock text = getTimeTextBlock(round);
252               final XDimension2D dim = text.calculateDimension(ug.getStringBounder());
253               text.drawU(ug.apply(new UTranslate(getPosInPixelInternal(round) - dim.getWidth() / 2, tickHeight + 1)));
254           }
255       }

257       private String getLabel(TimeTick tick, Map<String, TimeTick> codes) {
258           for (Entry<String, TimeTick> ent : codes.entrySet())
259               if (tick.equals(ent.getValue()))
260                   return ent.getKey();
261   
262           return format.formatTime(tick.getTime());
263       }

265       private BigDecimal getFirstPositiveOrZeroValue() {
266           for (TimeTick time : times)
267               if (time.getTime().signum() >= 0)
268                   return time.getTime();
269   
270           throw new IllegalStateException();
271       }

273       private Collection<Long> roundValues() {
274           final SortedSet<Long> result = new TreeSet<>();
275           if (tickUnitary == 0) {
276               for (TimeTick tick : times) {
277                   final long round = tick.getTime().longValue();
278                   result.add(round);
279               }
280           } else {
281               final int nb = getNbTick();
282               for (int i = 0; i <= nb; i++) {
283                   final long round = tickToTime(i);
284                   result.add(round);
285               }
286           }
287           if (result.first() < 0 && result.last() > 0)
288               result.add(0L);
289   
290           return result;
291       }

314       private static long computeHighestCommonFactor(long a, long b) {
315           long r = a;
316           while (r != 0) {
317               r = a % b;
318               a = b;
319               b = r;
320           }
321           return (Math.abs(a));
322       }
