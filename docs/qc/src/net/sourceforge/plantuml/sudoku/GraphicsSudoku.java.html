71   public class GraphicsSudoku {
72   
73       private final ISudoku sudoku;
74       private final UFont numberFont = UFont.sansSerif(20).bold();
75       private final UFont font = UFont.sansSerif(11);
76   
77       public GraphicsSudoku(ISudoku sudoku) {
78           this.sudoku = sudoku;
79       }
80   
81       public ImageData writeImageEps(OutputStream os) throws IOException {
82           final UGraphicEps ug = new UGraphicEps(HColors.WHITE, ColorMapper.IDENTITY,
83                   FileFormat.EPS_TEXT.getDefaultStringBounder(), EpsStrategy.WITH_MACRO_AND_TEXT);
84           drawInternal(ug);
85           os.write(ug.getEPSCode().getBytes());
86           return ImageDataSimple.ok();
87       }
88   
89       public ImageData writeImageSvg(OutputStream os) throws IOException {
90           final UGraphicSvg ug = new UGraphicSvg(HColors.WHITE, true, new XDimension2D(0, 0),
91                   ColorMapper.IDENTITY, false, 1.0, null, null, 0, "none", FileFormat.SVG.getDefaultStringBounder(),
92                   LengthAdjust.defaultValue(), false);
93           drawInternal(ug);
94           ug.writeToStream(os, null, -1); // dpi param is not used
95           return ImageDataSimple.ok();
96       }
97   
98       public ImageData writeImageLatex(OutputStream os, FileFormat fileFormat) throws IOException {
99           final UGraphicTikz ug = new UGraphicTikz(HColors.WHITE, ColorMapper.IDENTITY, FileFormat.LATEX.getDefaultStringBounder(), 1,
100                   fileFormat == FileFormat.LATEX);
101           drawInternal(ug);
102           ug.writeToStream(os, null, -1); // dpi param is not used
103           return ImageDataSimple.ok();
104       }
105   
106       public ImageData writeImagePng(OutputStream os) throws IOException {
107           final StringBounder stringBounder = FileFormat.PNG.getDefaultStringBounder();
108           final EmptyImageBuilder builder = new EmptyImageBuilder(null, sudoWidth, sudoHeight + textTotalHeight,
109                   Color.WHITE, stringBounder);
110           final BufferedImage im = builder.getBufferedImage();
111           final Graphics2D g3d = builder.getGraphics2D();
112   
113           final UGraphic ug = new UGraphicG2d(HColors.WHITE, ColorMapper.IDENTITY, stringBounder, g3d, 1.0);
114   
115           drawInternal(ug);
116           g3d.dispose();
117           PngIO.write(im, os, 96);
118           return new ImageDataSimple(im.getWidth(), im.getHeight());
119       }
120   
121       final private int xOffset = 5;
122       final private int yOffset = 5;
123   
124       final private int cellWidth = 30;
125       final private int cellHeight = 32;
126   
127       final private int numberxOffset = 10;
128       final private int numberyOffset = 5;
129   
130       final private int textTotalHeight = 50;
131   
132       final private int boldWidth = 3;
133       final private int sudoHeight = 9 * cellHeight + 2 * yOffset + boldWidth;
134       final private int sudoWidth = 9 * cellWidth + 2 * xOffset + boldWidth;
135   
136       private void drawInternal(UGraphic ug) {
137           ug = ug.apply(new UTranslate(xOffset, yOffset));
138   
139           for (int x = 0; x < 9; x++) {
140               for (int y = 0; y < 9; y++) {
141                   final int num = sudoku.getGiven(x, y);
142                   if (num > 0) {
143                       final TextBlock text = Display.create("" + num).create(FontConfiguration.blackBlueTrue(numberFont),
144                               HorizontalAlignment.CENTER, new SpriteContainerEmpty());
145                       text.drawU(ug
146                               .apply(new UTranslate((numberxOffset + x * cellWidth), (numberyOffset + y * cellHeight))));
147                   }
148               }
149           }
150   
151           ug = ug.apply(HColors.BLACK.bg()).apply(HColors.none());
152           for (int i = 0; i < 10; i++) {
153               final boolean bold = i % boldWidth == 0;
154               final int w = bold ? boldWidth : 1;
155               ug.apply(UTranslate.dy(i * cellHeight)).draw(new URectangle(9 * cellWidth + boldWidth, w));
156           }
157           for (int i = 0; i < 10; i++) {
158               final boolean bold = i % boldWidth == 0;
159               final int w = bold ? boldWidth : 1;
160               ug.apply(UTranslate.dx(i * cellWidth)).draw(new URectangle(w, 9 * cellHeight + boldWidth));
161           }
162   
163           ug = ug.apply(UTranslate.dy(sudoHeight));
164           final List<String> texts = new ArrayList<>();
165           texts.add("http://plantuml.com");
166           texts.add("Seed " + Long.toString(sudoku.getSeed(), 36));
167           texts.add("Difficulty " + sudoku.getRatting());
168           final TextBlock textBlock = Display.create(texts).create(FontConfiguration.blackBlueTrue(font),
169                   HorizontalAlignment.LEFT, new SpriteContainerEmpty());
170           textBlock.drawU(ug);
171       }
