36   package net.sourceforge.plantuml.graphml;
37   
38   import static java.nio.charset.StandardCharsets.UTF_8;
39   
40   import java.io.OutputStream;
41   
42   import javax.xml.parsers.DocumentBuilder;
43   import javax.xml.parsers.ParserConfigurationException;
44   import javax.xml.transform.OutputKeys;
45   import javax.xml.transform.Result;
46   import javax.xml.transform.Source;
47   import javax.xml.transform.Transformer;
48   import javax.xml.transform.TransformerException;
49   import javax.xml.transform.dom.DOMSource;
50   import javax.xml.transform.stream.StreamResult;
51   
52   import org.w3c.dom.Document;
53   import org.w3c.dom.Element;
54   
55   import net.sourceforge.plantuml.cucadiagram.GroupRoot;
56   import net.sourceforge.plantuml.cucadiagram.IEntity;
57   import net.sourceforge.plantuml.descdiagram.DescriptionDiagram;
58   import net.sourceforge.plantuml.xmi.XmlDiagramTransformer;
59   import net.sourceforge.plantuml.xml.XmlFactories;
60   
61   public class GraphmlDescriptionDiagram implements XmlDiagramTransformer {
62   
63       private final DescriptionDiagram diagram;
64       private final Document document;
65   
66       public GraphmlDescriptionDiagram(DescriptionDiagram diagram) throws ParserConfigurationException {
67           this.diagram = diagram;
68   
69           final DocumentBuilder builder = XmlFactories.newDocumentBuilder();
70           this.document = builder.newDocument();
71           document.setXmlVersion("1.0");
72           document.setXmlStandalone(true);
73   
74           final Element graphml = document.createElement("graphml");
75           graphml.setAttribute("xmlns", "http://graphml.graphdrawing.org/xmlns");
76           graphml.setAttribute("xmlns:xsi", "http://www.w3.org/2001/XMLSchema-instance");
77           graphml.setAttribute("xsi:schemaLocation", "http://graphml.graphdrawing.org/xmlns/1.0/graphml.xsd");
78           document.appendChild(graphml);
79   
80           final Element graph = document.createElement("graph");
81           graph.setAttribute("edgedefault", "undirected");
82           graphml.appendChild(graph);
83   
84           for (final IEntity ent : diagram.getLeafsvalues())
85               if (ent.getParentContainer() instanceof GroupRoot)
86                   addElement(ent, graph);
87   
88       }
89   
90       private void addElement(IEntity tobeAdded, Element container) {
91           final Element element = createEntityNode(tobeAdded);
92           container.appendChild(element);
93   
94       }
95   
96       private Element createEntityNode(IEntity entity) {
97           final Element cla = document.createElement("node");
98           cla.setAttribute("id", entity.getCode().getName());
99           return cla;
100       }
101   
102       public void transformerXml(OutputStream os) throws TransformerException, ParserConfigurationException {
103           final Source source = new DOMSource(document);
104   
105           final Result resultat = new StreamResult(os);
106   
107           final Transformer transformer = XmlFactories.newTransformer();
108           transformer.setOutputProperty(OutputKeys.INDENT, "yes");
109           transformer.setOutputProperty(OutputKeys.ENCODING, UTF_8.name());
110           transformer.transform(source, resultat);
111       }
112   
113   }
