61   public class CreoleParser implements SheetBuilder {
62   
63       private final FontConfiguration fontConfiguration;
64       private final ISkinSimple skinParam;
65       private final HorizontalAlignment horizontalAlignment;
66       private final CreoleMode creoleMode;
67       private final FontConfiguration stereotype;

69       public CreoleParser(FontConfiguration fontConfiguration, HorizontalAlignment horizontalAlignment,
70               ISkinSimple skinParam, CreoleMode creoleMode, FontConfiguration stereotype) {
71           this.stereotype = stereotype;
72           this.creoleMode = creoleMode;
73           this.fontConfiguration = fontConfiguration;
74           this.skinParam = Objects.requireNonNull(skinParam);
75           this.horizontalAlignment = horizontalAlignment;
76       }

78       private Stripe createStripe(String line, CreoleContext context, Stripe lastStripe,
79               FontConfiguration fontConfiguration) {
80           if (lastStripe instanceof StripeRaw) {
81               final StripeRaw code = (StripeRaw) lastStripe;
82               if (code.isTerminated()) {
83                   lastStripe = null;
84               } else {
85                   final boolean terminated = code.addAndCheckTermination(line);
86                   return null;
87               }
88           }
89   
90           if (lastStripe instanceof StripeTable && isTableLine(line)) {
91               final StripeTable table = (StripeTable) lastStripe;
92               table.analyzeAndAddLine(line);
93               return null;
94           } else if (lastStripe instanceof StripeTree && Parser.isTreeStart(StringUtils.trinNoTrace(line))) {
95               final StripeTree tree = (StripeTree) lastStripe;
96               tree.analyzeAndAdd(line);
97               return null;
98           } else if (isTableLine(line)) {
99               return new StripeTable(fontConfiguration, skinParam, line);
100           } else if (Parser.isTreeStart(line)) {
101               return new StripeTree(fontConfiguration, skinParam, line);
102           } else if (Parser.isCodeStart(line)) {
103               return new StripeCode(fontConfiguration.changeFamily(Parser.MONOSPACED));
104           } else if (Parser.isLatexStart(line)) {
105               return new StripeLatex(fontConfiguration);
106           }
107           return new CreoleStripeSimpleParser(line, context, fontConfiguration, skinParam, creoleMode)
108                   .createStripe(context);
109       }
110   
111       public static boolean isTableLine(String line) {
112           return line.matches("^(\\<#\\w+(,#?\\w+)?\\>)?\\|(\\=)?.*\\|$");
113       }
114   
115       public static boolean doesStartByColor(String line) {
116           return line.matches("^\\=?\\s*(\\<#\\w+(,#?\\w+)?\\>).*");
117       }

119       public Sheet createSheet(Display display) {
120           final Sheet sheet = new Sheet(horizontalAlignment);
121           if (Display.isNull(display) == false) {
122               final CreoleContext context = new CreoleContext();
123               for (CharSequence cs : display) {
124                   final Stripe stripe;
125                   if (cs instanceof EmbeddedDiagram) {
126                       final Atom atom = ((EmbeddedDiagram) cs).asDraw(skinParam);
127                       stripe = new Stripe() {
128                           public Atom getLHeader() {
129                               return null;
130                           }
131   
132                           public List<Atom> getAtoms() {
133                               return Arrays.asList(atom);
134                           }
135                       };
136                   } else if (cs instanceof Stereotype) {
137                       if (display.showStereotype())
138                           for (String st : ((Stereotype) cs).getLabels(skinParam.guillemet()))
139                               sheet.add(createStripe(st, context, sheet.getLastStripe(), stereotype));
140   
141                       continue;
142                   } else {
143                       stripe = createStripe(cs.toString(), context, sheet.getLastStripe(), fontConfiguration);
144                   }
145   
146                   if (stripe != null)
147                       sheet.add(stripe);
148   
149               }
150           }
151           return sheet;
152       }

154       public static void checkColor(Display result) throws NoSuchColorException {
155           FontConfiguration fc = FontConfiguration.blackBlueTrue(UFont.byDefault(10));
156           try {
157               new CreoleParser(fc, HorizontalAlignment.LEFT, new SpriteContainerEmpty(), CreoleMode.FULL, fc)
158                       .createSheet(result);
159           } catch (NoSuchColorRuntimeException e) {
160               throw new NoSuchColorException();
161           }
162       }
