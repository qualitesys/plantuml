183       private void addCommand(Command cmd) {
184           final String starters = cmd.startingChars();
185           for (int i = 0; i < starters.length(); i++) {
186               final char ch = starters.charAt(i);
187               List<Command> localList = commands.get(ch);
188               if (localList == null) {
189                   localList = new ArrayList<Command>();
190                   commands.put(ch, localList);
191               }
192               localList.add(cmd);
193           }
194       }

211       public void analyzeAndAdd(String line) {
212           if (Objects.requireNonNull(line).contains("" + BackSlash.hiddenNewLine()))
213               throw new IllegalArgumentException(line);
214   
215           line = CharHidder.hide(line);
216           if (style.getType() == StripeStyleType.HEADING) {
217               fontConfiguration = fontConfigurationForHeading(fontConfiguration, style.getOrder());
218               modifyStripe(line);
219           } else if (style.getType() == StripeStyleType.HORIZONTAL_LINE) {
220               atoms.add(CreoleHorizontalLine.create(fontConfiguration, line, style.getStyle(), skinParam));
221           } else {
222               modifyStripe(line);
223           }
224       }

226       private static FontConfiguration fontConfigurationForHeading(FontConfiguration fontConfiguration, int order) {
227           switch (order) {
228           case 0:
229               return fontConfiguration.bigger(4).bold();
230           case 1:
231               return fontConfiguration.bigger(2).bold();
232           case 2:
233               return fontConfiguration.bigger(1).bold();
234           default:
235               return fontConfiguration.italic();
236           }
237       }

267       public void addEmoji(String emojiName, String forcedColor) {
268           final Emoji emoji = Emoji.retrieve(emojiName);
269           if (emoji == null)
270               return;
271   
272           HColor col = null;
273           if (forcedColor == null)

275           else if (forcedColor.equals("#0") || forcedColor.equals("#000") || forcedColor.equals("#black"))
276               col = fontConfiguration.getColor();
277           else

279                   col = skinParam.getIHtmlColorSet().getColor(forcedColor);
280               } catch (NoSuchColorException e) {
281                   col = null;
282               }
283

291       private void modifyStripe(String line) {
292           final StringBuilder pending = new StringBuilder();
293   
294           while (line.length() > 0) {
295               final Command cmd = searchCommand(line);
296               if (cmd == null) {
297                   pending.append(line.charAt(0));
298                   line = line.substring(1);
299               } else {
300                   addPending(pending);
301                   line = cmd.executeAndGetRemaining(line, this);
302               }
303           }
304           addPending(pending);
305       }
306   
307       private void addPending(StringBuilder pending) {
308           if (pending.length() == 0)
309               return;
310   
311           atoms.add(AtomTextUtils.createLegacy(pending.toString(), fontConfiguration));
312           pending.setLength(0);
313       }
314   
315       private Command searchCommand(String line) {
316           final List<Command> localList = commands.get(line.charAt(0));
317           if (localList != null)
318               for (Command cmd : localList)
319                   if (cmd.matchingSize(line) != 0)
320                       return cmd;
321   
322           return null;
323       }
