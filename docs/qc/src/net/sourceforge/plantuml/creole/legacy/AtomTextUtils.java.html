78       public static Atom createUrl(Url url, FontConfiguration fontConfiguration, ISkinSimple skinSimple) {
79           fontConfiguration = fontConfiguration.hyperlink();
80           final Display display = Display.getWithNewlines(url.getLabel());
81           if (display.size() > 1) {
82               final List<Atom> all = new ArrayList<>();
83               for (CharSequence s : display.asList()) {
84                   all.add(createAtomText(s.toString(), url, fontConfiguration, skinSimple));
85               }
86               return new AtomVerticalTexts(all);
87   
88           }
89           return createAtomText(url.getLabel(), url, fontConfiguration, skinSimple);
90       }

92       private static final Pattern p = Pattern.compile(Splitter.openiconPattern + "|" + Splitter.spritePattern2 + "|"
93               + Splitter.imgPatternNoSrcColon + "|" + Splitter.emojiPattern);

95       private static Atom createAtomText(final String text, Url url, FontConfiguration fontConfiguration,
96               ISkinSimple skinSimple) {
97           final Matcher m = p.matcher(text);
98           final List<Atom> result = new ArrayList<>();
99           while (m.find()) {
100               final StringBuffer sb = new StringBuffer();
101               m.appendReplacement(sb, "");
102               if (sb.length() > 0) {
103                   result.add(new AtomText(sb.toString(), fontConfiguration, url, ZERO, ZERO, true));
104               }
105               final String valOpenicon = m.group(1);
106               final String valSprite = m.group(3);
107               final String valImg = m.group(5);
108               final String valEmoji = m.group(7);
109               if (valEmoji != null)
110                   throw new UnsupportedOperationException();
111   
112               if (valOpenicon != null) {
113                   final OpenIcon openIcon = OpenIcon.retrieve(valOpenicon);
114                   if (openIcon != null) {
115                       final double scale = Parser.getScale(m.group(2), 1);
116                       result.add(new AtomOpenIcon(null, scale, openIcon, fontConfiguration, url));
117                   }
118               } else if (valSprite != null) {
119                   final Sprite sprite = skinSimple.getSprite(valSprite);
120                   if (sprite != null) {
121                       final double scale = Parser.getScale(m.group(4), 1);
122                       result.add(new AtomSprite(null, scale, fontConfiguration, sprite, url));
123                   }
124               } else if (valImg != null) {
125                   final double scale = Parser.getScale(m.group(6), 1);
126                   result.add(AtomImg.create(valImg, ImgValign.TOP, 0, scale, url));
127   
128               }
129           }
130           final StringBuffer sb = new StringBuffer();
131           m.appendTail(sb);
132           if (sb.length() > 0) {
133               result.add(new AtomText(sb.toString(), fontConfiguration, url, ZERO, ZERO, true));
134           }
135           if (result.size() == 1) {
136               return result.get(0);
137           }
138           return new AtomHorizontalTexts(result);
139       }

141       public static Atom createListNumber(final FontConfiguration fontConfiguration, final int order, int localNumber) {
142           final DelayedDouble left = new DelayedDouble() {
143               public double getDouble(StringBounder stringBounder) {
144                   final XDimension2D dim = stringBounder.calculateDimension(fontConfiguration.getFont(), "9. ");
145                   return dim.getWidth() * order;
146               }
147           };
148           final DelayedDouble right = new DelayedDouble() {
149               public double getDouble(StringBounder stringBounder) {
150                   final XDimension2D dim = stringBounder.calculateDimension(fontConfiguration.getFont(), ".");
151                   return dim.getWidth();
152               }
153           };
154           return new AtomText("" + (localNumber + 1) + ".", fontConfiguration, null, left, right, true);
155       }
