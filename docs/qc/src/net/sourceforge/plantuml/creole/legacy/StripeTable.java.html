63           HEADER, NORMAL

72       public StripeTable(FontConfiguration fontConfiguration, ISkinSimple skinParam, String line) {
73           this.skinParam = skinParam;
74           this.fontConfiguration = fontConfiguration;
75           HColor lineColor = getBackOrFrontColor(line, 1);
76           if (lineColor == null)
77               lineColor = fontConfiguration.getColor();
78   
79           this.table = new AtomTable(lineColor);
80           this.marged = new AtomWithMargin(table, 2, 2);
81           analyzeAndAddInternal(line);
82       }

92       static Atom asAtom(List<StripeSimple> cells, double padding) {
93           final Sheet sheet = new Sheet(HorizontalAlignment.LEFT);
94           for (StripeSimple cell : cells)
95               sheet.add(cell);
96   
97           return new SheetBlock1(sheet, LineBreakStrategy.NONE, padding);
98       }

100       private HColor getBackOrFrontColor(String line, int idx) {
101           if (CreoleParser.doesStartByColor(line)) {
102               final int idx1 = line.indexOf('#');
103               final int idx2 = line.indexOf('>');
104               if (idx2 == -1)
105                   throw new IllegalStateException();
106   
107               final String[] color = line.substring(idx1, idx2).split(",");
108               if (idx < color.length) {
109                   final String s = color[idx];
110                   return s == null ? null : skinParam.getIHtmlColorSet().getColorOrWhite(s);
111               }
112           }
113           return null;
114       }
115   
116       private String withouBackColor(String line) {
117           final int idx2 = line.indexOf('>');
118           if (idx2 == -1)
119               throw new IllegalStateException();
120   
121           return line.substring(idx2 + 1);
122       }

124       private static final String hiddenBar = "\uE000";
125   
126       private void analyzeAndAddInternal(String line) {
127           line = line.replace("\\|", hiddenBar);
128           HColor lineBackColor = getBackOrFrontColor(line, 0);
129           if (lineBackColor != null)
130               line = withouBackColor(line);
131   
132           table.newLine(lineBackColor);
133           for (final StringTokenizer st = new StringTokenizer(line, "|"); st.hasMoreTokens();) {
134               Mode mode = Mode.NORMAL;
135               String v = st.nextToken().replace(hiddenBar.charAt(0), '|');
136               if (v.startsWith("=")) {
137                   v = v.substring(1);
138                   mode = Mode.HEADER;
139               }
140               HColor cellBackColor = getBackOrFrontColor(v, 0);
141               if (cellBackColor != null)
142                   v = withouBackColor(v);
143   
144               final List<String> lines = getWithNewlinesInternal(v);
145               final List<StripeSimple> cells = new ArrayList<>();
146               for (String s : lines) {
147                   final StripeSimple cell = new StripeSimple(getFontConfiguration(mode), stripeStyle, new CreoleContext(),
148                           skinParam, CreoleMode.FULL);
149                   if (s.startsWith("<r>")) {
150                       cell.setCellAlignment(HorizontalAlignment.RIGHT);
151                       s = s.substring("<r>".length());
152                   }
153                   cell.analyzeAndAdd(s);
154                   cells.add(cell);
155               }
156               table.addCell(asAtom(cells, skinParam.getPadding()), cellBackColor);
157           }
158       }

160       static List<String> getWithNewlinesInternal(String s) {
161           final List<String> result = new ArrayList<>();
162           final StringBuilder current = new StringBuilder();
163           for (int i = 0; i < s.length(); i++) {
164               final char c = s.charAt(i);
165               if (c == '\\' && i < s.length() - 1) {
166                   final char c2 = s.charAt(i + 1);
167                   i++;
168                   if (c2 == 'n') {
169                       result.add(current.toString());
170                       current.setLength(0);
171                   } else if (c2 == '\\') {
172                       current.append(c2);
173                   } else {
174                       current.append(c);
175                       current.append(c2);
176                   }
177               } else if (c == BackSlash.hiddenNewLine()) {
178                   result.add(current.toString());
179                   current.setLength(0);
180               } else {
181                   current.append(c);
182               }
183           }
184           result.add(current.toString());
185           return result;
186       }

188       private FontConfiguration getFontConfiguration(Mode mode) {
189           if (mode == Mode.NORMAL)
190               return fontConfiguration;
191   
192           return fontConfiguration.bold();
193       }

195       public void analyzeAndAddLine(String line) {
196           analyzeAndAddInternal(line);
