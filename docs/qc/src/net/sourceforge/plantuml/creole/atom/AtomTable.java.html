89       public XDimension2D calculateDimension(StringBounder stringBounder) {
90           initMap(stringBounder);
91           final double width = getEndingX(getNbCols() - 1);
92           final double height = getEndingY(getNbLines() - 1);
93           return new XDimension2D(width, height);
94       }

100       public void drawU(UGraphic ug) {
101           initMap(ug.getStringBounder());
102           for (int i = 0; i < getNbLines(); i++) {
103               final Line line = lines.get(i);
104               if (line.lineBackColor != null) {
105                   final double y1 = getStartingY(i);
106                   final double y2 = getStartingY(i + 1);
107                   final double x1 = getStartingX(0);
108                   final double x2 = getStartingX(getNbCols());
109                   ug.apply(HColors.none()).apply(line.lineBackColor.bg()).apply(new UTranslate(x1, y1))
110                           .draw(new URectangle(x2 - x1, y2 - y1));
111               }
112               for (int j = 0; j < getNbCols(); j++) {
113                   if (j >= line.cells.size()) {
114                       continue;
115                   }
116                   final Atom cell = line.cells.get(j);
117                   HorizontalAlignment align = HorizontalAlignment.LEFT;
118                   if (cell instanceof SheetBlock1) {
119                       align = ((SheetBlock1) cell).getCellAlignment();
120                   }
121                   final HColor cellBackColor = line.cellsBackColor.get(j);
122                   final double x1 = getStartingX(j);
123                   final double x2 = getStartingX(j + 1);
124                   final double cellWidth = x2 - x1;
125                   if (cellBackColor != null) {
126                       final double y1 = getStartingY(i);
127                       final double y2 = getStartingY(i + 1);
128                       ug.apply(HColors.none()).apply(cellBackColor.bg()).apply(new UTranslate(x1, y1))
129                               .draw(new URectangle(x2 - x1, y2 - y1));
130                   }
131                   final Position pos = positions.get(cell);
132                   final XDimension2D dimCell = cell.calculateDimension(ug.getStringBounder());
133                   final double dx;
134                   if (align == HorizontalAlignment.RIGHT) {
135                       dx = cellWidth - dimCell.getWidth();
136                   } else {
137                       dx = 0;
138                   }
139                   if (cellBackColor == null)
140                       cell.drawU(ug.apply(pos.getTranslate().compose(UTranslate.dx(dx))));
141                   else
142                       cell.drawU(ug.apply(cellBackColor.bg()).apply(pos.getTranslate().compose(UTranslate.dx(dx))));
143               }
144           }
145           ug = ug.apply(lineColor);
146           final ULine hline = ULine.hline(getEndingX(getNbCols() - 1));
147           for (int i = 0; i <= getNbLines(); i++) {
148               ug.apply(UTranslate.dy(getStartingY(i))).draw(hline);
149           }
150           final ULine vline = ULine.vline(getEndingY(getNbLines() - 1));
151           for (int i = 0; i <= getNbCols(); i++) {
152               ug.apply(UTranslate.dx(getStartingX(i))).draw(vline);
153           }
154   
155       }

157       private void initMap(StringBounder stringBounder) {
158           if (positions.size() > 0) {
159               return;
160           }
161           for (Line line : lines) {
162               for (Atom cell : line.cells) {
163                   final XDimension2D dim = cell.calculateDimension(stringBounder);
164                   final Position pos = new Position(0, 0, dim);
165                   positions.put(cell, pos);
166               }
167           }
168           for (int i = 0; i < lines.size(); i++) {
169               for (int j = 0; j < lines.get(i).size(); j++) {
170                   final Atom cell = lines.get(i).cells.get(j);
171                   final XDimension2D dim = cell.calculateDimension(stringBounder);
172                   final double x = getStartingX(j);
173                   final double y = getStartingY(i);
174                   final Position pos = new Position(x, y, dim);
175                   positions.put(cell, pos);
176               }
177           }
178       }

180       private double getStartingX(int col) {
181           double result = 0;
182           for (int i = 0; i < col; i++) {
183               result += getColWidth(i);
184           }
185           return result;
186       }
187   
188       private double getEndingX(int col) {
189           double result = 0;
190           for (int i = 0; i <= col; i++) {
191               result += getColWidth(i);
192           }
193           return result;
194       }
195   
196       private double getStartingY(int line) {
197           double result = 0;
198           for (int i = 0; i < line; i++) {
199               result += getLineHeight(i);
200           }
201           return result;
202       }
203   
204       private double getEndingY(int line) {
205           double result = 0;
206           for (int i = 0; i <= line; i++) {
207               result += getLineHeight(i);
208           }
209           return result;
210       }
211   
212       private double getColWidth(int col) {
213           double result = 0;
214           for (int i = 0; i < getNbLines(); i++) {
215               final Position position = getPosition(i, col);
216               if (position == null) {
217                   continue;
218               }
219               final double width = position.getWidth();
220               result = Math.max(result, width);
221           }
222           return result;
223       }
224   
225       private double getLineHeight(int line) {
226           double result = 0;
227           for (int i = 0; i < getNbCols(); i++) {
228               final Position position = getPosition(line, i);
229               if (position == null) {
230                   continue;
231               }
232               final double height = position.getHeight();
233               result = Math.max(result, height);
234           }
235           return result;
236       }
237   
238       private Position getPosition(int line, int col) {
239           if (line >= lines.size()) {
240               return null;
241           }
242           final Line l = lines.get(line);
243           if (col >= l.cells.size()) {
244               return null;
245           }
246           final Atom atom = l.cells.get(col);
247           return positions.get(atom);
248       }
249   
250       private int getNbCols() {
251           return lines.get(0).size();
252       }
253   
254       private int getNbLines() {
255           return lines.size();
256       }

260       }

264           positions.clear();
265       }
