60   public class PSystemDitaa extends AbstractPSystem {
61   
62       // private ProcessingOptions processingOptions;
63       private Object processingOptions;
64       private final boolean dropShadows;
65       private final String data;
66       private final float scale;
67       private final boolean transparentBackground;
68       private final Font font;
69       private final boolean forceFontSize;
70       private final boolean performSeparationOfCommonEdges;
71       private final boolean allCornersAreRound;
72   
73       public PSystemDitaa(UmlSource source, String data, boolean performSeparationOfCommonEdges, boolean dropShadows,
74               boolean allCornersAreRound, boolean transparentBackground, float scale, Font font, boolean forceFontSize) {
75           super(source);
76           this.data = data;
77           this.dropShadows = dropShadows;
78           this.performSeparationOfCommonEdges = performSeparationOfCommonEdges;
79           this.allCornersAreRound = allCornersAreRound;
80           try {
81               this.processingOptions = Class.forName("org.stathissideris.ascii2image.core.ProcessingOptions")
82                       .newInstance();
83               // this.processingOptions.setPerformSeparationOfCommonEdges(performSeparationOfCommonEdges);
84               this.processingOptions.getClass().getMethod("setPerformSeparationOfCommonEdges", boolean.class)
85                       .invoke(this.processingOptions, performSeparationOfCommonEdges);
86               this.processingOptions.getClass().getMethod("setAllCornersAreRound", boolean.class)
87                       .invoke(this.processingOptions, allCornersAreRound);
88           } catch (Exception e) {
89               Logme.error(e);
90               this.processingOptions = null;
91           }
92           this.transparentBackground = transparentBackground;
93           this.scale = scale;
94           this.font = font;
95           this.forceFontSize = forceFontSize;
96       }
97   
98       PSystemDitaa add(String line) {
99           return new PSystemDitaa(getSource(), data + line + BackSlash.NEWLINE, performSeparationOfCommonEdges,
100                   dropShadows, allCornersAreRound, transparentBackground, scale, font, forceFontSize);
101       }
102   
103       public DiagramDescription getDescription() {
104           return new DiagramDescription("(Ditaa)");
105       }
106   
107       @Override
108       final protected ImageData exportDiagramNow(OutputStream os, int num, FileFormatOption fileFormat)
109               throws IOException {
110           if (fileFormat.getFileFormat() == FileFormat.ATXT) {
111               os.write(getSource().getPlainString().getBytes());
112               return ImageDataSimple.ok();
113           }
114   
115           // ditaa can only export png so file format is mostly ignored
116           try {
117               // ditaa0_9.jar
118               // final ConversionOptions options = new ConversionOptions();
119               final Object options = Class.forName("org.stathissideris.ascii2image.core.ConversionOptions").newInstance();
120   
121               // final RenderingOptions renderingOptions = options.renderingOptions;
122               final Field f_renderingOptions = options.getClass().getField("renderingOptions");
123               final Object renderingOptions = f_renderingOptions.get(options);
124   
125               // renderingOptions.setBackgroundColor(font);
126               final Method setBackgroundColor = renderingOptions.getClass().getMethod("setBackgroundColor", Color.class);
127               setBackgroundColor.invoke(renderingOptions, transparentBackground ? new Color(0, 0, 0, 0) : Color.WHITE);
128   
129               // renderingOptions.setFont(font);
130               final Method setFont = renderingOptions.getClass().getMethod("setFont", Font.class);
131               setFont.invoke(renderingOptions, font);
132   
133               // renderingOptions.setForceFontSize(font);
134               final Method setForceFontSize = renderingOptions.getClass().getMethod("setForceFontSize", boolean.class);
135               setForceFontSize.invoke(renderingOptions, forceFontSize);
136   
137               // renderingOptions.setScale(scale);
138               final Method setScale = renderingOptions.getClass().getMethod("setScale", float.class);
139               setScale.invoke(renderingOptions, scale);
140   
141               // options.setDropShadows(dropShadows);
142               final Method setDropShadows = options.getClass().getMethod("setDropShadows", boolean.class);
143               setDropShadows.invoke(options, dropShadows);
144   
145               // final TextGrid grid = new TextGrid();
146               final Object grid = Class.forName("org.stathissideris.ascii2image.text.TextGrid").newInstance();
147   
148               // grid.initialiseWithText(data, null);
149               final Method initialiseWithText = grid.getClass().getMethod("initialiseWithText", String.class,
150                       Class.forName("org.stathissideris.ascii2image.core.ProcessingOptions"));
151               initialiseWithText.invoke(grid, data, null);
152   
153               // final Diagram diagram = new Diagram(grid, options, processingOptions);
154               final Class<?> clDiagram = Class.forName("org.stathissideris.ascii2image.graphics.Diagram");
155               clDiagram.getConstructor(grid.getClass(), options.getClass(), processingOptions.getClass())
156                       .newInstance(grid, options, processingOptions);
157               final Object diagram = clDiagram
158                       .getConstructor(grid.getClass(), options.getClass(), processingOptions.getClass())
159                       .newInstance(grid, options, processingOptions);
160   
161               // final BitmapRenderer bitmapRenderer = new BitmapRenderer();
162               final Object bitmapRenderer = Class.forName("org.stathissideris.ascii2image.graphics.BitmapRenderer")
163                       .newInstance();
164   
165               // final BufferedImage image = (BufferedImage)
166               // bitmapRenderer.renderToImage(diagram, renderingOptions);
167               final Method renderToImage = bitmapRenderer.getClass().getMethod("renderToImage", diagram.getClass(),
168                       renderingOptions.getClass());
169               final BufferedImage image = (BufferedImage) renderToImage.invoke(bitmapRenderer, diagram, renderingOptions);
170   
171               SImageIO.write(image, "png", os);
172               final int width = image.getWidth();
173               final int height = image.getHeight();
174               return new ImageDataSimple(width, height);
175           } catch (Throwable e) {
176               final List<String> strings = new ArrayList<>();
177               strings.add("DITAA has crashed");
178               strings.add(" ");
179               GraphvizCrash.youShouldSendThisDiagram(strings);
180               strings.add(" ");
181               UmlDiagram.exportDiagramError(os, e, new FileFormatOption(FileFormat.PNG), seed(), null, null, strings);
182               return ImageDataSimple.error();
183           }
184   
185       }
