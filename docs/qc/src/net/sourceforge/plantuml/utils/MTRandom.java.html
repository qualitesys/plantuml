37    * <p>
38    * A Java implementation of the MT19937 (Mersenne Twister) pseudo
39    * random number generator algorithm based upon the original C code
40    * by Makoto Matsumoto and Takuji Nishimura (see
41    * <a href="http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/emt.html">
42    * http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/emt.html</a> for
43    * more information.
44    * <p>
45    * As a subclass of java.util.Random this class provides a single
46    * canonical method next() for generating bits in the pseudo random
47    * number sequence.  Anyone using this class should invoke the public
48    * inherited methods (nextInt(), nextFloat etc.) to obtain values as
49    * normal.  This class should provide a drop-in replacement for the
50    * standard implementation of java.util.Random with the additional
51    * advantage of having a far longer period and the ability to use a
52    * far larger seed value.
53    * <p>
54    * This is <b>not</b> a cryptographically strong source of randomness
55    * and should <b>not</b> be used for cryptographic systems or in any
56    * other situation where true random numbers are required.

101       private final static int N = 624;
102       private final static int M = 397;
103       private final static int MAGIC[] = { 0x0, 0x9908b0df };
104       private final static int MAGIC_FACTOR1 = 1812433253;
105       private final static int MAGIC_FACTOR2 = 1664525;

128       /**
129        * This version of the constructor can be used to implement identical
130        * behaviour to the original C code version of this algorithm including
131        * exactly replicating the case where the seed value had not been set
132        * prior to calling genrand_int32.
133        * <p>
134        * If the compatibility flag is set to true, then the algorithm will be
135        * seeded with the same default value as was used in the original C
136        * code.  Furthermore the setSeed() method, which must take a 64 bit
137        * long value, will be limited to using only the lower 32 bits of the
138        * seed to facilitate seamless migration of existing C code into Java
139        * where identical behaviour is required.
140        * <p>
141        * Whilst useful for ensuring backwards compatibility, it is advised
142        * that this feature not be used unless specifically required, due to
143        * the reduction in strength of the seed value.
144        * 
145        * @param compatible Compatibility flag for replicating original
146        * behaviour.
147        */

151           setSeed(compat?DEFAULT_SEED:System.currentTimeMillis());

193       // Initializes mt[N] with a simple integer seed. This method is
194       // required as part of the Mersenne Twister algorithm but need
195       // not be made public.
196       private final void setSeed(int seed) {
197   
198           // Annoying runtime check for initialisation of internal data
199           // caused by java.util.Random invoking setSeed() during init.
200           // This is unavoidable because no fields in our instance will
201           // have been initialised at this point, not even if the code
202           // were placed at the declaration of the member variable.
203           if (mt == null) mt = new int[N];
204   
205           // ---- Begin Mersenne Twister Algorithm ----
206           mt[0] = seed;
207           for (mti = 1; mti < N; mti++) {
208               mt[mti] = (MAGIC_FACTOR1 * (mt[mti-1] ^ (mt[mti-1] >>> 30)) + mti);
209           }
210           // ---- End Mersenne Twister Algorithm ----
211       }
212   
213       /**
214        * This method resets the state of this instance using the 64
215        * bits of seed data provided.  Note that if the same seed data
216        * is passed to two different instances of MTRandom (both of
217        * which share the same compatibility state) then the sequence
218        * of numbers generated by both instances will be identical.
219        * <p>
220        * If this instance was initialised in 'compatibility' mode then
221        * this method will only use the lower 32 bits of any seed value
222        * passed in and will match the behaviour of the original C code
223        * exactly with respect to state initialisation.
224        * 
225        * @param seed The 64 bit value used to initialise the random
226        * number generator state. 
227        */
228       public final synchronized void setSeed(long seed) {
229           if (compat) {
230               setSeed((int)seed);
231           } else {
232   
233               // Annoying runtime check for initialisation of internal data
234               // caused by java.util.Random invoking setSeed() during init.
235               // This is unavoidable because no fields in our instance will
236               // have been initialised at this point, not even if the code
237               // were placed at the declaration of the member variable.
238               if (ibuf == null) ibuf = new int[2];
239   
240               ibuf[0] = (int)seed;
241               ibuf[1] = (int)(seed >>> 32);
242               setSeed(ibuf);
243           }
244       }
245   
246       /**
247        * This method resets the state of this instance using the byte
248        * array of seed data provided.  Note that calling this method
249        * is equivalent to calling "setSeed(pack(buf))" and in particular
250        * will result in a new integer array being generated during the
251        * call.  If you wish to retain this seed data to allow the pseudo
252        * random sequence to be restarted then it would be more efficient
253        * to use the "pack()" method to convert it into an integer array
254        * first and then use that to re-seed the instance.  The behaviour
255        * of the class will be the same in both cases but it will be more
256        * efficient.
257        *
258        * @param buf The non-empty byte array of seed information.
259        * @throws NullPointerException if the buffer is null.
260        * @throws IllegalArgumentException if the buffer has zero length.
261        */
262       public final void setSeed(byte[] buf) {
263           setSeed(pack(buf));
264       }

266       /**
267        * This method resets the state of this instance using the integer
268        * array of seed data provided.  This is the canonical way of
269        * resetting the pseudo random number sequence.
270        * 
271        * @param buf The non-empty integer array of seed information.
272        * @throws NullPointerException if the buffer is null.
273        * @throws IllegalArgumentException if the buffer has zero length.
274        */
275       public final synchronized void setSeed(int[] buf) {
276           int length = buf.length;
277           if (length == 0) throw new IllegalArgumentException("Seed buffer may not be empty");
278           // ---- Begin Mersenne Twister Algorithm ----
279           int i = 1, j = 0, k = (N > length ? N : length);
280           setSeed(MAGIC_SEED);
281           for (; k > 0; k--) {
282               mt[i] = (mt[i] ^ ((mt[i-1] ^ (mt[i-1] >>> 30)) * MAGIC_FACTOR2)) + buf[j] + j;
283               i++; j++;
284               if (i >= N) { mt[0] = mt[N-1]; i = 1; }
285               if (j >= length) j = 0;
286           }
287           for (k = N-1; k > 0; k--) {
288               mt[i] = (mt[i] ^ ((mt[i-1] ^ (mt[i-1] >>> 30)) * MAGIC_FACTOR3)) - i;
289               i++;
290               if (i >= N) { mt[0] = mt[N-1]; i = 1; }
291           }
292           mt[0] = UPPER_MASK; // MSB is 1; assuring non-zero initial array
293           // ---- End Mersenne Twister Algorithm ----
294       }
295   
296       /**
297        * This method forms the basis for generating a pseudo random number
298        * sequence from this class.  If given a value of 32, this method
299        * behaves identically to the genrand_int32 function in the original
300        * C code and ensures that using the standard nextInt() function
301        * (inherited from Random) we are able to replicate behaviour exactly.
302        * <p>
303        * Note that where the number of bits requested is not equal to 32
304        * then bits will simply be masked out from the top of the returned
305        * integer value.  That is to say that:
306        * <pre>
307        * mt.setSeed(12345);
308        * int foo = mt.nextInt(16) + (mt.nextInt(16) &lt;&lt; 16);</pre>
309        * will not give the same result as
310        * <pre>
311        * mt.setSeed(12345);
312        * int foo = mt.nextInt(32);</pre>
313        * 
314        * @param bits The number of significant bits desired in the output.
315        * @return The next value in the pseudo random sequence with the
316        * specified number of bits in the lower part of the integer.
317        */
318       protected final synchronized int next(int bits) {
319           // ---- Begin Mersenne Twister Algorithm ----
320           int y, kk;
321           if (mti >= N) {             // generate N words at one time
322   
323               // In the original C implementation, mti is checked here
324               // to determine if initialisation has occurred; if not
325               // it initialises this instance with DEFAULT_SEED (5489).
326               // This is no longer necessary as initialisation of the
327               // Java instance must result in initialisation occurring
328               // Use the constructor MTRandom(true) to enable backwards
329               // compatible behaviour.
330               
331               for (kk = 0; kk < N-M; kk++) {
332                   y = (mt[kk] & UPPER_MASK) | (mt[kk+1] & LOWER_MASK);
333                   mt[kk] = mt[kk+M] ^ (y >>> 1) ^ MAGIC[y & 0x1];
334               }
335               for (;kk < N-1; kk++) {
336                   y = (mt[kk] & UPPER_MASK) | (mt[kk+1] & LOWER_MASK);
337                   mt[kk] = mt[kk+(M-N)] ^ (y >>> 1) ^ MAGIC[y & 0x1];
338               }
339               y = (mt[N-1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
340               mt[N-1] = mt[M-1] ^ (y >>> 1) ^ MAGIC[y & 0x1];
341   
342               mti = 0;
343           }
344     
345           y = mt[mti++];
346   
347           // Tempering
348           y ^= (y >>> 11);
349           y ^= (y << 7) & MAGIC_MASK1;
350           y ^= (y << 15) & MAGIC_MASK2;
351           y ^= (y >>> 18);
352           // ---- End Mersenne Twister Algorithm ----
353           return (y >>> (32-bits));
354       }

356       // This is a fairly obscure little code section to pack a
357       // byte[] into an int[] in little endian ordering.  
358   
359       /**
360        * This simply utility method can be used in cases where a byte
361        * array of seed data is to be used to repeatedly re-seed the
362        * random number sequence.  By packing the byte array into an
363        * integer array first, using this method, and then invoking
364        * setSeed() with that; it removes the need to re-pack the byte
365        * array each time setSeed() is called.
366        * <p>
367        * If the length of the byte array is not a multiple of 4 then
368        * it is implicitly padded with zeros as necessary.  For example:
369        * <pre>    byte[] { 0x01, 0x02, 0x03, 0x04, 0x05, 0x06 }</pre>
370        * becomes
371        * <pre>    int[]  { 0x04030201, 0x00000605 }</pre>
372        * <p>
373        * Note that this method will not complain if the given byte array
374        * is empty and will produce an empty integer array, but the
375        * setSeed() method will throw an exception if the empty integer
376        * array is passed to it.
377        * 
378        * @param buf The non-null byte array to be packed.
379        * @return A non-null integer array of the packed bytes.
