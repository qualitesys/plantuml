205       List<String> cleanForKey(String key) {
206           List<String> result = cacheCleanForKey.get(key);
207           if (result == null) {
208               result = cleanForKeySlow(key);
209               cacheCleanForKey.put(key, result);

214       private static final Pattern patternCleanUnderscoreDot = Pattern.compile("_|\\.");
215       private static final Pattern patternCleanSequence = Pattern.compile("sequence(participant|actor)");
216       private static final Pattern patternCleanArrow = Pattern
217               .compile("(activity|class|component|object|sequence|state|usecase)arrow");
218       private static final Pattern patternCleanAlign = Pattern.compile("align$");
219   
220       List<String> cleanForKeySlow(String key) {
221           key = StringUtils.trin(StringUtils.goLowerCase(key));
222           key = patternCleanUnderscoreDot.matcher(key).replaceAll("");
223           key = patternCleanSequence.matcher(key).replaceAll("$1");
224           key = patternCleanArrow.matcher(key).replaceAll("arrow");
225           key = patternCleanAlign.matcher(key).replaceAll("alignment");
226   
227           final Matcher2 mm = stereoPattern.matcher(key);
228           final List<String> result = new ArrayList<>();
229           while (mm.find()) {
230               final String s = mm.group(1);
231               result.add(key.replaceAll(stereoPatternString, "") + "<<" + s + ">>");
232           }
233           if (result.size() == 0)
234               result.add(key);
235   
236           return Collections.unmodifiableList(result);
237       }
