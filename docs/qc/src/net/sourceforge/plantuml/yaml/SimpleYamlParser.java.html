35   package net.sourceforge.plantuml.yaml;
36   
37   import java.util.ArrayList;
38   import java.util.List;
39   import java.util.regex.Matcher;
40   import java.util.regex.Pattern;
41   
42   import net.sourceforge.plantuml.StringUtils;
43   import net.sourceforge.plantuml.json.JsonArray;
44   import net.sourceforge.plantuml.json.JsonObject;
45   import net.sourceforge.plantuml.json.JsonString;
46   import net.sourceforge.plantuml.json.JsonValue;
47   
48   public class SimpleYamlParser {
49   
50       private JsonValue result;
51       private final List<Integer> pendingIndents = new ArrayList<>();
52   
53       public JsonValue parse(List<String> lines) {
54           result = new JsonObject();
55           pendingIndents.clear();
56           pendingIndents.add(0);
57           final YamlLines yamlLines = new YamlLines(lines);
58           for (String s : yamlLines) {
59               parseSingleLine(s);
60           }
61           return result;
62   
63       }
64   
65       private String[] nameAndValue(String s) {
66           final Pattern p1 = Pattern.compile("^\\s*" + YamlLines.KEY + "\\s*: \\s*(\\S.*)$");
67           final Matcher m1 = p1.matcher(s);
68           if (m1.matches()) {
69               final String name = m1.group(1);
70               final String data = m1.group(2).trim();
71               return new String[] { name, data };
72           }
73           return null;
74       }
75   
76       private void parseSingleLine(String s) {
77           final int indent = getIndent(s);
78   //        System.err.println("s=" + s);
79   
80           if (isListStrict(s)) {
81               strictMuteToArray(indent);
82               return;
83           }
84   
85           final String listedValue = listedValue(s);
86           if (listedValue != null) {
87               final JsonArray array = getForceArray(indent);
88               array.add(listedValue);
89               return;
90           }
91   
92           final JsonObject working = (JsonObject) getWorking(indent);
93           if (working == null) {
94               System.err.println("ERROR: ignoring " + s);
95               return;
96           }
97   
98           final String[] nameAndValue = nameAndValue(s);
99           if (nameAndValue != null) {
100               final String name = nameAndValue[0];
101               final String data = StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(nameAndValue[1], "\"");
102               working.add(name, data);
103               return;
104           }
105   
106           final String nameOnly = YamlLines.nameOnly(s);
107           if (nameOnly != null) {
108               working.add(nameOnly, new JsonObject());
109               return;
110           }
111   
112           throw new UnsupportedOperationException(s);
113       }
114   
115       private String listedValue(String s) {
116           final Pattern p1 = Pattern.compile("^\\s*[-]\\s*(\\S.*)$");
117           final Matcher m1 = p1.matcher(s);
118           if (m1.matches()) {
119               final String name = m1.group(1).trim();
120               return name;
121           }
122           return null;
123       }
124   
125       private JsonArray getForceArray(int indent) {
126           if (indent == 0 && getLastIndent() == 0) {
127               if (result instanceof JsonArray == false) {
128                   result = new JsonArray();
129               }
130               return (JsonArray) result;
131           }
132           while (getLastIndent() > indent - 1)
133               pendingIndents.remove(pendingIndents.size() - 1);
134   
135           final JsonObject last = (JsonObject) search(result, pendingIndents.size());
136           final String field = last.names().get(last.size() - 1);
137           if (last.get(field) instanceof JsonArray == false) {
138               last.set(field, new JsonArray());
139           }
140           return (JsonArray) last.get(field);
141       }
142   
143       private void strictMuteToArray(int indent) {
144           if (indent == 0 && getLastIndent() == 0) {
145               if (result instanceof JsonArray == false) {
146                   result = new JsonArray();
147               }
148               return;
149           }
150           while (getLastIndent() > indent)
151               pendingIndents.remove(pendingIndents.size() - 1);
152   
153           if (result instanceof JsonArray) {
154               ((JsonArray) result).add(new JsonObject());
155               return;
156           }
157   
158           final JsonObject last = (JsonObject) search(result, pendingIndents.size());
159           final String field = last.names().get(last.size() - 1);
160           if (last.get(field) instanceof JsonArray == false) {
161               last.set(field, new JsonArray());
162           } else {
163               ((JsonArray) last.get(field)).add(new JsonObject());
164           }
165       }
166   
167       private boolean isListStrict(String s) {
168           return s.trim().equals("-");
169       }
170   
171       private int getLastIndent() {
172           return pendingIndents.get(pendingIndents.size() - 1);
173       }
174   
175       private JsonValue getWorking(int indent) {
176           if (indent > getLastIndent()) {
177               pendingIndents.add(indent);
178               return search(result, pendingIndents.size());
179           }
180           if (indent == getLastIndent()) {
181               return search(result, pendingIndents.size());
182           }
183           final int idx = pendingIndents.indexOf(indent);
184           if (idx == -1) {
185               return null;
186           }
187   
188           while (pendingIndents.size() > idx + 1)
189               pendingIndents.remove(pendingIndents.size() - 1);
190   
191           return search(result, pendingIndents.size());
192       }
193   
194       private static JsonValue search(JsonValue current1, int size) {
195           if (current1 instanceof JsonArray) {
196               JsonArray array = (JsonArray) current1;
197   
198               final JsonValue tmp;
199               if (array.size() == 0) {
200                   tmp = new JsonObject();
201                   array.add(tmp);
202               } else {
203                   tmp = array.get(array.size() - 1);
204               }
205               return tmp;
206           }
207           if (size <= 1) {
208               return current1;
209           }
210   
211           final JsonObject current = (JsonObject) current1;
212           final String last = current.names().get(current.size() - 1);
213           // System.err.println("last=" + last);
214           JsonValue tmp = current.get(last);
215           if (tmp instanceof JsonArray) {
216               JsonArray array = (JsonArray) tmp;
217               if (array.size() == 0) {
218                   tmp = new JsonObject();
219                   array.add(tmp);
220               } else {
221                   tmp = array.get(array.size() - 1);
222               }
223           }
224           if (tmp instanceof JsonString) {
225               System.err.println("JsonString? " + tmp);
226               return null;
227           }
228           return search(tmp, size - 1);
229       }
230   
231       private int getIndent(String s) {
232           int indent = 0;
233           for (int i = 0; i < s.length(); i++) {
234               final char ch = s.charAt(i);
235               if (ch == ' ' || ch == '\t') {
236                   indent++;
237               } else {
238                   return indent;
239               }
240           }
241           return 0;
242       }
