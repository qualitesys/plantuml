35   package net.sourceforge.plantuml.yaml;
36   
37   import java.util.ArrayList;
38   import java.util.Collections;
39   import java.util.Iterator;
40   import java.util.List;
41   import java.util.ListIterator;
42   import java.util.regex.Matcher;
43   import java.util.regex.Pattern;
44   
45   public class YamlLines implements Iterable<String> {
46   
47       public static final String KEY = "([^:]+)";
48   
49       private List<String> lines = new ArrayList<>();
50   
51       public YamlLines(List<String> rawLines) {
52           for (String s : rawLines) {
53               if (s.startsWith("#"))
54                   continue;
55   
56               s = removeDiese(s);
57               if (s.trim().length() == 0)
58                   continue;
59               lines.add(s);
60           }
61           mergeMultiline();
62           manageSpaceTwoPoints();
63           manageList();
64           final int startingEmptyCols = startingEmptyCols();
65           removeFirstCols(startingEmptyCols);
66       }
67   
68       private void manageSpaceTwoPoints() {
69           for (ListIterator<String> it = lines.listIterator(); it.hasNext();) {
70               String s = it.next();
71               if (s.contains("\"") == false && s.contains("'") == false && s.contains(":")
72                       && s.indexOf(':') == s.lastIndexOf(':') && s.contains(": ") == false) {
73                   s = s.replace(":", ": ");
74                   it.set(s);
75               }
76           }
77   
78       }
79   
80       private String removeDiese(String s) {
81           final int idx = s.indexOf(" #");
82           if (idx == -1)
83               return s;
84   
85           return s.substring(0, idx);
86       }
87   
88       private void manageList() {
89           final List<String> result = new ArrayList<>();
90           for (String s : lines) {
91               final Pattern p1 = Pattern.compile("^(\\s*[-])(\\s*\\S.*)$");
92               final Matcher m1 = p1.matcher(s);
93               if (s.contains(": ") && m1.matches()) {
94                   result.add(m1.group(1));
95                   result.add(s.replaceFirst("[-]", " "));
96               } else if (m1.matches()) {
97                   result.add(" " + s);
98               } else {
99                   result.add(s);
100               }
101   
102           }
103           this.lines = result;
104       }
105   
106       private void removeFirstCols(int startingEmptyCols) {
107           if (startingEmptyCols == 0)
108               return;
109   
110           for (ListIterator<String> it = lines.listIterator(); it.hasNext();) {
111               final String s = it.next().substring(startingEmptyCols);
112               it.set(s);
113           }
114       }
115   
116       private int startingEmptyCols() {
117           int result = Integer.MAX_VALUE;
118           for (String s : lines) {
119               result = Math.min(result, startingSpaces(s));
120               if (result == 0)
121                   return 0;
122           }
123           return result;
124       }
125   
126       private static int startingSpaces(String s) {
127           final Pattern p1 = Pattern.compile("^(\\s*).*");
128           final Matcher m1 = p1.matcher(s);
129           if (m1.matches())
130               return m1.group(1).length();
131   
132           return 0;
133       }
134   
135       private void mergeMultiline() {
136           final List<String> result = new ArrayList<>();
137           for (int i = 0; i < lines.size(); i++) {
138               final String init = isMultilineStart(i);
139               if (init != null) {
140                   final StringBuilder sb = new StringBuilder(init);
141                   while (i + 1 < lines.size() && textOnly(lines.get(i + 1))) {
142                       sb.append(" " + lines.get(i + 1).trim());
143                       i++;
144                   }
145                   result.add(sb.toString());
146               } else {
147                   result.add(lines.get(i));
148               }
149           }
150           this.lines = result;
151       }
152   
153       private String isMultilineStart(int i) {
154           if (nameOnly(lines.get(i)) != null && textOnly(lines.get(i + 1))) {
155               final int idx = lines.get(i).indexOf(':');
156               return lines.get(i).substring(0, idx + 1);
157           }
158           return null;
159       }
160   
161       public static String nameOnly(String s) {
162           final Pattern p1 = Pattern.compile("^\\s*" + KEY + "\\s*:\\s*[|>]?\\s*$");
163           final Matcher m1 = p1.matcher(s);
164           if (m1.matches()) {
165               final String name = m1.group(1);
166               return name;
167           }
168           return null;
169       }
170   
171       private boolean textOnly(String s) {
172           if (isList(s))
173               return false;
174           return s.indexOf(':') == -1;
175       }
176   
177       private boolean isList(String s) {
178           return s.trim().startsWith("-");
179       }
180   
181       public Iterator<String> iterator() {
182           return Collections.unmodifiableList(lines).iterator();
183       }
184   
185   }
