24   
25       static {

27           try (BufferedReader br = new BufferedReader(new InputStreamReader(is))) {
28               String s = null;

30                   new Emoji(s);

34           }

46       private Emoji(String unicode) {
47           final int x = unicode.indexOf(';');
48           if (x == -1) {
49               this.shortcut = null;
50           } else {
51               this.shortcut = unicode.substring(x + 1);
52               ALL.put(this.shortcut, this);
53               unicode = unicode.substring(0, x);
54           }
55           this.unicode = unicode;
56           ALL.put(unicode, this);
57       }

59       public static String pattern() {
60           final StringBuilder sb = new StringBuilder("\\<(#\\w+)?:(");
61           for (String s : ALL.keySet()) {
62               if (sb.toString().endsWith("(") == false)
63                   sb.append("|");
64               sb.append(s);
65           }
66           sb.append("):\\>");
67           return sb.toString();
68       }

79           try (BufferedReader br = new BufferedReader(
