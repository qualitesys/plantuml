41       private String extractData(String name, String s) {
42           final Pattern p = Pattern.compile(name + "=\"([^\"]+)\"");
43           final Matcher m = p.matcher(s);
44           if (m.find())
45               return m.group(1);
46   
47           return null;
48       }

54       public SvgNanoParser(List<String> svg, boolean keepColors) {
55           this.svgStart = svg.get(0);
56           this.keepColors = keepColors;
57   
58           for (String singleLine : svg) {
59               final Pattern p = Pattern
60                       .compile("(\\<text .*?\\</text\\>)|(\\<(svg|path|g|circle|ellipse)[^<>]*\\>)|(\\</[^<>]*\\>)");
61               final Matcher m = p.matcher(singleLine);
62               while (m.find()) {
63                   final String s = m.group(0);
64                   if (s.startsWith("<path") || s.startsWith("<g ") || s.startsWith("<g>") || s.startsWith("</g>")
65                           || s.startsWith("<circle ") || s.startsWith("<ellipse ") || s.startsWith("<text "))
66                       data.add(s);
67                   else if (s.startsWith("<svg") || s.startsWith("</svg")) {
68                       // Ignore
69                   } else
70                       System.err.println("???=" + s);
71               }
72           }
73       }

75       public void drawU(UGraphic ug, double scale, HColor colorForMonochrome) {
76           UGraphicWithScale ugs = new UGraphicWithScale(ug, scale);
77   
78           synchronized (this) {
79               if (colorForMonochrome != null && maxGray == -1)
80                   computeMinMaxGray();
81           }
82   
83           final List<UGraphicWithScale> stack = new ArrayList<>();
84           for (String s : data) {
85               if (s.startsWith("<path ")) {
86                   drawPath(ugs, s, colorForMonochrome);
87               } else if (s.startsWith("</g>")) {
88                   ugs = stack.remove(0);
89               } else if (s.startsWith("<g>")) {
90                   stack.add(0, ugs);
91               } else if (s.startsWith("<g ")) {
92                   stack.add(0, ugs);
93                   ugs = applyFill(ugs, s, colorForMonochrome);
94                   ugs = applyTransform(ugs, s);
95               } else if (s.startsWith("<circle ")) {
96                   drawCircle(ugs, s, colorForMonochrome);
97               } else if (s.startsWith("<ellipse ")) {
98                   drawEllipse(ugs, s, colorForMonochrome);
99               } else if (s.startsWith("<text ")) {
100                   drawText(ugs, s, colorForMonochrome);
101               } else {
102                   System.err.println("**?=" + s);
103               }
104           }
105       }

107       private void computeMinMaxGray() {
108           for (String s : data) {
109               if (s.contains("<path ") || s.contains("<g ") || s.contains("<circle ") || s.contains("<ellipse ")) {
110                   final HColor color = justExtractColor(s);
111                   if (color != null) {
112                       final int gray = getGray(color);
113                       minGray = Math.min(minGray, gray);
114                       maxGray = Math.max(maxGray, gray);
115                   }
116               } else {
117                   // Nothing
118               }
119           }
120       }

122       private int getGray(HColor col) {
123           final Color tmp = ColorUtils.getGrayScaleColor(col.toColor(ColorMapper.MONOCHROME));
124           return tmp.getGreen();
125       }

127       private UGraphicWithScale applyFill(UGraphicWithScale ugs, String s, HColor colorForMonochrome) {
128           final String fillString = extractData("fill", s);
129           if (fillString == null)
130               return ugs;
131   
132           if (fillString.equals("none")) {
133               final String strokeString = extractData("stroke", s);
134               if (strokeString == null)
135                   return ugs;
136               ugs = ugs.apply(HColors.none().bg());
137               final HColor stroke = getTrueColor(strokeString, colorForMonochrome);
138               ugs = ugs.apply(stroke);
139               final String strokeWidth = extractData("stroke-width", s);
140               if (strokeWidth != null)
141                   ugs = ugs.apply(new UStroke(Double.parseDouble(strokeWidth)));
142   
143           } else {
144               final HColor fill = getTrueColor(fillString, colorForMonochrome);
145               ugs = ugs.apply(fill).apply(fill.bg());
146           }
147   
148           return ugs;
149       }

151       private HColor justExtractColor(String s) {
152           final String fillString = extractData("fill", s);
153           if (fillString == null)
154               return null;
155   
156           if (fillString.equals("none")) {
157               final String strokeString = extractData("stroke", s);
158               if (strokeString == null)
159                   return null;
160   
161               final HColor stroke = getTrueColor(strokeString, null);
162               return stroke;
163   
164           } else {
165               final HColor fill = getTrueColor(fillString, null);
166               return fill;
167           }

171       private HColor getTrueColor(String code, HColor colorForMonochrome) {
172           final HColorSimple result = (HColorSimple) HColorSet.instance().getColorOrWhite(code);
173           if (colorForMonochrome == null)
174               return result;
175           final HColorSimple color = (HColorSimple) colorForMonochrome;
176           if (color.isGray())
177               return result.asMonochrome();
178           return result.asMonochrome(color, this.minGray, this.maxGray);
179       }

181       private void drawCircle(UGraphicWithScale ugs, String s, HColor colorForMonochrome) {
182           ugs = applyFill(ugs, s, colorForMonochrome);
183           ugs = applyTransform(ugs, s);
184   
185           final double scalex = ugs.getAffineTransform().getScaleX();
186           final double scaley = ugs.getAffineTransform().getScaleY();
187   
188           final double deltax = ugs.getAffineTransform().getTranslateX();
189           final double deltay = ugs.getAffineTransform().getTranslateY();
190   
191           final double cx = Double.parseDouble(extractData("cx", s)) * scalex;
192           final double cy = Double.parseDouble(extractData("cy", s)) * scaley;
193           final double rx = Double.parseDouble(extractData("r", s)) * scalex;
194           final double ry = Double.parseDouble(extractData("r", s)) * scaley;
195   
196           final UTranslate translate = new UTranslate(deltax + cx - rx, deltay + cy - ry);
197           ugs.apply(translate).draw(new UEllipse(rx * 2, ry * 2));
198       }

220       private void drawText(UGraphicWithScale ugs, String s, HColor colorForMonochrome) {
221           final double x = Double.parseDouble(extractData("x", s));
222           final double y = Double.parseDouble(extractData("y", s));
223           final String fill = extractData("fill", s);
224           final int fontSize = Integer.parseInt(extractData("font-size", s));
225   
226           final Pattern p = Pattern.compile("\\<text[^<>]*\\>(.*?)\\</text\\>");
227           final Matcher m = p.matcher(s);
228           if (m.find()) {
229               final String text = m.group(1);
230               HColor color = HColorSet.instance().getColorOrWhite(fill);
231               final FontConfiguration fc = FontConfiguration.create(UFont.sansSerif(fontSize), color, color, null);
232               final UText utext = new UText(text, fc);
233               UGraphic ug = ugs.getUg();
234               ug = ug.apply(new UTranslate(x, y));
235               ug.draw(utext);
236           }

242           ugs = applyTransform(ugs, s);
243

245           final int x2 = s.indexOf('"', x1 + 3);
246           final String tmp = s.substring(x1 + 3, x2);
247

249           svgPath.drawMe(ugs.getUg(), ugs.getAffineTransform());

253       private UGraphicWithScale applyTransform(UGraphicWithScale ugs, String s) {
254           final String transform = extractData("transform", s);
255           if (transform == null)
256               return ugs;
257   
258           if (transform.contains("rotate("))
259               return applyRotate(ugs, transform);
260   
261           if (transform.contains("matrix("))
262               return applyMatrix(ugs, transform);
263   
264           final double[] scale = getScale(transform);
265           final UTranslate translate = getTranslate(transform);
266           ugs = ugs.applyTranslate(translate.getDx(), translate.getDy());
267   
268           return ugs.applyScale(scale[0], scale[1]);
269       }

271       private UGraphicWithScale applyMatrix(UGraphicWithScale ugs, final String transform) {
272           final Pattern p3 = Pattern.compile(
273                   "matrix\\(([-.0-9]+)[ ,]+([-.0-9]+)[ ,]+([-.0-9]+)[ ,]+([-.0-9]+)[ ,]+([-.0-9]+)[ ,]+([-.0-9]+)\\)");
274           final Matcher m3 = p3.matcher(transform);
275           if (m3.find()) {
276               final double v1 = Double.parseDouble(m3.group(1));
277               final double v2 = Double.parseDouble(m3.group(2));
278               final double v3 = Double.parseDouble(m3.group(3));
279               final double v4 = Double.parseDouble(m3.group(4));
280               final double v5 = Double.parseDouble(m3.group(5));
281               final double v6 = Double.parseDouble(m3.group(6));
282               ugs = ugs.applyMatrix(v1, v2, v3, v4, v5, v6);
283           } else
284               System.err.println("WARNING: " + transform);
285           return ugs;
286       }

288       private UGraphicWithScale applyRotate(UGraphicWithScale ugs, final String transform) {
289           final Pattern p3 = Pattern.compile("rotate\\(([-.0-9]+)[ ,]+([-.0-9]+)[ ,]+([-.0-9]+)\\)");
290           final Matcher m3 = p3.matcher(transform);
291           if (m3.find()) {
292               final double angle = Double.parseDouble(m3.group(1));
293               final double x = Double.parseDouble(m3.group(2));
294               final double y = Double.parseDouble(m3.group(3));
295               ugs = ugs.applyRotate(angle, x, y);
296           } else
297               System.err.println("WARNING: " + transform);
298           return ugs;
299       }

301       private UTranslate getTranslate(String transform) {
302           double x = 0;
303           double y = 0;
304   
305           final Pattern p3 = Pattern.compile("translate\\(([-.0-9]+)[ ,]+([-.0-9]+)\\)");
306           final Matcher m3 = p3.matcher(transform);
307           if (m3.find()) {
308               x = Double.parseDouble(m3.group(1));
309               y = Double.parseDouble(m3.group(2));
310           } else {
311               final Pattern p4 = Pattern.compile("translate\\(([-.0-9]+)\\)");
312               final Matcher m4 = p4.matcher(transform);
313               if (m4.find()) {
314                   x = Double.parseDouble(m4.group(1));
315                   y = Double.parseDouble(m4.group(1));
316               }
317           }
318           return new UTranslate(x, y);
319       }

321       private double[] getScale(String transform) {
322           final double scale[] = new double[] { 1, 1 };
323           final Pattern p1 = Pattern.compile("scale\\(([-.0-9]+)\\)");
324           final Matcher m1 = p1.matcher(transform);
325           if (m1.find()) {
326               scale[0] = Double.parseDouble(m1.group(1));
327               scale[1] = scale[0];
328           } else {
329               final Pattern p2 = Pattern.compile("scale\\(([-.0-9]+)[ ,]+([-.0-9]+)\\)");
330               final Matcher m2 = p2.matcher(transform);
331               if (m2.find()) {
332                   scale[0] = Double.parseDouble(m2.group(1));
333                   scale[1] = Double.parseDouble(m2.group(2));
334               }
335           }
336           return scale;
337       }
