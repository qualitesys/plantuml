56   public class EmptyImageBuilder {
57   
58       private final BufferedImage im;
59       private final Graphics2D g2d;
60       private final Color background;
61       private final StringBounder stringBounder;
62   
63       private static EmptyImageBuilder create(String watermark, int width, int height, Color background,
64               StringBounder stringBounder, double dpiFactor) {
65           EmptyImageBuilder result = new EmptyImageBuilder(watermark, (int) (width * dpiFactor),
66                   (int) (height * dpiFactor), background, stringBounder);
67           if (dpiFactor != 1.0)
68               result.g2d.setTransform(AffineTransform.getScaleInstance(dpiFactor, dpiFactor));
69           return result;
70       }
71   
72       public EmptyImageBuilder(String watermark, int width, int height, Color background, StringBounder stringBounder) {
73           if (width <= 0 || height <= 0)
74               throw new IllegalArgumentException("width and height must be positive");
75   
76           if (width > GraphvizUtils.getenvImageLimit()) {
77               Log.info("Width too large " + width + ". You should set PLANTUML_LIMIT_SIZE");
78               width = GraphvizUtils.getenvImageLimit();
79           }
80           if (height > GraphvizUtils.getenvImageLimit()) {
81               Log.info("Height too large " + height + ". You should set PLANTUML_LIMIT_SIZE");
82               height = GraphvizUtils.getenvImageLimit();
83           }
84           this.background = background;
85           this.stringBounder = stringBounder;
86           Log.info("Creating image " + width + "x" + height);
87           im = new BufferedImage(width, height, getType(background));
88           g2d = im.createGraphics();
89           UAntiAliasing.ANTI_ALIASING_ON.apply(g2d);
90           if (background != null) {
91               g2d.setColor(background);
92               g2d.fillRect(0, 0, width, height);
93           }
94           if (watermark != null) {
95               final int gray = 200;
96               g2d.setColor(new Color(gray, gray, gray));
97               printWatermark(watermark, width, height);
98           }
99       }
100   
101       private int getType(Color background) {
102           if (background == null) {
103               return BufferedImage.TYPE_INT_ARGB;
104           }
105           if (background.getAlpha() != 255) {
106               return BufferedImage.TYPE_INT_ARGB;
107           }
108           return BufferedImage.TYPE_INT_RGB;
109       }
110   
111       private void printWatermark(String watermark, int maxWidth, int maxHeight) {
112           final Font javaFont = g2d.getFont();
113           final FontMetrics fm = g2d.getFontMetrics(javaFont);
114           final Rectangle2D rect = fm.getStringBounds(watermark, g2d);
115           final int height = (int) rect.getHeight();
116           final int width = (int) rect.getWidth();
117           if (height < 2 || width < 2) {
118               return;
119           }
120           if (width <= maxWidth)
121               for (int y = height; y < maxHeight; y += height + 1) {
122                   for (int x = 0; x < maxWidth; x += width + 10) {
123                       g2d.drawString(watermark, x, y);
124                   }
125               }
126           else {
127               final List<String> withBreaks = withBreaks(watermark, javaFont, fm, maxWidth);
128               int y = 0;
129               while (y < maxHeight) {
130                   for (String s : withBreaks) {
131                       g2d.drawString(s, 0, y);
132                       y += (int) fm.getStringBounds(s, g2d).getHeight();
133                   }
134                   y += 10;
135               }
136           }
137       }
138   
139       private int getWidth(String line, Font javaFont, FontMetrics fm) {
140           final Rectangle2D rect = fm.getStringBounds(line, g2d);
141           return (int) rect.getWidth();
142       }
143   
144       private List<String> withBreaks(String watermark, Font javaFont, FontMetrics fm, int maxWidth) {
145           final String[] words = watermark.split("\\s+");
146           final List<String> result = new ArrayList<>();
147           String pending = "";
148           for (String word : words) {
149               final String candidate = pending.length() == 0 ? word : pending + " " + word;
150               if (getWidth(candidate, javaFont, fm) < maxWidth) {
151                   pending = candidate;
152               } else {
153                   result.add(pending);
154                   pending = word;
155               }
156           }
157           if (pending.length() > 0) {
158               result.add(pending);
159           }
160           return result;
161       }

163       public BufferedImage getBufferedImage() {
164           return im;
165       }
166   
167       public Graphics2D getGraphics2D() {
168           return g2d;
169       }
170   
171       public UGraphicG2d getUGraphicG2d() {
172           final HColor back = HColors.simple(background);
173           final UGraphicG2d result = new UGraphicG2d(back, ColorMapper.IDENTITY, stringBounder, g2d, 1.0);
174           result.setBufferedImage(im);
175           return result;
176       }
