238               final long maxMemory = Runtime.getRuntime().maxMemory();
239               final long totalMemory = Runtime.getRuntime().totalMemory();
240               final long usedMemory = totalMemory - freeMemory;
241               final int threadActiveCount = Thread.activeCount();
242               strings.add("Max Memory: " + format(maxMemory));
243               strings.add("Total Memory: " + format(totalMemory));
244               strings.add("Free Memory: " + format(freeMemory));
245               strings.add("Used Memory: " + format(usedMemory));
246               strings.add("Thread Active Count: " + threadActiveCount);
247           }
248           return Collections.unmodifiableCollection(strings);
249       }

257       public static synchronized String getHostName() {
258           if (hostname == null) {
259               hostname = getHostNameSlow();
260           }
261           return hostname;
262       }
263   
264       private static String getHostNameSlow() {
265           try {
266               final InetAddress addr = InetAddress.getLocalHost();
267               return addr.getHostName();
268           } catch (Throwable e) {
269               final Map<String, String> env = System.getenv();
270               if (env.containsKey("COMPUTERNAME")) {
271                   return env.get("COMPUTERNAME");
272               } else if (env.containsKey("HOSTNAME")) {
273                   return env.get("HOSTNAME");
274               }
275           }
276           return "Unknown Computer";
277       }
