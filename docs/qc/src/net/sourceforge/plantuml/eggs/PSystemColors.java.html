97       public void drawU(UGraphic ug) {
98           if (paletteCentralColor != null && colors.getColorOrWhite(paletteCentralColor) instanceof HColorSimple)
99               drawPalette(ug);
100           else
101               drawFull(ug);
102       }

104       private void drawPalette(UGraphic ug) {
105           double x = (centerHexa(2, 0).getX() + centerHexa(3, 0).getX()) / 2;
106           double y = centerHexa(0, 2).getY() + corner(1).getY();
107           ug = ug.apply(new UTranslate(x, y));
108           final UPolygon hexa = getHexa();
109   
110           final List<String> friends = getColorsCloseTo(paletteCentralColor);
111           int idx = 0;
112           drawOneHexa(ug, friends.get(idx++), 0, 0, hexa);
113   
114           drawOneHexa(ug, friends.get(idx++), 1, 0, hexa);
115           drawOneHexa(ug, friends.get(idx++), 0, 1, hexa);
116           drawOneHexa(ug, friends.get(idx++), -1, 1, hexa);
117           drawOneHexa(ug, friends.get(idx++), -1, 0, hexa);
118           drawOneHexa(ug, friends.get(idx++), -1, -1, hexa);
119           drawOneHexa(ug, friends.get(idx++), 0, -1, hexa);
120   
121           drawOneHexa(ug, friends.get(idx++), 2, 0, hexa);
122           drawOneHexa(ug, friends.get(idx++), 1, 1, hexa);
123           drawOneHexa(ug, friends.get(idx++), 1, 2, hexa);
124           drawOneHexa(ug, friends.get(idx++), 0, 2, hexa);
125           drawOneHexa(ug, friends.get(idx++), -1, 2, hexa);
126           drawOneHexa(ug, friends.get(idx++), -2, 1, hexa);
127           drawOneHexa(ug, friends.get(idx++), -2, 0, hexa);
128           drawOneHexa(ug, friends.get(idx++), -2, -1, hexa);
129           drawOneHexa(ug, friends.get(idx++), -1, -2, hexa);
130           drawOneHexa(ug, friends.get(idx++), 0, -2, hexa);
131           drawOneHexa(ug, friends.get(idx++), 1, -2, hexa);
132           drawOneHexa(ug, friends.get(idx++), 1, -1, hexa);
133       }

135       private XPoint2D centerHexa(int i, int j) {
136           final double width = getWidth();
137           final double x = width * i + (j % 2 == 0 ? 0 : width / 2);
138           final double y = size * j * 1.5;
139           return new XPoint2D(x, y);
140   
141       }
142   
143       private double getWidth() {
144           return Math.sqrt(3) / 2 * 2 * size;
145       }

147       private void drawOneHexa(UGraphic ug, String colorName, int i, int j, UPolygon hexa) {
148           final HColor color = colors.getColorOrWhite(colorName);
149           ug = applyColor(ug, color);
150           ug = ug.apply(new UTranslate(centerHexa(i, j)));
151           ug.draw(hexa);
152   
153           final UFont font = UFont.sansSerif(14).bold();
154   
155           TextBlock tt = getTextName(font, colorName, color);
156           XDimension2D dimText = tt.calculateDimension(ug.getStringBounder());
157           if (dimText.getWidth() > getWidth()) {
158               tt = getTextName(font, findShortest(ug.getStringBounder(), font, colorName), color);
159               dimText = tt.calculateDimension(ug.getStringBounder());
160           }
161           tt.drawU(ug.apply(new UTranslate(-dimText.getWidth() / 2, -dimText.getHeight() / 2)));
162       }

164       private String findShortest(StringBounder stringBounder, UFont font, String colorName) {
165           String result = null;
166           double min = Double.MAX_VALUE;
167           for (int i = 1; i < colorName.length() - 1; i++) {
168               if (Character.isLowerCase(colorName.charAt(i)))
169                   continue;
170   
171               final String candidat = colorName.substring(0, i) + BackSlash.BS_BS_N + colorName.substring(i);
172               final TextBlock tt = getTextName(font, candidat, (HColorSimple) HColors.BLACK);
173               final double width = tt.calculateDimension(stringBounder).getWidth();
174               if (width < min) {
175                   result = candidat;
176                   min = width;
177               }
178           }
179           return result;
180       }

182       private UGraphic applyColor(UGraphic ug, HColor color) {
183           return ug.apply(color).apply(color.bg());
184       }

186       private XPoint2D corner(int i) {
187           double angle_deg = 60 * i + 30;
188           double angle_rad = Math.PI / 180 * angle_deg;
189           return new XPoint2D(size * Math.cos(angle_rad), size * Math.sin(angle_rad));
190       }

192       private UPolygon getHexa() {
193           final UPolygon result = new UPolygon();
194           for (int i = 0; i < 6; i++)
195               result.addPoint(corner(i));
196   
197           return result;
198       }

200       private List<String> getColorsCloseTo(String other) {
201           final List<String> result = new ArrayList<>(colors.names());
202           for (Iterator<String> it = result.iterator(); it.hasNext();) {
203               final String candidat = it.next();
204               final String similar = candidat.replaceAll("Gray", "Grey");
205               if (candidat.equals(similar))
206                   continue;
207   
208               if (result.contains(similar))
209                   it.remove();
210   
211           }
212           if (containsCaseInsensitive(result, other) == false)
213               result.add(other);
214   
215           Collections.sort(result, closeComparator(paletteCentralColor));
216           return result;
217       }
218   
219       private boolean containsCaseInsensitive(Collection<String> source, String target) {
220           for (String s : source)
221               if (s.equalsIgnoreCase(target))
222                   return true;
223   
224           return false;
225       }
226   
227       private Comparator<String> closeComparator(String center) {
228           final HColorSimple centerColor = (HColorSimple) colors.getColorOrWhite(center);
229           return new Comparator<String>() {
230               public int compare(String col1, String col2) {
231                   final int dist1 = centerColor.distanceTo((HColorSimple) colors.getColorOrWhite(col1));
232                   final int dist2 = centerColor.distanceTo((HColorSimple) colors.getColorOrWhite(col2));
233                   return (int) Math.signum(dist1 - dist2);
234               }
235           };
236       }

238       private void drawFull(UGraphic ug) {
239           final UFont font = UFont.sansSerif(14).bold();
240   
241           ug = ug.apply(HColors.BLACK);
242           int i = 0;
243           int j = 0;
244           for (String name : colors.names()) {
245               UGraphic tmp = getPositioned(ug, i, j);
246               final HColor color = colors.getColorOrWhite(name);
247               applyColor(tmp, color).draw(new URectangle(rectangleWidth, rectangleHeight));
248               final TextBlock tt = getTextName(font, name, color);
249               final XDimension2D dimText = tt.calculateDimension(ug.getStringBounder());
250               final double dy = (rectangleHeight - dimText.getHeight()) / 2;
251               final double dx = (rectangleWidth - dimText.getWidth()) / 2;
252               tt.drawU(tmp.apply(new UTranslate(dx, dy)));
253               if (j++ == 20) {
254                   j = 0;
255                   i++;
256               }
257           }
258       }

260       private TextBlock getTextName(final UFont font, String name, final HColor color) {
261           final HColor opposite = color.opposite();
262           final FontConfiguration fc = FontConfiguration.create(font, opposite, HColors.BLUE, new UStroke());
263           final TextBlock tt = Display.getWithNewlines(name).create(fc, HorizontalAlignment.CENTER,
264                   new SpriteContainerEmpty());
265           return tt;
266       }

268       private UGraphic getPositioned(UGraphic ug, int i, int j) {
269           return ug.apply(new UTranslate(rectangleWidth * i, rectangleHeight * j));
270       }
