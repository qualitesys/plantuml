95           } else if (mb.getYMode() != Globals.B_PRED) {
96               decodeMacroBlockTokens(frame, true);

102       private void decodeMacroBlockTokens(VP8Frame frame, boolean withY2)
103               throws IOException {
104           skipInnerLoopFilter = false;
105           if (withY2) {
106               skipInnerLoopFilter = skipInnerLoopFilter
107                       | decodePlaneTokens(frame, 1, SubBlock.PLANE.Y2, false);
108           }
109           skipInnerLoopFilter = skipInnerLoopFilter
110                   | decodePlaneTokens(frame, 4, SubBlock.PLANE.Y1, withY2);
111           skipInnerLoopFilter = skipInnerLoopFilter
112                   | decodePlaneTokens(frame, 2, SubBlock.PLANE.U, false);
113           skipInnerLoopFilter = skipInnerLoopFilter
114                   | decodePlaneTokens(frame, 2, SubBlock.PLANE.V, false);
115           skipInnerLoopFilter = !skipInnerLoopFilter;
116       }

118       private boolean decodePlaneTokens(VP8Frame frame, int dimentions,
119               SubBlock.PLANE plane, boolean withY2) throws IOException {
120           MacroBlock mb = this;
121           boolean r = false;
122           for (int y = 0; y < dimentions; y++) {
123               for (int x = 0; x < dimentions; x++) {
124                   int L = 0;
125                   int A = 0;
126                   int lc = 0;
127                   SubBlock sb = mb.getSubBlock(plane, x, y);
128                   SubBlock left = frame.getLeftSubBlock(sb, plane);
129                   SubBlock above = frame.getAboveSubBlock(sb, plane);
130                   if (left.hasNoZeroToken()) {
131   
132                       L = 1;
133                   }
134   
135                   lc += L;
136   
137                   if (above.hasNoZeroToken()) {
138   
139                       A = 1;
140                   }
141   
142                   lc += A;
143                   sb.decodeSubBlock(frame.getTokenBoolDecoder(),
144                           frame.getCoefProbs(), lc,
145                           SubBlock.planeToType(plane, withY2), withY2);
146                   r = r | sb.hasNoZeroToken();
147               }
148           }
149           return r;
150       }

152       public void dequantMacroBlock(VP8Frame frame) {
153           MacroBlock mb = this;
154           if (mb.getYMode() != Globals.B_PRED) {
155               SubBlock sb = mb.getY2SubBlock();
156               int acQValue = frame.getSegmentQuants().getSegQuants()[this.getSegmentId()]
157                       .getY2ac_delta_q();
158               int dcQValue = frame.getSegmentQuants().getSegQuants()[this.getSegmentId()].getY2dc();
159   
160               int input[] = new int[16];
161               input[0] = sb.getTokens()[0] * dcQValue;
162   
163               for (int x = 1; x < 16; x++)
164                   input[x] = sb.getTokens()[x] * acQValue;
165   
166               sb.setDiff(IDCT.iwalsh4x4(input));
167               for (int j = 0; j < 4; j++) {
168                   for (int i = 0; i < 4; i++) {
169                       SubBlock ysb = mb.getYSubBlock(i, j);
170                       ysb.dequantSubBlock(frame, sb.getDiff()[i][j]);
171                   }
172               }
173   
174               mb.predictY(frame);
175               mb.predictUV(frame);
176               for (int i = 0; i < 2; i++) {
177                   for (int j = 0; j < 2; j++) {
178                       SubBlock uvsb = mb.getUSubBlock(j, i);
179                       uvsb.dequantSubBlock(frame, null);
180                       uvsb = mb.getVSubBlock(i, j);
181                       uvsb.dequantSubBlock(frame, null);
182                   }
183               }
184               mb.recon_mb();
185   
186           } else {
187               for (int j = 0; j < 4; j++) {
188                   for (int i = 0; i < 4; i++) {
189                       SubBlock sb = mb.getYSubBlock(i, j);
190                       sb.dequantSubBlock(frame, null);
191                       sb.predict(frame);
192                       sb.reconstruct();
193                   }
194               }
195               mb.predictUV(frame);
196               for (int i = 0; i < 2; i++) {
197                   for (int j = 0; j < 2; j++) {
198                       SubBlock sb = mb.getUSubBlock(j, i);
199                       sb.dequantSubBlock(frame, null);
200                       sb.reconstruct();
201                   }
202               }
203               for (int i = 0; i < 2; i++) {
204                   for (int j = 0; j < 2; j++) {
205                       SubBlock sb = mb.getVSubBlock(j, i);
206                       sb.dequantSubBlock(frame, null);
207                       sb.reconstruct();
208                   }
209               }
210           }
211       }

391       public void predictUV(VP8Frame frame) {
392           MacroBlock aboveMb = frame.getMacroBlock(x, y - 1);
393           MacroBlock leftMb = frame.getMacroBlock(x - 1, y);
394   
395           switch (this.uvMode) {
396           case Globals.DC_PRED:
397               // System.out.println("UV DC_PRED");
398   
399               boolean up_available = false;
400               boolean left_available = false;
401               int Uaverage = 0;
402               int Vaverage = 0;
403               int expected_udc = 0;
404               int expected_vdc = 0;
405               if (x > 0)
406                   left_available = true;
407               if (y > 0)
408                   up_available = true;
409               if (up_available || left_available) {
410                   if (up_available) {
411                       for (int j = 0; j < 2; j++) {
412                           SubBlock usb = aboveMb.getUSubBlock(j, 1);
413                           SubBlock vsb = aboveMb.getVSubBlock(j, 1);
414                           for (int i = 0; i < 4; i++) {
415                               Uaverage += usb.getDest()[i][3];
416                               Vaverage += vsb.getDest()[i][3];
417                           }
418                       }
419                   }
420   
421                   if (left_available) {
422                       for (int j = 0; j < 2; j++) {
423                           SubBlock usb = leftMb.getUSubBlock(1, j);
424                           SubBlock vsb = leftMb.getVSubBlock(1, j);
425                           for (int i = 0; i < 4; i++) {
426                               Uaverage += usb.getDest()[3][i];
427                               Vaverage += vsb.getDest()[3][i];
428                           }
429                       }
430                   }
431   
432                   int shift = 2;
433                   if (up_available)
434                       shift++;
435                   if (left_available)
436                       shift++;
437   
438                   expected_udc = (Uaverage + (1 << (shift - 1))) >> shift;
439                   expected_vdc = (Vaverage + (1 << (shift - 1))) >> shift;
440               } else {
441                   expected_udc = 128;
442                   expected_vdc = 128;
443               }
444   
445               int ufill[][] = new int[4][4];
446               for (int y = 0; y < 4; y++)
447                   for (int x = 0; x < 4; x++)
448                       ufill[x][y] = expected_udc;
449   
450               int vfill[][] = new int[4][4];
451               for (int y = 0; y < 4; y++)
452                   for (int x = 0; x < 4; x++)
453                       vfill[x][y] = expected_vdc;
454   
455               for (int y = 0; y < 2; y++)
456                   for (int x = 0; x < 2; x++) {
457                       SubBlock usb = uSubBlocks[x][y];
458                       SubBlock vsb = vSubBlocks[x][y];
459                       usb.setPredict(ufill);
460                       vsb.setPredict(vfill);
461                   }
462   
463               break;
464           case Globals.V_PRED:
465               // System.out.println("UV V_PRED");
466   
467               SubBlock[] aboveUSb = new SubBlock[2];
468               SubBlock[] aboveVSb = new SubBlock[2];
469               for (int x = 0; x < 2; x++) {
470                   aboveUSb[x] = aboveMb.getUSubBlock(x, 1);
471                   aboveVSb[x] = aboveMb.getVSubBlock(x, 1);
472               }
473   
474               for (int y = 0; y < 2; y++)
475                   for (int x = 0; x < 2; x++) {
476                       SubBlock usb = uSubBlocks[y][x];
477                       SubBlock vsb = vSubBlocks[y][x];
478                       int ublock[][] = new int[4][4];
479                       int vblock[][] = new int[4][4];
480                       for (int j = 0; j < 4; j++)
481                           for (int i = 0; i < 4; i++) {
482                               ublock[j][i] = aboveUSb[y]
