72       private int DCTextra(BoolDecoder bc2, int p[]) throws IOException {
73           int v = 0;
74           int offset = 0;
75           do {
76               v += v + bc2.readBool(p[offset]);
77               offset++;
78           } while (p[offset] > 0);
79           return v;

82       public void decodeSubBlock(BoolDecoder bc2, int[][][][] coef_probs,
83               int ilc, int type, boolean withY2) throws IOException {
84           SubBlock sb = this;
85           int startAt = 0;
86           if (withY2)
87               startAt = 1;
88           int lc = ilc;
89           int c = 0;
90           int v = 1;
91   
92           boolean skip = false;
93   
94           while (!(v == Globals.dct_eob) && c + startAt < 16) {
95   
96               if (!skip)
97                   v = bc2.readTree(Globals.vp8CoefTree,
98                           coef_probs[type][Globals.vp8CoefBands[c + startAt]][lc]);
99               else
100                   v = bc2.readTreeSkip(
101                           Globals.vp8CoefTree,
102                           coef_probs[type][Globals.vp8CoefBands[c + startAt]][lc],
103                           1);
104   
105               int dv = decodeToken(bc2, v);
106               lc = 0;
107               skip = false;
108               if (dv == 1 || dv == -1)
109                   lc = 1;
110               else if (dv > 1 || dv < -1)
111                   lc = 2;
112               else if (dv == Globals.DCT_0)
113                   skip = true;
114   
115               int tokens[] = sb.getTokens();
116   
117               if (v != Globals.dct_eob)
118                   tokens[Globals.vp8defaultZigZag1d[c + startAt]] = dv;
119               c++;
120           }
121           hasNoZeroToken = false;
122           for (int x = 0; x < 16; x++)
123               if (tokens[x] != 0)
124                   hasNoZeroToken = true;
125       }

127       private int decodeToken(BoolDecoder bc2, int v) throws IOException {
128           int r = v;
129   
130           if (v == Globals.dct_cat1) {
131               r = 5 + DCTextra(bc2, Globals.Pcat1);
132           }
133           if (v == Globals.dct_cat2) {
134               r = 7 + DCTextra(bc2, Globals.Pcat2);
135           }
136           if (v == Globals.dct_cat3) {
137               r = 11 + DCTextra(bc2, Globals.Pcat3);
138           }
139           if (v == Globals.dct_cat4) {
140               r = 19 + DCTextra(bc2, Globals.Pcat4);
141           }
142           if (v == Globals.dct_cat5) {
143               r = 35 + DCTextra(bc2, Globals.Pcat5);
144           }
145           if (v == Globals.dct_cat6) {
146               r = 67 + DCTextra(bc2, Globals.Pcat6);
147           }
148           if (v != Globals.DCT_0 && v != Globals.dct_eob) {
149               if (bc2.readBit() > 0)
150                   r = -r;
151           }
152   
153           return r;
154       }

156       public void dequantSubBlock(VP8Frame frame, Integer Dc) {
157           SubBlock sb = this;
158   
159           int[] adjustedValues = new int[16];
160           for (int i = 0; i < 16; i++) {
161               int QValue;
162               if (plane == PLANE.U || plane == PLANE.V) {
163                   QValue = frame.getSegmentQuants().getSegQuants()[this.getMacroBlock().getSegmentId()]
164                           .getUvac_delta_q();
165                   if (i == 0)
166                       QValue = frame.getSegmentQuants().getSegQuants()[this.getMacroBlock().getSegmentId()]
167                               .getUvdc_delta_q();
168               } else {
169                   QValue = frame.getSegmentQuants().getSegQuants()[this.getMacroBlock().getSegmentId()].getY1ac();
170                   if (i == 0)
171                       QValue = frame.getSegmentQuants().getSegQuants()[this.getMacroBlock().getSegmentId()]
172                               .getY1dc();
173               }
174   
175               int inputValue = sb.getTokens()[i];
176               adjustedValues[i] = inputValue * QValue;
177   
178           }
179   
180           if (Dc != null)
181               adjustedValues[0] = Dc;
182   
183           int[][] diff = IDCT.idct4x4llm(adjustedValues);
184           sb.setDiff(diff);
185   
186       }

325       public void predict(VP8Frame frame) {
326           SubBlock sb = this;
327           SubBlock aboveSb = frame.getAboveSubBlock(sb, sb.getPlane());
328           SubBlock leftSb = frame.getLeftSubBlock(sb, sb.getPlane());
329   
330           int[] above = new int[4];
331           int[] left = new int[4];
332   
333           above[0] = aboveSb.getPredict(sb.getMode(), false)[0][3];
334           above[1] = aboveSb.getPredict(sb.getMode(), false)[1][3];
335           above[2] = aboveSb.getPredict(sb.getMode(), false)[2][3];
336           above[3] = aboveSb.getPredict(sb.getMode(), false)[3][3];
337           left[0] = leftSb.getPredict(sb.getMode(), true)[3][0];
338           left[1] = leftSb.getPredict(sb.getMode(), true)[3][1];
339           left[2] = leftSb.getPredict(sb.getMode(), true)[3][2];
340           left[3] = leftSb.getPredict(sb.getMode(), true)[3][3];
341           SubBlock AL = frame.getLeftSubBlock(aboveSb, sb.getPlane());
342   
343           // for above left if left and above is null use left (129?) else use
344           // above (127?)
345           int al;
346           if (!leftSb.isDest() && !aboveSb.isDest()) {
347   
348               al = AL.getPredict(sb.getMode(), false)[3][3];
349           } else if (!aboveSb.isDest()) {
350   
351               al = AL.getPredict(sb.getMode(), false)[3][3];
352           } else
353               al = AL.getPredict(sb.getMode(), true)[3][3];
354           SubBlock AR = frame.getAboveRightSubBlock(sb, sb.plane);
355           int ar[] = new int[4];
356           ar[0] = AR.getPredict(sb.getMode(), false)[0][3];
357           ar[1] = AR.getPredict(sb.getMode(), false)[1][3];
358           ar[2] = AR.getPredict(sb.getMode(), false)[2][3];
359           ar[3] = AR.getPredict(sb.getMode(), false)[3][3];
360           int[][] p = new int[4][4];
361           int pp[];
362           switch (sb.getMode()) {
363           case Globals.B_DC_PRED:
364               // System.out.println("B_DC_PRED");
365               int expected_dc = 0;
366   
367               for (int i = 0; i < 4; i++) {
368                   expected_dc += above[i];
369                   expected_dc += left[i];
370               }
371               expected_dc = (expected_dc + 4) >> 3;
372   
373               for (int y = 0; y < 4; y++)
374                   for (int x = 0; x < 4; x++)
375                       p[x][y] = expected_dc;
376   
377               break;
378           case Globals.B_TM_PRED:
379   
380               // System.out.println("B_TM_PRED");
381   
382               // prediction similar to true_motion prediction
383   
384               for (int r = 0; r < 4; r++) {
385                   for (int c = 0; c < 4; c++) {
386   
387                       int pred = above[c] - al + left[r];
388                       if (pred < 0)
389                           pred = 0;
390   
391                       if (pred > 255)
392                           pred = 255;
393   
394                       p[c][r] = pred;
395                   }
396               }
397               break;
398           case Globals.B_VE_PRED:
399               // System.out.println("B_VE_PRED");
400   
401               int ap[] = new int[4];
402               ap[0] = (al + 2 * above[0] + above[1] + 2) >> 2;
403               ap[1] = (above[0] + 2 * above[1] + above[2] + 2) >> 2;
404               ap[2] = (above[1] + 2 * above[2] + above[3] + 2) >> 2;
405               ap[3] = (above[2] + 2 * above[3] + ar[0] + 2) >> 2;
406   
407               for (int r = 0; r < 4; r++) {
408                   for (int c = 0; c < 4; c++) {
409   
410                       p[c][r] = ap[c];
411   
412                   }
413               }
414               break;
415           case Globals.B_HE_PRED:
416               // System.out.println("B_HE_PRED");
417   
418               int lp[] = new int[4];
419               lp[0] = (al + 2 * left[0] + left[1] + 2) >> 2;
420               lp[1] = (left[0] + 2 * left[1] + left[2] + 2) >> 2;
421               lp[2] = (left[1] + 2 * left[2] + left[3] + 2) >> 2;
422               lp[3] = (left[2] + 2 * left[3] + left[3] + 2) >> 2;
423   
424               for (int r = 0; r < 4; r++) {
425                   for (int c = 0; c < 4; c++) {
426                       p[c][r] = lp[r];
427                   }
428               }
429               break;
430           case Globals.B_LD_PRED:
431               // System.out.println("B_LD_PRED");
432               p[0][0] = (above[0] + above[1] * 2 + above[2] + 2) >> 2;
433               p[1][0] = p[0][1] = (above[1] + above[2] * 2 + above[3] + 2) >> 2;
434               p[2][0] = p[1][1] = p[0][2] = (above[2] + above[3] * 2 + ar[0] + 2) >> 2;
435               p[3][0] = p[2][1] = p[1][2] = p[0][3] = (above[3] + ar[0] * 2
436                       + ar[1] + 2) >> 2;
437               p[3][1] = p[2][2] = p[1][3] = (ar[0] + ar[1] * 2 + ar[2] + 2) >> 2;
438               p[3][2] = p[2][3] = (ar[1] + ar[2] * 2 + ar[3] + 2) >> 2;
439               p[3][3] = (ar[2] + ar[3] * 2 + ar[3] + 2) >> 2;
440   
441               break;
442           case Globals.B_RD_PRED:
443               // System.out.println("B_RD_PRED");
444               pp = new int[9];
445   
446               pp[0] = left[3];
447               pp[1] = left[2];
448               pp[2] = left[1];
449               pp[3] = left[0];
450               pp[4] = al;
451               pp[5] = above[0];
452               pp[6] = above[1];
453               pp[7] = above[2];
454               pp[8] = above[3];
455   
456               p[0][3] = (pp[0] + pp[1] * 2 + pp[2] + 2) >> 2;
457               p[1][3] = p[0][2] = (pp[1] + pp[2] * 2 + pp[3] + 2) >> 2;
458               p[2][3] = p[1][2] = p[0][1] = (pp[2] + pp[3] * 2 + pp[4] + 2) >> 2;
459               p[3][3] = p[2][2] = p[1][1] = p[0][0] = (pp[3] + pp[4] * 2 + pp[5] + 2) >> 2;
460               p[3][2] = p[2][1] = p[1][0] = (pp[4] + pp[5] * 2 + pp[6] + 2) >> 2;
461               p[3][1] = p[2][0] = (pp[5] + pp[6] * 2 + pp[7] + 2) >> 2;
462               p[3][0] = (pp[6] + pp[7] * 2 + pp[8] + 2) >> 2;
463               break;
464   
465           case Globals.B_VR_PRED:
466               // System.out.println("B_VR_PRED");
