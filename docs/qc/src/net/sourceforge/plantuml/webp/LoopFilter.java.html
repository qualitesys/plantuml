23       private static int c(int v) {
24           // return (int) (v < -128 ? -128 : (v > 127 ? v : 127));

33       private static int common_adjust(boolean use_outer_taps, /*
34                                                                * filter is 2 or 4
35                                                                * taps wide
36                                                                */
37       Segment seg) {
38           int p1 = u2s(seg.P1); /* retrieve and convert all 4 pixels */
39           int p0 = u2s(seg.P0);
40           int q0 = u2s(seg.Q0);
41           int q1 = u2s(seg.Q1);
42   
43           /*
44            * Disregarding clamping, when "use_outer_taps" is false, "a" is
45            * 3*(q0-p0). Since we are about to divide "a" by 8, in this case we end
46            * up multiplying the edge difference by 5/8. When "use_outer_taps" is
47            * true (as for the simple filter), "a" is p1 - 3*p0 + 3*q0 - q1, which
48            * can be thought of as a refinement of 2*(q0 - p0) and the adjustment
49            * is something like (q0 - p0)/4.
50            */
51           int a = c((use_outer_taps ? c(p1 - q1) : 0) + 3 * (q0 - p0));
52           /*
53            * b is used to balance the rounding of a/8 in the case where the
54            * "fractional" part "f" of a/8 is exactly 1/2.
55            */
56           int b = (c(a + 3)) >> 3;
57           /*
58            * Divide a by 8, rounding up when f >= 1/2. Although not strictly part
59            * of the "C" language, the right-shift is assumed to propagate the sign
60            * bit.
61            */
62           a = c(a + 4) >> 3;
63           /* Subtract "a" from q0, "bringing it closer" to p0. */
64           seg.Q0 = s2u(q0 - a);
65           /*
66            * Add "a" (with adjustment "b") to p0, "bringing it closer" to q0. The
67            * clamp of "a+b", while present in the reference decoder, is
68            * superfluous; we have -16 <= a <= 15 at this point.
69            */
70           seg.P0 = s2u(p0 + b);
71   
72           return a;
73       }
74   
75       /*
76        * All functions take (among other things) a segment (of length at most 4 +
77        * 4 = 8) symmetrically straddling an edge. The pixel values (or pointers)
78        * are always given in order, from the "beforemost" to the "aftermost". So,
79        * for a horizontal edge (written "|"), an 8-pixel segment would be ordered
80        * p3 p2 p1 p0 | q0 q1 q2 q3.
81        */
82       /*
83        * Filtering is disabled if the difference between any two adjacent
84        * "interior" pixels in the 8-pixel segment exceeds the relevant threshold
85        * (I). A more complex thresholding calculation is done for the group of
86        * four pixels that straddle the edge, in line with the calculation in
87        * simple_segment() above.
88        */
89       public static boolean filter_yes(int I, /* limit on interior differences */
90       int E, /* limit at the edge */
91       int p3, int p2, int p1, int p0, /* pixels before edge */
92       int q0, int q1, int q2, int q3 /* pixels after edge */
93       ) {
94           return (abs(p0 - q0) * 2 + abs(p1 - q1) / 2) <= E && abs(p3 - p2) <= I
95                   && abs(p2 - p1) <= I && abs(p1 - p0) <= I && abs(q3 - q2) <= I
96                   && abs(q2 - q1) <= I && abs(q1 - q0) <= I;
97       }

99       private static Segment getSegH(SubBlock rsb, SubBlock lsb, int a) {
100           Segment seg = new Segment();
101           int[][] rdest = rsb.getDest();
102           int[][] ldest = lsb.getDest();
103           seg.P0 = ldest[3][a];
104           seg.P1 = ldest[2][a];
105           seg.P2 = ldest[1][a];
106           seg.P3 = ldest[0][a];
107           seg.Q0 = rdest[0][a];
108           seg.Q1 = rdest[1][a];
109           seg.Q2 = rdest[2][a];
110           seg.Q3 = rdest[3][a];
111           return seg;
112       }

114       private static Segment getSegV(SubBlock bsb, SubBlock tsb, int a) {
115           Segment seg = new Segment();
116           int[][] bdest = bsb.getDest();
117           int[][] tdest = tsb.getDest();
118   
119           seg.P0 = tdest[a][3];
120           seg.P1 = tdest[a][2];
121           seg.P2 = tdest[a][1];
122           seg.P3 = tdest[a][0];
123           seg.Q0 = bdest[a][0];
124           seg.Q1 = bdest[a][1];
125           seg.Q2 = bdest[a][2];
126           seg.Q3 = bdest[a][3];
127           return seg;
128       }

130       /*
131        * Filtering is altered if (at least) one of the differences on either side
132        * of the edge exceeds a threshold (we have "high edge variance").
133        */
134       public static boolean hev(int threshold, int p1, int p0, /*
135                                                                * pixels before
136                                                                * edge
137                                                                */
138       int q0, int q1 /* pixels after edge */
139       ) {
140           return abs(p1 - p0) > threshold || abs(q1 - q0) > threshold;
141       }

144           frame.fireLFProgressUpdate(0);

158                       .getMacroBlockRows()))));

160                   // System.out.println("x: "+x+" y: "+y);

162                   MacroBlock bmb = frame.getMacroBlock(x, y);

170                           interior_limit >>= sharpnessLevel > 4 ? 2 : 1;
171                           if (interior_limit > 9 - sharpnessLevel)

179                       if(sub_bedge_limit < 1)
180                           sub_bedge_limit = 1;
181   
182                       /* Luma and Chroma use the same inter-macroblock edge limit */
183                       int mbedge_limit = sub_bedge_limit+4;
184   
185                       // left
186                       if (x > 0) {
187                           MacroBlock lmb = frame.getMacroBlock(x - 1, y);

192                                   Segment seg = getSegH(rsb, lsb, a);

194                                   // mbedge_limit, seg);
195                                   // System.out.println(mbedge_limit);
196                                   simple_segment(mbedge_limit, seg);
197                                   setSegH(rsb, lsb, seg, a);
198                               }
199                           }
200                       }
201   
202                       // sb left
203                       if (!rmb.isSkip_inner_lf()) {
204   
205                           for (int a = 1; a < 4; a++) {
206                               for (int b = 0; b < 4; b++) {
207                                   SubBlock lsb = rmb.getSubBlock(SubBlock.PLANE.Y1,
208                                           a - 1, b);
209                                   SubBlock rsb = rmb.getSubBlock(SubBlock.PLANE.Y1,
210                                           a, b);

221                           }

267           for (int y = 0; y < frame.getMacroBlockRows(); y++) {

270               for (int x = 0; x < frame.getMacroBlockCols(); x++) {
271                   MacroBlock rmb = frame.getMacroBlock(x, y);
272                   MacroBlock bmb = frame.getMacroBlock(x, y);
273                   int sharpnessLevel = frame.getSharpnessLevel();
274                   int loop_filter_level = rmb.getFilterLevel();
275                   if(loop_filter_level!=0) {
276                       int interior_limit = rmb.getFilterLevel();
277                       if (sharpnessLevel > 0) {
278                           interior_limit >>= sharpnessLevel > 4 ? 2 : 1;
279                           if (interior_limit > 9 - sharpnessLevel)
280                               interior_limit = 9 - sharpnessLevel;
281                       }
282                       if (interior_limit == 0)
283                           interior_limit = 1;
284   
285                       int hev_threshold = 0;
286                       if (frame.getFrameType() == 0) /* current frame is a key frame */

289                               hev_threshold = 2;

318                                           mbedge_limit, seg);

320                                   seg = getSegH(rsbV, lsbV, a);
321                                   MBfilter(hev_threshold, interior_limit,
322                                           mbedge_limit, seg);

331                           for (int a = 1; a < 2; a++) {

333                                   SubBlock lsbU = rmb.getSubBlock(SubBlock.PLANE.U,
334                                           a - 1, b);

336                                           a, b);

518       static void MBfilter(int hev_threshold, /* detect high edge variance */
519       int interior_limit, /* possibly disable filter */
520       int edge_limit, Segment seg) {
521           int p3 = u2s(seg.P3), p2 = u2s(seg.P2), p1 = u2s(seg.P1), p0 = u2s(seg.P0);
522           int q0 = u2s(seg.Q0), q1 = u2s(seg.Q1), q2 = u2s(seg.Q2), q3 = u2s(seg.Q3);
523           if (filter_yes(interior_limit, edge_limit, q3, q2, q1, q0, p0, p1, p2,
524                   p3)) {
525               if (!hev(hev_threshold, p1, p0, q0, q1)) {
526                   // Same as the initial calculation in "common_adjust",
527                   // w is something like twice the edge difference
528                   int w = c(c(p1 - q1) + 3 * (q0 - p0));
529   
530                   // 9/64 is approximately 9/63 = 1/7 and 1<<7 = 128 = 2*64.
531                   // So this a, used to adjust the pixels adjacent to the edge,
532                   // is something like 3/7 the edge difference.
533                   int a = (27 * w + 63) >> 7;
534   
535                   seg.Q0 = s2u(q0 - a);
536                   seg.P0 = s2u(p0 + a);
537                   // Next two are adjusted by 2/7 the edge difference
538                   a = (18 * w + 63) >> 7;
539                   // System.out.println("a: "+a);
540                   seg.Q1 = s2u(q1 - a);
541                   seg.P1 = s2u(p1 + a);
542                   // Last two are adjusted by 1/7 the edge difference
543                   a = (9 * w + 63) >> 7;
544                   seg.Q2 = s2u(q2 - a);
545                   seg.P2 = s2u(p2 + a);
546               } else
547                   // if hev, do simple filter
548                   common_adjust(true, seg); // using outer taps
549           }
550       }
