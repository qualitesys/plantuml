102       public boolean decodeFrame(boolean debug) throws IOException {

104           this.debug=debug;

106           int c;
107           frame.seek(offset++); c=frame.readUnsignedByte();
108           frameType = getBitAsInt(c, 0);
109           if(frameType!=0)
110               return false;
111           int versionNumber = getBitAsInt(c, 1) << 1;
112   
113           versionNumber += getBitAsInt(c, 2) << 1;
114           versionNumber += getBitAsInt(c, 3);

116           firstPartitionLengthInBytes = getBitAsInt(c, 5) << 0;
117           firstPartitionLengthInBytes += getBitAsInt(c, 6) << 1;
118           firstPartitionLengthInBytes += getBitAsInt(c, 7) << 2;
119           frame.seek(offset++); c=frame.readUnsignedByte();
120           firstPartitionLengthInBytes += c << 3;
121           frame.seek(offset++); c=frame.readUnsignedByte();
122           firstPartitionLengthInBytes += c << 11;
123           frame.seek(offset++); c=frame.readUnsignedByte();
124           frame.seek(offset++); c=frame.readUnsignedByte();
125           frame.seek(offset++); c=frame.readUnsignedByte();
126           frame.seek(offset++); c=frame.readUnsignedByte();
127           int hBytes = c;
128           frame.seek(offset++); c=frame.readUnsignedByte();
129           hBytes += c << 8;
130           width = (hBytes & 0x3fff);
131           frame.seek(offset++); c=frame.readUnsignedByte();
132           int vBytes = c;
133           frame.seek(offset++); c=frame.readUnsignedByte();
134           vBytes += c << 8;
135           height = (vBytes & 0x3fff);
136           int tWidth = width;
137           int tHeight = height;
138           if ((tWidth & 0xf) != 0)
139               tWidth += 16 - (tWidth & 0xf);
140   
141           if ((tHeight & 0xf) != 0)
142               tHeight += 16 - (tHeight & 0xf);
143           macroBlockRows=tHeight >> 4;
144           macroBlockCols=tWidth >> 4;
145           createMacroBlocks();
146   
147           BoolDecoder bc = new BoolDecoder(frame, offset);
148   
149           if (frameType == 0) {
150               int clr_type = bc.readBit();
151               int clamp_type = bc.readBit();
152   
153           }
154           segmentationIsEnabled = bc.readBit();
155           if (segmentationIsEnabled > 0) {
156               updateMacroBlockSegmentationMap = bc.readBit();
157               updateMacroBlockSegmentatonData = bc.readBit();
158               if(updateMacroBlockSegmentatonData > 0 ) {
159                   
160                       macroBlockSegementAbsoluteDelta = bc.readBit();
161                       /* For each segmentation feature (Quant and loop filter level) */
162                       for (int i = 0; i < Globals.MAX_MB_SEGMENTS; i++) {
163                           int value =0;
164                           if (bc.readBit() > 0) {
165                               value = bc.readLiteral(Globals.vp8MacroBlockFeatureDataBits[0]);
166                               if(bc.readBit()>0)
167                               value=-value;
168                           }
169                           this.segmentQuants.getSegQuants()[i].setQindex(value);
170                       }
171                       for (int i = 0; i < Globals.MAX_MB_SEGMENTS; i++) {
172                           int value = 0;
173                           if (bc.readBit() > 0) {
174                               value = bc.readLiteral(Globals.vp8MacroBlockFeatureDataBits[1]);
175                               if(bc.readBit()>0)
176                                   value=-value;
177                           }
178                           this.segmentQuants.getSegQuants()[i].setFilterStrength(value);
179                       }
180   
181                       if(updateMacroBlockSegmentationMap > 0) {
182                           macroBlockSegmentTreeProbs = new int[Globals.MB_FEATURE_TREE_PROBS];
183                           for (int i = 0; i < Globals.MB_FEATURE_TREE_PROBS; i++) {
184                               int value=255;
185                               if (bc.readBit()>0) {
186                                   value = bc.readLiteral(8);
187                               }
188                               else
189                                   value = 255;
190                               macroBlockSegmentTreeProbs[i] = value;
191                       }
192                   }
193               }
194           }
195           simpleFilter = bc.readBit();
196           filterLevel = bc.readLiteral(6);
197           
198           sharpnessLevel = bc.readLiteral(3);
199           modeRefLoopFilterDeltaEnabled = bc.readBit();
200           if (modeRefLoopFilterDeltaEnabled > 0) {
201               // Do the deltas need to be updated
202               modeRefLoopFilterDeltaUpdate = bc.readBit();
203               if (modeRefLoopFilterDeltaUpdate > 0) {
204                   for (int i = 0; i < MAX_REF_LF_DELTAS; i++) {
205   
206                       if (bc.readBit() > 0) {
207                           refLoopFilterDeltas[i] = bc.readLiteral(6);
208                           if (bc.readBit() > 0) // Apply sign
209                               refLoopFilterDeltas[i] = refLoopFilterDeltas[i] * -1;
210                       }
211                   }
212                   for (int i = 0; i < MAX_MODE_LF_DELTAS; i++) {
213   
214                       if (bc.readBit() > 0) {
215                           modeLoopFilterDeltas[i] = bc.readLiteral(6);
216                           if (bc.readBit() > 0) // Apply sign
217                               modeLoopFilterDeltas[i] = modeLoopFilterDeltas[i] * -1;
218                       }
219                   }
220               }
221           }
222   
223           filterType = (filterLevel == 0) ? 0 : (simpleFilter>0) ? 1 : 2;
224           setupTokenDecoder(bc, firstPartitionLengthInBytes,
225                   offset);
226           bc.seek();
227   
228           segmentQuants.parse(bc, segmentationIsEnabled==1, macroBlockSegementAbsoluteDelta==1);
229   
230           // Determine if the golden frame or ARF buffer should be updated and
231           // how.
232           // For all non key frames the GF and ARF refresh flags and sign bias
233           // flags must be set explicitly.
234           if (frameType != 0) {
235               throw new IllegalArgumentException("bad input: not intra");
236           }
237           refreshEntropyProbs = bc.readBit();
238           if (refreshEntropyProbs > 0) {
239   
240           }
241           refreshLastFrame = 0;
242           if (frameType == 0)
243               refreshLastFrame = 1;
244           else
245               refreshLastFrame = bc.readBit();
246           for (int i = 0; i < BLOCK_TYPES; i++)
247               for (int j = 0; j < COEF_BANDS; j++)
248                   for (int k = 0; k < PREV_COEF_CONTEXTS; k++)
249                       for (int l = 0; l < MAX_ENTROPY_TOKENS - 1; l++) {
250   
251                           if (bc.readBool(Globals.vp8CoefUpdateProbs[i][j][k][l]) > 0) {
252                               int newp = bc.readLiteral(8);
253                               this.coefProbs[i][j][k][l] = newp;
254                           }
255                       }
256   
257           // Read the mb_no_coeff_skip flag
258           macroBlockNoCoeffSkip = (int) bc.readBit();
259           if (frameType == 0) {
260               readModes(bc);
261           } else {
262               throw new IllegalArgumentException("bad input: not intra");
263           }
264   
265           int ibc = 0;
266           int num_part = 1 << multiTokenPartition;
267   
268           for (int mb_row = 0; mb_row < macroBlockRows; mb_row++) {
269   
270               if (num_part > 1) {
271                   
272                   tokenBoolDecoder = tokenBoolDecoders.elementAt(ibc);
273                   tokenBoolDecoder.seek();
274   
275                   decodeMacroBlockRow(mb_row);
276   
277                   ibc++;
278                   if(ibc==num_part)
279                       ibc=0;
280               }
281               else
282                   decodeMacroBlockRow(mb_row);
283               fireProgressUpdate(mb_row);
284   
285           }
286   
287           if(this.getFilterType()>0 && this.getFilterLevel()!=0)
288               this.loopFilter();
289           return true;
290       }

857       private void readModes(BoolDecoder bc) throws IOException {
858           int mb_row = -1;
859           int prob_skip_false = 0;
860           
861           if (macroBlockNoCoeffSkip > 0) {
862               prob_skip_false = bc.readLiteral(8);
863           }
864   
865           while (++mb_row < macroBlockRows) {
866               int mb_col = -1;
867               while (++mb_col < macroBlockCols) {
868   
869                   //if (this.segmentation_enabled > 0) {
870                   //    logger.log(Level.SEVERE, "TODO:");
871                   //    throw new IllegalArgumentException("bad input: segmentation_enabled()");
872                   //}
873                   // Read the macroblock coeff skip flag if this feature is in
874                   // use, else default to 0
875                   MacroBlock mb = getMacroBlock(mb_col, mb_row);
876                   
877                   if ((segmentationIsEnabled >0) &&( updateMacroBlockSegmentationMap > 0)) {
878                       int value = bc.readTree(Globals.macroBlockSegmentTree, this.macroBlockSegmentTreeProbs);
879                       mb.setSegmentId(value);
880                   }
881                   
882                   if(modeRefLoopFilterDeltaEnabled > 0) {
883                       int level = filterLevel;
884                       level = level + refLoopFilterDeltas[0];
885                       level = (level < 0) ? 0 : (level > 63) ? 63 : level;
886                       mb.setFilterLevel(level);
887                   }
888                   else {
889                       mb.setFilterLevel(segmentQuants.getSegQuants()[mb.getSegmentId()].getFilterStrength());
890   //                    logger.error("TODO:");
891                   }
892                   
893                   int mb_skip_coeff = 0;
894                   if (macroBlockNoCoeffSkip > 0)
895                       mb_skip_coeff = bc.readBool(prob_skip_false);
896                   else
897                       mb_skip_coeff = 0;
898                   mb.setSkipCoeff(mb_skip_coeff);
899   
900                   int y_mode = readYMode(bc);
901                   
902                   mb.setYMode(y_mode);
903   
904                   if (y_mode == Globals.B_PRED) {
905   
906                       for (int i = 0; i < 4; i++) {

1007               if (i < num_part - 1) {

1010                   bc.seek();

1013               }
