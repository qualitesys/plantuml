18   import java.io.IOException;
19   
20   public class SegmentQuants {
21   
22       private static DeltaQ get_delta_q(BoolDecoder bc, int prev)
23               throws IOException {
24           DeltaQ ret = new DeltaQ();
25           ret.v = 0;
26           ret.update = false;
27   
28           if (bc.readBit() > 0) {
29               ret.v = bc.readLiteral(4);
30   
31               if (bc.readBit() > 0)
32                   ret.v = -ret.v;
33           }
34   
35           /* Trigger a quantizer update if the delta-q value has changed */
36           if (ret.v != prev)
37               ret.update = true;
38   
39           return ret;
40       }
41   
42       private int qIndex;
43   
44       private SegmentQuant[] segQuants = new SegmentQuant[Globals.MAX_MB_SEGMENTS];
45   
46       public SegmentQuants() {
47           for (int x = 0; x < Globals.MAX_MB_SEGMENTS; x++)
48               segQuants[x] = new SegmentQuant();
49       }
50   
51       public int getqIndex() {
52           return qIndex;
53       }
54   
55       public SegmentQuant[] getSegQuants() {
56           return segQuants;
57       }
58   
59       public void parse(BoolDecoder bc, boolean segmentation_enabled,
60               boolean mb_segement_abs_delta) throws IOException {
61           qIndex = bc.readLiteral(7);
62           boolean q_update = false;
63           DeltaQ v = get_delta_q(bc, 0);
64           int y1dc_delta_q = v.v;
65           q_update = q_update || v.update;
66           v = get_delta_q(bc, 0);
67           int y2dc_delta_q = v.v;
68           q_update = q_update || v.update;
69           v = get_delta_q(bc, 0);
70           int y2ac_delta_q = v.v;
71           q_update = q_update || v.update;
72           v = get_delta_q(bc, 0);
73           int uvdc_delta_q = v.v;
74           q_update = q_update || v.update;
75           v = get_delta_q(bc, 0);
76           int uvac_delta_q = v.v;
77           q_update = q_update || v.update;
78   
79           for (SegmentQuant s : segQuants) {
80               if (!segmentation_enabled) {
81                   s.setQindex(qIndex);
82               } else if (!mb_segement_abs_delta) {
83                   s.setQindex(s.getQindex() + qIndex);
84               }
85   
86               s.setY1dc(y1dc_delta_q);
87               s.setY2dc(y2dc_delta_q);
88               s.setY2ac_delta_q(y2ac_delta_q);
89               s.setUvdc_delta_q(uvdc_delta_q);
90               s.setUvac_delta_q(uvac_delta_q);
91   
92           }
93       }
94   
95       public void setSegQuants(SegmentQuant[] segQuants) {
96           this.segQuants = segQuants;
97       }
98   }
