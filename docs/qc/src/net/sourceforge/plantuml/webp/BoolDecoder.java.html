20   import javax.imageio.stream.ImageInputStream;

22   public class BoolDecoder {
23       int bit_count; /* # of bits shifted out of value, at most 7 */
24       ImageInputStream data;
25       private long offset; /* pointer to next compressed data byte */
26       private int range; /* always identical to encoder's range */
27       private int value; /* contains at least 24 significant bits */
28   
29       BoolDecoder(ImageInputStream frame, long offset) throws IOException {
30           this.data = frame;
31           this.offset = offset;
32           initBoolDecoder();
33       }
34   
35       private void initBoolDecoder() throws IOException {
36   
37           value = 0; /* value = first 16 input bits */
38   
39           data.seek(offset);
40           value = data.readUnsignedByte() << 8;
41           // value = (data[offset]) << 8;
42           offset++;
43   
44           range = 255; /* initial range is full */
45           bit_count = 0; /* have not yet shifted out any bits */
46       }
47   
48       public int readBit() throws IOException {
49           return readBool(128);
50       }
51   
52       public int readBool(int probability) throws IOException {
53   
54           int bit = 0;
55           int split;
56           int bigsplit;
57           int range = this.range;
58           int value = this.value;
59           split = 1 + (((range - 1) * probability) >> 8);
60           bigsplit = (split << 8);
61           range = split;
62   
63           if (value >= bigsplit) {
64               range = this.range - split;
65               value = value - bigsplit;
66               bit = 1;
67           }
68   
69           {
70               int count = this.bit_count;
71               int shift = Globals.vp8dxBitreaderNorm[range];
72               range <<= shift;
73               value <<= shift;
74               count -= shift;
75   
76               if (count <= 0) {
77                   // data.seek(offset);
78                   value |= data.readUnsignedByte() << (-count);
79                   // value |= data[offset] << (-count);
80                   offset++;
81                   count += 8;
82               }
83   
84               this.bit_count = count;
85           }
86           this.value = value;
87           this.range = range;
88           return bit;
89       }

91       /*
92        * Convenience function reads a "literal", that is, a "num_bits" wide
93        * unsigned value whose bits come high- to low-order, with each bit encoded
94        * at probability 128 (i.e., 1/2).
95        */
96       public int readLiteral(int num_bits) throws IOException {
97           int v = 0;
98           while (num_bits-- > 0)
99               v = (v << 1) + readBool(128);
100           return v;
101       }

103       int readTree(int t[], /* tree specification */
104       int p[] /* corresponding interior node probabilities */
105       ) throws IOException {
106           int i = 0; /* begin at root */
107   
108           /* Descend tree until leaf is reached */
109           while ((i = t[i + readBool(p[i >> 1])]) > 0) {
110           }
111           return -i; /* return value is negation of nonpositive index */
112   
113       }
114   
115       int readTreeSkip(int t[], /* tree specification */
116       int p[], /* corresponding interior node probabilities */
117       int skip_branches) throws IOException {
118           int i = skip_branches * 2; /* begin at root */
119   
120           /* Descend tree until leaf is reached */
121           while ((i = t[i + readBool(p[i >> 1])]) > 0) {
122           }
123           return -i; /* return value is negation of nonpositive index */
124   
125       }
