265       private static void encodeSprite(List<String> result) throws IOException {
266           SpriteGrayLevel level = SpriteGrayLevel.GRAY_16;
267           boolean compressed = false;
268           final String path;
269           if (result.size() > 1 && result.get(0).matches("(4|8|16)z?")) {
270               if (result.get(0).startsWith("8")) {
271                   level = SpriteGrayLevel.GRAY_8;
272               }
273               if (result.get(0).startsWith("4")) {
274                   level = SpriteGrayLevel.GRAY_4;
275               }
276               compressed = StringUtils.goLowerCase(result.get(0)).endsWith("z");
277               path = result.get(1);
278           } else {
279               path = result.get(0);
280           }
281   
282           final String fileName;
283           final URL source;
284           final String lowerPath = StringUtils.goLowerCase(path);
285           if (lowerPath.startsWith(httpProtocol) || lowerPath.startsWith(httpsProtocol)) {
286               source = new java.net.URL(path);
287               final String p = source.getPath();
288               fileName = p.substring(p.lastIndexOf('/') + 1, p.length());
289           } else {
290               final SFile f = new SFile(path);
291               source = f.toURI().toURL();
292               fileName = f.getName();
293           }
294   
295           if (source == null) {
296               return;
297           }
298   
299           final BufferedImage im;
300           try (InputStream stream = source.openStream()) {
301               im = SImageIO.read(stream);
302           }
303           final String name = getSpriteName(fileName);
304           final String s = compressed ? SpriteUtils.encodeCompressed(im, name, level)
305                   : SpriteUtils.encode(im, name, level);
306           System.out.println(s);
307       }

377       private static void processArgs(Option option, ErrorStatus error)
378               throws NoPlantumlCompressionException, InterruptedException {
379           if (option.isDecodeurl() == false && option.getNbThreads() > 1 && option.isCheckOnly() == false
380                   && OptionFlags.getInstance().isExtractFromMetadata() == false) {
381               multithread(option, error);
382               return;
383           }
384           final List<File> files = new ArrayList<>();
385           for (String s : option.getResult()) {
386               if (option.isDecodeurl()) {
387                   error.goOk();
388                   final Transcoder transcoder = TranscoderUtil.getDefaultTranscoder();
389                   System.out.println("@startuml");
390                   System.out.println(transcoder.decode(s));
391                   System.out.println("@enduml");
392               } else {
393                   final FileGroup group = new FileGroup(s, option.getExcludes(), option);
394                   incTotal(group.getFiles().size());
395                   files.addAll(group.getFiles());
396               }
397           }
398           foundNbFiles(files.size());
399           for (File f : files) {
400               try {
401                   manageFileInternal(f, option, error);
402                   incDone(error.hasError());
403                   if (error.hasError() && option.isFailfastOrFailfast2()) {
404                       return;
405                   }
406               } catch (IOException e) {
407                   Logme.error(e);
408               }
409           }
410       }

412       private static void multithread(final Option option, final ErrorStatus error) throws InterruptedException {
413           Log.info("Using several threads: " + option.getNbThreads());
414           final ExecutorService executor = Executors.newFixedThreadPool(option.getNbThreads());
415   
416           int nb = 0;
417           for (String s : option.getResult()) {
418               final FileGroup group = new FileGroup(s, option.getExcludes(), option);
419               for (final File f : group.getFiles()) {
420                   incTotal(1);
421                   nb++;
422                   executor.submit(new Runnable() {
423                       public void run() {
424                           if (error.hasError() && option.isFailfastOrFailfast2()) {
425                               return;
426                           }
427                           try {
428                               manageFileInternal(f, option, error);
429                           } catch (IOException e) {
430                               Logme.error(e);
431                           } catch (InterruptedException e) {
432                               Logme.error(e);
433                           }
434                           incDone(error.hasError());
435                       }
436                   });
437               }
438           }
439           foundNbFiles(nb);
440           executor.shutdown();
441           executor.awaitTermination(Long.MAX_VALUE, TimeUnit.SECONDS);
442       }

458       private static void manageFileInternal(File f, Option option, ErrorStatus error)
459               throws IOException, InterruptedException {
460           Log.info("Working on " + f.getPath());
461           if (OptionFlags.getInstance().isExtractFromMetadata()) {
462               System.out.println("------------------------");
463               System.out.println(f);
464               // new Metadata().readAndDisplayMetadata(f);
465               System.out.println();
466               error.goOk();
467               final String data = new MetadataTag(f, "plantuml").getData();
468               // File file = SecurityUtils.File("tmp.txt");
469               // PrintWriter pw = SecurityUtils.PrintWriter(file, "UTF-8");
470               // pw.println(NastyEncoder.fromISO_8859_1(data));
471               // pw.close();
472   
473               System.out.println(data);
474               System.out.println("------------------------");
475               return;
476           }
477           final ISourceFileReader sourceFileReader;
478           if (option.getOutputFile() == null) {
479               File outputDir = option.getOutputDir();
480               if (outputDir != null && outputDir.getPath().endsWith("$")) {
481                   final String path = outputDir.getPath();
482                   outputDir = new File(path.substring(0, path.length() - 1)).getAbsoluteFile();
483                   sourceFileReader = new SourceFileReaderCopyCat(option.getDefaultDefines(f), f, outputDir,
484                           option.getConfig(), option.getCharset(), option.getFileFormatOption());
485               } else {
486                   sourceFileReader = new SourceFileReader(option.getDefaultDefines(f), f, outputDir, option.getConfig(),
487                           option.getCharset(), option.getFileFormatOption());
488               }
489           } else {
490               sourceFileReader = new SourceFileReaderHardFile(option.getDefaultDefines(f), f, option.getOutputFile(),
491                       option.getConfig(), option.getCharset(), option.getFileFormatOption());
492           }

533       private static void extractPreproc(Option option, final ISourceFileReader sourceFileReader) throws IOException {
534           final String charset = option.getCharset();
535           for (BlockUml blockUml : sourceFileReader.getBlocks()) {
536               final SuggestedFile suggested = ((SourceFileReaderAbstract) sourceFileReader).getSuggestedFile(blockUml)
537                       .withPreprocFormat();
538               final SFile file = suggested.getFile(0);
539               Log.info("Export preprocessing source to " + file.getPrintablePath());
540               try (final PrintWriter pw = charset == null ? file.createPrintWriter() : file.createPrintWriter(charset)) {
541                   int level = 0;
542                   for (CharSequence cs : blockUml.getDefinition(true)) {
543                       String s = cs.toString();
544                       if (cypher != null) {
545                           if (s.contains("skinparam") && s.contains("{")) {
546                               level++;
547                           }
548                           if (level == 0 && s.contains("skinparam") == false) {
549                               s = cypher.cypher(s);
550                           }
551                           if (level > 0 && s.contains("}")) {
552                               level--;
553                           }
554                       }
555                       pw.println(s);
556                   }
557               }
558           }
559       }
