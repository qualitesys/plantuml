85       public static synchronized LicenseInfo retrieveQuick() {
86           if (cache == null) {
87               cache = retrieveDistributor();
88           }
89           if (cache == null) {
90               cache = retrieveNamedSlow();
91           }
92           return cache;
93       }

95       public static boolean retrieveNamedOrDistributorQuickIsValid() {
96           return retrieveQuick().isValid();

99       public static synchronized LicenseInfo retrieveNamedSlow() {
100           cache = LicenseInfo.NONE;
101           if (OptionFlags.ALLOW_INCLUDE == false) {
102               return cache;
103           }
104           final String key = prefs.get("license", "");
105           if (key.length() > 0) {
106               cache = setIfValid(retrieveNamed(key), cache);
107               if (cache.isValid()) {
108                   return cache;
109               }
110           }
111           for (SFile f : fileCandidates()) {
112               try {
113                   if (f.exists() && f.canRead()) {
114                       final LicenseInfo result = retrieve(f);
115                       if (result == null) {
116                           return null;
117                       }
118                       cache = setIfValid(result, cache);
119                       if (cache.isValid()) {
120                           return cache;
121                       }
122                   }
123               } catch (IOException e) {
124                   Log.info("Error " + e);
125                   // Logme.error(e);
126               }
127           }
128           return cache;
129       }

131       public static LicenseInfo retrieveNamed(final String key) {
132           if (key.length() > 99 && key.matches("^[0-9a-z]+$")) {
133               try {
134                   final String sig = SignatureUtils.toHexString(PLSSignature.signature());
135                   return PLSSignature.retrieveNamed(sig, key, true);
136               } catch (Exception e) {
137                   // Logme.error(e);
138                   Log.info("Error retrieving license info" + e);
139               }
140           }
141           return LicenseInfo.NONE;
142       }

144       public static BufferedImage retrieveDistributorImage(LicenseInfo licenseInfo) {
145           if (licenseInfo.getLicenseType() != LicenseType.DISTRIBUTOR) {
146               return null;
147           }
148           try {
149               final byte[] s1 = PLSSignature.retrieveDistributorImageSignature();
150               if (SignatureUtils.toHexString(s1).equals(SignatureUtils.toHexString(licenseInfo.sha)) == false) {
151                   return null;
152               }
153               final InputStream dis = PSystemVersion.class.getResourceAsStream("/distributor.png");
154               if (dis == null) {
155                   return null;
156               }
157               try {
158                   final BufferedImage result = SImageIO.read(dis);
159                   return result;
160               } finally {
161                   dis.close();
162               }
163           } catch (Exception e) {
164               Logme.error(e);
165           }
166           return null;
167       }

169       public static LicenseInfo retrieveDistributor() {
170           final InputStream dis = PSystemVersion.class.getResourceAsStream("/distributor.txt");
171           if (dis == null) {
172               return null;
173           }
174           try {
175               final BufferedReader br = new BufferedReader(new InputStreamReader(dis));
176               final String licenseString = br.readLine();
177               br.close();
178               final LicenseInfo result = PLSSignature.retrieveDistributor(licenseString);
179               final Throwable creationPoint = new Throwable();
180               creationPoint.fillInStackTrace();
181               for (StackTraceElement ste : creationPoint.getStackTrace()) {
182                   if (ste.toString().contains(result.context)) {
183                       return result;
184                   }
185               }
186               return null;
187           } catch (Exception e) {
188               Logme.error(e);
189               return null;
190           }
191       }

193       public static Collection<SFile> fileCandidates() {
194           final Set<SFile> result = new TreeSet<>();
195           final String classpath = System.getProperty("java.class.path");
196           String[] classpathEntries = classpath.split(SFile.pathSeparator);
197           for (String s : classpathEntries) {
198               if (s == null)
199                   continue;
200               SFile dir = new SFile(s);
201               if (dir.isFile()) {
202                   dir = dir.getParentFile();
203               }
204               if (dir != null && dir.isDirectory()) {
205                   result.add(dir.file("license.txt"));
206               }
207           }
208           return result;
209       }

211       private static LicenseInfo setIfValid(LicenseInfo value, LicenseInfo def) {
212           if (value.isValid() || def.isNone()) {
213               return value;
214           }
215           return def;
216       }

218       private static LicenseInfo retrieve(SFile f) throws IOException {
219           final BufferedReader br = f.openBufferedReader();
220           if (br == null) {
221               return null;
222           }
223           try {
224               final String s = br.readLine();
225               final LicenseInfo result = retrieveNamed(s);
226               if (result != null) {
227                   Log.info("Reading license from " + f.getAbsolutePath());
228               }
229               return result;
230           } finally {
231               br.close();
232           }
233       }
