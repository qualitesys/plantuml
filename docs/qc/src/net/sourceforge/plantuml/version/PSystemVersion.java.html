117       private static BufferedImage getImage(final String name) {
118           try {
119               final InputStream is = PSystemVersion.class.getResourceAsStream(name);
120               final BufferedImage image = SImageIO.read(is);
121               is.close();
122               return image;
123           } catch (IOException e) {
124               Logme.error(e);
125           }
126           return new BufferedImage(10, 10, BufferedImage.TYPE_INT_ARGB);
127       }

140       public static BufferedImage getPlantumlSmallIcon2() {
141           if (transparentIcon != null) {
142               return transparentIcon;
143           }
144           final BufferedImage ico = getPlantumlSmallIcon();
145           if (ico == null) {
146               return new BufferedImage(10, 10, BufferedImage.TYPE_INT_ARGB);
147           }
148           transparentIcon = new BufferedImage(ico.getWidth(), ico.getHeight(), BufferedImage.TYPE_INT_ARGB_PRE);
149           for (int i = 0; i < ico.getWidth(); i++) {
150               for (int j = 0; j < ico.getHeight(); j++) {
151                   final int col = ico.getRGB(i, j);
152                   if (col != ico.getRGB(0, 0)) {
153                       transparentIcon.setRGB(i, j, col);
154                   }
155               }
156           }
157           return transparentIcon;
158       }

160       public static PSystemVersion createShowVersion2(UmlSource source) {
161           final List<String> strings = new ArrayList<>();
162           strings.add("<b>PlantUML version " + Version.versionString() + "</b> (" + Version.compileTimeString() + ")");
163           strings.add("(" + License.getCurrent() + " source distribution)");
164           GraphvizCrash.checkOldVersionWarning(strings);
165           if (OptionFlags.ALLOW_INCLUDE) {
166               if (SecurityUtils.getSecurityProfile() == SecurityProfile.UNSECURE) {
167                   strings.add("Loaded from " + Version.getJarPath());
168               }
169               if (OptionFlags.getInstance().isWord()) {
170                   strings.add("Word Mode");
171                   strings.add("Command Line: " + Run.getCommandLine());
172                   strings.add("Current Dir: " + new SFile(".").getAbsolutePath());
173                   strings.add("plantuml.include.path: " + PreprocessorUtils.getenv(SecurityUtils.PATHS_INCLUDES));
174               }
175           }
176           strings.add(" ");
177           // strings.add("<b>Stdlib:");
178           // Stdlib.addInfoVersion(strings, false);
179           // strings.add(" ");
180   
181           GraphvizUtils.addDotStatus(strings, true);
182           strings.add(" ");
183           for (String name : OptionPrint.interestingProperties()) {
184               strings.add(name);
185           }
186           for (String v : OptionPrint.interestingValues()) {
187               strings.add(v);
188           }
189           
190           return new PSystemVersion(source, true, strings);
191       }

257       public static PSystemVersion createKeyDistributor(UmlSource source) throws IOException {
258           final LicenseInfo license = LicenseInfo.retrieveDistributor();
259           BufferedImage im = null;
260           final List<String> strings = new ArrayList<>();
261           if (license == null) {
262               strings.add("No license found");
263           } else {
264               strings.add(license.getOwner());
265               strings.add(license.getContext());
266               strings.add(license.getGenerationDate().toString());
267               strings.add(license.getExpirationDate().toString());
268               im = LicenseInfo.retrieveDistributorImage(license);
269           }
270           return new PSystemVersion(source, strings, im);
271       }
