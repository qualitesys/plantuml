88       public static PLSSignature fromRaw512(byte[] data) throws NoSuchAlgorithmException, IOException {
89           if (data.length != 512) {
90               throw new IllegalArgumentException();
91           }
92   
93           final byte resultA[] = new byte[64];
94           final byte resultB[] = new byte[512 - 64];
95           System.arraycopy(data, 0, resultA, 0, resultA.length);
96           System.arraycopy(data, 64, resultB, 0, resultB.length);
97   
98           final byte[] sig = SignatureUtils.getSHA512raw(resultB);
99           if (SignatureUtils.toHexString(resultA).equals(SignatureUtils.toHexString(sig)) == false) {
100               return null;
101               // throw new IOException();
102           }
103   
104           final ByteArrayInputStream bais = new ByteArrayInputStream(resultB);
105           final int type = bais.read();
106           if (type == 0) {
107               final int version = bais.read();
108               final byte sha[] = readBytes(bais, 64);
109               final long now = readLong(bais);
110               final long exp = readLong(bais);
111               final String owner = readString(bais);
112               return new PLSSignature(type, sha, now, exp, owner, null);
113           }
114           if (type == 2) {
115               final int version = bais.read();
116               final byte sha[] = readBytes(bais, 64);
117               final long now = readLong(bais);
118               final long exp = readLong(bais);
119               final String owner = readString(bais);
120               final String context = readString(bais);
121               return new PLSSignature(type, sha, now, exp, owner, context);
122           }
123           return null;
124       }

134       private static byte[] readBytes(ByteArrayInputStream bais, int size) throws IOException {
135           byte[] result = new byte[size];
136           final int read = bais.read(result);
137           if (read != size) {
138               throw new IOException();
139           }
140           return result;
141       }

143       private static String readString(ByteArrayInputStream bais) throws IOException {
144           final int size = bais.read();
145           if (size > 80) {
146               throw new IOException();
147           }
148           byte[] result = new byte[size];
149           final int read = bais.read(result);
150           if (read != size) {
151               throw new IOException();
152           }
153           return new String(result, UTF_8);
154       }

156       private static long readLong(ByteArrayInputStream bais) throws IOException {
157           final byte[] result = new byte[8];
158           final int read = bais.read(result);
159           if (read != 8) {
160               throw new IOException();
161           }
162           long ll = 0;
163           for (int i = 7; i >= 0; i--) {
164               final long mask = ((long) (result[i] & 0xFF)) << (8 * (7 - i));
165               ll = ll | mask;
166           }
167           return ll;
168       }

170       public static LicenseInfo retrieveNamed(String sig, String key, boolean doCheck) throws NoSuchAlgorithmException,
171               InvalidKeySpecException, IOException {
172           byte[] block = decode(key);
173           xor(block, SignatureUtils.getSHA512raw(SignatureUtils.salting(sig, getSalt(sig))));
174           final PLSSignature sig2 = PLSSignature.fromRaw512(block);
175           if (sig2 == null) {
176               return LicenseInfo.NONE;
177           }
178   
179           return sig2.toLicenseInfo();
180       }

191       private static byte[] decode(String key) throws IOException {
192           final BigInteger lu = new BigInteger(key, 36);
193           final QBlock qb2 = new QBlock(lu);
194           final QBlock qb3 = qb2.change(Dedication.E, Dedication.N);
195           byte block[] = qb3.getData512();
196           if (block.length != 512) {
197               throw new IOException();
198           }
199           return block;
200       }

202       private static void xor(byte buffer[], TurningBytes turningBytes) {
203           for (int i = 0; i < buffer.length; i++) {
204               buffer[i] ^= turningBytes.nextByte();
205           }
206       }

208       public static void xor(byte buffer[], byte[] key) {
209           xor(buffer, new TurningBytes(key));
210       }

212       public static byte[] getSalt(final String signature) throws UnsupportedEncodingException {
213           final Random rnd = new Random(getSeed(signature.getBytes(UTF_8)));
214           final byte salt[] = new byte[512];
215           rnd.nextBytes(salt);
216           return salt;
217       }

219       private static long getSeed(byte[] bytes) {
220           long result = 19;
221           for (byte b : bytes) {
222               result = result * 41 + b;
223           }
224           return result;
225       }

227       public static byte[] signature() throws IOException {
228           final String signature = OptionPrint.getHostName() + getMacAddress();
229           try {
230               return SignatureUtils.getSHA512raw(SignatureUtils.salting(signature, getSalt(signature)));
231           } catch (Exception e) {
232               Logme.error(e);
233               throw new IOException();
234           }
235       }

237       private static String getMacAddress() throws IOException {
238   
239           final Enumeration<NetworkInterface> net = NetworkInterface.getNetworkInterfaces();
240           final StringBuilder result = new StringBuilder();
241           while (net.hasMoreElements()) {
242               final NetworkInterface element = net.nextElement();
243               byte[] mac = element.getHardwareAddress();
244               if (mac != null) {
245                   for (byte b : mac) {
246                       result.append(String.format("%02x", b));
247                   }
248               }
249           }
250           return result.toString();
251       }
