62   public class PSystemKeygen extends PlainDiagram {
63   
64       final private String key;
65   
66       public PSystemKeygen(UmlSource source, String key) {
67           super(source);
68           this.key = key;
69       }
70   
71       @Override
72       protected UDrawable getRootDrawable(FileFormatOption fileFormatOption) {
73           return new UDrawable() {
74               public void drawU(UGraphic ug) {
75                   try {
76                       drawInternal(ug);
77                   } catch (IOException e) {
78                       Logme.error(e);
79                   }
80               }
81           };
82       }
83   
84       public DiagramDescription getDescription() {
85           return new DiagramDescription("(Key)");
86       }
87   
88       private void drawInternal(UGraphic ug) throws IOException {
89           final LicenseInfo installed = LicenseInfo.retrieveNamedSlow();
90           if (key.length() == 0) {
91               drawFlash(ug, installed);
92               return;
93           }
94           final LicenseInfo info = LicenseInfo.retrieveNamed(key);
95           if (info.isNone()) {
96               drawFlash(ug, installed);
97               return;
98           }
99           final List<String> strings = header();
100           strings.add("<u>Provided license information</u>:");
101           License.addLicenseInfo(strings, info);
102           strings.add(" ");
103           strings.add("========================================================================");
104           try {
105               LicenseInfo.persistMe(key);
106           } catch (BackingStoreException e) {
107               strings.add("<i>Error: Cannot store license key.</i>");
108           }
109   
110           if (installed.isNone()) {
111               strings.add("No license currently installed.");
112               strings.add(" ");
113               strings.add("<b>Please copy license.txt to one of those files</b>:");
114               for (SFile f : LicenseInfo.fileCandidates()) {
115                   strings.add(f.getAbsolutePath());
116               }
117               strings.add(" ");
118           } else {
119               strings.add("<u>Installed license</u>:");
120               License.addLicenseInfo(strings, installed);
121               strings.add(" ");
122           }
123   
124           final TextBlock disp = GraphicStrings.createBlackOnWhite(strings);
125           disp.drawU(ug);
126       }
127   
128       private ArrayList<String> header() {
129           final ArrayList<String> strings = new ArrayList<>();
130           strings.add("<b>PlantUML version " + Version.versionString() + "</b> (" + Version.compileTimeString() + ")");
131           strings.add("(" + License.getCurrent() + " source distribution)");
132   //        if (OptionFlags.ALLOW_INCLUDE) {
133   //            strings.add("Loaded from " + Version.getJarPath());
134   //        }
135           strings.add(" ");
136           return strings;
137       }
138   
139       private void drawFlash(UGraphic ug, LicenseInfo info) throws IOException {
140           final List<String> strings = header();
141           strings.add("To get your <i>Professional Edition License</i>,");
142           strings.add("please send this qrcode to <b>plantuml@gmail.com</b> :");
143   
144           TextBlock disp = GraphicStrings.createBlackOnWhite(strings);
145           disp.drawU(ug);
146   
147           ug = ug.apply(UTranslate.dy(disp.calculateDimension(ug.getStringBounder()).getHeight()));
148           final FlashCodeUtils utils = FlashCodeFactory.getFlashCodeUtils();
149           final BufferedImage im = utils.exportFlashcode(
150                   Version.versionString() + "\n" + SignatureUtils.toHexString(PLSSignature.signature()), Color.BLACK,
151                   Color.WHITE);
152           if (im != null) {
153               final UImage flash = new UImage(new PixelImage(im, AffineTransformType.TYPE_NEAREST_NEIGHBOR)).scale(4);
154               ug.draw(flash);
155               ug = ug.apply(UTranslate.dy(flash.getHeight()));
156           }
157   
158           if (info.isNone() == false) {
159               strings.clear();
160               strings.add("<u>Installed license</u>:");
161               License.addLicenseInfo(strings, info);
162               strings.add(" ");
163               disp = GraphicStrings.createBlackOnWhite(strings);
164               disp.drawU(ug);
165           }
166   
167       }
168   }
