12   public class Finalize {
13   
14       public static void finalize(Instance instance, Argon2 argon2) {
15   
16           Block finalBlock = instance.memory[instance.getLaneLength() - 1];
17   
18           /* XOR the last blocks */
19           for (int i = 1; i < instance.getLanes(); i++) {
20               int lastBlockInLane = i * instance.getLaneLength() + (instance.getLaneLength() - 1);
21               finalBlock.xorWith(instance.memory[lastBlockInLane]);
22           }
23   
24           byte[] finalBlockBytes = finalBlock.toBytes();
25           byte[] finalResult = Functions.blake2bLong(finalBlockBytes, argon2.getOutputLength());
26   
27           argon2.setOutput(finalResult);
28   
29           if (argon2.isClearMemory()) {
30               instance.clear();
31               argon2.clear();
32           }
33       }
34   }
