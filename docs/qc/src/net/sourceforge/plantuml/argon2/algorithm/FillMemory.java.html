10   import java.util.ArrayList;
11   import java.util.List;

20   public class FillMemory {
21   
22       public static void fillMemoryBlocks(Instance instance) {
23           if (instance.getLanes() == 1) {
24               fillMemoryBlockSingleThreaded(instance);
25           } else {
26               fillMemoryBlockMultiThreaded(instance);
27           }
28       }
29   
30       private static void fillMemoryBlockSingleThreaded(Instance instance) {
31           for (int i = 0; i < instance.getIterations(); i++) {
32               for (int j = 0; j < ARGON2_SYNC_POINTS; j++) {
33                   Position position = new Position(i, 0, j, 0);
34                   FillSegment.fillSegment(instance, position);
35               }
36           }
37       }
38   
39       private static void fillMemoryBlockMultiThreaded(final Instance instance) {
40   
41           ExecutorService service = Executors.newFixedThreadPool(instance.getLanes());
42           List<Future<?>> futures = new ArrayList<Future<?>>();
43   
44           for (int i = 0; i < instance.getIterations(); i++) {
45               for (int j = 0; j < ARGON2_SYNC_POINTS; j++) {
46                   for (int k = 0; k < instance.getLanes(); k++) {
47   
48                       final Position position = new Position(i, k, j, 0);
49   
50                       Future future = service.submit(new Runnable() {
51                           @Override
52                           public void run() {
53                               FillSegment.fillSegment(instance, position);
54                           }
55                       });
56   
57                       futures.add(future);
58                   }
59   
60                   joinThreads(instance, futures);
61               }
62           }
63   
64           service.shutdownNow();
65       }
66   
67       private static void joinThreads(Instance instance, List<Future<?>> futures) {
68           try {
69               for (Future<?> f : futures) {
70                   f.get();
71               }
72           } catch (InterruptedException e) {
73               instance.clear();
74               throw new RuntimeException(e);
75           } catch (ExecutionException e) {
76               instance.clear();
77               throw new RuntimeException(e);
78           }
79       }
80   }
