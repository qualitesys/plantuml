72        * H' - blake2bLong - variable length hash function
73        */
74       static byte[] blake2bLong(byte[] input, int outputLength) {
75   
76           assert (input.length == ARGON2_PREHASH_SEED_LENGTH || input.length == ARGON2_BLOCK_SIZE);
77   
78           byte[] result = new byte[outputLength];
79           byte[] outlenBytes = Util.intToLittleEndianBytes(outputLength);
80   
81           int blake2bLength = 64;
82   
83           if (outputLength <= blake2bLength) {
84               result = blake2b(input, outlenBytes, outputLength);
85           } else {
86               byte[] outBuffer;
87   
88               /* V1 */
89               outBuffer = blake2b(input, outlenBytes, blake2bLength);
90               System.arraycopy(outBuffer, 0, result, 0, blake2bLength / 2);
91   
92               int r = (outputLength / 32) + (outputLength % 32 == 0 ? 0 : 1) - 2;
93   
94               int position = blake2bLength / 2;
95               for (int i = 2; i <= r; i++, position += blake2bLength / 2) {
96                   /* V2 to Vr */
97                   outBuffer = blake2b(outBuffer, null, blake2bLength);
98                   System.arraycopy(outBuffer, 0, result, position, blake2bLength / 2);
99               }
100   
101               int lastLength = outputLength - 32 * r;
102   
103               /* Vr+1 */
104               outBuffer = blake2b(outBuffer, null, lastLength);
105               System.arraycopy(outBuffer, 0, result, position, lastLength);
106           }
107   
108           assert (result.length == outputLength);
109           return result;
110       }

112       private static byte[] blake2b(byte[] input, byte[] outlenBytes, int outputLength) {
113           Blake2b.Param params = new Blake2b.Param()
114                   .setDigestLength(outputLength);
115   
116           final Blake2b blake2b = Blake2b.Digest.newInstance(params);
117   
118           if (outlenBytes != null)
119               blake2b.update(outlenBytes);
120   
121           blake2b.update(input);
122   
123           return blake2b.digest();
124       }

143       private static void F(Block block, int a, int b, int c, int d) {
144           fBlaMka(block, a, b);
145           rotr64(block, d, a, 32);
146   
147           fBlaMka(block, c, d);
148           rotr64(block, b, c, 24);
149   
150           fBlaMka(block, a, b);
151           rotr64(block, d, a, 16);
152   
153           fBlaMka(block, c, d);
154           rotr64(block, b, c, 63);
155       }
156   
157       /*designed by the Lyra PHC team */
158       /* a <- a + b + 2*aL*bL
159        * + == addition modulo 2^64
160        * aL = least 32 bit */
161       private static void fBlaMka(Block block, int x, int y) {
162           final long m = 0xFFFFFFFFL;
163           final long xy = (block.v[x] & m) * (block.v[y] & m);
164   
165           block.v[x] = block.v[x] + block.v[y] + 2 * xy;
166       }

168       private static void rotr64(Block block, int v, int w, long c) {
169           final long temp = block.v[v] ^ block.v[w];
170           block.v[v] = (temp >>> c) | (temp << (64 - c));
171       }
