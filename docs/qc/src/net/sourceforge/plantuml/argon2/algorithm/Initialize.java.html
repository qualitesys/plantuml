19       public static void initialize(Instance instance, Argon2 argon2) {
20           byte[] initialHash = Functions.initialHash(
21                   Util.intToLittleEndianBytes(argon2.getLanes()),
22                   Util.intToLittleEndianBytes(argon2.getOutputLength()),
23                   Util.intToLittleEndianBytes(argon2.getMemory()),
24                   Util.intToLittleEndianBytes(argon2.getIterations()),
25                   Util.intToLittleEndianBytes(argon2.getVersion()),
26                   Util.intToLittleEndianBytes(argon2.getType().ordinal()),
27                   Util.intToLittleEndianBytes(argon2.getPasswordLength()),
28                   argon2.getPassword(),
29                   Util.intToLittleEndianBytes(argon2.getSaltLength()),
30                   argon2.getSalt(),
31                   Util.intToLittleEndianBytes(argon2.getSecretLength()),
32                   argon2.getSecret(),
33                   Util.intToLittleEndianBytes(argon2.getAdditionalLength()),
34                   argon2.getAdditional()
35           );
36           fillFirstBlocks(instance, initialHash);
37       }
38   
39       /**
40        * (H0 || 0 || i) 72 byte -> 1024 byte
41        * (H0 || 1 || i) 72 byte -> 1024 byte
42        */
43       private static void fillFirstBlocks(Instance instance, byte[] initialHash) {
44   
45           final byte[] zeroBytes = {0, 0, 0, 0};
46           final byte[] oneBytes = {1, 0, 0, 0};
47   
48           byte[] initialHashWithZeros = getInitialHashLong(initialHash, zeroBytes);
49           byte[] initialHashWithOnes = getInitialHashLong(initialHash, oneBytes);
50   
51           for (int i = 0; i < instance.getLanes(); i++) {
52   
53               byte[] iBytes = Util.intToLittleEndianBytes(i);
54   
55               System.arraycopy(iBytes, 0, initialHashWithZeros, ARGON2_PREHASH_DIGEST_LENGTH + 4, 4);
56               System.arraycopy(iBytes, 0, initialHashWithOnes, ARGON2_PREHASH_DIGEST_LENGTH + 4, 4);
57   
58               byte[] blockhashBytes = Functions.blake2bLong(initialHashWithZeros, ARGON2_BLOCK_SIZE);
59               instance.memory[i * instance.getLaneLength() + 0].fromBytes(blockhashBytes);
60   
61               blockhashBytes = Functions.blake2bLong(initialHashWithOnes, ARGON2_BLOCK_SIZE);
62               instance.memory[i * instance.getLaneLength() + 1].fromBytes(blockhashBytes);
63           }
64       }
65   
66       private static byte[] getInitialHashLong(byte[] initialHash, byte[] appendix) {
67           byte[] initialHashLong = new byte[ARGON2_PREHASH_SEED_LENGTH];
68   
69           System.arraycopy(initialHash, 0, initialHashLong, 0, ARGON2_PREHASH_DIGEST_LENGTH);
70           System.arraycopy(appendix, 0, initialHashLong, ARGON2_PREHASH_DIGEST_LENGTH, 4);
71   
72           return initialHashLong;
73       }
