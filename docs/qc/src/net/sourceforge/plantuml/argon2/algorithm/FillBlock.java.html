6   package net.sourceforge.plantuml.argon2.algorithm;
7   
8   import net.sourceforge.plantuml.argon2.model.Block;
9   
10   class FillBlock {
11   
12       static void fillBlock(Block X, Block Y, Block currentBlock, boolean withXor) {
13   
14           Block R = new Block();
15           Block Z = new Block();
16   
17           R.xor(X, Y);
18           Z.copyBlock(R);
19   
20           /* Apply Blake2 on columns of 64-bit words: (0,1,...,15) , then
21           (16,17,..31)... finally (112,113,...127) */
22           for (int i = 0; i < 8; i++) {
23   
24               Functions.roundFunction(Z,
25                       16 * i, 16 * i + 1, 16 * i + 2,
26                       16 * i + 3, 16 * i + 4, 16 * i + 5,
27                       16 * i + 6, 16 * i + 7, 16 * i + 8,
28                       16 * i + 9, 16 * i + 10, 16 * i + 11,
29                       16 * i + 12, 16 * i + 13, 16 * i + 14,
30                       16 * i + 15
31               );
32           }
33   
34           /* Apply Blake2 on rows of 64-bit words: (0,1,16,17,...112,113), then
35           (2,3,18,19,...,114,115).. finally (14,15,30,31,...,126,127) */
36           for (int i = 0; i < 8; i++) {
37   
38               Functions.roundFunction(Z,
39                       2 * i, 2 * i + 1, 2 * i + 16,
40                       2 * i + 17, 2 * i + 32, 2 * i + 33,
41                       2 * i + 48, 2 * i + 49, 2 * i + 64,
42                       2 * i + 65, 2 * i + 80, 2 * i + 81,
43                       2 * i + 96, 2 * i + 97, 2 * i + 112,
44                       2 * i + 113
45               );
46   
47           }
48   
49           if (withXor) {
50               currentBlock.xor(R, Z, currentBlock);
51           } else {
52               currentBlock.xor(R, Z);
53           }
54       }
55   }
