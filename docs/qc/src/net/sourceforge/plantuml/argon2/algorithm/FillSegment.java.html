19   class FillSegment {
20   
21       static void fillSegment(Instance instance, Position position) {
22   
23           Block addressBlock = null, inputBlock = null, zeroBlock = null;
24   
25           boolean dataIndependentAddressing = isDataIndependentAddressing(instance, position);
26           int startingIndex = getStartingIndex(position);
27           int currentOffset = position.lane * instance.getLaneLength() + position.slice * instance.getSegmentLength() + startingIndex;
28           int prevOffset = getPrevOffset(instance, currentOffset);
29   
30           if (dataIndependentAddressing) {
31               addressBlock = new Block();
32               zeroBlock = new Block();
33               inputBlock = new Block();
34   
35               initAddressBlocks(instance, position, zeroBlock, inputBlock, addressBlock);
36           }
37   
38           for (position.index = startingIndex; position.index < instance.getSegmentLength(); position.index++, currentOffset++, prevOffset++) {
39               prevOffset = rotatePrevOffset(instance, currentOffset, prevOffset);
40   
41               long pseudoRandom = getPseudoRandom(instance, position, addressBlock, inputBlock, zeroBlock, prevOffset, dataIndependentAddressing);
42               int refLane = getRefLane(instance, position, pseudoRandom);
43               int refColumn = getRefColumn(instance, position, pseudoRandom, refLane == position.lane);
44   
45               /* 2 Creating a new block */
46               Block prevBlock = instance.memory[prevOffset];
47               Block refBlock = instance.memory[((instance.getLaneLength()) * refLane + refColumn)];
48               Block currentBlock = instance.memory[currentOffset];
49   
50               boolean withXor = isWithXor(instance, position);
51               FillBlock.fillBlock(prevBlock, refBlock, currentBlock, withXor);
52           }
53       }
54   
55       private static boolean isDataIndependentAddressing(Instance instance, Position position) {
56           return (instance.getType() == Argon2Type.Argon2i) ||
57                   (instance.getType() == Argon2Type.Argon2id
58                           && (position.pass == 0)
59                           && (position.slice < Constants.ARGON2_SYNC_POINTS / 2)
60                   );
61       }
62   
63       private static void initAddressBlocks(Instance instance, Position position, Block zeroBlock, Block inputBlock, Block addressBlock) {
64           inputBlock.v[0] = Util.intToLong(position.pass);
65           inputBlock.v[1] = Util.intToLong(position.lane);
66           inputBlock.v[2] = Util.intToLong(position.slice);
67           inputBlock.v[3] = Util.intToLong(instance.memory.length);
68           inputBlock.v[4] = Util.intToLong(instance.getIterations());
69           inputBlock.v[5] = Util.intToLong(instance.getType().ordinal());
70   
71           if ((position.pass == 0) && (position.slice == 0)) {
72               /* Don't forget to generate the first block of addresses: */
73               nextAddresses(zeroBlock, inputBlock, addressBlock);
74           }
75       }
76   
77       private static boolean isWithXor(Instance instance, Position position) {
78           return !(position.pass == 0 || instance.getVersion() == ARGON2_VERSION_10);
79       }
80   
81       private static int getPrevOffset(Instance instance, int currentOffset) {
82           if (currentOffset % instance.getLaneLength() == 0) {
83               /* Last block in this lane */
84               return currentOffset + instance.getLaneLength() - 1;
85           } else {
86               /* Previous block */
87               return currentOffset - 1;
88           }
89       }
90   
91       private static int rotatePrevOffset(Instance instance, int currentOffset, int prevOffset) {
92           if (currentOffset % instance.getLaneLength() == 1) {
93               prevOffset = currentOffset - 1;
94           }
95           return prevOffset;
96       }
97   
98       private static int getStartingIndex(Position position) {
99           if ((position.pass == 0) && (position.slice == 0)) {
100               return 2; /* we have already generated the first two blocks */
101           } else {
102               return 0;
103           }
104       }
105   
106       private static void nextAddresses(Block zeroBlock, Block inputBlock, Block addressBlock) {
107           inputBlock.v[6]++;
108           FillBlock.fillBlock(zeroBlock, inputBlock, addressBlock, false);
109           FillBlock.fillBlock(zeroBlock, addressBlock, addressBlock, false);
110       }
111   
112       /* 1.2 Computing the index of the reference block */
113       /* 1.2.1 Taking pseudo-random value from the previous block */
114       private static long getPseudoRandom(Instance instance, Position position, Block addressBlock, Block inputBlock, Block zeroBlock, int prevOffset, boolean dataIndependentAddressing) {
115           if (dataIndependentAddressing) {
116               if (position.index % ARGON2_ADDRESSES_IN_BLOCK == 0) {
117                   nextAddresses(zeroBlock, inputBlock, addressBlock);
118               }
119               return addressBlock.v[position.index % ARGON2_ADDRESSES_IN_BLOCK];
120           } else {
121               return instance.memory[prevOffset].v[0];
122           }
123       }
124   
125       private static int getRefLane(Instance instance, Position position, long pseudoRandom) {
126           int refLane = (int) (((pseudoRandom >>> 32)) % instance.getLanes());
127   
128           if ((position.pass == 0) && (position.slice == 0)) {
129               /* Can not reference other lanes yet */
130               refLane = position.lane;
131           }
132           return refLane;
133       }
134   
135       private static int getRefColumn(Instance instance, Position position, long pseudoRandom,
136                                       boolean sameLane) {
137   
138           int referenceAreaSize;
139           int startPosition;
140   
141           if (position.pass == 0) {
142               startPosition = 0;
143   
144               if (sameLane) {
145                   /* The same lane => add current segment */
146                   referenceAreaSize = position.slice * instance.getSegmentLength() + position.index - 1;
147               } else {
148                   /* pass == 0 && !sameLane => position.slice > 0*/
149                   referenceAreaSize = position.slice * instance.getSegmentLength() + ((position.index == 0) ? (-1) : 0);
150               }
151   
152           } else {
153               startPosition = ((position.slice + 1) * instance.getSegmentLength()) % instance.getLaneLength();
154   
155               if (sameLane) {
156                   referenceAreaSize = instance.getLaneLength() - instance.getSegmentLength() + position.index - 1;
157               } else {
158                   referenceAreaSize = instance.getLaneLength() - instance.getSegmentLength() + ((position.index == 0) ? (-1) : 0);
159               }
160           }
161   
162           long relativePosition = pseudoRandom & 0xFFFFFFFFL;
163   //        long relativePosition = pseudoRandom << 32 >>> 32;
164           relativePosition = (relativePosition * relativePosition) >>> 32;
165           relativePosition = referenceAreaSize - 1 - (referenceAreaSize * relativePosition >>> 32);
166   
167           return (int) (startPosition + relativePosition) % instance.getLaneLength();
168       }
