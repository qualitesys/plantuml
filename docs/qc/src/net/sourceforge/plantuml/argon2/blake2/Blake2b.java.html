406           @Override final public void update (final byte[] b, int off, int len) {
407               if (b == null) {
408                   throw new IllegalArgumentException("input buffer (b) is null");
409               }
410               /* zero or more calls to compress */
411               // REVU: possibly the double buffering of c-ref is more sensible ..
412               //       regardless, the hotspot is in the compress, as expected.
413               while (len > 0) {
414                   if ( buflen == 0) {
415                       /* try compressing direct from input ? */
416                       while ( len > Spec.block_bytes ) {
