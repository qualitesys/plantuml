32   class Validation {
33   
34       static void validateInput(Argon2 argon2){
35           String message = null;
36   
37           if (argon2.getLanes() < MIN_PARALLELISM)
38               message = P_MIN_MSG;
39           else if (argon2.getLanes() > MAX_PARALLELISM)
40               message = P_MAX_MSG;
41           else if(argon2.getMemory() < 2 * argon2.getLanes())
42               message = M_MIN_MSG;
43           else if(argon2.getIterations() < MIN_ITERATIONS)
44               message = T_MIN_MSG;
45           else if(argon2.getIterations() > MAX_ITERATIONS)
46               message = T_MAX_MSG;
47           else if(argon2.getPasswordLength() < MIN_PWD_LENGTH)
48               message = PWD_MIN_MSG;
49           else if(argon2.getPasswordLength() > MAX_PWD_LENGTH)
50               message = PWD_MAX_MSG;
51           else if(argon2.getSaltLength() < MIN_SALT_LENGTH)
52               message = SALT_MIN_MSG;
53           else if(argon2.getSaltLength() > MAX_SALT_LENGTH)
54               message = SALT_MAX_MSG;
55           else if(argon2.getSecretLength() > MAX_SECRET_LENGTH)
56               message = SECRET_MAX_MSG;
57           else if(argon2.getAdditionalLength() > MAX_AD_LENGTH)
58               message = ADDITIONAL_MAX_MSG;
59   
60               if(message != null)
61                   throw new Argon2InvalidParameterException(message);
62       }
63   }
