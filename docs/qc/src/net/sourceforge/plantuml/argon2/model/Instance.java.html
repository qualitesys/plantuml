6   package net.sourceforge.plantuml.argon2.model;
7   
8   import static net.sourceforge.plantuml.argon2.Constants.ARGON2_SYNC_POINTS;
9   
10   import net.sourceforge.plantuml.argon2.Argon2;
11   
12   public class Instance {
13   
14       public Block[] memory;
15       private int version;
16       private int iterations;
17       private int segmentLength;
18       private int laneLength;
19       private int lanes;
20   
21       private Argon2Type type;
22   
23       public Instance(Argon2 argon2) {
24           this.version = argon2.getVersion();
25           this.iterations = argon2.getIterations();
26           this.lanes = argon2.getLanes();
27           this.type = argon2.getType();
28   
29           /* 2. Align memory size */
30           /* Minimum memoryBlocks = 8L blocks, where L is the number of lanes */
31           int memoryBlocks = argon2.getMemory();
32   
33           if (memoryBlocks < 2 * ARGON2_SYNC_POINTS * argon2.getLanes()) {
34               memoryBlocks = 2 * ARGON2_SYNC_POINTS * argon2.getLanes();
35           }
36   
37           this.segmentLength = memoryBlocks / (argon2.getLanes() * ARGON2_SYNC_POINTS);
38           this.laneLength = segmentLength * ARGON2_SYNC_POINTS;
39           /* Ensure that all segments have equal length */
40           memoryBlocks = segmentLength * (argon2.getLanes() * ARGON2_SYNC_POINTS);
41   
42           initMemory(memoryBlocks);
43       }
44   
45       private void initMemory(int memoryBlocks) {
46           this.memory = new Block[memoryBlocks];
47   
48           for (int i = 0; i < memory.length; i++) {
49               memory[i] = new Block();
50           }
51       }
52   
53       public void clear() {
54           for (Block b : memory) {
55               b.clear();
56           }
57   
58           memory = null;
59       }
60   
61       public Block[] getMemory() {
62           return memory;
63       }
64   
65       public int getVersion() {
66           return version;
67       }
68   
69       public int getIterations() {
70           return iterations;
71       }
72   
73       public int getSegmentLength() {
74           return segmentLength;
75       }
76   
77       public int getLaneLength() {
78           return laneLength;
79       }
80   
81       public int getLanes() {
82           return lanes;
83       }
84   
85       public Argon2Type getType() {
86           return type;
87       }
88   }
