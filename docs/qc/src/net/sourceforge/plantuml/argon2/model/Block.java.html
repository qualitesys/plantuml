6   package net.sourceforge.plantuml.argon2.model;

8   
9   import static net.sourceforge.plantuml.argon2.Constants.ARGON2_BLOCK_SIZE;
10   import static net.sourceforge.plantuml.argon2.Constants.ARGON2_QWORDS_IN_BLOCK;
11   
12   import java.util.Arrays;
13   
14   import net.sourceforge.plantuml.argon2.Util;
15   
16   public class Block {
17   
18       /* 128 * 8 Byte QWords */
19       public long[] v;
20   
21       public Block() {
22           v = new long[ARGON2_QWORDS_IN_BLOCK];
23       }
24   
25       public void fromBytes(byte[] input) {
26           assert (input.length == ARGON2_BLOCK_SIZE);
27   
28           for (int i = 0; i < v.length; i++) {
29               byte[] slice = Arrays.copyOfRange(input, i * 8, (i + 1) * 8);
30               v[i] = Util.littleEndianBytesToLong(slice);
31           }
32       }
33   
34       public byte[] toBytes() {
35           byte[] result = new byte[ARGON2_BLOCK_SIZE];
36   
37           for (int i = 0; i < v.length; i++) {
38               byte[] bytes = Util.longToLittleEndianBytes(v[i]);
39               System.arraycopy(bytes, 0, result, i * bytes.length, bytes.length);
40           }
41   
42           return result;
43       }
44   
45       public void copyBlock(Block other) {
46           System.arraycopy(other.v, 0, v, 0, v.length);
47       }
48   
49       public void xor(Block b1, Block b2) {
50           for (int i = 0; i < v.length; i++) {
51               v[i] = b1.v[i] ^ b2.v[i];
52           }
53       }
54   
55       public void xor(Block b1, Block b2, Block b3) {
56           for (int i = 0; i < v.length; i++) {
57               v[i] = b1.v[i] ^ b2.v[i] ^ b3.v[i];
58           }
59       }
60   
61       public void xorWith(Block other) {
62           for (int i = 0; i < v.length; i++) {
63               v[i] = v[i] ^ other.v[i];
64           }
65       }
66   
67       @Override
68       public String toString() {
69           StringBuilder result = new StringBuilder();
70           for (long value : v) {
71               result.append(Util.bytesToHexString(Util.longToLittleEndianBytes(value)));
72           }
73   
74           return result.toString();
75       }
76   
77       void clear() {
78           Arrays.fill(v, 0);
79       }
80   }
