57   /**
58    * A Base64 encoder/decoder.
59    * 
60    * <p>
61    * This class is used to encode and decode data in Base64 format as described in RFC 1521.
62    * <p>
63    * Project home page: <a href="http://www.source-code.biz/base64coder/java/">www.source-code.biz/base64coder/java</a><br>
64    * Author: Christian d'Heureuse, Inventec Informatik AG, Zurich, Switzerland<br>
65    * Multi-licensed: EPL / LGPL / GPL / AL / BSD / MIT.
66    * 
67    * @author Christian d'Heureuse, Inventec Informatik AG, Zurich, Switzerland, www.source-code.biz
68    */
69   public class Base64Coder {
70   
71       // The line separator string of the operating system.
72       private static final String systemLineSeparator = System.getProperty("line.separator");
73   
74       // Mapping table from 6-bit nibbles to Base64 characters.
75       private static final char[] map1 = new char[64];
76       static {
77           int i = 0;
78           for (char c = 'A'; c <= 'Z'; c++)
79               map1[i++] = c;
80           for (char c = 'a'; c <= 'z'; c++)
81               map1[i++] = c;
82           for (char c = '0'; c <= '9'; c++)
83               map1[i++] = c;
84           map1[i++] = '+';
85           map1[i++] = '/';
86       }
87   
88       // Mapping table from Base64 characters to 6-bit nibbles.
89       private static final byte[] map2 = new byte[128];
90       static {
91           for (int i = 0; i < map2.length; i++)
92               map2[i] = -1;
93           for (int i = 0; i < 64; i++)
94               map2[map1[i]] = (byte) i;
95       }

97       /**
98        * Encodes a string into Base64 format. No blanks or line breaks are inserted.
99        * 
100        * @param s
101        *            A String to be encoded.
102        * @return A String containing the Base64 encoded data.
103        */
104       public static String encodeString(String s) {
105           return new String(encode(s.getBytes()));
106       }
107   
108       /**
109        * Encodes a byte array into Base 64 format and breaks the output into lines of 76 characters. This method is
110        * compatible with <code>sun.misc.BASE64Encoder.encodeBuffer(byte[])</code>.
111        * 
112        * @param in
113        *            An array containing the data bytes to be encoded.
114        * @return A String containing the Base64 encoded data, broken into lines.
115        */
116       public static String encodeLines(byte[] in) {
117           return encodeLines(in, 0, in.length, 76, systemLineSeparator);
118       }
119   
120       /**
121        * Encodes a byte array into Base 64 format and breaks the output into lines.
122        * 
123        * @param in
124        *            An array containing the data bytes to be encoded.
125        * @param iOff
126        *            Offset of the first byte in <code>in</code> to be processed.
127        * @param iLen
128        *            Number of bytes to be processed in <code>in</code>, starting at <code>iOff</code>.
129        * @param lineLen
130        *            Line length for the output data. Should be a multiple of 4.
131        * @param lineSeparator
132        *            The line separator to be used to separate the output lines.
133        * @return A String containing the Base64 encoded data, broken into lines.
134        */
135       public static String encodeLines(byte[] in, int iOff, int iLen, int lineLen, String lineSeparator) {
136           int blockLen = (lineLen * 3) / 4;
137           if (blockLen <= 0)
138               throw new IllegalArgumentException();
139           int lines = (iLen + blockLen - 1) / blockLen;
140           int bufLen = ((iLen + 2) / 3) * 4 + lines * lineSeparator.length();
141           StringBuilder buf = new StringBuilder(bufLen);
142           int ip = 0;
143           while (ip < iLen) {
144               int l = Math.min(iLen - ip, blockLen);
145               buf.append(encode(in, iOff + ip, l));
146               buf.append(lineSeparator);
147               ip += l;
148           }
149           return buf.toString();
150       }

153        * Encodes a byte array into Base64 format. No blanks or line breaks are inserted in the output.
154        * 
155        * @param in
156        *            An array containing the data bytes to be encoded.
157        * @return A character array containing the Base64 encoded data.
158        */
159       public static char[] encode(byte[] in) {
160           return encode(in, 0, in.length);
161       }

163       /**
164        * Encodes a byte array into Base64 format. No blanks or line breaks are inserted in the output.
165        * 
166        * @param in
167        *            An array containing the data bytes to be encoded.
168        * @param iLen
169        *            Number of bytes to process in <code>in</code>.
170        * @return A character array containing the Base64 encoded data.
171        */
172       public static char[] encode(byte[] in, int iLen) {
173           return encode(in, 0, iLen);
174       }
175   
176       /**
177        * Encodes a byte array into Base64 format. No blanks or line breaks are inserted in the output.
178        * 
179        * @param in
180        *            An array containing the data bytes to be encoded.
181        * @param iOff
182        *            Offset of the first byte in <code>in</code> to be processed.
183        * @param iLen
184        *            Number of bytes to process in <code>in</code>, starting at <code>iOff</code>.
185        * @return A character array containing the Base64 encoded data.
186        */
187       public static char[] encode(byte[] in, int iOff, int iLen) {
188           int oDataLen = (iLen * 4 + 2) / 3; // output length without padding
189           int oLen = ((iLen + 2) / 3) * 4; // output length including padding
190           char[] out = new char[oLen];
191           int ip = iOff;
192           int iEnd = iOff + iLen;
193           int op = 0;
194           while (ip < iEnd) {
195               int i0 = in[ip++] & 0xff;
196               int i1 = ip < iEnd ? in[ip++] & 0xff : 0;
197               int i2 = ip < iEnd ? in[ip++] & 0xff : 0;
198               int o0 = i0 >>> 2;
199               int o1 = ((i0 & 3) << 4) | (i1 >>> 4);
200               int o2 = ((i1 & 0xf) << 2) | (i2 >>> 6);
201               int o3 = i2 & 0x3F;
202               out[op++] = map1[o0];
203               out[op++] = map1[o1];
204               out[op] = op < oDataLen ? map1[o2] : '=';
205               op++;
206               out[op] = op < oDataLen ? map1[o3] : '=';
207               op++;
208           }
209           return out;
210       }

212       /**
213        * Decodes a string from Base64 format. No blanks or line breaks are allowed within the Base64 encoded input data.
214        * 
215        * @param s
216        *            A Base64 String to be decoded.
217        * @return A String containing the decoded data.
218        * @throws IllegalArgumentException
219        *             If the input is not valid Base64 encoded data.
220        */
221       public static String decodeString(String s) {
222           return new String(decode(s));
223       }
224   
225       /**
226        * Decodes a byte array from Base64 format and ignores line separators, tabs and blanks. CR, LF, Tab and Space
227        * characters are ignored in the input data. This method is compatible with
228        * <code>sun.misc.BASE64Decoder.decodeBuffer(String)</code>.
229        * 
230        * @param s
231        *            A Base64 String to be decoded.
232        * @return An array containing the decoded data bytes.
233        * @throws IllegalArgumentException
234        *             If the input is not valid Base64 encoded data.
235        */
236       public static byte[] decodeLines(String s) {
237           char[] buf = new char[s.length()];
238           int p = 0;
239           for (int ip = 0; ip < s.length(); ip++) {
240               char c = s.charAt(ip);
241               if (c != ' ' && c != '\r' && c != '\n' && c != '\t')
242                   buf[p++] = c;
243           }
244           return decode(buf, 0, p);
245       }

248        * Decodes a byte array from Base64 format. No blanks or line breaks are allowed within the Base64 encoded input
249        * data.
250        * 
251        * @param s
252        *            A Base64 String to be decoded.
253        * @return An array containing the decoded data bytes.
254        * @throws IllegalArgumentException
255        *             If the input is not valid Base64 encoded data.
256        */
257       public static byte[] decode(String s) {
258           return decode(s.toCharArray());
259       }

261       /**
262        * Decodes a byte array from Base64 format. No blanks or line breaks are allowed within the Base64 encoded input
263        * data.
264        * 
265        * @param in
266        *            A character array containing the Base64 encoded data.
267        * @return An array containing the decoded data bytes.
268        * @throws IllegalArgumentException
269        *             If the input is not valid Base64 encoded data.
270        */
271       public static byte[] decode(char[] in) {
272           return decode(in, 0, in.length);
273       }

275       /**
276        * Decodes a byte array from Base64 format. No blanks or line breaks are allowed within the Base64 encoded input
277        * data.
278        * 
279        * @param in
280        *            A character array containing the Base64 encoded data.
281        * @param iOff
282        *            Offset of the first character in <code>in</code> to be processed.
283        * @param iLen
284        *            Number of characters to process in <code>in</code>, starting at <code>iOff</code>.
285        * @return An array containing the decoded data bytes.
286        * @throws IllegalArgumentException
287        *             If the input is not valid Base64 encoded data.
288        */
289       public static byte[] decode(char[] in, int iOff, int iLen) {
290           if (iLen % 4 != 0)
291               throw new IllegalArgumentException("Length of Base64 encoded input string is not a multiple of 4.");
292           while (iLen > 0 && in[iOff + iLen - 1] == '=')
293               iLen--;
294           int oLen = (iLen * 3) / 4;
295           byte[] out = new byte[oLen];
296           int ip = iOff;
297           int iEnd = iOff + iLen;
298           int op = 0;
299           while (ip < iEnd) {
300               int i0 = in[ip++];
301               int i1 = in[ip++];
302               int i2 = ip < iEnd ? in[ip++] : 'A';
303               int i3 = ip < iEnd ? in[ip++] : 'A';
304               if (i0 > 127 || i1 > 127 || i2 > 127 || i3 > 127)
305                   throw new IllegalArgumentException("Illegal character in Base64 encoded data.");
306               int b0 = map2[i0];
307               int b1 = map2[i1];
308               int b2 = map2[i2];
309               int b3 = map2[i3];
310               if (b0 < 0 || b1 < 0 || b2 < 0 || b3 < 0)
311                   throw new IllegalArgumentException("Illegal character in Base64 encoded data.");
312               int o0 = (b0 << 2) | (b1 >>> 4);
313               int o1 = ((b1 & 0xf) << 4) | (b2 >>> 2);
314               int o2 = ((b2 & 3) << 6) | b3;
315               out[op++] = (byte) o0;
316               if (op < oLen)
317                   out[op++] = (byte) o1;
318               if (op < oLen)
319                   out[op++] = (byte) o2;
320           }
321           return out;
322       }
