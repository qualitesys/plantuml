11   import java.io.IOException;
12   import java.util.Objects;

15   /**
16    * Stores a finite recent history of a byte stream. Useful as an implicit
17    * dictionary for Lempel-Ziv schemes. Mutable and not thread-safe.
18    */
19   final class ByteHistory {
20       
21       /*---- Fields ----*/
22       
23       // Circular buffer of byte data.
24       private byte[] data;
25       
26       // Index of next byte to write to, always in the range [0, data.length).
27       private int index;
28       
29       
30       
31       /*---- Constructor ----*/
32       
33       /**
34        * Constructs a byte history of the specified size, initialized to zeros.
35        * @param size the size, which must be positive
36        * @throws IllegalArgumentException if size is zero or negative
37        */
38       public ByteHistory(int size) {
39           if (size < 1)
40               throw new IllegalArgumentException("Size must be positive");
41           data = new byte[size];
42           index = 0;
43       }
44       
45       
46       
47       /*---- Methods ----*/
48       
49       /**
50        * Appends the specified byte to this history.
51        * This overwrites the byte value at {@code size} positions ago.
52        * @param b the byte value to append
53        */
54       public void append(int b) {
55           if (index < 0 || index >= data.length)
56               throw new IllegalStateException();
57           data[index] = (byte)b;
58           index = (index + 1) % data.length;
59       }
60       
61       
62       /**
63        * Copies {@code len} bytes starting at {@code dist} bytes ago to
64        * the specified output stream and also back into this buffer itself.
65        * <p>Note that if the length exceeds the distance, then some of the output
66        * data will be a copy of data that was copied earlier in the process.</p>
67        * @param dist the distance to go back, in the range [1, size]
68        * @param len the length to copy, which must be at least 0
69        * @param out the output stream to write to (not {@code null})
70        * @throws NullPointerException if the output stream is {@code null}
71        * @throws IllegalArgumentException if the length is negative,
72        * distance is not positive, or distance is greater than the buffer size
73        * @throws IOException if an I/O exception occurs
74        */
75       public void copy(int dist, int len, OutputStreamProtected out) throws IOException {
76           Objects.requireNonNull(out);
77           if (len < 0 || dist < 1 || dist > data.length)
78               throw new IllegalArgumentException();
79           
80           int readIndex = (index - dist + data.length) % data.length;
81           if (readIndex < 0 || readIndex >= data.length)
82               throw new IllegalStateException();
83           
84           for (int i = 0; i < len; i++) {
85               byte b = data[readIndex];
86               readIndex = (readIndex + 1) % data.length;
87               out.write(b);
88               append(b);
89           }
90       }
91       
92   }
