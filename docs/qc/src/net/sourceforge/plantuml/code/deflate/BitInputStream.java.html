11   import java.io.Closeable;

16   /**
17    * A stream of bits that can be read. Bits are packed in little endian within a byte.
18    * For example, the byte 0x87 reads as the sequence of bits [1,1,1,0,0,0,0,1].
19    */
20   public interface BitInputStream extends Closeable {
21       
22       /**
23        * Returns the current bit position, which ascends from 0 to 7 as bits are read.
24        * @return the current bit position, which is between 0 and 7
25        */
26       public int getBitPosition();
27       
28       
29       /**
30        * Discards the remainder of the current byte (if any) and reads the next
31        * whole byte from the stream. Returns -1 if the end of stream is reached.
32        * @return the next byte from the stream, or -1 if the end of stream is reached
33        */
34       public int readByte() throws IOException;
35       
36       
37       /**
38        * Reads a bit from this stream. Returns 0 or 1 if a bit is available, or -1 if
39        * the end of stream is reached. The end of stream always occurs on a byte boundary.
40        * @return the next bit of 0 or 1, or -1 for the end of stream
41        * @throws IOException if an I/O exception occurred
42        */
43       public int read() throws IOException;
44       
45       
46       /**
47        * Reads a bit from this stream. Returns 0 or 1 if a bit is available, or throws an {@code EOFException}
48        * if the end of stream is reached. The end of stream always occurs on a byte boundary.
49        * @return the next bit of 0 or 1
50        * @throws IOException if an I/O exception occurred
51        * @throws EOFException if the end of stream is reached
52        */
53       public int readNoEof() throws IOException;
54       
55       
56       /**
57        * Closes this stream and the underlying input stream.
58        * @throws IOException if an I/O exception occurred
59        */
60       public void close() throws IOException;
61       
62   }
