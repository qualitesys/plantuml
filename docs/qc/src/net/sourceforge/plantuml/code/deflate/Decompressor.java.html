10   /**
11    * Decompresses raw DEFLATE data (without zlib or gzip container) into bytes.
12    */
13   public final class Decompressor {
14       
15       /*---- Public functions ----*/
16       
17       /**
18        * Reads from the specified input stream, decompress the data, and returns a new byte array.
19        * @param in the bit input stream to read from (not {@code null})
20        * @throws NullPointerException if the input stream is {@code null}
21        * @throws DataFormatException if the DEFLATE data is malformed
22        */
23       public static byte[] decompress(BitInputStream in) throws IOException, DataFormatException {
24           OutputStreamProtected out = new OutputStreamProtected();
25           decompress(in, out);
26           return out.toByteArray();
27       }
28       
29       
30       /**
31        * Reads from the specified input stream, decompress the data, and writes to the specified output stream.
32        * @param in the bit input stream to read from (not {@code null})
33        * @param out the byte output stream to write to (not {@code null})
34        * @throws NullPointerException if the input or output stream is {@code null}
35        * @throws DataFormatException if the DEFLATE data is malformed
36        */
37       public static void decompress(BitInputStream in, OutputStreamProtected out) throws IOException, DataFormatException {
38           new Decompressor(in, out);
39       }

43       /*---- Private implementation ----*/
44       
45       private BitInputStream input;
46       
47       private OutputStreamProtected output;
48       
49       private ByteHistory dictionary;
50       
51       
52       
53       // Constructor, which immediately performs decompression
54       private Decompressor(BitInputStream in, OutputStreamProtected out) throws IOException, DataFormatException {
55           // Initialize fields
56           input = Objects.requireNonNull(in);
57           output = Objects.requireNonNull(out);
58           dictionary = new ByteHistory(32 * 1024);
59           
60           // Process the stream of blocks
61           boolean isFinal;
62           do {
63               // Read the block header
64               isFinal = in.readNoEof() == 1;  // bfinal
65               int type = readInt(2);  // btype
66               
67               // Decompress rest of block based on the type
68               if (type == 0)
69                   decompressUncompressedBlock();
70               else if (type == 1)
71                   decompressHuffmanBlock(FIXED_LITERAL_LENGTH_CODE, FIXED_DISTANCE_CODE);
72               else if (type == 2) {
73                   CanonicalCode[] litLenAndDist = decodeHuffmanCodes();
74                   decompressHuffmanBlock(litLenAndDist[0], litLenAndDist[1]);
75               } else if (type == 3)
76                   throw new DataFormatException("Reserved block type");
77               else
78                   throw new IllegalStateException("Impossible value");
79           } while (!isFinal);
80       }
81       
82       
83       /*-- The constant code trees for static Huffman codes (btype = 1) --*/
84       
85       private static final CanonicalCode FIXED_LITERAL_LENGTH_CODE;
86       private static final CanonicalCode FIXED_DISTANCE_CODE;
87       
88       static {  // Make temporary tables of canonical code lengths
89           int[] llcodelens = new int[288];
90           Arrays.fill(llcodelens,   0, 144, 8);
91           Arrays.fill(llcodelens, 144, 256, 9);
92           Arrays.fill(llcodelens, 256, 280, 7);
93           Arrays.fill(llcodelens, 280, 288, 8);
94           FIXED_LITERAL_LENGTH_CODE = new CanonicalCode(llcodelens);
95           
96           int[] distcodelens = new int[32];
97           Arrays.fill(distcodelens, 5);
98           FIXED_DISTANCE_CODE = new CanonicalCode(distcodelens);
99       }
100

102       /*-- Method for reading and decoding dynamic Huffman codes (btype = 2) --*/
103       
104       // Reads from the bit input stream, decodes the Huffman code
105       // specifications into code trees, and returns the trees.
106       private CanonicalCode[] decodeHuffmanCodes() throws IOException, DataFormatException {
107           int numLitLenCodes = readInt(5) + 257;  // hlit + 257
108           int numDistCodes = readInt(5) + 1;      // hdist + 1
109           
110           // Read the code length code lengths
111           int numCodeLenCodes = readInt(4) + 4;   // hclen + 4
112           int[] codeLenCodeLen = new int[19];  // This array is filled in a strange order
113           codeLenCodeLen[16] = readInt(3);
114           codeLenCodeLen[17] = readInt(3);
115           codeLenCodeLen[18] = readInt(3);
116           codeLenCodeLen[ 0] = readInt(3);
117           for (int i = 0; i < numCodeLenCodes - 4; i++) {
118               int j = (i % 2 == 0) ? (8 + i / 2) : (7 - i / 2);
119               codeLenCodeLen[j] = readInt(3);
120           }
121           
122           // Create the code length code
123           CanonicalCode codeLenCode;
124           try {
125               codeLenCode = new CanonicalCode(codeLenCodeLen);
126           } catch (IllegalArgumentException e) {
127               throw new DataFormatException(e.getMessage());
128           }
129           
130           // Read the main code lengths and handle runs
131           int[] codeLens = new int[numLitLenCodes + numDistCodes];
132           for (int codeLensIndex = 0; codeLensIndex < codeLens.length; ) {
133               int sym = codeLenCode.decodeNextSymbol(input);
134               if (0 <= sym && sym <= 15) {
135                   codeLens[codeLensIndex] = sym;
136                   codeLensIndex++;
137               } else {
138                   int runLen;
139                   int runVal = 0;
140                   if (sym == 16) {
141                       if (codeLensIndex == 0)
142                           throw new DataFormatException("No code length value to copy");
143                       runLen = readInt(2) + 3;
144                       runVal = codeLens[codeLensIndex - 1];
145                   } else if (sym == 17)
146                       runLen = readInt(3) + 3;
147                   else if (sym == 18)
148                       runLen = readInt(7) + 11;
149                   else
150                       throw new IllegalStateException("Symbol out of range");
151                   int end = codeLensIndex + runLen;
152                   if (end > codeLens.length)
153                       throw new DataFormatException("Run exceeds number of codes");
154                   Arrays.fill(codeLens, codeLensIndex, end, runVal);
155                   codeLensIndex = end;
156               }
157           }
158           
159           // Create literal-length code tree
160           int[] litLenCodeLen = Arrays.copyOf(codeLens, numLitLenCodes);
161           CanonicalCode litLenCode;
162           try {
163               litLenCode = new CanonicalCode(litLenCodeLen);
164           } catch (IllegalArgumentException e) {
165               throw new DataFormatException(e.getMessage());
166           }
167           
168           // Create distance code tree with some extra processing
169           int[] distCodeLen = Arrays.copyOfRange(codeLens, numLitLenCodes, codeLens.length);
170           CanonicalCode distCode;
171           if (distCodeLen.length == 1 && distCodeLen[0] == 0)
172               distCode = null;  // Empty distance code; the block shall be all literal symbols
173           else {
174               // Get statistics for upcoming logic
175               int oneCount = 0;
176               int otherPositiveCount = 0;
177               for (int x : distCodeLen) {
178                   if (x == 1)
179                       oneCount++;
180                   else if (x > 1)
181                       otherPositiveCount++;
182               }
183               
184               // Handle the case where only one distance code is defined
185               if (oneCount == 1 && otherPositiveCount == 0) {
186                   // Add a dummy invalid code to make the Huffman tree complete
187                   distCodeLen = Arrays.copyOf(distCodeLen, 32);
188                   distCodeLen[31] = 1;
189               }
190               try {
191                   distCode = new CanonicalCode(distCodeLen);
192               } catch (IllegalArgumentException e) {
193                   throw new DataFormatException(e.getMessage());
194               }
195           }
196           
197           return new CanonicalCode[]{litLenCode, distCode};
198       }

202       
203       // Handles and copies an uncompressed block from the bit input stream.
204       private void decompressUncompressedBlock() throws IOException, DataFormatException {
205           // Discard bits to align to byte boundary
206           while (input.getBitPosition() != 0)
207               input.readNoEof();
208           
209           // Read length
210           int len  = readInt(16);
211           int nlen = readInt(16);
212           if ((len ^ 0xFFFF) != nlen)
213               throw new DataFormatException("Invalid length in uncompressed block");
214           
215           // Copy bytes
216           for (int i = 0; i < len; i++) {
217               int b = input.readByte();
218               if (b == -1)
219                   throw new EOFException();
220               output.write(b);
221               dictionary.append(b);
222           }
223       }
224       
225       
226       // Decompresses a Huffman-coded block from the bit input stream based on the given Huffman codes.
227       private void decompressHuffmanBlock(CanonicalCode litLenCode, CanonicalCode distCode)
228               throws IOException, DataFormatException {
229           Objects.requireNonNull(litLenCode);
230           // distCode is allowed to be null
231           
232           while (true) {
233               int sym = litLenCode.decodeNextSymbol(input);
234               if (sym == 256)  // End of block
235                   break;
236               
237               if (sym < 256) {  // Literal byte
238                   output.write(sym);
239                   dictionary.append(sym);
240               } else {  // Length and distance for copying
241                   int run = decodeRunLength(sym);
242                   if (run < 3 || run > 258)
243                       throw new IllegalStateException("Invalid run length");
244                   if (distCode == null)
245                       throw new DataFormatException("Length symbol encountered with empty distance code");
246                   int distSym = distCode.decodeNextSymbol(input);
247                   int dist = decodeDistance(distSym);
248                   if (dist < 1 || dist > 32768)
249                       throw new IllegalStateException("Invalid distance");
250                   dictionary.copy(dist, run, output);
251               }
252           }
253       }
254       
255       
256       /*-- Symbol decoding methods --*/
257       
258       // Returns the run length based on the given symbol and possibly reading more bits.
259       private int decodeRunLength(int sym) throws IOException, DataFormatException {
260           if (sym < 257 || sym > 287)  // Cannot occur in the bit stream; indicates the decompressor is buggy
261               throw new IllegalStateException("Invalid run length symbol: " + sym);
262           else if (sym <= 264)
263               return sym - 254;
264           else if (sym <= 284) {
265               int numExtraBits = (sym - 261) / 4;
266               return (((sym - 265) % 4 + 4) << numExtraBits) + 3 + readInt(numExtraBits);
267           } else if (sym == 285)
268               return 258;
269           else  // sym is 286 or 287
270               throw new DataFormatException("Reserved length symbol: " + sym);
271       }
272       
273       
274       // Returns the distance based on the given symbol and possibly reading more bits.
275       private int decodeDistance(int sym) throws IOException, DataFormatException {
276           if (sym < 0 || sym > 31)  // Cannot occur in the bit stream; indicates the decompressor is buggy
277               throw new IllegalStateException("Invalid distance symbol: " + sym);
278           if (sym <= 3)
279               return sym + 1;
280           else if (sym <= 29) {
281               int numExtraBits = sym / 2 - 1;
282               return ((sym % 2 + 2) << numExtraBits) + 1 + readInt(numExtraBits);
283           } else  // sym is 30 or 31
284               throw new DataFormatException("Reserved distance symbol: " + sym);
285       }

289       
290       // Reads the given number of bits from the bit input stream as a single integer, packed in little endian.
291       private int readInt(int numBits) throws IOException {
292           if (numBits < 0 || numBits > 31)
293               throw new IllegalArgumentException();
294           int result = 0;
295           for (int i = 0; i < numBits; i++)
296               result |= input.readNoEof() << i;
297           return result;
298       }
