11   import java.io.EOFException;
12   import java.io.IOException;
13   import java.io.InputStream;
14   import java.util.Objects;
15   
16   
17   /**
18    * A stream of bits that can be read. Because they come from an underlying byte stream,
19    * the total number of bits is always a multiple of 8. The bits are read in little endian.
20    * Mutable and not thread-safe.
21    */
22   public final class ByteBitInputStream implements BitInputStream {
23       
24       /*---- Fields ----*/
25       
26       // The underlying byte stream to read from (not null).
27       private InputStream input;
28       
29       // Either in the range [0x00, 0xFF] if bits are available, or -1 if end of stream is reached.
30       private int currentByte;
31       
32       // Number of remaining bits in the current byte, always between 0 and 7 (inclusive).
33       private int numBitsRemaining;
34       
35       
36       
37       /*---- Constructor ----*/
38       
39       /**
40        * Constructs a bit input stream based on the specified byte input stream.
41        * @param in the byte input stream (not {@code null})
42        * @throws NullPointerException if the input stream is {@code null}
43        */
44       public ByteBitInputStream(InputStream in) {
45           input = Objects.requireNonNull(in);
46           currentByte = 0;
47           numBitsRemaining = 0;
48       }
49       
50       
51       
52       /*---- Methods ----*/
53       
54       public int getBitPosition() {
55           if (numBitsRemaining < 0 || numBitsRemaining > 7)
56               throw new IllegalStateException();
57           return (8 - numBitsRemaining) % 8;
58       }
59       
60       
61       public int readByte() throws IOException {
62           currentByte = 0;
63           numBitsRemaining = 0;
64           return input.read();
65       }
66       
67       
68       public int read() throws IOException {
69           if (currentByte == -1)
70               return -1;
71           if (numBitsRemaining == 0) {
72               currentByte = input.read();
73               if (currentByte == -1)
74                   return -1;
75               numBitsRemaining = 8;
76           }
77           if (numBitsRemaining <= 0)
78               throw new IllegalStateException();
79           numBitsRemaining--;
80           return (currentByte >>> (7 - numBitsRemaining)) & 1;
81       }
82       
83       
84       public int readNoEof() throws IOException {
85           int result = read();
86           if (result == -1)
87               throw new EOFException();
88           return result;
89       }
90       
91       
92       public void close() throws IOException {
93           input.close();
94           currentByte = -1;
95           numBitsRemaining = 0;
96       }
97       
98   }
