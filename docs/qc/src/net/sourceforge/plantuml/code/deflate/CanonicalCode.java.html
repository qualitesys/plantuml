11   import java.io.IOException;
12   import java.util.Arrays;
13   import java.util.Objects;
14   
15   
16   /**
17    * A canonical Huffman code, where the code values for each symbol is derived
18    * from a given sequence of code lengths. This data structure is immutable.
19    * This could be transformed into an explicit Huffman code tree.
20    * <p>Example:</p>
21    * <pre>  Code lengths (canonical code):
22    *    Symbol A: 1
23    *    Symbol B: 0 (no code)
24    *    Symbol C: 3
25    *    Symbol D: 2
26    *    Symbol E: 3
27    *  
28    *  Generated Huffman codes:
29    *    Symbol A: 0
30    *    Symbol B: (Absent)
31    *    Symbol C: 110
32    *    Symbol D: 10
33    *    Symbol E: 111
34    *  
35    *  Huffman code tree:
36    *      .
37    *     / \
38    *    A   .
39    *       / \
40    *      D   .
41    *         / \
42    *        C   E</pre>
43    */
44   final class CanonicalCode {
45       
46       /* 
47        * These arrays store the Huffman codes and values necessary for decoding.
48        * symbolCodeBits contains Huffman codes, each padded with a 1 bit at the
49        * beginning to disambiguate codes of different lengths (e.g. otherwise we
50        * can't distinguish 0b01 from 0b0001). Each symbolCodeBits[i] decodes to its
51        * corresponding symbolValues[i]. Values in symbolCodeBits are strictly increasing.
52        * 
53        * For the example of codeLengths=[1,0,3,2,3], we would have:
54        *   i | symbolCodeBits[i] | symbolValues[i]
55        *   --+-------------------+----------------
56        *   0 |             0b1_0 |               0
57        *   1 |            0b1_10 |               3
58        *   2 |           0b1_110 |               2
59        *   3 |           0b1_111 |               4
60        */
61       private int[] symbolCodeBits;
62       private int[] symbolValues;
63       
64       
65       
66       /**
67        * Constructs a canonical Huffman code from the specified array of symbol code lengths.
68        * Each code length must be non-negative. Code length 0 means no code for the symbol.
69        * The collection of code lengths must represent a proper full Huffman code tree.
70        * <p>Examples of code lengths that result in correct full Huffman code trees:</p>
71        * <ul>
72        *   <li>[1, 1] (result: A=0, B=1)</li>
73        *   <li>[2, 2, 1, 0, 0, 0] (result: A=10, B=11, C=0)</li>
74        *   <li>[3, 3, 3, 3, 3, 3, 3, 3] (result: A=000, B=001, C=010, ..., H=111)</li>
75        * </ul>
76        * <p>Examples of code lengths that result in under-full Huffman code trees:</p>
77        * <ul>
78        *   <li>[0, 2, 0] (result: B=00, unused=01, unused=1)</li>
79        *   <li>[0, 1, 0, 2] (result: B=0, D=10, unused=11)</li>
80        * </ul>
81        * <p>Examples of code lengths that result in over-full Huffman code trees:</p>
82        * <ul>
83        *   <li>[1, 1, 1] (result: A=0, B=1, C=overflow)</li>
84        *   <li>[1, 1, 2, 2, 3, 3, 3, 3] (result: A=0, B=1, C=overflow, ...)</li>
85        * </ul>
86        * @param codeLengths array of symbol code lengths (not {@code null})
87        * @throws NullPointerException if the array is {@code null}
88        * @throws IllegalArgumentException if any element is negative, any value exceeds MAX_CODE_LENGTH,
89        * or the collection of code lengths would yield an under-full or over-full Huffman code tree
90        */
91       public CanonicalCode(int[] codeLengths) {
92           // Check argument values
93           Objects.requireNonNull(codeLengths);
94           for (int x : codeLengths) {
95               if (x < 0)
96                   throw new IllegalArgumentException("Negative code length");
97               if (x > MAX_CODE_LENGTH)
98                   throw new IllegalArgumentException("Maximum code length exceeded");
99           }
100           
101           // Allocate code values to symbols. Symbols are processed in the order
102           // of shortest code length first, breaking ties by lowest symbol value.
103           symbolCodeBits = new int[codeLengths.length];
104           symbolValues   = new int[codeLengths.length];
105           int numSymbolsAllocated = 0;
106           int nextCode = 0;
107           for (int codeLength = 1; codeLength <= MAX_CODE_LENGTH; codeLength++) {
108               nextCode <<= 1;
109               int startBit = 1 << codeLength;
110               for (int symbol = 0; symbol < codeLengths.length; symbol++) {
111                   if (codeLengths[symbol] != codeLength)
112                       continue;
113                   if (nextCode >= startBit)
114                       throw new IllegalArgumentException("This canonical code produces an over-full Huffman code tree");
115                   
116                   symbolCodeBits[numSymbolsAllocated] = startBit | nextCode;
117                   symbolValues  [numSymbolsAllocated] = symbol;
118                   numSymbolsAllocated++;
119                   nextCode++;
120               }
121           }
122           if (nextCode != 1 << MAX_CODE_LENGTH)
123               throw new IllegalArgumentException("This canonical code produces an under-full Huffman code tree");
124           
125           // Trim unused trailing elements
126           symbolCodeBits = Arrays.copyOf(symbolCodeBits, numSymbolsAllocated);
127           symbolValues   = Arrays.copyOf(symbolValues  , numSymbolsAllocated);
128       }
129       
130       
131       
132       /**
133        * Decodes the next symbol from the specified bit input stream based on this
134        * canonical code. The returned symbol value is in the range [0, codeLengths.length).
135        * @param in the bit input stream to read from
136        * @return the next decoded symbol
137        * @throws IOException if an I/O exception occurred
138        */
139       public int decodeNextSymbol(BitInputStream in) throws IOException {
140           Objects.requireNonNull(in);
141           int codeBits = 1;  // The start bit
142           while (true) {
143               // Accumulate one bit at a time on the right side until a match is
144               // found in the symbolCodeBits array. Because the Huffman code tree is
145               // full, this loop must terminate after at most MAX_CODE_LENGTH iterations.
146               codeBits = codeBits << 1 | in.readNoEof();
147               int index = Arrays.binarySearch(symbolCodeBits, codeBits);
148               if (index >= 0)
149                   return symbolValues[index];
150           }
151       }
152       
153       
154       /**
155        * Returns a string representation of this canonical code,
156        * useful for debugging only, and the format is subject to change.
157        * @return a string representation of this canonical code
158        */
159       public String toString() {
160           StringBuilder sb = new StringBuilder();
161           for (int i = 0; i < symbolCodeBits.length; i++) {
162               sb.append(String.format("Code %s: Symbol %d%n",
163                   Integer.toBinaryString(symbolCodeBits[i]).substring(1),
164                   symbolValues[i]));
165           }
166           return sb.toString();
167       }
168       
169       
170       
171       // The maximum Huffman code length allowed in the DEFLATE standard.
172       private static final int MAX_CODE_LENGTH = 15;
173       
174   }
