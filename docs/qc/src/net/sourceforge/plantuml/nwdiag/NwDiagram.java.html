116       public CommandExecutionResult openGroup(String name) {
117           if (initDone == false)
118               return errorNoInit();
119   
120           for (NStackable element : stack)
121               if (element instanceof NwGroup)
122                   return CommandExecutionResult.error("Cannot nest group");
123   
124           final NwGroup newGroup = new NwGroup(name);
125           stack.add(0, newGroup);
126           groups.add(newGroup);
127           currentGroup = newGroup;
128           return CommandExecutionResult.ok();
129       }

159       public CommandExecutionResult link(String name1, String name2) {
160           if (initDone == false)
161               return errorNoInit();
162   
163           final NServer server2;
164           if (lastNetwork() == null) {
165               createNetwork(name1);
166               server2 = NServer.create(name2);
167           } else {
168               final NServer server1 = servers.get(name1);
169               final Network network1 = createNetwork("");
170               network1.goInvisible();
171               if (server1 != null)
172                   server1.connectTo(lastNetwork());
173   
174               server2 = new NServer(name2, server1.getBar());
175           }
176           servers.put(name2, server2);
177           server2.connectTo(lastNetwork());
178           playField.addInPlayfield(server2.getBar());
179           return CommandExecutionResult.ok();
180       }

182       public CommandExecutionResult addElement(String name, String definition) {
183           if (initDone == false)
184               return errorNoInit();
185   
186           if (currentGroup != null) {
187               if (alreadyInSomeGroup(name))
188                   return CommandExecutionResult.error("Element already in another group.");
189   
190               currentGroup.addName(name);
191           }
192   
193           NServer server = null;
194           if (lastNetwork() == null) {
195               if (currentGroup != null)
196                   return CommandExecutionResult.ok();
197   
198               assert currentGroup == null;
199               final Network network1 = createNetwork("");
200               network1.goInvisible();
201               server = NServer.create(name);
202               servers.put(name, server);
203               server.doNotPrintFirstLink();
204           } else {
205               server = servers.get(name);
206               if (server == null) {
207                   server = NServer.create(name);
208                   servers.put(name, server);
209               }
210           }
211           final Map<String, String> props = toSet(definition);
212           server.connectTo(lastNetwork(), props.get("address"));
213           server.updateProperties(props);
214           playField.addInPlayfield(server.getBar());
215           return CommandExecutionResult.ok();
216       }

218       private boolean alreadyInSomeGroup(String name) {
219           for (NwGroup g : groups) 
220               if (g.names().contains(name))
221                   return true;
222   
223           return false;
224       }

230       private Map<String, String> toSet(String definition) {
231           final Map<String, String> result = new HashMap<String, String>();
232           if (definition == null)
233               return result;
234   
235           final Pattern p = Pattern.compile("\\s*(\\w+)\\s*=\\s*(\"([^\"]*)\"|[^\\s,]+)");
236           final Matcher m = p.matcher(definition);
237           while (m.find()) {
238               final String name = m.group(1);
239               final String value = m.group(3) == null ? m.group(2) : m.group(3);
240               result.put(name, value);
241           }
242           return result;
243   
244       }

303       private void drawMe(UGraphic ug) {
304           ug = ug.apply(new UTranslate(margin, margin));
305   
306           final StringBounder stringBounder = ug.getStringBounder();
307   
308           double deltaX = 0;
309           double deltaY = 0;
310   
311           final GridTextBlockDecorated grid = buildGrid(stringBounder);
312   
313           for (int i = 0; i < networks.size(); i++) {
314               final Network current = networks.get(i);
315               final String address = current.getOwnAdress();
316               final TextBlock desc = toTextBlockForNetworkName(current.getDisplayName(), address);
317               final XDimension2D dim = desc.calculateDimension(stringBounder);
318               if (i == 0)
319                   deltaY = (dim.getHeight() - GridTextBlockDecorated.NETWORK_THIN) / 2;
320   
321               deltaX = Math.max(deltaX, dim.getWidth());
322           }
323           double y = 0;
324           for (int i = 0; i < networks.size(); i++) {
325               final Network current = networks.get(i);
326               final String address = current.getOwnAdress();
327               final TextBlock desc = toTextBlockForNetworkName(current.getDisplayName(), address);
328               final XDimension2D dim = desc.calculateDimension(stringBounder);
329               desc.drawU(ug.apply(new UTranslate(deltaX - dim.getWidth(), y)));
330   
331               y += grid.lineHeight(stringBounder, i);
332           }
333           deltaX += 5;
334   
335           grid.drawU(ug.apply(new UTranslate(deltaX, deltaY)));
336           final XDimension2D dimGrid = grid.calculateDimension(stringBounder);
337   
338           ug.apply(new UTranslate(dimGrid.getWidth() + deltaX + margin, dimGrid.getHeight() + deltaY + margin))
339                   .draw(new UEmpty(1, 1));
340   
341       }

343       private Map<Network, String> getLinks(NServer element) {
344           final Map<Network, String> result = new LinkedHashMap<>();
345           for (Network network : networks) {
346               final String s = element.getAdress(network);
347               if (s != null)
348                   result.put(network, s);
349   
350           }
351           return result;
352       }

354       private GridTextBlockDecorated buildGrid(StringBounder stringBounder) {
355   
356           playField.fixGroups(groups, servers.values());
357   
358           final GridTextBlockDecorated grid = new GridTextBlockDecorated(networks.size(), servers.size(), groups,
359                   networks, getSkinParam());
360   
361           final Map<NBar, Integer> layout = playField.doLayout();
362           for (int i = 0; i < networks.size(); i++) {
363               final Network current = networks.get(i);
364               for (Map.Entry<String, NServer> ent : servers.entrySet()) {
365                   final NServer server = ent.getValue();
366                   if (server.getMainNetworkNext() == current) {
367                       final Map<Network, String> conns = getLinks(server);
368                       final Integer col = layout.get(server.getBar());
369                       if (col == null)
370                           continue;
371                       double topMargin = LinkedElement.MAGIC;
372                       NwGroup group = getGroupOf(server);
373                       if (group != null)
374                           topMargin += group.getTopHeaderHeight(stringBounder, getSkinParam());
375                       grid.add(i, col, server.getLinkedElement(topMargin, conns, networks, getSkinParam()));
376                   }
377               }
378           }
379   
380           return grid;
381       }

383       private NwGroup getGroupOf(NServer server) {
384           for (NwGroup group : groups)
385               if (group.contains(server))
386                   return group;
387   
388           return null;
389       }

391       public CommandExecutionResult setProperty(String property, String value) {
392           if (initDone == false)
393               return errorNoInit();
394   
395           if ("address".equalsIgnoreCase(property) && lastNetwork() != null)
396               lastNetwork().setOwnAdress(value);
397   
398           if ("width".equalsIgnoreCase(property) && lastNetwork() != null)
399               lastNetwork().setFullWidth("full".equalsIgnoreCase(value));
400   
401           if ("color".equalsIgnoreCase(property)) {
402               final HColor color = value == null ? null
403                       : getSkinParam().getIHtmlColorSet().getColorOrWhite(value);
404               if (currentGroup != null)
405                   currentGroup.setColor(color);
406               else if (lastNetwork() != null)
407                   lastNetwork().setColor(color);
408   
409           }
410           if ("description".equalsIgnoreCase(property))
411               if (currentGroup == null)
412                   lastNetwork().setDescription(value);
413               else
414                   currentGroup.setDescription(value);
415   
416           return CommandExecutionResult.ok();
417       }
