289       private void updateSimpleLine() {

310       private ScrollablePicture buildScrollablePicture() {
311           final GeneratedImage generatedImage = simpleLine.getGeneratedImage();
312           if (generatedImage == null) {
313               return null;
314           }
315           final File png = generatedImage.getPngFile();
316           BufferedImage image = null;
317           try {
318               image = SImageIO.read(new SFile(png.getAbsolutePath()));
319               if (sizeMode == SizeMode.ZOOM_FIT) {
320                   final Dimension imageDim = new Dimension(image.getWidth(), image.getHeight());
321                   final Dimension newImgDim = ImageHelper.getScaledDimension(imageDim,
322                           scrollPane.getViewport().getSize());
323                   image = ImageHelper.getScaledInstance(image, newImgDim, getHints(), true);
324               } else if (sizeMode == SizeMode.WIDTH_FIT) {
325                   final Dimension imageDim = new Dimension(image.getWidth(), image.getHeight());
326                   final Dimension newImgDim = ImageHelper.getScaledDimensionWidthFit(imageDim,
327                           scrollPane.getViewport().getSize());
328                   image = ImageHelper.getScaledInstance(image, newImgDim, getHints(), false);
329               } else if (zoomFactor != 0) {
330                   final Dimension imageDim = new Dimension(image.getWidth(), image.getHeight());
331                   final Dimension newImgDim = ImageHelper.getScaledDimension(imageDim, getZoom());
332                   image = ImageHelper.getScaledInstance(image, newImgDim, getHints(), false);
333               }
334           } catch (IOException ex) {
335               final String msg = "Error reading file: " + ex.toString();
336               final TextBlockBackcolored error = GraphicStrings.createForError(Arrays.asList(msg), false);
337               try {
338                   final byte[] bytes = plainPngBuilder(error).writeByteArray();
339                   image = SImageIO.read(bytes);
340               } catch (IOException e) {
341                   Logme.error(e);
342               }
343           }
344           final ImageIcon imageIcon = new ImageIcon(image, simpleLine.toString());
345           final ScrollablePicture scrollablePicture = new ScrollablePicture(imageIcon, 1);
346   
347           scrollablePicture.addMouseListener(new MouseAdapter() {
348               public void mousePressed(MouseEvent me) {
349                   super.mousePressed(me);
350                   startX = me.getX();
351                   startY = me.getY();
352               }
353           });
354           scrollablePicture.addMouseMotionListener(new MouseMotionAdapter() {
355               public void mouseDragged(MouseEvent me) {
356                   super.mouseDragged(me);
357                   final int diffX = me.getX() - startX;
358                   final int diffY = me.getY() - startY;
359   
360                   final JScrollBar hbar = scrollPane.getHorizontalScrollBar();
361                   hbar.setValue(hbar.getValue() - diffX);
362                   final JScrollBar vbar = scrollPane.getVerticalScrollBar();
363                   vbar.setValue(vbar.getValue() - diffY);
364               }
365           });
366   
367           return scrollablePicture;
368       }

370       private RenderingHints getHints() {
371           final RenderingHints hints = new RenderingHints(RenderingHints.KEY_INTERPOLATION,
372                   RenderingHints.VALUE_INTERPOLATION_BILINEAR);
373           hints.put(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);

375           return hints;
376       }

378       private double getZoom() {
379           // if (zoomFactor <= -10) {
380           // return 0.05;

408       public void refreshImage(boolean external) {
409           final JScrollBar bar1 = scrollPane.getVerticalScrollBar();

411           if (external && isError() == false) {

413               v2 = bar2.getValue();

417           if (external) {

419                   public void run() {
420                       refreshSimpleLine();
421                       if (isError() == false) {
422                           bar1.setValue(v1);
423                           bar2.setValue(v2);
424                       }
425                   }
426               });
427           }
428       }
