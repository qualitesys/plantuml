38   import java.awt.Dimension;
39   import java.awt.Graphics2D;
40   import java.awt.Image;
41   import java.awt.RenderingHints;
42   import java.awt.Transparency;
43   import java.awt.image.BufferedImage;
44   
45   /**
46    * A collection of methods to help with processing images. A majority of this code was originally found online.
47    */
48   public class ImageHelper {
49       /**
50        * Returns a scaled instance of a {@code BufferedImage}.
51        * 
52        * Modified from: https://today.java.net/pub/a/today/2007/04/03/perils-of-image-getscaledinstance.html
53        * 
54        * @param img
55        *            the original image to be scaled
56        * @param targetDim
57        *            the desired dimensions of the scaled instance, in pixels
58        * @param hints
59        *            RenderingHints used when scaling the image
60        * @param higherQuality
61        *            if true, this method will use a multi-step scaling technique that provides higher quality than the
62        *            usual one-step technique (only useful in downscaling cases, targetDim is smaller than the original
63        *            dimensions, and generally only when the {@code BILINEAR} hint is specified)
64        * @return a scaled version of the original {@code BufferedImage}
65        */
66       public static BufferedImage getScaledInstance(BufferedImage img, Dimension targetDim, RenderingHints hints,
67               boolean higherQuality) {
68           final int targetWidth = targetDim.width;
69           final int targetHeight = targetDim.height;
70           final int type = (img.getTransparency() == Transparency.OPAQUE) ? BufferedImage.TYPE_INT_RGB
71                   : BufferedImage.TYPE_INT_ARGB;
72           BufferedImage ret = (BufferedImage) img;
73           int w;
74           int h;
75           if (higherQuality) {
76               // Use multi-step technique: start with original size, then
77               // scale down in multiple passes with drawImage()
78               // until the target size is reached
79               w = img.getWidth();
80               h = img.getHeight();
81           } else {
82               // Use one-step technique: scale directly from original
83               // size to target size with a single drawImage() call
84               w = targetWidth;
85               h = targetHeight;
86           }
87   
88           do {
89               if (higherQuality && w > targetWidth) {
90                   w /= 2;
91                   if (w < targetWidth) {
92                       w = targetWidth;
93                   }
94               }
95   
96               if (higherQuality && h > targetHeight) {
97                   h /= 2;
98                   if (h < targetHeight) {
99                       h = targetHeight;
100                   }
101               }
102   
103               final BufferedImage tmp = new BufferedImage(w, h, type);
104               final Graphics2D g2 = tmp.createGraphics();
105               g2.setRenderingHints(hints);
106               g2.drawImage(ret, 0, 0, w, h, null);
107               g2.dispose();
108   
109               ret = tmp;
110           } while (w != targetWidth || h != targetHeight);
111   
112           return ret;
113       }
114   
115       /**
116        * Converts an Image to a BufferedImage.
117        * 
118        * From: http://stackoverflow.com/questions/13605248/java-converting-image-to-bufferedimage
119        */
120       public static BufferedImage toBufferedImage(Image img) {
121           if (img instanceof BufferedImage) {
122               return (BufferedImage) img;
123           }
124           // Create a buffered image with transparency
125           final BufferedImage bimage = new BufferedImage(img.getWidth(null), img.getHeight(null),
126                   BufferedImage.TYPE_INT_ARGB);
127   
128           // Draw the image on to the buffered image
129           final Graphics2D bGr = bimage.createGraphics();
130           bGr.drawImage(img, 0, 0, null);
131           bGr.dispose();
132   
133           return bimage;
134       }

136       /**
137        * Calculates the dimensions of a scaled image given the dimensions of an image and the area it is to be drawn in
138        * while preserving aspect ratio.
139        * 
140        * From: http://stackoverflow.com/questions/10245220/java-image-resize-maintain-aspect-ratio
141        * 
142        * @param imgSize
143        *            dimensions of the original image.
144        * @param boundary
145        *            dimensions of the area the image is to be drawn in.
146        */
147       public static Dimension getScaledDimension(Dimension imgSize, Dimension boundary) {
148           final int originalWidth = imgSize.width;
149           final int originaHeight = imgSize.height;
150           final int boundWidth = boundary.width;
151           final int boundHeight = boundary.height;
152           int newWidth = originalWidth;
153           int newHeight = originaHeight;
154   
155           // first check if we need to scale width
156           if (originalWidth > boundWidth) {
157               // scale width to fit
158               newWidth = boundWidth;
159               // scale height to maintain aspect ratio
160               newHeight = (newWidth * originaHeight) / originalWidth;
161           }
162   
163           // then check if we need to scale even with the new height
164           if (newHeight > boundHeight) {
165               // scale height to fit instead
166               newHeight = boundHeight;
167               // scale width to maintain aspect ratio
168               newWidth = (newHeight * originalWidth) / originaHeight;
169           }
170   
171           return new Dimension(newWidth, newHeight);
172       }
173   
174       public static Dimension getScaledDimensionWidthFit(Dimension imgSize, Dimension boundary) {
175           final int originalWidth = imgSize.width;
176           final int originaHeight = imgSize.height;
177           final int boundWidth = boundary.width;
178           final int boundHeight = boundary.height;
179           int newWidth = originalWidth;
180           int newHeight = originaHeight;
181   
182           // first check if we need to scale width
183           if (originalWidth != boundWidth) {
184               // scale width to fit
185               newWidth = boundWidth;
186               // scale height to maintain aspect ratio
187               newHeight = (newWidth * originaHeight) / originalWidth;
188           }
189   
190           return new Dimension(newWidth, newHeight);
191       }
192   
193       public static Dimension getScaledDimension(Dimension dim, double zoom) {
194           return new Dimension((int) (dim.getWidth() * zoom), (int) (dim.getHeight() * zoom));
195       }
