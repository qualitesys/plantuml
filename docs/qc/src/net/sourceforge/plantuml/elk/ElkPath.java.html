115       private Style getStyle() {
116           final StyleSignatureBasic signature = StyleSignatureBasic.of(SName.root, SName.element, styleName, SName.arrow);
117           return signature.getMergedStyle(diagram.getCurrentStyleBuilder());
118       }

120       public void drawU(UGraphic ug) {
121   
122           if (link.isHidden())
123               return;
124   
125           final ISkinParam skinParam = diagram.getSkinParam();
126   
127           HColor color = getStyle().value(PName.LineColor).asColor(skinParam.getIHtmlColorSet());
128   
129           if (this.link.getColors() != null) {
130               final HColor newColor = this.link.getColors().getColor(ColorType.ARROW, ColorType.LINE);
131               if (newColor != null)
132                   color = newColor;
133           } else if (this.link.getSpecificColor() != null) {
134               color = this.link.getSpecificColor();
135           }
136   
137           final LinkType linkType = link.getType();
138           UStroke stroke = linkType.getStroke3(skinParam.getThickness(LineParam.arrow, null));
139           if (link.getColors() != null && link.getColors().getSpecificLineStroke() != null)
140               stroke = link.getColors().getSpecificLineStroke();
141   
142           ug = ug.apply(stroke).apply(color);
143   
144           final List<ElkEdgeSection> sections = edge.getSections();
145           if (sections.size() == 0) {
146               System.err.println("Strange: no section?");
147               System.err.println("Maybe a 'Long hierarchical edge' " + edge.isHierarchical());
148               return;
149           } else {
150               drawSections(ug, sections);
151           }
152   
153           final UDrawable extremityFactory1 = getDecors(link.getType().getDecor1(), Math.PI / 2, HColors.WHITE);
154           final UDrawable extremityFactory2 = getDecors(link.getType().getDecor2(), -Math.PI / 2, HColors.WHITE);
155   
156           if (extremityFactory1 != null) {
157               final double x = sections.get(0).getEndX();
158               final double y = sections.get(0).getEndY();
159               extremityFactory1.drawU(ug.apply(new UTranslate(x, y)));
160           }
161   
162           if (extremityFactory2 != null) {
163               final double x = sections.get(0).getStartX();
164               final double y = sections.get(0).getStartY();
165               extremityFactory2.drawU(ug.apply(new UTranslate(x, y)));
166           }
167   
168           drawLabels(ug);
169   
170       }
171   
172       private UDrawable getDecors(LinkDecor decors, double angle, HColor backColor) {
173           // For legacy reason, extends are treated differently
174           if (decors == LinkDecor.EXTENDS)
175               return new ExtremityFactoryExtends(backColor).createUDrawable(new XPoint2D(), angle, null);
176   
177           final ExtremityFactory extremityFactory = decors.getExtremityFactory(backColor);
178           if (extremityFactory == null)
179               return null;
180   
181           return extremityFactory.createUDrawable(new XPoint2D(), angle, null);
182       }
183   
184       private void drawLabels(UGraphic ug) {
185           for (ElkLabel label : edge.getLabels()) {
186               final double x = label.getX();
187               final double y = label.getY();
188               final TextBlock labelLink;
189               // Nasty trick: we store the type of label (center/head/tail) in the text
190               final String type = label.getText();
191               if ("X".equals(type))
192                   labelLink = centerLabel;
193               else if ("1".equals(type))
194                   labelLink = tailLabel;
195               else if ("2".equals(type))
196                   labelLink = headLabel;
197               else
198                   continue;
199   
200               labelLink.drawU(ug.apply(new UTranslate(x, y)));
201           }
202       }
203   
204       private void drawSections(UGraphic ug, final Collection<ElkEdgeSection> sections) {
205           for (ElkEdgeSection section : sections) {
206               final Collection<ElkBendPoint> points = section.getBendPoints();
207   
208               double x1 = section.getStartX();
209               double y1 = section.getStartY();
210   
211               for (ElkBendPoint pt : points) {
212                   drawLine(ug, x1, y1, pt.getX(), pt.getY());
213                   x1 = pt.getX();
214                   y1 = pt.getY();
215               }
216   
217               drawLine(ug, x1, y1, section.getEndX(), section.getEndY() + magicY2);
218           }
219       }
220   
221       private void drawLine(UGraphic ug, double x1, double y1, double x2, double y2) {
222           final ULine line = new ULine(x2 - x1, y2 - y1);
223           ug.apply(new UTranslate(x1, y1)).draw(line);
224       }
