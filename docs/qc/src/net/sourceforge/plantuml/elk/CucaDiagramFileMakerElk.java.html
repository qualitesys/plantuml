194       // Retrieve the real position of a node, depending on its parents
195       private XPoint2D getPosition(ElkNode elkNode) {
196           final ElkNode parent = elkNode.getParent();
197   
198           final double x = elkNode.getX();
199           final double y = elkNode.getY();
200   
201           // This nasty test checks that parent is "root"
202           if (parent == null || parent.getLabels().size() == 0) {
203               return new XPoint2D(x, y);
204           }
205   
206           // Right now, this is recursive
207           final XPoint2D parentPosition = getPosition(parent);
208           return new XPoint2D(parentPosition.getX() + x, parentPosition.getY() + y);
209   
210       }
