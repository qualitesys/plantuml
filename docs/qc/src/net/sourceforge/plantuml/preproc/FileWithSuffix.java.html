64       @Override
65       public String toString() {
66           if (file == null) {
67               return super.toString();
68           }
69           return file.toString();
70       }

72       public Reader getReader(Charset charset) throws IOException {
73           if (file == null) {
74               return null;
75           }
76           final InputStream tmp = file.openFile();
77           if (tmp == null) {
78               return null;
79           }
80           if (entry == null) {
81               return new InputStreamReader(tmp, charset);
82           }
83           final InputStream is = getDataFromZip(tmp, entry);
84           if (is == null) {
85               return null;
86           }
87           return new InputStreamReader(is, charset);
88       }
89   
90       private InputStream getDataFromZip(InputStream is, String name) throws IOException {
91           final ZipInputStream zis = new ZipInputStream(is);
92           ZipEntry ze = zis.getNextEntry();
93   
94           while (ze != null) {
95               final String fileName = ze.getName();
96               if (ze.isDirectory()) {
97               } else if (fileName.equals(name)) {
98                   return zis;
99               }
100               ze = zis.getNextEntry();
101           }
102           zis.closeEntry();
103           zis.close();
104           return null;
105       }

107       public boolean fileOk() {
108           return file != null && file.isOk();
109       }

111       FileWithSuffix(SFile file, String suffix) {
112           this.file = new AFileRegular(file);
113           this.suffix = suffix;
114           this.entry = null;
115           this.description = file.getName();
116       }
117   
118       FileWithSuffix(String description, String suffix, AFile file, String entry) {
119           this.description = description;
120           this.suffix = suffix;
121           this.file = file;
122           this.entry = entry;
123       }

125       static FileWithSuffix none() {
126           return new FileWithSuffix("NONE", null, null, null);
127       }

129       @Override
130       public int hashCode() {
131           int v = 0;
132           if (file != null) {
133               v += file.hashCode();
134           }
135           if (suffix != null) {
136               v += suffix.hashCode() * 43;
137           }
138           if (entry != null) {
139               v += entry.hashCode();
140           }
141           return v;
142       }

144       @Override
145       public boolean equals(Object arg) {
146           final FileWithSuffix other = (FileWithSuffix) arg;
147           return this.file.equals(other.file) && equals(suffix, other.suffix) && same(entry, other.entry);
148       }

150       private static boolean same(String s1, String s2) {
151           if (s1 == null && s2 == null) {
152               return true;
153           }
154           if (s1 != null && s2 != null) {
155               return s1.equals(s2);
156           }
157           return false;
158       }

160       public AParentFolder getParentFile() {
161           if (file == null) {
162               return null;
163           }
164           Log.info("Getting parent of " + file);
165           Log.info("-->The parent is " + file.getParentFile());
166           return file.getParentFile();
167       }

177       private static boolean equals(String s1, String s2) {
178           if (s1 == null && s2 == null) {
179               return true;
180           }
181           if (s1 != null && s2 != null) {
182               return s1.equals(s2);
183           }
184           assert (s1 == null && s2 != null) || (s1 != null && s2 == null);
185           return false;
186       }

188       public static Set<File> convert(Set<FileWithSuffix> all) throws FileNotFoundException {
189           final Set<File> result = new HashSet<>();
190           for (FileWithSuffix f : all) {
191               result.add(f.file.getUnderlyingFile().conv());
192           }
193           return result;
194       }
