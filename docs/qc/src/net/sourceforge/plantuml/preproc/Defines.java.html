185       public List<String> applyDefines(String line) {
186           // System.err.println("line=" + line + " " + values.size());
187           line = manageDate(line);
188           line = manageEnvironment(line);
189           line = method1(line);
190           // line = values.size() < 10 ? method1(line) : method2(line);
191           return Arrays.asList(line.split("\n"));
192       }

194       private String method1(String line) {
195           for (Define def : values.values()) {
196               line = def.apply(line);
197           }
198           return line;
199       }

201       private Map<String, Collection<Define>> getAll() {
202           final Map<String, Collection<Define>> result = new LinkedHashMap<String, Collection<Define>>();
203           for (Define def : values.values()) {
204               Collection<Define> tmp = result.get(def.getFunctionName());
205               if (tmp == null) {
206                   tmp = new ArrayList<>();
207                   result.put(def.getFunctionName(), tmp);
208               }
209               tmp.add(def);
210           }
211           return result;
212       }

216       private String method2(String line) {
217           final Set<String> words = words(line);
218           if (magic == null) {
219               magic = getAll();
220   
221           }
222           for (String w : words) {
223               Collection<Define> tmp = magic.get(w);
224               if (tmp == null) {
225                   continue;
226               }
227               for (Define def : tmp) {
228                   line = def.apply(line);
229               }
230           }
231           return line;
232       }

234       private Set<String> words(String line) {
235           final String ID = "[A-Za-z_][A-Za-z_0-9]*";
236           Pattern p = Pattern.compile(ID);
237           Matcher m = p.matcher(line);
238           final Set<String> words = new HashSet<>();
239           while (m.find()) {
240               words.add(m.group(0));
241           }
242           return words;
243       }

245       private String manageEnvironment(String line) {
246           for (Map.Entry<String, String> ent : environment.entrySet()) {
247               final String key = Pattern.quote("%" + ent.getKey() + "%");
248               line = line.replaceAll(key, ent.getValue());
249           }
250           return line;
251       }
252   
253       private static final String DATE = "(?i)%date(\\[(.+?)\\])?%";
254       private final static Pattern datePattern = Pattern.compile(DATE);
255   
256       private String manageDate(String line) {
257           final Matcher m = datePattern.matcher(line);
258           if (m.find()) {
259               final String format = m.group(2);
260               String replace;
261               if (format == null) {
262                   replace = new Date().toString();
263               } else {
264                   try {
265                       replace = new SimpleDateFormat(format).format(new Date());
266                   } catch (Exception e) {
267                       replace = "(BAD DATE PATTERN:" + format + ")";
268                   }
269               }
270               line = line.replaceAll(DATE, replace);
271           }
272           return line;
273       }
