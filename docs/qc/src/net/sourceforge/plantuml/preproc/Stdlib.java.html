77       private String loadResource(String file) throws IOException {
78           final SoftReference<String> cached = cache.get(file.toLowerCase());
79           if (cached != null) {
80               final String cachedResult = cached.get();
81               if (cachedResult != null) {
82                   // Log.info("Using cache for " + file);
83                   return cachedResult;
84               }
85           }
86           Log.info("No cache for " + file);
87           final DataInputStream dataStream = getDataStream();
88           if (dataStream == null)
89               return null;
90   
91           dataStream.readUTF();
92           final InputStream spriteStream = getSpriteStream();
93           if (spriteStream == null) {
94               dataStream.close();
95               return null;
96           }
97           try {
98               StringBuilder found = null;
99               while (true) {
100                   final String filename = dataStream.readUTF();
101                   if (filename.equals(SEPARATOR)) {
102                       Log.info("Not found " + filename);
103                       return null;
104                   }
105                   if (filename.equalsIgnoreCase(file))
106                       found = new StringBuilder();
107   
108                   while (true) {
109                       final String s = dataStream.readUTF();
110                       if (s.equals(SEPARATOR)) {
111                           if (found != null) {
112                               final String result = found.toString();
113                               cache.put(file.toLowerCase(), new SoftReference<>(result));
114                               return result;
115                           }
116                           break;
117                       }
118                       if (found != null) {
119                           found.append(s);
120                           found.append("\n");
121                       }
122                       if (isSpriteLine(s)) {
123                           final Matcher m = sizePattern.matcher(s);
124                           final boolean ok = m.find();
125                           if (ok == false)
126                               throw new IOException(s);
127   
128                           final int width = Integer.parseInt(m.group(1));
129                           final int height = Integer.parseInt(m.group(2));
130                           if (found == null) {
131                               skipSprite(width, height, spriteStream);
132                           } else {
133                               final String sprite = readSprite(width, height, spriteStream);
134                               found.append(sprite);
135                               found.append("}\n");
136                           }
137                       }
138                   }
139               }
140           } finally {
141               dataStream.close();
142               spriteStream.close();
143           }
144   
145       }

157       private String readSprite(int width, int height, InputStream inputStream) throws IOException {
158           final int nbLines = (height + 1) / 2;
159           final StringBuilder result = new StringBuilder();
160           int line = 0;
161           for (int j = 0; j < nbLines; j++) {
162               final StringBuilder sb1 = new StringBuilder();
163               final StringBuilder sb2 = new StringBuilder();
164               for (int i = 0; i < width; i++) {
165                   final int b = inputStream.read();
166                   final int b1 = (b & 0xF0) >> 4;
167                   final int b2 = (b & 0x0F);
168                   sb1.append(toHexString(b1));
169                   sb2.append(toHexString(b2));
170               }
171               result.append(sb1.toString());
172               result.append("\n");
173               line++;
174               if (line < height) {
175                   result.append(sb2.toString());
176                   result.append("\n");
177                   line++;
178               }
179           }
180           return result.toString();
181       }
