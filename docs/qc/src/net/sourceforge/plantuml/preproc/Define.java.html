36   package net.sourceforge.plantuml.preproc;
37   
38   import java.util.Iterator;
39   import java.util.List;
40   import java.util.regex.Matcher;
41   import java.util.regex.Pattern;
42   
43   import net.sourceforge.plantuml.AParentFolder;
44   import net.sourceforge.plantuml.BackSlash;
45   import net.sourceforge.plantuml.tim.expression.TValue;
46   
47   public class Define {
48   
49       private final DefineSignature signature;
50       private final String definition;
51       private final String definitionQuoted;
52       private final boolean emptyParentheses;
53       private Pattern pattern;
54       private final AParentFolder currentDir;
55   
56       public Define(String key, List<String> lines, boolean emptyParentheses, AParentFolder currentDir) {
57           this.currentDir = currentDir;
58           this.emptyParentheses = emptyParentheses;
59           if (lines == null) {
60               this.definition = null;
61               this.definitionQuoted = null;
62           } else {
63               final StringBuilder sb = new StringBuilder();
64               for (final Iterator<String> it = lines.iterator(); it.hasNext();) {
65                   sb.append(it.next());
66                   if (it.hasNext()) {
67                       sb.append('\n');
68                   }
69               }
70               this.definition = sb.toString();
71               this.definitionQuoted = Matcher.quoteReplacement(definition);
72           }
73           this.signature = new DefineSignature(key, this.definitionQuoted);
74   
75       }
76   
77       @Override
78       public String toString() {
79           return signature.toString();
80       }
81   
82       public String apply(String line) {
83           if (definition == null) {
84               return line;
85           }
86           // if (getFunctionName().indexOf('_') >= 0 && line.indexOf('_') == -1) {
87           // return line;
88           // }
89           if (/* line.length() < getFunctionName().length() || */line.contains(getFunctionName()) == false) {
90               return line;
91           }
92           if (signature.isMethod()) {
93               if (line.indexOf('(') == -1) {
94                   return line;
95               }
96               line = apply1(line);
97           } else {
98               line = apply2(line);
99           }
100           return line;
101       }
102   
103       private String apply2(String line) {
104           if (pattern == null) {
105               final String regex = "\\b" + signature.getKey() + "\\b" + (emptyParentheses ? "(\\(\\))?" : "");
106               pattern = Pattern.compile(regex);
107           }
108   
109           line = BackSlash.translateBackSlashes(line);
110           line = pattern.matcher(line).replaceAll(definitionQuoted);
111           line = BackSlash.untranslateBackSlashes(line);
112           return line;
113       }
114   
115       private String apply1(String line) {
116           for (Variables vars : signature.getVariationVariables()) {
117               line = vars.applyOn(line);
118           }
119           return line;
120       }
121   
122       public final String getFunctionName() {
123           return signature.getFonctionName();
124       }
125   
126       public TValue asTVariable() {
127           return TValue.fromString(definition);
128       }
129   }
