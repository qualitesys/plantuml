43   import net.sourceforge.plantuml.sequencediagram.graphic.Segment;
44   import net.sourceforge.plantuml.ugraphic.UGraphic;

52   public class RectangleTask {
53   
54       private final List<Segment> segments;
55       private final double round;
56       private final int completion;
57   
58       public RectangleTask(double startPos, double endPos, double round, int completion, Collection<Segment> paused) {
59           this.round = round;
60           this.completion = completion;
61           if (startPos < endPos) {
62               this.segments = new ArrayList<>(new Segment(startPos, endPos).cutSegmentIfNeed(paused));
63           } else {
64               this.segments = Collections.singletonList(new Segment(startPos, startPos + 1));
65           }
66       }
67   
68       private void draw2hlines(UGraphic ug, double height, ULine hline) {
69           ug.draw(hline);
70           ug.apply(UTranslate.dy(height)).draw(hline);
71       }
72   
73       private void drawRect(double widthCompletion, UGraphic ug, HColor documentBackground, double width, double height) {
74           if (widthCompletion == -1 || widthCompletion == 0) {
75               if (widthCompletion == 0)
76                   ug = ug.apply(documentBackground.bg());
77               final URectangle rect = new URectangle(width, height);
78               ug.draw(rect);
79           } else {
80               final URectangle rect1 = new URectangle(widthCompletion, height);
81               ug.draw(rect1);
82               final URectangle rect2 = new URectangle(width - widthCompletion, height);
83               ug.apply(documentBackground.bg()).apply(UTranslate.dx(widthCompletion)).draw(rect2);
84           }
85       }
86   
87       public void draw(UGraphic ug, double height, HColor documentBackground, boolean oddStart, boolean oddEnd) {
88   
89           if (round == 0) {
90               drawWithoutRound(ug, height, documentBackground, oddStart, oddEnd);
91               return;
92           }
93   
94           if (segments.size() != 1) {
95               drawWithRound(ug, height, documentBackground);
96               return;
97           }
98   
99           assert segments.size() == 1;
100           assert round > 0;
101           final Segment segment = segments.get(0);
102   
103           final double width = segment.getLength();
104           final URectangle partial = new URectangle(width, height).rounded(round);
105           if (completion == 100 || completion == 0) {
106               if (completion == 0)
107                   ug = ug.apply(documentBackground.bg());
108               if (oddStart && !oddEnd)
109                   ug.apply(UTranslate.dx(segment.getPos1())).draw(PathUtils.UtoRight(width, height, round));
110               else if (!oddStart && oddEnd)
111                   ug.apply(UTranslate.dx(segment.getPos1())).draw(PathUtils.UtoLeft(width, height, round));
112               else
113                   ug.apply(UTranslate.dx(segment.getPos1())).draw(partial);
114           } else {
115               final double x1 = width * completion / 100;
116               ug.apply(HColors.none()).apply(UTranslate.dx(segment.getPos1()))
117                       .draw(PathUtils.UtoLeft(x1, height, round));
118               ug.apply(documentBackground.bg()).apply(HColors.none()).apply(UTranslate.dx(segment.getPos1() + x1))
119                       .draw(PathUtils.UtoRight(width * (100 - completion) / 100, height, round));
120               ug.apply(HColors.none().bg()).apply(UTranslate.dx(segment.getPos1())).draw(partial);
121           }
122   
123       }
124   
125       private void drawWithRound(UGraphic ug, double height, HColor documentBackground) {
126   
127           final Segment first = segments.get(0);
128           ug.apply(UTranslate.dx(first.getPos1())).draw(PathUtils.UtoLeft(first.getLength(), height, round));
129   
130           for (int i = 1; i < segments.size() - 1; i++) {
131               final Segment segment = segments.get(i);
132               drawPartly(segment.getLength() * completion / 100, ug, segment, height, documentBackground, i);
133           }
134   
135           final Segment last = segments.get(segments.size() - 1);
136           ug.apply(UTranslate.dx(last.getPos1())).draw(PathUtils.UtoRight(last.getLength(), height, round));
137   
138           drawIntermediateDotted(ug, height);
139       }
140   
141       private void drawWithoutRound(UGraphic ug, double height, HColor documentBackground, boolean oddStart,
142               boolean oddEnd) {
143           final ULine vline = ULine.vline(height);
144   
145           final double sum = getFullSegmentsLength();
146           final double lim = completion == 100 ? sum : sum * completion / 100;
147           double current = 0;
148   
149           for (int i = 0; i < segments.size(); i++) {
150               final Segment segment = segments.get(i);
151               final double next = current + segment.getLength();
152               final double widthCompletion;
153               if (lim >= next)
154                   widthCompletion = -1;
155               else if (current >= lim)
156                   widthCompletion = 0;
157               else {
158                   assert current < lim && lim < next;
159                   widthCompletion = lim - current;
160               }
161   
162               drawPartly(widthCompletion, ug, segment, height, documentBackground, i);
163   
164               if (!oddStart && i == 0) {
165                   ug.apply(UTranslate.dx(segment.getPos1())).draw(vline);
166               }
167               if (!oddEnd && i == segments.size() - 1) {
168                   ug.apply(UTranslate.dx(segment.getPos2())).draw(vline);
169               }
170               current = next;
171           }
172           drawIntermediateDotted(ug, height);
173       }
174   
175       private double getFullSegmentsLength() {
176           double result = 0;
177           for (Segment seg : segments)
178               result += seg.getLength();
179           return result;
180       }
181   
182       private void drawIntermediateDotted(UGraphic ug, double height) {
183           ug = ug.apply(new UStroke(2, 3, 1));
184           for (int i = 0; i < segments.size() - 1; i++) {
185               final double v1 = segments.get(i).getPos2() + 3;
186               final double v2 = segments.get(i + 1).getPos1() - 3;
187               if (v2 > v1) {
188                   draw2hlines(ug.apply(UTranslate.dx(v1)), height, ULine.hline(v2 - v1));
189               }
190           }
191       }
192   
193       private void drawPartly(double widthCompletion, UGraphic ug, final Segment segment, double height,
194               HColor documentBackground, int i) {
195           double width = segment.getLength();
196           if (i != segments.size() - 1) {
197               width++;
198           }
199           if (width > 0) {
200               drawRect(widthCompletion, ug.apply(HColors.none()).apply(UTranslate.dx(segment.getPos1())),
201                       documentBackground, width, height);
202           }
203   
204           double pos1 = segment.getPos1();
205           double len = segment.getLength();
206           if (i == 0) {
207               if (segments.size() > 1) {
208                   len--;
209               }
210           } else {
211               pos1++;
212               len--;
213           }
214           if (len > 0) {
215               draw2hlines(ug.apply(UTranslate.dx(pos1)), height, ULine.hline(len));
216           }
217       }
