55   public class ResourceDraw implements UDrawable {
56   
57       private final Resource res;
58       private final TimeScale timeScale;
59       private final double y;
60       private final Day min;
61       private final Day max;
62       private final GanttDiagram gantt;
63   
64       public ResourceDraw(GanttDiagram gantt, Resource res, TimeScale timeScale, double y, Day min, Day max) {
65           this.res = res;
66           this.timeScale = timeScale;
67           this.y = y;
68           this.min = min;
69           this.max = max;
70           this.gantt = gantt;
71       }
72   
73       public void drawU(UGraphic ug) {
74           final TextBlock title = Display.getWithNewlines(res.getName()).create(getFontConfiguration(13),
75                   HorizontalAlignment.LEFT, new SpriteContainerEmpty());
76           title.drawU(ug);
77           final ULine line = ULine.hline(timeScale.getEndingPosition(max) - timeScale.getStartingPosition(min));
78           ug.apply(HColors.BLACK).apply(UTranslate.dy(title.calculateDimension(ug.getStringBounder()).getHeight()))
79                   .draw(line);
80   
81           double startingPosition = -1;
82           int totalLoad = 0;
83           int totalLimit = 0;
84           for (Day i = min; i.compareTo(max) <= 0; i = i.increment()) {
85               final boolean isBreaking = timeScale.isBreaking(i);
86               totalLoad += gantt.getLoadForResource(res, i);
87               totalLimit += 100;
88               if (isBreaking) {
89                   if (totalLoad > 0) {
90                       final boolean over = totalLoad > totalLimit;
91                       final FontConfiguration fontConfiguration = getFontConfiguration(9,
92                               over ? HColors.RED : HColors.BLACK);
93                       final TextBlock value = Display.getWithNewlines("" + totalLoad).create(fontConfiguration,
94                               HorizontalAlignment.LEFT, new SpriteContainerEmpty());
95                       if (startingPosition == -1)
96                           startingPosition = timeScale.getStartingPosition(i);
97                       final double endingPosition = timeScale.getEndingPosition(i);
98                       final double start = (startingPosition + endingPosition) / 2
99                               - value.calculateDimension(ug.getStringBounder()).getWidth() / 2;
100                       value.drawU(ug.apply(new UTranslate(start, 16)));
101                   }
102                   startingPosition = -1;
103                   totalLoad = 0;
104                   totalLimit = 0;
105               } else {
106                   if (startingPosition == -1)
107                       startingPosition = timeScale.getStartingPosition(i);
108               }
109           }
110   
111       }
112   
113       private FontConfiguration getFontConfiguration(int size) {
114           return getFontConfiguration(size, HColors.BLACK);
115       }
116   
117       private FontConfiguration getFontConfiguration(int size, HColor color) {
118           final UFont font = UFont.serif(size);
119           return FontConfiguration.create(font, color, color, null);
120       }
121   
122       public double getHeight() {
123           return 16 * 2;
124       }
125   
126       public final double getY() {
127           return y;
128       }
129   
130   }
