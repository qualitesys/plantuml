88       @Override
89       public void drawTitle(UGraphic ug, LabelStrategy labelStrategy, double colTitles, double colBars) {
90           final ClockwiseTopRightBottomLeft padding = getStyle().getPadding();
91           final ClockwiseTopRightBottomLeft margin = getStyle().getMargin();
92           final double dx = margin.getLeft() + padding.getLeft();
93           final double dy = margin.getTop() + padding.getTop();
94           final double x;
95           if (labelStrategy.titleInFirstColumn()) {
96               x = colTitles;
97           } else {
98               x = 0;
99           }
100           getTitle().drawU(ug.apply(new UTranslate(x + dx, dy)));
101       }

119               return TextBlockUtils.empty(0, 0);

129       @Override
130       public void drawU(UGraphic ug) {
131           final StringBounder stringBounder = ug.getStringBounder();
132           final double widthTitle = getTitle().calculateDimension(stringBounder).getWidth();
133           final double start = timeScale.getStartingPosition(min);
134           // final double start2 = start1 + widthTitle;
135           final double end = timeScale.getEndingPosition(max);
136   
137           final ClockwiseTopRightBottomLeft padding = getStyle().getPadding();
138           final ClockwiseTopRightBottomLeft margin = getStyle().getMargin();
139           ug = ug.apply(new UTranslate(0, margin.getTop()));
140   
141           final HColor backColor = getStyle().value(PName.BackGroundColor)
142                   .asColor(colorSet);
143   
144           if (backColor.isTransparent() == false) {
145               final double height = padding.getTop() + getTextHeight(stringBounder) + padding.getBottom();
146               if (height > 0) {
147                   final URectangle rect = new URectangle(end - start, height);
148                   ug.apply(backColor.bg()).draw(rect);
149               }
150           }
151   
152           final HColor lineColor = getStyle().value(PName.LineColor).asColor(colorSet);
153           ug = ug.apply(lineColor);
154           ug = ug.apply(UTranslate.dy(padding.getTop() + getTextHeight(stringBounder) / 2));
155   
156           if (widthTitle == 0) {
157               final ULine line = ULine.hline(end - start);
158               ug.draw(line);
159           } else {
160               if (padding.getLeft() > 1) {
161                   final ULine line1 = ULine.hline(padding.getLeft());
162                   ug.draw(line1);
163               }
164               final double x1 = padding.getLeft() + margin.getLeft() + widthTitle + margin.getRight();
165               final double x2 = end - 1;
166               final ULine line2 = ULine.hline(x2 - x1);
167               ug.apply(UTranslate.dx(x1)).draw(line2);
168           }
169       }
