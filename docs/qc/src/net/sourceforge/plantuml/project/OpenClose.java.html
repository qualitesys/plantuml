36   package net.sourceforge.plantuml.project;
37   
38   import java.util.Collection;
39   import java.util.EnumSet;
40   import java.util.HashSet;
41   
42   import net.sourceforge.plantuml.project.core3.Histogram;
43   import net.sourceforge.plantuml.project.core3.TimeLine;
44   import net.sourceforge.plantuml.project.time.Day;
45   import net.sourceforge.plantuml.project.time.DayOfWeek;
46   
47   public class OpenClose implements Histogram, LoadPlanable {
48   
49       private final Collection<DayOfWeek> closedDayOfWeek = EnumSet.noneOf(DayOfWeek.class);
50       private final Collection<DayOfWeek> openedDayOfWeek = EnumSet.noneOf(DayOfWeek.class);
51       private final Collection<Day> closedDays = new HashSet<>();
52       private final Collection<Day> openedDays = new HashSet<>();
53       private Day startingDay;
54   
55       public int daysInWeek() {
56           return 7 - closedDayOfWeek.size();
57       }
58   
59       private boolean isThereSomeChangeAfter(Day day) {
60           if (closedDayOfWeek.size() > 0)
61               return true;
62   
63           for (Day tmp : closedDays)
64               if (tmp.compareTo(day) >= 0)
65                   return true;
66   
67           for (Day tmp : openedDays)
68               if (tmp.compareTo(day) >= 0)
69                   return true;
70   
71           return false;
72       }
73   
74       private boolean isThereSomeChangeBefore(Day day) {
75           if (closedDayOfWeek.size() > 0)
76               return true;
77   
78           for (Day tmp : closedDays)
79               if (tmp.compareTo(day) <= 0)
80                   return true;
81   
82           for (Day tmp : openedDays)
83               if (tmp.compareTo(day) <= 0)
84                   return true;
85   
86           return false;
87       }
88   
89       public boolean isClosed(Day day) {
90           if (openedDays.contains(day))
91               return false;
92   
93           final DayOfWeek dayOfWeek = day.getDayOfWeek();
94           return closedDayOfWeek.contains(dayOfWeek) || closedDays.contains(day);
95       }
96   
97       public void close(DayOfWeek day) {
98           closedDayOfWeek.add(day);
99       }
100   
101       public void open(DayOfWeek day) {
102           closedDayOfWeek.remove(day);
103           openedDayOfWeek.add(day);
104       }
105   
106       public void close(Day day) {
107           closedDays.add(day);
108       }
109   
110       public void open(Day day) {
111           openedDays.add(day);
112       }
113   
114       public final Day getStartingDay() {
115           return startingDay;
116       }
117   
118       public final void setStartingDay(Day startingDay) {
119           this.startingDay = startingDay;
120       }
121   
122       public long getNext(long moment) {
123           Day day = Day.create(moment);
124           if (isThereSomeChangeAfter(day) == false)
125               return TimeLine.MAX_TIME;
126   
127           final long current = getLoatAtInternal(day);
128           System.err.println("getNext:day=" + day + " current=" + current);
129           while (true) {
130               day = day.increment();
131               final int tmp = getLoatAtInternal(day);
132               System.err.println("..day=" + day + " " + tmp);
133               if (tmp != current)
134                   return day.getMillis();
135   
136           }
137       }
138   
139       public long getPrevious(long moment) {
140           Day day = Day.create(moment);
141           if (isThereSomeChangeBefore(day) == false)
142               return -TimeLine.MAX_TIME;
143   
144           final long current = getLoatAtInternal(day);
145           System.err.println("getPrevious=" + day + " current=" + current);
146           while (true) {
147               day = day.decrement();
148               final int tmp = getLoatAtInternal(day);
149               System.err.println("..day=" + day + " " + tmp);
150               if (tmp != current)
151                   return day.getMillis();
152   
153           }
154       }
155   
156       public long getValueAt(long moment) {
157           final Day day = Day.create(moment);
158           if (isClosed(day))
159               return 0;
160   
161           return 100;
162       }
163   
164       public int getLoadAt(Day day) {
165           if (getStartingDay() == null)
166               return 100;
167   
168           return getLoatAtInternal(day);
169       }
170   
171       private int getLoatAtInternal(Day day) {
172           if (isClosed(day))
173               return 0;
174   
175           return 100;
176       }
177   
178       public LoadPlanable mutateMe(final OpenClose except) {
179           if (except != null)
180               return new LoadPlanable() {
181                   @Override
182                   public int getLoadAt(Day instant) {
183                       if (except.openedDays.contains(instant))
184                           return 100;
185                       if (except.closedDays.contains(instant))
186                           return 0;
187                       if (except.openedDayOfWeek.size() > 0 && except.openedDayOfWeek.contains(instant.getDayOfWeek()))
188                           return 100;
189                       return OpenClose.this.getLoadAt(instant);
190                   }
191               };
192           return this;
193       }

195   }
