426       private void initTaskAndResourceDraws(TimeScale timeScale, double headerHeight, StringBounder stringBounder) {
427           Real y = origin.addFixed(headerHeight);
428           for (Task task : tasks.values()) {
429               final TaskDraw draw;
430               if (task instanceof TaskSeparator) {
431                   final TaskSeparator taskSeparator = (TaskSeparator) task;
432                   draw = new TaskDrawSeparator(taskSeparator.getName(), timeScale, y, min, max, task.getStyleBuilder(),
433                           getSkinParam().getIHtmlColorSet());
434               } else if (task instanceof TaskGroup) {
435                   final TaskGroup taskGroup = (TaskGroup) task;
436                   draw = new TaskDrawGroup(timeScale, y, taskGroup.getCode().getSimpleDisplay(), getStart(taskGroup),
437                           getEnd(taskGroup), getSkinParam(), task, this, task.getStyleBuilder());
438               } else {
439                   final TaskImpl tmp = (TaskImpl) task;
440                   final String disp = hideResourceName ? tmp.getCode().getSimpleDisplay() : tmp.getPrettyDisplay();
441                   if (tmp.isDiamond()) {
442                       draw = new TaskDrawDiamond(timeScale, y, disp, getStart(tmp), getSkinParam(), task, this,
443                               task.getStyleBuilder());
444                   } else {
445                       final boolean oddStart = printStart != null && min.compareTo(getStart(tmp)) == 0;
446                       final boolean oddEnd = printStart != null && max.compareTo(getEnd(tmp)) == 0;
447                       draw = new TaskDrawRegular(timeScale, y, disp, getStart(tmp), getEnd(tmp), oddStart, oddEnd,
448                               getSkinParam(), task, this, getConstraints(task), task.getStyleBuilder());
449                   }
450                   draw.setColorsAndCompletion(tmp.getColors(), tmp.getCompletion(), tmp.getUrl(), tmp.getNote());
451               }
452               if (task.getRow() == null)
453                   y = y.addAtLeast(draw.getFullHeightTask(stringBounder));
454   
455               draws.put(task, draw);
456           }
457           origin.compileNow();
458           magicPush(stringBounder);
459           double yy = lastY(stringBounder);
460           if (yy == 0) {
461               yy = headerHeight;
462           } else if (this.hideResourceFoobox == false)
463               for (Resource res : resources.values()) {
464                   final ResourceDraw draw = new ResourceDraw(this, res, timeScale, yy, min, max);
465                   res.setTaskDraw(draw);
466                   yy += draw.getHeight();
