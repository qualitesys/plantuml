57   public class GanttArrow implements UDrawable {
58   
59       private final TimeScale timeScale;
60       private final Direction atStart;
61       private final TaskInstant source;
62       private final Direction atEnd;
63       private final TaskInstant dest;
64   
65       private final HColorSet colorSet;
66       private final Style style;
67       private final ToTaskDraw toTaskDraw;
68       private final StyleBuilder styleBuilder;

70       public GanttArrow(HColorSet colorSet, Style style, TimeScale timeScale, TaskInstant source, TaskInstant dest,
71               ToTaskDraw toTaskDraw, StyleBuilder styleBuilder) {
72           this.styleBuilder = styleBuilder;
73           this.toTaskDraw = toTaskDraw;
74           this.style = style;
75           this.colorSet = colorSet;
76           this.timeScale = timeScale;
77           this.source = source;
78           this.dest = dest;
79           if (source.getAttribute() == TaskAttribute.END && dest.getAttribute() == TaskAttribute.START) {
80               this.atStart = source.sameRowAs(dest) ? Direction.LEFT : Direction.DOWN;
81               this.atEnd = Direction.RIGHT;
82           } else if (source.getAttribute() == TaskAttribute.END && dest.getAttribute() == TaskAttribute.END) {
83               this.atStart = Direction.RIGHT;
84               this.atEnd = Direction.LEFT;
85           } else if (source.getAttribute() == TaskAttribute.START && dest.getAttribute() == TaskAttribute.START) {
86               this.atStart = Direction.LEFT;
87               this.atEnd = Direction.RIGHT;
88           } else if (source.getAttribute() == TaskAttribute.START && dest.getAttribute() == TaskAttribute.END) {
89               this.atStart = source.sameRowAs(dest) ? Direction.RIGHT : Direction.DOWN;
90               this.atEnd = Direction.LEFT;
91           } else {
92               throw new IllegalArgumentException();
93           }
94       }
95   
96       private TaskDraw getSource() {
97           if (source.getMoment() instanceof Task)
98               return toTaskDraw.getTaskDraw((Task) source.getMoment());
99   
100           return null;
101   
102       }
103   
104       private TaskDraw getDestination() {
105           if (dest.getMoment() instanceof Task)
106               return toTaskDraw.getTaskDraw((Task) dest.getMoment());
107   
108           return null;
109       }

111       public void drawU(UGraphic ug) {
112           ug = style.applyStrokeAndLineColor(ug, colorSet);
113   
114           final TaskDraw start = getSource();
115           final TaskDraw end = getDestination();
116           if (start == null || end == null)
117               return;
118   
119           double x1 = getX(source.getAttribute(), start, atStart);
120           final StringBounder stringBounder = ug.getStringBounder();
121           double y1 = start.getY(stringBounder, atStart);
122   
123           final double x2 = getX(dest.getAttribute(), end, atEnd.getInv());
124           final double y2 = end.getY(stringBounder, atEnd);
125   
126           if (atStart == Direction.DOWN && y2 < y1)
127               y1 = start.getY(stringBounder, atStart.getInv());
128   
129           final double minimalWidth = 8;
130   
131           if (this.atStart == Direction.DOWN && this.atEnd == Direction.RIGHT) {
132               if (x2 > x1) {
133                   if (x2 - x1 < minimalWidth)
134                       x1 = x2 - minimalWidth;
135   
136                   drawLine(ug, x1, y1, x1, y2, x2, y2);
137               } else {
138                   x1 = getX(source.getAttribute(), start, Direction.RIGHT);
139                   y1 = start.getY(stringBounder, Direction.RIGHT);
140                   final double y1b = end.getY(stringBounder).getCurrentValue();
141                   drawLine(ug, x1, y1, x1 + 6, y1, x1 + 6, y1b, x2 - 8, y1b, x2 - 8, y2, x2, y2);
142               }
143           } else if (this.atStart == Direction.RIGHT && this.atEnd == Direction.LEFT) {
144               final double xmax = Math.max(x1, x2) + 8;
145               drawLine(ug, x1, y1, xmax, y1, xmax, y2, x2, y2);
146           } else if (this.atStart == Direction.LEFT && this.atEnd == Direction.RIGHT) {
147               final double xmin = Math.min(x1, x2) - 8;
148               drawLine(ug, x1, y1, xmin, y1, xmin, y2, x2, y2);
149           } else if (this.atStart == Direction.DOWN && this.atEnd == Direction.LEFT) {
150               drawLine(ug, x1, y1, x1, y2, x2, y2);
151           } else {
152               throw new IllegalArgumentException();
153           }
154   
155           ug = ug.apply(new UStroke(1.5)).apply(
156                   style.value(PName.LineColor).asColor(colorSet).bg());
157           ug.apply(new UTranslate(x2, y2)).draw(Arrows.asTo(atEnd));
158   
159       }
160   
161       private void drawLine(UGraphic ug, double... coord) {
162           for (int i = 0; i < coord.length - 2; i += 2) {
163               final double x1 = coord[i];
164               final double y1 = coord[i + 1];
165               final double x2 = coord[i + 2];
166               final double y2 = coord[i + 3];
167               ug.apply(new UTranslate(x1, y1)).draw(new ULine(x2 - x1, y2 - y1));
168           }
169   
170       }

172       private StyleSignatureBasic getStyleSignatureTask() {
173           return StyleSignatureBasic.of(SName.root, SName.element, SName.ganttDiagram, SName.task);
174       }

176       private double getX(TaskAttribute taskAttribute, TaskDraw task, Direction direction) {
177           if (direction == Direction.LEFT) {
178               return task.getX1(taskAttribute) - 1;
179           }
180           if (direction == Direction.RIGHT) {
181               return task.getX2(taskAttribute) + 1;
182           }
183           return (task.getX1(taskAttribute) + (task.getX2(taskAttribute))) / 2;
184       }
