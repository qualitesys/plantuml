38   import java.io.IOException;
39   import java.io.OutputStream;

41   import java.text.SimpleDateFormat;
42   import java.util.Arrays;
43   import java.util.Collection;
44   import java.util.Date;
45   import java.util.List;
46   import java.util.Locale;
47   import java.util.Map;
48   import java.util.TimeZone;
49   
50   import ext.plantuml.com.ctreber.acearth.ACearth;
51   import ext.plantuml.com.ctreber.acearth.ConfigurationACearth;
52   import ext.plantuml.com.ctreber.acearth.plugins.markers.Marker;
53   import net.sourceforge.plantuml.AbstractPSystem;
54   import net.sourceforge.plantuml.FileFormatOption;
55   import net.sourceforge.plantuml.api.ImageDataSimple;
56   import net.sourceforge.plantuml.core.DiagramDescription;
57   import net.sourceforge.plantuml.core.ImageData;
58   import net.sourceforge.plantuml.core.UmlSource;
59   
60   public class PSystemXearth extends AbstractPSystem {
61   
62       final private int width;
63       final private int height;
64       final private Map<String, String> config;
65       final private List<Marker> markers;
66   
67       final private Collection<String> enums = Arrays.asList("viewPositionType");
68       final private Collection<String> doubles = Arrays.asList("sunPosRelLat", "sunPosRelLong", "orbitPeriod",
69               "orbitInclination", "viewPosLat", "viewPosLong", "starFrequency", "viewMagnification");
70       final private Collection<String> integers = Arrays.asList("daySideBrightness", "nightSideBrightness",
71               "terminatorDiscontinuity", "gridDivision", "gridPixelDivision", "bigStars");
72       final private Collection<String> booleans = Arrays.asList("shadeP", "gridP", "starsP");
73   
74       public PSystemXearth(UmlSource source, int width, int height, Map<String, String> config, List<Marker> markers) {
75           super(source);
76           this.width = width;
77           this.height = height;
78           this.config = config;
79           this.markers = markers;
80       }
81   
82       @Override
83       final protected ImageData exportDiagramNow(OutputStream os, int num, FileFormatOption fileFormat)
84               throws IOException {
85           synchronized (PSystemXearth.class) {
86               final ACearth earth = new ACearth(markers);
87               final ConfigurationACearth conf = earth.getConf();
88               conf.setInt("imageWidth", width);
89               conf.setInt("imageHeight", height);
90   
91               for (Map.Entry<String, String> ent : config.entrySet()) {
92                   final String key = ent.getKey();
93                   final String value = ent.getValue();
94                   if (key.equalsIgnoreCase("GMT")) {
95                       final Date date = extractGmt(value);
96                       conf.setInt("fixedTime", (int) (date.getTime() / 1000L));
97                   } else if (enums.contains(key)) {
98                       conf.getMOEnum(key).set(value);
99                   } else if (doubles.contains(key)) {
100                       conf.setDouble(key, Double.parseDouble(value));
101                   } else if (integers.contains(key)) {
102                       conf.setInt(key, Integer.parseInt(value));
103                   } else if (booleans.contains(key)) {
104                       conf.setBoolean(key, value.equalsIgnoreCase("true"));
105                   } else {
106                       throw new UnsupportedOperationException(key);
107                   }
108               }
109               earth.exportPng(os);
110               return new ImageDataSimple(width, height);
111           }
112       }
113   
114       private Date extractGmt(String s) {
115           final SimpleDateFormat timeFormat;
116           if (s.matches("\\d{4}/\\d{2}/\\d{2} \\d{2}:\\d{2}:\\d{2}")) {
117               timeFormat = new SimpleDateFormat("yyyy/MM/dd HH:mm:ss", Locale.US);
118           } else if (s.matches("\\d{4}/\\d{2}/\\d{2} \\d{2}:\\d{2}")) {
119               timeFormat = new SimpleDateFormat("yyyy/MM/dd HH:mm", Locale.US);
120           } else {
121               throw new UnsupportedOperationException(s);
122           }
123           timeFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
124           try {
125               return timeFormat.parse(s);
126           } catch (ParseException e) {
127               throw new UnsupportedOperationException(s);
128           }
129   
130       }
131   
132       public DiagramDescription getDescription() {
133           return new DiagramDescription("(XEarth)");
134       }
