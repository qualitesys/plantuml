63   public class ScientificEquationSafe {
64   
65       private final ScientificEquation equation;
66       private final String formula;
67   
68       private ScientificEquationSafe(String formula, ScientificEquation equation) {
69           this.formula = formula;
70           this.equation = equation;
71       }
72   
73       public static ScientificEquationSafe fromAsciiMath(String formula) {
74           try {
75               return new ScientificEquationSafe(formula, new AsciiMath(formula));
76           } catch (Exception e) {
77               Logme.error(e);
78               Log.info("Error parsing " + formula);
79               return new ScientificEquationSafe(formula, null);
80           }
81       }
82   
83       public static ScientificEquationSafe fromLatex(String formula) {
84           try {
85               return new ScientificEquationSafe(formula, new LatexBuilder(formula));
86           } catch (Exception e) {
87               Logme.error(e);
88               Log.info("Error parsing " + formula);
89               return new ScientificEquationSafe(formula, null);
90           }
91       }
92   
93       private ImageData dimSvg;
94   
95       public UImageSvg getSvg(double scale, Color foregroundColor, Color backgroundColor) {
96           if (equation != null)
97               try {
98                   final UImageSvg svg = equation.getSvg(scale, foregroundColor, backgroundColor);
99                   dimSvg = new ImageDataSimple(equation.getDimension());
100                   return svg;
101               } catch (Exception e) {
102                   printTrace(e);
103               }
104           final ByteArrayOutputStream baos = new ByteArrayOutputStream();
105           try {
106               dimSvg = plainImageBuilder(getRollback(), new FileFormatOption(FileFormat.SVG)).write(baos);
107           } catch (IOException e1) {
108               return null;
109           }
110           return new UImageSvg(new String(baos.toByteArray()), scale);
111       }
112   
113       public MutableImage getImage(Color foregroundColor, Color backgroundColor) {
114           if (equation != null)
115               try {
116                   return equation.getImage(foregroundColor, backgroundColor);
117               } catch (Exception e) {
118                   printTrace(e);
119               }
120           try {
121               final byte[] bytes = plainPngBuilder(getRollback()).writeByteArray();
122               return new PixelImage(SImageIO.read(bytes), AffineTransformType.TYPE_BILINEAR);
123           } catch (IOException e1) {
124               return null;
125           }
126       }
127   
128       private void printTrace(Exception e) {
129           System.err.println("formula=" + formula);
130           if (equation != null) {
131               System.err.println("Latex=" + equation.getSource());
132           }
133           Logme.error(e);
134       }
135   
136       private TextBlockBackcolored getRollback() {
137           return GraphicStrings.createBlackOnWhiteMonospaced(Arrays.asList(formula));
138       }
139   
140       public ImageData export(OutputStream os, FileFormatOption fileFormat, float scale, Color foregroundColor,
141               Color backgroundColor) throws IOException {
142           if (fileFormat.getFileFormat() == FileFormat.PNG) {
143               final BufferedImage image = getImage(foregroundColor, backgroundColor).withScale(scale).getImage();
144               SImageIO.write(image, "png", os);
145               return new ImageDataSimple(image.getWidth(), image.getHeight());
146           }
147           if (fileFormat.getFileFormat() == FileFormat.SVG) {
148               os.write(getSvg(1, foregroundColor, backgroundColor).getSvg(true).getBytes());
149               return dimSvg;
150           }
151           if (fileFormat.getFileFormat() == FileFormat.EPS) {
152               final BufferedImage image = getImage(foregroundColor, backgroundColor).withScale(scale).getImage();
153               final EpsGraphics out = new EpsGraphics();
154               out.drawImage(image, 0, 0);
155               out.close();
156               os.write(out.getEPSCode().getBytes());
157               return new ImageDataSimple(image.getWidth(), image.getHeight());
158           }
159           throw new UnsupportedOperationException();
160       }
161   
162       public final String getFormula() {
163           return formula;
164       }
