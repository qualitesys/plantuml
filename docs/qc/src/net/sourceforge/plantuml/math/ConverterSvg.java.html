55   public class ConverterSvg {
56   
57       private final Icon icon;
58   
59       public ConverterSvg(Icon icon) {
60           this.icon = icon;
61       }
62   
63       static {
64           try {
65               // DefaultTeXFont.registerAlphabet(new CyrillicRegistration());
66               // DefaultTeXFont.registerAlphabet(new GreekRegistration());
67               final Class<?> clDefaultTeXFont = Class.forName("org.scilab.forge.jlatexmath.DefaultTeXFont");
68               final Class<?> clAlphabetRegistration = Class.forName("org.scilab.forge.jlatexmath.AlphabetRegistration");
69               final Method registerAlphabet = clDefaultTeXFont.getMethod("registerAlphabet", clAlphabetRegistration);
70               registerAlphabet.invoke(null,
71                       Class.forName("org.scilab.forge.jlatexmath.cyrillic.CyrillicRegistration").newInstance());
72               registerAlphabet.invoke(null,
73                       Class.forName("org.scilab.forge.jlatexmath.greek.GreekRegistration").newInstance());
74   
75           } catch (Exception e) {
76               Logme.error(e);
77           }
78   
79       }
80   
81       private Dimension dimension;
82   
83       public String getSvg(double scale, boolean fontAsShapes, Color backgroundColor)
84               throws ClassNotFoundException, IllegalAccessException, IllegalArgumentException, InvocationTargetException,
85               NoSuchMethodException, SecurityException, InstantiationException, IOException {
86           // DOMImplementation domImpl = GenericDOMImplementation.getDOMImplementation();
87           final Class<?> clGenericDOMImplementation = Class.forName("org.apache.batik.dom.GenericDOMImplementation");
88           final DOMImplementation domImpl = (DOMImplementation) clGenericDOMImplementation
89                   .getMethod("getDOMImplementation").invoke(null);
90           final String svgNS = "http://www.w3.org/2000/svg";
91           final Document document = domImpl.createDocument(svgNS, "svg", null);
92   
93           // SVGGeneratorContext ctx = SVGGeneratorContext.createDefault(document);
94           final Class<?> clSVGGeneratorContext = Class.forName("org.apache.batik.svggen.SVGGeneratorContext");
95           final Object ctx = clSVGGeneratorContext.getMethod("createDefault", Document.class).invoke(null, document);
96   
97           // SVGGraphics2D g2 = new SVGGraphics2D(ctx, fontAsShapes);
98           final Class<?> clSVGGraphics2D = Class.forName("org.apache.batik.svggen.SVGGraphics2D");
99           final Graphics g2 = (Graphics) clSVGGraphics2D.getConstructor(clSVGGeneratorContext, boolean.class)
100                   .newInstance(ctx, fontAsShapes);
101   
102           dimension = new Dimension(icon.getIconWidth(), icon.getIconHeight());
103           // g2.setSVGCanvasSize(dimension);
104           g2.getClass().getMethod("setSVGCanvasSize", Dimension.class).invoke(g2, dimension);
105           if (backgroundColor != null) {
106               g2.setColor(backgroundColor);
107               g2.fillRect(0, 0, icon.getIconWidth(), icon.getIconHeight());
108           }
109   
110           icon.paintIcon(null, g2, 0, 0);
111   
112           final Writer out = new CharArrayWriter();
113   
114           final boolean useCSS = true;
115           // g2.stream(out, useCSS);
116           g2.getClass().getMethod("stream", Writer.class, boolean.class).invoke(g2, out, useCSS);
117   
118           out.flush();
119           out.close();
120   
121           return out.toString();
122       }
123   
124       public XDimension2D getDimension() {
125           return XDimension2D.fromDimension(dimension);
126       }

128   }
