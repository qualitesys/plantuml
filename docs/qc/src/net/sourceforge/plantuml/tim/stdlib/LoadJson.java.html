42   import java.util.Set;

60   /**
61    * Loads JSON data from file or URL source.
62    * <p>
63    * Supports three parameters for datasource, default JSON value and charset. The
64    * datasource will be checked against the security rules.
65    * <p>
66    * Examples:<br/>
67    * 
68    * <pre>
69    *     &#64; startuml
70    *     ' loads a local file
71    *     !$JSON_LOCAL_RELATIVE=%load_json("file.json")
72    *
73    *     ' loads a local file from an absolute file path
74    *     !$JSON_LOCAL_ABS=%load_json("c:/loaded/data/file.json")
75    *
76    *     ' tries to load a local file and returns an empty JSON
77    *     !$JSON_LOCAL_REL_EMPTY=%load_json("file-not-existing.json")
78    *
79    *     ' tries to load a local file and returns an default JSON
80    *     !$DEF_JSON={"status":"No data found"}
81    *     !$JSON_LOCAL_REL_DEF=%load_json("file-not-existing.json", $DEF_JSON)
82    *
83    *     ' loads a local file with a specific charset (default is UTF-8)
84    *     !$JSON_LOCAL_RELATIVE_CHARSET=%load_json("file.json", "{}", "iso-8859-1")
85    *
86    *     ' loads a remote JSON from an endpoint (and default, if not reachable)
87    *     !$STATUS_NO_CONNECTION={"status": "No connection"}
88    *     !$JSON_REMOTE_DEF=%load_json("https://localhost:7778/management/health", $STATUS_NO_CONNECTION)
89    *     status -> $JSON_REMOTE_DEF.status
90    *     &#64; enduml
91    * </pre>
92    * 
93    * @author Aljoscha Rittner
94    */
95   public class LoadJson extends SimpleReturnFunction {
96   
97       private static final String VALUE_CHARSET_DEFAULT = "UTF-8";
98   
99       private static final String VALUE_DEFAULT_DEFAULT = "{}";
100   
101       public TFunctionSignature getSignature() {
102           return new TFunctionSignature("%load_json", 3);
103       }
104   
105       public boolean canCover(int nbArg, Set<String> namedArgument) {
106           return nbArg == 1 || nbArg == 2 || nbArg == 3;
107       }
108   
109       public TValue executeReturnFunction(TContext context, TMemory memory, LineLocation location, List<TValue> values,
110               Map<String, TValue> named) throws EaterException, EaterExceptionLocated {
111           final String path = values.get(0).toString();
112           try {
113               String data = loadStringData(path, getCharset(values));
114               if (data == null)
115                   data = getDefaultJson(values);
116   
117               JsonValue jsonValue = Json.parse(data);
118               return TValue.fromJson(jsonValue);
119           } catch (ParseException pe) {
120               Logme.error(pe);
121               throw EaterException.unlocated("JSON parse issue in source " + path + " on location " + pe.getLocation());
122           } catch (UnsupportedEncodingException e) {
123               Logme.error(e);
124               throw EaterException.unlocated("JSON encoding issue in source " + path + ": " + e.getMessage());
125           }
126       }
127   
128       /**
129        * Returns the JSON default, if the data source contains no data.
130        * 
131        * @param values value parameters
132        * @return the defined default JSON or {@code "{}"}
133        */
134       private String getDefaultJson(List<TValue> values) {
135           if (values.size() > 1)
136               return values.get(1).toString();
137   
138           return VALUE_DEFAULT_DEFAULT;
139       }
140   
141       /**
142        * Returns the charset name (if set)
143        * 
144        * @param values value parameters
145        * @return defined charset or {@code "UTF-8"}
146        */
147       private String getCharset(List<TValue> values) {
148           if (values.size() == 3)
149               return values.get(2).toString();
150   
151           return VALUE_CHARSET_DEFAULT;
152       }
153   
154       /**
155        * Loads String data from a data source {@code path} (file or URL) and expects
156        * the data encoded in {@code charset}.
157        * 
158        * @param path    path to data source (http(s)-URL or file).
159        * @param charset character set to encode the string data
160        * @return the decoded String from the data source
161        * @throws EaterException if something went wrong on reading data
162        */
163       private String loadStringData(String path, String charset) throws EaterException, UnsupportedEncodingException {
164   
165           byte[] byteData = null;
166           if (path.startsWith("http://") || path.startsWith("https://")) {
167               final SURL url = SURL.create(path);
168               if (url == null)
169                   throw EaterException.located("load JSON: Invalid URL " + path);
170               byteData = url.getBytes();
171           } else {
172               try {
173                   final SFile file = FileSystem.getInstance().getFile(path);
174                   if (file != null && file.exists() && file.canRead() && !file.isDirectory()) {
175                       final ByteArrayOutputStream out = new ByteArrayOutputStream(1024 * 8);
176                       FileUtils.copyToStream(file, out);
177                       byteData = out.toByteArray();
178                   }
179               } catch (IOException e) {
180                   Logme.error(e);
181                   throw EaterException.located("load JSON: Cannot read file " + path + ". " + e.getMessage());
182               }
183           }
184   
185           if (byteData == null || byteData.length == 0)
186               return null; // no length, no data (we want the default)
187   
188           return new String(byteData, charset);
189   
190       }
191   }
