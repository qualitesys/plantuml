308       private TValue executeOneLineNotSafe(TMemory memory, StringLocated s, TFunctionType ftype, boolean modeSpecial)
309               throws EaterException, EaterExceptionLocated {
310           final TLineType type = s.getType();
311   
312           if (type == TLineType.INCLUDESUB) {
313               this.executeIncludesub(memory, s);
314               return null;
315           } else if (type == TLineType.THEME) {
316               this.executeTheme(memory, s);
317               return null;
318           } else if (type == TLineType.INCLUDE) {
319               this.executeInclude(memory, s);
320               return null;
321           } else if (type == TLineType.INCLUDE_DEF) {
322               this.executeIncludeDef(memory, s);
323               return null;
324           } else if (type == TLineType.IMPORT) {
325               this.executeImport(memory, s);
326               return null;
327           }
328           if (type == TLineType.DUMP_MEMORY) {
329               this.executeDumpMemory(memory, s.getTrimmed());
330               return null;
331           } else if (type == TLineType.ASSERT) {
332               this.executeAssert(memory, s.getTrimmed());
333               return null;
334           } else if (type == TLineType.UNDEF) {
335               this.executeUndef(memory, s);
336               return null;
337           } else if (ftype != TFunctionType.RETURN_FUNCTION && type == TLineType.PLAIN) {
338               this.addPlain(memory, s);
339               return null;
340           } else if (ftype == TFunctionType.RETURN_FUNCTION && type == TLineType.RETURN) {
341               if (modeSpecial) {
342                   final EaterReturn eaterReturn = new EaterReturn(s);
343                   eaterReturn.analyze(this, memory);
344                   final TValue result = eaterReturn.getValue2();
345                   return result;
346               }
347               // Actually, ignore because we are in a if
348               return null;
349           } else if (ftype == TFunctionType.RETURN_FUNCTION && type == TLineType.PLAIN) {
350               this.simulatePlain(memory, s);
351               return null;
352           } else if (type == TLineType.AFFECTATION_DEFINE) {
353               this.executeAffectationDefine(memory, s);
354               return null;
355           } else if (ftype == null && type == TLineType.END_FUNCTION) {
356               CommandExecutionResult.error("error endfunc");
357               return null;
358           } else if (type == TLineType.LOG) {
359               this.executeLog(memory, s);
360               return null;
361           } else if (s.getString().matches("^\\s+$")) {
362               return null;
363           } else {
364               throw EaterException.located("Compile Error " + ftype + " " + type);
365           }
366       }

405       private StringLocated[] applyFunctionsAndVariablesInternal(TMemory memory, StringLocated located)
406               throws EaterException, EaterExceptionLocated {
407           if (memory.isEmpty() && functionsSet.size() == 0)
408               return new StringLocated[] { located };
409   
410           final String result = applyFunctionsAndVariables(memory, located.getLocation(), located.getString());
411           if (result == null)
412               return null;
413   
414           final String[] splited = result.split("\n");
415           final StringLocated[] tab = new StringLocated[splited.length];
416           for (int i = 0; i < splited.length; i++)
417               tab[i] = new StringLocated(splited[i], located.getLocation());
418   
419           return tab;
420       }

424       public String applyFunctionsAndVariables(TMemory memory, LineLocation location, final String str)
425               throws EaterException, EaterExceptionLocated {
426           // https://en.wikipedia.org/wiki/Boyer%E2%80%93Moore%E2%80%93Horspool_algorithm
427           // https://stackoverflow.com/questions/1326682/java-replacing-multiple-different-substring-in-a-string-at-once-or-in-the-most
428           // https://en.wikipedia.org/wiki/String-searching_algorithm
429           // https://www.quora.com/What-is-the-most-efficient-algorithm-to-replace-all-occurrences-of-a-pattern-P-in-a-string-with-a-pattern-P
430           // https://en.wikipedia.org/wiki/Trie
431           if (memory.isEmpty() && functionsSet.size() == 0) {
432               return str;
433           }
434           final StringBuilder result = new StringBuilder();
435           for (int i = 0; i < str.length(); i++) {
436               final char c = str.charAt(i);
437               final String presentFunction = getFunctionNameAt(str, i);
438               if (presentFunction != null) {
439                   final String sub = str.substring(i);

441                           isLegacyDefine(presentFunction), isUnquoted(presentFunction));

443                   final TFunctionSignature signature = new TFunctionSignature(presentFunction, call.getValues().size(),
444                           call.getNamedArguments().keySet());

446                   if (function == null)

448   
449                   if (function.getFunctionType() == TFunctionType.PROCEDURE) {

455                           appendToLastResult(remaining);
456

460                       this.pendingAdd = str.substring(0, i);
461                       executeVoid3(location, memory, sub, function, call);
462                       return null;
463                   }
464                   assert function.getFunctionType() == TFunctionType.RETURN_FUNCTION
465                           || function.getFunctionType() == TFunctionType.LEGACY_DEFINE;
466                   final TValue functionReturn = function.executeReturnFunction(this, memory, location, call.getValues(),
467                           call.getNamedArguments());
468                   result.append(functionReturn.toString());
469                   i += call.getCurrentPosition() - 1;
470               } else if (new VariableManager(this, memory, location).getVarnameAt(str, i) != null) {
471                   i = new VariableManager(this, memory, location).replaceVariables(str, i, result);
472               } else {
473                   result.append(c);
474               }
475           }
476           return result.toString();
477       }
