59       public TFunctionImpl(String functionName, List<TFunctionArgument> args, boolean unquoted,
60               TFunctionType functionType) {
61           final Set<String> names = new HashSet<>();
62           for (TFunctionArgument tmp : args) {
63               names.add(tmp.getName());
64           }
65           this.signature = new TFunctionSignature(functionName, args.size(), names);
66           this.args = args;
67           this.unquoted = unquoted;
68           this.functionType = functionType;
69       }

71       public boolean canCover(int nbArg, Set<String> namedArguments) {
72           for (String n : namedArguments) {
73               if (signature.getNamedArguments().contains(n) == false) {
74                   return false;
75               }
76           }
77           if (nbArg > args.size()) {
78               return false;
79           }
80           assert nbArg <= args.size();
81           int neededArgument = 0;
82           for (TFunctionArgument arg : args) {
83               if (namedArguments.contains(arg.getName())) {
84                   continue;
85               }
86               if (arg.getOptionalDefaultValue() == null) {
87                   neededArgument++;
88               }
89           }
90           if (nbArg < neededArgument) {
91               return false;
92           }
93           assert nbArg >= neededArgument;
94           return true;
95       }

97       private TMemory getNewMemory(TMemory memory, List<TValue> values, Map<String, TValue> namedArguments) {
98           final Map<String, TValue> result = new HashMap<String, TValue>();
99           int ivalue = 0;
100           for (TFunctionArgument arg : args) {
101               final TValue value;
102               if (namedArguments.containsKey(arg.getName())) {
103                   value = namedArguments.get(arg.getName());
104               } else if (ivalue < values.size()) {
105                   value = values.get(ivalue);
106                   ivalue++;
107               } else {
108                   value = arg.getOptionalDefaultValue();
109               }
110               if (value == null) {
111                   throw new IllegalStateException();
112               }
113               result.put(arg.getName(), value);
114           }
115           return memory.forkFromGlobal(result);
116       }

123       public void addBody(StringLocated s) throws EaterExceptionLocated {
124           body.add(s);
125           if (s.getType() == TLineType.RETURN) {
126               this.containsReturn = true;
127               if (functionType == TFunctionType.PROCEDURE) {
128                   throw EaterExceptionLocated
129                           .located("A procedure cannot have !return directive. Declare it as a function instead ?", s);
130                   // this.functionType = TFunctionType.RETURN;
131               }
132           }
133       }

135       public void executeProcedure(TContext context, TMemory memory, LineLocation location, String s)
136               throws EaterException, EaterExceptionLocated {
137           final EaterFunctionCall call = new EaterFunctionCall(new StringLocated(s, location),
138                   context.isLegacyDefine(signature.getFunctionName()), unquoted);
139           call.analyze(context, memory);
140           final String endOfLine = call.getEndOfLine();
141           final List<TValue> args = call.getValues();
142           final Map<String, TValue> named = call.getNamedArguments();
143           executeProcedureInternal(context, memory, args, named);
144           context.appendEndOfLine(endOfLine);
145       }

147       public void executeProcedureInternal(TContext context, TMemory memory, List<TValue> args, Map<String, TValue> named)
148               throws EaterException, EaterExceptionLocated {
149           if (functionType != TFunctionType.PROCEDURE && functionType != TFunctionType.LEGACY_DEFINELONG) {
150               throw new IllegalStateException();
151           }
152           final TMemory copy = getNewMemory(memory, args, named);
153           context.executeLines(copy, body, TFunctionType.PROCEDURE, false);
154       }

156       public TValue executeReturnFunction(TContext context, TMemory memory, LineLocation location, List<TValue> args,
157               Map<String, TValue> named) throws EaterException, EaterExceptionLocated {
158           if (functionType == TFunctionType.LEGACY_DEFINE) {
159               return executeReturnLegacyDefine(location, context, memory, args);
160           }
161           if (functionType != TFunctionType.RETURN_FUNCTION) {
162               throw EaterException.unlocated("Illegal call here. Is there a return directive in your function?");
163           }
164           final TMemory copy = getNewMemory(memory, args, named);
165           final TValue result = context.executeLines(copy, body, TFunctionType.RETURN_FUNCTION, true);
166           if (result == null) {
167               throw EaterException.unlocated("No return directive found in your function");
168           }
169           return result;
170       }

172       private TValue executeReturnLegacyDefine(LineLocation location, TContext context, TMemory memory, List<TValue> args)
173               throws EaterException, EaterExceptionLocated {
174           if (legacyDefinition == null) {
175               throw new IllegalStateException();
176           }
177           final TMemory copy = getNewMemory(memory, args, Collections.<String, TValue>emptyMap());
178           final String tmp = context.applyFunctionsAndVariables(copy, location, legacyDefinition);
179           if (tmp == null) {
180               return TValue.fromString("");
181           }
182           return TValue.fromString(tmp);
183           // eaterReturn.execute(context, copy);
184           // // System.err.println("s3=" + eaterReturn.getValue2());
185           // return eaterReturn.getValue2();
186       }
