80       final public TValue eatExpression(TContext context, TMemory memory) throws EaterException, EaterExceptionLocated {
81           if (peekChar() == '{') {
82               String data = eatAllToEnd();
83               // System.err.println("data=" + data);
84               final JsonValue json = Json.parse(data);
85               // System.err.println("json=" + json);
86               return TValue.fromJson(json);
87           }
88           final TokenStack tokenStack = eatTokenStack();
89           return tokenStack.getResult(getLineLocation(), context, memory);
90       }

92       final protected TokenStack eatTokenStack() throws EaterException {
93           final TokenStack tokenStack = new TokenStack();
94           addIntoTokenStack(tokenStack, false);
95           if (tokenStack.size() == 0) {
96               throw EaterException.located("Missing expression");
97           }
98           return tokenStack;
99       }

119       final public String eatAndGetQuotedString() throws EaterException {
120           final char separator = peekChar();
121           if (TLineType.isQuote(separator) == false) {
122               throw EaterException.located("quote10");
123           }
124           checkAndEatChar(separator);
125           final StringBuilder value = new StringBuilder();
126           addUpTo(separator, value);
127           checkAndEatChar(separator);
128           return value.toString();
129       }

306       final protected TFunctionImpl eatDeclareFunction(TContext context, TMemory memory, boolean unquoted,
307               LineLocation location, boolean allowNoParenthesis, TFunctionType type)
308               throws EaterException, EaterExceptionLocated {
309           final List<TFunctionArgument> args = new ArrayList<>();
310           final String functionName = eatAndGetFunctionName();
311           skipSpaces();
312           if (safeCheckAndEatChar('(') == false) {
313               if (allowNoParenthesis) {
314                   return new TFunctionImpl(functionName, args, unquoted, type);
315               }
316               throw EaterException.located("Missing opening parenthesis");
317           }
318           while (true) {
319               skipSpaces();
320               char ch = peekChar();
321               if (TLineType.isLetterOrUnderscoreOrDollar(ch)) {
322                   final String varname = eatAndGetVarname();
323                   skipSpaces();
324                   final TValue defValue;
325                   if (peekChar() == '=') {
326                       eatOneChar();
327                       final TokenStack def = TokenStack.eatUntilCloseParenthesisOrComma(this);
328                       def.guessFunctions();
329                       defValue = def.getResult(getLineLocation(), context, memory);
330                       // System.err.println("result=" + defValue);
331                   } else {
332                       defValue = null;
333                   }
334                   args.add(new TFunctionArgument(varname, defValue));
335               } else if (ch == ',') {
336                   checkAndEatChar(',');
337               } else if (ch == ')') {
338                   checkAndEatChar(")");
339                   break;
340               } else {
341                   throw EaterException.located("Error in function definition");
342               }
343           }
344           skipSpaces();
345           return new TFunctionImpl(functionName, args, unquoted, type);
346       }

348       final protected TFunctionImpl eatDeclareReturnFunctionWithOptionalReturn(TContext context, TMemory memory,
349               boolean unquoted, LineLocation location) throws EaterException, EaterExceptionLocated {
350           final TFunctionImpl result = eatDeclareFunction(context, memory, unquoted, location, false,
351                   TFunctionType.RETURN_FUNCTION);
352           if (peekChar() == 'r') {
353               checkAndEatChar("return");
354               skipSpaces();
355               final String line = "!return " + eatAllToEnd();
356               result.addBody(new StringLocated(line, location));
357           } else if (peekChar() == '!') {
358               checkAndEatChar("!return");
359               skipSpaces();
360               final String line = "!return " + eatAllToEnd();
361               result.addBody(new StringLocated(line, location));
362           }
363           return result;
364       }
