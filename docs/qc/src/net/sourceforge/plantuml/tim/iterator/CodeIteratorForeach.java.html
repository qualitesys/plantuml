36   
37   import java.util.List;
38   
39   import net.sourceforge.plantuml.StringLocated;
40   import net.sourceforge.plantuml.json.JsonValue;
41   import net.sourceforge.plantuml.tim.EaterException;
42   import net.sourceforge.plantuml.tim.EaterExceptionLocated;
43   import net.sourceforge.plantuml.tim.EaterForeach;
44   import net.sourceforge.plantuml.tim.ExecutionContextForeach;
45   import net.sourceforge.plantuml.tim.TContext;
46   import net.sourceforge.plantuml.tim.TLineType;
47   import net.sourceforge.plantuml.tim.TMemory;
48   import net.sourceforge.plantuml.tim.TVariableScope;
49   import net.sourceforge.plantuml.tim.expression.TValue;
50   
51   public class CodeIteratorForeach extends AbstractCodeIterator {
52   
53       private final TContext context;
54       private final TMemory memory;
55       private final List<StringLocated> logs;
56   
57       public CodeIteratorForeach(CodeIterator source, TContext context, TMemory memory, List<StringLocated> logs) {
58           super(source);
59           this.context = context;
60           this.memory = memory;
61           this.logs = logs;
62       }
63   
64       public StringLocated peek() throws EaterException, EaterExceptionLocated {
65           int level = 0;
66           while (true) {
67               final StringLocated result = source.peek();
68               if (result == null) {
69                   return null;
70               }
71   
72               final ExecutionContextForeach foreach = memory.peekForeach();
73               if (foreach != null && foreach.isSkipMe()) {
74                   if (result.getType() == TLineType.FOREACH) {
75                       level++;
76                   } else if (result.getType() == TLineType.ENDFOREACH) {
77                       level--;
78                       if (level == -1) {
79                           memory.pollForeach();
80                           level = 0;
81                       }
82                   }
83                   next();
84                   continue;
85               }
86   
87               if (result.getType() == TLineType.FOREACH) {
88                   logs.add(result);
89                   executeForeach(memory, result.getTrimmed());
90                   next();
91                   continue;
92               } else if (result.getType() == TLineType.ENDFOREACH) {
93                   logs.add(result);
94                   if (foreach == null) {
95                       throw EaterException.located("No foreach related to this endforeach");
96                   }
97                   foreach.inc();
98                   if (foreach.isSkipMe()) {
99                       memory.pollForeach();
100                   } else {
101                       setLoopVariable(memory, foreach, result);
102                       source.jumpToCodePosition(foreach.getStartForeach());
103                   }
104                   next();
105                   continue;
106               }
107   
108               return result;
109           }
110       }
111   
112       private void executeForeach(TMemory memory, StringLocated s) throws EaterException, EaterExceptionLocated {
113           final EaterForeach condition = new EaterForeach(s);
114           condition.analyze(context, memory);
115           final ExecutionContextForeach foreach = ExecutionContextForeach.fromValue(condition.getVarname(),
116                   condition.getJsonArray(), source.getCodePosition());
117           if (condition.isSkip()) {
118               foreach.skipMeNow();
119           } else {
120               setLoopVariable(memory, foreach, s);
121           }
122           memory.addForeach(foreach);
123       }
124   
125       private void setLoopVariable(TMemory memory, ExecutionContextForeach foreach, StringLocated position)
126               throws EaterException {
127           final JsonValue first = foreach.getJsonArray().get(foreach.currentIndex());
128           memory.putVariable(foreach.getVarname(), TValue.fromJson(first), TVariableScope.GLOBAL);
129       }
130   
131   }
