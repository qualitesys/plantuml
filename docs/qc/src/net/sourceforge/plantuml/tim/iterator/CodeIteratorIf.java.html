35   package net.sourceforge.plantuml.tim.iterator;
36   
37   import java.util.List;
38   
39   import net.sourceforge.plantuml.StringLocated;
40   import net.sourceforge.plantuml.tim.EaterElseIf;
41   import net.sourceforge.plantuml.tim.EaterException;
42   import net.sourceforge.plantuml.tim.EaterExceptionLocated;
43   import net.sourceforge.plantuml.tim.EaterIf;
44   import net.sourceforge.plantuml.tim.EaterIfdef;
45   import net.sourceforge.plantuml.tim.EaterIfndef;
46   import net.sourceforge.plantuml.tim.ExecutionContextIf;
47   import net.sourceforge.plantuml.tim.TContext;
48   import net.sourceforge.plantuml.tim.TLineType;
49   import net.sourceforge.plantuml.tim.TMemory;

51   public class CodeIteratorIf extends AbstractCodeIterator {
52   
53       private final TContext context;
54       private final TMemory memory;
55       private final List<StringLocated> logs;
56   
57       public CodeIteratorIf(CodeIterator source, TContext context, TMemory memory, List<StringLocated> logs) {
58           super(source);
59           this.context = context;
60           this.memory = memory;
61           this.logs = logs;
62       }
63   
64       public StringLocated peek() throws EaterException, EaterExceptionLocated {
65           while (true) {
66               final StringLocated result = source.peek();
67               if (result == null) {
68                   return null;
69               }
70               if (result.getType() == TLineType.IF) {
71                   logs.add(result);
72                   executeIf(context, memory, result.getTrimmed());
73                   next();
74                   continue;
75               } else if (result.getType() == TLineType.IFDEF) {
76                   logs.add(result);
77                   executeIfdef(context, memory, result.getTrimmed());
78                   next();
79                   continue;
80               } else if (result.getType() == TLineType.IFNDEF) {
81                   logs.add(result);
82                   executeIfndef(context, memory, result.getTrimmed());
83                   next();
84                   continue;
85               } else if (result.getType() == TLineType.ELSE) {
86                   logs.add(result);
87                   executeElse(context, memory, result.getTrimmed());
88                   next();
89                   continue;
90               } else if (result.getType() == TLineType.ELSEIF) {
91                   logs.add(result);
92                   executeElseIf(context, memory, result.getTrimmed());
93                   next();
94                   continue;
95               } else if (result.getType() == TLineType.ENDIF) {
96                   logs.add(result);
97                   executeEndif(context, memory, result.getTrimmed());
98                   next();
99                   continue;
100               } else if (memory.peekIf() != null && (memory.areAllIfOk(context, memory) == false)) {
101                   logs.add(result);
102                   next();
103                   continue;
104               }
105   
106               return result;
107           }
108       }
109   
110       private void executeIf(TContext context, TMemory memory, StringLocated s) throws EaterException, EaterExceptionLocated {
111           final EaterIf condition = new EaterIf(s);
112           condition.analyze(context, memory);
113           final boolean isTrue = condition.isTrue();
114           memory.addIf(ExecutionContextIf.fromValue(isTrue));
115       }
116   
117       private void executeElseIf(TContext context, TMemory memory, StringLocated s) throws EaterException, EaterExceptionLocated {
118           final ExecutionContextIf poll = (ExecutionContextIf) memory.peekIf();
119           if (poll == null) {
120               throw EaterException.located("No if related to this else");
121           }
122   
123           poll.enteringElseIf();
124           if (poll.hasBeenBurn() == false) {
125               final EaterElseIf condition = new EaterElseIf(s);
126               condition.analyze(context, memory);
127               final boolean isTrue = condition.isTrue();
128               if (isTrue) {
129                   poll.nowInSomeElseIf();
130               }
131           }
132       }
133   
134       private void executeIfdef(TContext context, TMemory memory, StringLocated s) throws EaterException {
135           final EaterIfdef condition = new EaterIfdef(s);
136           condition.analyze(context, memory);
137           final boolean isTrue = condition.isTrue(context, memory);
138           memory.addIf(ExecutionContextIf.fromValue(isTrue));
139       }
140   
141       private void executeIfndef(TContext context, TMemory memory, StringLocated s) throws EaterException {
142           final EaterIfndef condition = new EaterIfndef(s);
143           condition.analyze(context, memory);
144           final boolean isTrue = condition.isTrue(context, memory);
145           memory.addIf(ExecutionContextIf.fromValue(isTrue));
146       }
147   
148       private void executeElse(TContext context, TMemory memory, StringLocated s) throws EaterException {
149           final ExecutionContextIf poll = (ExecutionContextIf) memory.peekIf();
150           if (poll == null) {
151               throw EaterException.located("No if related to this else");
152           }
153           poll.nowInElse();
154       }
155   
156       private void executeEndif(TContext context, TMemory memory, StringLocated s) throws EaterException {
157           final ExecutionContextIf poll = (ExecutionContextIf) memory.pollIf();
158           if (poll == null) {
159               throw EaterException.located("No if related to this endif");
160           }
161       }
162   
163   }
