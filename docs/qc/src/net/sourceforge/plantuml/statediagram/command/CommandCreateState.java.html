71       private static IRegex getRegexConcat() {
72           return RegexConcat.build(CommandCreateState.class.getName(), RegexLeaf.start(), //
73                   new RegexLeaf("state"), //
74                   RegexLeaf.spaceOneOrMore(), //
75   
76                   new RegexOr(//
77                           new RegexConcat(//
78                                   new RegexLeaf("CODE1", "([%pLN_.]+)"), //
79                                   RegexLeaf.spaceOneOrMore(), new RegexLeaf("as"), RegexLeaf.spaceOneOrMore(), //
80                                   new RegexLeaf("DISPLAY1", "[%g]([^%g]+)[%g]")), //
81                           new RegexConcat(//
82                                   new RegexLeaf("DISPLAY2", "[%g]([^%g]+)[%g]"), //
83                                   RegexLeaf.spaceOneOrMore(), new RegexLeaf("as"), RegexLeaf.spaceOneOrMore(), //
84                                   new RegexLeaf("CODE2", "([%pLN_.]+)")), //
85                           new RegexLeaf("CODE3", "([%pLN_.]+)"), //
86                           new RegexLeaf("CODE4", "[%g]([^%g]+)[%g]")), //
87                   RegexLeaf.spaceZeroOrMore(), //
88                   new RegexLeaf("TAGS1", Stereotag.pattern() + "?"), //
89                   RegexLeaf.spaceZeroOrMore(), //
90                   new RegexLeaf("STEREOTYPE", "(\\<\\<.*\\>\\>)?"), //
91                   RegexLeaf.spaceZeroOrMore(), //
92                   new RegexLeaf("TAGS2", Stereotag.pattern() + "?"), //
93                   RegexLeaf.spaceZeroOrMore(), //
94                   new RegexLeaf("URL", "(" + UrlBuilder.getRegexp() + ")?"), //

108       private static ColorParser color() {
109           return ColorParser.simpleColor(ColorType.BACK);
110       }
111   
112       @Override
113       protected CommandExecutionResult executeArg(StateDiagram diagram, LineLocation location, RegexResult arg)
114               throws NoSuchColorException {
115           final String idShort = arg.getLazzy("CODE", 0);
116           final Ident ident = diagram.buildLeafIdent(idShort);
117           final Code code = diagram.V1972() ? ident : diagram.buildCode(idShort);
118           String display = arg.getLazzy("DISPLAY", 0);
119           if (display == null) 
120               display = code.getName();
121           
122           final String stereotype = arg.get("STEREOTYPE", 0);
123           final LeafType type = getTypeFromStereotype(stereotype);
124           if (diagram.checkConcurrentStateOk(ident, code) == false) 
125               return CommandExecutionResult.error("The state " + code.getName()
126                       + " has been created in a concurrent state : it cannot be used here.");
127           
128           final IEntity ent = diagram.getOrCreateLeaf(diagram.buildLeafIdent(idShort), code, type, null);
129           ent.setDisplay(Display.getWithNewlines(display));
130   
131           if (stereotype != null) 
132               ent.setStereotype(Stereotype.build(stereotype));
133           
134           final String urlString = arg.get("URL", 0);
135           if (urlString != null) {
136               final UrlBuilder urlBuilder = new UrlBuilder(diagram.getSkinParam().getValue("topurl"), UrlMode.STRICT);
137               final Url url = urlBuilder.getUrl(urlString);
138               ent.addUrl(url);
139           }
140   
141           Colors colors = color().getColor(arg, diagram.getSkinParam().getIHtmlColorSet());
142           final String s = arg.get("LINECOLOR", 1);
143   
144           final HColor lineColor = s == null ? null
145                   : diagram.getSkinParam().getIHtmlColorSet().getColor(s);
146           if (lineColor != null) 
147               colors = colors.add(ColorType.LINE, lineColor);
148           
149           if (arg.get("LINECOLOR", 0) != null) 
150               colors = colors.addLegacyStroke(arg.get("LINECOLOR", 0));
151           
152           ent.setColors(colors);
153   
154           // ent.setSpecificColorTOBEREMOVED(ColorType.BACK,
155           // diagram.getSkinParam().getIHtmlColorSet().getColorIfValid(arg.get("COLOR",
156           // 0)));
157           // ent.setSpecificColorTOBEREMOVED(ColorType.LINE,
158           // diagram.getSkinParam().getIHtmlColorSet().getColorIfValid(arg.get("LINECOLOR",
159           // 1)));
160           // ent.applyStroke(arg.get("LINECOLOR", 0));
161   
162           final String addFields = arg.get("ADDFIELD", 0);
163           if (addFields != null) 
164               ent.getBodier().addFieldOrMethod(addFields);
165           
166           CommandCreateClassMultilines.addTags(ent, arg.getLazzy("TAGS", 0));
167   
168           return CommandExecutionResult.ok();
169       }
170   
171       private LeafType getTypeFromStereotype(String stereotype) {
172           if ("<<choice>>".equalsIgnoreCase(stereotype)) 
173               return LeafType.STATE_CHOICE;
174           
175           if ("<<fork>>".equalsIgnoreCase(stereotype)) 
176               return LeafType.STATE_FORK_JOIN;
177           
178           if ("<<join>>".equalsIgnoreCase(stereotype)) 
179               return LeafType.STATE_FORK_JOIN;
180           
181           if ("<<start>>".equalsIgnoreCase(stereotype)) 
182               return LeafType.CIRCLE_START;
183           
184           if ("<<end>>".equalsIgnoreCase(stereotype)) 
185               return LeafType.CIRCLE_END;
186           
187           if ("<<history>>".equalsIgnoreCase(stereotype)) 
188               return LeafType.PSEUDO_STATE;
189           
190           if ("<<history*>>".equalsIgnoreCase(stereotype)) 
191               return LeafType.DEEP_HISTORY;
192           
193           return null;
194       }
