66       public boolean checkConcurrentStateOk(Ident ident, Code code) {
67           if (this.V1972()) {
68               return checkConcurrentStateOkInternal1972(ident);
69           }
70           final boolean result = checkConcurrentStateOkInternal(code);
71           // System.err.println("checkConcurrentStateOk " + code + " " + ident + " " +
72           // result);
73           return result;
74       }

76       private boolean checkConcurrentStateOkInternal(Code code) {
77           if (leafExist(code) == false) {
78               return true;
79           }
80           final IEntity existing = this.getLeaf(code);
81           if (getCurrentGroup().getGroupType() == GroupType.CONCURRENT_STATE
82                   && getCurrentGroup() != existing.getParentContainer()) {
83               return false;
84           }
85           if (existing.getParentContainer().getGroupType() == GroupType.CONCURRENT_STATE
86                   && getCurrentGroup() != existing.getParentContainer()) {
87               return false;
88           }
89           return true;
90       }

92       private boolean checkConcurrentStateOkInternal1972(Ident ident) {
93           if (leafExistSmart(ident) == false) {
94               return true;
95           }
96           final IEntity existing = this.getLeafSmart(ident);
97           if (getCurrentGroup().getGroupType() == GroupType.CONCURRENT_STATE
98                   && getCurrentGroup() != existing.getParentContainer()) {
99               return false;
100           }
101           if (existing.getParentContainer().getGroupType() == GroupType.CONCURRENT_STATE
102                   && getCurrentGroup() != existing.getParentContainer()) {
103               return false;
104           }
105           return true;
106       }

108       @Override
109       public IEntity getOrCreateLeaf(Ident ident, Code code, LeafType type, USymbol symbol) {
110           if (checkConcurrentStateOk(Objects.requireNonNull(ident), code) == false) {
111               throw new IllegalStateException("Concurrent State " + code);
112           }
113           if (!this.V1972() && type == null) {
114               if (code.getName().startsWith("[*]")) {
115                   throw new IllegalArgumentException();
116               }
117               if (isGroup(code)) {
118                   return getGroup(code);
119               }
120               return getOrCreateLeafDefault(ident, code, LeafType.STATE, null);
121           }
122           if (this.V1972() && type == null) {
123               if (ident.getName().startsWith("[*]")) {
124                   throw new IllegalArgumentException();
125               }
126               if (isGroupVerySmart(ident)) {
127                   return getGroupVerySmart(ident);
128               }
129               if (getNamespaceSeparator() == null) {
130                   final ILeaf result = getLeafVerySmart(ident);
131                   if (result != null) {
132                       return result;
133                   }
134   
135               }
136               return getOrCreateLeafDefault(ident, code, LeafType.STATE, null);
137           }
138           return getOrCreateLeafDefault(ident, code, type, symbol);
139       }

222       public boolean concurrentState(char direction) {
223           final IGroup cur = getCurrentGroup();
224           // printlink("BEFORE");
225           if (EntityUtils.groupRoot(cur) == false && cur.getGroupType() == GroupType.CONCURRENT_STATE) {
226               super.endGroup();
227           }
228           getCurrentGroup().setConcurrentSeparator(direction);
229           final String tmp1 = this.getUniqueSequence(CONCURRENT_PREFIX);
230           final Ident ident1 = buildLeafIdent(tmp1);
231           final Code code1 = this.V1972() ? ident1 : buildCode(tmp1);
232           gotoGroup(ident1, code1, Display.create(""), GroupType.CONCURRENT_STATE, getCurrentGroup(),
233                   NamespaceStrategy.SINGLE);
234           final IGroup conc1 = getCurrentGroup();
235           if (EntityUtils.groupRoot(cur) == false && cur.getGroupType() == GroupType.STATE) {
236               cur.moveEntitiesTo(conc1);
237               super.endGroup();
238               final String tmp2 = this.getUniqueSequence(CONCURRENT_PREFIX);
239               final Ident ident2 = buildLeafIdent(tmp2);
240               final Code code2 = this.V1972() ? ident2 : buildCode(tmp2);
241               gotoGroup(ident2, code2, Display.create(""), GroupType.CONCURRENT_STATE, getCurrentGroup(),
242                       NamespaceStrategy.SINGLE);
243           }
244           // printlink("AFTER");
245           return true;
246       }

275       @Override
276       public String checkFinalError() {
277           for (Link link : this.getLinks()) {
278               final IGroup parent1 = getGroupParentIfItIsConcurrentState(link.getEntity1());
279               final IGroup parent2 = getGroupParentIfItIsConcurrentState(link.getEntity2());
280               if (isCompatible(parent1, parent2) == false) {
281                   return "State within concurrent state cannot be linked out of this concurrent state (between "
282                           + link.getEntity1().getCodeGetName() + " and " + link.getEntity2().getCodeGetName() + ")";
283               }
284           }
285           return super.checkFinalError();
286       }

288       private static boolean isCompatible(IGroup parent1, IGroup parent2) {
289           if (parent1 == null && parent2 == null) {
290               return true;
291           }
292           if (parent1 != null ^ parent2 != null) {
293               return false;
294           }
295           assert parent1 != null && parent2 != null;
296           return parent1 == parent2;
297       }
298   
299       private static IGroup getGroupParentIfItIsConcurrentState(IEntity ent) {
300           IGroup parent = ent.getParentContainer();
301           while (parent != null) {
302               if (parent.getGroupType() == GroupType.CONCURRENT_STATE) {
303                   return parent;
304               }
305               parent = parent.getParentContainer();
306           }
307           return null;
308   
309       }
