21   class BlockSplitter {
22   
23     static int split(Cookie cookie, byte[] input, int from, int to) {
24       LzStore store = cookie.store1;
25       store.reset();
26       Deflate.greedy(cookie, null, input, from, to, store);
27   
28       int nPoints = splitLz(cookie, store.litLens, store.dists, store.size);
29   
30       int pos = from;
31       char[] dists = store.dists;
32       char[] litLens = store.litLens;
33       int[] points = cookie.splitPoints;
34   
35       points[0] = pos;
36       for (int i = 0, j = 1; j <= nPoints; ++j) {
37         for (int pj = points[j]; i < pj; ++i) {
38           pos += (dists[i] == 0) ? 1 : litLens[i];
39         }
40         points[j] = pos;
41       }
42       return nPoints;
43     }
44   
45     // TODO: May be use some kind of SORTED data-structure for splitPoints?
46     static int splitLz(Cookie cookie, char[] litLens, char[] dists, int llSize) {
47       int[] splitPoints = cookie.splitPoints;
48       int[] splitSize = cookie.splitSize;
49       splitPoints[0] = 0;
50       splitSize[0] = Deflate.calculateBlockSize(cookie, litLens, dists, 0, llSize);
51       splitPoints[1] = llSize;
52       splitSize[1] = -1;
53       int numBlocks = 1;
54       int maxBlocks = cookie.blockSplittingMax;
55   
56       if (llSize < 10) {
57         return numBlocks;
58       }
59   
60       int lStart = 0;
61       int lEnd = llSize;
62       int blockN = 0;
63       while (numBlocks < maxBlocks) {
64         int llPos = findMinimum(cookie, litLens, dists, lStart, lEnd);
65   
66         int splitL = Deflate.calculateBlockSize(cookie, litLens, dists, lStart, llPos);
67         int splitR = Deflate.calculateBlockSize(cookie, litLens, dists, llPos, lEnd);
68   
69         if (splitL + splitR > splitSize[blockN] || llPos == lStart + 1 || llPos == lEnd) {
70           splitSize[blockN] = -1;
71         } else {
72           splitSize[blockN] = splitL;
73           numBlocks++;
74           blockN++;
75           System.arraycopy(splitPoints, blockN, splitPoints, blockN + 1, numBlocks - blockN);
76           System.arraycopy(splitSize, blockN, splitSize, blockN + 1, numBlocks - blockN);
77           splitPoints[blockN] = llPos;
78           splitSize[blockN] = splitR;
79         }
80   
81         int longest = 0;
82         boolean found = false;
83         for (int i = 0; i < numBlocks; i++) {
84           int start = splitPoints[i];
85           int end = splitPoints[i + 1];
86           if ((splitSize[i] != -1) && end - start > longest) {
87             lStart = start;
88             lEnd = end;
89             found = true;
90             longest = end - start;
91             blockN = i;
92           }
93         }
94         if (!found) {
95           break;
96         }
97   
98         if (lEnd - lStart < 10) {
99           break;
100         }
101       }
102   
103       return numBlocks;
104     }
105   
106     private static int findMinimum(Cookie cookie, char[] litLens, char[] dists, int from, int to) {
107       int start = from + 1;
108       int end = to;
109       if (end - start < 1024) {
110         int best = Integer.MAX_VALUE;
111         int result = start;
112         for (int i = start; i < end; i++) {
113           int v = Deflate.calculateBlockSize(cookie, litLens, dists, from, i)
114               + Deflate.calculateBlockSize(cookie, litLens, dists, i, to);
115           if (v < best) {
116             best = v;
117             result = i;
118           }
119         }
120         return result;
121       } else {
122         int n = Cookie.SPLIT_PARTITIONS;
123         int[] p = cookie.p;
124         int[] vp = cookie.vp;
125         int lastBest = Integer.MAX_VALUE;
126         int pos = start;
127   
128         while (true) {
129           if (end - start <= n) {
130             break;
131           }
132   
133           for (int i = 0; i < n; i++) {
134             p[i] = start + (i + 1) * ((end - start) / (n + 1));
135             vp[i] = Deflate.calculateBlockSize(cookie, litLens, dists, from, p[i])
136                 + Deflate.calculateBlockSize(cookie, litLens, dists, p[i], to);
137           }
138           int bestI = 0;
139           int best = vp[0];
140           for (int i = 1; i < n; i++) {
141             if (vp[i] < best) {
142               best = vp[i];
143               bestI = i;
144             }
145           }
146           if (best > lastBest) {
147             break;
148           }
149   
150           start = bestI == 0 ? start : p[bestI - 1];
151           end = bestI == n - 1 ? end : p[bestI + 1];
152   
153           pos = p[bestI];
154           lastBest = best;
155         }
156         return pos;
157       }
158     }
