19   package net.sourceforge.plantuml.zopfli;
20   
21   final class LzStore {
22     final char[] litLens;
23     final char[] dists;
24     int size;
25   
26     LzStore(final int maxBlockSize) {
27       litLens = new char[maxBlockSize];
28       dists = new char[maxBlockSize];
29     }
30   
31     final void append(final char length, final char dist) {
32       litLens[size] = length;
33       dists[size++] = dist;
34     }
35   
36     final void reset() {
37       size = 0;
38     }
39   
40     final void copy(final LzStore source) {
41       size = source.size;
42       System.arraycopy(source.litLens, 0, litLens, 0, size);
43       System.arraycopy(source.dists, 0, dists, 0, size);
44     }
45   }
