21   final class SymbolStats {
22     private final static double INV_LOG_2 = 1.4426950408889 * 0x10000L; /* 1.0 / log(2.0) */
23     private final int[] litLens = new int[288];
24     private final int[] dists = new int[32]; // Why 32? Expect 30.
25     final long[] lLiterals = new long[288];
26     final long[] lLengths = new long[259];
27     final long[] dSymbols = new long[32];
28   
29     void getFreqs(LzStore store) {
30       int[] sLitLens = this.litLens;
31       int[] sDists = this.dists;
32       System.arraycopy(Cookie.intZeroes, 0, sLitLens, 0, 288);
33       System.arraycopy(Cookie.intZeroes, 0, sDists, 0, 32);
34   
35       int size = store.size;
36       char[] litLens = store.litLens;
37       char[] dists = store.dists;
38       int[] lengthSymbol = Util.LENGTH_SYMBOL;
39       int[] cachedDistSymbol = Util.CACHED_DIST_SYMBOL;
40       for (int i = 0; i < size; i++) {
41         int d = dists[i];
42         int l = litLens[i];
43         if (d == 0) {
44           sLitLens[l]++;
45         } else {
46           sLitLens[lengthSymbol[l]]++;
47           sDists[cachedDistSymbol[d]]++;
48         }
49       }
50       sLitLens[256] = 1;
51       calculate();
52     }
53   
54     final void copy(final SymbolStats source) {
55       System.arraycopy(source.litLens, 0, litLens, 0, 288);
56       System.arraycopy(source.dists, 0, dists, 0, 32);
57       System.arraycopy(source.lLiterals, 0, lLiterals, 0, 288);
58       System.arraycopy(source.lLengths, 0, lLengths, 0, 259);
59       System.arraycopy(source.dSymbols, 0, dSymbols, 0, 32);
60     }
61   
62     final void calculate() {
63       calculateLens();
64       calculateDists();
65     }
66   
67     final void calculateLens() {
68       int sum = 0;
69       int[] litLens = this.litLens;
70       for (int i = 0; i < 288; ++i) {
71         sum += litLens[i];
72       }
73       double log2sum = (sum == 0 ? Math.log(288) : Math.log(sum)) * INV_LOG_2;
74       long[] lLiterals = this.lLiterals;
75       for (int i = 0; i < 288; ++i) {
76         if (litLens[i] == 0) {
77           lLiterals[i] = (long)log2sum;
78         } else {
79           lLiterals[i] = (long)(log2sum - Math.log(litLens[i]) * INV_LOG_2);
80         }
81         if (lLiterals[i] < 0) {
82           lLiterals[i] = 0;
83         }
84       }
85       long[] lLengths = this.lLengths;
86       int[] lengthSymbol = Util.LENGTH_SYMBOL;
87       int[] lengthExtraBits = Util.LENGTH_EXTRA_BITS;
88       for (int i = 0; i < 259; ++i) {
89         lLengths[i] = lLiterals[lengthSymbol[i]] + (lengthExtraBits[i] * 0x10000L);
90       }
91     }
92   
93     final void calculateDists() {
94       int sum = 0;
95       int[] dists = this.dists;
96       for (int i = 0; i < 32; ++i) {
97         sum += dists[i];
98       }
99       double log2sum = (sum == 0 ? Math.log(32) : Math.log(sum)) * INV_LOG_2;
100       long[] dSymbols = this.dSymbols;
101       for (int i = 0; i < 32; ++i) {
102         if (dists[i] == 0) {
103           dSymbols[i] = (long)log2sum;
104         } else {
105           dSymbols[i] = (long)(log2sum - Math.log(dists[i]) * INV_LOG_2);
106         }
107         if (dSymbols[i] < 0) {
108           dSymbols[i] = 0;
109         }
110       }
111       for (int i = 4; i < 30; ++i) {
112         dSymbols[i] += 0x10000L * ((i / 2) - 1);
113       }
114     }

116     final void alloy(final SymbolStats ligand) {
117       int[] ligandLitLens = ligand.litLens;
118       for (int i = 0; i < 288; i++) {
119         litLens[i] += ligandLitLens[i] / 2;
120       }
121       litLens[256] = 1;
122   
123       int[] ligandDists = ligand.dists;
124       for (int i = 0; i < 32; i++) {
125         dists[i] += ligandDists[i] / 2;
126       }
127     }

129     final int randomizeFreqs(int z) {
130       int[] data = litLens;
131       int n = data.length;
132       for (int i = 0; i < n; i++) {
133         z = 0x7FFFFFFF & (1103515245 * z + 12345);
134         if ((z >>> 4) % 3 == 0) {
135           z = 0x7FFFFFFF & (1103515245 * z + 12345);
136           int p = z % n;
137           if (data[i] < data[p]) {
138             data[i] = data[p];
139           }
140         }
141       }
142       data[256] = 1;
143   
144       data = dists;
145       n = data.length;
146       for (int i = 0; i < n; i++) {
147         z = 0x7FFFFFFF & (1103515245 * z + 12345);
148         if ((z >>> 4) % 3 == 0) {
149           z = 0x7FFFFFFF & (1103515245 * z + 12345);
150           int p = z % n;
151           if (data[i] < data[p]) {
152             data[i] = data[p];
153           }
154         }
155       }
156   
157       return z;
158     }

160     final long minCost() {
161       long[] lLengths = this.lLengths;
162       long minLengthCost = lLengths[3];
163       for (int i = 4; i < 259; i++) {
164         long c = lLengths[i];
165         if (c < minLengthCost) {
166           minLengthCost = c;
167         }
168       }
169   
170       long[] dSymbols = this.dSymbols;
171       long minDistCost = dSymbols[0];
172       for (int i = 1; i < 30; i++) {
173         long c = dSymbols[i];
174         if (c < minDistCost) {
175           minDistCost = c;
176         }
177       }
178   
179       return minDistCost + minLengthCost;
180     }
