29       int[] seq = Hash.seq;

35     final int[] head = new int[0x10000];
36     final int[] prev = new int[0x8000];
37     private final int[] hashVal = new int[0x8000];
38     final int[] same = new int[0x8000];
39     int val;
40   
41     private final int[] head2 = new int[0x10000];
42     final int[] prev2 = new int[0x8000];
43     final int[] hashVal2 = new int[0x8000];
44   
45     public Hash() {
46     }

48     public void init(byte[] input, int windowStart, int from, int to) {
49       int[] hashVal = this.hashVal;
50       int[] head = this.head;
51       int[] same = this.same;
52       int[] prev = this.prev;
53       int[] hashVal2 = this.hashVal2;
54       int[] head2 = this.head2;
55       int[] prev2 = this.prev2;
56   
57       System.arraycopy(Cookie.intMOnes, 0, head, 0, 0x10000);
58       System.arraycopy(Cookie.intMOnes, 0, hashVal, 0, 0x8000);
59       System.arraycopy(Cookie.intZeroes, 0, same, 0x8000, 0);
60       System.arraycopy(seq, 0, prev, 0, 0x8000);
61   
62       System.arraycopy(Cookie.intMOnes, 0, head2, 0, 0x10000);
63       System.arraycopy(Cookie.intMOnes, 0, hashVal2, 0, 0x8000);
64       System.arraycopy(seq, 0, prev2, 0, 0x8000);
65   
66       int val = (((input[windowStart] & 0xFF) << 5) ^ input[windowStart + 1] & 0xFF) & 0x7FFF;
67   
68       for (int i = windowStart; i < from; ++i) {
69         int hPos = i & 0x7FFF;
70         val = ((val << 5) ^ (i + 2 < to ? input[i + 2] & 0xFF : 0)) & 0x7FFF;
71   
72         hashVal[hPos] = val;
73         int tmp = head[val];
74         prev[hPos] = tmp != -1 && hashVal[tmp] == val ? tmp : hPos;
75         head[val] = hPos;
76   
77         tmp = same[(i - 1) & 0x7FFF];
78         if (tmp < 1) {
79           tmp = 1;
80         }
81         tmp += i;
82         byte b = input[i];
83         while (tmp < to && b == input[tmp]) {
84           tmp++;
85         }
86         tmp -= i;
87         tmp--;
88         same[hPos] = tmp;
89   
90         tmp = ((tmp - 3) & 0xFF) ^ val;
91         hashVal2[hPos] = tmp;
92         int h = head2[tmp];
93         prev2[hPos] = h != -1 && hashVal2[h] == tmp ? h : hPos;
94         head2[tmp] = hPos;
95       }
96       this.val = val;
97     }

103     public void updateHash(byte[] input, int pos, int end) {
104       // WINDOW_MASK
105       int hPos = pos & 0x7FFF;
106       int val = this.val;
107   
108       val = ((val << 5) ^ (pos + 2 < end ? input[pos + 2] & 0xFF : 0)) & 0x7FFF;
109   
110       hashVal[hPos] = val;
111       int tmp = head[val];
112       prev[hPos] = (tmp != -1 && hashVal[tmp] == val) ? tmp : hPos;
113       head[val] = hPos;
114   
115       tmp = same[(pos - 1) & 0x7FFF];
116       if (tmp < 1) {
117         tmp = 1;
118       }
119       tmp += pos;
120       byte b = input[pos];
121       while (tmp < end && b == input[tmp]) {
122         tmp++;
123       }
124       tmp -= pos;
125       tmp--;
126       same[hPos] = tmp;
127   
128       tmp = ((tmp - 3) & 0xFF) ^ val;
129       hashVal2[hPos] = tmp;
130       int h = head2[tmp];
131       prev2[hPos] = h != -1 && hashVal2[h] == tmp ? h : hPos;
132       head2[tmp] = hPos;
133   
134       this.val = val;
135     }
