52     public static void greedy(Cookie cookie, LongestMatchCache lmc, byte[] input, int from, int to, LzStore store) {
53       Hash h = cookie.h;
54       h.init(input, Math.max(from - 0x8000, 0), from, to);
55       int prevLength = 0;
56       int prevMatch = 0;
57       char[] dummySubLen = cookie.c259a;
58       boolean matchAvailable = false;
59   
60       for (int i = from; i < to; i++) {
61         h.updateHash(input, i, to);
62         findLongestMatch(cookie, lmc, from, h, input, i, to, 258, dummySubLen);
63         int len = cookie.lenVal;
64         int dist = cookie.distVal;
65         int lengthScore = dist > 1024 ? len - 1 : len;
66         int prevLengthScore = prevMatch > 1024 ? prevLength - 1 : prevLength;
67   
68         if (matchAvailable) {
69           matchAvailable = false;
70           if (lengthScore > prevLengthScore + 1) {
71             store.append((char) (input[i - 1] & 0xFF), (char) 0);
72             if (lengthScore >= 3 && len < 258) {
73               matchAvailable = true;
74               prevLength = len;
75               prevMatch = dist;
76               continue;
77             }
78           } else {
79             store.append((char) prevLength, (char) prevMatch);
80             for (int j = 2; j < prevLength; j++) {
81               i++;
82               h.updateHash(input, i, to);
83             }
84             continue;
85           }
86         } else if (lengthScore >= 3 && len < 258) {
87           matchAvailable = true;
88           prevLength = len;
89           prevMatch = dist;
90           continue;
91         }
92   
93         if (lengthScore >= 3) {
94           store.append((char) len, (char) dist);
95         } else {
96           len = 1;
97           store.append((char) (input[i] & 0xFF), (char) 0);
98         }
99         for (int j = 1; j < len; j++) {
100           i++;
101           h.updateHash(input, i, to);
102         }
103       }
104     }

106     static void findLongestMatch(Cookie cookie, LongestMatchCache lmc, int blockStart, Hash h, byte[] array,
107         int pos, int size, int limit, char[] subLen) {
108       //# WINDOW_SIZE = 0x8000
109       //# WINDOW_MASK = 0x7FFF
110       //# MIN_MATCH = 3
111       //# MAX_MATCH = 258
112   
113       int offset = pos - blockStart;
114       char[] lmcLength = lmc != null ? lmc.length : null;
115       if (lmc != null && ((lmcLength[offset] == 0 || lmc.dist[offset] != 0))
116           && (limit == 258 || lmcLength[offset] <= limit
117           || subLen != null && lmc.maxCachedSubLen(offset) >= limit)) {
118         if (subLen == null || lmcLength[offset] <= lmc.maxCachedSubLen(offset)) {
119           cookie.lenVal = lmcLength[offset];
120           if (cookie.lenVal > limit) {
121             cookie.lenVal = limit;
122           }
123           if (subLen != null) {
124             lmc.cacheToSubLen(offset, cookie.lenVal, subLen);
125             cookie.distVal = subLen[cookie.lenVal];
126           } else {
127             cookie.distVal = lmc.dist[offset];
128           }
129           return;
130         }
131         limit = lmcLength[offset];
132       }
133   
134       if (size - pos < 3) {
135         cookie.lenVal = 0;
136         cookie.distVal = 0;
137         return;
138       }
139   
140       if (pos + limit > size) {
141         limit = size - pos;
142       }
143   
144       int bestDist = 0;
145       int bestLength = 1;
146       int arrayEnd = pos + limit;
147       int chainCounter = 8192;
148       int[] hPrev = h.prev;
149       int[] hPrev2 = h.prev2;
150       int pp = h.head[h.val];
151       int threshold = h.same[pp];
152       int[] hashVal2 = h.hashVal2;
153       int marker = hashVal2[pp];
154       int p = hPrev[pp];
155       pp -= p;
156       int dist = pp > 0 ? pp : pp + 0x8000;
157   
158       while (dist < 0x8000 && chainCounter > 0) {
159         int scan = pos;
160         int match = pos - dist;
161   
162         if (array[scan + bestLength] == array[match + bestLength]) {
163           int same0 = h.same[pos & 0x7FFF];
164           if (same0 > 2 && array[scan] == array[match]) {
165             int same1 = h.same[match & 0x7FFF];
166             int same = same0 < same1 ? same0 : same1;
167             if (same > limit) {
168               same = limit;
169             }
170             scan += same;
171             match += same;
172           }
173           while (scan != arrayEnd && array[scan] == array[match]) {
174             scan++;
175             match++;
176           }
177           scan -= pos;
178   
179           if (scan > bestLength) {
180             if (subLen != null) {
181               for (int j = bestLength + 1; j <= scan; j++) {
182                 subLen[j] = (char) dist;
183               }
184             }
185             bestDist = dist;
186             bestLength = scan;
187             if (scan >= limit) {
188               break;
189             }
190           }
191         }
192   
193         if (hPrev != hPrev2 && bestLength >= threshold && marker == hashVal2[p]) {
194           hPrev = hPrev2;
195         }
196   
197         pp = p;
198         p = hPrev[p];
199         if (p == pp) {
200           break;
201         }
202         pp -= p;
203         dist += pp > 0 ? pp : 0x8000 + pp;
204   
205         --chainCounter;
206       }
207   
208       if (lmc != null && limit == 258 && subLen != null && lmcLength[offset] != 0 && lmc.dist[offset] == 0) {
209         if (bestLength < 3) {
210           lmc.dist[offset] = 0;
211           lmcLength[offset] = 0;
212         } else {
213           lmc.dist[offset] = (char)bestDist;
214           lmcLength[offset] = (char)bestLength;
215         }
216         lmc.subLenToCache(subLen, offset, bestLength);
217       }
218   
219       cookie.distVal = bestDist;
220       cookie.lenVal = bestLength;
221     }

223     private static void deflatePart(Cookie cookie, Options options, byte[] input, int from, int to, boolean flush,
224         Buffer output) {
225       // assert from != to
226       switch (options.blockSplitting) {
227         case FIRST:
228           deflateSplittingFirst(cookie, options, flush, input, from, to, output);
229           break;
230   
231         case LAST:
232           deflateSplittingLast(cookie, options, flush, input, from, to, output);
233           break;
234   
235         case NONE:
236           deflateDynamicBlock(cookie, options, flush, input, from, to, output);
237           break;
238       }
239     }

241     private static void deflateDynamicBlock(Cookie cookie, Options options, boolean flush, byte[] input,
242         int from, int to, Buffer output) {
243       // assert from != to
244       LongestMatchCache lmc = cookie.lmc;
245       lmc.init(to - from);
246   
247       BlockType type = BlockType.DYNAMIC;
248       LzStore store = Squeeze.optimal(cookie, options.numIterations, lmc, input, from, to);
249   
250       if (store.size < 1000) {
251         LzStore fixedStore = cookie.store1;
252         fixedStore.reset();
253         Squeeze.bestFixedLengths(cookie, lmc, input, from, to, cookie.lengthArray, cookie.costs);
254         Squeeze.optimalRun(cookie, lmc, input, from, to, cookie.lengthArray, fixedStore);
255         int dynCost = calculateBlockSize(cookie, store.litLens, store.dists, 0, store.size);
256         int fixedCost = calculateFixedBlockSize(cookie, fixedStore.litLens,
257             fixedStore.dists, fixedStore.size);
258         if (fixedCost < dynCost) {
259           type = BlockType.FIXED;
260           store = fixedStore;
261         }
262       }
263   
264       addLzBlock(cookie, type, flush, store.litLens, store.dists, 0, store.size, output);
265     }

295     static int calculateBlockSize(Cookie cookie, char[] litLens, char[] dists, int lStart, int lEnd) {
296       int[] llLengths = cookie.i288a;
297       System.arraycopy(Cookie.intZeroes, 0, llLengths, 0, 288);
298       int[] dLengths = cookie.i32a;
299       System.arraycopy(Cookie.intZeroes, 0, dLengths, 0, 32);
300   
301       int result = 3;
302   
303       int[] llCounts = cookie.i288b;
304       System.arraycopy(Cookie.intZeroes, 0, llCounts, 0, 288);
305       int[] dCounts = cookie.i32b;
306       System.arraycopy(Cookie.intZeroes, 0, dCounts, 0, 32);
307   
308       int[] lengthSymbol = Util.LENGTH_SYMBOL;
309       int[] cachedDistSymbol = Util.CACHED_DIST_SYMBOL;
310       int[] lengthExtraBits = Util.LENGTH_EXTRA_BITS;
311       for (int i = lStart; i < lEnd; i++) {
312         int d = dists[i];
313         int l = litLens[i];
314         if (d == 0) {
315           llCounts[l]++;
316         } else {
317           llCounts[lengthSymbol[l]]++;
318           int distSymbol = cachedDistSymbol[d];
319           dCounts[distSymbol]++;
320           result += lengthExtraBits[l];
321           if (distSymbol > 3) {
322             result += (distSymbol / 2) - 1;
323           }
324         }
325       }
326       llCounts[256] = 1;
327   
328   
329       int[] llCountsCopy = cookie.i288c;
330       System.arraycopy(llCounts, 0, llCountsCopy, 0, 288);
331       optimizeHuffmanForRle(cookie, llCountsCopy);
332       Katajainen.lengthLimitedCodeLengths(cookie, llCountsCopy, 15, llLengths);
333   
334       int[] dCountsCopy = cookie.i32c;
335       System.arraycopy(dCounts, 0, dCountsCopy, 0, 32);
336       optimizeHuffmanForRle(cookie, dCountsCopy);
337       Katajainen.lengthLimitedCodeLengths(cookie, dCountsCopy, 15, dLengths);
338       patchDistanceCodesForBuggyDecoders(dLengths);
339   
340       result += simulateAddDynamicTree(cookie, llLengths, dLengths);
341   
342       for (int i = 0; i < 288; ++i) {
343         result += llCounts[i] * llLengths[i];
344       }
345       for (int i = 0; i < 32; ++i) {
346         result += dCounts[i] * dLengths[i];
347       }
348       return result;
349     }

351     private static int calculateFixedBlockSize(Cookie cookie, char[] litLens, char[] dists, int size) {
352       int[] llLengths = cookie.i288a;
353       int[] dLengths = cookie.i32a;
354       getFixedTree(llLengths, dLengths);
355   
356       int result = 3;
357   
358       int[] cachedDistExtraBits = Util.CACHED_DIST_EXTRA_BITS;
359       int[] lengthExtraBits = Util.LENGTH_EXTRA_BITS;
360       int[] lengthSymbol = Util.LENGTH_SYMBOL;
361   
362       for (int i = 0; i < size; i++) {
363         int d = dists[i];
364         int l = litLens[i];
365         if (d == 0) {
366           result += llLengths[l];
367         } else {
368           result += llLengths[lengthSymbol[l]];
369           result += lengthExtraBits[l];
370           result += 5;
371           result += d < 4097 ? cachedDistExtraBits[d] : d < 16385 ? d < 8193 ? 11 : 12 : 13;
372         }
373       }
374       result += llLengths[256];
375   
376       return result;
377     }

429         int size = simulateEncodeTree(cookie, llLengths, dLengths, (i & 1) != 0, (i & 2) != 0, (i & 4) != 0);

432           best = i;
433         }

439     private static void encodeTree(Cookie cookie, int[] llLengths, int[] dLengths,
440         boolean use16, boolean use17, boolean use18, Buffer output) {
441       int hLit = 29;
442       int hDist = 29;
443   
444       while (hLit > 0 && llLengths[256 + hLit] == 0) {
445         hLit--;
446       }
447       while (hDist > 0 && dLengths[hDist] == 0) {
448         hDist--;
449       }
450   
451       int lldTotal = hLit + 258 + hDist;
452       int[] lldLengths = cookie.i320b;
453       System.arraycopy(llLengths, 0, lldLengths, 0, 257 + hLit);
454       System.arraycopy(dLengths, 0, lldLengths, 257 + hLit, hDist + 1);
455   
456       int rleSize = 0;
457       int[] rle = cookie.i320a;
458       int[] rleBits = cookie.i320c;
459   
460       for (int i = 0; i < lldTotal; i++) {
461         int count = 1;
462         int symbol = lldLengths[i];
463         if (use16 || (symbol == 0 && (use17 || use18))) {
464           for (int j = i + 1; j < lldTotal && symbol == lldLengths[j]; j++) {
465             count++;
466           }
467         }
468         i += count - 1;
469   
470         if (symbol == 0 && count > 2) {
471           if (use18) {
472             while (count > 10) {
473               int delta = count > 138 ? 138 : count;
474               rle[rleSize] = 18;
475               rleBits[rleSize++] = delta - 11;

734     private static void optimizeHuffmanForRle(Cookie cookie, int[] counts) {
735       int[] goodForRle = cookie.i289a;
736       int length = counts.length;
737       for (; length >= 0; --length) {
738         if (length == 0) {
739           return;
740         }
741         if (counts[length - 1] != 0) {
742           break;
743         }
744       }
745       System.arraycopy(Cookie.intZeroes, 0, goodForRle, 0, length + 1);
746   
747       int symbol = counts[0];
748       int stride = 0;
749       for (int i = 0; i < length + 1; ++i) {
750         if (i == length || counts[i] != symbol) {
751           if ((symbol == 0 && stride >= 5) || (symbol != 0 && stride >= 7)) {
752             for (int k = 0; k < stride; ++k) {
753               goodForRle[i - k - 1] = 1;
754             }
755           }
756           stride = 1;
757           if (i != length) {
758             symbol = counts[i];
759           }
760         } else {
761           ++stride;
762         }
763       }
764   
765       stride = 0;
766       int limit = counts[0];
767       int sum = 0;
768       for (int i = 0; i < length + 1; ++i) {
769         if ((i == length) || (goodForRle[i] != 0) || (counts[i] - limit >= 4) || (limit - counts[i] >= 4)) {
770           if ((stride >= 4) || ((stride >= 3) && (sum == 0))) {
771             int count = (sum + stride / 2) / stride;
772             if (count < 1) count = 1;
773             if (sum == 0) {
774               count = 0;
775             }
776             for (int k = 0; k < stride; ++k) {
777               counts[i - k - 1] = count;
778             }
779           }
780           stride = 0;
781           sum = 0;
782           if (i < length - 3) {
783             limit = (counts[i] + counts[i + 1] + counts[i + 2] + counts[i + 3] + 2) / 4;
784           } else if (i < length) {
785             limit = counts[i];
786           } else {
787             limit = 0;
788           }
789         }
790         ++stride;
791         if (i != length) {
792           sum += counts[i];
793         }
794       }
795     }
