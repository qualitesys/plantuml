23     static LzStore optimal(Cookie cookie, int numIterations, LongestMatchCache lmc, byte[] input, int from, int to) {
24       LzStore currentStore = cookie.store1;
25       currentStore.reset();
26       LzStore store = cookie.store2;
27       Deflate.greedy(cookie, lmc, input, from, to, currentStore);
28       SymbolStats stats = cookie.stats;
29       SymbolStats bestStats = cookie.bestStats;
30       SymbolStats lastStats = cookie.lastStats;
31       stats.getFreqs(currentStore);
32   
33       char[] lengthArray = cookie.lengthArray;
34       long[] costs = cookie.costs;
35   
36       int cost;
37       int bestCost = Integer.MAX_VALUE;
38       int lastCost = 0;
39       int lastRandomStep = -1;
40   
41       for (int i = 0; i < numIterations; i++) {
42         currentStore.reset();
43         bestLengths(cookie, lmc, from, input, from, to, stats.minCost(), stats, lengthArray, costs);
44         optimalRun(cookie, lmc, input, from, to, lengthArray, currentStore);
45         cost = Deflate.calculateBlockSize(cookie, currentStore.litLens, currentStore.dists, 0, currentStore.size);
46         if (cost < bestCost) {
47           store.copy(currentStore);
48           bestStats.copy(stats);
49           bestCost = cost;
50         }
51         lastStats.copy(stats);
52         stats.getFreqs(currentStore);
53         if (lastRandomStep != -1) {
54           stats.alloy(lastStats);
55           stats.calculate();
56         }
57         if (i > 5 && cost == lastCost) {
58           stats.copy(bestStats);
59           cookie.rnd = stats.randomizeFreqs(cookie.rnd);
60           stats.calculate();
61           lastRandomStep = i;
62         }
63         lastCost = cost;
64       }
65       return store;
66     }

68     static void optimalRun(Cookie cookie, LongestMatchCache lmc, byte[] input, int from, int to,
69         char[] lengthArray, LzStore store) {
70       // assert from != to
71       char[] path = cookie.path;
72       int pathSize = 0;
73       int size = to - from;
74       do {
75         char las = lengthArray[size];
76         path[pathSize++] = las;
77         size -= las;
78       } while (size != 0);
79   
80       int windowStart = Math.max(from - 0x8000, 0);
81       Hash h = cookie.h;
82       h.init(input, windowStart, from, to);
83       int pos = from;
84   
85       do {
86         h.updateHash(input, pos, to);
87         int length = path[--pathSize];
88         if (length >= 3) {
89           Deflate.findLongestMatch(cookie, lmc, from, h, input, pos, to, length, null);
90           store.append((char) length, (char) cookie.distVal);
91         } else {
92           length = 1;
93           store.append((char) (input[pos] & 0xFF), (char) 0);
94         }
95   
96         for (int j = 1; j < length; ++j) {
97           h.updateHash(input, pos + j, to);
98         }
99         pos += length;
100       } while (pathSize != 0);
101     }

103     private static long fixedCost(int litLen, int dist) {
104       if (dist == 0) {
105         if (litLen <= 143) {
106           return 8;
107         }
108         return 9;
109       } else {
110         long cost = 12 + (dist < 4097 ? Util.CACHED_DIST_EXTRA_BITS[dist] : dist < 16385 ? dist < 8193 ? 11 : 12 : 13)
111             + Util.LENGTH_EXTRA_BITS[litLen];
112         if (Util.LENGTH_SYMBOL[litLen] > 279) {
113           return cost + 1;
114         }
115         return cost;
116       }
117     }

119     private static void bestLengths(Cookie cookie, LongestMatchCache lmc, int blockStart, byte[] input, int from, int to,
120         long minCost, SymbolStats stats, char[] lengthArray, long[] costs) {
121       //# WINDOW_SIZE = 0x8000
122       //# WINDOW_MASK = 0x7FFF
123       //# MAX_MATCH = 258
124   
125       int windowStart = Math.max(from - 0x8000, 0);
126       Hash h = cookie.h;
127       h.init(input, windowStart, from, to);
128       Cookie.fillCostMax(costs, to - from + 1);
129       costs[0] = 0L;
130       lengthArray[0] = 0;
131       int[] same = h.same;
132   
133       char[] subLen = cookie.c259a;
134       System.arraycopy(Cookie.charZeroes, 0, subLen, 0, 259);
135   
136       long[] slLiterals = stats.lLiterals;
137       long[] slLengths = stats.lLengths;
138       long[] sdSymbols = stats.dSymbols;
139       long stepCost = slLengths[258] + sdSymbols[0];
140   
141       int[] cachedDistSymbol = Util.CACHED_DIST_SYMBOL;
142   
143       int i = from;
144       int j = 0;
145       while (i < to) {
146         h.updateHash(input, i, to);
147   
148         if (same[i & 0x7FFF] > 516 && i > from + 259 && i + 517 < to && same[(i - 258) & 0x7FFF] > 258) {
149           for (int k = 0; k < 258; ++k) {
150             costs[j + 258] = costs[j] + stepCost;
151             lengthArray[j + 258] = 258;
152             i++;
153             j++;
154             h.updateHash(input, i, to);
155           }
156         }
157   
158         Deflate.findLongestMatch(cookie, lmc, blockStart, h, input, i, to, 258, subLen);
159   
160         long costsJ = costs[j];
161         if (i + 1 <= to) {
162           long newCost = costsJ + slLiterals[input[i] & 0xFF];
163           if (newCost < costs[j + 1]) {
164             costs[j + 1] = newCost;
165             lengthArray[j + 1] = 1;
166           }
167         }
168         int lenValue = cookie.lenVal;
169         long baseCost = minCost + costsJ;
170         if (lenValue > to - i) {
171           lenValue = to - i;
172         }
173         int jpk = j + 3;
174         for (char k = 3; k <= lenValue; k++, jpk++) {
175           if (costs[jpk] > baseCost) {
176             long newCost = costsJ + (slLengths[k] + sdSymbols[cachedDistSymbol[subLen[k]]]);
177             if (costs[jpk] > newCost) {
178               costs[jpk] = newCost;
179               lengthArray[jpk] = k;
180             }
181           }
182         }
183   
184         i++;
185         j++;
186       }
187     }

189     static void bestFixedLengths(Cookie cookie, LongestMatchCache lmc, byte[] input, int from, int to,
190         char[] lengthArray, long[] costs) {
191       int windowStart = Math.max(from - 0x8000, 0);
192       Hash h = cookie.h;
193       h.init(input, windowStart, from, to);
194       Cookie.fillCostMax(costs, to - from + 1);
195       costs[0] = 0L;
196       lengthArray[0] = 0;
197   
198       char[] subLen = cookie.c259a;
199       for (int i = from; i < to; i++) {
200         int j = i - from;
201         h.updateHash(input, i, to);
202   
203         if (h.same[i & 0x7FFF] > 258 * 2
204             && i > from + 258 + 1
205             && i + 258 * 2 + 1 < to
206             && h.same[(i - 258) & 0x7FFF]
207             > 258) {
208           long symbolCost = fixedCost(258, 1);
209           for (int k = 0; k < 258; k++) {
210             costs[j + 258] = costs[j] + symbolCost;
211             lengthArray[j + 258] = 258;
212             i++;
213             j++;
214             h.updateHash(input, i, to);
215           }
216         }
217   
218         Deflate.findLongestMatch(cookie, lmc, from, h, input, i, to, 258, subLen);
219   
220         if (i + 1 <= to) {
221           long newCost = costs[j] + fixedCost(input[i] & 0xFF, 0);
222           if (newCost < costs[j + 1]) {
223             costs[j + 1] = newCost;
224             lengthArray[j + 1] = 1;
225           }
226         }
227         int lenValue = cookie.lenVal;
228         for (char k = 3; k <= lenValue && i + k <= to; k++) {
229           if (costs[j + k] - costs[j] <= 12.0) {
230             continue;
231           }
232   
233           long newCost = costs[j] + fixedCost(k, subLen[k]);
234           if (newCost < costs[j + k]) {
235             costs[j + k] = newCost;
236             lengthArray[j + k] = k;
237           }
238         }
239       }
240     }
