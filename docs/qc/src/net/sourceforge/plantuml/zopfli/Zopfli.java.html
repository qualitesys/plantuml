21   public class Zopfli {
22   
23     private static class Crc {
24   
25       private final static int[] table = makeTable();
26   
27       private static int[] makeTable() {
28         int[] result = new int[256];
29   
30         for (int n = 0; n < 256; ++n) {
31           int c = n;
32           for (int k = 0; k < 8; ++k) {
33             if ((c & 1) == 1) {
34               c = 0xEDB88320 ^ (c >>> 1);
35             } else {
36               c = c >>> 1;
37             }
38           }
39           result[n] = c;
40         }
41   
42         return result;
43       }
44   
45       public static int calculate(byte[] input) {
46         int c = ~0;
47         for (int i = 0, n = input.length; i < n; ++i) {
48           c = table[(c ^ input[i]) & 0xFF] ^ (c >>> 8);
49         }
50         return ~c;
51       }
52     }
53   
54     private final Cookie cookie;
55   
56     public synchronized Buffer compress(Options options, byte[] input) {
57       Buffer output = new Buffer();
58       switch (options.outputType) {
59         case GZIP:
60           compressGzip(options, input, output);
61           break;
62   
63         case ZLIB:
64           compressZlib(options, input, output);
65           break;
66         case DEFLATE:
67           Deflate.compress(cookie, options, input, output);
68           break;
69         default:
70           throw new IllegalArgumentException(
71               "Unexpected output format: " + options.outputType);
72       }
73       return output;
74     }
75   
76     /**
77      * Calculates the adler32 checksum of the data
78      */
79     private static int adler32(byte[] data) {
80       int s1 = 1;
81       int s2 = 1 >> 16;
82       int i = 0;
83       while (i < data.length) {
84         int tick = Math.min(data.length, i + 1024);
85         while (i < tick) {
86           s1 += data[i++];
87           s2 += s1;
88         }
89         s1 %= 65521;
90         s2 %= 65521;
91       }
92   
93       return (s2 << 16) | s1;
94     }
95   
96   
97     private void compressZlib(Options options, byte[] input,
98         Buffer output) {
99       output.append((byte) 0x78);
100       output.append((byte) 0x1E);
101   
102       Deflate.compress(cookie, options, input, output);
103   
104       int checksum = adler32(input);
105       output.append((byte) ((checksum >> 24) & 0xFF));
106       output.append((byte) ((checksum >> 16) & 0xFF));
107       output.append((byte) ((checksum >> 8) & 0xFF));
108       output.append((byte) (checksum & 0xFF));
109     }
110   
111     private void compressGzip(Options options, byte[] input,
112         Buffer output) {
113       output.append((byte) 31);
114       output.append((byte) 139);
115       output.append((byte) 8);
116       output.append((byte) 0);
117   
118       output.append((byte) 0);
119       output.append((byte) 0);
120       output.append((byte) 0);
121       output.append((byte) 0);
122   
123       output.append((byte) 2);
124       output.append((byte) 3);
125   
126       Deflate.compress(cookie, options, input, output);
127   
128       int crc = Crc.calculate(input);
129       output.append((byte) (crc & 0xFF));
130       output.append((byte) ((crc >> 8) & 0xFF));
131       output.append((byte) ((crc >> 16) & 0xFF));
132       output.append((byte) ((crc >> 24) & 0xFF));
133   
134       int size = input.length;
135       output.append((byte) (size & 0xFF));
136       output.append((byte) ((size >> 8) & 0xFF));
137       output.append((byte) ((size >> 16) & 0xFF));
138       output.append((byte) ((size >> 24) & 0xFF));
139     }

141     public Zopfli(int masterBlockSize) {
142       cookie = new Cookie(masterBlockSize);
143     }
