25     static void lengthLimitedCodeLengths(Cookie cookie, int[] frequencies, int maxBits,
26         int[] bitLengths) {
27       cookie.resetPool();
28       int n = frequencies.length;
29       int nn = 0;
30       Node[] leaves = cookie.leaves1;
31       for (int i = 0; i < n; i++) {
32         if (frequencies[i] != 0) {
33           leaves[nn] = cookie.node(frequencies[i], i, null);
34           nn++;
35         }
36       }
37   
38       if (nn == 0) {
39         return;
40       }
41       if (nn == 1) {
42         bitLengths[leaves[0].count] = 1;
43         return;
44       }
45   
46       Node[] leaves2 = cookie.leaves2;
47       System.arraycopy(leaves, 0, leaves2, 0, nn);
48       sort(leaves2, leaves, 0, nn);
49   
50       Node[] list0 = cookie.list0;
51       Node node0 = cookie.node(leaves[0].weight, 1, null);
52   
53       Node[] list1 = cookie.list1;
54       Node node1 = cookie.node(leaves[1].weight, 2, null);
55   
56       for (int i = 0; i < maxBits; ++i) {
57         list0[i] = node0;
58         list1[i] = node1;
59       }
60   
61       int numBoundaryPmRuns = 2 * nn - 4;
62       for (int i = 0; i < numBoundaryPmRuns; i++) {
63         boolean last = i == numBoundaryPmRuns - 1;
64         boundaryPm(cookie, leaves, list0, list1, nn, maxBits - 1, last);
65       }
66   
67       for (Node node = list1[maxBits - 1]; node != null; node = node.tail) {
68         for (int i = node.count - 1; i >= 0; --i) {
69           bitLengths[leaves[i].count]++;
70         }
71       }
72     }

74     private static void boundaryPm(Cookie cookie, Node[] leaves, Node[] list0, Node[] list1, int numSymbols, int index,
75         boolean last) {
76       int lastCount = list1[index].count;
77   
78       if (index == 0 && lastCount >= numSymbols) {
79         return;
80       }
81   
82       list0[index] = list1[index];
83   
84       if (index == 0) {
85         list1[index] = cookie.node(leaves[lastCount].weight, lastCount + 1, null);
86       } else {
87         int sum = list0[index - 1].weight + list1[index - 1].weight;
88         if (lastCount < numSymbols && sum > leaves[lastCount].weight) {
89           list1[index] = cookie.node(leaves[lastCount].weight, lastCount + 1, list1[index].tail);
90         } else {
91           list1[index] = cookie.node(sum, lastCount, list1[index - 1]);
92           if (!last) {
93             boundaryPm(cookie, leaves, list0, list1, numSymbols, index - 1, false);
94             boundaryPm(cookie, leaves, list0, list1, numSymbols, index - 1, false);
95           }
96         }
97       }
98     }

100     private static void sort(Node[] src, Node[] dest, int low, int high) {
101       int length = high - low;
102   
103       if (length < 7) {
104         for (int i = low + 1; i < high; i++)
105           for (int j = i, k = i - 1; j > low && (dest[k].weight > dest[j].weight); --j, --k) {
106             Node t = dest[j];
107             dest[j] = dest[k];
108             dest[k] = t;
109           }
110         return;
111       }
112   
113       int mid = (low + high) >>> 1;
114       sort(dest, src, low, mid);
115       sort(dest, src, mid, high);
116   
117       for (int i = low, p = low, q = mid; i < high; i++) {
118         if (q >= high || p < mid && (src[p].weight <= src[q].weight))
119           dest[i] = src[p++];
120         else
121           dest[i] = src[q++];
122       }
123     }
124   }
