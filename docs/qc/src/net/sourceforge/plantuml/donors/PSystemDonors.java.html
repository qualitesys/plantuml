109       @Override
110       protected UDrawable getRootDrawable(FileFormatOption fileFormatOption) throws IOException {
111           final List<TextBlock> cols = getCols(getDonors(), COLS, FREE_LINES);
112           return new UDrawable() {
113               public void drawU(UGraphic ug) {
114                   final TextBlockBackcolored header = GraphicStrings
115                           .createBlackOnWhite(Arrays.asList("<b>Special thanks to our sponsors and donors !"));
116                   header.drawU(ug);
117                   final StringBounder stringBounder = ug.getStringBounder();
118                   ug = ug.apply(UTranslate.dy(header.calculateDimension(stringBounder).getHeight()));
119                   double x = 0;
120                   double lastX = 0;
121                   double y = 0;
122                   for (TextBlock tb : cols) {
123                       final XDimension2D dim = tb.calculateDimension(stringBounder);
124                       tb.drawU(ug.apply(UTranslate.dx(x)));
125                       lastX = x;
126                       x += dim.getWidth() + 10;
127                       y = Math.max(y, dim.getHeight());
128                   }
129                   final UImage logo = new UImage(
130                           new PixelImage(PSystemVersion.getPlantumlImage(), AffineTransformType.TYPE_BILINEAR));
131                   ug.apply(new UTranslate(lastX, y - logo.getHeight())).draw(logo);
132               }
133           };
134       }

136       public static List<TextBlock> getCols(List<String> lines, final int nbCol, final int reserved) throws IOException {
137           final List<TextBlock> result = new ArrayList<>();
138           final int maxLine = (lines.size() + (nbCol - 1) + reserved) / nbCol;
139           for (int i = 0; i < lines.size(); i += maxLine) {
140               final List<String> current = lines.subList(i, Math.min(lines.size(), i + maxLine));
141               result.add(GraphicStrings.createBlackOnWhite(current));
142           }
143           return result;
144       }

146       private List<String> getDonors() throws IOException {
147           final List<String> lines = new ArrayList<>();
148           final Transcoder t = TranscoderImpl.utf8(new AsciiEncoder(), new StringCompressorNone(),
149                   new CompressionBrotli());
150           try {
151               final String s = t.decode(DONORS).replace('*', '.');
152               final StringTokenizer st = new StringTokenizer(s, BackSlash.NEWLINE);
153               while (st.hasMoreTokens()) {
154                   lines.add(st.nextToken());
155               }
156           } catch (NoPlantumlCompressionException e) {
157               Logme.error(e);
158           }
159           return lines;
160       }
