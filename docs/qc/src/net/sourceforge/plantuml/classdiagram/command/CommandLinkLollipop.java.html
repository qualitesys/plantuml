69           return RegexConcat.build(CommandLinkLollipop.class.getName() + umlDiagramType, RegexLeaf.start(), //
70                   new RegexOptional( //
71                           new RegexConcat( //
72                                   new RegexLeaf("HEADER", "@([\\d.]+)"), //
73                                   RegexLeaf.spaceOneOrMore() //
74                           )), //
75                   new RegexLeaf("ENT1",
76                           "(?:" + optionalKeywords(umlDiagramType) + "[%s]+)?"
77                                   + "(\\.?[%pLN_]+(?:\\.[%pLN_]+)*|[%g][^%g]+[%g])[%s]*(\\<\\<.*\\>\\>)?"), //
78                   RegexLeaf.spaceZeroOrMore(), //

115       @Override
116       protected CommandExecutionResult executeArg(AbstractClassOrObjectDiagram diagram, LineLocation location,
117               RegexResult arg) {
118   
119           final String ent1 = arg.get("ENT1", 1);
120           final String ent2 = arg.get("ENT2", 1);
121   
122           final IEntity cl1;
123           final IEntity cl2;
124           final IEntity normalEntity;
125   
126           final String suffix = "lol" + diagram.getUniqueSequence();
127           if (arg.get("LOL_THEN_ENT", 1) == null) {
128               assert arg.get("ENT_THEN_LOL", 0) != null;
129               final Ident ident1 = diagram.buildLeafIdent(ent1);
130               final Code ent1code = diagram.V1972() ? ident1 : diagram.buildCode(ent1);
131               cl1 = getFoo1(diagram, ent1code, ident1);
132               final Ident idNewLong = diagram.buildLeafIdent(ent1 + suffix);
133               cl2 = diagram.createLeaf(idNewLong, idNewLong.toCode(diagram), Display.getWithNewlines(ent2),
134                       getType(arg.get("ENT_THEN_LOL", 1)), null);
135               normalEntity = cl1;
136           } else {
137               final Ident ident2 = diagram.buildLeafIdent(ent2);
138               final Code ent2code = diagram.V1972() ? ident2 : diagram.buildCode(ent2);
139               cl2 = getFoo1(diagram, ent2code, ident2);
140               final Ident idNewLong = diagram.buildLeafIdent(ent2 + suffix);
141               cl1 = diagram.createLeaf(idNewLong, idNewLong.toCode(diagram), Display.getWithNewlines(ent1),
142                       getType(arg.get("LOL_THEN_ENT", 0)), null);
143               normalEntity = cl2;
144           }
145   
146           final LinkType linkType = getLinkType(arg);
147           final String queue = getQueue(arg);
148   
149           int length = queue.length();
150           if (length == 1 && diagram.getNbOfHozizontalLollipop(normalEntity) > 1)
151               length++;
152   
153           String firstLabel = arg.get("FIRST_LABEL", 0);
154           String secondLabel = arg.get("SECOND_LABEL", 0);
155   
156           String labelLink = null;
157   
158           if (arg.get("LABEL_LINK", 0) != null) {
159               labelLink = arg.get("LABEL_LINK", 0);
160               if (firstLabel == null && secondLabel == null) {
161                   final Pattern2 p1 = MyPattern.cmpile("^\"([^\"]+)\"([^\"]+)\"([^\"]+)\"$");
162                   final Matcher2 m1 = p1.matcher(labelLink);
163                   if (m1.matches()) {
164                       firstLabel = m1.group(1);
165                       labelLink = StringUtils.trin(
166                               StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(StringUtils.trin(m1.group(2))));
167                       secondLabel = m1.group(3);
168                   } else {
169                       final Pattern2 p2 = MyPattern.cmpile("^\"([^\"]+)\"([^\"]+)$");
170                       final Matcher2 m2 = p2.matcher(labelLink);
171                       if (m2.matches()) {
172                           firstLabel = m2.group(1);
173                           labelLink = StringUtils.trin(StringUtils
174                                   .eventuallyRemoveStartingAndEndingDoubleQuote(StringUtils.trin(m2.group(2))));
175                           secondLabel = null;
176                       } else {
177                           final Pattern2 p3 = MyPattern.cmpile("^([^\"]+)\"([^\"]+)\"$");
178                           final Matcher2 m3 = p3.matcher(labelLink);
179                           if (m3.matches()) {
180                               firstLabel = null;
181                               labelLink = StringUtils.trin(StringUtils
182                                       .eventuallyRemoveStartingAndEndingDoubleQuote(StringUtils.trin(m3.group(1))));
183                               secondLabel = m3.group(2);
184                           }
185                       }
186                   }
187               }
188               labelLink = StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(labelLink);
189           }
190           final LinkArg linkArg = LinkArg.build(Display.getWithNewlines(labelLink), length,
191                   diagram.getSkinParam().classAttributeIconSize() > 0);
192           final Link link = new Link(diagram.getSkinParam().getCurrentStyleBuilder(), cl1, cl2, linkType,
193                   linkArg.withQuantifier(firstLabel, secondLabel).withDistanceAngle(diagram.getLabeldistance(),
194                           diagram.getLabelangle()));
195           diagram.resetPragmaLabel();
196           addLink(diagram, link, arg.get("HEADER", 0));
197   
198           return CommandExecutionResult.ok();
199       }

201       private IEntity getFoo1(AbstractClassOrObjectDiagram diagram, final Code code, final Ident ident) {
202           if (diagram.V1972()) {
203               final IEntity result = ident.size() == 1 ? diagram.getLeafVerySmart(ident) : diagram.getLeafStrict(ident);
204               if (result != null)
205                   return result;
206   
207           }
208           return diagram.getOrCreateLeaf(ident, code, null, null);
209       }
