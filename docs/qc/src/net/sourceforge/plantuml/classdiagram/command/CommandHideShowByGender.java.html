60   public class CommandHideShowByGender extends SingleLineCommand2<UmlDiagram> {
61   
62       public CommandHideShowByGender() {
63           super(getRegexConcat());
64       }

66       static IRegex getRegexConcat() {
67           return RegexConcat.build(CommandHideShowByGender.class.getName(), RegexLeaf.start(), //
68                   new RegexLeaf("COMMAND", "(hide|show)"), //
69                   RegexLeaf.spaceOneOrMore(), //
70                   new RegexLeaf("GENDER",
71                           "(?:(class|object|interface|enum|annotation|abstract|[%pLN_.]+|[%g][^%g]+[%g]|\\<\\<.*\\>\\>)[%s]+)*?"), //
72                   new RegexOptional( //
73                           new RegexConcat( //
74                                   new RegexLeaf("EMPTY", "(empty)"), //
75                                   RegexLeaf.spaceOneOrMore()) //
76                   ), //
77                   new RegexLeaf("PORTION", "(members?|attributes?|fields?|methods?|circles?|circled?|stereotypes?)"), //
78                   RegexLeaf.end());
79       }

81       private final EntityGender emptyByGender(EntityPortion portion) {
82           if (portion == EntityPortion.METHOD) {
83               return EntityGenderUtils.emptyMethods();
84           }
85           if (portion == EntityPortion.FIELD) {
86               return EntityGenderUtils.emptyFields();
87           }
88           if (portion == EntityPortion.MEMBER) {
89               throw new IllegalArgumentException();
90               // return EntityGenderUtils.emptyMembers();
91           }
92           return EntityGenderUtils.all();
93       }

95       @Override
96       protected CommandExecutionResult executeArg(UmlDiagram diagram, LineLocation location, RegexResult arg) {
97           if (diagram instanceof AbstractClassOrObjectDiagram) {
98               return executeClassDiagram((AbstractClassOrObjectDiagram) diagram, arg);
99           }
100           if (diagram instanceof DescriptionDiagram) {
101               return executeDescriptionDiagram((DescriptionDiagram) diagram, arg);
102           }
103           if (diagram instanceof SequenceDiagram) {
104               return executeSequenceDiagram((SequenceDiagram) diagram, arg);
105           }
106           // Just ignored
107           return CommandExecutionResult.ok();
108       }

110       private CommandExecutionResult executeSequenceDiagram(SequenceDiagram diagram, RegexResult arg) {
111           final EntityPortion portion = getEntityPortion(arg.get("PORTION", 0));
112           diagram.hideOrShow(portion.asSet(), arg.get("COMMAND", 0).equalsIgnoreCase("show"));
113           return CommandExecutionResult.ok();
114       }

116       private CommandExecutionResult executeDescriptionDiagram(DescriptionDiagram diagram, RegexResult arg) {
117           final EntityPortion portion = getEntityPortion(arg.get("PORTION", 0));
118           final EntityGender gender;
119           final String arg1 = arg.get("GENDER", 0);
120           if (arg1 == null) {
121               gender = EntityGenderUtils.all();
122           } else if (arg1.equalsIgnoreCase("class")) {
123               gender = EntityGenderUtils.byEntityType(LeafType.CLASS);
124           } else if (arg1.equalsIgnoreCase("object")) {
125               gender = EntityGenderUtils.byEntityType(LeafType.OBJECT);
126           } else if (arg1.equalsIgnoreCase("interface")) {
127               gender = EntityGenderUtils.byEntityType(LeafType.INTERFACE);
128           } else if (arg1.equalsIgnoreCase("enum")) {
129               gender = EntityGenderUtils.byEntityType(LeafType.ENUM);
130           } else if (arg1.equalsIgnoreCase("abstract")) {
131               gender = EntityGenderUtils.byEntityType(LeafType.ABSTRACT_CLASS);
132           } else if (arg1.equalsIgnoreCase("annotation")) {
133               gender = EntityGenderUtils.byEntityType(LeafType.ANNOTATION);
134           } else if (arg1.equalsIgnoreCase("protocol")) {
135               gender = EntityGenderUtils.byEntityType(LeafType.PROTOCOL);
136           } else if (arg1.equalsIgnoreCase("struct")) {
137               gender = EntityGenderUtils.byEntityType(LeafType.STRUCT);
138           } else if (arg1.equalsIgnoreCase("exception")) {
139               gender = EntityGenderUtils.byEntityType(LeafType.EXCEPTION);
140           } else if (arg1.equalsIgnoreCase("metaclass")) {
141               gender = EntityGenderUtils.byEntityType(LeafType.METACLASS);
142           } else if (arg1.equalsIgnoreCase("stereotype")) {
143               gender = EntityGenderUtils.byEntityType(LeafType.STEREOTYPE);
144           } else if (arg1.startsWith("<<")) {
145               gender = EntityGenderUtils.byStereotype(arg1);
146           } else {
147               final IEntity entity = diagram.getOrCreateLeaf(diagram.buildLeafIdent(arg1), diagram.buildCode(arg1), null,
148                       null);
149               gender = EntityGenderUtils.byEntityAlone(entity);
150           }
151   
152           diagram.hideOrShow(gender, portion, arg.get("COMMAND", 0).equalsIgnoreCase("show"));
153           return CommandExecutionResult.ok();
154       }

156       private CommandExecutionResult executeClassDiagram(AbstractClassOrObjectDiagram diagram, RegexResult arg) {
157   
158           final EntityPortion portion = getEntityPortion(arg.get("PORTION", 0));
159   
160           EntityGender gender = null;
161           String arg1 = arg.get("GENDER", 0);
162           if (arg1 == null) {
163               gender = EntityGenderUtils.all();
164           } else if (arg1.equalsIgnoreCase("class")) {
165               gender = EntityGenderUtils.byEntityType(LeafType.CLASS);
166           } else if (arg1.equalsIgnoreCase("object")) {
167               gender = EntityGenderUtils.byEntityType(LeafType.OBJECT);
168           } else if (arg1.equalsIgnoreCase("interface")) {
169               gender = EntityGenderUtils.byEntityType(LeafType.INTERFACE);
170           } else if (arg1.equalsIgnoreCase("enum")) {
171               gender = EntityGenderUtils.byEntityType(LeafType.ENUM);
172           } else if (arg1.equalsIgnoreCase("abstract")) {
173               gender = EntityGenderUtils.byEntityType(LeafType.ABSTRACT_CLASS);
174           } else if (arg1.equalsIgnoreCase("annotation")) {
175               gender = EntityGenderUtils.byEntityType(LeafType.ANNOTATION);
176           } else if (arg1.equalsIgnoreCase("protocol")) {
177               gender = EntityGenderUtils.byEntityType(LeafType.PROTOCOL);
178           } else if (arg1.equalsIgnoreCase("struct")) {
179               gender = EntityGenderUtils.byEntityType(LeafType.STRUCT);
180           } else if (arg1.equalsIgnoreCase("exception")) {
181               gender = EntityGenderUtils.byEntityType(LeafType.EXCEPTION);
182           } else if (arg1.equalsIgnoreCase("metaclass")) {
183               gender = EntityGenderUtils.byEntityType(LeafType.METACLASS);
184           } else if (arg1.equalsIgnoreCase("stereotype")) {
185               gender = EntityGenderUtils.byEntityType(LeafType.STEREOTYPE);
186           } else if (arg1.startsWith("<<")) {
187               gender = EntityGenderUtils.byStereotype(arg1);
188           } else {
189               arg1 = StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(arg1);
190               final Ident ident = diagram.buildLeafIdent(arg1);
191               final Code code = diagram.V1972() ? ident : diagram.buildCode(arg1);
192               final IEntity entity = diagram.getOrCreateLeaf(ident, code, null, null);
193               gender = EntityGenderUtils.byEntityAlone(entity);
194           }
195           if (gender != null) {
196               final boolean empty = arg.get("EMPTY", 0) != null;
197               final boolean emptyMembers = empty && portion == EntityPortion.MEMBER;
198               if (empty == true && emptyMembers == false) {
199                   gender = EntityGenderUtils.and(gender, emptyByGender(portion));
200               }
201               if (EntityUtils.groupRoot(diagram.getCurrentGroup()) == false) {
202                   gender = EntityGenderUtils.and(gender, EntityGenderUtils.byPackage(diagram.getCurrentGroup()));
203               }
204   
205               if (emptyMembers) {
206                   diagram.hideOrShow(EntityGenderUtils.and(gender, emptyByGender(EntityPortion.FIELD)),
207                           EntityPortion.FIELD, arg.get("COMMAND", 0).equalsIgnoreCase("show"));
208                   diagram.hideOrShow(EntityGenderUtils.and(gender, emptyByGender(EntityPortion.METHOD)),
209                           EntityPortion.METHOD, arg.get("COMMAND", 0).equalsIgnoreCase("show"));
210               } else {
211                   diagram.hideOrShow(gender, portion, arg.get("COMMAND", 0).equalsIgnoreCase("show"));
212               }
213           }
214           return CommandExecutionResult.ok();
215       }

217       private EntityPortion getEntityPortion(String s) {
218           final String sub = StringUtils.goLowerCase(s.substring(0, 3));
219           if (sub.equals("met")) {
220               return EntityPortion.METHOD;
221           }
222           if (sub.equals("mem")) {
223               return EntityPortion.MEMBER;
224           }
225           if (sub.equals("att") || sub.equals("fie")) {
226               return EntityPortion.FIELD;
227           }
228           if (sub.equals("cir")) {
229               return EntityPortion.CIRCLED_CHARACTER;
230           }
231           if (sub.equals("ste")) {
232               return EntityPortion.STEREOTYPE;
233           }
234           throw new IllegalArgumentException();
235       }
