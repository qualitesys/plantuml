162           if (ent1String == null && ent2String == null)
163               return executeArgSpecial3(diagram, arg);
164   
165           if (ent1String == null)
166               return executeArgSpecial1(diagram, arg);
167   
168           if (ent2String == null)
169               return executeArgSpecial2(diagram, arg);
170   
171           Ident ident1 = diagram.buildLeafIdentSpecial(ent1String);
172           Ident ident2 = diagram.buildLeafIdentSpecial(ent2String);
173           Ident ident1pure = Ident.empty().add(ent1String, diagram.getNamespaceSeparator());
174           Ident ident2pure = Ident.empty().add(ent2String, diagram.getNamespaceSeparator());
175           Code code1 = diagram.V1972() ? ident1 : diagram.buildCode(ent1String);
176           Code code2 = diagram.V1972() ? ident2 : diagram.buildCode(ent2String);
177           if (isGroupButNotTheCurrentGroup(diagram, code1, ident1)
178                   && isGroupButNotTheCurrentGroup(diagram, code2, ident2)) {
179               return executePackageLink(diagram, arg);
180           }
181   
182           String port1 = null;
183           String port2 = null;
184   
185           if (diagram.V1972()) {
186               if ("::".equals(diagram.getNamespaceSeparator())) {
187                   if (removeMemberPartIdentSpecial(diagram, ident1) != null) {
188                       port1 = ident1.getLast();
189                       ident1 = removeMemberPartIdentSpecial(diagram, ident1);
190                       code1 = ident1;
191                   }
192                   if (removeMemberPartIdentSpecial(diagram, ident2) != null) {
193                       port2 = ident2.getLast();
194                       ident2 = removeMemberPartIdentSpecial(diagram, ident2);
195                       code2 = ident1;
196                   }
197               } else {
198                   if (removeMemberPartIdent(diagram, ident1) != null) {
199                       port1 = ident1.getPortMember();
200                       ident1 = removeMemberPartIdent(diagram, ident1);
201                       code1 = ident1;
202                   }
203                   if (removeMemberPartIdent(diagram, ident2) != null) {
204                       port2 = ident2.getPortMember();
205                       ident2 = removeMemberPartIdent(diagram, ident2);
206                       code2 = ident2;
207                   }
208               }
209           } else {
210               if (removeMemberPartLegacy1972(diagram, ident1) != null) {
211                   port1 = ident1.getPortMember();
212                   code1 = removeMemberPartLegacy1972(diagram, ident1);
213                   ident1 = ident1.removeMemberPart();
214               }
215               if (removeMemberPartLegacy1972(diagram, ident2) != null) {
216                   port2 = ident2.getPortMember();
217                   code2 = removeMemberPartLegacy1972(diagram, ident2);
218                   ident2 = ident2.removeMemberPart();
219               }
220           }
221   
222           final IEntity cl1 = getFoo1(diagram, code1, ident1, ident1pure);
223           final IEntity cl2 = getFoo1(diagram, code2, ident2, ident2pure);
224   
225           final LinkType linkType = getLinkType(arg);
226           final Direction dir = getDirection(arg);
227           final int queue;
228           if (dir == Direction.LEFT || dir == Direction.RIGHT)
229               queue = 1;
230           else
231               queue = getQueueLength(arg);
232   
233           final Labels labels = new Labels(arg);
234   
235           final String kal1 = arg.get("QUALIFIER1", 0);
236           final String kal2 = arg.get("QUALIFIER2", 0);
237   
238           final LinkArg linkArg = LinkArg
239                   .build(labels.getDisplay(), queue, diagram.getSkinParam().classAttributeIconSize() > 0)
240                   .withQuantifier(labels.getFirstLabel(), labels.getSecondLabel())
241                   .withDistanceAngle(diagram.getLabeldistance(), diagram.getLabelangle()).withKal(kal1, kal2);
242   
243           Link link = new Link(diagram.getSkinParam().getCurrentStyleBuilder(), cl1, cl2, linkType, linkArg);
244           if (arg.get("URL", 0) != null) {
245               final UrlBuilder urlBuilder = new UrlBuilder(diagram.getSkinParam().getValue("topurl"), UrlMode.STRICT);
246               final Url url = urlBuilder.getUrl(arg.get("URL", 0));
247               link.setUrl(url);
248           }
249           link.setPortMembers(port1, port2);
250   
251           if (dir == Direction.LEFT || dir == Direction.UP)
252               link = link.getInv();
253   
254           link.setLinkArrow(labels.getLinkArrow());
255           link.setColors(color().getColor(arg, diagram.getSkinParam().getIHtmlColorSet()));
256           link.applyStyle(arg.getLazzy("ARROW_STYLE", 0));
257           link.setCodeLine(location);
258   
259           addLink(diagram, link, arg.get("HEADER", 0));
260   
261           return CommandExecutionResult.ok();
262       }

365                   : diagram.getGroup(diagram.buildCode(ent1String));

367                   : diagram.getGroup(diagram.buildCode(ent2String));

373               queue = 1;
374           else
375               queue = getQueueLength(arg);
376

378           final String firstLabel = arg.get("FIRST_LABEL", 0);

380           final LinkArg linkArg = LinkArg.build(labelLink, queue, diagram.getSkinParam().classAttributeIconSize() > 0);
381           final Link link = new Link(diagram.getSkinParam().getCurrentStyleBuilder(), cl1, cl2, linkType,

383                           diagram.getLabelangle()));

385   
386           diagram.resetPragmaLabel();
387   
388           link.applyStyle(arg.getLazzy("ARROW_STYLE", 0));
389   
390           addLink(diagram, link, arg.get("HEADER", 0));
391           return CommandExecutionResult.ok();
392       }
