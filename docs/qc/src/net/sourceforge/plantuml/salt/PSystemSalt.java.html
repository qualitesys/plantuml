164       private List<String> manageSprite() {
165   
166           final Command<WithSprite> cmd = new CommandFactorySprite().createMultiLine(false);
167   
168           final List<String> result = new ArrayList<>();
169           for (Iterator<String> it = data.iterator(); it.hasNext();) {
170               String s = it.next();
171               if (s.equals("hide stereotype")) {
172                   // System.err.println("skipping " + s);
173               } else if (s.startsWith("skinparam ")) {
174                   // System.err.println("skipping " + s);
175               } else if (s.startsWith("scale ")) {
176                   final double scale = Double.parseDouble(s.substring("scale ".length()));
177                   this.setScale(new ScaleSimple(scale));
178                   // System.err.println("skipping " + s);
179               } else if (s.startsWith("sprite $")) {
180                   BlocLines bloc = BlocLines.singleString(s);
181                   do {
182                       s = it.next();
183                       bloc = bloc.addString(s);
184                   } while (s.equals("}") == false);
185                   try {
186                       final CommandExecutionResult cmdResult = cmd.execute(this, bloc);
187                   } catch (NoSuchColorException e) {
188                   }
189               } else {
190                   result.add(s);
191               }
192           }
193           return result;
194       }

196       private Element createElement(List<String> data) {
197   
198           final DataSourceImpl source = new DataSourceImpl(data);
199   
200           final Collection<AbstractElementFactoryComplex> cpx = new ArrayList<>();
201   
202           // cpx.add(new ElementFactorySimpleFrame(source, dictionnary));
203           cpx.add(new ElementFactoryPyramid(source, dictionary));
204           cpx.add(new ElementFactoryScroll(source, dictionary));
205           cpx.add(new ElementFactoryBorder(source, dictionary));
206   
207           for (AbstractElementFactoryComplex f : cpx) {
208               addSimpleFactory(f, source, dictionary);
209           }
210           for (AbstractElementFactoryComplex f1 : cpx) {
211               for (AbstractElementFactoryComplex f2 : cpx) {
212                   f1.addFactory(f2);
213               }
214           }
215   
216           for (ElementFactory f : cpx) {
217               if (f.ready()) {
218                   Log.info("Using " + f);
219                   return f.create().getElement();
220               }
221           }
222   
223           Log.println("data=" + data);
224           throw new IllegalArgumentException();
225   
226       }

237           cpxFactory.addFactory(new ElementFactoryRadioOn(source, dictionary));

244       }
