36   package net.sourceforge.plantuml.salt.element;
37   
38   import java.util.HashSet;
39   import java.util.Set;
40   
41   import net.sourceforge.plantuml.awt.geom.XDimension2D;
42   import net.sourceforge.plantuml.graphic.TextBlock;
43   import net.sourceforge.plantuml.salt.Cell;
44   import net.sourceforge.plantuml.ugraphic.UGraphic;
45   import net.sourceforge.plantuml.ugraphic.ULine;
46   import net.sourceforge.plantuml.ugraphic.URectangle;
47   import net.sourceforge.plantuml.ugraphic.UTranslate;
48   import net.sourceforge.plantuml.ugraphic.color.HColor;
49   
50   public class Grid {
51   
52       private final double[] rowsStart;
53       private final double[] colsStart;
54       private final TableStrategy strategy;
55       private final TextBlock title;
56   
57       private final Set<Segment> horizontals = new HashSet<>();
58       private final Set<Segment> verticals = new HashSet<>();
59   
60       public Grid(double[] rowsStart, double[] colsStart, TableStrategy strategy, TextBlock title) {
61           this.title = title;
62           this.rowsStart = rowsStart;
63           this.colsStart = colsStart;
64           this.strategy = strategy;
65           if (strategy == TableStrategy.DRAW_OUTSIDE || strategy == TableStrategy.DRAW_OUTSIDE_WITH_TITLE
66                   || strategy == TableStrategy.DRAW_ALL) {
67               addOutside();
68           }
69       }
70   
71       private void addOutside() {
72           final int nbRow = rowsStart.length;
73           final int nbCol = colsStart.length;
74           for (int c = 0; c < nbCol - 1; c++) {
75               horizontals.add(new Segment(0, c));
76               horizontals.add(new Segment(nbRow - 1, c));
77           }
78           for (int r = 0; r < nbRow - 1; r++) {
79               verticals.add(new Segment(r, 0));
80               verticals.add(new Segment(r, nbCol - 1));
81           }
82   
83       }
84   
85       public void drawU(UGraphic ug, double x, double y, HColor white) {
86           // Hlines
87           for (Segment seg : horizontals) {
88               final int row1 = seg.getRow();
89               final int col1 = seg.getCol();
90               final double width = colsStart[col1 + 1] - colsStart[col1];
91               ug.apply(new UTranslate(x + colsStart[col1], y + rowsStart[row1])).draw(ULine.hline(width));
92           }
93           // Vlines
94           for (Segment seg : verticals) {
95               final int row1 = seg.getRow();
96               final int col1 = seg.getCol();
97               final double height = rowsStart[row1 + 1] - rowsStart[row1];
98               ug.apply(new UTranslate(x + colsStart[col1], y + rowsStart[row1])).draw(ULine.vline(height));
99           }
100   
101           final XDimension2D dim = title.calculateDimension(ug.getStringBounder());
102   
103           if (dim.getWidth() > 0 && dim.getHeight() > 0) {
104               final UGraphic ug2 = ug.apply(new UTranslate(x + 6, y - dim.getHeight() * 0));
105               ug2.apply(white.bg()).apply(white).draw(new URectangle(dim));
106               title.drawU(ug2);
107           }
108   
109       }
110   
111       public void addCell(Cell cell) {
112   
113           if (strategy == TableStrategy.DRAW_NONE || strategy == TableStrategy.DRAW_OUTSIDE
114                   || strategy == TableStrategy.DRAW_OUTSIDE_WITH_TITLE) {
115               return;
116           }
117   
118           if (strategy == TableStrategy.DRAW_HORIZONTAL || strategy == TableStrategy.DRAW_ALL) {
119               // Hlines
120               for (int c = cell.getMinCol(); c <= cell.getMaxCol(); c++) {
121                   horizontals.add(new Segment(cell.getMinRow(), c));
122                   horizontals.add(new Segment(cell.getMaxRow() + 1, c));
123               }
124           }
125           if (strategy == TableStrategy.DRAW_VERTICAL || strategy == TableStrategy.DRAW_ALL) {
126               // Vlines
127               for (int r = cell.getMinRow(); r <= cell.getMaxRow(); r++) {
128                   verticals.add(new Segment(r, cell.getMinCol()));
129                   verticals.add(new Segment(r, cell.getMaxCol() + 1));
130               }
131           }
132       }
133   }
