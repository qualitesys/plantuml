120       public static int swapInt(int v) {
121           return (v >>> 24) | (v << 24) | ((v << 8) & 0x00FF0000) | ((v >> 8) & 0x0000FF00);
122       }

124       public static short swapShort(short v) {
125           return (short) ((v >>> 8) | (v << 8));
126       }
