65       public static synchronized byte[] salting(String pass, byte[] salt)
66               throws NoSuchAlgorithmException, InvalidKeySpecException {
67           final int iterations = 500;
68           final int keyLength = 512;
69           final SecretKeyFactory skf = SecretKeyFactory.getInstance("PBKDF2WithHmacSHA1");
70           final PBEKeySpec spec = new PBEKeySpec(pass.toCharArray(), salt, iterations, keyLength);
71           final SecretKey key = skf.generateSecret(spec);
72           final byte[] tmp = key.getEncoded();
73           return tmp;
74       }

94       public static String toHexString(byte data[]) {

154       public static synchronized String getSignatureSha512(InputStream is) throws IOException {
155           try {
156               final MessageDigest msgDigest = MessageDigest.getInstance("SHA-512");
157               int read = 0;
158               while ((read = is.read()) != -1) {
159                   msgDigest.update((byte) read);
160               }
161               final byte[] digest = msgDigest.digest();
162               return toString(digest);
163           } catch (NoSuchAlgorithmException e) {
164               Logme.error(e);
165               throw new UnsupportedOperationException(e);
166           } catch (UnsupportedEncodingException e) {
167               Logme.error(e);
168               throw new UnsupportedOperationException(e);
169           }
170       }

177       public static String purge(String s) {
178           final String regex = "(?i)\\<img\\s+src=\"(?:[^\"]+[/\\\\])?([^/\\\\\\d.]+)\\d*(\\.\\w+)\"/\\>";
179           s = s.replaceAll(regex, "<img src=\"$1$2\"/>");
180           final String regex2 = "(?i)image=\"(?:[^\"]+[/\\\\])?([^/\\\\\\d.]+)\\d*(\\.\\w+)\"";
181           s = s.replaceAll(regex2, "image=\"$1$2\"");
182           return s;
183       }

185       public static synchronized String getSignature(SFile f) throws IOException {
186           try (final InputStream is = f.openFile()) {
187               final MessageDigest msgDigest = MessageDigest.getInstance("MD5");
188               if (is == null) {
189                   throw new FileNotFoundException();
190               }
191               int read = -1;
192               while ((read = is.read()) != -1) {
193                   msgDigest.update((byte) read);
194               }
195               final byte[] digest = msgDigest.digest();
196               return toString(digest);
197           } catch (NoSuchAlgorithmException e) {
198               Logme.error(e);
199               throw new UnsupportedOperationException(e);
200           } catch (UnsupportedEncodingException e) {
201               Logme.error(e);
202               throw new UnsupportedOperationException(e);
203           }
204       }
