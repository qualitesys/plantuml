144               } else if (s.equalsIgnoreCase("-teps") || s.equalsIgnoreCase("-eps")) {

148               } else if (s.equalsIgnoreCase("-ttxt") || s.equalsIgnoreCase("-txt")) {

153                   setFileFormatOption(new FileFormatOption(FileFormat.BRAILLE_PNG));
154               } else if (s.equalsIgnoreCase("-png") || s.equalsIgnoreCase("-tpng")) {

166               } else if (s.equalsIgnoreCase("-darkmode")) {
167                   setFileFormatOption(this.fileFormatOption.withColorMapper(ColorMapper.DARK_MODE));
168               } else if (s.equalsIgnoreCase("-overwrite")) {
169                   OptionFlags.getInstance().setOverwrite(true);
170               } else if (s.equalsIgnoreCase("-output") || s.equalsIgnoreCase("-o")) {
171                   i++;

173                       continue;
174

176               } else if (s.equalsIgnoreCase("-ofile")) {
177                   i++;
178                   if (i == arg.length)

182               } else if (s.equalsIgnoreCase("-graphvizdot") || s.equalsIgnoreCase("-graphviz_dot")) {
183                   i++;
184                   if (i == arg.length)
185                       continue;
186   
187                   GraphvizUtils.setDotExecutable(StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(arg[i]));
188               } else if (s.equalsIgnoreCase("-charset")) {
189                   i++;
190                   if (i == arg.length)
191                       continue;
192   
193                   charset = StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(arg[i]);
194               } else if (s.equalsIgnoreCase("-filename")) {
195                   i++;
196                   if (i == arg.length)
197                       continue;
198   
199                   filename = StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(arg[i]);
200               } else if (s.equalsIgnoreCase("-filedir")) {
201                   i++;
202                   if (i == arg.length)
203                       continue;
204   
205                   fileDir = StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(arg[i]);
206               } else if (s.startsWith("-o") && s.length() > 3) {
207                   s = s.substring(2);
208                   outputDir = new File(StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(s));
209               } else if (s.equalsIgnoreCase("-exclude") || s.equalsIgnoreCase("-x")) {
210                   i++;
211                   if (i == arg.length)

234   
235               } else if (s.equalsIgnoreCase("-failfast")) {
236                   this.failfast = true;

238                   this.failfast2 = true;
239               } else if (s.equalsIgnoreCase("-noerror")) {
240                   this.noerror = true;
241               } else if (s.equalsIgnoreCase("-checkonly")) {
242                   this.checkOnly = true;
243               } else if (s.equalsIgnoreCase("-theme")) {
244                   i++;
245                   if (i == arg.length)
246                       continue;
247   
248                   this.config.add(0, "!theme " + arg[i]);
249               } else if (s.equalsIgnoreCase("-config")) {
250                   i++;
251                   if (i == arg.length)
252                       continue;
253   
254                   initConfig(StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(arg[i]));
255               } else if (s.startsWith("-I")) {
256                   initInclude(s.substring(2));
257               } else if (s.equalsIgnoreCase("-computeurl") || s.equalsIgnoreCase("-encodeurl")) {
258                   this.computeurl = true;
259               } else if (s.startsWith("-x")) {
260                   s = s.substring(2);
261                   excludes.add(StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(s));
262               } else if (s.equalsIgnoreCase("-verbose") || s.equalsIgnoreCase("-v")) {
263                   OptionFlags.getInstance().setVerbose(true);
264               } else if (s.equalsIgnoreCase("-pipe") || s.equalsIgnoreCase("-p")) {

371                   stdrpt = 1;

406       public Stdrpt getStdrpt() {
407           if (stdrpt == 1)
408               return new StdrptV1();
409   
410           if (stdrpt == 2)
411               return new StdrptV2();
412   
413           // Legacy case
414           if (isPipe() || isPipeMap() || isSyntax())
415               return new StdrptPipe0();
416   
417           return new StdrptNull();
418       }
