55   /**
56    * Represents the textual source of some diagram. The source should start with a
57    * <code>@startfoo</code> and end with <code>@endfoo</code>.
58    * <p>
59    * So the diagram does not have to be a UML one.
60    *
61    * @author Arnaud Roques
62    *
63    */
64   final public class UmlSource {
65   
66       final private List<StringLocated> source;
67       final private List<StringLocated> rawSource;
68   
69       public UmlSource removeInitialSkinparam() {
70           if (hasInitialSkinparam(source) == false)
71               return this;
72   
73           final List<StringLocated> copy = new ArrayList<>(source);
74           while (hasInitialSkinparam(copy))
75               copy.remove(1);
76   
77           return new UmlSource(copy, rawSource);
78       }
79   
80       public boolean containsIgnoreCase(String searched) {
81           for (StringLocated s : source)
82               if (StringUtils.goLowerCase(s.getString()).contains(searched))
83                   return true;
84   
85           return false;
86       }
87   
88       private static boolean hasInitialSkinparam(final List<StringLocated> copy) {
89           return copy.size() > 1 && (copy.get(1).getString().startsWith("skinparam ")
90                   || copy.get(1).getString().startsWith("skinparamlocked "));
91       }
92   
93       private UmlSource(List<StringLocated> source, List<StringLocated> rawSource) {
94           this.source = source;
95           this.rawSource = rawSource;
96       }
97   
98       public static UmlSource create(List<StringLocated> source, boolean checkEndingBackslash) {
99           return createWithRaw(source, checkEndingBackslash, emptyArrayList());
100       }
101   
102       /**
103        * Build the source from a text.
104        * 
105        * @param source               the source of the diagram
106        * @param checkEndingBackslash <code>true</code> if an ending backslash means
107        *                             that a line has to be collapsed with the
108        *                             following one.
109        */
110       public static UmlSource createWithRaw(List<StringLocated> source, boolean checkEndingBackslash,
111               List<StringLocated> rawSource) {
112           final UmlSource result = new UmlSource(emptyArrayList(), rawSource);
113           result.loadInternal(source, checkEndingBackslash);
114           return result;
115       }
116   
117       private static ArrayList<StringLocated> emptyArrayList() {
118           return new ArrayList<>();
119       }
120   
121       private void loadInternal(List<StringLocated> source, boolean checkEndingBackslash) {
122           if (checkEndingBackslash) {
123               final StringBuilder pending = new StringBuilder();
124               for (StringLocated cs : source) {
125                   final String s = cs.getString();
126                   if (StringUtils.endsWithBackslash(s)) {
127                       pending.append(s, 0, s.length() - 1);
128                   } else {
129                       pending.append(s);
130                       this.source.add(new StringLocated(pending.toString(), cs.getLocation()));
131                       pending.setLength(0);
132                   }
133               }
134           } else {
135               this.source.addAll(source);
136           }
137       }

139       /**
140        * Retrieve the type of the diagram. This is based on the first line
141        * <code>@startfoo</code>.
142        *
143        * @return the type of the diagram.
144        */
145       public DiagramType getDiagramType() {
146           return DiagramType.getTypeFromArobaseStart(source.get(0).getString());
147       }

155           return new IteratorCounter2Impl(source);
156       }

162       /**
163        * Return the source as a single String with <code>\n</code> as line separator.
164        *
165        * @return the whole diagram source
166        */
167       public String getPlainString() {
168           final StringBuilder sb = new StringBuilder();
169           for (StringLocated s : source) {
170               sb.append(s.getString());
171               sb.append('\r');
172               sb.append(BackSlash.CHAR_NEWLINE);
173           }
174           return sb.toString();
175       }
176   
177       public String getRawString() {
178           final StringBuilder sb = new StringBuilder();
179           for (StringLocated s : rawSource) {
180               sb.append(s.getString());
181               sb.append('\r');
182               sb.append(BackSlash.CHAR_NEWLINE);
183           }
184           return sb.toString();
185       }

187       public long seed() {
188           return StringUtils.seed(getPlainString());
189       }

191       public String getLine(LineLocation n) {
192           for (StringLocated s : source)
193               if (s.getLocation().compareTo(n) == 0)
194                   return s.getString();
195   
196           return null;
197       }

200        * Return the number of line in the diagram.

202       public int getTotalLineCount() {
203           return source.size();
204       }

207           return getTotalLineCount() < 5;
208       }

210       /**
211        * Check if a source diagram description is empty. Does not take comment line
212        * into account.
213        *
214        * @return <code>true</code> if the diagram does not contain information.
215        */
216       public boolean isEmpty() {
217           for (StringLocated s : source) {
218               if (StartUtils.isArobaseStartDiagram(s.getString()))
219                   continue;
220   
221               if (StartUtils.isArobaseEndDiagram(s.getString()))
222                   continue;
223   
224               if (s.getString().matches("\\s*'.*"))
225                   continue;
226   
227               if (StringUtils.trin(s.getString()).length() != 0)
228                   return false;
229   
230           }
231           return true;
232       }
233   
234       /**
235        * Retrieve the title, if defined in the diagram source. Never return
236        * <code>null</code>.
237        */
238       public Display getTitle() {
239           final Pattern2 p = MyPattern.cmpile("^[%s]*title[%s]+(.+)$");
240           for (StringLocated s : source) {
241               final Matcher2 m = p.matcher(s.getString());
242               final boolean ok = m.matches();
243               if (ok)
244                   return Display.create(m.group(1));
245           }
246           return Display.empty();
247       }
248   
249       public boolean isStartDef() {
250           return source.get(0).getString().startsWith("@startdef");
251       }
252   
253       public String getId() {
254           final Pattern p = Pattern.compile("id=([\\w]+)\\b");
255           final Matcher m = p.matcher(source.get(0).getString());
256           if (m.find())
257               return m.group(1);
258   
259           return null;
260       }
