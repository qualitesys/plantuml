273       public HColor calculateBackColor() {
274           final Style style = StyleSignatureBasic.of(SName.root, SName.document, this.getUmlDiagramType().getStyleName())
275                   .getMergedStyle(this.getSkinParam().getCurrentStyleBuilder());
276   
277           HColor backgroundColor = style.value(PName.BackGroundColor).asColor(this.getSkinParam().getIHtmlColorSet());
278           if (backgroundColor == null)
279               backgroundColor = HColors.transparent();
280   
281           return backgroundColor;
282       }

284       @Override
285       protected ColorMapper muteColorMapper(ColorMapper init) {
286           if ("dark".equalsIgnoreCase(getSkinParam().getValue("mode")))
287               return ColorMapper.DARK_MODE;
288           final String monochrome = getSkinParam().getValue("monochrome");
289           if ("true".equals(monochrome))
290               return ColorMapper.MONOCHROME;
291           if ("reverse".equals(monochrome))
292               return ColorMapper.MONOCHROME_REVERSE;
293   
294           final String reversecolor = getSkinParam().getValue("reversecolor");
295           if (reversecolor == null)
296               return init;
297   
298           if ("dark".equalsIgnoreCase(reversecolor))
299               return ColorMapper.LIGTHNESS_INVERSE;
300   
301           final ColorOrder order = ColorOrder.fromString(reversecolor);
302           if (order == null)
303               return init;
304   
305           return ColorMapper.reverse(order);
306   
307       }
