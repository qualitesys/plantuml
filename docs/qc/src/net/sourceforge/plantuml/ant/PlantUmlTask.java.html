102       // The method executing the task
103       @Override
104       public void execute() throws BuildException {
105   
106           if (option.isSplash()) {
107               Splash.createSplash();
108           }
109   
110           this.log("Starting PlantUML");
111   
112           try {
113               if (dir != null) {
114                   final File error = processingSingleDirectory(new File(dir));
115                   eventuallyFailfast(error);
116               }
117               for (FileSet fileSet : filesets) {
118                   final File error = manageFileSet(fileSet);
119                   eventuallyFailfast(error);
120               }
121               for (FileList fileList : filelists) {
122                   final File error = manageFileList(fileList);
123                   eventuallyFailfast(error);
124               }
125               if (executorService != null) {
126                   executorService.shutdown();
127                   executorService.awaitTermination(Long.MAX_VALUE, TimeUnit.MILLISECONDS);
128                   if (option.isSplash()) {
129                       Splash.disposeSplash();
130                   }
131               }
132               this.log("Nb images generated: " + nbFiles.get());
133           } catch (IOException e) {
134               Logme.error(e);
135               throw new BuildException(e.toString());
136           } catch (InterruptedException e) {
137               Logme.error(e);
138               throw new BuildException(e.toString());
139           }
140   
141       }

143       private void eventuallyFailfast(final File error) throws IOException {
144           if (error != null && option.isFailfastOrFailfast2()) {
145               this.log("Error in file " + error.getCanonicalPath());
146               throw new BuildException("Error in file " + error.getCanonicalPath());
147           }
148       }

150       private File manageFileList(FileList fl) throws IOException, InterruptedException {
151           final File fromDir = fl.getDir(getProject());
152   
153           final String[] srcFiles = fl.getFiles(getProject());
154   
155           for (String src : srcFiles) {
156               final File f = new File(fromDir, src);
157               final boolean error = processingSingleFile(f);
158               if (error) {
159                   return f;
160               }
161           }
162           return null;
163       }
164   
165       private File manageFileSet(FileSet fs) throws IOException, InterruptedException {
166           final DirectoryScanner ds = fs.getDirectoryScanner(getProject());
167           final File fromDir = fs.getDir(getProject());
168   
169           final String[] srcFiles = ds.getIncludedFiles();
170           final String[] srcDirs = ds.getIncludedDirectories();
171   
172           for (String src : srcFiles) {
173               final File f = new File(fromDir, src);
174               final boolean error = processingSingleFile(f);
175               if (error) {
176                   return f;
177               }
178           }
179   
180           for (String src : srcDirs) {
181               final File dir = new File(fromDir, src);
182               final File errorFile = processingSingleDirectory(dir);
183               if (errorFile != null) {
184                   return errorFile;
185               }
186           }
187           return null;
188   
189       }

191       private boolean processingSingleFile(final File f) throws IOException, InterruptedException {
192           if (OptionFlags.getInstance().isVerbose()) {
193               this.log("Processing " + f.getAbsolutePath());
194           }
195           final SourceFileReader sourceFileReader = new SourceFileReader(Defines.createWithFileName(f), f,
196                   option.getOutputDir(), option.getConfig(), option.getCharset(), option.getFileFormatOption());
197   
198           if (option.isCheckOnly()) {
199               return sourceFileReader.hasError();
200           }
201           if (executorService == null) {
202               return doFile(f, sourceFileReader);
203           }
204   
205           Splash.incTotal(1);
206           executorService.submit(new Callable<Boolean>() {
207               public Boolean call() throws Exception {
208                   return doFile(f, sourceFileReader);
209               }
210           });
211   
212           return false;
213       }
214   
215       private boolean doFile(final File f, final SourceFileReader sourceFileReader)
216               throws IOException, InterruptedException {
217           final Collection<GeneratedImage> result = sourceFileReader.getGeneratedImages();
218           boolean error = false;
219           for (GeneratedImage g : result) {
220               if (OptionFlags.getInstance().isVerbose()) {
221                   myLog(g + " " + g.getDescription());
222               }
223               nbFiles.addAndGet(1);
224               if (g.lineErrorRaw() != -1) {
225                   error = true;
226               }
227           }
228           Splash.incDone(error);
229           if (error) {
230               myLog("Error: " + f.getCanonicalPath());
231           }
232           if (error && option.isFailfastOrFailfast2()) {
233               return true;
234           }
235           return false;
236       }

238       private synchronized void myLog(String s) {
239           this.log(s);
240       }

242       private File processingSingleDirectory(File dir) throws IOException, InterruptedException {
243           if (dir.exists() == false) {
244               final String s = "The file " + dir.getAbsolutePath() + " does not exists.";
245               this.log(s);
246               throw new BuildException(s);
247           }
248           if (dir.listFiles() != null)
249               for (File f : dir.listFiles()) {
250                   if (f.isFile() == false)
251                       continue;
252   
253                   if (fileToProcess(f.getName()) == false)
254                       continue;
255   
256                   final boolean error = processingSingleFile(f);
257                   if (error)
258                       return f;
259   
260               }
261           return null;
262       }
