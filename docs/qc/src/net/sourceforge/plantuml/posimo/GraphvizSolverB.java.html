88       public XDimension2D solve(Cluster root, Collection<Path> paths) throws IOException {
89           final String dotString = new DotxMaker(root, paths).createDotString("nodesep=0.2;", "ranksep=0.2;");
90   
91           // if (OptionFlags.getInstance().isKeepTmpFiles()) {
92           // traceDotString(dotString);
93           // }
94   
95           final MinFinder minMax = new MinFinder();
96   
97           // Log.println("dotString=" + dotString);
98   
99           // exportPng(dotString, SecurityUtils.File("png", "test1.png"));
100   
101           final Graphviz graphviz = GraphvizUtils.create(null, dotString, "svg");
102           final ByteArrayOutputStream baos = new ByteArrayOutputStream();
103           final ProcessState state = graphviz.createFile3(baos);
104           baos.close();
105           if (state.differs(ProcessState.TERMINATED_OK())) {
106               throw new IllegalStateException("Timeout2 " + state);
107           }
108           final byte[] result = baos.toByteArray();
109           final String s = new String(result, UTF_8);
110           // Log.println("result=" + s);
111   
112           // if (OptionFlags.getInstance().isKeepTmpFiles()) {
113           // traceSvgString(s);
114           // }
115   
116           final Pattern pGraph = Pattern.compile("(?m)\\<svg\\s+width=\"(\\d+)pt\"\\s+height=\"(\\d+)pt\"");
117           final Matcher mGraph = pGraph.matcher(s);
118           if (mGraph.find() == false) {
119               throw new IllegalStateException();
120           }
121           final int width = Integer.parseInt(mGraph.group(1));
122           final int height = Integer.parseInt(mGraph.group(2));
123   
124           final YDelta yDelta = new YDelta(height);
125           for (Block b : root.getRecursiveContents()) {
126               final String start = "b" + b.getUid();
127               final int p1 = s.indexOf("<title>" + start + "</title>");
128               if (p1 == -1) {
129                   throw new IllegalStateException();
130               }
131               final List<XPoint2D> pointsList = extractPointsList(s, p1, yDelta);
132               b.setX(getMinX(pointsList));
133               b.setY(getMinY(pointsList));
134               minMax.manage(b.getPosition());
135           }
136   
137           for (Cluster cl : root.getSubClusters()) {
138               final String start = "cluster" + cl.getUid();
139               final int p1 = s.indexOf("<title>" + start + "</title>");
140               if (p1 == -1) {
141                   throw new IllegalStateException();
142               }
143               final List<XPoint2D> pointsList = extractPointsList(s, p1, yDelta);
144               cl.setX(getMinX(pointsList));
145               cl.setY(getMinY(pointsList));
146               final double w = getMaxX(pointsList) - getMinX(pointsList);
147               final double h = getMaxY(pointsList) - getMinY(pointsList);
148               cl.setHeight(h);
149               cl.setWidth(w);
150               minMax.manage(cl.getPosition());
151           }
152   
153           for (Path p : paths) {
154               final String start = "b" + p.getStart().getUid();
155               final String end = "b" + p.getEnd().getUid();
156               final String searched = "<title>" + start + "&#45;&gt;" + end + "</title>";
157               final int p1 = s.indexOf(searched);
158               if (p1 == -1) {
159                   throw new IllegalStateException(searched);
160               }
161               final int p2 = s.indexOf(" d=\"", p1);
162               final int p3 = s.indexOf("\"", p2 + " d=\"".length());
163               final String points = s.substring(p2 + " d=\"".length(), p3);
164               final DotPath dotPath = new DotPath(new SvgResult(points, yDelta));
165               p.setDotPath(dotPath);
166               minMax.manage(dotPath.getMinFinder());
167   
168               // Log.println("pointsList=" + pointsList);
169               if (p.getLabel() != null) {
170                   final List<XPoint2D> pointsList = extractPointsList(s, p1, yDelta);
171                   final double x = getMinX(pointsList);
172                   final double y = getMinY(pointsList);
173                   p.setLabelPosition(x, y);
174                   minMax.manage(x, y);
175               }
176           }
177           return new XDimension2D(width, height);
178       }

180       static private List<XPoint2D> extractPointsList(final String svg, final int starting, final YDelta yDelta) {
181           return new SvgResult(svg, yDelta).substring(starting).extractList(SvgResult.POINTS_EQUALS);
182       }

184       static private double getMaxX(List<XPoint2D> points) {
185           double result = points.get(0).x;
186           for (int i = 1; i < points.size(); i++) {
187               if (points.get(i).x > result) {
188                   result = points.get(i).x;
189               }
190           }
191           return result;
192       }

225           final Graphviz graphviz = GraphvizUtils.create(null, dotString, "png");
226           try (OutputStream os = f.createBufferedOutputStream()) {
227               final ProcessState state = graphviz.createFile3(os);
228               if (state.differs(ProcessState.TERMINATED_OK())) {
229                   throw new IllegalStateException("Timeout3 " + state);
230               }
231           }

235           for (Path p : paths) {
236               if (p.getStart().getUid() == start && p.getEnd().getUid() == end) {
237                   return p;
238               }
239           }
240           throw new IllegalArgumentException();
