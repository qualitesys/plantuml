66       /**
67        * Returns the square of the distance from a point to a line segment. The
68        * distance measured is the distance between the specified point and the closest
69        * point between the specified end points. If the specified point intersects the
70        * line segment in between the end points, this method returns 0.0.
71        *
72        * @param x1 the X coordinate of the start point of the specified line segment
73        * @param y1 the Y coordinate of the start point of the specified line segment
74        * @param x2 the X coordinate of the end point of the specified line segment
75        * @param y2 the Y coordinate of the end point of the specified line segment
76        * @param px the X coordinate of the specified point being measured against the
77        *           specified line segment
78        * @param py the Y coordinate of the specified point being measured against the
79        *           specified line segment
80        * @return a double value that is the square of the distance from the specified
81        *         point to the specified line segment.
82        * @see #ptLineDistSq(double, double, double, double, double, double)
83        * @since 1.2
84        */
85       public static double ptSegDistSq(double x1, double y1, double x2, double y2, double px, double py) {
86           // Adjust vectors relative to x1,y1
87           // x2,y2 becomes relative vector from x1,y1 to end of segment
88           x2 -= x1;
89           y2 -= y1;
90           // px,py becomes relative vector from x1,y1 to test point
91           px -= x1;
92           py -= y1;
93           double dotprod = px * x2 + py * y2;
94           double projlenSq;
95           if (dotprod <= 0.0) {
96               // px,py is on the side of x1,y1 away from x2,y2
97               // distance to segment is length of px,py vector
98               // "length of its (clipped) projection" is now 0.0
99               projlenSq = 0.0;
100           } else {
101               // switch to backwards vectors relative to x2,y2
102               // x2,y2 are already the negative of x1,y1=>x2,y2
103               // to get px,py to be the negative of px,py=>x2,y2
104               // the dot product of two negated vectors is the same
105               // as the dot product of the two normal vectors
106               px = x2 - px;
107               py = y2 - py;
108               dotprod = px * x2 + py * y2;
109               if (dotprod <= 0.0) {
110                   // px,py is on the side of x2,y2 away from x1,y1
111                   // distance to segment is length of (backwards) px,py vector
112                   // "length of its (clipped) projection" is now 0.0
113                   projlenSq = 0.0;
114               } else {
115                   // px,py is between x1,y1 and x2,y2
116                   // dotprod is the length of the px,py vector
117                   // projected on the x2,y2=>x1,y1 vector times the
118                   // length of the x2,y2=>x1,y1 vector
119                   projlenSq = dotprod * dotprod / (x2 * x2 + y2 * y2);
120               }
121           }
122           // Distance to line is now the length of the relative point
123           // vector minus the length of its projection onto the line
124           // (which is zero if the projection falls outside the range
125           // of the line segment).
126           double lenSq = px * px + py * py - projlenSq;
127           if (lenSq < 0) {
128               lenSq = 0;
129           }
130           return lenSq;
131       }
