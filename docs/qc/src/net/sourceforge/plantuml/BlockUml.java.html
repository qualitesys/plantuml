105       public static List<StringLocated> convert(List<String> strings) {
106           final List<StringLocated> result = new ArrayList<>();
107           LineLocationImpl location = new LineLocationImpl("block", null);

126       public BlockUml(List<StringLocated> strings, Defines defines, ISkinSimple skinParam, PreprocessorModeSet mode,
127               Charset charset) {
128           this.rawSource = new ArrayList<>(strings);
129           this.localDefines = defines;
130           this.skinParam = skinParam;
131           final String s0 = strings.get(0).getTrimmed().getString();
132           if (StartUtils.startsWithSymbolAnd("start", s0) == false)
133               throw new IllegalArgumentException();
134   
135           if (mode == null) {
136               this.data = new ArrayList<>(strings);
137           } else {
138               final TimLoader timLoader = new TimLoader(mode.getImportedFiles(), defines, charset,
139                       (DefinitionsContainer) mode);
140               this.included.addAll(timLoader.load(strings));
141               this.data = timLoader.getResultList();
142               this.debug = timLoader.getDebug();
143               this.preprocessorError = timLoader.isPreprocessorError();
144           }
145       }

147       public String getFileOrDirname() {
148           if (OptionFlags.getInstance().isWord())
149               return null;
150   
151           final Matcher2 m = StartUtils.patternFilename.matcher(StringUtils.trin(data.get(0).getString()));
152           final boolean ok = m.find();
153           if (ok == false)
154               return null;
155   
156           String result = m.group(1);
157           final int x = result.indexOf(',');
158           if (x != -1)
159               result = result.substring(0, x);
160   
161           for (int i = 0; i < result.length(); i++) {
162               final char c = result.charAt(i);
163               if ("<>|".indexOf(c) != -1)
164                   return null;
165   
166           }
167           if (result.startsWith("file://"))
168               result = result.substring("file://".length());
169   
170           result = result.replaceAll("\\.\\w\\w\\w$", "");
171           return result;
172       }

178               else {
179                   system = new PSystemBuilder().createPSystem(data, rawSource,
180                           skinParam == null ? Collections.<String, String>emptyMap() : skinParam.values());
181               }
182           }
183           return system;
184       }

190       private String internalEtag() {
191           try {
192               final AsciiEncoder coder = new AsciiEncoder();
193               final MessageDigest msgDigest = MessageDigest.getInstance("MD5");
194               for (StringLocated s : data)
195                   msgDigest.update(s.getString().getBytes(UTF_8));
196   
197               final byte[] digest = msgDigest.digest();
198               return coder.encode(digest);
199           } catch (Exception e) {
200               Logme.error(e);
201               return "NOETAG";
202           }
203       }
