257       public Cypher getCypher() {
258           final Cypher cypher = new Cypher();
259           for (String s : type) {
260               cypher.addException(s);
261           }
262           for (String s : keyword) {
263               cypher.addException(s.replace("@", ""));
264           }
265           for (String s : preproc) {
266               cypher.addException(s.substring(1));
267           }
268           for (String s : SkinParam.getPossibleValues()) {
269               cypher.addException(s);
270           }
271           for (String s : HColorSet.instance().names()) {
272               cypher.addException(s);
273           }
274           cypher.addException("o");
275           return cypher;
276       }

278       public void print(PrintStream ps) {
279           print(ps, "type", type);
280           print(ps, "keyword", keyword);
281           print(ps, "preprocessor", preproc);
282           print(ps, "skinparameter", SkinParam.getPossibleValues());
283           print(ps, "color", HColorSet.instance().names());
284           ps.println(";EOF");
285       }
286   
287       private static void print(PrintStream ps, String name, Collection<String> data) {
288           ps.println(";" + name);
289           ps.println(";" + data.size());
290           for (String k : data) {
291               ps.println(k);
292           }
293           ps.println();
294       }
