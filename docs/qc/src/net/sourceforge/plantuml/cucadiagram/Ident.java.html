58       public boolean startsWith(Ident other) {
59           if (other.parts.size() > this.parts.size()) {
60               return false;
61           }
62           for (int i = 0; i < other.parts.size(); i++) {
63               if (other.parts.get(i).equals(this.parts.get(i)) == false) {
64                   return false;
65               }
66           }
67           return true;
68       }

151       static private Ident from(String full, String separator) {
152           final Ident result = new Ident(new ArrayList<String>());
153           if (isOdd(full, separator)) {
154               result.parts.add(full);
155               return result;
156           }
157           while (true) {
158               int idx = full.indexOf(separator);
159               if (idx == -1) {
160                   result.parts.add(full);
161                   result.checkResult(separator);
162                   return result;
163               }
164               if (idx > 0) {
165                   result.parts.add(full.substring(0, idx));
166               }
167               full = full.substring(idx + separator.length());
168           }
169       }

171       private static boolean isOdd(String full, String separator) {
172           return separator == null || full.contains(separator + separator) || full.endsWith(separator);
173       }
174   
175       private void checkResult(String separator) {
176           for (String s : this.parts) {
177               if (s.length() == 0) {
178                   throw new IllegalStateException(toString());
179               }
180               if (separator != null && s.contains(separator) && s.endsWith(separator) == false
181                       && s.contains(separator + separator) == false) {
182                   throw new IllegalStateException(toString());
183               }
184           }
185       }

187       public Ident add(String sup, String separator) {
188           this.checkResult(separator);
189           final Ident added = from(sup, separator);
190           final List<String> list = new ArrayList<>(this.parts.size() + added.parts.size());
191           list.addAll(this.parts);
192           list.addAll(added.parts);
193           final Ident result = new Ident(list);
194           result.checkResult(separator);
195           return result;
196       }

226           }
227           return sb.toString();
228       }

230       public void checkSameAs(Code code, String separator, CucaDiagram diagram) {
231           if (diagram.V1972()) {
232               return;
233           }
234           final String last = parts.get(parts.size() - 1);
235           if (separator == null) {
236               if (code.getName().equals(last) != true && code.getName().equals(toString(separator)) == false) {
237   //                System.err.println("code1=" + code);
238   //                System.err.println("this1=" + this);
239                   EntityFactory.bigError();
240               }
241           } else {
242               if (getLastPart(code.getName(), separator).equals(last) != true
243                       && code.getName().equals(toString(separator)) == false) {
244   //                System.err.println("code2=" + code);
245   //                System.err.println("this2=" + this);
246                   EntityFactory.bigError();
247               }
248           }
249       }

251       private String getLastPart(String fullName, String separator) {
252           if (separator == null) {
253               return fullName;
254           }
255           final int x = fullName.lastIndexOf(separator);
256           if (x == -1) {
257               return fullName;
258           }
259           return fullName.substring(x + separator.length());
260       }

274       public Ident move(Ident from, Ident to) {
275           if (this.startsWith(from) == false) {
276               throw new IllegalArgumentException();
277           }
278           final List<String> result = new ArrayList<>(to.parts);
279           for (int i = from.parts.size(); i < this.parts.size(); i++) {
280               result.add(this.parts.get(i));
281           }
282           return new Ident(result);
283       }
