180       public static Display getWithNewlines(String s) {
181           if (s == null)
182               return NULL;
183   
184           final List<String> result = new ArrayList<>();
185           final StringBuilder current = new StringBuilder();
186           HorizontalAlignment naturalHorizontalAlignment = null;
187           boolean rawMode = false;
188           for (int i = 0; i < s.length(); i++) {
189               final char c = s.charAt(i);
190               final String sub = s.substring(i);
191               if (sub.startsWith("<math>") || sub.startsWith("<latex>") || sub.startsWith("[["))
192                   rawMode = true;
193               else if (sub.startsWith("</math>") || sub.startsWith("</latex>") || sub.startsWith("]]"))
194                   rawMode = false;
195   
196               if (rawMode == false && c == '\\' && i < s.length() - 1) {
197                   final char c2 = s.charAt(i + 1);
198                   i++;
199                   if (c2 == 'n' || c2 == 'r' || c2 == 'l') {
200                       if (c2 == 'r')
201                           naturalHorizontalAlignment = HorizontalAlignment.RIGHT;
202                       else if (c2 == 'l')
203                           naturalHorizontalAlignment = HorizontalAlignment.LEFT;
204   
205                       result.add(current.toString());
206                       current.setLength(0);
207                   } else if (c2 == 't') {
208                       current.append('\t');
209                   } else if (c2 == '\\') {
210                       current.append(c2);
211                   } else {
212                       current.append(c);
213                       current.append(c2);
214                   }
215               } else if (c == BackSlash.hiddenNewLine()) {
216                   result.add(current.toString());
217                   current.setLength(0);
218               } else {
219                   current.append(c);
220               }
221           }
222           result.add(current.toString());
223           return new Display(true, result, naturalHorizontalAlignment, false, CreoleMode.FULL);
224       }

248       private static List<CharSequence> manageEmbeddedDiagrams(final Collection<? extends CharSequence> strings) {
249           final List<CharSequence> result = new ArrayList<>();
250           final Iterator<? extends CharSequence> it = strings.iterator();
251           while (it.hasNext()) {
252               CharSequence s = it.next();
253               final String type = EmbeddedDiagram.getEmbeddedType(s);
254               if (type != null) {
255                   final List<CharSequence> other = new ArrayList<>();
256                   other.add("@start" + type);
257                   while (it.hasNext()) {
258                       final CharSequence s2 = it.next();
259                       if (s2 != null && StringUtils.trin(s2.toString()).equals("}}"))
260                           break;
261   
262                       other.add(s2);
263                   }
264                   other.add("@end" + type);
265                   s = new EmbeddedDiagram(Display.create(other));
266               }
267               result.add(s);
268           }
269           return result;
270       }

455       public List<Display> splitMultiline(Pattern2 separator) {
456           final List<Display> result = new ArrayList<>();
457           Display pending = new Display(showStereotype, this.naturalHorizontalAlignment, this.isNull,
458                   this.defaultCreoleMode);
459           result.add(pending);
460           for (CharSequence line : displayData) {
461               final Matcher2 m = separator.matcher(line);
462               if (m.find()) {
463                   final CharSequence s1 = line.subSequence(0, m.start());
464                   pending.displayData.add(s1);
465                   final CharSequence s2 = line.subSequence(m.end(), line.length());
466                   pending = new Display(showStereotype, this.naturalHorizontalAlignment, this.isNull,
467                           this.defaultCreoleMode);
468                   result.add(pending);
469                   pending.displayData.add(s2);
470               } else {
471                   pending.displayData.add(line);
472               }
473           }
474           return Collections.unmodifiableList(result);
475       }

600           if (all.size() <= 1)

602   
603           for (CharSequence cs : all) {
604               final String s = cs.toString();
605               if (s.startsWith("< "))
606                   return true;
607   
608               if (s.startsWith("> "))
609                   return true;
610

616           }
