213               if (leafType != LeafType.ANNOTATION && leafType != LeafType.ABSTRACT_CLASS && leafType != LeafType.CLASS
214                       && leafType != LeafType.ENUM && leafType != LeafType.INTERFACE) {
215                   return false;
216                   // throw new IllegalArgumentException("type=" + leafType);
217               }
218               if (newType != LeafType.ANNOTATION && newType != LeafType.ABSTRACT_CLASS && newType != LeafType.CLASS
219                       && newType != LeafType.ENUM && newType != LeafType.INTERFACE && newType != LeafType.OBJECT) {
220                   return false;

228           this.symbol = newSymbol;
