74       public static Graphviz createForSystemDot(ISkinParam skinParam, String dotString, String... type) {
75           if (useVizJs(skinParam)) {
76               Log.info("Using " + VIZJS);
77               return new GraphvizJs(dotString);
78           }
79           final AbstractGraphviz result;
80           if (isWindows())
81               result = new GraphvizWindowsOld(skinParam, dotString, type);
82           else
83               result = new GraphvizLinux(skinParam, dotString, type);
84   
85           if (result.getExeState() != ExeState.OK && VizJsEngine.isOk()) {
86               Log.info("Error with file " + result.getDotExe() + ": " + result.getExeState().getTextMessage());
87               Log.info("Using " + VIZJS);
88               return new GraphvizJs(dotString);
89           }
90           return result;
91       }

96               return new GraphvizJs(dotString);

99           if (isWindows())

103

106               Log.info("Using " + VIZJS);
107               return new GraphvizJs(dotString);
108           }
109           return result;

112       private static boolean useVizJs(ISkinParam skinParam) {
113           if (skinParam != null && skinParam.isUseVizJs() && VizJsEngine.isOk())
114               return true;
115   
116           if (VIZJS.equalsIgnoreCase(getenvGraphvizDot()) && VizJsEngine.isOk())
117               return true;
118   
119           return false;

133

135           if (StringUtils.isNotEmpty(getenv))
136               return StringUtils.eventuallyRemoveStartingAndEndingDoubleQuote(getenv);

138           return null;

173       public static String dotVersion() throws IOException, InterruptedException {
174           if (dotVersion == null) {
175               final File dotExe = GraphvizUtils.getDotExe();
176               final ExeState exeState = ExeState.checkFile(dotExe);
177               if (exeState == ExeState.OK)
178                   dotVersion = create(null, "png").dotVersion();
179               else
180                   dotVersion = "Error:" + exeState.getTextMessage(dotExe);
181   
182           }
183           return dotVersion;
184       }

195       public static int retrieveVersion(String s) {
196           if (s == null)
197               return -1;
198   
199           final Pattern p = Pattern.compile("\\s(\\d)\\.(\\d\\d?)\\D");
200           final Matcher m = p.matcher(s);
201           if (m.find() == false)
202               return -1;
203   
204           return 100 * Integer.parseInt(m.group(1)) + Integer.parseInt(m.group(2));
205       }

211       static public int addDotStatus(List<String> result, boolean withRichText) {
212           String red = "";
213           String bold = "";
214           if (withRichText) {
215               red = "<b><color:red>";
216               bold = "<b>";
217           }
218   
219           int error = 0;
220   
221           if (useVizJs(null)) {
222               result.add("VizJs library is used!");
223               try {
224                   final String err = getTestCreateSimpleFile();
225                   if (err == null)
226                       result.add(bold + "Installation seems OK. File generation OK");
227                   else
228                       result.add(red + err);
229   
230               } catch (Exception e) {
231                   result.add(red + e.toString());
232                   Logme.error(e);
233                   error = -1;
234               }
235               return error;
236           }
237   
238           final File dotExe = GraphvizUtils.getDotExe();
239           if (SecurityUtils.getSecurityProfile() == SecurityProfile.UNSECURE) {
240               final String ent = GraphvizUtils.getenvGraphvizDot();
241               if (ent == null)
242                   result.add("The environment variable GRAPHVIZ_DOT has not been set");
243               else
244                   result.add("The environment variable GRAPHVIZ_DOT has been set to " + ent);
245   
246               result.add("Dot executable is " + dotExe);
247           }
248           final ExeState exeState = ExeState.checkFile(dotExe);
249   
250           if (exeState == ExeState.OK) {
251               try {
252                   final String version = GraphvizUtils.dotVersion();
253                   result.add("Dot version: " + version);
254                   final int v = GraphvizUtils.getDotVersion();
255                   if (v == -1) {
256                       result.add("Warning : cannot determine dot version");
257                       error = -2;
258                   } else if (v < DOT_VERSION_LIMIT) {
259                       result.add(bold + "Warning : Your dot installation seems old");
260                       result.add(bold + "Some diagrams may have issues");
261                       error = -3;
262                   } else {
263                       final String err = getTestCreateSimpleFile();
264                       if (err == null) {
265                           result.add(bold + "Installation seems OK. File generation OK");
266                       } else {
267                           result.add(red + err);
268                           error = -4;
269                       }
270                   }
271               } catch (Exception e) {
272                   result.add(red + e.toString());
273                   Logme.error(e);
274                   error = -5;
275               }
276           } else {
277               result.add(red + "Error: " + exeState.getTextMessage());
278               result.add("Error: only sequence diagrams will be generated");
279               error = -6;
280           }
281   
282           return error;
283       }

285       static String getTestCreateSimpleFile() throws IOException {
286           final Graphviz graphviz2 = GraphvizUtils.create(null, "digraph foo { test; }", "svg");
287           final ByteArrayOutputStream baos = new ByteArrayOutputStream();
288           final ProcessState state = graphviz2.createFile3(baos);
289           if (state.differs(ProcessState.TERMINATED_OK()))
290               return "Error: timeout " + state;
291   
292           final byte data[] = baos.toByteArray();
293   
294           if (data.length == 0)
295               return "Error: dot generates empty file. Check you dot installation.";
296   
297           final String s = new String(data);
298           if (s.indexOf("<svg") == -1)
299               return "Error: dot generates unreadable SVG file. Check you dot installation.";
300   
301           return null;
302       }
