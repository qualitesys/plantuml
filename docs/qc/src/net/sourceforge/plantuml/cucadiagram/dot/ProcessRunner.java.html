50   public class ProcessRunner {
51   
52       private final String[] cmd;
53   
54       private String error;
55       private String out;
56   
57       private volatile ProcessState state = ProcessState.INIT();
58       private final Lock changeState = new ReentrantLock();
59   
60       public ProcessRunner(String[] cmd) {
61           this.cmd = cmd;
62       }
63   
64       public ProcessState run(byte in[], OutputStream redirection) {
65           return run(in, redirection, null);
66       }
67   
68       public ProcessState run(byte in[], OutputStream redirection, SFile dir) {
69           if (this.state.differs(ProcessState.INIT())) {
70               throw new IllegalStateException();
71           }
72           this.state = ProcessState.RUNNING();
73           final MainThread mainThread = new MainThread(cmd, dir, redirection, in);
74           try {
75               // http://steveliles.github.io/invoking_processes_from_java.html
76               final long timeoutMs = OptionFlags.getInstance().getTimeoutMs();
77               final boolean done = new TimeoutExecutor(timeoutMs).executeNow(mainThread);
78           } finally {
79               changeState.lock();
80               try {
81                   if (state.equals(ProcessState.RUNNING())) {
82                       state = ProcessState.TIMEOUT();
83                       // mainThread.cancel();
84                   }
85               } finally {
86                   changeState.unlock();
87               }
88           }
89           if (state.equals(ProcessState.TERMINATED_OK())) {
90               assert mainThread != null;
91               this.error = mainThread.getError();
92               this.out = mainThread.getOut();
93           }
94           return state;
95       }

97       class MainThread implements MyRunnable {
98   
99           private final String[] cmd;
100           private final SFile dir;
101           private final OutputStream redirection;
102           private final byte[] in;
103           private volatile Process process;
104           private volatile ThreadStream errorStream;
105           private volatile ThreadStream outStream;
106   
107           public MainThread(String[] cmd, SFile dir, OutputStream redirection, byte[] in) {
108               this.cmd = cmd;
109               this.dir = dir;
110               this.redirection = redirection;
111               this.in = in;
112           }
113   
114           public String getOut() {
115               return outStream.getString();
116           }
117   
118           public String getError() {
119               return errorStream.getString();
120           }
121   
122           public void runJob() throws InterruptedException {
123               try {
124                   startThreads();
125                   if (state.equals(ProcessState.RUNNING())) {
126                       final int result = joinInternal();
127                   }
128               } finally {
129                   changeState.lock();
130                   try {
131                       if (state.equals(ProcessState.RUNNING())) {
132                           state = ProcessState.TERMINATED_OK();
133                       }
134                   } finally {
135                       changeState.unlock();
136                   }
137                   if (process != null) {
138                       process.destroy();
139                       close(process.getErrorStream());
140                       close(process.getOutputStream());
141                       close(process.getInputStream());
142                   }
143               }
144   
145           }
146   
147           public void cancelJob() {
148               // The changeState lock is ok
149               // assert changeState.tryLock();
150               // assert state == ProcessState.TIMEOUT;
151               if (process != null) {
152                   errorStream.cancel();
153                   outStream.cancel();
154                   process.destroy();
155                   // interrupt();
156                   close(process.getErrorStream());
157                   close(process.getOutputStream());
158                   close(process.getInputStream());
159               }
160           }
161   
162           private void startThreads() {
163               try {
164                   process = Runtime.getRuntime().exec(cmd, null, dir == null ? null : dir.conv());
165               } catch (IOException e) {
166                   Logme.error(e);
167                   changeState.lock();
168                   try {
169                       state = ProcessState.IO_EXCEPTION1(e);
170                   } finally {
171                       changeState.unlock();
172                   }
173                   Logme.error(e);
174                   return;
175               }
176               errorStream = new ThreadStream(process.getErrorStream(), null);
177               outStream = new ThreadStream(process.getInputStream(), redirection);
178               errorStream.start();
179               outStream.start();
180               if (in != null) {
181                   final OutputStream os = process.getOutputStream();
182                   try {
183                       try {
184                           os.write(in);
185                       } finally {
186                           os.close();
187                       }
188                   } catch (IOException e) {
189                       changeState.lock();
190                       try {
191                           state = ProcessState.IO_EXCEPTION2(e);
192                       } finally {
193                           changeState.unlock();
194                       }
195                       Logme.error(e);
196                   }
197               }
198           }
199   
200           public int joinInternal() throws InterruptedException {
201               errorStream.join();
202               outStream.join();
203               final int result = process.waitFor();
204               return result;
205           }
206   
207       }

209       class ThreadStream extends Thread {
210   
211           private volatile InputStream streamToRead;
212           private volatile OutputStream redirection;
213           private volatile StringBuffer sb = new StringBuffer();
214   
215           ThreadStream(InputStream streamToRead, OutputStream redirection) {
216               this.streamToRead = streamToRead;
217               this.redirection = redirection;
218           }
219   
220           public String getString() {
221               if (sb == null) {
222                   return "";
223               }
224               return sb.toString();
225           }
226   
227           public void cancel() {
228               assert state.equals(ProcessState.TIMEOUT()) || state.equals(ProcessState.RUNNING()) : state;
229               this.interrupt();
230               sb = null;
231               streamToRead = null;
232               redirection = null;
233               // Because of this, some NPE may occurs in run() method, but we do not care
234           }
235   
236           @Override
237           public void run() {
238               int read = 0;
239               try {
240                   while ((read = streamToRead.read()) != -1) {
241                       if (state.equals(ProcessState.TIMEOUT())) {
242                           return;
243                       }
244                       if (redirection == null) {
245                           sb.append((char) read);
246                       } else {
247                           redirection.write(read);
248                       }
249                   }
250               } catch (Throwable e) {
251                   System.err.println("ProcessRunnerA " + e);
252                   Logme.error(e);
253                   sb.append('\n');
254                   sb.append(e.toString());
255               }
256           }
257       }
258   
259       public final String getError() {
260           return error;
261       }
262   
263       public final String getOut() {
264           return out;
265       }

268           try {
269               if (is != null) {
270                   is.close();
271               }
272           } catch (IOException e) {
273               Logme.error(e);
274           }
275       }

277       private void close(OutputStream os) {
278           try {
279               if (os != null) {
280                   os.close();
281               }
282           } catch (IOException e) {
283               Logme.error(e);
284           }
285       }
