49   class GraphvizWindowsOld extends AbstractGraphviz {
50   
51       static private File specificDotExe;
52   
53       @Override
54       protected File specificDotExe() {
55           synchronized (GraphvizWindowsOld.class) {
56               if (specificDotExe == null) {
57                   specificDotExe = specificDotExeSlow();
58               }
59               if (specificDotExe == null)
60                   specificDotExe = WindowsDotArchive.getInstance().getWindowsExeLite();
61   
62               return specificDotExe;
63           }
64       }
65   
66       public boolean graphviz244onWindows() {
67           try {
68               return GraphvizUtils.getDotVersion() == 244;
69           } catch (Exception e) {
70               Logme.error(e);
71               return false;
72           }
73       }
74   
75       private File specificDotExeSlow() {
76           for (File tmp : new File("c:/").listFiles(new FileFilter() {
77               public boolean accept(java.io.File pathname) {
78                   return pathname.isDirectory() && pathname.canRead();
79               }
80           })) {
81               final File result = searchInDir(tmp);
82               if (result != null) {
83                   return result;
84               }
85           }
86           return null;
87       }
88   
89       private static File searchInDir(final File dir) {
90           if (dir.exists() == false || dir.isDirectory() == false) {
91               return null;
92           }
93           final List<File> dots = new ArrayList<>();
94           final File[] files = dir.listFiles(new FileFilter() {
95               public boolean accept(java.io.File pathname) {
96                   return pathname.isDirectory() && StringUtils.goLowerCase(pathname.getName()).startsWith("graphviz");
97               }
98           });
99           if (files == null) {
100               return null;
101           }
102           for (File f : files) {
103               final File result = new File(new File(f, "bin"), "dot.exe");
104               if (result.exists() && result.canRead()) {
105                   dots.add(result.getAbsoluteFile());
106               }
107               final File result2 = new File(new File(f, "release/bin"), "dot.exe");
108               if (result2.exists() && result2.canRead()) {
109                   dots.add(result2.getAbsoluteFile());
110               }
111           }
112           return higherVersion(dots);
113       }
114   
115       static File higherVersion(List<File> dots) {
116           if (dots.size() == 0) {
117               return null;
118           }
119           Collections.sort(dots, Collections.reverseOrder());
120           return dots.get(0);
121       }

123       GraphvizWindowsOld(ISkinParam skinParam, String dotString, String... type) {
124           super(skinParam, dotString, type);
125       }
126   
127       @Override
128       protected String getExeName() {
129           return "dot.exe";
130       }

132   }
