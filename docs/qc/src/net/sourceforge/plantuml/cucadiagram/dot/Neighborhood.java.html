59       private final List<Link> allButSametails;

61       public Neighborhood(ILeaf leaf, List<Link> sametailLinks, List<Link> all) {
62           this.leaf = leaf;
63           this.sametailLinks = sametailLinks;
64           this.allButSametails = new ArrayList<>(all);
65           allButSametails.removeAll(sametailLinks);
66       }

68       public void drawU(UGraphic ug, double minX, double minY, Bibliotekon bibliotekon, XDimension2D shapeDim) {
69           final Set<XPoint2D> contactPoints = new HashSet<>();
70           for (Link link : sametailLinks) {
71               final SvekLine line = bibliotekon.getLine(link);
72               final XPoint2D contact = line.getStartContactPoint();
73               contactPoints.add(contact);
74           }
75           final XRectangle2D rect = new XRectangle2D(minX, minY, shapeDim.getWidth(), shapeDim.getHeight());
76           final XPoint2D center = new XPoint2D(rect.getCenterX(), rect.getCenterY());
77   
78           for (XPoint2D pt : contactPoints) {
79               final XPoint2D inter = intersection(rect, center, pt);
80               if (inter == null) {
81                   // System.err.println("rect=" + rect);
82                   // System.err.println("center=" + center);
83                   // System.err.println("pt=" + pt);
84                   assert false;
85                   continue;
86               }
87               final double theta = Math.atan2(center.getX() - pt.getX(), -(center.getY() - pt.getY()));
88               final XPoint2D middle = drawExtends(ug, inter, theta);
89               drawLine(ug, middle, pt);
90           }
91   
92           for (Link link : allButSametails) {
93               final SvekLine line = bibliotekon.getLine(link);
94               final XPoint2D contact = link.getEntity1() == leaf ? line.getStartContactPoint()
95                       : line.getEndContactPoint();
96               if (contact == null) {
97                   // assert false;
98                   continue;
99               }
100               final XPoint2D inter = intersection(rect, center, contact);
101               if (inter == null) {
102                   // assert false;
103                   continue;
104               }
105               drawLine(ug, inter, contact);
106           }
107       }

109       private XPoint2D drawExtends(UGraphic ug, XPoint2D contact, double theta) {
110           final UPolygon poly = new UPolygon();
111           poly.addPoint(0, 0);
112           poly.addPoint(7, 20);
113           poly.addPoint(-7, 20);
114           poly.rotate(theta);
115           final UTranslate translate = new UTranslate(contact);
116           ug.apply(translate).draw(poly);
117           final XPoint2D p1 = translate.getTranslated(poly.getPoints().get(1));
118           final XPoint2D p2 = translate.getTranslated(poly.getPoints().get(2));
119           return new XPoint2D((p1.getX() + p2.getX()) / 2, (p1.getY() + p2.getY()) / 2);
120       }

122       static XPoint2D intersection(XRectangle2D rect, XPoint2D pt1, XPoint2D pt2) {
123           XPoint2D p;
124           p = intersection(new XPoint2D(rect.getMinX(), rect.getMinY()), new XPoint2D(rect.getMaxX(), rect.getMinY()),
125                   pt1, pt2);
126           if (p != null)
127               return p;
128   
129           p = intersection(new XPoint2D(rect.getMinX(), rect.getMaxY()), new XPoint2D(rect.getMaxX(), rect.getMaxY()),
130                   pt1, pt2);
131           if (p != null)
132               return p;
133   
134           p = intersection(new XPoint2D(rect.getMinX(), rect.getMinY()), new XPoint2D(rect.getMinX(), rect.getMaxY()),
135                   pt1, pt2);
136           if (p != null)
137               return p;
138   
139           p = intersection(new XPoint2D(rect.getMaxX(), rect.getMinY()), new XPoint2D(rect.getMaxX(), rect.getMaxY()),
140                   pt1, pt2);
141           if (p != null)
142               return p;
143   
144           return null;
145       }
146   
147       static private XPoint2D intersection(XPoint2D pt1, XPoint2D pt2, XPoint2D pt3, XPoint2D pt4) {
148           // System.err.println("Checking intersection of " + pt1 + "-" + pt2 + " and " +
149           // pt3 + "-" + pt4);
150           return intersection(pt1.getX(), pt1.getY(), pt2.getX(), pt2.getY(), pt3.getX(), pt3.getY(), pt4.getX(),
151                   pt4.getY());
152       }
153   
154       private static final double epsilon = .001;
155   
156       static private XPoint2D intersection(double x1, double y1, double x2, double y2, double x3, double y3, double x4,
157               double y4) {
158           final double d = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4);
159           if (d == 0)
160               return null;
161   
162           final double xi = ((x3 - x4) * (x1 * y2 - y1 * x2) - (x1 - x2) * (x3 * y4 - y3 * x4)) / d;
163           final double yi = ((y3 - y4) * (x1 * y2 - y1 * x2) - (y1 - y2) * (x3 * y4 - y3 * x4)) / d;
164   
165           final XPoint2D p = new XPoint2D(xi, yi);
166           if (xi + epsilon < Math.min(x1, x2) || xi - epsilon > Math.max(x1, x2))
167               return null;
168   
169           if (xi + epsilon < Math.min(x3, x4) || xi - epsilon > Math.max(x3, x4))
170               return null;
171   
172           if (yi + epsilon < Math.min(y1, y2) || yi - epsilon > Math.max(y1, y2))
173               return null;
174   
175           if (yi + epsilon < Math.min(y3, y4) || yi - epsilon > Math.max(y3, y4))
176               return null;
177   
178           return p;
179       }

181       private void drawLine(UGraphic ug, XPoint2D pt1, XPoint2D pt2) {
182           drawLine(ug, pt1.getX(), pt1.getY(), pt2.getX(), pt2.getY());
183       }
184   
185       private void drawLine(UGraphic ug, double x1, double y1, double x2, double y2) {
186           final ULine line = new ULine(x2 - x1, y2 - y1);
187           ug.apply(new UTranslate(x1, y1)).draw(line);
188       }
