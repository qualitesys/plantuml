165       public void removeIrrelevantSametail() {
166           final Map<String, Integer> sametails = new HashMap<String, Integer>();
167           for (Link link : links) {
168               if (link.getType().getDecor2().isExtendsLike()) {
169                   link.setSametail(link.getEntity1().getUid());
170               }
171               final String sametail = link.getSametail();
172               if (sametail == null) {
173                   continue;
174               }
175               final Integer value = sametails.get(sametail);
176               sametails.put(sametail, value == null ? 1 : value + 1);
177           }
178           final Collection<String> toremove = new HashSet<>();
179           final int limit = skinParam.groupInheritance();
180           for (Map.Entry<String, Integer> ent : sametails.entrySet()) {
181               final String key = ent.getKey();
182               if (ent.getValue() < limit) {
183                   toremove.add(key);
184               } else {
185                   final List<Link> some = new ArrayList<>();
186                   for (Link link : links) {
187                       if (key.equals(link.getSametail())) {
188                           some.add(link);
189                       }
190                   }
191                   final ILeaf leaf = getLeaf(key);
192                   final Neighborhood neighborhood = new Neighborhood(leaf, some, getLinksOfThisLeaf(leaf));
193                   leaf.setNeighborhood(neighborhood);
194               }
195           }
196   
197           for (Link link : links) {
198               final String sametail = link.getSametail();
199               if (sametail == null) {
200                   continue;
201               }
202               if (toremove.contains(sametail)) {
203                   link.setSametail(null);
204               }
205           }
206       }

208       private List<Link> getLinksOfThisLeaf(ILeaf leaf) {
209           final List<Link> result = new ArrayList<>();
210           for (Link link : links) {
211               if (link.contains(leaf)) {
212                   result.add(link);
213               }
214           }
215           return result;
216       }

218       private ILeaf getLeaf(String key) {
219           for (ILeaf entity : leafs) {
220               if (entity.getUid().equals(key)) {
221                   return entity;
222               }
223           }
224           return null;
225   
226       }
