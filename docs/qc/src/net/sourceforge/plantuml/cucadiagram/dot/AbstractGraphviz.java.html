61       private static String findExecutableOnPath(String name) {
62           final String path = System.getenv("PATH");
63           if (path != null) {
64               for (String dirname : path.split(SFile.pathSeparator)) {
65                   final File file = new File(dirname, name);
66                   if (file.isFile() && file.canExecute()) {
67                       return file.getAbsolutePath();
68                   }
69               }
70           }
71           return null;
72       }

84       final protected File searchDotExe() {
85           String getenv = GraphvizUtils.getenvGraphvizDot();
86           if (findExecutableOnPath() && getenv == null) {
87               getenv = findExecutableOnPath(getExeName());
88           }
89           if (getenv == null) {
90               return specificDotExe();
91           }
92           return new File(getenv);
93       }

99       final public ProcessState createFile3(OutputStream os) {
100           Objects.requireNonNull(dotString);
101   
102           if (getExeState() != ExeState.OK) {
103               // createPngNoGraphviz(os, new
104               // FileFormatOption(FileFormat.valueOf(type[0].goUpperCase())));
105               throw new IllegalStateException();
106           }
107           final String cmd[] = getCommandLine();
108           ProcessRunner p = null;
109           ProcessState state = null;
110           try {
111               Log.info("Starting Graphviz process " + Arrays.asList(cmd));
112               Log.info("DotString size: " + dotString.length());
113               p = new ProcessRunner(cmd);
114               state = p.run(dotString.getBytes(), os);
115               // if (state == ProcessState.TERMINATED_OK) {
116               // result = true;
117               // }
118               Log.info("Ending process ok");
119           } catch (Throwable e) {
120               Logme.error(e);
121               Log.error("Error: " + e);
122               Log.error("The command was " + cmd);
123               Log.error("");
124               Log.error("Try java -jar plantuml.jar -testdot to figure out the issue");
125               Log.error("");
126           } finally {
127               Log.info("Ending Graphviz process");
128           }
129           if (OptionFlags.getInstance().isCheckDotError() && p != null && p.getError().length() > 0) {
130               Log.error("GraphViz error stream : " + p.getError());
131               if (OptionFlags.getInstance().isCheckDotError()) {
132                   throw new IllegalStateException("Dot error " + p.getError());
133               }
134           }
135           if (OptionFlags.getInstance().isCheckDotError() && p != null && p.getOut().length() > 0) {
136               Log.error("GraphViz out stream : " + p.getOut());
137               if (OptionFlags.getInstance().isCheckDotError()) {
138                   throw new IllegalStateException("Dot out " + p.getOut());
139               }
140           }
141           return state;
142       }

153       private String executeCmd(final String cmd[]) {
154           final ProcessRunner p = new ProcessRunner(cmd);
155           final ProcessState state = p.run(null, null);
156           if (state.differs(ProcessState.TERMINATED_OK())) {
157               return "?";
158           }
159           final StringBuilder sb = new StringBuilder();
160           if (StringUtils.isNotEmpty(p.getOut())) {
161               sb.append(p.getOut());
162           }
163           if (StringUtils.isNotEmpty(p.getError())) {
164               if (sb.length() > 0) {
165                   sb.append(' ');
166               }
167               sb.append(p.getError());
168           }
169           return StringUtils.trin(sb.toString().replace('\n', ' '));
170       }

172       final String[] getCommandLine() {
173           if (OptionFlags.ADD_NICE_FOR_DOT) {
174               final String[] result = new String[type.length + 1 + 3];
175               result[0] = "/bin/nice";
176               result[1] = "-n";
177               result[2] = "10";
178               result[3] = getDotExe().getAbsolutePath();
179               for (int i = 0; i < type.length; i++) {
180                   result[i + 4] = "-T" + type[i];
181               }
182               return result;
183           }
184           final String[] result = new String[type.length + 1];
185           result[0] = getDotExe().getAbsolutePath();
186           for (int i = 0; i < type.length; i++) {
187               result[i + 1] = "-T" + type[i];
188           }
189           return result;
190       }
