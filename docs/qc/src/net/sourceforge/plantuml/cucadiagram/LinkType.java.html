163       public String getSpecificDecorationSvek() {
164           final StringBuilder sb = new StringBuilder();
165   
166           final boolean isEmpty1 = decor1 == LinkDecor.NONE && hat1 == LinkHat.NONE;
167           final boolean isEmpty2 = decor2 == LinkDecor.NONE && hat2 == LinkHat.NONE;
168   
169           if (isEmpty1 && isEmpty2) {
170               sb.append("arrowtail=none");
171               sb.append(",arrowhead=none");
172           } else if (isEmpty1 == false && isEmpty2 == false) {
173               sb.append("dir=both,");
174               sb.append("arrowtail=empty");
175               sb.append(",arrowhead=empty");
176           } else if (isEmpty1 && isEmpty2 == false) {
177               sb.append("arrowtail=empty");
178               sb.append(",arrowhead=none");
179               sb.append(",dir=back");
180               // } else if (isEmpty1 == false && isEmpty2) {
181               // sb.append("arrowtail=none");
182               // sb.append(",arrowhead=empty");
183           }
184   
185           final double arrowsize = Math.max(decor1.getArrowSize(), decor2.getArrowSize());
186           if (arrowsize > 0) {
187               if (sb.length() > 0)
188                   sb.append(",");
189   
190               sb.append("arrowsize=" + arrowsize);
191           }
192           return sb.toString();
193       }

207       private boolean isExtendsOrAggregationOrCompositionOrPlus() {
208           return isExtends() || isAggregationOrComposition() || isPlus() || isOf(LinkDecor.DEFINEDBY)
209                   || isOf(LinkDecor.REDEFINES);
210       }
211   
212       private boolean isOf(LinkDecor ld) {
213           return decor1 == ld || decor2 == ld;
214       }
215   
216       private boolean isExtends() {
217           return decor1 == LinkDecor.EXTENDS || decor2 == LinkDecor.EXTENDS;
218       }
219   
220       private boolean isPlus() {
221           return decor1 == LinkDecor.PLUS || decor2 == LinkDecor.PLUS;
222       }
223   
224       private boolean isAggregationOrComposition() {
225           return decor1 == LinkDecor.AGREGATION || decor2 == LinkDecor.AGREGATION || decor1 == LinkDecor.COMPOSITION
226                   || decor2 == LinkDecor.COMPOSITION;
227       }
