40   public class LinkArg {
41   
42       private final Display label;
43       private final String quantifier1;
44       private final String quantifier2;
45       private final String labeldistance;
46       private final String labelangle;
47   
48       private final String kal1;
49       private final String kal2;
50   
51       private VisibilityModifier visibilityModifier;
52       private int length;
53   
54       public static LinkArg build(final Display label, int length) {
55           return build(label, length, true);
56       }
57   
58       public static LinkArg noDisplay(int length) {
59           return build(Display.NULL, length, true);
60       }
61   
62       public static LinkArg build(final Display label, int length, boolean manageVisibilityModifier) {
63           VisibilityModifier visibilityModifier = null;
64           final Display newLabel;
65           if (Display.isNull(label)) {
66               newLabel = Display.NULL;
67           } else {
68               newLabel = label.manageGuillemet(manageVisibilityModifier);
69               if (manageVisibilityModifier && VisibilityModifier.isVisibilityCharacter(label.get(0)))
70                   visibilityModifier = VisibilityModifier.getVisibilityModifier(label.get(0), false);
71           }
72           return new LinkArg(newLabel, length, null, null, null, null, visibilityModifier, null, null);
73       }
74   
75       public LinkArg withQuantifier(String quantifier1, String quantifier2) {
76           return new LinkArg(label, length, quantifier1, quantifier2, labeldistance, labelangle, visibilityModifier, kal1,
77                   kal2);
78       }
79   
80       public LinkArg withKal(String kal1, String kal2) {
81           return new LinkArg(label, length, quantifier1, quantifier2, labeldistance, labelangle, visibilityModifier, kal1,
82                   kal2);
83       }
84   
85       public LinkArg withDistanceAngle(String labeldistance, String labelangle) {
86           return new LinkArg(label, length, quantifier1, quantifier2, labeldistance, labelangle, visibilityModifier, kal1,
87                   kal2);
88       }

90       private LinkArg(Display label, int length, String quantifier1, String quantifier2, String labeldistance,
91               String labelangle, VisibilityModifier visibilityModifier, String kal1, String kal2) {
92   
93           this.label = label;
94           this.visibilityModifier = visibilityModifier;
95           this.length = length;
96           this.quantifier1 = quantifier1;
97           this.quantifier2 = quantifier2;
98           this.labeldistance = labeldistance;
99           this.labelangle = labelangle;
100           this.kal1 = kal1;
101           this.kal2 = kal2;
102       }
