103       private boolean hasSmallIcon() {

118       public XDimension2D calculateDimension(StringBounder stringBounder) {
119           double smallIcon = 0;
120           if (hasSmallIcon())
121               smallIcon = skinParam.getCircledCharacterRadius() + 3;
122   
123           double x = 0;
124           double y = 0;
125           for (CharSequence cs : members) {
126               final TextBlock bloc = createTextBlock(cs);
127               final XDimension2D dim = bloc.calculateDimension(stringBounder);
128               x = Math.max(dim.getWidth(), x);
129               y += dim.getHeight();
130           }
131           x += smallIcon;
132           return new XDimension2D(x, y);
133       }

135       private Collection<String> sortBySize(Collection<String> all) {
136           final List<String> result = new ArrayList<String>(all);
137           Collections.sort(result, new Comparator<String>() {
138               @Override
139               public int compare(String s1, String s2) {
140                   final int diff = s2.length() - s1.length();
141                   if (diff != 0)
142                       return diff;
143                   return s1.compareTo(s2);
144               }
145           });
146           return result;
147       }

164           }
165           return ports;

183       private int getScore(String cs, String shortName) {
184           if (cs.matches(".*\\b" + shortName + "\\b.*"))
185               return 100;
186   
187           if (cs.contains(shortName))
188               return 50;
189   
190           return 0;
191       }

193       private TextBlock createTextBlock(CharSequence cs) {
194   
195           FontConfiguration config = FontConfiguration.create(skinParam, style, leaf.getColors());
196   
197           if (cs instanceof Member) {
198               final Member m = (Member) cs;
199               final boolean withVisibilityChar = skinParam.classAttributeIconSize() == 0;
200               String s = m.getDisplay(withVisibilityChar);
201               if (withVisibilityChar && s.startsWith("#"))
202                   s = CharHidder.addTileAtBegin(s);
203   
204               if (m.isAbstract())
205                   config = config.italic();
206   
207               if (m.isStatic())
208                   config = config.underline();
209   
210               TextBlock bloc = Display.getWithNewlines(s).create8(config, align, skinParam, CreoleMode.SIMPLE_LINE,
211                       skinParam.wrapWidth());
212               bloc = TextBlockUtils.fullInnerPosition(bloc, m.getDisplay(false));
213               return new TextBlockTracer(m, bloc);
214           }
215   
216           if (cs instanceof EmbeddedDiagram)
217               return ((EmbeddedDiagram) cs).asDraw(skinParam);
218   
219           return Display.getWithNewlines(cs.toString()).create8(config, align, skinParam, CreoleMode.SIMPLE_LINE,
220                   skinParam.wrapWidth());
221   
222       }

302       private ULayoutGroup getLayout(final StringBounder stringBounder) {
303           final ULayoutGroup group;
304           if (hasSmallIcon()) {
305               group = new ULayoutGroup(
306                       new PlacementStrategyVisibility(stringBounder, skinParam.getCircledCharacterRadius() + 3));
307               for (CharSequence cs : members) {
308                   final TextBlock bloc = createTextBlock(cs);
309                   if (cs instanceof EmbeddedDiagram) {
310                       group.add(getUBlock(null, null));
311                   } else {
312                       final Member att = (Member) cs;
313                       final VisibilityModifier modifier = att.getVisibilityModifier();
314                       group.add(getUBlock(modifier, att.getUrl()));
315                   }
316                   group.add(bloc);
317               }
318           } else {
319               final PlacementStrategy placementStrategy;
320               if (align == HorizontalAlignment.LEFT)
321                   placementStrategy = new PlacementStrategyY1Y2Left(stringBounder);
322               else if (align == HorizontalAlignment.CENTER)
323                   placementStrategy = new PlacementStrategyY1Y2Center(stringBounder);
324               else
325                   placementStrategy = new PlacementStrategyY1Y2Right(stringBounder);
326   
327               group = new ULayoutGroup(placementStrategy);
328               for (CharSequence cs : members) {
329                   final TextBlock bloc = createTextBlock(cs);
330                   group.add(bloc);
331               }
332           }
333           return group;
334       }
