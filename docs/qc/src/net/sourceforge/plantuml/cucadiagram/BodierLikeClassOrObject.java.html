58   public class BodierLikeClassOrObject implements Bodier {
59   
60       private final List<CharSequence> rawBody = new ArrayList<>();
61       private final Set<VisibilityModifier> hides;
62       private LeafType type;
63       private List<Member> methodsToDisplay;
64       private List<Member> fieldsToDisplay;
65       private ILeaf leaf;
66   
67       @Override
68       public void muteClassToObject() {
69           methodsToDisplay = null;
70           fieldsToDisplay = null;
71           type = LeafType.OBJECT;
72       }
73   
74       BodierLikeClassOrObject(LeafType type, Set<VisibilityModifier> hides) {
75           if (type == LeafType.MAP)
76               throw new IllegalArgumentException();
77   
78           this.type = Objects.requireNonNull(type);
79           assert type.isLikeClass() || type == LeafType.OBJECT;
80           this.hides = hides;
81       }
82   
83       @Override
84       public void setLeaf(ILeaf leaf) {
85           this.leaf = Objects.requireNonNull(leaf);
86   
87       }
88   
89       @Override
90       public boolean addFieldOrMethod(String s) {
91           // Empty cache
92           methodsToDisplay = null;
93           fieldsToDisplay = null;
94           rawBody.add(s);
95           return true;
96       }
97   
98       private boolean isBodyEnhanced() {
99           for (CharSequence s : rawBody)
100               if (BodyEnhanced1.isBlockSeparator(s) || CreoleParser.isTableLine(s.toString())
101                       || Parser.isTreeStart(s.toString()))
102                   return true;
103   
104           return false;
105       }
106   
107       private boolean isMethod(CharSequence s) {
108           final String purged = s.toString().replaceAll(UrlBuilder.getRegexp(), "");
109           if (purged.contains("{method}"))
110               return true;
111   
112           if (purged.contains("{field}"))
113               return false;
114   
115           return purged.contains("(") || purged.contains(")");
116       }
117   
118       @Override
119       public Display getMethodsToDisplay() {
120           if (methodsToDisplay == null) {
121               methodsToDisplay = new ArrayList<>();
122               for (int i = 0; i < rawBody.size(); i++) {
123                   final CharSequence s = rawBody.get(i);
124                   if (isMethod(i, rawBody) == false)
125                       continue;
126   
127                   if (s.length() == 0 && methodsToDisplay.size() == 0)
128                       continue;
129   
130                   final Member m = Member.method(s);
131                   if (hides == null || hides.contains(m.getVisibilityModifier()) == false)
132                       methodsToDisplay.add(m);
133   
134               }
135               removeFinalEmptyMembers(methodsToDisplay);
136           }
137           return Display.create(methodsToDisplay);
138       }
139   
140       private boolean isMethod(int i, List<CharSequence> rawBody) {
141           if (i > 0 && i < rawBody.size() - 1 && rawBody.get(i).length() == 0 && isMethod(rawBody.get(i - 1))
142                   && isMethod(rawBody.get(i + 1))) {
143               return true;
144           }
145           return isMethod(rawBody.get(i));
146       }
147   
148       @Override
149       public Display getFieldsToDisplay() {
150           if (fieldsToDisplay == null) {
151               fieldsToDisplay = new ArrayList<>();
152               for (CharSequence s : rawBody) {
153                   if (type != LeafType.OBJECT && isMethod(s) == true)
154                       continue;
155   
156                   if (s.length() == 0 && fieldsToDisplay.size() == 0)
157                       continue;
158   
159                   final Member m = Member.field(s);
160                   if (hides == null || hides.contains(m.getVisibilityModifier()) == false)
161                       fieldsToDisplay.add(m);
162   
163               }
164               removeFinalEmptyMembers(fieldsToDisplay);
165           }
166           return Display.create(fieldsToDisplay);
167       }
168   
169       private void removeFinalEmptyMembers(List<Member> result) {
170           while (result.size() > 0 && StringUtils.trin(result.get(result.size() - 1).getDisplay(false)).length() == 0)
171               result.remove(result.size() - 1);
172   
173       }
174   
175       @Override
176       public boolean hasUrl() {
177           for (CharSequence cs : getFieldsToDisplay())
178               if (cs instanceof Member) {
179                   final Member m = (Member) cs;
180                   if (m.hasUrl())
181                       return true;
182   
183               }
184   
185           for (CharSequence cs : getMethodsToDisplay())
186               if (cs instanceof Member) {
187                   final Member m = (Member) cs;
188                   if (m.hasUrl())
189                       return true;
190   
191               }
192           return false;
193       }
194   
195       private List<CharSequence> rawBodyWithoutHidden() {
196           final List<CharSequence> result = new ArrayList<>();
197           for (CharSequence s : rawBody) {
198               final Member m;
199               if (isMethod(s))
200                   m = Member.method(s);
201               else
202                   m = Member.field(s);
203   
204               if (hides.contains(m.getVisibilityModifier()) == false)
205                   result.add(m);
206   
207           }
208           return result;
209       }

211       @Override
212       public TextBlock getBody(ISkinParam skinParam, boolean showMethods, boolean showFields, Stereotype stereotype,
213               Style style, FontConfiguration fontConfiguration) {
214   
215           if (BodyFactory.BODY3)
216               return new Body3(rawBody, skinParam, stereotype, style);
217   
218           if (type.isLikeClass() && isBodyEnhanced()) {
219               if (showMethods || showFields)
220                   return BodyFactory.create1(skinParam.getDefaultTextAlignment(HorizontalAlignment.LEFT),
221                           rawBodyWithoutHidden(), skinParam, stereotype, leaf, style);
222   
223               return null;
224           }
225           if (leaf == null)
226               throw new IllegalStateException();
227   
228           if (type == LeafType.OBJECT) {
229               if (showFields == false)
230                   return new TextBlockLineBefore(style.value(PName.LineThickness).asDouble(), TextBlockUtils.empty(0, 0));
231   
232               return BodyFactory.create1(skinParam.getDefaultTextAlignment(HorizontalAlignment.LEFT),
233                       rawBodyWithoutHidden(), skinParam, stereotype, leaf, style);
234           }
235           assert type.isLikeClass();
236   
237           final MethodsOrFieldsArea fields = new MethodsOrFieldsArea(getFieldsToDisplay(), skinParam, leaf, style);
238   
239           final MethodsOrFieldsArea methods = new MethodsOrFieldsArea(getMethodsToDisplay(), skinParam, leaf, style);
240           if (showFields && showMethods == false)
241               return fields.asBlockMemberImpl();
242           else if (showMethods && showFields == false)
243               return methods.asBlockMemberImpl();
244           else if (showFields == false && showMethods == false)
245               return TextBlockUtils.empty(0, 0);
246   
247           final TextBlock bb1 = fields.asBlockMemberImpl();
248           final TextBlock bb2 = methods.asBlockMemberImpl();
249           return TextBlockUtils.mergeTB(bb1, bb2, HorizontalAlignment.LEFT);
250       }
251   
252       @Override
253       public List<CharSequence> getRawBody() {
254           return Collections.unmodifiableList(rawBody);
255       }
