121       @Override
122       final protected TextBlock getArea(StringBounder stringBounder) {
123           if (area != null)
124               return area;
125   
126           urls.clear();
127           final List<TextBlock> blocks = new ArrayList<>();
128   
129           char separator = lineFirst ? '_' : 0;
130           TextBlock title = null;
131           Display display = null;
132           for (ListIterator<CharSequence> it = rawBody2.iterator(); it.hasNext();) {
133               final CharSequence cs = it.next();
134               if (cs instanceof EmbeddedDiagram) {
135                   if (display == null)
136                       display = Display.empty();
137                   if (display.size() > 0 || separator != 0) {
138                       blocks.add(buildTextBlock(display, separator, title, stringBounder));
139                       separator = 0;
140                       title = null;
141                       display = null;
142                   }
143                   blocks.add(TextBlockUtils.withMargin(((EmbeddedDiagram) cs).asDraw(skinParam), 2, 2));
144               } else {
145                   final String s = cs.toString();
146                   if (isBlockSeparator(s)) {
147                       if (display == null)
148                           display = Display.empty();
149                       blocks.add(buildTextBlock(display, separator, title, stringBounder));
150                       separator = s.charAt(0);
151                       title = getTitle(s, skinParam);
152                       display = null;
153                   } else if (isTreeOrTable(s)) {
154                       final boolean isTable = CreoleParser.isTableLine(s);
155                       if (display == null)
156                           display = Display.empty();
157                       blocks.add(buildTextBlock(display, separator, title, stringBounder));
158   
159                       separator = 0;
160                       title = null;
161                       display = null;
162                       final List<CharSequence> allTree = buildTreeOrTable(s, it);
163                       final FontConfiguration fontConfiguration = style
164                               .getFontConfiguration(skinParam.getIHtmlColorSet());
165                       TextBlock bloc = Display.create(allTree).create7(fontConfiguration, align, skinParam,
166                               CreoleMode.FULL);
167                       if (isTable)
168                           bloc = TextBlockUtils.withMargin(bloc, 10, 10, 0, 5);
169   
170                       blocks.add(bloc);
171                   } else {
172                       if (display == null)
173                           display = Display.empty();
174                       display = display.add(cs);
175                       if (cs instanceof Member && ((Member) cs).getUrl() != null)
176                           urls.add(((Member) cs).getUrl());
177   
178                   }
179               }
180           }
181   
182           if (display == null)
183               display = Display.empty();
184           if (inEllipse && display.size() == 0)
185               display = display.add("");
186   
187           blocks.add(buildTextBlock(display, separator, title, stringBounder));
188   
189           if (blocks.size() == 1)
190               this.area = blocks.get(0);
191           else
192               this.area = new TextBlockVertical2(blocks, align);
193   
194           final double minClassWidth = style.value(PName.MinimumWidth).asDouble();
195           if (minClassWidth > 0)
196               this.area = TextBlockUtils.withMinWidth(this.area, minClassWidth, align);
197   
198           return area;
199       }

201       private TextBlock buildTextBlock(Display display, char separator, TextBlock title, StringBounder stringBounder) {
202           // TextBlock result = display.create9(titleConfig, align, skinParam,
203           // LineBreakStrategy.NONE);
204           TextBlock result = new MethodsOrFieldsArea(display, skinParam, align, entity, style);
205           result = decorate(result, separator, title, stringBounder);
206           return result;
207       }

209       private static List<CharSequence> buildTreeOrTable(String init, ListIterator<CharSequence> it) {
210           final List<CharSequence> result = new ArrayList<>();
211           final Pattern p = Pattern.compile("^(\\s+)");
212           final Matcher m = p.matcher(init);
213           String start = "";
214           if (m.find())
215               start = m.group(1);
216   
217           result.add(purge(init, start));
218           while (it.hasNext()) {
219               String s = it.next().toString();
220               if (isTreeOrTable(s)) {
221                   result.add(purge(s, start));
222               } else {
223                   it.previous();
224                   return result;
225               }
226   
227           }
228           return result;
229       }
230   
231       private static String purge(String s, String start) {
232           if (s.startsWith(start))
233               return s.substring(start.length());
234   
235           return s;
236       }
