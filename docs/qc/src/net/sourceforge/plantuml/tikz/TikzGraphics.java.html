91       private void addCommand(final StringBuilder sb) {
92           final String s = sb.toString();
93           cmd.add(s);
94           if (s.startsWith("\\draw[") || s.startsWith("\\shade[")) {
95               final int end = s.indexOf(']');
96               if (end != -1) {
97                   final int start = s.indexOf('[');
98                   final String style = s.substring(start + 1, end);
99                   Integer count = styles.get(style);
100                   if (count == null) {
101                       count = 1;
102                       stylesID.put(style, "pstyle" + stylesID.size());
103                   } else {
104                       count++;
105                   }
106                   styles.put(style, count);
107               }
108           }

111       private String getColorName(HColor hcolor) {
112           final Color color = hcolor.toColor(mapper);
113           if (color.equals(Color.WHITE))
114               return "white";
115   
116           if (color.equals(Color.BLACK))
117               return "black";
118

120           return Objects.requireNonNull(result);

218       private String useStyle(String s) {
219           for (String style : styles.keySet()) {
220               final String start1 = "\\draw[" + style + "]";
221               if (s.startsWith(start1)) {
222                   final String newStart = "\\draw[" + stylesID.get(style) + "]";
223                   return newStart + s.substring(start1.length());
224               }
225               final String start2 = "\\shade[" + style + "]";
226               if (s.startsWith(start2)) {
227                   final String newStart = "\\shade[" + stylesID.get(style) + "]";
228                   return newStart + s.substring(start2.length());
229               }
230           }
231           return s;
232       }

270               sb.append("below right,");
271               sb.append("inner sep=2ex,");
272               sb.append("minimum width=" + format(width) + "pt,");
273               sb.append("minimum height=" + format(height) + "pt,");

315       public void text(double x, double y, String text, boolean underline, boolean italic, boolean bold) {
316           final StringBuilder sb = new StringBuilder("\\node at " + couple(x, y));

321           }
322           sb.append("]{");

324               if (underline)
325                   sb.append("\\underline{");
326   
327               if (italic)
328                   sb.append("\\textit{");
329   
330               if (bold)
331                   sb.append("\\textbf{");
332   
333               sb.append(protectText(text));
334               if (bold)
335                   sb.append("}");
336   
337               if (italic)
338                   sb.append("}");
339   
340               if (underline)
341                   sb.append("}");
342   
343           } else {
344               appendPendingUrl(sb);
345               sb.append("{");
346               sb.append(protectText(text));
347               sb.append("}");
348           }
349           sb.append("};");
350           addCommand(sb);
351       }

384       private String protectText(String text) {
385           text = text.replaceAll("\\\\", "\\\\\\\\");
386           text = text.replaceAll("_", "\\\\_");
387           text = text.replaceAll("\u00AB", "\\\\guillemotleft ");
388           text = text.replaceAll("\u00BB", "\\\\guillemotright ");
389           text = text.replaceAll("<", "\\\\textless ");
390           text = text.replaceAll(">", "\\\\textgreater ");
391           text = text.replaceAll("&", "\\\\&");
392           text = text.replaceAll("%", "\\\\%");
393           text = text.replace("$", "\\$");
394           text = text.replace("{", "\\{");
395           text = text.replace("}", "\\}");
396           // text = text.replaceAll("~", "\\\\~{}");
397           text = text.replace("~", "{\\raise.35ex\\hbox{$\\scriptstyle\\mathtt{\\sim}$}}");
398           // {\raise.35ex\hbox{$\scriptstyle\mathtt{\sim}$}}
399           // {\\raise.35ex\\hbox{$\\scriptstyle\\mathtt{\\sim}$}}
400           return text;
401       }

463       private void appendShadeOrDraw(final StringBuilder sb) {
464           final boolean gradient = this.fillcolorGradient2 != null;
465           sb.append(gradient ? "\\shade[" : "\\draw[");
466           if (color != null) {
467               sb.append(gradient ? "draw=" : "color=");
468               sb.append(getColorName(color) + ",");
469           }
470           if (gradient) {
471               sb.append("top color=" + getColorName(fillcolor) + ",");
472               sb.append("bottom color=" + getColorName(fillcolorGradient2) + ",");
