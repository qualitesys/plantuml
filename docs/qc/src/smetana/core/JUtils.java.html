175       public static boolean EQ(Object o1, Object o2) {
176           final boolean result = EQ_(o1, o2);
177           if (o1 instanceof UnsupportedStarStruct && o2 instanceof UnsupportedStarStruct) {
178               UnsupportedStarStruct ooo1 = (UnsupportedStarStruct) o1;
179               UnsupportedStarStruct ooo2 = (UnsupportedStarStruct) o2;
180               if ((ooo1.UID == ooo2.UID) != result) {
181                   throw new UnsupportedOperationException();
182               }
183           }
184           return result;
185       }

187       private static boolean EQ_(Object o1, Object o2) {
188           if (o1 == o2) {
189               return true;
190           }
191           if (o1 == null && o2 != null) {
192               return false;
193           }
194           if (o2 == null && o1 != null) {
195               return false;
196           }
197           if (o1 instanceof CString && o2 instanceof CString) {
198               return ((CString) o1).isSameThan((CString) o2);
199           }
200           if (o1 instanceof CArrayOfStar && o2 instanceof CArrayOfStar) {
201               return ((CArrayOfStar) o1).comparePointer_((CArrayOfStar) o2) == 0;
202           }
203           if (o1 instanceof __ptr__ && o2 instanceof __ptr__) {
204               return ((__ptr__) o1).isSameThan((__ptr__) o2);
205           }
206   
207           System.err.println("o1=" + o1.getClass() + " " + o1);
208           System.err.println("o2=" + o2.getClass() + " " + o2);
209           throw new UnsupportedOperationException();
210       }

216       public static void qsort1(CArrayOfStar array, int nb, CFunction compare) {
217           SmetanaDebug.LOG("qsort1 "+nb);
218           boolean change;
219           do {
220               change = false;
221               for (int i = 0; i < nb - 1; i++) {
222                   __ptr__ element1 = array.plus_(i);
223                   __ptr__ element2 = array.plus_(i + 1);
224                   Integer cmp = (Integer) compare.exe(element1, element2);
225                   if (cmp.intValue() > 0) {
226                       change = true;
227                       array._swap(i, i + 1);
228                   }
229               }
230           } while (change);
231           SmetanaDebug.LOG("qsort1 ok");
232   
233       }

235       public static void qsort2(int array[], int nb, CFunction compare) {
236           SmetanaDebug.LOG("qsort2 "+nb);
237           boolean change;
238           do {
239               change = false;
240               for (int i = 0; i < nb - 1; i++) {
241                   Integer element1 = array[i];
242                   Integer element2 = array[i + 1];
243                   Integer cmp = (Integer) compare.exe(element1, element2);
244                   if (cmp.intValue() > 0) {
245                       change = true;
246                       final int tmp = array[i];
247                       array[i] = array[i + 1];
248                       array[i + 1] = tmp;
249                   }
250               }
251           } while (change);
252           SmetanaDebug.LOG("qsort2 ok");
253   
254       }
