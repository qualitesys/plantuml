222   public static void map_path(ST_Agnode_s from, ST_Agnode_s to, ST_Agedge_s orig, ST_Agedge_s ve, int type) {
223   ENTERING("6o86r59v2ujlxqcw7761y6o5b","map_path");
224   try {
225       int r;
226       ST_Agnode_s u, v;
227       ST_Agedge_s e;
228       assert(ND_rank(from) < ND_rank(to));
229       if (EQ(agtail(ve), from) && EQ(aghead(ve), to))
230       return;
231       if (ED_count(ve) > 1) {
232       ED_to_virt(orig, null);
233       if (ND_rank(to) - ND_rank(from) == 1) {
234           if ((e = find_fast_edge(from, to))!=null && (ports_eq(orig, e))) {
235           merge_oneway(orig, e);
236           if ((ND_node_type(from) == 0)
237               && (ND_node_type(to) == 0))
238               other_edge(orig);
239           return;
240           }
241       }
242       u = from;
243       for (r = ND_rank(from); r < ND_rank(to); r++) {
244           if (r < ND_rank(to) - 1)
245           v = clone_vn(dot_root(from), aghead(ve));
246           else
247           v = to;
248           e = virtual_edge(u, v, orig);
249           ED_edge_type(e, type);
250           u = v;
251           ED_count(ve, ED_count(ve) - 1);
252           ve = (ST_Agedge_s) ND_out(aghead(ve)).list.get_(0);
253       }
254       } else {
255       if (ND_rank(to) - ND_rank(from) == 1) {
256           if ((ve = find_fast_edge(from, to))!=null && (ports_eq(orig, ve))) {
257           /*ED_to_orig(ve) = orig; */
258           ED_to_virt(orig, ve);
259           ED_edge_type(ve, type);
260           ED_count(ve, ED_count(ve)+1);
261           if ((ND_node_type(from) == 0)
262               && (ND_node_type(to) == 0))
263               other_edge(orig);
264           } else {
265           ED_to_virt(orig, null);
266           ve = virtual_edge(from, to, orig);
267           ED_edge_type(ve, type);
268           }
269       }
270       if (ND_rank(to) - ND_rank(from) > 1) {
271           e = ve;
272           if (NEQ(agtail(ve), from)) {
273           ED_to_virt(orig, null);
274           e = virtual_edge(from, aghead(ve), orig);
275           ED_to_virt(orig, e);
276           delete_fast_edge(ve);
277           } else
278           e = ve;
279           while (ND_rank(aghead(e)) != ND_rank(to))
280           e = (ST_Agedge_s) ND_out(aghead(e)).list.get_(0);
281           if (NEQ(aghead(e), to)) {
282           ve = e;
283           e = virtual_edge(agtail(e), to, orig);
284           ED_edge_type(e, type);
285           delete_fast_edge(ve);
286           }
287       }
288       }
289   } finally {
290   LEAVING("6o86r59v2ujlxqcw7761y6o5b","map_path");
291   }

329   ENTERING("6g2m2y44x66lajznvnon2gubv","interclexp");
330   try {

332       ST_Agnode_s n;
333       ST_Agedge_s e, prev, next;
334       
335       g = dot_root(subg);
336       for (n = agfstnode(subg); n!=null; n = agnxtnode(subg, n)) {
337       
338       /* N.B. n may be in a sub-cluster of subg */
339       prev = null;
340       for (e = agfstedge(g, n); e!=null; e = next) {
341           next = agnxtedge(g, e, n);
342           if (agcontains(subg, e))
343           continue;
344           
345           /* canonicalize edge */
346           e = AGMKOUT(e);
347           /* short/flat multi edges */
348           if (mergeable(prev, e)) {
349           if (ND_rank(agtail(e)) == ND_rank(aghead(e)))
350               ED_to_virt(e, prev);
351           else
352               ED_to_virt(e, null);
353            if (ED_to_virt(prev) == null)
354                continue;    /* internal edge */
355            merge_chain(subg, e, ED_to_virt(prev), false);
356            safe_other_edge(e);
357            continue;
358           }
359           
360           /* flat edges */
361           if (ND_rank(agtail(e)) == ND_rank(aghead(e))) {
362           ST_Agedge_s fe;
363           if ((fe = find_flat_edge(agtail(e), aghead(e))) == null) {
364               flat_edge(g, e);
365               prev = e;
366           } else if (NEQ(e, fe)) {
367                        safe_other_edge(e);
368                        if (N(ED_to_virt(e))) merge_oneway(e, fe);
369           }
370           continue;
371           }
372           
373           /* forward edges */
374           if (ND_rank(aghead(e)) > ND_rank(agtail(e))) {
375           make_interclust_chain(g, agtail(e), aghead(e), e);
376           prev = e;
377           continue;
378           }
379           
380           /* backward edges */
381           else {
382   /*
383   I think that make_interclust_chain should create call other_edge(e) anyway 
384                   if (agcontains(subg,agtail(e))
385                       && agfindedge(g,aghead(e),agtail(e))) other_edge(e);
386   */
387           make_interclust_chain(g, aghead(e), agtail(e), e);
388           prev = e;
389           }
390       }
391       }
392   } finally {
393   LEAVING("6g2m2y44x66lajznvnon2gubv","interclexp");
394   }
