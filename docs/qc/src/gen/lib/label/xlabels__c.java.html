238   private static int hd_hil_s_from_xy_(ST_point p, int n) {
239        int i, x = p.x, y = p.y, xi, yi;
240        int s;
241        s = 0;            /* Initialize. */
242        for (i = n - 1; i >= 0; i--) {
243        xi = (x >> i) & 1;    /* Get bit i of x. */
244        yi = (y >> i) & 1;    /* Get bit i of y. */
245        s = 4 * s + 2 * xi + (xi ^ yi);    /* Append two bits to s. */
246        x = x ^ y;        /* These 3 lines swap */
247        y = y ^ (x & (yi - 1));    /* x and y if yi = 0. */
248        x = x ^ y;
249        x = x ^ (-xi & (yi - 1));    /* Complement x and y if */
250        y = y ^ (-xi & (yi - 1));    /* xi = 1 and yi = 0. */
251        }
252        return s;
253    }
